201         private boolean stopFlag;
202 
203         /**
204          * Sets the stopFlag variable to true
205          */
206         public void setStopFlag() {
207             stopFlag = true;
208         }
209 
210         @Override
211         public void run() {
212         /*
213         This snap inserts the batch of documents either when the batch reaches the full size or
214          when the batch timeout occurs (whichever is earlier).
215         The timeout condition is handled in this TimerTask object.
216         This task should not proceed if the main snap thread is already processing the batch.
217         This is achieved by synchronizing the code block using the shared object "documents".
218          */
219             synchronized (documents) {
220                 if (stopFlag) {
221                     cancel();
222                     return;
223                 }
224                 if (!documents.isEmpty()) {
225                     flushBatch();
226                     reset();
227                 }
228             }
229         }
230     }

441     private List<Field> generateFieldList(Map<String, Object> data) {
442         List<Field> fieldList = new ArrayList<>();
443         Set<Map.Entry<String, Object>> entries = data.entrySet();
444 
445         for (Map.Entry<String, Object> entry : entries) {
446             Object value = entry.getValue();
447             String fieldName = entry.getKey();
448             LegacySQLTypeName fieldType;
449             Field field;
450             if (value instanceof Map) {
451                 List<Field> subFields = generateFieldList((Map) value);
452                 field = Field.of(
453                         fieldName, LegacySQLTypeName.RECORD, FieldList.of(subFields));
454             } else if (value instanceof List) {
455                 List list = (List) value;
456                 if (list.isEmpty()) {
457                     fieldType = LegacySQLTypeName.STRING;
458                     field = Field.newBuilder(fieldName, fieldType)
459                             .setMode(Field.Mode.REPEATED).build();
460                 } else {
461                     Object firstObject = list.get(0);
462                     if (firstObject instanceof List) {
463                         throw new ExecutionException(ERR_UNSUPPORTED_DATA_TYPE_MSG)
464                                 .withReason(ERR_LIST_OF_LIST_NOT_SUPPORTED_REASON)
465                                 .withResolution(ERR_LIST_OF_LIST_NOT_SUPPORTED_RESOLUTION);
466                     } else if (firstObject instanceof Map) {
467                         List<Field> subFields = generateFieldList((Map) firstObject);
468                         field = Field.newBuilder(
469                                 fieldName, LegacySQLTypeName.RECORD, FieldList.of(subFields))
470                                 .setMode(Field.Mode.REPEATED).build();
471                     } else {
472                         fieldType = getPrimitiveFieldType(firstObject);
473                         field = Field.newBuilder(fieldName, fieldType)
474                                 .setMode(Field.Mode.REPEATED).build();
475                     }
476                 }
477             } else {
478                 fieldType = getPrimitiveFieldType(value);
479                 field = Field.of(fieldName, fieldType);
480             }
481             fieldList.add(field);
482         }
483         return fieldList;
484     }

486     private LegacySQLTypeName getPrimitiveFieldType(Object value) {
487         if (value == null || value instanceof String) {
488             return LegacySQLTypeName.STRING;
489         } else if (value instanceof BigInteger || value instanceof Long ||
490                 value instanceof Integer) {
491             return LegacySQLTypeName.INTEGER;
492         } else if (value instanceof BigDecimal || value instanceof Double ||
493                 value instanceof Float) {
494             return LegacySQLTypeName.FLOAT;
495         } else if (value instanceof Boolean) {
496             return LegacySQLTypeName.BOOLEAN;
497         } else if (value instanceof byte[]) {
498             return LegacySQLTypeName.BYTES;
499         } else if (value instanceof LocalDate) {
500             return LegacySQLTypeName.DATE;
501         } else if (value instanceof LocalTime) {
502             return LegacySQLTypeName.TIME;
503         } else if (value instanceof DateTime || value instanceof LocalDateTime) {
504             return LegacySQLTypeName.DATETIME;
505         } else {
506             throw new ExecutionException(ERR_UNSUPPORTED_DATA_TYPE_MSG)
507                     .withReason(String.format(ERR_UNSUPPORTED_DATA_TYPE_BIG_QUERY_REASON,
508                             value.getClass().getSimpleName()))
509                     .withResolution(ERR_UNSUPPORTED_DATA_TYPE_RESOLUTION);
510         }
511     }
