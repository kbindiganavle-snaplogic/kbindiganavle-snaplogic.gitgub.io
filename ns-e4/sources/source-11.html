478     /**
479      * Returns the private key (if exists) for the given alias in the given keystore
480      *
481      * @param keyStorePath
482      * @param keyStorePassphrase
483      * @param privateKeyAlias
484      * @return instance of {@link PrivateKey}
485      */
486     private PrivateKey getPrivateKeyFromKeyStore(String keyStorePath, char[] keyStorePassphrase,
487             String privateKeyAlias) {
488         try {
489             KeyStore keyStore = getKeyStoreInstance(keyStorePath, keyStorePassphrase);
490             KeyStore.ProtectionParameter protectionParameter =
491                     new KeyStore.PasswordProtection(keyStorePassphrase);
492             KeyStore.Entry entry = keyStore.getEntry(privateKeyAlias, protectionParameter);
493             if (entry instanceof KeyStore.PrivateKeyEntry) {
494                 return ((KeyStore.PrivateKeyEntry) entry).getPrivateKey();
495             } else {
496                 throw new ConfigurationException(ERR_PRIVATE_KEY_NOT_FOUND_MSG)
497                         .withResolution(ERR_PRIVATE_KEY_NOT_FOUND_RESOLUTION);
498             }
499         } catch (IOException e1) {
