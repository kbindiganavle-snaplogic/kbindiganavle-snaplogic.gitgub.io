249     private String convertValue(String field, Object object) {
250         String ret;
251 
252         if (object instanceof Timestamp) {
253             ret = formatTimestamp(field, (Timestamp)object);
254         } else if (object instanceof DateTime) {
255             String fieldType = dbFieldTypes.get(field);
256             switch (fieldType) {
257                 case DB_DATE_FIELD_TYPE:
258                     ret = ((DateTime) object)
259                             .toString(DateTimeFormat.forPattern(PATTERN_DATE_FORMAT));
260                     break;
261 
262                 case DB_TIME_FIELD_TYPE:
263                     ret = ((DateTime) object)
264                             .toString(DateTimeFormat.forPattern(PATTERN_TIME_FORMAT));
265                     break;
266 
267                 case DB_TIMESTAMP_FIELD_TYPE:
268                     ret = ((DateTime) object)
269                             .toString(DateTimeFormat.forPattern(PATTERN_TIMESTAMP_FORMAT));
270                     break;
271                 default:
272                     ret = object.toString();
273             }
274         } else {
275             ret = object.toString();
276         }
277         return escapeInput(ret);

468     private void preLoadAction() {
469         // Create temp file path for input data
470         tempDir = System.getProperty(TEMP_DIR_PROPERTY);
471         tempDir = tempDir.endsWith(File.separator) ? tempDir : tempDir.concat(File.separator);
472         uuid = UUID.randomUUID().toString();
473         dataFileName = String.format(FORMAT_DATA_FILE, uuid);
474         // format TPT Load script
475         script = tptLoadScript.formatScript(tempDir, dataFileName, logTableName,
476                 errorTableName1, errorTableName2, account.getDatabaseConfig(), inputViewFields,
477                 isWinPlex, deleteTableData);
478         obfuscatedScript = tptLoadScript.getObfuscatedScript();
479         LOG.debug(obfuscatedScript);
480 
481         scriptFileName = String.format(FORMAT_SCRIPT_FILE, UUID.randomUUID().toString());
482         scriptFileName = tempDir.concat(scriptFileName);
483         tbuildCommand = String.format(TBUILD_COMMAND_FORMAT, tbuildPath, scriptFileName,
484                 tempDir, uuid, tempDir);
485         // write the script into a file
486         BufferedWriter scriptWriter = null;
487         try {
488             scriptWriter = new BufferedWriter(new FileWriter(scriptFileName));
489             scriptWriter.write(script);
490         } catch (IOException ioException) {
491             SnapDataException dataException = new SnapDataException(ioException,
492                     ERR_CREATING_SCRIPT_FILE_MSG)
493                     .withReason(ioException.getMessage());
494             if (StringUtils.containsIgnoreCase(ioException.getMessage(), NOT_ENOUGH_SPACE)) {
495                 dataException = dataException.withResolution(ERR_NOT_ENOUGH_SPACE_RESOLUTION);
496             } else {
497                 dataException = dataException.withResolutionAsDefect();
498             }
499             errorViews.write(dataException, errorDoc);
500             return;
501         } finally {
502             IOUtils.closeQuietly(scriptWriter);
503         }
504 
505         // create FIFO special file
506         dataFileName = tempDir.concat(dataFileName);
507         if (!isWinPlex) {
508             try {
509                 Process process = Runtime.getRuntime()
510                         .exec(String.format(CMD_MKFIFO_FMT, dataFileName));
511                 process.waitFor();
512             } catch (IOException | InterruptedException e) {
513                 throw new ExecutionException(e, ERR_CREATE_DATA_FILE_MSG)
514                         .withReason(e.getMessage())
515                         .withResolution(RESOLUTION_ADDRESS_ISSUE);
516             }
517         }
518         if (!isWinPlex) {
519             // Reader of the named pipe must be created before writer.
520             loadAction();
521         }
522         if (isWinPlex) {
523             try {
524                 dataWriter = new BufferedWriter(new FileWriter(dataFileName));
525             } catch (IOException e) {
526                 throw new ExecutionException(e, ERR_CREATE_DATA_FILE_MSG)
527                         .withReason(e.getMessage())
528                         .withResolution(RESOLUTION_ADDRESS_ISSUE);
529             }
530         } else {
531             // The call to open a writer of a named pipe will block the thread until the named pipe
532             // has a reader, so we do it in a separate thread, give tbuild time to initialize and
533             // close the thread when thing goes wrong.
534             // After initiating the tbuild command in a separate process, wait for a given time and
535             // verify the status. If tbuild could not be invoked or dataWriter can't be initiated
536             // in the given timeout, set the loader state to failed.
537             Thread waitThread = new Thread(new Runnable() {
538                 @Override
539                 public void run() {
540                     try {
541                         dataWriter = new BufferedWriter(new FileWriter(dataFileName));
542                     } catch (IOException e) {
543                         throw new ExecutionException(e, ERR_CREATE_DATA_FILE_MSG)
544                                 .withReason(e.getMessage())
545                                 .withResolution(RESOLUTION_ADDRESS_ISSUE);
546                     }
547                 }
548             });
549             waitThread.start();
550             long waitTime = 0;
551             while (waitTime < TIMEOUT_FOR_OPENING_WRITTER_IN_MS) {
552                 try {
553                     Thread.sleep(INTERVAL_IN_MS);
554                     try {
555                         loaderProcess.exitValue();
556                         // Loader didn't run successfully, stop running
557                         loaderState = TeradataService.LoaderState.LOADER_STATE_FAILED;
558                         break;
559                     } catch (IllegalThreadStateException e) {
560                         // No op
561                     }
562                     if (!waitThread.isAlive()) {
563                         break;
564                     }
565                     waitTime += INTERVAL_IN_MS;
566                 } catch (InterruptedException e) {
567                     e.printStackTrace();
568                 }
569             }
570             if (waitTime >= TIMEOUT_FOR_OPENING_WRITTER_IN_MS) {
571                 // Timeout exceeded , stop the snap
572                 waitThread.interrupt();
573                 if (loaderState != TeradataService.LoaderState.LOADER_STATE_FAILED) {
574                     isSuccessful = false;
575                 }
576             } else {
577                 if (loaderState == TeradataService.LoaderState.LOADER_STATE_DEFAULT) {
578                     // success
579                     loaderState = TeradataService.LoaderState.LOADER_STATE_STARTED;
580                     deleteScriptFile();
581                 } else {
582                     // Loader failed initializing
583                     waitThread.interrupt();
584                 }
585             }
586         }
587     }

589     private void postLoadAction() {
590         if (loaderState != TeradataService.LoaderState.LOADER_STATE_DEFAULT) {
591             if (loaderState != TeradataService.LoaderState.LOADER_STATE_FAILED
592                     && (dataWriter == null || !isSuccessful)) {
593                 loaderProcess.destroy();
594                 return;
595             }
596         }
597 
598         if (!isErrorOccurred) {
599             Map<String, String> fastloadSummary = getFastloadSummary();
600             if (fastloadSummary != null) {
601                 Map<String, Object> fastloadOutput = new HashMap<>();
602                 fastloadOutput.put(TeradataService.TPTField.SCRIPT.toString(), obfuscatedScript);
603                 fastloadOutput.put(TeradataService.TPTField.STATUS.toString(), fastloadSummary);
604                 fastloadOutput.put(TeradataService.TPTField.OUT.toString(), output.toString());
605                 outputViews.write(documentUtility.newDocument(fastloadOutput));
606 
607                 if (writeErrorRecords) {
608                     writeErrorRecordsToErrorView(errorTableName1);
609                     writeErrorRecordsToErrorView(errorTableName2);
610                 }
611             } else if (StringUtils.isBlank(script) && autoCreateTable) {
612                 // snap requested table to be created but there is no input data.
613                 // in this case, TBUILD isn't run (no output) but table is created.
614                 // this is a valid condition. don't write to error view.
615             } else {
616                 String errorMessages = getErrorMessages();
617                 SnapDataException loadException = new SnapDataException(ERR_LOADING_DATA_REASON);
618                 if (StringUtils.containsIgnoreCase(errorMessages, NO_MORE_ROOM_IN_DATABASE)) {
619                     loadException.withReason(errorMessages)
620                             .withResolution(RESOLUTION_BROKEN_PIPE);
621                 } else {
622                     loadException.withReason(errorMessages)
623                             .withResolution(ERR_LOADING_DATA_RESOLUTION);
624                 }
625                 errorViews.write(loadException);
626             }
627         }

685     private void loadAction() {
686         try {
687             Runtime runtime = Runtime.getRuntime();
688             loaderProcess = runtime.exec(tbuildCommand);
689             if (isWinPlex) {
690                 loaderState = TeradataService.LoaderState.LOADER_STATE_STARTED;
691             }
692             if (loaderProcess != null) {
693                 // Runtime.exec() has limited buffer size for stdin and stdout streams, and
694                 // the external process will get stuck when the buffer is full, so we keep
695                 // cleaning up the stdout buffer to keep TPT Load from hanging.
696                 new Thread(new Runnable() {
697                     @Override
698                     public void run() {
699                         BufferedReader reader = null;
700                         try {
701                             reader = new BufferedReader(
702                                     new InputStreamReader(loaderProcess.getInputStream()));
703                             String line;
704                             while ((line = reader.readLine()) != null) {
705                                 LOG.debug(FASTLOAD_OUTPUT_FMT, line);
706                                 output.append(line).append(LINE_BREAK);
707                             }
708                         } catch (IOException ex) {
709                             // no op
710                         } finally {
711                             IOUtils.closeQuietly(reader);
712                         }
713                     }
714                 }).start();
715             }
716         } catch (Exception e5) {
717             SnapDataException dataException = new SnapDataException(e5,
718                     ERR_EXECUTE_FASTLOAD_CMD_MSG).withReason(e5.getMessage())
719                     .withResolution(ERR_EXECUTE_FASTLOAD_CMD_RESOLUTION);
720             errorViews.write(dataException, errorDoc);
721         }

754     @Override
755     public void execute() {
756         if (inputViews.isEmpty()) {
757             return;
758         }
759 
760         for (String viewName : getInputViewOrder()) {
761             for (Document document : inputViews.documents(inputViews.get(viewName))) {
762                 process(document, viewName);
763             }
764         }
765 
766         if (!firstRecord && autoCreateTable && schema != null) {
767             // We want to create the table even if there are no rows to be loaded.
768             createTableIfAbsent(null);
769         }
770 
771         if (isWinPlex && isSuccessful && dataWriter != null) {
772             loadAction();
773         }
774 
775         // Wait for fastload to finish if loading was started successfully.
776         if (loaderState == TeradataService.LoaderState.LOADER_STATE_STARTED) {
777             final CountDownLatch latch = new CountDownLatch(1);
778             Thread waitThread = new Thread(new Runnable() {
779                 @Override
780                 public void run() {
781                     try {
782                         loaderProcess.waitFor();
783                     } catch (InterruptedException e) {
784                         // ignore
785                     } finally {
786                         latch.countDown();
787                     }
788                 }
789             });
790             waitThread.start();
791             if (!isWinPlex) {
792                 // Close writer of the named pipe so that fastload won't wait for input forever.
793                 IOUtils.closeQuietly(dataWriter);
794             }
795             try {
796                 latch.await();
797                 loaderState = TeradataService.LoaderState.LOADER_STATE_SUCCEEDED;
798             } catch (InterruptedException e) {
799                 isSuccessful = false;
800             }
801         }
802     }
