276     protected void processDocument(Document document, String inputViewName) {
277         // To use "Limit offset" a valid "Order by" is required
278         if (getIntValue(limitOffset.eval(document)) != null && orderBy.size() == 0) {
279             throw new ConfigurationException(ERR_ORDERBY_MISSING)
280                     .withReason(ERR_ORDERBY_MISSING_REASON)
281                     .withResolution(RESOLUTION_ADDRESS_ISSUE);
282         }
283 
284         int viewOrderSize = viewOrder.size();
285         if (viewOrderSize > 1) {
286             // The user added the second (schema) view, so we need to write that out.
287             String schemaViewName = viewOrder.get(1);
288             DocumentOutputView schemaView = outputViews.getDocumentViewFor(schemaViewName);
289             Connection conn = jdbcOperations.acquireConnection(account);
290             Map<String, Object> schema = jdbcOperations.getRawTableMetaData(conn,
291                     account.getDatabaseConfig().getDatabaseName(), tableName);
292             schemaView.write(documentUtility.newDocument(schema));
293             try {
294                 // We'll only ever write one doc, so close down this view.
295                 schemaView.close();
296             } catch (IOException e) {
297                 throw new ExecutionException(e, UNABLE_TO_CLOSE_SCHEMA_VIEW, schemaViewName)
298                         .withReason(e.getMessage());
299             }
300             // Pop the schema view name so we don't enter this branch again.
301             viewOrder.remove(viewOrderSize - 1);
302         }
303         List<String> outputFieldNames = getColumnNamesFromTable(outputFields, OUTPUT_FIELD_PROP,
304                 document);
305         generateSelectBuilder(document);
306 
307         // Teradata sql does not use Limit or Offset convert to Teradata Sql format.
308         String query = ((TeradataOperations) jdbcOperations).convertSqlToTeradataSql(
309                 selectBuilder.toString());
310 
311         //Fetch all field values from the snap
312         final DatabaseConfig databaseConfig = account.getDatabaseConfig();
313         String outputDir = String.valueOf(outputDirExpr.<Object>eval(document));
314         URI uriDir = Utils.toUri(outputDir);
315         String outputFileName = String.valueOf(outputFileExpr.<Object>eval(document));
316         String outputFilePath = getOutputFilePath(outputDir, outputFileName);
317         String fileFormat = dataFormat.getOption();
318         String tdpID = databaseConfig.getHostName();
319         String userName = databaseConfig.getUsername();
320         String password = databaseConfig.getPassword();
321         String quotedData = quotedDataProp ? TPTScript.VALUE_YES : TPTScript.VALUE_NO;
322         String escapeDelimiter = escapeTextDelimiterExpr.eval
323                 (document) == null ? EMPTY_STRING :
324                 String.valueOf(escapeTextDelimiterExpr.<Object>eval(document));
325         String textDelimiter = textDelimiterExpr.eval(document) == null ?
326                 EMPTY_STRING : String.valueOf(textDelimiterExpr.<Object>eval(document));
327         String textDelimiterTypeData = textDelimiterType.getOption();
328         String dateFormat = tptDateFormat.getOption();
329         String escapeTextDelimiterTypeData = escpaeTextDelimiterType.getOption();
330 
331         //Setting up the TPTScript Attributes
332         Connection conn = jdbcOperations.acquireConnection(account);
333         TableMetaData tableMetaData = jdbcOperations.getTableMetaData(conn,
334                 account.getDatabaseConfig().getDatabaseName(), tableName);
335         TPTScript tptScript = new TPTScript(tableName, outputFieldNames, tableMetaData,
336                 tptDateFormat);
337         tptScript.setSelectAttrib(tdpID, userName, password, query, dateFormat);
338         tptScript.setFileAttrib(outputFilePath, TPTScript.FILE_MODE_WRITE, fileFormat);
339         tptScript.setFileTextAttrib(quotedData, escapeTextDelimiterTypeData, escapeDelimiter);
340         tptScript.setTextDelimiterAttrib(textDelimiterTypeData, textDelimiter);
341         if (HDFS.equals(uriDir.getScheme())) {
342             tptScript.setHdfsAttrib(uriDir, outputFileName);
343         }
344 
345         Map<TPTField, Object> results = service.tptFastExport(query, account, fileAction,
346                 tbuildPath, Utils.toPath(outputFilePath), tptScript);
347 
348         // creates a SnapDataException if the TPT script execution failed
349         // status code of 4 or less means successful execution
350         if ((Integer) results.get(TPTField.STATUS) > 4) {
351             String err = (String) results.get(TPTField.ERR);
352             String out = (String) results.get(TPTField.OUT);
353             String reason = String.format(FORMAT_REASON, StringUtils.isBlank(err) ? out : err,
354                     results.get(TPTField.SCRIPT));
355             SnapDataException snapDataException = new SnapDataException(ERR_SCRIPT_EXCUTE)
356                     .formatWith(results.get(TPTField.STATUS))
357                     .withReason(reason)
358                     .withResolution(RESOLUTION_ADDRESS_ISSUE);
359             errorViews.write(snapDataException, document);
360             return;
361         }
362 
363         // if the execution is successful, convert results to map using string instead of enum.
364         results.remove(TPTField.ERR);
365         Map<String, Object> data = new LinkedHashMap<>();
366         for (Map.Entry<TPTField, Object> entry : results.entrySet()) {
367             data.put(entry.getKey().getSchema(), entry.getValue());
368         }
369 
370         // create document.
371         Document doc = documentUtility.newDocumentFor(document, data);
372         // write document to output view.
373         if (viewOrder.size() > 0) {
374             String schemaViewName = viewOrder.get(0);
375             DocumentOutputView schemaView = outputViews.getDocumentViewFor(schemaViewName);
376             schemaView.write(doc);
377         }
378     }

400     private String getOutputFilePath(String outputDir, String outputFileName) {
401         if (StringUtils.isBlank(outputDir)) {
402             outputDir = URL_SEPARATOR;
403         }
404         if (outputDir.startsWith(URL_PROTOCOL_FILE) &&
405                 outputDir.length() > URL_PROTOCOL_FILE.length()) {
406             outputDir = outputDir.substring(URL_PROTOCOL_FILE.length());
407         }
408         return outputDir.endsWith(URL_SEPARATOR) ? StringUtils.join(outputDir, outputFileName) :
409                 StringUtils.join(outputDir, URL_SEPARATOR, outputFileName);
