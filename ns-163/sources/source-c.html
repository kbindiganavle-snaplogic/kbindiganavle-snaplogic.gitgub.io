504     protected void waitForProcess(final Process process, String script) throws IOException, InterruptedException {
505         //when configuration exception the process would not be initialized
506         if (process == null) {
507             return;
508         }
509         //all data is successfully written to the file, close the file to signal the tbuild to
510         // finish up the process
511         if (dataWriter != null) {
512             IOUtils.closeQuietly(dataWriter);
513 
514         } else {
515             //Not able to get the dataWritter, So the process would wait indefinitely, so kill
516             // the process. and show a error
517             process.destroy();
518         }
519         if (fileWriter != null) {
520             IOUtils.closeQuietly(fileWriter);
521         }
522 
523         //Update the status of the loaderProcess,this use to update the error view
524         final int[] status = new int[1];
525         status[0] = 99;
526         // Wait for the TPT to finish its job in another thread so that the snap can be
527         // stopped immediately when manually stopping the pipeline.
528         final CountDownLatch latch = new CountDownLatch(1);
529         Thread waitThread = new Thread(new Runnable() {
530             @Override
531             public void run() {
532                 try {
533                     status[0] = process.waitFor();
534                 } catch (InterruptedException e) {
535                     // ignore
536                 } finally {
537                     latch.countDown();
538                 }
539             }
540         });
541         waitThread.start();
542         if (!isWinPlex) {
543             // Close writer of the named pipe so that sqlldr won't wait for input forever.
544             IOUtils.closeQuietly(dataWriter);
545         }
546         try {
547             latch.await();
548             processOutput(script, status[0]);
549         } catch (InterruptedException e) {
550             processOutput(script, status[0]);
551         }
552 
553     }

555     private void processOutput(final String script, final int status) {
556         Map<String, String> tptUpdateSummary = getTptUpdateSummary(cmdOut.toString());
557         String err = cmdErr.toString();
558         String out = cmdOut.toString();
559         if (status > 4) {

561             SnapDataException snapDataException = new SnapDataException(ERR_SCRIPT_EXCUTE)
562                     .formatWith(status)
563                     .withReason(reason);
564             if (out.contains(TPT_MULTILOAD_ERROR_CODE)) {
565                 snapDataException.withResolution(ERR_WRITING_INPUT_DATA_RESOLUTION);
566             } else {
567                 snapDataException.withResolution(RESOLUTION_ADDRESS_ISSUE);
568             }
569             errorViews.write(snapDataException);
570         } else {
571             final Map<TeradataService.TPTField, Object> results = new LinkedHashMap<>();
572             results.put(TeradataService.TPTField.OUT, out);
573             results.put(TeradataService.TPTField.ERR, err);
574             results.put(TeradataService.TPTField.SUMMARY, tptUpdateSummary);
575             results.put(TeradataService.TPTField.SCRIPT, script);

613         // inserting a delay to complete any pending or in-processing statements of Teradata
