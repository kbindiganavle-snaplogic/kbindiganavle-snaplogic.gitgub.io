88             if (config.getClassLoader() != null) {

100             throw new ClassNotFoundException("unable to find " + TDCH_IMPORT_CLASSNAME);
101         }

298      * Run TDCH executable.
299      *
300      * Implementation note: the Teradata documentation refers to "hadoop jar ..."
301      * instead of calling java directly. This shouldn't matter since the "hadoop"
302      * executable is just a thin shell that sets up the environment when executing
303      * a jar. Calling java directly could potentially miss something but it avoids
304      * the need to locate the "hadoop" executable.
305      *
306      * @param args
307      * @return
308      * @throws ConfigurationException
309      * @throws IOException
310      */
311     public Map<TeradataExportToHdfsField, Object> runTdchExecutable(List<String> args,
312             DatabaseConfig dbConfig) throws ConfigurationException, IOException {
313 
314         // create necessary configuration information.
315         final File configdir = makeConfigDirectoryForTdchExecutable();
316         final String classpath = createClasspathForTdchExecutable(configdir, dbConfig);
317         final List<String> envp = makeEnvironmentForTdchExecutable(classpath);
318 
319         // determine location of java binary.
320         final String javaHome = System.getenv("JAVA_HOME");
321         final String java = (javaHome != null) ? javaHome + "/bin/java" : "/usr/bin/java";
322 
323         // prepend location of java binary and tool to argument list.
324         final List<String> cmdlist = new ArrayList<>();
325         cmdlist.add(java);
326         cmdlist.add("com.teradata.hadoop.tool.TeradataImportTool");
327         cmdlist.addAll(args);
328         final String[] cmdarray = cmdlist.toArray(new String[0]);
329 
330         // execute TDCH
331         final Map<TeradataExportToHdfsField, Object> results = runTdchExecutable(cmdarray,
332                 envp.toArray(new String[0]));
333 
334         // list classpath
335         results.put(TeradataExportToHdfsField.CLASSPATH, classpath.toString());
336 
337         // list environment variables provided to process.
338         StringWriter w = new StringWriter();
339         for (String env : envp) {
340             w.write(env + "\n");
341         }
342         results.put(TeradataExportToHdfsField.ENVIRONMENT, w.toString());
343 
344         return results;
345     }

515     List<String> makeEnvironmentForTdchExecutable(String classpath) {
516         final List<String> envp = new ArrayList<>();
517

519 
520         final List<String> keys = new ArrayList<>(System.getenv().keySet());
521         Collections.sort(keys);
522         for (String key : keys) {
523             if (key.startsWith("CDH_") || key.startsWith("HADOOP_") ||
524                     key.startsWith("JAVA_") || key.startsWith("NM_") || key.startsWith("YARN_")) {
525                 envp.add(key + "=" + System.getenv(key));
526             }
527         }

533      * Actually execute TDCH executable.
534      *
535      * @param cmdarray
536      * @param envp
537      * @return
538      */
539     Map<TeradataExportToHdfsField, Object> runTdchExecutable(String cmdarray[], String[] envp)
540             throws IOException {
541 
542         final Process p = Runtime.getRuntime().exec(cmdarray, envp);
543         final ProcessStreamReader r1 = new ProcessStreamReader(p, p.getInputStream());
544         final ProcessStreamReader r2 = new ProcessStreamReader(p, p.getErrorStream());
545         r1.start();
546         r2.start();
547         int retCode = 0;
548         try {
549             retCode = p.waitFor();
550             r1.interrupt();
551             r1.join(200L);
552             r2.interrupt();
553             r2.join(200L);
554         } catch (InterruptedException e) {
555             // ignore
556         }
557 
558         LOGGER.info("TDCH return value: {}", retCode);
559 
560         final Map<TeradataExportToHdfsField, Object> results = new LinkedHashMap<>();
561         results.put(TeradataExportToHdfsField.OUT, r1.toString());
562         results.put(TeradataExportToHdfsField.ERR, r2.toString());
563         results.put(TeradataExportToHdfsField.STATUS, retCode);
564 
565         return results;
566     }
