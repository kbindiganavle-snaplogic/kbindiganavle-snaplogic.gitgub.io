176     @Override
177     public void write(byte[] array, int offset, int length) throws IOException {
178         if (byteBuffer.remaining() < length) {
179             if (byteBuffer.position() > 0) {
180                 startUploadThread(byteBuffer, false);
181                 byteBuffer = getByteBuffer();
182             } else {
183                 // Since the current buffer is empty and the data array is larger than the
184                 // buffer, the current buffer should be replaced by a new buffer. This case
185                 // seldom happens, but should be covered.
186                 ByteBuffer buffer = ByteBuffer.wrap(array, offset, length).duplicate();
187                 replaceByteBuffer(buffer);
188                 startUploadThread(buffer, false);
189                 return;
190             }
191         }
192         byteBuffer.put(array, offset, length);
193     }

195     @Override
196     public void close() throws IOException {
197         if (isClosed) {
198             return;
199         }
200         try {
201             if (!isFailed) {
202                 if (byteBuffer.position() >= 0 || !tasks.isEmpty()) {
203                     if (initRequest == null) {
204                         if (byteBuffer.position() > 0 || uploadRequest.isWriteEmptyFile()) {
205                             // If multipart file upload hasn't been started, the target file is
206                             // smaller than one buffer or empty. putObject() should be used for
207                             // better performance.
208                             putObjectFailSafe();
209                         }
210                         return;
211                     }
212                     // If byteBuffer.position() is positive this means there is more data to upload.
213                     // Upload it and signal it is the last part.
214                     // Non-empty task list and empty byteBuffer means all data has already been
215                     // written. Write an empty last part to signal the last part to the S3 client.
216                     startUploadThread(byteBuffer, true);
217                 }
218             }
219 
220             // wait for multipart upload to complete
221             List<PartETag> partETags = tasks
222                     .stream()
223                     .map(task -> {
224                         UploadPart uploadPart = task.join();
225                         bytesUploaded += uploadPart.partSize;
226                         PartETag partETag = uploadPart.getPartEtag();
227                         return partETag;
228                     })
229                     .collect(Collectors.toList());
230             if (initRequest != null) {
231                 completeMultipartUpload(partETags);
232             }
233         } finally {
234             super.close();
235             if (bufferQueue != null) {
236                 bufferQueue.clear();
237             }
238             byteBuffer.clear();
239             byteBuffer = null;
240             if (executor != null) {
241                 if (isFailed) {
242                     abortMultipartUpload();
243                 }
244                 shutdownThreadPool(executor);
245             }
246             isClosed = true;
247         }
248     }

250     private void putObjectFailSafe() throws IOException {
251         try {
252             // byteBuffer.array() returns the entire array.
253             // byteBuffer should be copied to a byte array of actual data size.
254             byte[] bytes = new byte[byteBuffer.position()];
255             byteBuffer.rewind();
256             byteBuffer.get(bytes);
257             // FailSafe retries
258             ContextualRunnable runnable = context -> uploadSmallFile(bytes);
259             CheckedConsumer<Throwable> failedAttemptListener = e -> LOGGER.warn(
260                     "{} - Failed to upload file: s3:///{}/{}, {}",
261                     S3_RETRY_SIGNATURE, bucketName, key, e.getMessage());
262             ContextualResultListener<Object, Throwable> retryLogListener =
263                     Utils.retryLogListener(S3_RETRY_SIGNATURE);
264             failsafeExecute(runnable, failedAttemptListener, retryLogListener,
265                     s3Service.getMaxRetries(), 1);
266         } catch (Throwable t) {
267             throw new IOException(String.format(ERR_UPLOAD_FILE, bucketName, key), t);
268         }
269     }

308     private ByteBuffer waitForFreeBuffer() throws IOException {
309         ByteBuffer byteBuffer = null;
310         try {

312         } catch (InterruptedException e) {

322     private void startUploadThread(ByteBuffer buffer, boolean isLastPart) throws IOException {
323         if (initRequest == null) {
324             try {
325                 initMultipartUpload();
326             } catch (AmazonS3Exception e) {
327                 isFailed = true;
328                 String reason = acl == null ? e.getMessage() : REASON_ACL;
329                 throw new IOException(String.format(ERR_CREATE_MULTIPART_UPLOAD, bucketName, key,
330                         reason), e);
331             }
332         }
333         if (partNumber > MAX_PART_NUMBER) {
334             isFailed = true;
335             throw new IOException(String.format(ERR_MAX_UPLOAD_PARTS, MAX_PART_NUMBER,
336                     bufferSize, MAX_PART_NUMBER));
337         }
338         if (isFailed) {
339             close();
340             return;
341         }
342         // start part upload thread
343         UploadPart uploadPart = new UploadPart(amazonS3, bucketName, key, uploadId, buffer,
344                 partNumber++, bufferQueue, isLastPart, this, this);
345         CompletableFuture<UploadPart> task = CompletableFuture.supplyAsync(uploadPart, executor);
346         // save the upload thread info
347         tasks.add(task);
348     }

363     protected void shutdownThreadPool(final ExecutorService threadPool) {
364         // first, no more new tasks accepted
365         threadPool.shutdown();
366         try {
367             // wait for some time for the scheduled tasks to complete
368             if (!threadPool.awaitTermination(DEFAULT_THREAD_POOL_COMPLETE_TIME,
369                     DEFAULT_THREAD_POOL_COMPLETE_TIME_UNIT)) {
370                 // still tasks didn't complete, so cancel them
371                 threadPool.shutdownNow();
372                 // wait for some time for cancelling the scheduled tasks
373                 if (!threadPool.awaitTermination(DEFAULT_THREAD_POOL_CANCEL_TIME,
374                         DEFAULT_THREAD_POOL_CANCEL_TIME_UNIT)) {
375                     LOGGER.error("Thread pool did not shutdown gracefully");
376                 }
377             }
378         } catch (InterruptedException e1) {
379             // attempt to cancel the tasks again as the thread got interrupted
380             threadPool.shutdownNow();
381         }
382     }

384     private void initUploadMetadata() throws IOException {
385         objectMetadata = new ObjectMetadata();
386         Map<String, Object> headerMap = uploadRequest.getHeaderMap();
387         if (headerMap != null) {
388             headerMap.entrySet().stream()
389                     .filter(p -> p.getValue() != null)
390                     .forEach(p -> {
391                         String key = p.getKey();
392                         String value = p.getValue().toString();
393                         if (CONTENT_LENGTH.equalsIgnoreCase(key)) {
394                             objectMetadata.setContentLength(Long.valueOf(value));
395                         } else if (CONTENT_TYPE.equalsIgnoreCase(key)) {
396                             objectMetadata.setContentType(value);
397                         } else if (CONTENT_DISPOSITION.equalsIgnoreCase(key)) {
398                             objectMetadata.setContentDisposition(value);
399                         } else if (HttpHeaders.CONTENT_ENCODING.equalsIgnoreCase(key)) {
400                             objectMetadata.setContentEncoding(value);
401                         }
402                     });
403         }
404         if (StringUtils.isEmpty(objectMetadata.getContentType())) {
405             objectMetadata.setContentType(MediaType.OCTET_STREAM.toString());
406         }
407 
408         // process S3 encryption
409         S3ClientConfig s3ClientConfig = s3Service.getS3ClientConfig();
410         encryptType = S3EncryptionType.toEnum(s3ClientConfig.getEncryptionType());
411         switch (encryptType) {
412             case SSE:
413                 objectMetadata.setSSEAlgorithm(ObjectMetadata.AES_256_SERVER_SIDE_ENCRYPTION);
414                 break;
415             case SSE_KMS:
416                 String kmsKey = s3ClientConfig.getKmsKey();
417                 sseAwsKeyManagementParams = new SSEAwsKeyManagementParams(kmsKey);
418                 break;
419         }
420         acl = uploadRequest.getAcl();
421         if (acl != null) {
422             try {
423                 acl.setOwner(amazonS3.getS3AccountOwner());
424             } catch (Exception e) {
425                 String reason = e instanceof AmazonS3Exception ?
426                         String.format(FORMAT_EXTENDED_REQUEST_ID, e.getMessage(),
427                                 ((AmazonS3Exception) e).getExtendedRequestId()) :
428                         e.getMessage();
429                 throw new IOException(String.format(ERR_ACCOUNT_OWNER, reason), e);
430             }
431         }
432         // process user-defined metadata
433         if (MapUtils.isNotEmpty(uploadRequest.getUserDefinedMetadataMap())) {
434             objectMetadata.setUserMetadata(uploadRequest.getUserDefinedMetadataMap());
435         }
436         // init object tagging
437         if (uploadRequest.getObjectTagMap() != null) {
438             List<Tag> objectTags = uploadRequest.getObjectTagMap()
439                     .entrySet()
440                     .stream()
441                     .map(p -> new Tag(p.getKey(), p.getValue()))
442                     .collect(Collectors.toList());
443             if (CollectionUtils.isNotEmpty(objectTags)) {
444                 objectTagging = new ObjectTagging(objectTags);
445             }
446         }
447     }

482         @Override
483         public UploadPart get() {
484             //read bytes till data
485             byte[] data = new byte[buffer.position()];
486             //flip from 'write' mode to 'read' mode
487             buffer.flip();
488             buffer.get(data,  buffer.position(), buffer.limit());
489             try (ByteArrayInputStream inputStream = new ByteArrayInputStream(data)) {
490                 // create request to upload a part
491                 UploadPartRequest uploadRequest = new UploadPartRequest()
492                         .withBucketName(bucketName)
493                         .withKey(key)
494                         .withUploadId(uploadId)
495                         .withPartNumber(partNumber)
496                         .withInputStream(inputStream)
497                         .withLastPart(isLastPart)
498                         .withPartSize(partSize)
499                         .withGeneralProgressListener(progressListener);
500                 // upload part
501                 partETag = s3.uploadPart(uploadRequest)
502                         .getPartETag();
503                 LOGGER.debug("Uploaded part size: {} for {}, partNum: {}",
504                         data.length, key, partNumber);
505             } catch (SdkClientException | IOException e) {
506                 parent.isFailed = true;
507                 LOGGER.debug("S3 Upload part failed for bucket: {}, key: {}, part number: {}",
508                         bucketName, key, partNumber);
509                 throw new CompletionException(String.format(ERR_UPLOAD_PART,
510                         bucketName, key, parent.bytesUploaded, partNumber), e);
511             } finally {
512                 buffer.clear();
513                 try {
514                     queue.offer(buffer, BLOCKING_QUEUE_TIMEOUT, TimeUnit.HOURS);
515                 } catch (InterruptedException e) {
516                     throw new CompletionException(e.getMessage(), e);
517                 }
518             }
519             return this;
520         }

546     private void initMultipartUpload() {
547         // initiate multipart upload request
548         InitiateMultipartUploadRequest initRequest =
549                 new InitiateMultipartUploadRequest(bucketName, key);
550         // set object metadata
551         initRequest.setObjectMetadata(objectMetadata);
552         // set object tags
553         if (objectTagging != null) {
554             initRequest.setTagging(objectTagging);
555         }
556         if (sseAwsKeyManagementParams != null) {
557             initRequest.setSSEAwsKeyManagementParams(sseAwsKeyManagementParams);
558         }
559         // set ACL to InitiateMultipartUploadRequest
560         if (acl != null) {
561             initRequest.setAccessControlList(acl);
562         }
563         if (uploadRequest.getCannedACL() != null) {
564             initRequest.setCannedACL(uploadRequest.getCannedACL());
565         }
566         uploadId = amazonS3.initiateMultipartUpload(initRequest).getUploadId();
567         // initialize fixed thread pool
568         executor = (ThreadPoolExecutor) Executors.newFixedThreadPool(threadPoolSize);
569         bufferQueue = new LinkedBlockingDeque<>(threadPoolSize);
