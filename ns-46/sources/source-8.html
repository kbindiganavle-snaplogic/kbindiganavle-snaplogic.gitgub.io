246          */
247         String nonProxyHosts = System.getProperty(HTTP_NON_PROXY);
248         String proxyHost = System.getProperty(HTTPS_PROXY);
249         String proxyPort = System.getProperty(HTTPS_PORT);
250         String username = System.getProperty(HTTPS_PROXY_USER);
251         String password = System.getProperty(HTTPS_PROXY_PASSWORD);
252         if (StringUtils.isBlank(proxyHost) || StringUtils.isBlank(proxyPort)) {
253             proxyHost = System.getProperty(HTTP_PROXY);
254             proxyPort = System.getProperty(HTTP_PORT);
255             username = System.getProperty(HTTP_PROXY_USER);
256             password = System.getProperty(HTTP_PROXY_PASSWORD);
257         }
258         if (isNoneBlank(proxyHost, proxyPort)) {
259             config.withProxyHost(proxyHost)
260                     .withProxyPort(Integer.parseInt(proxyPort));
261             if (!StringUtils.isBlank(username) && !StringUtils.isBlank(password)) {
262                 config.withProxyUsername(username)
263                         .withProxyPassword(password);
264             }
265         }

267             config.withNonProxyHosts(nonProxyHosts);

270         RetryPolicy retryPolicy = PredefinedRetryPolicies

272         return config

274                 .withRetryPolicy(retryPolicy)
275                 .withSocketTimeout(s3SocketTimeoutSecs * 1000);

302     private AWSCredentialsProvider getCredentialsProviderFromAssumingRole(String roleArn,
303             String externalId, AWSCredentialsProvider credentialsProvider) {
304         Region defaultRegion = Regions.getCurrentRegion();
305         String region = defaultRegion == null ? Regions.DEFAULT_REGION.getName() :
306                 defaultRegion.getName();
307         ClientConfiguration configuration = new ClientConfiguration();
308         String apnHeader = getApnUserAgentHeaderFor(COMPANY_NAME, PROJECT_NAME, VERSION);
309         configuration.setUserAgentPrefix(apnHeader);
310         AWSSecurityTokenService sts =
311                 AWSSecurityTokenServiceClientBuilder
312                         .standard()
313                         .withRegion(region)
314                         .withCredentials(credentialsProvider)
315                         .withClientConfiguration(configuration)
316                         .build();
317         String sessionName = String.format(ASSUME_ROLE_SESSION_NAME_FMT, this.toString());
318         AssumeRoleRequest assumeRequest = new AssumeRoleRequest()
319                 .withRoleArn(roleArn)
320                 .withRoleSessionName(sessionName);
321         if (StringUtils.isNotEmpty(externalId)) {
322             assumeRequest = assumeRequest.withExternalId(externalId);
323         }
324         LOG.debug("Assuming role {} with session name: {}", roleArn, sessionName);
325         try {
326             AssumeRoleResult assumeRoleResult = findSessionDuration(sts, assumeRequest);
327             LOG.warn("maxSessionDuration found: {} seconds", durationSeconds);
328             Credentials credentials = assumeRoleResult.getCredentials();
329             timestampCreated = System.currentTimeMillis();
330             return new AWSStaticCredentialsProvider(new BasicSessionCredentials(
331                     credentials.getAccessKeyId(),
332                     credentials.getSecretAccessKey(),
333                     credentials.getSessionToken()));
334         } catch (Throwable th) {
335             throw new SnapDataException(th, ERR_FAIL_TO_ASSUME_ROLE)
336                     .withReason(th.toString())
337                     .withResolution(RESOLUTION_ASSUME_ROLE);
338         }
339     }

341     @VisibleForTesting
342     AssumeRoleResult findSessionDuration(AWSSecurityTokenService sts,
343             AssumeRoleRequest assumeRequest) {
344         if (durationSeconds > 0) {
345             // If the max session duration has already been found, re-use it.
346             return sts.assumeRole(assumeRequest.withDurationSeconds(durationSeconds));
347         }
348         // In most cases, the maxSessionDuration set in the role is at default, which is 1 hour.
349         // Therefore, first try 2 hours for sessionDuration. If it fails, use 1 hour for
350         // the sessionDuration. If it succeeds, start the binary search trying 12 hours, next.
351         int hours = 2;
352         while (true) {
353             Pair<Object, Object> pair = SESSION_DURATION_LIST.get(hours);
354             try {
355                 durationSeconds = hours * SECONDS_IN_HOUR;
356                 assumeRequest.withDurationSeconds(durationSeconds);
357                 AssumeRoleResult assumeRoleResult = sts.assumeRole(assumeRequest);
358                 Object success = pair.getLeft();
359                 if (R.equals(success)) {
360                     return assumeRoleResult;
361                 } else {
362                     hours = (int) success;
363                 }
364             } catch (AWSSecurityTokenServiceException e) {
365                 if (StringUtils.containsIgnoreCase(e.getMessage(), EXCEEDS_MAX_SESSION_DURATION)) {
366                     Object fail = pair.getRight();
367                     if (fail instanceof List) {
368                         hours = (int) ((List) fail).get(0);
369                         durationSeconds = hours * SECONDS_IN_HOUR;
370                         return sts.assumeRole(assumeRequest.withDurationSeconds(durationSeconds));
371                     } else {
372                         hours = (int) fail;
373                     }
374                 } else {
375                     throw e;
376                 }
377             }
378         }
379     }

382      * Creates an Amazon S3 client. If client-side encryption is used then this will create
383      * an encryption client using the KMS key.
384      * @param credentialsProvider - AWSCredentialsProvider object
385      * @param clientConfiguration - ClientConfiguration object
386      * @param s3ClientConfig      - S3ClientConfig object
387      * @param region              - AWS region name
388      * @return AmazonS3 object
389      */
390     private AmazonS3 createClient(AWSCredentialsProvider credentialsProvider,
391             ClientConfiguration clientConfiguration, S3ClientConfig s3ClientConfig, String region) {
392         S3EncryptionType encryptType = S3EncryptionType.toEnum(s3ClientConfig.getEncryptionType());
393         AwsClientBuilder builder;
394         if (S3EncryptionType.CSE_KMS.equals(encryptType)) {
395             // client-side KMS encryption
396             String kmsKey = s3ClientConfig.getKmsKey();
397             String kmsRegion = s3ClientConfig.getKmsRegion();
398             if (StringUtils.isBlank(kmsKey)) {
399                 //If no KMS key is provided, supply a default one. In the case where the encryption
400                 //client is only reading s3 objects it does not need a KMS key. The client is smart
401                 //enough to look at the object metadata and download the KMS key/certificate.
402                 //However, there is no way to create an encryption client in the AWS SDK without
403                 //supplying a KMS key. We supply a default value so the client can be used for
404                 //read-only scenarios without forcing the end user to supply a key that won't be
405                 //used.
406                 kmsKey = DEFAULT_CSE_KMS;
407             }
408             if (StringUtils.isBlank(kmsRegion)) {
409                 //If no KMS region is provided, use the default us-east-1
410                 kmsRegion = Regions.US_EAST_1.getName();
411             }
412             KMSEncryptionMaterialsProvider kms = new KMSEncryptionMaterialsProvider(kmsKey);
413             // minimal crypto configuration in V2; default CryptoMode.StrictAuthenticatedEncryption
414             CryptoConfigurationV2 cryptoConfig = new CryptoConfigurationV2()
415                     .withAwsKmsRegion(RegionUtils.getRegion(kmsRegion))
416                     // The following setting allows the client to read V1 encrypted objects
417                     .withCryptoMode(CryptoMode.AuthenticatedEncryption);
418             // create AmazonS3Client with encryption V2
419             AmazonS3EncryptionClientV2Builder encryptionBuilder = AmazonS3EncryptionClientV2
420                     .encryptionBuilder()
421                     .withCredentials(credentialsProvider)
422                     .withClientConfiguration(clientConfiguration)
423                     .withEncryptionMaterialsProvider(kms)
424                     .withCryptoConfiguration(cryptoConfig);
425             if (isEndpoint(region)) {
426                 encryptionBuilder.withPathStyleAccessEnabled(true);
427             } else {
428                 encryptionBuilder.withForceGlobalBucketAccessEnabled(true);
429             }
430             builder = encryptionBuilder;
431         } else {
432             // not encrypted on client-side
433             AmazonS3ClientBuilder s3ClientBuilder = AmazonS3ClientBuilder.standard()
434                     .withCredentials(credentialsProvider)
435                     .withClientConfiguration(clientConfiguration);
436             if (isEndpoint(region)) {
437                 s3ClientBuilder.withPathStyleAccessEnabled(true);
438             } else {
439                 s3ClientBuilder.withForceGlobalBucketAccessEnabled(true);
440             }
441             builder = s3ClientBuilder;
442         }
443         if (StringUtils.isBlank(region) || DEFAULT.equalsIgnoreCase(region)) {
444             builder.withRegion(Regions.DEFAULT_REGION);
445         } else if (isEndpoint(region)) {
446             // If signing region is not null, region must be a VPC S3 endpoint.
447             String signingRegion = parseSigningRegion(region);
448             builder.withEndpointConfiguration(
449                     new AwsClientBuilder.EndpointConfiguration(region, signingRegion));
450         } else {
451             builder.withRegion(region);
452         }
453         return (AmazonS3) builder.build();
454     }

498     private static String parseSigningRegion(String region) {
499         // VPC S3 endpoint looks like:
500         //     bucket.vpce-028b7814794578709-vu0vvauy.s3.us-west-2.vpce.amazonaws.com
501         // signing region can be parsed between ".s3." and ".vpce.amazonaws.com"
502         int index1 = region.indexOf(DOT_S3_DOT);
503         int index2 = region.indexOf(VPCE_AMAZON_AWS);
504         return index1 > 0 && index2 > 0 && index1 + DOT_S3_DOT.length() < index2 ?
505                 region.substring(index1 + DOT_S3_DOT.length(), index2) : null;
506     }
