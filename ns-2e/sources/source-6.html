74     @Inject
75     public SnapObjectMapper() {
76         super();
77         super.configure(JsonGenerator.Feature.AUTO_CLOSE_TARGET, false);
78         super.configure(SerializationFeature.WRITE_DATES_AS_TIMESTAMPS, false);
79         super.configure(SerializationFeature.FAIL_ON_EMPTY_BEANS, false);
80         // Read all non-floating as BigInteger.
81         super.enable(DeserializationFeature.USE_BIG_INTEGER_FOR_INTS);
82         // Read all floating as BigDecimal.
83         super.enable(DeserializationFeature.USE_BIG_DECIMAL_FOR_FLOATS);
84         // Do not include colon in timezone offset for backward compatibility
85         super.setDateFormat(StdDateFormat.instance.withColonInTimeZone(false));
86         super.registerModule(new JodaModule());
87     }

94             addSerializer(LocalTime.class, new LocalTimeSerializer());
95         }
