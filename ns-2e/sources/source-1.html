44 /**
45  * Immutable java.nio.file.Path facade for URLs. This is a filesystem-agnostic class that
46  * understands how to manipulate URL elements but does nothing more than that. It should be
47  * used by all of our FileSystem implementations since we have unit tests to verify the
48  * correct behavior.
49  *
50  * Important note: this class should only handle path manipulation. It is not necessary for
51  * it to do more than that. All of the 'smarts' should be in the FileSystem implementation.
52  *
53  * For proper encapsulation the concrete classes should be thin wrappers that do little more
54  * than wrap the constructors.
55  *
56  * Implementation note: this is called an 'abstract' class but isn't actually an abstract
57  * class due to the HadoopDirectoryStream.
58  *
59  * Implementation note: the java.nio javadoc states that an "absolute" path is one
60  * that contains a scheme (and hostname), not one that starts with a leading slash.
61  * For our purposes we assume that a 'scheme' will always be present and only check
62  * for the presence of a hostname.
63  *
64  * @param <FS> concrete filesystem
65  */

502     public Path normalize() {
503         final Stack<String> stack = new Stack<>();
504         for (String element : elements) {
505             switch(element) {
506                 // current directory - noop
507                 case ".":
508                     break;
509 
510                 // parent element - pop one element.
511                 case "..":
512                     if (stack.isEmpty()) {
513                         throw new IllegalStateException(ERR_TOO_MANY_PARENT_ELEMENTS_MSG);
514                     }
515                     stack.pop();
516                     break;
517 
518                 // anything else - push one element.
519                 default:
520                     stack.push(element);
521             }
522         }
523 
524         // convert stack to array of strings.
525         final String[] elements = new String[stack.size()];
526         for (int i = 0; i < stack.size(); i++) {
527            elements[i] = stack.get(i);
528         }
529 
530         return newAbsolutePath(elements);
531     }

760     /**
761      * Compare two paths. Note that this does not recognize that multiple hostnames can be
762      * used within a single cluster. It might make more sense to order paths by hostname last.
763      *
764      * @see java.nio.file.Path#compareTo(java.nio.file.Path)
765      */
766     @Override
767     public int compareTo(Path other) {
768         if ((other == null) || !(other instanceof AbstractUrlPath)) {
769             return -1;
770         }
771 
772         final AbstractUrlPath<?> that = (AbstractUrlPath<?>) other;
773 
774         checkSameHost(that);
775 
776         // always sort absolute and relative paths separately.
777         if (this.isAbsolute() && !that.isAbsolute()) {
778             return -1;
779         } else if (!this.isAbsolute() && that.isAbsolute()) {
780             return 1;
781         }
782 
783         // compare elements piecewise. We can't just call getPath() and compare that
784         // since the collation sequence will be off for chars between ' ' and '/'.
785         for (int i = 0; i < Math.min(this.elements.length, that.elements.length); i++) {
786             final int r = this.elements[i].compareTo(that.elements[i]);
787             if (r < 0) {
788                 return -1;
789             } else if (r > 0) {
790                 return 1;
791             }
792         }
793 
794         // either the paths are the same or one is longer than the other.
795         final int r = this.elements.length - that.elements.length;
796         if (r < 0) {
797             return -1;
798         } else if (r > 0) {
799             return 1;
800         }
801         return 0;
802     }

876     /**
877      * Merge 'first' element and optional additional elements.
878      */
879     static String[] merge(String scheme, String host, int port, String first, String... more) {
880         // if a hostname is specified verify it matches.
881         final String prefix = schemeToPrefix(scheme);
882         if (first.startsWith(prefix)) {
883             // verify host and port match
884             String actualFirst = first.substring(prefix.length());
885             final String[] tmp = actualFirst.split(SEPARATOR);
886             final String[] hostport = tmp[0].split(":");
887             if ((host == null) || host.equals(hostport)) {
888                 throw new InvalidPathException(ERR_HOSTS_DO_NOT_MATCH_MSG);
889             }
890             if ((hostport.length == 2) && (port != Integer.parseInt(hostport[1]))) {
891                 throw new InvalidPathException(ERR_PORTS_DO_NOT_MATCH_MSG);
892             }
893 
894             final String[] tmp2 = new String[tmp.length - 1];
895             System.arraycopy(tmp, 1, tmp2, 0, tmp.length - 1);
896             return merge(tmp2, more);
897         }
898 
899         String actualFirst = first;
900         if (SEPARATOR.equals(first)) {
901             // do nothing
902         } else if (first.startsWith(SEPARATOR)) {
903             actualFirst = first.substring(SEPARATOR.length());
904         } else {
905             actualFirst = first;
906         }
907 
908         if ((actualFirst != null) && actualFirst.contains(SEPARATOR)) {
909             throw new IllegalArgumentException(ERR_PATH_ELEMENT_CONTAINS_SEPARATOR_MSG);
910         }
911 
912         for (String element : more) {
913             if (element.contains(SEPARATOR)) {
914                 throw new IllegalArgumentException(
915                         ERR_PATH_ELEMENT_CONTAINS_SEPARATOR_MSG);
916             }
917         }
918 
919         // create defensive copy
920         String[] elements = null;
921         if (actualFirst == null) {
922             elements = new String[more.length];
923         } else {
924             elements = new String[1 + more.length];
925             elements[0] = actualFirst;
926             System.arraycopy(more, 0, elements, 1, more.length);
927         }
928 
929         return elements;
930     }

933      * Break path apart into elements.
934      * @param path
935      * @return
936      */
937     static String[] pathToElements(String scheme, String path) {
938         final String[] src = path.split(SEPARATOR);
939         final String prefix = schemeToPrefix(scheme);
940 
941         // if the path doesn't start with 'scheme://' then it's a relative path and
942         // we can immediately return the array.
943         if (!path.startsWith(prefix)) {
944             return src;
945         }
946 
947         // if it starts with 'scheme://' then we must have at least the host:port as well.
948         if (src.length < 3) {
949             throw new IllegalArgumentException(ERR_INCOMPLETE_PATH_MSG + path);
950         }
951 
952         // create a second array and copy everything but the first two elements.
953         final String[] dest = new String[src.length - 3];
954         System.arraycopy(src, 3, dest,  0, src.length - 3);
955         return dest;
956     }
