386     private static String fileConcat(String path1, String path2) {
387         if (StringUtils.isBlank(path1)) {
388             return path2;
389         }
390         if (StringUtils.isBlank(path2)) {
391             return path1;
392         }
393         if (!path1.endsWith(URL_SEPARATOR) && !path2.startsWith(URL_SEPARATOR)) {
394             path1 = path1.concat(URL_SEPARATOR);
395         } else if (path1.endsWith(URL_SEPARATOR) && path2.startsWith(URL_SEPARATOR)) {
396             path2 = path2.substring(1);
397         }
398         return path1.concat(path2);

401     /**
402      * Gets file path string with user info removed ("username:password@"), if any.
403      * If user info is not found, it returns the same file path string unmodified.
404      * This static method can be used in error or log message formatting
405      *
406      * @param file - file path string with or without user info
407      * @return file path string without user info
408      */
409     public static String getFileNoAuth(String file) {
410         if (file == null) {
411             return StringUtils.EMPTY;
412         }
413         int end = file.lastIndexOf('@');
414         if (end > 0) {
415             if (StringUtils.startsWithIgnoreCase(file, PROTOCOL_S3)) {
416                 // in s3, the bucket name field may have '@' character
417                 int another = file.lastIndexOf('@', end - 1);
418                 if (another > 0 && another < end) {
419                     end = another;
420                 }
421             }
422             int length = file.length();
423             int begin = file.indexOf(URL_SEPARATOR);
424             if (begin > 0) {
425                 begin += URL_SEPARATOR.length();
426                 if ('/' == file.charAt(begin)) {
427                     begin++;
428                 }
429                 if (begin < end && end < length) {
430                     int indexColon = file.indexOf(':', begin);
431                     if (indexColon > begin && indexColon < end) {
432                         String tail = file.substring(end + 1, length);
433                         if (tail.startsWith(PROTOCOL_S3)) {
434                             tail = tail.substring(PROTOCOL_S3.length());
435                         } else if (tail.startsWith(PROTOCOL_WASBS)) {
436                             tail = tail.substring(PROTOCOL_WASBS.length());
437                         } else if (tail.startsWith(PROTOCOL_WASB)) {
438                             tail = tail.substring(PROTOCOL_WASB.length());
439                         }
440                         return file.substring(0, begin).concat(tail);
441                     }
442                 }
443             }
444         }
445         return file;
446     }

448     /**
449      * Converts URI to string without username and password. It should be in the form of
450      * [scheme:][//authority][path][?query][#fragment]
451      *
452      * @param uri - URI object
453      *
454      * @return URI string without username and password
455      */
456     public static String getFileNoAuth(URI uri) {
457         if (uri == null) {
458             return StringUtils.EMPTY;
459         }
460         // Get the [scheme:][//authority] without username and password
461         String scheme = uri.getScheme();
462         String authority;
463         if (scheme == null) {
464             authority = StringUtils.EMPTY;
465         } else {
466             switch(scheme) {
467                 case PROTOCOL_FILE:
468                     authority = scheme.concat(PROTOCOL_SEPARATOR);
469                     if (uri.getHost() != null) {
470                         // This case is for Windows UNC path where we need to add the hostname.
471                         // Since hostname does not starts with '//' so we need to concat that.
472                         authority = authority.concat(URL_SEPARATOR).concat(URL_SEPARATOR).
473                                 concat(uri.getHost());
474                     }
475                     break;
476                 case PROTOCOL_SLDB:
477                 case PROTOCOL_S3:
478                 case PROTOCOL_WASB:
479                 case PROTOCOL_WASBS:
480                 case PROTOCOL_GS:
481                     authority = scheme.concat(PROTOCOL_SEPARATOR);
482                     break;
483                 default:
484                     String host = uri.getHost();
485                     Integer port = uri.getPort();
486                     try {
487                         authority =  new URI(scheme, null, host, port, URL_SEPARATOR, null, null)
488                                              .toString();
489                     } catch (URISyntaxException e) {
490                         throw new SnapDataException(e, ERR_CONVERT_URI)
491                                       .formatWith(host, port)
492                                       .withResolution(RESOLUTION_ADDRESS_ISSUE);
493                     }
494                     if (StringUtils.endsWith(authority, URL_SEPARATOR)) {
495                         authority = authority.substring(0, authority.length() - 1);
496                     }
497             }
498         }
499         // Get the decoded path to the file. Do not use
500         // org.apache.commons.httpclient.util.URIUtil.decode() here since it decodes '+' to a space.
501         String path = uri.getPath();
502         // Get the query part of the URI
503         String query = uri.getQuery();
504         query = StringUtils.isBlank(query) ? StringUtils.EMPTY : QUESTION_MARK.concat(query);
505         // Get the fragment part of URI
506         String fragment = uri.getFragment();
507         fragment = StringUtils.isBlank(fragment) ? StringUtils.EMPTY : POUND.concat(fragment);
508         // Compose the full URI string without username and password.
509         return String.format(FORMAT_CONCAT4, authority, path, query, fragment);
510     }

796      * Converts a set of posix file permissions into equivalent octal string such as 744, etc.
797      *
798      * @param  permissions set containing posix file permissions
799      * @return string representing permissions in octal system
800      */
801     public static String convertPosixPermissionsIntoOctal(Set<PosixFilePermission> permissions) {
802         int octalOwnerPermission = 0;
803         int octalGroupPermission = 0;
804         int octalOthersPermission = 0;
805         for (PosixFilePermission permission : permissions) {
806             switch (permission) {
807                 case OWNER_READ:
808                     octalOwnerPermission += 4;
809                     break;
810                 case OWNER_WRITE:
811                     octalOwnerPermission += 2;
812                     break;
813                 case OWNER_EXECUTE:
814                     octalOwnerPermission += 1;
815                     break;
816                 case GROUP_READ:
817                     octalGroupPermission += 4;
818                     break;
819                 case GROUP_WRITE:
820                     octalGroupPermission += 2;
821                     break;
822                 case GROUP_EXECUTE:
823                     octalGroupPermission += 1;
824                     break;
825                 case OTHERS_READ:
826                     octalOthersPermission += 4;
827                     break;
828                 case OTHERS_WRITE:
829                     octalOthersPermission += 2;
830                     break;
831                 default:
832                     octalOthersPermission += 1;
833                     break;
834             }
835         }
836         StringBuilder finalPermissions = new StringBuilder();
837         finalPermissions.append(String.valueOf(octalOwnerPermission))
838                 .append(String.valueOf(octalGroupPermission))
839                 .append(String.valueOf(octalOthersPermission));
840         return finalPermissions.toString();
841     }

911     /**
912      * Executes a copy operation from source to target.
913      * It can be used by FileSystemProvider.copy() of a protocol which does not have a built-in
914      * copy command in its API. (e.g. sftp and ftp)
915      * This implementation tries to perform the copy operation similar to the behavior of the
916      * default local file system Java API.
917      * The target parent directory must exist. Otherwise, NoSuchFileException is thrown.
918      *
919      * @param source  - source Path object
920      * @param target  - target Path object
921      * @param options - If REPLACE_EXISTING, overwrites.
922      *                Otherwise, throws FileAlreadyExistsException.
923      * @throws NoSuchFileException if the source or target parent directory does not exist
924      *         FileAlreadyExistsException if no REPLACE_EXISTING in options and the target exists
925      *         DirectoryNotEmptyException if REPLACE_EXISTING is in options and the target is an
926      *                                    non-empty directory
927      */
928     public void copy(Path source, Path target, CopyOption... options) throws IOException {
929         URI sourceUri = source.toUri();
930         String sourceFile = sourceUri.getPath();
931         URI targetUri = target.toUri();
932         String targetFile = targetUri.getPath();
933         if (!Files.exists(source, NOFOLLOW_LINKS)) {
934             throw new NoSuchFileException(sourceFile);
935         }
936         boolean replaceExisting = Arrays.asList(options).contains(REPLACE_EXISTING);
937         if (replaceExisting) {
938             if (Files.isDirectory(target, NOFOLLOW_LINKS)) {
939                 if (Files.list(target).count() == 0) {
940                     if (Files.isRegularFile(source, NOFOLLOW_LINKS)) {
941                         // if source is a file and target is an empty directory, delete the target
942                         // so that the source can be copied to the target below
943                         Files.delete(target);
944                     } else if (Files.isDirectory(source, NOFOLLOW_LINKS)) {
945                         // if source is directory and target is an empty directory, do nothing
946                         return;
947                     }
948                 } else {
949                     throw new DirectoryNotEmptyException(targetFile);
950                 }
951             }
952         } else {
953             if (Files.exists(target, NOFOLLOW_LINKS)) {
954                 throw new FileAlreadyExistsException(targetFile, null, ERR_TARGET_EXISTS);
955             }
956         }
957         if (Files.isDirectory(source, NOFOLLOW_LINKS)) {
958             validateParents(target);
959             if (Files.isRegularFile(target, NOFOLLOW_LINKS)) {
960                 // If no REPLACE_EXISTING and source is a directory and target is a regular file,
961                 // delete the target file so that an empty target directory can be created.
962                 // This is the way how the local file system Java API behaves.
963                 Files.delete(target);
964             }
965             Files.createDirectory(target);
966         } else if (Files.isRegularFile(source, NOFOLLOW_LINKS)) {
967             validateParents(target);
968             // Now, it is ready to copy source file to targe file
969             simpleCopy(sourceUri, targetUri);
970         } else {
971             throw new IOException(ERR_SOURCE);
972         }
973     }

975     private void simpleCopy(URI sourceUri, URI targetUri) throws IOException {
976         URLConnection sourceConnection = null;
977         URLConnection targetConnection = null;
978         InputStream inputStream = null;
979         OutputStream outputStream = null;
980         try {
981             sourceConnection = getUrlConnection(sourceUri);
982             inputStream = sourceConnection.getInputStream();
983             targetConnection = getUrlConnection(targetUri);
984             outputStream = targetConnection.getOutputStream();
985             IOUtils.copyLarge(inputStream, outputStream);
986         } finally {
987             IOUtils.closeQuietly(outputStream);
988             IOUtils.closeQuietly(inputStream);
989             if (sourceConnection instanceof Disconnectable) {
990                 ((Disconnectable) sourceConnection).disconnect();
991             }
992             if (targetConnection instanceof Disconnectable) {
993                 ((Disconnectable) targetConnection).disconnect();
994             }
995         }
996     }
