63     private static void buildUnaryTable(PropertyBuilder builder, StringFunctionProvider provider) {
64         SnapProperty unaryFunctionName = builder
65                 .describe(STRING_UNARY_NAME_PROP, FUNCTION_LABEL,
66                         UNARY_STRING_NAME_DESC)
67                 .required()
68                 .withSuggestions((suggestionBuilder, propertyValues) ->
69                         suggestStringUnaryExpressions(suggestionBuilder, provider), false)
70                 .build();
71         SnapProperty unaryFunctionArg = builder
72                 .describe(UNARY_STRING_FUNC_ARG_PROP, ARGUMENT_LABEL, STRING_FUNC_ARG_DESC)
73                 .type(SnapType.STRING)
74                 .required()
75                 .expression()
76                 .build();
77         SnapProperty alias =
78                 builder.describe(UNARY_STRING_ALIAS_PROP, ALIAS_FIELD_LABEL, STRING_ALIAS_DESC)
79                         .type(SnapType.STRING)
80                         .required()
81                         .build();
82         builder.describe(UNARY_STRING_FUNCTIONS_PROP, STRING_UNARY_FUNCTIONS_LABEL,
83                 UNARY_STRING_FUNCTIONS_DESC)
84                 .type(SnapType.TABLE)
85                 .withEntry(unaryFunctionName)
86                 .withEntry(unaryFunctionArg)
87                 .withEntry(alias)
88                 .add();
89     }
