199      * This method returns string XML representation of the given LDAP entry.
200      * @param  entry
201      * @return A String representing LDAP entry in XML format.
202      */
203     public final String convertLdapEntryIntoXml(Entry entry) {
204         StringBuilder xmlResponse = new StringBuilder(XML_ENTRY_START);
205         String separator = System.lineSeparator();
206         String entryLines[] = entry.toString().split(separator);
207         for (int i = 1; i < entryLines.length; ++i) {
208             if (entryLines[i].length() == 0) {
209                 continue;
210             }
211             String tokens[] = entryLines[i].split(COLON);
212             if (tokens[0].trim().equalsIgnoreCase(DN)) {
213                 xmlResponse
214                         .append(separator)
215                         .append(SPACE)
216                         .append(XML_DN_START);
217                 String dnParts[] = tokens[1].split(COMMA);
218                 for (int j = 0; j < dnParts.length; ++j) {
219                     String pair[] = dnParts[j].split(EQUAL_TO);
220                     xmlResponse
221                             .append(separator)
222                             .append(SPACE)
223                             .append(SPACE)
224                             .append(String.format(XML_START, pair[0].trim()))
225                             .append(pair[1].trim())
226                             .append(String.format(XML_END, pair[0].trim()));
227                 }
228                 xmlResponse
229                         .append(separator)
230                         .append(SPACE)
231                         .append(XML_DN_END);
232             } else {
233                 xmlResponse
234                         .append(separator)
235                         .append(SPACE)
236                         .append(String.format(XML_START, tokens[0].trim()))
237                         .append(tokens[1].trim())
238                         .append(String.format(XML_END, tokens[0].trim()));
239             }
240         }
241         xmlResponse
242                 .append(separator)
243                 .append(XML_ENTRY_END);
244         return xmlResponse.toString();
245     }

248      * This method returns a map of values representing the given LDAP entry.
249      *
250      * @param entry The LDAP entry to be converted.
251      * @return A Map representing LDAP entry.
252      */
253     public final Map<String, Object> convertLdapEntryIntoJson(Entry entry) {
254         Map<String, Object> attributeValueMap = new LinkedHashMap<>();
255         boolean isDnFoundInAttrs = false;
256         // for some ldap servers, "dn" will appear as "distinguishedName" in the attribute set
257         // if "distinguishedName" is not found then explicitly "dn" is added
258         for (Attribute attribute : entry.getAttributes()) {
259             String attrName = attribute.getId();
260             if (attrName.equalsIgnoreCase(ATTR_DISTINGUISHED_NAME)) {
261                 isDnFoundInAttrs = true;
262             }
263             List<Object> attributeValues = new ArrayList<>();
264             List<String> binaryStringReps = new ArrayList<>();
265             for (Value<?> value : attribute) {
266                 if (value instanceof BinaryValue) {
267                     attributeValues.add(((BinaryValue) value).getValue());
268                     binaryStringReps.add(value.toString());
269                 } else {
270                     attributeValues.add(value.toString());
271                 }
272             }
273             Object trimmedAttrValue = attributeValues.isEmpty() ? StringUtils.EMPTY :
274                     (attributeValues.size() > 1 ? attributeValues : attributeValues.get(0));
275             attributeValueMap.put(attrName, trimmedAttrValue);
276             if (!binaryStringReps.isEmpty() && includeStringRep) {
277                 attributeValueMap.put(String.format(LDAP_BINARY_STR_FORMAT, attrName),
278                         (binaryStringReps.size() > 1 ? binaryStringReps : binaryStringReps.get(0)));
279             }
280         }
281         if (!isDnFoundInAttrs) {
282             attributeValueMap.put(LDAP_DISTINGUISHED_NAME_PROP.toLowerCase(),
283                     entry.getDn().getName());
284         }
285 
286         return attributeValueMap;
287     }

290      * This method loads the attribute names and value expressions given by the user in the
291      * attributes table.
292      *
293      * @param  propertyValues
294      * @param  attrs
295      * @param  entryType
296      * @return Map object containing the attribute names and value expressions
297      */
298     protected Map<String, ExpressionProperty> populateAttributesFromTable(
299             final PropertyValues propertyValues,
300             List<Map<String, Map<String, Object>>> attrs,
301             String entryType) {
302         Map<String, ExpressionProperty> attributes = new HashMap<String, ExpressionProperty>(attrs.size());
303         int index = 0;
304         String linkedName;
305         for (Map<String, Map<String, Object>> map : attrs) {
306             String attrName = (String) (map.get(LDAP_ATTRIBUTE_NAME_PROP).get(VALUE));
307             if (attrName == null) {
308                 throw new ConfigurationException(LDAP_ATTR_NAME_MISSING_ERROR_MSG)
309                         .withReason(LDAP_ATTR_NAME_MISSING_ERROR_MSG)
310                         .withResolution(LDAP_ATTR_NAME_MISSING_ERROR_RESOLUTION);
311             }
312             ExpressionProperty attrValueAsExp = propertyValues
313                     .getExpressionPropertyFor(attrs.get(index++),
314                             LDAP_ATTRIBUTE_VALUE_PROP);
315             if (!entryType.equalsIgnoreCase(LDAP_ENTRY_TYPE_OTHER)) {
316                 attrName = attrName.replace(ASTERISK, StringUtils.EMPTY);
317                 attrName = attrName.replaceAll(SPACE, UNDERSCORE);
318                 linkedName = AttributeNameLink.LINKS.get(attrName);
319                 if (linkedName != null) {
320                     attrName = linkedName;
321                 }
322             }
323             attributes.put(attrName, attrValueAsExp);
324         }
325         return attributes;
326     }
