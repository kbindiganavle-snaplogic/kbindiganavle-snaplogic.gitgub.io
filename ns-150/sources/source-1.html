653     /**
654      * Write input data into a temp file on local disk. The file will be uploaded to the staging
655      * at the end of the whole execution, just before the loading operation.
656      *
657      * @param data Data from the input document
658      */
659     protected void writeToLocalFile(Map<Object, Object> data, int maxChunkSize,
660                                     Connection snowflakeConnection) {
661         if (firstRecord) {
662             try {
663                 dataFileName = String.format(DATA_FILE_FMT, tempDir,
664                         tempDir.endsWith(File.separator) ? StringUtils.EMPTY
665                                 : File.separator,
666                         UUID.randomUUID().toString());
667                 if (IS_WIN_PLEX) {
668                     // Path name containing TILDE(~) character is an indication that ShortPath has
669                     // been provided. As Java is not compatible with this windows-specific notation
670                     // We need to convert it to long path.
671                     if (dataFileName.contains(TILDE)) {
672                         getLogger().info(String.format("Short Path is: %s", dataFileName));
673                         dataFileName = BulkLoadUtils.getLongPath(dataFileName);
674                         getLogger().info(String.format("Long Path after conversion is: %s",
675                                 dataFileName));
676                     }
677                     dataFileName = dataFileName.replace("\\", "/");
678                 }
679                 createChunkedFileWriter();
680             } catch (IOException e) {
681                 throw new StagingException(e, ERR_CREATE_DATA_FILE_MSG)
682                         .withReason(e.getMessage())
683                         .withResolution(RESOLUTION_ADDRESS_ISSUE);
684             }
685             firstRecord = false;
686         }
687         try {
688             if (cos.getCount() > maxChunkSize) {
689                 dataWriter.close();
690                 putLocalDataToSnowflakeStage(chunkedFileName, false);
691                 chunkNumber++;
692                 createChunkedFileWriter();
693             }
694             boolean firstColumn = true;
695             for (String column : copyColumns) {
696                 Object value = data.get(column);
697                 if (!firstColumn) {
698                     dataWriter.write(DEFAULT_FIELD_DELIMITER);
699                 }
700                 firstColumn = false;
701                 if (value != null) {
702                     String valueString = value.toString();
703                     dataWriter.write(formatString(valueString,
704                             loadEmptyString && (value instanceof String),
705                             DEFAULT_FIELD_DELIMITER));
706                 } else if (loadEmptyString) {
707                     // substitute null value with some other value for string type columns
708                     // present string types supported by snowflake are
709                     //   (CHAR, CHARACTER, VARCHAR, STRING, and TEXT)
710                     dataWriter.write(substituteValueForNull);
711                 }
712             }
713             dataWriter.write(DEFAULT_RECORD_DELIMITER);
714             dataWriter.flush();
715         } catch (IOException | InterruptedException ioe) {
716             throw new ExecutionException(ioe, ERROR_STREAMING_DATA)
717                     .withReason(ioe.getLocalizedMessage())
718                     .withResolution(RESOLUTION_ADDRESS_ISSUE);
719         }
720     }

785     private String getStageNameForNamedStage(String staging) {
786         String stageName;
787         if (SnowflakeUtils.isTableStage(staging)) {
788             String[] stagingParts = SnowflakeUtils.splitTableStageName(staging);
789             if (stagingParts.length == 1) {
790                 // for stage name without schema like @%table
791                 stageName = PERCENT_SIGN + quotationHandler.escape(
792                         jdbcOperations.normalize(stagingParts[0]));
793                 if (StringUtils.isNotBlank(schemaName)) {
794                     stageName = String.format(FULLY_QUALIFIED_STAGING_LOCATION_FMT,
795                             schemaName, stageName);
796                 }
797             } else {
798                 // for stage name with schema like @schema.%table

800                         stagingParts[0]));

802                         jdbcOperations.normalize(stagingParts[1]));

814                         schemaName, stageName);

935     // this method is executed in a child thread; see initializePutQueue()
936     private int putChunks() throws SQLException, InterruptedException {
937         int chunkCount = 0;
938         while (true) {
939             // take() will wait for an item to be added with putQueue.transfer() by putLocalDataToSnowflakeStage() in the parent thread
940             PutParams putParams = putQueue.take();
941             chunkCount++;
942             String query = getQueryForPut(putParams.localFilePath, putParams.stagePrefix);
943             try (Statement stmt = snowflakeConnection.createStatement()) {
944                 stmt.execute(query);
945             }
946             File f = new File(putParams.localFilePath);
947             if (f.exists()) {
948                 if (f.delete()) {
949                     getLogger().info("Chunked file {} is deleted", f);
950                 } else {
951                     getLogger().error("Error while deleting local file {}", f);
952                 }
953             }
954             if (putParams.last) {
955                 getLogger().debug("put last chunk; returning");
956                 break;
957             }
958         }
959         return chunkCount;
960     }
