426     private void doProcessDocument(final Document document, final String inputViewName) {
427         statusReporter.notify(PREPARING_UNLOD_OPERATION);
428         kmsKey = kmsKeyExpression.eval(document);
429         randomUUID = System.nanoTime();
430         if (snowflakeConnection == null) {
431             // Connection might have been created already if truncate was done
432             snowflakeConnection = jdbcOperations.acquireConnection(account);
433         }
434         if (queryManagementOption.equals(SqlQueryManagementOptions.CANCEL_QUEUED_QUERIES) &&
435                 !queryIDThreadStarted) {
436             startQueryIDThread(snowflakeConnection);
437         }
438         try (Statement stmt = snowflakeConnection.createStatement()) {
439             StringBuilder cmdBuilder = formatBasicCopyCmd(document);
440             cmdBuilder.append(formatFileFormatOptions(document));
441             cmdBuilder = SnowflakeUtils.addEncryptionProperty(cmdBuilder,
442                     encryptionType, kmsKey);
443             cmdBuilder.append(formatCopyOptions());
444             if (header) {
445                 cmdBuilder.append(HEADER_TRUE);
446             }
447             cmdBuilder.append(formatValidationMode());
448             String fullUnloadQuery = cmdBuilder.toString();
449             Map<String, Object> outputInfo = Maps.newLinkedHashMap();
450             // Hide the confidential info (s3 secret key, etc)
451             String fullQueryForOutput = fullUnloadQuery;
452             switch (accountType) {
453                 case S3:
454                     if(StringUtils.isNotEmpty(accountS3AccessID)) {
455                         fullQueryForOutput = fullUnloadQuery
456                                 .replace(accountS3AccessID, HIDDEN_MASK);
457                     }
458                     if(StringUtils.isNotEmpty(accountS3SecretKey)) {
459                         fullQueryForOutput = fullQueryForOutput
460                                 .replace(accountS3SecretKey, HIDDEN_MASK);
461                     }
462                     if(StringUtils.isNotEmpty(accountAWSToken)) {
463                         fullQueryForOutput = fullQueryForOutput
464                                 .replace(accountAWSToken, HIDDEN_MASK);
465                     }
466                     if (StringUtils.isNotBlank(kmsKey)) {
467                         fullQueryForOutput = fullQueryForOutput.replace(kmsKey, HIDDEN_MASK);
468                     }
469                     break;
470             }
471             outputInfo.put(UNLOAD_REQUEST_KEY, fullQueryForOutput);
472             outputInfo.put(UNLOAD_DESTINATION, unloadDestination);
473             // add query to Queue, getQueryThread reads query from this Queue
474             DatabaseUtils.submitQueryToThread(fullQueryForOutput, queryQueue);
475             statusReporter.notify(SENT_UNLOAD_OPERATION);
476             // SNAP-6458: We have to use executeQuery here to get the resultSet for DDL statement
477             // Since Snowflake JDBC 3.8.1, execute() method won't return resultSet for DDL.
478             // Corresponding Snowflake ticket: SNOW-76035
479             stmt.executeQuery(fullUnloadQuery);
480             try (ResultSet rs = stmt.getResultSet()) {
481                 if (rs != null) {
482                     outputInfo.put(RESPONSE_KEY, wrapResponse(rs));
483                 }
484             }
485             statusReporter.notify(EXECUTED_UNLOAD_OPERATION);
486             outputViews.write(documentUtility.newDocumentFor(document, outputInfo));
487             if (tokenRefreshAttempted) {
488                 tokenRefreshAttempted = false;
489             }
490         } catch (SQLException e) {
491             catchSQLException(e, ERROR_PERFORMING_UNLOAD);
492         } catch (SnapDataException e) {
493             if (e.getCause() instanceof SQLException) {
494                 catchSQLException((SQLException) e.getCause(), e.getMessage());
495             }
496             errorViews.write(e, document);
497         } finally {
498             if (StringUtils.isNotBlank(tempStageName)) {
499                 try (Statement stmt = snowflakeConnection.createStatement()) {
500                     stmt.execute(String.format(DROP_STAGE_COMMAND, schemaName, tempStageName));
501                     LOGGER.info("Successfully dropped temporary external azure storage {}",
502                             tempStageName);
503                 } catch (SQLException e1) {
504                     LOGGER.error("Error deleting temporary external azure storage {}", tempStageName,
505                             e1);
506                 } finally {
507                     tempStageName = null;
508                 }
509             }
510         }
511         if (tokenRefreshAttempted) {
512             doProcessDocument(document, inputViewName);
513         }
514     }

516     protected void catchSQLException(final SQLException e, final String errorMessage) {
517         boolean tokenExpired = StringUtils.containsIgnoreCase(e.getMessage(),
518                 INVALID_SAS_TOKEN_ERROR_CONTENTS_FROM_MICROSOFT);
519         if ((accountType == SnowflakeAccountType.AZURE) && tokenExpired &&
520                 !tokenRefreshAttempted && azureDatabaseAccount.isSystemGeneratedToken()) {
521             tokenRefreshAttempted = true;
522             LOGGER.info("System generated SAS token might have expired as the error message " +
523                     "is {}; attempting the unload operation once more by generating a " +
524                     "new SAS token", e.getMessage());
525         } else {
526             String resolution = ERROR_PERFORMING_UNLOAD_RESOLUTION;
527             if (tokenRefreshAttempted && tokenExpired) {
528                 LOGGER.info("Expiry timeout for system generated token needs to be increased " +
529                         "in the code");
530                 resolution = null;
531             } else if (tokenExpired) {
532                 resolution = ERR_UNLOAD_OPERATION_FAILED_FOR_INVALID_SAS_TOKEN_RESOLUTION;
533             }
534             if (tokenRefreshAttempted) {
535                 tokenRefreshAttempted = false;
536             }
537             ExecutionException executionException = new ExecutionException(e, errorMessage)
538                     .withReason(e.getLocalizedMessage());
539             executionException = (resolution == null) ?
540                     executionException.withResolutionAsDefect() :
541                     executionException.withResolution(resolution);
542             throw executionException;
543         }
544     }

843     private String toStagingString(String staging) {
844         String stagePrefix;
845         switch (stagingInputType) {
846             case NAMED_STAGE:

850                 String[] stagingParts = quotationHandler.splitFullyQualifiedName(
851                         stageName, 2);

853                     throw new SnapDataException(ERROR_SCHEMA_NOT_SPECIFIED)

869             case AZURE_URI:
870                 stagePrefix = staging;

875             default:

877                     stagePrefix = String.format(FORMAT_INTERNAL_PATH, staging);
