325         formatName = formatNameExpression.eval(document);
326         fileFormatOption = fileFormatOptionExpression.eval(document);
327         if (StringUtils.isNotBlank(fileFormatOption)) {
328             verifyFileFormatOption();
329         }

331         bufferSizeInMB = bufferSizeExpr.eval(document);
332         if(queryManagementOption.equals(SqlQueryManagementOptions.CANCEL_QUEUED_QUERIES) &&
333                 !queryIDThreadStarted) {
334             startQueryIDThread(snowflakeConnection);
335         }
336         // if the document is from schema input view then generate column translation
337         if (inputViewName != null && inputViewName.equals(schemaViewName)) {
338             if (document != null) {
339                 schema = document;
340                 columnTranslation = databaseUtils.buildColumnTranslation(schema);
341             }
342             return;
343         }
344         // if the document is from data input view then write it to temp file
345         Map<String, Object> mapData = document.get(Map.class);
346         Map<Object, Object> data = new TreeMap<>(mapData);
347         if (data.isEmpty()) {
348             return;
349         }
350         if (firstRecord) {
351             if (lazyLoadTableMetadata()) {
352                 Connection conn = jdbcOperations.acquireConnection(account);
353                 tableMetaData = jdbcOperations.getTableMetaData(conn,
354                         account.getDatabaseConfig().getDatabaseName(), tableName);
355             }
356             if (preserveCaseSensitivity) {
357                 snowflakeMetadataProvider.setUseUpperCaseIdentifiers(false);
358             }
359             if (tmpTableCreated = createTempTable() && generateCopyColumns(data, document)) {
360                 databaseUtils.translateColumnNamesInDoc(columnTranslation, document);
361                 List<String> normalizeKeyColumnNames = Lists.newArrayList();
362                 for (String column : keyColumnNames) {
363                     if (StringUtils.isBlank(column)) {
364                         throw new ExecutionException(ERROR)
365                                 .withReason(EMPTY_PRIMARY_KEY)
366                                 .withResolution(ERR_INVALID_COLUMN_RESOLUTION);

369                     if (!tableMetaData.containsKey(normalizedColumn)) {

422     @Override
423     public void handle(LifecycleEvent event) {
424         switch (event) {
425             case SUCCESS:
426                 LOGGER.debug("Got success lifecycle event: {}", String.valueOf(event));
427                 closeDataWriter();
428                 if (inputCount > 0 && !copyColumns.isEmpty()) {
429                     if (STAGING_LOCATION_INTERNAL.equals(stagingLocation)) {
430                         try {
431                             waitForPutChunksCompletion();
432                             deleteTempFile();
433                         } catch (java.util.concurrent.ExecutionException | InterruptedException e) {
434                             throw new ExecutionException(e, ERR_SENDING_LAST_CHUNK)
435                                     .withReason(e.getMessage())
436                                     .withResolution(ERR_SENDING_LAST_CHUNK_RESOLUTION);
437                         }
438                     }
439                     if (loadTempFileIntoTempTable()) {
440                         mergeTempTableIntoTargetTable();
441                     }
442                 }
443                 deleteTempTable();
444                 break;
445             case CLOSE:
446             case STOP:
447             case FAILURE:
448                 LOGGER.debug(STOP_LIFECYCLE_MSG, String.valueOf(event));
449                 closeDataWriter();
450                 deleteTempFile();
451                 if (queryManagementOption.equals(SqlQueryManagementOptions.CANCEL_QUEUED_QUERIES)
452                         && queryIDThreadStarted) {
453                     stopQuery(snowflakeConnection);
454                 }
455                 deleteTempTable();
456                 break;
457             default:
458                 LOGGER.debug(OTHER_LIFECYCLE_MSG, String.valueOf(event));
459         }
460         super.handle(event);
461     }

631     @Override
632     protected void loadAction(String targetTable, Document ignored) {
633         StringBuilder joinList = new StringBuilder();
634         boolean firstColumn = true;
635         String mergeStmt = null;
636         if (!preserveCaseSensitivity) {
637             // If case-insensitive, get copyColumns from quotedCopyColumns because column names from
638             // table metadata (not from input document) should be used.
639             copyColumns = quotedCopyColumns.stream()
640                     .map(p -> p.substring(1, p.length() - 1))
641                     .collect(Collectors.toList());
642         }
643         for (String column : copyColumns) {
644             if (!keyColumnNames.contains(jdbcOperations.normalize(column))) {
645                 if (!firstColumn) {
646                     joinList.append(COMMA_DELIMITER);
647                 }
648                 firstColumn = false;
649                 joinList.append(String.format(UPDATE_ENTRY, column, column));
650             }
651         }
652         StringBuilder joinExpList = getPredicateList(targetTable, keyColumnNames);
653 
654         try (Statement stmt = snowflakeConnection.createStatement()) {
655             deleteCond = encloseQuotes(deleteCond);
656             String deleteCommand = String.format(DELETE_MERGE_COMMAND, deleteCond);
657             String updateStmt = createUpdateStmt();
658             String insertStmt = createInsertStmt();
659 
660             if (StringUtils.isNotEmpty(deleteCond)) {
661 
662                 mergeStmt = String.format(MERGE_COMMAND_DELETE, targetTable, mergeTable,
663                         joinExpList, deleteCommand, updateStmt, insertStmt);
664             } else {
665                 mergeStmt = String.format(MERGE_COMMAND, targetTable, mergeTable,
666                         joinExpList, updateStmt, insertStmt);
667             }
668             DatabaseUtils.submitQueryToThread(mergeStmt, queryQueue);
669             // SNAP-6456: We have to use executeQuery here to get the resultSet for DDL statement
670             // Since Snowflake JDBC 3.8.1, execute() method won't return resultSet for DDL.
671             // Corresponding Snowflake ticket: SNOW-76035
672             try (ResultSet rs = stmt.executeQuery(mergeStmt)) {
673                 if (rs != null && rs.next()) {
674                     insertCount = rs.getInt(RESPONSE_ROWS_INSERTED_COUNT);
675                     updateCount = rs.getInt(RESPONSE_ROWS_UPDATED_COUNT);
676                 } else {
677                     insertCount = 0;
678                     updateCount = 0;
679                 }
680             }
681 
682         } catch (SQLException e) {
683             throw new ConfigurationException(INVALID_MERGE_CONDITION)
684                     .formatWith(mergeStmt)
685                     .withReason(e.getMessage())
686                     .withResolution(PLEASE_CORRECT_THE_MERGE_CONDITION_SYNTAX);
687         }
688     }
