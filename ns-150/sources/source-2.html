402         super.configureSnap(propertyValues);
403         setTempDir(propertyValues);
404         doNullSchemaCheck();
405         readStagingLocationProperty(propertyValues);
406         dataSource = propertyValues.get(DATA_SOURCE_PROP);
407         if (!DATA_SOURCE_STAGED_FILES.equals(dataSource) &&
408                 STAGING_LOCATION_INTERNAL.equals(stagingLocation)) {
409             readFlushChunkSizeProperty(propertyValues);
410         }
411         truncate = propertyValues.getBoolean(TRUNCATE_PROP, false);
412         stagingExpression = propertyValues.getAsExpression(STAGING_PROP);
413         List<Map<String, String>> fileList = propertyValues.get(FILE_LIST_PATH);
414         fileNamePatternExpression = propertyValues.getAsExpression(FILE_NAME_PATTERN_PROP);
415         selectQuery = propertyValues.getAsExpression(SELECT_STATEMENT_PROP).eval(null);
416         aggregateErrorType = propertyValues.get(AGGREGATE_ERRORS_PROP) == null
417                 ? FULL_ERROR_VIEW : propertyValues.get(AGGREGATE_ERRORS_PROP);
418         if (!DATA_SOURCE_STAGED_FILES.equals(dataSource) && StringUtils.isNotBlank(selectQuery)) {
419             throw new ConfigurationException(ERROR_DATA_QUERY_TRANSFORMATION)
420                     .withReason(ERROR_DATA_QUERY_TRANSFORMATION_REASON)
421                     .withResolution(ERROR_DATA_QUERY_TRANSFORMATION_RESOLUTION);
422         }
423         Connection conn = jdbcOperations.acquireConnection(account);
424         if (DATA_SOURCE_STAGED_FILES.equals(dataSource)) {
425             if (autoCreateTable && !jdbcOperations.tableExists(conn,
426                     account.getDatabaseConfig().getDatabaseName(), tableName)) {
427                 throw new ConfigurationException(ERROR_AUTO_CREATE_TABLE_NOT_SUPPORTED)
428                         .withReason(ERROR_AUTO_CREATE_TABLE_NOT_SUPPORTED_REASON)
429                         .withResolution(ERROR_AUTO_CREATE_TABLE_NOT_SUPPORTED_RESOLUTION);
430             }
431             if (CollectionUtils.isNotEmpty(fileList)) {
432                 filePathExpressions = new ArrayList<>(fileList.size());
433                 for (Map<String, String> item : fileList) {
434                     ExpressionProperty pathExpression = propertyValues.getExpressionPropertyFor(
435                             item, STAGED_FILE_PROP);
436                     filePathExpressions.add(pathExpression);
437                 }
438             } else if (CollectionUtils.isEmpty(fileList) &&
439                     fileNamePatternExpression.isEmpty() && StringUtils.isEmpty(selectQuery)) {
440                 throw new ConfigurationException(ERROR_NO_STAGED_FILE_OR_PATTERN)
441                         .withReason(ERROR_NO_STAGED_FILE_OR_PATTERN_REASON)
442                         .withResolution(ERROR_NO_STAGED_FILE_OR_PATTERN_RESOLUTION);

467                 schemaViewName = viewOrder.get(1);

524     @Override
525     public void execute() {
526         if (inputViews.isEmpty()) {
527             process(null, null);
528         } else {
529             for (String viewName : getInputViewOrder()) {
530                 for (Document document : inputViews.documents(inputViews.get(viewName))) {
531                     process(document, viewName);
532                 }
533             }
534         }
535 
536         if (!firstRecord && autoCreateTable && schema != null) {
537             // We want to create the table even if there are no rows to be loaded.
538             createTableIfAbsent(null);
539         }
540         if (DATA_SOURCE_INPUT_VIEW.equals(dataSource)) {
541             if (!successful || (bulkLoadWrapper == null && dataWriter == null)) {
542                 return;
543             }
544             if (STAGING_LOCATION_EXTERNAL.equals(stagingLocation)) {
545                 LOGGER.debug(LOG_WRITING_TO_EXTERNAL_LOCATION, accountType.name());
546                 if (bulkLoadWrapper != null) {
547                     bulkLoadWrapper.close();
548                     bulkLoadWrapperClosed = true;
549                 }
550                 LOGGER.debug(LOG_WRITING_TO_EXTERNAL_LOCATION_DONE, accountType.name());
551             } else {
552                 closeDataWriter();
553                 try {
554                     waitForPutChunksCompletion();
555                 } catch (java.util.concurrent.ExecutionException | InterruptedException e) {
556                     throw new ExecutionException(e, ERR_SENDING_LAST_CHUNK)
557                             .withReason(e.getMessage())
558                             .withResolution(RESOLUTION_ADDRESS_ISSUE);
559                 }
560             }
561             try {
562                 // This is for loading the data from inputview, which is batched in temp file
563                 preLoadAction();
564                 loadActionWithOAuthRetry(tableName, null);
565             } catch (SnapDataException e) {
566                 errorViews.write(e);
567             }
568         }
569     }

571     @Override
572     protected void processDocument(final Document document, final String inputViewName) {
573         // Initialize expression properties
574         fileNamePattern = fileNamePatternExpression.eval(document);
575         formatName = formatNameExpression.eval(document);
576         fileFormatOption = fileFormatOptionExpression.eval(document);
577         if (StringUtils.isNotBlank(fileFormatOption)) {
578             verifyFileFormatOption();
579         }
580         kmsKey = kmsKeyExpression.eval(document);
581         bufferSizeInMB = bufferSizeExpr.eval(document);
582         if (queryManagementOption.equals(SqlQueryManagementOptions.CANCEL_QUEUED_QUERIES) &&
583                 !queryIDThreadStarted) {
584             startQueryIDThread(snowflakeConnection);
585         }
586 
587         // Check if this is the schema document and save it for later.
588         if (inputViewName != null && inputViewName.equals(schemaViewName)) {
589             if (document != null) {
590                 schema = document;
591                 columnTranslation = databaseUtils.buildColumnTranslation(schema);
592             }
593             return;
594         }
595 
596         if (document != null) {
597             inputCount++;
598         }
599 
600         if (DATA_SOURCE_INPUT_VIEW.equals(dataSource)) {
601             // If the data is from inputview, we need to batch the input into a temp file
602             if (STAGING_LOCATION_EXTERNAL.equals(stagingLocation)
603                     && NAMED_STAGE.equals(stagingInputType)) {
604                 SnapDataException sde = new SnapDataException(OPERATION_NOT_SUPPORTED)
605                         .withReason(REASON_CANNOT_UPLOAD_DOC_TO_EXTERNAL_NAMED_STAGE)
606                         .withResolution(RESOLUTION_CANNOT_UPLOAD_DOC_TO_EXTERNAL_NAMED_STAGE);
607                 errorViews.write(sde, document);
608                 return;
609             }
610             storeInputDocument(document);
611         } else {
612             // For staged file, we note down the file list if there is any, and go ahead to load
613             // them
614             if (CollectionUtils.isNotEmpty(filePathExpressions)) {
615                 // Evaluate file paths
616                 filePaths = new ArrayList<>();
617                 for (ExpressionProperty pathExpression : filePathExpressions) {
618                     String path = pathExpression.eval(document);
619                     if (StringUtils.isBlank(path)) {
620                         continue;
621                     }
622                     filePaths.add(path);
623                 }
624             }
625             try {
626                 loadActionWithOAuthRetry(tableName, document);
627             } catch (SnapDataException e) {
628                 errorViews.write(e, document);
629             }
630         }
631     }

648     /**
649      * Store the input document into a temporary file. The temp file will be on local disk if
650      * internal stage, on S3 bucket defined in the account if external.
651      *
652      * NOTE: Staging is not used when data source is inputview
653      *
654      * @param document Single document object
655      */
656     private void storeInputDocument(Document document) {
657         Map<String, Object> data = document == null ? null : document.get(Map.class);
658         if (data == null) {
659             return;
660         }
661         databaseUtils.translateColumnNamesInDoc(columnTranslation, document);
662         if (firstRecord) {
663             // tableMetaData may have been set by configureSnap, but might be stale
664             // if snaps running in parallel recreated the table since then.
665             if (!metadataFound) {
666                 tableMetaData = lazyLoadTableMetadata() ? null : tableMetaData;
667                 if (autoCreateTable) {
668                     // this will set tableMetaData if the table is created
669                     createTableIfAbsent(document);
670                 }
671                 if (tableMetaData == null) {
672                     Connection conn = jdbcOperations.acquireConnection(account);
673                     tableMetaData = jdbcOperations.getTableMetaData(conn,
674                             account.getDatabaseConfig().getDatabaseName(), tableName);
675                     if (tableMetaData.isEmpty()) {
676                         throw new ConfigurationException(TABLE_NOT_FOUND)
677                                 .formatWith(tableName)
678                                 .withReason(ERROR_TABLE_DOES_NOT_EXIST)
679                                 .withResolution(ERROR_TABLE_DOES_NOT_EXIST_RESOLUTION);
680                     }
681                 }
682                 metadataFound = true;
683             }
684             if (preserveCaseSensitivity) {
685                 snowflakeMetadataProvider.setUseUpperCaseIdentifiers(false);
686             }
687             // Initialize copyColumns and quotedCopyColumns
688             initCopyColumns(data);
689             if (copyColumns.isEmpty()) {
690                 throw new ExecutionException(ERROR_INVALID_INPUT)
691                         .withReason(ERROR_INVALID_INPUT_REASON)
692                         .withResolution(ERROR_INVALID_INPUT_RESOLUTION);
693             }
694         }
695 
696         writeDataToFile(data, chunkFlushSize, snowflakeConnection);
697     }

761     /**
762      * The action that loads staged file (either specified by user or generated by snap) to the
763      * table in Snowflake.
764      */
765     @Override
766     protected void loadAction(String targetTable, Document original) {
767         if (snowflakeConnection == null) {
768             // Connection might have been created already if truncate was done
769             snowflakeConnection = jdbcOperations.acquireConnection(account);
770         }
771 
772         try (Statement stmt = snowflakeConnection.createStatement()) {
773             if (DATA_SOURCE_INPUT_VIEW.equals(dataSource)) {
774                 loadFromInputView(targetTable, stmt);
775             } else {
776                 queryForStagedFileExecuted = false;
777                 loadFromStagedFile(targetTable, stmt, original);
778             }
779 
780             // Output results
781             try (ResultSet rs = stmt.getResultSet()) {
782                 if (canExecuteValidateCommand()) {
783                     writeErrorView(targetTable, original);
784                 } else if (!DONT_VALIDATE.equals(aggregateErrorType)){
785                     WARNING_MESSAGE_S3.report();
786                 }
787                 if (!purge && DATA_SOURCE_INPUT_VIEW.equals(dataSource) &&
788                         STAGING_LOCATION_EXTERNAL.equals(stagingLocation)) {
789                     PURGE_LINT.report();
790                 }
791                 if (DATA_SOURCE_INPUT_VIEW.equals(dataSource) ||
792                         (DATA_SOURCE_STAGED_FILES.equals(dataSource) &&
793                                 queryForStagedFileExecuted)) {
794                     writeResultToOutputView(rs, original);
795                 }
796             } catch (SQLException e) {
797                 throw new ExecutionException(e, ERROR_COPYING_EXTERNAL_FILE_SNOWFLAKE)
798                         .formatWith(accountType.name())
799                         .withReason(e.getMessage())
800                         .withResolution(ERROR_FIX_DATA_ISSUES);
801             }
802 
803             cleanupTempStageFolder();
804         } catch (ExecutionException | SnapDataException e) {
805             throw e;
806         } catch (SQLException e) {
807             // Check if the token expired and operation needs to be re-attempted if its
808             //  system-generated token option
809             boolean sasTokenExpired = (accountType == SnowflakeAccountType.AZURE) &&
810                     (StringUtils.containsIgnoreCase(e.getMessage(),
811                             INVALID_SAS_TOKEN_ERROR_CONTENTS_FROM_MICROSOFT));
812             if (sasTokenExpired && !sasTokenRefreshed && azureAccount.isSystemGeneratedToken()) {
813                 sasTokenRefreshed = true;
814                 LOGGER.info("System generated SAS token might have expired as the error message " +
815                         "is {}; attempting the unload operation once more by generating a " +
816                         "new SAS token", e.getMessage());
817                 try {
818                     loadActionWithOAuthRetry(targetTable, original);
819                 } finally {
820                     sasTokenRefreshed = false;
821                 }
822             } else {
823                 // Handle the Data type exception from Snowflake. Route them to Errorview
824                 String resolution = RESOLUTION_ADDRESS_ISSUE;
825                 if (sasTokenExpired) {
826                     if (!azureAccount.isSystemGeneratedToken()) {
827                         resolution = ERR_UNLOAD_OPERATION_FAILED_FOR_INVALID_SAS_TOKEN_RESOLUTION;
828                     } else if (sasTokenRefreshed) {
829                         LOGGER.info("System generated token seems to be valid for very short " +
830                                 "amount of time.  Consider increasing its longivity.");
831                         resolution = null;
832                     }
833                 }
834 
835                 SnapDataException exception = new SnapDataException(e, ERROR_PERFORMING_LOAD)
836                         .withReason(e.getLocalizedMessage());
837                 exception = resolution != null ?
838                         exception.withResolution(resolution) :
839                         exception.withResolutionAsDefect();
840                 throw exception;
841             }
842         } catch (NumberFormatException ne) {
843             throw new SnapDataException(ne, ERROR_PERFORMING_LOAD)
844                     .withReason(ne.getLocalizedMessage())
845                     .withResolution(RESOLUTION_ADDRESS_ISSUE);
846         } catch (Exception e) {
847             throw new ExecutionException(e, ERROR_PERFORMING_LOAD)
848                     .withReason(e.getLocalizedMessage())
849                     .withResolution(RESOLUTION_ADDRESS_ISSUE);
850         }
851         LOGGER.debug(LOG_COPY_DONE);
852     }

855      * Execute 'COPY INTO table' to load a staged file into the target table.
856      *
857      * @param targetTable
858      * @param stmt
859      *
860      * @throws SQLException
861      */
862     private void loadFromStagedFile(String targetTable, Statement stmt, Document document) throws

866         if (query == null) {
867             return;

872         // Corresponding Snowflake ticket: SNOW-76035
873         stmt.executeQuery(query);

877      * Get the full 'COPY INTO table' string for loading staged files.
878      *
879      * @param targetTable target table name for loading
880      * @param document
881      *
882      * @return 'COPY INTO table' string that is sent to the endpoint
883      */
884     protected String getQueryForStagedFile(String targetTable, Document document)
885             throws SQLException {
886         StringBuilder queryBuilder;
887         String stagePrefix = toStagingString(staging);
888         queryBuilder = buildQueryBuilderFromFile(targetTable, stagePrefix, document);
889 
890         if (CollectionUtils.isNotEmpty(filePaths)) {
891             // Format: FILES = ( 'file1','file2','file3'... )
892             queryBuilder.append(String.format(FILES_FMT, StringUtils.join(filePaths, "','")));
893         } else if (StringUtils.isNotBlank(fileNamePattern)) {
894             queryBuilder.append(String.format(FORMAT_SPACE_QUOTE, KEY_PATTERN, fileNamePattern));
895         } else if (StringUtils.isBlank(selectQuery)) {
896             SnapDataException snapDataException = new SnapDataException(
897                     ERROR_NO_STAGED_FILE_OR_PATTERN)
898                     .withReason(ERROR_NO_STAGED_FILE_OR_PATTERN_REASON)
899                     .withResolution(ERROR_NO_STAGED_FILE_OR_PATTERN_RESOLUTION);
900             errorViews.write(snapDataException);
901             return null;
902         }
903         queryBuilder = SnowflakeUtils.addEncryptionProperty(queryBuilder, encryptionType, kmsKey);
904 
905         if (formatName != null) {
906             queryBuilder.append(String.format(FILE_FORMAT_OPTION, KEY_FORMAT_NAME,
907                     validateAndReturnFullFormatName(formatName),
908                     StringUtils.EMPTY, StringUtils.EMPTY));
909         } else if (fileFormatType != null) {
910             if (!NONE.equals(fileFormatType)) {
911                 queryBuilder.append(String.format(FILE_FORMAT_OPTION, KEY_TYPE, fileFormatType,
912                         StringUtils.isNotBlank(fileFormatOption) ?
913                                 fileFormatOption.trim() : StringUtils.EMPTY,
914                         StringUtils.EMPTY));
915             } else if (StringUtils.isNotBlank(fileFormatOption)) {
916                 queryBuilder.append(String.format(FILE_FORMAT_NONE, fileFormatOption.trim(),
917                         StringUtils.EMPTY));
918             }
919         }
920 
921         appendQueryOptions(queryBuilder);
922         String query = queryBuilder.toString();
923         logQuery(query);
924         return query;
925     }

927     /**
928      * Build the initial COPY query for external/internal staged files.
929      *
930      * With columns:
931      * If table columns are provided in the snap configuration:
932      * COPY INTO table("column1", "column2", ...) FROM stage/
933      *
934      * Legacy behavior:
935      * If all files contains the same columns and those columns are all included in the table:
936      * COPY INTO table(column1, column2, ...) FROM stage/
937      *
938      * otherwise:
939      * COPY INTO table FROM stage/
940      *
941      * @param targetTable Table name we are loading to
942      * @param stagePrefix Staging prefix for the staged file(s)
943      * @param document
944      *
945      * @return The String builder for the query string; null if no file is specified
946      * @throws SQLException
947      */
948     private StringBuilder buildQueryBuilderFromFile(String targetTable, String stagePrefix,
949             Document document) throws SQLException {
950         List<String> tableColumns = getQueryColumns(targetTable, document);
951         String headerColumns = null;
952         boolean queryWithColumn = CollectionUtils.isNotEmpty(tableColumns);
953         if (queryWithColumn) {
954             headerColumns = StringUtils.join(tableColumns, COMMA);
955         }
956 
957         StringBuilder queryBuilder;
958         String filePathStage;
959         if (StringUtils.isNotEmpty(selectQuery)) {
960             queryBuilder = new StringBuilder(String.format
961                     (COPY_FROM_STAGED_FILE_COMMAND_WITH_DATA_QUERY_TRANSFORMATION,
962                             targetTable, headerColumns, selectQuery));
963         } else if (STAGING_LOCATION_EXTERNAL.equals(stagingLocation)
964                 && !NAMED_STAGE.equals(stagingInputType)) {
965             // Perform the query building for External Staging location.
966             if (S3_URI.equals(stagingInputType) || AZURE_URI.equals(stagingInputType) ||
967                     GCS_URI.equals(stagingInputType)) {
968                 if (queryWithColumn) {
969                     queryBuilder = new StringBuilder(String.format
970                             (COPY_FROM_STAGED_FILE_COMMAND_WITH_DATA_TRANSFORMATION,
971                                     targetTable, headerColumns, stagePrefix + SLASH));
972                 } else {
973                     queryBuilder = new StringBuilder(String.format(COPY_FROM_STAGED_FILE_COMMAND,
974                             targetTable, stagePrefix + SLASH));
975                 }
976             } else {
977                 if (stagePrefix.length() > 0 && !(stagePrefix.startsWith(SLASH))) {
978                     // if Staging is specified
979                     stagePrefix = SLASH + stagePrefix;
980                 }
981                 if (accountType == SnowflakeAccountType.AZURE) {
982                     filePathStage = setAzurePath(azureAccount.getStorageBlobUrl(), stagePrefix);
983                 } else if (accountType == SnowflakeAccountType.GoogleCloudStorage) {
984                     retrieveGCSInfo();
985                     if (StringUtils.isBlank(gcsFolder)) {
986                         if (StringUtils.isBlank(stagePrefix)) {
987                             filePathStage =
988                                     String.format(GCS_FILE_KEY_FULL_PATH_NO_FOLDER,
989                                             gcsBucket + SLASH);
990                         } else {
991                             filePathStage = String.format(GCS_FILE_KEY_FULL_PATH, gcsBucket,
992                                     stagePrefix.substring(1) + SLASH);
993                         }
994                     } else {
995                         filePathStage = String.format(GCS_FILE_KEY_FULL_PATH, gcsBucket,
996                                 gcsFolder + stagePrefix + SLASH);
997                     }
