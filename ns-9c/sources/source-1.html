256         StringBuffer escapedValue = new StringBuffer();
257         char[] chars = value.toCharArray();
258 
259         for (char c : chars) {
260             if (SPECIAL_REGEX_CHARS_TO_ESCAPE.contains(c)) {

269     protected void writeToOutputView(final String s) {
270         Scanner reader = null;
271         String line = null;
272         List<String> columns;
273         Map<String, Object> data;
274         try {
275             columns = new ArrayList<>();
276             Connection conn = jdbcOperations.acquireConnection(account);
277             TableMetaData schema = jdbcOperations.getTableMetaData(conn,
278                     account.getDatabaseConfig().getDatabaseName(), tableName);
279 
280             reader = new Scanner(new File(metadataFileName));
281             reader.useDelimiter(FORMAT_FILE_ROW_TERMINATOR);
282             String[] fields;
283             while (reader.hasNext()) {
284                 line = reader.next();
285                 if (StringUtils.isBlank(line)) {
286                     continue;
287                 }
288                 fields = line.split("\\s+");
289                 // skip version and row count in format file
290                 if (fields.length > 1) {
291                     columns.add(fields[fields.length - 2]);
292                 }
293             }
294             IOUtils.closeQuietly(reader);
295             if (UTF8) {
296                 reader = new Scanner(new File(dataFileName), UTF_8);
297             } else {
298                 reader = new Scanner(new File(dataFileName));
299             }
300             reader.useDelimiter(doExtraRegexEscape(dataFileRowTerminator));
301             while (reader.hasNext()) {
302                 line = reader.next();
303                 if (StringUtils.isBlank(line)) {
304                     continue;
305                 }
306                 String[] dataRecord = line.split(doExtraRegexEscape(fieldDelimiter), -2);
307                 data = new LinkedHashMap<>();
308                 for (int i = 0, len = dataRecord.length; i < len; i++) {
309                     Object convertedValue = convertToSnapType(schema.get(
310                             columns.get(i)).getDataType(), dataRecord[i]);
311                     data.put(columns.get(i), convertedValue);
312                 }
313                 if (rowsOutputView != null) {
314                     rowsOutputView.write(documentUtility.newDocument(data));
315                 }
316             }
317         } catch (IOException ie) {
318             SnapDataException dataException = new SnapDataException(
319                     ie, ERR_PROCESSING_BCP_FETCH_DATA_FILE_MSG)
320                     .withReason(ie.getMessage())
321                     .withResolution(ERR_PROCESSING_BCP_FETCH_DATA_FILE_RESOLUTION);
322             errorViews.write(dataException);
323         } finally {
324             IOUtils.closeQuietly(reader);
325         }
326     }

343     private Object convertToSnapType(int columnDataType, String value) {
344         if (StringUtils.isEmpty(value)) {
345             return value;
346         }
347         switch (columnDataType) {
348             case Types.TINYINT:
349             case Types.INTEGER:
350             case Types.BIGINT:
351             case Types.SMALLINT:
352                 return new BigInteger(value);
353             case Types.BIT:
354                 return value.equals("1");
355             case Types.NUMERIC:
356             case Types.DECIMAL:
357             case Types.REAL:
358             case Types.FLOAT:
359                 return new BigDecimal(value);
360             case Types.DATE:
361                 return DateTime.parse(value, DateTimeFormat.forPattern(
362                         dateFormats.DATE_FORMAT.toString())).toString(DATE_TIME_FORMAT);
363             case Types.TIMESTAMP:
364                 return DateTime.parse(value, DateTimeFormat.forPattern(
365                         dateFormats.DATETIME_FORMAT.toString())).toString(DATE_TIME_FORMAT);
366             default:
367                 return value;
368 
369         }
370     }
