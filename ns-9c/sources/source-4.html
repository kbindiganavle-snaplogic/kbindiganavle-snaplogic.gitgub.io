326     public void execute() {
327         if (inputViews.isEmpty() || !initAzureStorage()) {
328             return;
329         }
330         Connection conn = jdbcOperations.acquireConnection(account);
331         if(jdbcOperations.tableExists(conn,
332                 account.getDatabaseConfig().getDatabaseName(), tableName)) {
333             tableMetaData = getTableMetaData();
334         }
335         long recordCount = 0L;
336         for (InputView inputView : inputViews) {
337             for (Document document : inputViews.documents(inputView)) {
338                 processDocument(document, inputView.getName());
339             }
340             recordCount = inputView.getObjectCount();
341         }
342         try {
343             azureStorageOutputStream.close();
344         } catch (IOException e1) {
345             SnapDataException snapDataException = new SnapDataException(e1, ERR_BLOB_CREATION_MSG)
346                     .withReason(e1.getMessage())
347                     .withResolution(ERR_BLOB_CREATION_RESOLUTION);
348             errorViews.write(snapDataException);
349             return;
350         }
351         /**
352          * At least one record should be written to azure storage to confirm the azure blob
353          * creation.
354          **/
355         if (recordCount > 0) {
356             isAzureBlobCreated = true;
357         }

397      * Extracts values from the incoming document and return them as a stream of bytes
398      * Delimiter is used to separate these values
399      *
400      * @param data
401      *
402      * @return stream of bytes representing values from data
403      */
404     private byte[] getValuesFromDocument(Map<String, Object> data) {
405         List<Object> finalValues = new ArrayList<>();
406         for (Object value : data.values()) {
407             if (value instanceof DateTime || value instanceof LocalDateTime) {
408                 value = converterRegistry.to(DateTime.class).to(value);
409             } else if (value instanceof Date || value instanceof LocalDate) {
410                 value = converterRegistry.to(LocalDate.class).to(value);
411             } else if (value instanceof Time || value instanceof LocalTime) {
412                 value = converterRegistry.to(LocalTime.class).to(value);
413             } else if (value instanceof BigInteger || value instanceof BigDecimal ||
414                     value instanceof Number || value instanceof Long || value instanceof Integer
415                     || value instanceof Short) {
416                 value = new BigDecimal(value.toString()).toPlainString();
417             }
418             finalValues.add(value);
419         }
420         return StringUtils.join(finalValues, DEFAULT_DATA_DELIMITER).getBytes(charset);
421     }

522     private void generateExternalTableSchema(Map<String, Object> data) {
523         externalTableSchema = new StringBuilder();
524         boolean isFirstKey = true;
525         for (String key : documentKeys) {
526             if (!isFirstKey) {
527                 externalTableSchema.append(COMMA);

529                 isFirstKey = false;
530             }
531             externalTableSchema.append(key);
532             externalTableSchema.append(SPACE);
533             Connection conn = jdbcOperations.acquireConnection(account);
534             if (bulkLoadSchema.equalsIgnoreCase(BULK_LOAD_SCHEMA_EXISTING_TABLE) &&
535                     jdbcOperations.tableExists(conn, account.getDatabaseConfig().getDatabaseName(),
536                             tableName) && tableMetaData != null) {
537                 String dataType = mapDataType(tableMetaData.getColumnTypeName(key).toUpperCase());
538                 externalTableSchema.append(dataType);
539                 LOG.debug("Data Type generated from existing table: " + dataType);
540             } else {
541                 externalTableSchema.append(getDataType(data.get(key)));
542             }
543             LOG.debug("Data Type generated from input schema: " + getDataType(data.get(key)));
544             columnNames.add(String.format(FMT_QUOTED_COLUMN_NAME, key));

559                 value instanceof Integer || value instanceof Short) {
560             return DATA_TYPE_BIGINT;
561         } else if (value instanceof Number || value instanceof BigDecimal) {
562             return DATA_TYPE_FLOAT;
563         } else if (value instanceof DateTime || value instanceof LocalDateTime) {
564             return DATA_TYPE_DATETIME;
565         } else if (value instanceof Date || value instanceof LocalDate) {
566             return DATA_TYPE_DATE;
567         } else if (value instanceof Time || value instanceof LocalTime) {
568             return DATA_TYPE_TIME;
569         } else if (value instanceof Boolean) {
570             return DATA_TYPE_BIT;
571         } else {
572             return DATA_TYPE_VARCHAR;
573         }

778     protected boolean isAzureSqlDataWarehouse() {
779         /*
780         There are two steps to determine if a given Azure Sql instance is a Data Warehouse or not.
781         Step 1: Connect to master database of the given Azure Sql instance
782         Step 2: Run the following query:
783                 SELECT ds.edition [Edition] FROM sys.database_service_objectives AS ds WHERE ds.database_id =
784                  (SELECT db.database_id FROM sys.databases AS db WHERE UPPER(db.name) = 'target DB name in upper-case')
785 
786         If the above query throws exception then its not a Data Warehouse.
787         Or, if the output column of the query is not equal to 'DataWarehouse' then is not a Data Warehouse.
788          */
789         if (account instanceof BaseDatabaseAccount) {
790             String jdbcUrl = ((BaseDatabaseAccount) account).createJdbcUrlWithMasterDatabase();
791             DatabaseConfig databaseConfig = account.getDatabaseConfig();
792             try (Connection connToMasterDb = DriverManager.getConnection(jdbcUrl,
793                     databaseConfig.getUsername(), databaseConfig.getPassword())) {
794                 try (Statement statement = connToMasterDb.createStatement()) {
795                     String hostName = databaseConfig.getDatabaseName().toUpperCase();
796                     String query = String.format(QUERY_TO_CHECK_IF_CURRENT_INSTANCE_DATA_WAREHOUSE,
797                             hostName);
798                     try (ResultSet resultSet = statement.executeQuery(query)) {
799                         resultSet.next();
800                         return resultSet.getString(1).equalsIgnoreCase(DATA_WAREHOUSE);
801                     }
802                 }
803             } catch (SQLException e1) {
804                 // This instance may not be a Data Warehouse
805                 LOG.error("Exception encountered while trying to determine if the given " +
806                         "Azure Sql instance is a Data Warehouse", e1);
807                 return false;
808             }
809         } else {
810             return false;
811         }
812     }

826                 queryToDoBulkLoad = String.format(QUERY_BULK_INSERT_APPEND, tableName,
827                         columnList, columnList, externalTableName);

830                         externalTableName);
831             }
832             LOG.debug("Query to do bulk load: {}", queryToDoBulkLoad);
833             dbUtils.executeWriteQuery(queryToDoBulkLoad);
834             return true;
835         } catch (SQLException e1) {
836             isAzureBlobNotToBeDeleted = true;
837             String errorMessage = e1.getMessage();
838             if (errorMessage.contains(BULK_LOAD_OVERWRITE_ERROR_MSG)) {
839                 throwDbQueryExecuteException(e1, QUERY_NAME_BULK_LOAD_OVERWRITE,
840                         ERR_BULK_LOAD_OVERWRITE_RESOLUTION);
841             } else if (errorMessage.contains(BULK_LOAD_APPEND_ERROR_MSG)) {
842                 throwDbQueryExecuteException(e1, QUERY_NAME_BULK_LOAD_APPEND,
843                         ERR_BULK_LOAD_APPEND_RESOLUTION);
844             } else {
845                 throwDbQueryExecuteException(e1, isBulkLoadAppend ? QUERY_NAME_BULK_LOAD_APPEND :
846                         QUERY_NAME_BULK_LOAD_OVERWRITE,
847                         String.format(ERR_BULK_LOAD_GENERAL_RESOLUTION, putRejectedRowLocation()));
848             }
849             return false;
850         }
