358      * Removes currency symbol from currency, and returns the currency value
359      * @param value currency
360      * @return currency value if symbol is valid, input if symbol is not present, null otherwise
361      */
362     private boolean isCurrency(Object value) {
363         if (value instanceof BigInteger || value instanceof BigDecimal) {
364             return true;
365         }
366 
367         if (!(value instanceof String)) {
368             return false;
369         }
370 
371         String strVal = value.toString();
372         int numNegativeSymbols = StringUtils.countMatches(strVal, "-");
373 
374         if (numNegativeSymbols > 1 || (numNegativeSymbols == 1
375                 && strVal.substring(0, strVal.indexOf('-')).matches(".*\\d.*"))) {
376             return false;
377         }
378 
379         boolean isNegative = numNegativeSymbols == 1;
380         String input = isNegative ? strVal.replace("-", "").trim() : strVal;
381 
382         Pattern pattern = Pattern.compile("([0-9,.-]+)");
383         Matcher matcher = pattern.matcher(input);
384 
385         String currencyValue = null;
386 
387         // we only take the first currency value to prevent scenarios like "123.USD.456"
388         if (matcher.find()) {
389             currencyValue = matcher.group();
390         }
391 
392         if (currencyValue != null) {
393             String currencySymbol = input.replace(currencyValue, "").trim();
394 
395             if (currencySymbol.equals(StringUtils.EMPTY)) {
396                 // no symbols found. the whole input would be a currency value
397                 return isValidCurrency(input);
398             } else if (CURRENCY_SYMBOL_SET.contains(currencySymbol)
399                     || CURRENCY_CODE_SET.contains(currencySymbol)) {
400                 // a valid symbol found, let's see if the currency value is also valid
401                 return isValidCurrency(currencyValue);
402             }
403         }
404 
405         return false;
406     }
