227     private void processExport(String munchkinId, Document document) {
228         // Step 1 - Create the Extract using UI data
229         //      https://{munchkinId}.mktorest.com/bulk/v1/leads/export/create.json
230         // Step 2 - Enqueue the Extract using ID from step 1
231         //      https://{munchkinId}.mktorest.com/bulk/v1/leads/export/{exportId}/enqueue.json
232         // Step 3 - Poll the status until it contains success
233         //      https://{munchkinId}.mktorest.com/bulk/v1/leads/export/{exportId}/status.json
234         // Step 4 - Retrieve the data
235         //      https://{munchkinId}.mktorest.com/bulk/v1/leads/export/{exportId}/file.json
236         try {
237             String host = MARKETO_ROOT_URL.replace(MUNCHKIN_ID_TOKEN,
238                     munchkinId);
239 
240             String result = createExport(host);
241             exportId = getExportIdFromResult(result);
242 
243             enqueueExport(host, exportId);
244 
245             waitForExportToFinish(host, exportId);
246 
247             downloadExport(host, exportId, document);
248         } catch (SnapDataException e) {
249             if (exportId != null) {
250                 String reasonWithExportId = e.getReason() + "; exportID: " + exportId;
251                 e.withReason(reasonWithExportId);
252             }
253             throw e;
254         } catch (Exception e) {
255             SnapDataException sde = new SnapDataException(e, "Failed to %s: %s")
256                     .formatWith(getOperation().getName(), getEntity())
257                     .withResolution(ADDRESS_THE_REPORTED_ISSUE);
258             if (exportId != null) {
259                 String message = sde.getReason() + "; exportID: " + exportId;
260                 sde.withReason(message);
261             }
262             throw sde;
263         }
264     }

323     protected void waitForExportToFinish(final String host, final String exportId) {
324         String result;
325         int statusCalls = 0;
326 
327         statusReporter.begin(10, MINUTES);
328 
329         try {
330             do {
331                 if (statusCalls > 0) {
332                     try {
333                         Thread.sleep(1000L * 60L);  // Doc says wait 60 seconds between polls
334                     } catch (InterruptedException e) {
335                         Thread.currentThread().interrupt();
336                         Throwable root = Throwables.getRootCause(e);
337                         throw new SnapDataException(root, ERROR_CHECKING_STATUS)
338                             .withReason(root.getMessage())
339                             .withResolution(ADDRESS_THE_REPORTED_ISSUE);
340                     }
341                 }
342                 String url = (host + STATUS_COMMAND).replace(EXPORT_ID_PATTERN, exportId);
343                 result = performOAuthRetryableHttpGet(url);
344 
345                 statusReporter.worked(statusCalls + 1L);
346             } while (!result.contains(STATUS_COMPLETE) && statusCalls++ < 10);  // up to 10 minutes
347 
348             if (statusCalls >= 10) {
349                 throw new SnapDataException(ERROR_CHECKING_STATUS)
350                     .withReason(STATUS_TOOK_TOO_LONG)
351                     .withResolution(ADDRESS_THE_REPORTED_ISSUE);
352             }
353         } finally {

367             final Document document) {
368         final String result;
369         final Header[] headers;
370 
371         String url = (host + FILE_COMMAND).replace(EXPORT_ID_PATTERN, exportId);
372         try {
373             MarketoBulkResponse response = performOAuthRetryableHttpRequest(HttpMethod.GET, url,
374                     null);
375             result = response.getContent();
376             headers = response.getHeaders();
377         } catch (IOException e) {
378             Throwable root = Throwables.getRootCause(e);
379             throw new SnapDataException(root, ERROR_RETREIVING_EXTRACTION)
380                     .withReason(root.getMessage())
381                     .withResolution(ADDRESS_THE_REPORTED_ISSUE);
382         }
383         Map<String, String> headerData = new LinkedHashMap<>(headers.length);
384         for (Header header : headers) {
385             headerData.put(header.getName().toLowerCase(), header.getValue());
386         }
387         headerData.put(EXPORT_ID, exportId);
388         Document header = documentUtility.newDocumentFor(document, headerData);
389         outputViews.write(new BinaryOutput() {
390             @Override
391             public Document getHeader() {
392                 return header;
393             }
394 
395             @Override
396             public void write(WritableByteChannel writeChannel) throws IOException {
397                 CharsetEncoder encoder = StandardCharsets.UTF_8.newEncoder();
398                 writeChannel.write(encoder.encode(CharBuffer.wrap(result)));
399             }
400         });
401     }

419     private String getExtractionPayload() {
420         StringBuilder result = new StringBuilder();
421 
422         result.append(OPEN_BRACE);
423         result.append(FIELDS_KEY).append(OPEN_BRACKET);
424 
425         for (String field : evaluatedFieldValues) {
426             result.append(QUOTE).append(field).append(QUOTE).append(COMMA);
427         }
428         result.deleteCharAt(result.length() - 1);
429         result.append(CLOSE_BRACKET).append(COMMA);
430         result.append(FORMAT_KEY).append(QUOTE).append(this.fileFormat).append(QUOTE).append(COMMA);
431         result.append(COLUMNS_HEADER_NAMES_KEY).append(OPEN_BRACE);
432         for (String field : evaluatedFieldValues) {
433             result.append(QUOTE).append(field).append(QUOTE).append(COLON)
434                 .append(QUOTE).append(field).append(QUOTE).append(COMMA);
435         }
436         result.deleteCharAt(result.length() - 1);
437         result.append(CLOSE_BRACE).append(COMMA);
438 
439         String startEndString = FILTER_KEY.replace(START_AT, startDate);
440         startEndString = startEndString.replace(END_AT, endDate);
441         result.append(startEndString);
442         result.append(CLOSE_BRACE);
