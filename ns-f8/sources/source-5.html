260             throws Exception {
261         openApiPath = paths.get(0);
262         operation = HttpPost.METHOD_NAME.toLowerCase();
263         // Execute http request for openApiPath and operation which are already set in configure()
264         // for defineInputSchema().
265         inputMap = document.get(Map.class);
266         if (bulkExportConfig.entity.equals(LEADS) ||
267                 bulkExportConfig.entity.equals(CUSTOM_OBJECTS)) {
268             if (inputMap.containsKey(REQUEST_BODY)) {
269                 HashMap<String, Object> requestBodyMap =
270                         (HashMap<String, Object>) inputMap.get(REQUEST_BODY);
271                 if (requestBodyMap != null && requestBodyMap.containsKey(FILTER)) {
272                     HashMap<String, Object> filterMap =
273                             (HashMap<String, Object>) requestBodyMap.get(FILTER);
274                     if (filterMap != null && filterMap.size() > 1) {
275                         throw new SnapDataException(ERR_EXPORT_INVALID_REQUEST)
276                                 .withReason(ERR_EXPORT_INVALID_REQUEST_REASON_MULT_FILTER)
277                                 .withResolution(ERR_EXPORT_INVALID_REQUEST_RESOLUTION_MULT_FILTER);
278                     }
279                 }
280             }
281         }
282         writeApiNameToInput();
283         outDoc = failsafeOAuthProcess(inputMap, document);
284         Map<String, Object> outMap = outDoc.get(Map.class);
285         // check if the export batch job is successfully created
286         if (isSuccess(outMap)) {
287             Object value = readJsonPath(outMap, ENTITY, RESULT_0, EXPORT_ID);
288             if (value instanceof String) {
289                 // export ID found
290                 return (String) value;
291             }
292         }
293         // export ID not found
294         Object value = readJsonPath(outMap, ENTITY, ERRORS);
295         throw new SnapDataException(FMT_ERROR_CREATE_BATCH_JOB)
296                 .formatWith(bulkExportConfig.entity)
297                 .withReason(value == null ? ERROR_EXPORT_ID_NOT_FOUND : value.toString())

325     protected void checkStatus(Document document) throws Exception {
326         // set OpenAPI path and operation
327         openApiPath = paths.get(2);
328         operation = HttpGet.METHOD_NAME.toLowerCase();
329 
330         int delaySeconds = 0;
331         String status = null;
332         try {
333             do {
334                 // wait progressively
335                 delaySeconds += incrementSeconds;
336                 sleepNoInterrupt(delaySeconds, TimeUnit.SECONDS, LOG);
337                 // refresh access token if necessary
338                 updateAuthHeader();
339                 // execute http request
340                 outDoc = failsafeOAuthProcess(inputMap, document);
341                 Map<String, Object> outMap = outDoc.get(Map.class);
342                 Object value = readJsonPath(outMap, ENTITY, RESULT_0, STATUS);
343                 if (value instanceof String) {
344                     status = (String) value;
345                 } else {
346                     throw new SnapDataException(FMT_ERROR_CHECK_BULK_EXPORT_JOB_STATUS)
347                             .formatWith(bulkExportConfig.entity)
348                             .withReason(readJsonPath(outMap, ENTITY).toString())
349                             .withResolutionAsDefect();
350                 }
351                 LOG.info("Marketo bulk job status: {}", status);
352                 long lapse = System.currentTimeMillis() - startTime;
353                 statusReporter.notify(STATUS_MESSAGE, lapse / 1000,
354                         timeoutSeconds > 0 ? timeoutSeconds : UNDERSCORE);
355                 if (timeoutSeconds > 0 && lapse > timeoutMillis &&
356                         !COMPLETED.equalsIgnoreCase(status)) {
357                     throw new SnapDataException(FMT_ERROR_TIMEOUT_EXPORT_JOB_STATUS)
358                             .formatWith(bulkExportConfig.entity)
359                             .withReason(String.format(ERROR_EXPORT_JOB_TIMED_OUT, timeoutSeconds))
360                             .withResolution(RESOLUTION_TIMEOUT_EXPORT);
361                 }
362             } while (!COMPLETED.equalsIgnoreCase(status));
363         } finally {
364             statusReporter.done();
365         }

379         Object value = readJsonPath(outMap, STATUS_LINE, STATUS_CODE);
