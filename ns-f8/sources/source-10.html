327     protected void process(Document document, final String inputViewName) {
328         AtomicReference<HttpEntity> requestEntity = new AtomicReference<>();
329         try {
330             Object input = document == null ? null : document.get();
331             Map<String, Object> inputMap = openApiHelper.getAsMap(input);
332             Map<String, Object> inputMapReconciledQueryParams;
333             if (queryParams.isEmpty()) {
334                 // Request via input doc doesn't need query param adjustment
335                 inputMapReconciledQueryParams = inputMap;
336             } else {
337                 // Copy the input doc and merge query params from snap into input docs
338                 inputMapReconciledQueryParams = reconcileQueryParameters(inputMap, queryParams);
339             }
340 
341             // evaluate expression properties
342             openApiPath = openApiPathExpr.eval(document);
343             enableAutoPaging = Utils.toBoolean(enableAutoPagingExp.eval(document));
344             if (StringUtils.isBlank(openApiBase) || StringUtils.isBlank(openApiPath)) {
345                 throw new SnapDataException(ERR_SERVICE_URL)
346                     .formatWith(openApiBase, openApiPath)
347                     .withReason(REASON_INVALID_OPENAPI_URL)
348                     .withResolution(RESOLUTION_ADDRESS_ISSUE);
349             }
350             // get operation spec
351             computeOpenApiSpec();
352             httpUtils.setIsVersion2(isVersion2);
353             String endpointFromSummary = getEndpointFromSummary(openApiPath);
354 
355             Object contentKey = CONTENT_KEY_MAP.get(endpointFromSummary);
356             Object filenameKey = FILENAME_KEY_MAP.get(endpointFromSummary);
357             if (contentKey != null && getRESTMessageKind().equals("post")) {
358                 openApiHelper.withMultipartFileContentKey((String) contentKey);
359             }
360             if (filenameKey != null && getRESTMessageKind().equals("post")) {
361                 openApiHelper.withMultipartFilenameKey((String) filenameKey);
362             }
363             Map<String, Object> operationSpec = openApiHelper.getSpec(openApiSpec,
364                 PATHS, endpointFromSummary, getRESTMessageKind());
365             if (MapUtils.isEmpty(operationSpec)) {
366                 throw new SnapDataException(ERR_SPEC_NOT_FOUND)
367                     .formatWith(openApiPath, getRESTMessageKind())
368                     .withReason(REASON_SPEC_NOT_FOUND)
369                     .withResolution(RESOLUTION_ADDRESS_ISSUE);
370             }
371             String nextPageToken = StringUtils.EMPTY;
372             Document outDoc;
373             do {
374                 String finalNextPageToken = nextPageToken;
375                 outDoc = Failsafe.with(OAUTH_RETRY_POLICY)
376                         .onRetry(new CheckedConsumer<Throwable>() {
377                             @Override
378                             public void accept(
379                                     Throwable throwable) {
380                                 if (account instanceof MarketoOAuth2Account) {
381                                     LOG.error("Refreshing token and retrying marketo request.");
382                                     ((MarketoOAuth2Account) account).refreshAccessTokenWithRetry();
383                                 }
384                             }
385                         })
386                         .get(() -> {
387                             List<Header> requestHeaders = new ArrayList<>();
388                             requestHeaders.add(new BasicHeader(ACCEPT, ACCEPT_ALL));
389                             // Service starts with just the base and path
390                             String serviceUrl = JfsUtils.concat(openApiBase,
391                                     endpointFromSummary);
392                             resetNextPageToken(inputMapReconciledQueryParams, finalNextPageToken);
393                             prepareMaxReturnPagination(inputMapReconciledQueryParams);
394                             // process parameters
395                             serviceUrl = openApiHelper.processParameters(operationSpec,
396                                     inputMapReconciledQueryParams,
397                                     serviceUrl, requestHeaders);
398 
399                             // The URI Components encode commas which makes Marketo grumpy
400                             serviceUrl = serviceUrl.replaceAll(HEX_COMMA, COMMA);
401                             // execute http request
402                             // get auth header now to use an updated access token if necessary
403                             authHeader = account.connect();
404                             if (authHeader != null) {
405                                 requestHeaders.add(authHeader);
406                             }
407                             // get http entity
408                             requestEntity.set(openApiHelper.getHttpEntity(operationSpec, input,
409                                     requestHeaders));
410                             // When requestEntity is null, ContentType header is not getting added.
411                             // So manually adding the default value to avoid regression
412                             if (requestEntity.get() == null) {
413                                 requestHeaders.add(new BasicHeader(CONTENT_TYPE,
414                                         APPLICATION_JSON + CHARSET_UTF_8));
415                             }
416                             // get http method
417                             HttpUriRequest httpUriRequest = openApiHelper.getHttpUriRequest(
418                                     serviceUrl, getRESTMessageKind(), requestEntity.get());
419                             // set request headers
420                             httpUriRequest.setHeaders(requestHeaders.toArray(new Header[0]));
421                             HttpResponse response = executeHttpRequest(httpUriRequest);
422                             return httpUtils.processHttpResponse(response, requestHeaders,
423                                     document, openApiHelper.getSpec(operationSpec, RESPONSES));
424                         });
425                 /*
426                  * - Date-Based paging token is generated by calling the GetPagingToken endpoint.
427                  * and never have a 2nd page to process. Excluding it to process response to
428                  * skip pagination.
429                  *
430                  * - To retrieve activities from Marketo:
431                  * For the first call, use the Get Paging Token API to get nextPageToken. For
432                  * subsequent calls to this endpoint, use the nextPageToken returned from the
433                  * response. This endpoint will always return the nextPageToken. If the
434                  * moreResult attribute is true, this means more results are available.
435                  * Continue to call this endpoint until the moreResult attribute returns false,
436                  * which means there are no results available. The nextPageToken returned from
437                  * this API should always be reused for the next iteration of this call.
438                  *
439                  */
440                 if (GET_PAGINGTOKEN.equalsIgnoreCase(endpointFromSummary) ||
441                         (hasMoreResultsSupport(outDoc) && !hasMoreResults(outDoc))) {
442                     outputViews.write(outDoc, document);
443                     break;
444                 }
445                 if (enableAutoPaging) {
446                     nextPageToken = getNextPageToken(outDoc);
447                 }
448             } while (hasNext(nextPageToken, outDoc, document));
449         } catch (SuggestViewAbortException e) {
450             throw e;
451         } catch (SnapDataException e) {
452             errorViews.write(e, document);
453         } catch (Exception e) {
454             errorViews.write((SnapDataException)
455                 new SnapDataException(e, ERR_PROCESS_OPENAPI_SPEC)
456                     .withResolution(RESOLUTION_ADDRESS_ISSUE), document);
457         } finally {
458             EntityUtils.consumeQuietly(requestEntity.get());
459         }
460     }

462     // When the 'maxReturn' input query param value exists and is larger than 200, the server
463     // returns an error entity, we will use this param as the total number of records the user
464     // wants to download. maxReturn == 0 is regarded as all records from the offset. If the offset
465     // input query parameter is not given, its default value is 0.
466     // After the first page
467     private void prepareMaxReturnPagination(Map<String, Object> inputMap) {
468         boolean isFirstPage = false;
469         if (maxReturn == null) {
470             isFirstPage = true;
471             maxReturn = getLongFrom(jsonPathUtil.nullableRead(MAX_RETURN, inputMap));
472         }
473         if (maxReturn == null) {
474             return;
475         } else if (maxReturn == 0) {
476             jsonPathUtil.write(inputMap, MAX_RETURN, maxPageSize);
477             maxReturn = Long.MAX_VALUE;
478         } else if (maxReturn < 0) {
479             throw new SnapDataException(ERR_INVALID_INPUT_QUERY_PARAMETER)
480                     .formatWith(maxReturn)
481                     .withReason(REASON_CANNOT_BE_NEGATIVE)
482                     .withResolution(RESOLUTION_POSITIVE_OR_ZERO);
483         } else if (maxReturn > maxPageSize) {
484             jsonPathUtil.write(inputMap, MAX_RETURN, maxPageSize);
485             maxReturn -= maxPageSize;
486         } else {
487             // this is the last page
488             jsonPathUtil.write(inputMap, MAX_RETURN, maxReturn);
489             maxReturn = null;
490         }
491         // set offset for the next page
492         Long offset = getLongFrom(jsonPathUtil.nullableRead(OFFSET, inputMap));
493         if (offset == null) {
494             jsonPathUtil.write(inputMap, OFFSET, 0);
495         } else if (!isFirstPage) {
496             jsonPathUtil.write(inputMap, OFFSET, offset + maxPageSize);
497         }
498     }

500     private boolean hasNext(String nextPageToken, Document outDoc, Document inDoc) {
501         boolean hasNext = false;
502         if (StringUtils.isNotEmpty(nextPageToken)) {
503             hasNext = true;
504         } else if (maxReturn != null) {
505             // If it is a maxReturn-type pagination, the size of $entity.result in the output doc
506             // is the only indication for whether there is a next page or not.
507             Map<String, Object> outMap = outDoc.get(Map.class);
508             List<Object> result = jsonPathUtil.nullableRead(ENTITY_RESULT, outMap);
509             if (CollectionUtils.isNotEmpty(result)) {
510                 // check if there are more pages
511                 hasNext = result.size() >= maxPageSize;
512             }
513             if (!hasNext) {
514                 maxReturn = null;
515             }
516         }
517         // Done with checking outDoc, now we can write the output document to the output view
518         outputViews.write(outDoc, inDoc);
519         return hasNext;
520     }

544     @VisibleForTesting
545     protected Map<String, Object> reconcileQueryParameters(Map<String, Object> inputMap,
546                                                            Map<String, Object> snapQueryParams) {
547         // Request body is always copied over without change, shallow copy is fine
548         Map<String, Object> inputMapCopy = new HashMap<>();
549         inputMapCopy.put(REQUEST_BODY, inputMap.get(REQUEST_BODY));
550         // Map under "parameters" will need to be present no matter what
551         Map<String, Object> inputMapParametersCopy = new HashMap<>();
552         inputMapCopy.put("parameters", inputMapParametersCopy);
553 
554         Map<String, Object> inputMapParametersQueryCopy;
555 
556         if (inputMap.get("parameters") == null) {
557             // If parameters is missing, just make sure $.parameters.query exists to merge
558             // snap setting query params in
559             inputMapParametersQueryCopy = new HashMap<>();
560             inputMapParametersCopy.put("query", inputMapParametersQueryCopy);
561             inputMapCopy.put("parameters", inputMapParametersCopy);
562         } else {
563             // If parameters isn't empty, then we need to copy its contents before we make any edits
564             // to the copy
565             // Start off with a blank map linked to "parameters"
566             inputMapCopy.put("parameters", inputMapParametersCopy);
567 
568             // Then for each key (param type) within that map in the original input, replicate
569             // the map, if any, present under the key (param type)
570             Map<String, Object> inputMapParameters =
571                     (Map<String, Object>) inputMap.get("parameters");
572             inputMapParameters.keySet().forEach(paramType -> {
573                 if (inputMapParameters.get(paramType) != null) {
574                     // param type e.g. "path", "query"
575                     Map<String, Object> inputParameterTypeMap =
576                             (Map<String, Object>) inputMapParameters.get(paramType);
577                     Map<String, Object> inputParameterTypeMapCopy = new HashMap<>();
578                     // copy all the actual parameters as-is
579                     inputParameterTypeMap.keySet().forEach(paramName ->
580                             inputParameterTypeMapCopy.put(paramName,
581                                     inputParameterTypeMap.get(paramName)));
582                     inputMapParametersCopy.put(paramType, inputParameterTypeMapCopy);
583                 }
584             });
585 
586             if (inputMapParametersCopy.get("query") == null) {
587                 // If query path is missing, just make that and put it into parameters, but don't
588                 // otherwise alter parameters
589                 inputMapParametersQueryCopy = new HashMap<>();
590                 inputMapParametersCopy.put("query", inputMapParametersQueryCopy);
591             } else {
592                 // If query path is present, check for duplicates
593                 inputMapParametersQueryCopy =
594                         (Map<String, Object>) inputMapParametersCopy.get("query");
595                 Map<String, Object> duplicateParams = new HashMap<>(inputMapParametersQueryCopy);
596                 duplicateParams.keySet().retainAll(snapQueryParams.keySet());
597                 if (!duplicateParams.isEmpty()) {
598                     throw new SnapDataException(ERR_QUERY_PARAMS_CONFLICT)
599                             .withReason(REASON_INVALID_QUERY_PARAMS)
600                             .withResolution(ERR_QUERY_PARAMS_CONFLICT_RESOLUTION);
601                 }
602             }
603         }
604         // With everything copied and query params verified de-duped, merge the snap settings
605         // query params with the input doc query params
606         inputMapParametersQueryCopy.putAll(snapQueryParams);
607         return inputMapCopy;
608     }

764     private void createEndpointMap(Map<String, Object> paths) {
765         summaryEndpointMap = new TreeMap<>();   // so it's sorted
766         for (String key : paths.keySet()) {
767             if (!isExcluded(key)) {
768                 Map<String, Object> element = (Map<String, Object>) paths.get(key);
769                 for (String elementName : element.keySet()) {
770                     if (StringUtils.equals(elementName, getRESTMessageKind())) {
771                         Map<String, Object> values = (Map<String, Object>) element
772                             .get(elementName);
773                         String summary = (String) values.get(SUMMARY);
774                         summaryEndpointMap.put(summary, key);
775                     }
776                 }
777             }
778         }
