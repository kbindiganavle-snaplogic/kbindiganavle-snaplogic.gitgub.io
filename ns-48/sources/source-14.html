140     private static SSLConnectionSocketFactory getSslSocketFactory(boolean trustAll,
141             Account account) {
142         KeyManager[] keyManagers = null;
143         TrustManager[] trustManagers = null;
144         X509HostnameVerifier hostnameVerifier =
145                 SSLConnectionSocketFactory.BROWSER_COMPATIBLE_HOSTNAME_VERIFIER;
146 
147         if (account instanceof SSLAccount) {
148             SSLAccount sslAccount = (SSLAccount) account;
149             keyManagers = sslAccount.getKeyManagers();
150             trustManagers = sslAccount.getTrustManagers();
151         }
152         if (trustAll) {
153             trustManagers = TM_TRUST_ALL;
154             hostnameVerifier = SSLConnectionSocketFactory.ALLOW_ALL_HOSTNAME_VERIFIER;
155         }
156         try {
157             SSLContext sslContext = SSLContext.getInstance(SSLSocketFactory.TLS);
158             sslContext.init(keyManagers, trustManagers, new SecureRandom());
159             return new SSLConnectionSocketFactory(sslContext, hostnameVerifier);
160         } catch (NoSuchAlgorithmException | KeyManagementException e) {
161             throw new ExecutionException(e, TRUST_STORE_FAILURE)
162                     .withReason(e.getMessage())
163                     .withResolutionAsDefect();
164         }
165     }

213     public void execute(Document document, String envelope, List<Header> headerList,
214             List<Map<String, Object>> attachments) {
215         // As this flag is carried over from earlier documents, making sure it gets reset for
216         // each document so that every request to try first with empty SOAPAction
217         failedWithEmptySOAPAction = false;
218         if (HTTP_PROTOCOL_LABEL.equals(protocolType)) {
219             executeHttp(document, envelope, headerList, attachments);
220         } else {
221             executeCXF(document, envelope, headerList, attachments);
222         }
223     }

225     protected void executeCXF(final Document document, final String envelope, final List<Header>
226             headerList, final List<Map<String, Object>> attachments) {
227         Bus bus = dispatch.getClient().getBus();
228         List<Interceptor<? extends Message>> outInterceptors = bus.getOutInterceptors();
229         Header[] headers = headerList.toArray(new Header[0]);
230         if (outInterceptors != null) {
231             for (Interceptor<? extends Message> interceptor : outInterceptors) {
232                 if (interceptor instanceof SoapMessageSenderInterceptor) {
233                     ((SoapMessageSenderInterceptor) interceptor).setHeaders(headers);
234                 }
235             }
236         }
237 
238         LOGGER.debug("DISPATCHING_SOAP_REQUEST", envelope);
239         Callable<Boolean> callable = () -> {
240             try {
241                 try {
242                     soapExecuteAndProcessResponse(document, envelope, attachments);
243                     return true;
244                 } catch (Exception e) {
245                     Throwable rootCause = Throwables.getRootCause(e);
246                     if (!failedWithEmptySOAPAction && (rootCause instanceof SoapFault)
247                             && (StringUtils.containsIgnoreCase(rootCause.getMessage(),
248                             ERR_MSG_SOAPACTION)
249                             || StringUtils.containsIgnoreCase(rootCause.getMessage(),
250                             ERR_MSG_SOAP_ACTION))) {
251                         LOGGER.warn("SOAP request failed: {}, retrying after generating SOAPAction",
252                                 rootCause.getMessage());
253                         failedWithEmptySOAPAction = true;
254                         addSOAPAction(dispatch, clientBuilder);
255                         soapExecuteAndProcessResponse(document, envelope, attachments);
256                         return true;
257                     }
258                     throw e;
259                 }
260             } catch (Exception e) {
261                 Throwable rootCause = Throwables.getRootCause(e);
262                 if (rootCause instanceof HTTPException &&
263                         SC_LENGTH_REQUIRED == ((HTTPException) rootCause).getResponseCode()) {
264                     HTTPConduit httpConduit = (HTTPConduit) dispatch.getClient().getConduit();
265                     HTTPClientPolicy httpClientPolicy = httpConduit.getClient();
266                     httpClientPolicy.setAllowChunking(false);
267                     httpConduit.setClient(httpClientPolicy);
268                     soapExecuteAndProcessResponse(document, envelope, attachments);
269                     return true;
270                 }
271                 throw e;
272             }
273         };
274 
275         try {
276             retryer.call(callable);
277         } catch (RetryException e) {
278             soapResponseProcessor.processException(document, envelope, headerList,
279                     clientBuilder, outInterceptors, e);
280         } catch (java.util.concurrent.ExecutionException e) {
281             throw new SnapDataException(e, Messages.CONCURRENT_EXECUTION_EXCEPTION)
282                     .withReason(e.getMessage())
283                     .withResolutionAsDefect();
284         }
285     }

288             headerList, final List<Map<String, Object>> attachments) {

313     protected Callable<Boolean> getBooleanCallable(final Document document, final String url,
314             final HttpClient client, final HttpPost post, final List<Header> headers) {
315         return () -> {
316             try {
317                 soapHttpExecuteAndProcessResponse(document, url, client, post, headers);
318                 return true;
319             } catch (SOAPActionException e) {
320                 if (!failedWithEmptySOAPAction) {
321                     LOGGER.warn("SOAP request failed: {}, retrying with generated SOAPAction",
322                             e.getMessage());
323                     failedWithEmptySOAPAction = true;
324                     List<Header> updatedHeaders = setupHttpSoapHeader(post);
325                     soapHttpExecuteAndProcessResponse(document, url, client, post, updatedHeaders);
326                     return true;
327                 }
328                 throw e;
329             }
330         };
331     }

333     protected void soapHttpExecuteAndProcessResponse(final Document document, final String url,
334             final HttpClient client, final HttpPost post, final List<Header> headersList) throws
335             IOException {
336         HttpResponse response = client.execute(post, RequestClientUtil.getProxyContext());
337         Header[] headers = new Header[headersList.size()];
338         headers = headersList.toArray(headers);
339         DefaultSoapResponseProcessor defaultSoapResponseProcessor =
340                 (DefaultSoapResponseProcessor) soapResponseProcessor;
341         RestResponseProcessor restResponseProcessor = new RestResponseProcessor
342                 (defaultSoapResponseProcessor.xmlUtils, HttpPost.METHOD_NAME,
343                         defaultSoapResponseProcessor.outputViews,
344                         defaultSoapResponseProcessor.errorViews,
345                         defaultSoapResponseProcessor.documentUtility)
346                 .withShowAllHeaders(outputHttpResponseHeaders);
347         if (!failedWithEmptySOAPAction) {
348             restResponseProcessor = restResponseProcessor.withSnapName(SOAP_EXECUTE);
349         }
350         restResponseProcessor.process(response, document, headers, url);
351     }

366     protected String getSoapAction(final ClientBuilder clientBuilder) {
367         String soapAction = clientBuilder.getSoapAction();
368         if (failedWithEmptySOAPAction) {
369             if ((soapAction == null || soapAction.isEmpty()) &&
370                     clientBuilder.getRequestMessageBody() != null) {
371                 /* Build a custom SOAPAction like the SOAPUI App because some wsdl doesn't like
372                 empty SOAPAction. This case will be execute only after we see "The given SOAPAction
373                 does not match an operation." error for the first time.
374                 Sample wsdl file where it returned error when SOAPAction string sent with non-empty
375                 string https://ws.secureworks.com/api/TicketingService?wsdl */
376                 soapAction = clientBuilder.getRequestMessageBody().getNamespaceURI() + "/Soap/" +
377                         clientBuilder.getRequestMessageBody().getLocalPart();
378             }
379         }
380         return soapAction;
381     }

392     protected List<Header> setupHeaders(final HttpPost post, final List<Header> headers, Header
393             contentTypeHeader) {
394         //Set all the headers and set soapAction
395         setupHttpSoapHeader(post);
396         post.setHeader(contentTypeHeader);
397         for (Header header : headers) {
398             // Always override the SOAP_ACTION and CONTENT_TYPE with the user provided value as
399             // there are no duplicates allowed for those 2 headers.
400             if (SOAP_ACTION.equalsIgnoreCase(header.getName()) ||
401                     HttpHeaders.CONTENT_TYPE.equalsIgnoreCase(header.getName())) {
402                 post.setHeader(header);
403                 continue;
404             }
405             if (allowDuplicateHeaders) {
406                 post.addHeader(header);
407             } else {
408                 post.setHeader(header);
409             }
410         }
411         return Arrays.asList(post.getAllHeaders());
412     }

416         Header soapActionHeader = new BasicHeader(SOAP_ACTION, action);
417         post.setHeader(soapActionHeader);
418         return Arrays.asList(post.getAllHeaders());

691          * Builds the SoapExecutor and validates that the required parameters are set using the
692          * CXF library
693          *
694          * @return
695          * @throws ConfigurationException
696          */
697         public SoapExecutor buildUsingCXF() throws ConfigurationException {
698 
699             if (clientBuilder instanceof ClientBuilderImpl) {
700                 ((ClientBuilderImpl) clientBuilder).enableWSAddressing(enableWSAddressing);
701             }
702             DispatchImpl dispatch = Utility.newDispatch(clientBuilder);
703             Client client = dispatch.getClient();
704             Bus bus = client.getBus();
705 
706             if (isMTOM) {
707                 SOAPBinding soapBinding = (SOAPBinding) dispatch.getBinding();
708                 soapBinding.setMTOMEnabled(true);
709             }
710 
711             if (nullResponseType == null) {
712                 nullResponseType = Messages.NULL_RESPONSE_LEGACY_LABEL;
713             }
714 
715             bus.setProperty(AsyncHTTPConduit.USE_ASYNC, Boolean.TRUE);
716             HTTPConduit httpConduit = (HTTPConduit) dispatch.getClient().getConduit();
717             RequestClientUtil.setProxyConfig(httpConduit);
718 
719             HTTPClientPolicy httpClientPolicy = new HTTPClientPolicy();
720             httpClientPolicy.setConnectionTimeout(TimeUnit.SECONDS.toMillis(timeout));
721             httpClientPolicy.setReceiveTimeout(TimeUnit.SECONDS.toMillis(timeout));
722             httpClientPolicy.setAutoRedirect(true);
723             // If the endpoint doesn't support content chunking in the SOAP request, its http policy
724             // should be disabled here. Then, the Content-Length of the SOAP payload will be
725             // calculated and included in the http request header list.
726             httpClientPolicy.setAllowChunking(allowChunking);
727             for (Header header : soapHttpContextProvider.getHttpHeaders()) {
728                 if (header.getName().equalsIgnoreCase(HttpHeaders.CONTENT_TYPE)) {
729                     httpClientPolicy.setContentType(header.getValue());
730                 }
731             }
732             httpConduit.setClient(httpClientPolicy);
733             bus.getInInterceptors().add(new GZIPInInterceptor());
734             bus.getInInterceptors().add(new NetSuiteResponseInterceptor());
735             soapExecutor.addSOAPAction(dispatch, clientBuilder);
736 
737             // we always capture the contents of the request and response so we can make
738             // it available. The interceptors don't actually log the contents unless the
739             // DEBUG level is enabled.
740             RequestLogInterceptor requestLogInterceptor =
741                     new RequestLogInterceptor(requestLoggingMessageHolder);
742             bus.getOutInterceptors().add(requestLogInterceptor);
743             ResponseLogInterceptor responseLogInterceptor =
744                     new ResponseLogInterceptor(responseLoggingMessageHolder);
745             bus.getInInterceptors().add(responseLogInterceptor);
746 
747             if (account instanceof SoapExecutorCallbackAccount) {
748                 ((SoapExecutorCallbackAccount) account).soapExecuteConfigure(dispatch);
749             }
750 
751             if (trustAllCerts || (soapHttpContextProvider != null &&
752                     soapHttpContextProvider.getHttpHeaders() != null &&
753                     soapHttpContextProvider.getHttpHeaders().length > 0)) {
754                 SoapMessageSenderInterceptor soapMessageSenderInterceptor =
755                         new SoapMessageSenderInterceptor(trustAllCerts);
756                 soapMessageSenderInterceptor.setSSLProtocol(sslProtocol);
757                 bus.getOutInterceptors().add(soapMessageSenderInterceptor);
758             }
759 
760             Retryer<Boolean> retryer = RetryerBuilder.<Boolean>newBuilder()
761                     .retryIfExceptionOfType(Exception.class)
762                     .retryIfRuntimeException()
763                     .withWaitStrategy(WaitStrategies.fixedWait(retryInterval, TimeUnit.SECONDS))
764                     .withStopStrategy(StopStrategies.stopAfterAttempt(maxRequests))
765                     .build();
766 
767             if (soapResponseProcessor == null) {
768                 throw new ConfigurationException(Messages.INVALID_SOAP_EXECUTOR_CONFIGURATION)
769                         .withReason(Messages.SOAP_RESPONSE_PROCESOR_IS_NULL)
770                         .withResolutionAsDefect();
771             }
772             if (invocationService == null) {
773                 throw new ConfigurationException(Messages.INVALID_SOAP_EXECUTOR_CONFIGURATION)
774                         .withReason(Messages.INVOCATION_SERVICE_IS_NULL)
775                         .withResolutionAsDefect();
776             }
777             soapResponseProcessor.validate();
778             soapExecutor.retryer = retryer;
779             soapExecutor.nullResponseType = nullResponseType;
780             soapExecutor.protocolType = protocolType;
781             soapExecutor.clientBuilder = clientBuilder;
782             soapExecutor.dispatch = dispatch;
783             soapExecutor.invocationService = invocationService;
784             soapExecutor.soapResponseProcessor = soapResponseProcessor;
785             soapExecutor.outputHttpResponseHeaders = outputHttpResponseHeaders;
786             soapExecutor.allowChunking = allowChunking;
787             soapExecutor.allowDuplicateHeaders = allowDuplicateHeaders;
788             return soapExecutor;
789         }

802             // .INSTANCE inside of library if the "keep-alive" is "false" this is a known issue
803             // for NTLM accounts [SWAT-1742], where platform changed its default value to "false"
804             // [PLAT-3695]. So we use the DefaultConnectionReuseStrategy.INSTANCE Reuse strategy
805             httpClientBuilder.setConnectionReuseStrategy(DefaultConnectionReuseStrategy.INSTANCE);
806             if (clientBuilder instanceof ClientBuilderImpl) {
807                 ((ClientBuilderImpl) clientBuilder).enableWSAddressing(enableWSAddressing);
808             }
809 
810 
811             // TODO support MTOM for http request
812             // REF: https://www.soapui.org/docs/soap-and-wsdl/attachments.html
813 //            if (isMTOM) {
814 //                SOAPBinding soapBinding = (SOAPBinding) dispatch.getBinding();
815 //                soapBinding.setMTOMEnabled(true);
816 //            }
817 
818             if (nullResponseType == null) {
819                 nullResponseType = Messages.NULL_RESPONSE_LEGACY_LABEL;
820             }
821             RequestConfig requestConfig = RequestConfig.custom()
822                     .setConnectTimeout(timeout * 1000)
823                     .setConnectionRequestTimeout(timeout * 1000)
824                     .setRedirectsEnabled(true)
825                     // If the endpoint doesn't support content chunking in the SOAP request,
826                     // its http policy should be disabled here. Then, the Content-Length of the
827                     // SOAP payload will be calculated and included in the http request header list.
828                     .setStaleConnectionCheckEnabled(allowChunking)
829                     .build();

851                     .retryIfExceptionOfType(Exception.class)
852                     .retryIfRuntimeException()

864                         .withReason(Messages.INVOCATION_SERVICE_IS_NULL)

866             }

868             soapExecutor.retryer = retryer;
869             soapExecutor.nullResponseType = nullResponseType;
870             soapExecutor.protocolType = protocolType;
871             soapExecutor.clientBuilder = clientBuilder;
872             soapExecutor.dispatch = null;
873             soapExecutor.httpClientBuilder = httpClientBuilder;
874             soapExecutor.connectionManager = connMgr;
875             soapExecutor.soapHttpContextProvider = soapHttpContextProvider;
876             soapExecutor.invocationService = invocationService;
