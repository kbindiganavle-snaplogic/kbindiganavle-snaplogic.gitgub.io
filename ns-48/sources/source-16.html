322                                  final @Nullable HttpContextProvider httpContextProvider,
323                                  boolean importDocuments) {
324         try {
325             return WSDL_CACHE.get(wsdlUrl.toString() + importDocuments, () ->
326                     getWsdl(wsdlUrl, httpContextProvider, importDocuments));
327         } catch (java.util.concurrent.ExecutionException | UncheckedExecutionException e) {
328             log.debug("Unable to load data into cache, {}", e.getMessage());
329             Throwable cause = e.getCause();
330             if (cause instanceof ExecutionException) {
331                 throw (ExecutionException)cause;
332             } else if (cause instanceof SnapException) {
333                 throw (SnapException) cause;
334             } else {
335                 throw new ExecutionException(cause, CANNOT_PARSE_WSDL)
336                         .withReason(cause.getMessage())
337                         .withResolutionAsDefect();
338             }
339         }

343      * Parses the WSDL into a {@link Description} object.
344      *
345      * @param wsdlUrl             URL where the wsdl is available
346      * @param httpContextProvider as the optional http context
347      * @param importDocuments     indicates if feature IMPORT_DOCUMENTS should be enabled
348      *
349      * @return wsdlDescription
350      */
351     public Description getWsdl(final URL wsdlUrl,
352                                  final @Nullable HttpContextProvider httpContextProvider,
353                                  boolean importDocuments) {
354         log.debug("getWsdl: wsdlUrl={} importDocuments={}", wsdlUrl, importDocuments);
355         WSDLReader wsdlReader = wsdlReaderCreator.create();
356         Description description;
357         try {
358             Document wsdlDocument =
359                     domUtil.getResourceAsDocument(wsdlUrl, httpContextProvider, TIMEOUT);
360             HttpClientBuilder httpClientBuilder = getHttpClientBuilder(httpContextProvider);
361             try {
362                 wsdlReader.setFeature(IMPORT_DOCUMENTS, importDocuments);
363                 description = wsdlReader.read(wsdlDocument, httpClientBuilder);
364             } catch (Exception e) {
365                 // If we turn importDocuments to true, it is possible that the
366                 // imported Documents are not valid, which stopped the whole parsing process.
367                 // If this happened, we may just stop importing documents for this WSDL file
368                 // and try to parse it again.
369                 if (Throwables.getRootCause(e) instanceof SAXParseException && importDocuments) {
370                     wsdlReader.setFeature(IMPORT_DOCUMENTS, Boolean.FALSE);
371                     description = wsdlReader.read(wsdlDocument, httpClientBuilder);
372                 } else {
373                     throw e;
374                 }
375             }
376             log.debug("getWsdl from document: returning {}", description);
377             return description;
378         } catch (Exception e) {
379             // The exception caught by this catch block will be while obtaining an instance of a
380             //     Document or a Description in the try block.
381             // Observations have shown exceptions occurring when the WSDL is complex containing
382             //     import elements.
383             // In case of an exception, we will try to obtain Description object directly by
384             //     supplying WSDL URL.
385             if (Boolean.TRUE.equals(wsdlReader.getFeature(IMPORT_DOCUMENTS))) {
386                 try {
387                     HttpClientBuilder httpClientBuilder = null;
388                     try (InputStream inputStream =
389                                  getProtocolInputStream(httpContextProvider, wsdlUrl)){
390                         httpClientBuilder = getHttpClientBuilder(httpContextProvider);
391                         InputSource inputSource = new InputSource(inputStream);
392                         inputSource.setSystemId(wsdlUrl.toString());
393                         description = wsdlReader.read(inputSource, httpClientBuilder);
394                     } catch (IOException ioe) {
395                         throw new ExecutionException(ioe, CANNOT_PARSE_WSDL)
396                                 .withReason(ioe.getMessage())
397                                 .withResolution(PLEASE_VERIFY_SERVER_AND_RETRY);
398                     } catch (Exception rte) {
399                         if (Throwables.getRootCause(rte) instanceof SAXParseException) {
400                             wsdlReader.setFeature(IMPORT_DOCUMENTS, Boolean.FALSE);
401                             description = wsdlReader.read(wsdlUrl, httpClientBuilder);
402                         } else {
403                             throw rte;
404                         }
405                     }
406                     log.debug("getWsdl from url: returning {}", description);
407                     return description;
408                 } catch (SnapException se) {
409                     throw se;
410                 } catch (IOException ioe) {
411                     throw new ExecutionException(ioe, CANNOT_PARSE_WSDL)
412                             .withReason(ioe.getMessage())
413                             .withResolution(PLEASE_VERIFY_SERVER_AND_RETRY);
414                 } catch (Exception e1) {
415                     if (e1.getCause() != null) {
416                         if (e1.getCause() instanceof WSDLImportException) {
417                             throw new ExecutionException(e1, CANNOT_PARSE_WSDL)
418                                     .withReason(e1.getCause().getMessage())
419                                     .withResolution(VALIDATE_WSDL);
420                         }
421                     }
422                     Throwable t = Throwables.getRootCause(e);
423                     throw new ExecutionException(t, CANNOT_PARSE_WSDL)
424                             .withReason(t.getMessage())
425                             .withResolutionAsDefect();
426                 }
427             } else if (e instanceof ExecutionException) {
428                 throw (ExecutionException) e;
429             } else {
430                 Throwable t = Throwables.getRootCause(e);
431                 throw new ExecutionException(t, CANNOT_PARSE_WSDL)
432                         .withReason(t.getMessage())
433                         .withResolutionAsDefect();
434             }
435         }
436     }

439         int timeout = TIMEOUT;

441             SoapHttpContextProvider soapHttpContextProvider =
442                     (SoapHttpContextProvider) httpContextProvider;
443             int wsdlTimeout = soapHttpContextProvider.getTimeout() * SEC_MILLIS;
