428     private SOAPMessage getSoapMessage(final String content, Header[] headers, String contentType) {
429         Charset charset = StandardCharsets.UTF_8;
430         try {
431             ContentType ctype = ContentType.parse(contentType);
432             charset = ctype.getCharset();
433         } catch (UnsupportedCharsetException e) {
434             // do nothing and use the default UTF_8 charset value
435             LOG.error("Unsupported Charset for contenttype : " + contentType);
436         }
437         if (charset == null) {
438             charset = StandardCharsets.UTF_8;
439         }
440         //This is done to match the output of the Apache Http with CXF
441         InputStream is = new ByteArrayInputStream(content.getBytes(charset));
442         MimeHeaders mimeHeaders = new MimeHeaders();
443         for (Header header : headers) {
444             mimeHeaders.addHeader(header.getName(), header.getValue());
445         }
446         SOAPMessage soapResponse = null;
447         try {
448             soapResponse = MessageFactory.newInstance(SOAPConstants.DYNAMIC_SOAP_PROTOCOL)
449                     .createMessage(mimeHeaders, is);
450         } catch (IOException | SOAPException e) {
451             throw new SnapDataException(e, Messages.PROCESS_SOAP_RESPONSE_ERROR)
452                     .withReason(String.format(Messages.SOAP_EXCEPTION_RECEIVED_S), e.getMessage())
453                     .withResolution(Messages.PLEASE_REVIEW_MESSAGE_AND_ADDRESS);
454         }
455         return soapResponse;
456     }

476         }

482      * Process the multipart message. This basically spereates the headers and content and it as
483      * a part of the entity object.
484      *
485      * @param httpResponse
486      * @param entity
487      *
488      * @return Parsed Map
489      * @throws IOException
490      */
491     private Map<Integer, Object> processMultipartResponse(final HttpResponse httpResponse, final
492     HttpEntity entity) throws IOException, XMLStreamException {
493         Header[] contentTypeHeader = httpResponse.getHeaders(HttpHeaders.CONTENT_TYPE);
494         String contentTypeValue = contentTypeHeader.length > 0 ? contentTypeHeader[0]
495                 .getValue() : NO_BOUNDARY;
496         ContentType contentTypeParse = ContentType.parse(contentTypeValue);
497         String boundary = contentTypeParse.getParameter(BOUNDARY);
498         MultipartStream multipartStream = new MultipartStream(entity.getContent(),
499                 boundary.getBytes(), BUF_SIZE, null);
500         boolean nextPart = multipartStream.skipPreamble();
501         Map<Integer, Object> multipartContent = new HashMap<>();
502         int count = 0;
503         while (nextPart) {
504             count++;
505             Map<String, Object> part = new HashMap<>();
506             // Header
507             String header = multipartStream.readHeaders();
508             //Split is with new line as all header are added as separate lines in a multipart
509             // message, "\r" are replace as any new line (in different OS) would have "\n" as
510             // part of it so that the splits are consistent
511             String[] headerSplit = header.replace(StringUtils.CR, StringUtils.EMPTY).split
512                     (StringUtils.LF);
513             Map<String, Object> headersMap = new HashMap();
514             for (String myHeader : headerSplit) {
515                 String[] headerParse = myHeader.split(":");
516                 if (headerParse.length == 0) {
517                     continue;
518                 }
519                 String value = headerParse.length >= 2 ? headerParse[1] : StringUtils.EMPTY;
520                 headersMap.put(headerParse[0], value.trim());
521             }
522             part.put(HEADER_FIELD, headersMap);
523             //Body
524             ByteArrayOutputStream bytes = new ByteArrayOutputStream();
525             multipartStream.readBodyData(bytes);
526             nextPart = multipartStream.readBoundary();
527             Object multipartContentData = getMultipartContent(headersMap, bytes);
528             part.put(CONTENT_FIELD, multipartContentData);
529             multipartContent.put(count, part);
530         }
531         return multipartContent;
532     }

534     private Object getMultipartContent(final Map<String, Object> headersMap, final
535     ByteArrayOutputStream bytes) throws IOException, XMLStreamException {
536         String contentType = (String) headersMap.get(HttpHeaders.CONTENT_TYPE);
537         if (containsAny(contentType, BINARY_CONTENT_TYPES)) {
538             return bytes.toByteArray();
539         } else if (contentType.contains(XML_CONTENT)) {
540             Source source = new StreamSource(new StringReader(getContent(bytes.toByteArray(),
541                     headersMap)));
542             return xmlUtils.convertToJson(
543                     xmlInputFactory, jsonOutputFactory, source);
544         } else if (contentType.contains(JSON_CONTENT)) {
545             return OBJECT_MAPPER.readValue(getContent(bytes.toByteArray(), headersMap), TYPE_REF);
546         }
547         return getContent(bytes.toByteArray(), headersMap);
548     }
