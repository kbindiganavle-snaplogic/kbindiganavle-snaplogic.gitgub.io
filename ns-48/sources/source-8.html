176             QName portQName, QName operationQName, @Nullable HttpContextProvider
177             httpContextProvider, boolean importDocuments) {
178         URL wsdlUrl = introspectionService.urlObjectFor(wsdl);
179         description = introspectionService.parseWsdl(wsdlUrl, httpContextProvider,
180                 importDocuments);
181         Endpoint endpoint = introspectionService.getEndpointFrom(description, serviceQName,
182                 portQName);
183         BindingOperation bindingOperation = introspectionService.getOperationFrom(endpoint,
184                 operationQName);
185         QName requestMessageBody =
186                 introspectionService.getRequestBodyMessage(description, bindingOperation);
187         QName responseMessageBody =
188                 introspectionService.getResponseBodyMessage(description, bindingOperation);
189         ClientBuilder.OperationType type;
190         BindingProtocol.SOAPMEPConstants soapmepConstant;
191         switch (description.getVersion()) {
192             case WSDL11:
193                 soapmepConstant = bindingOperation.getMEP();
194                 break;
195             case WSDL20:
196                 try {
197                     Map<QName, String> attributes = bindingOperation.getOtherAttributes();
198                     String mepURI = attributes.get(WSDL20_MEP_QNAME);
199                     if (mepURI == null) {
200                         throw new ConfigurationException(COULD_NOT_IDENTIFY_OPERATION_TYPE)
201                                 .withResolutionAsDefect();
202                     }
203                     if (mepURI.endsWith(SOAP_RESPONSE) || mepURI.endsWith(
204                             SOAP_RESPONSE_WITH_SLASH)) {
205                         soapmepConstant = BindingProtocol.SOAPMEPConstants.SOAP_RESPONSE;
206                     } else if (mepURI.endsWith(REQUEST_RESPONSE) || mepURI.endsWith(
207                             REQUEST_RESPONSE_WITH_SLASH)) {
208                         soapmepConstant = BindingProtocol.SOAPMEPConstants.REQUEST_RESPONSE;
209                     } else if (mepURI.endsWith(ONE_WAY) || mepURI.endsWith(ONE_WAY_WITH_SLASH)) {
210                         soapmepConstant = BindingProtocol.SOAPMEPConstants.ONE_WAY;
211                     } else {
212                         throw new ConfigurationException(COULD_NOT_IDENTIFY_OPERATION_TYPE)
213                                 .withResolutionAsDefect();
214                     }
215                 } catch (XmlException e) {
216                     throw new ConfigurationException(e, COULD_NOT_READ_OPERATION_ATTRIBUTES)
217                             .withResolutionAsDefect();
218                 }
219                 break;
220             default:
221                 throw new IllegalArgumentException();
222         }
223 
224         switch (soapmepConstant) {
225             case SOAP_RESPONSE:
226                 type = ClientBuilder.OperationType.RESPONSE;
227                 break;
228             case REQUEST_RESPONSE:
229                 type = ClientBuilder.OperationType.REQUEST_RESPONSE;
230                 break;
231             case ONE_WAY:
232                 type = ClientBuilder.OperationType.REQUEST;
233                 break;
234             default:
235                 throw new ConfigurationException(String.format(INVALID_OPERATION_TYPE,
236                         bindingOperation.getMEP().toString()));
237         }
238 
239         AbsItfBinding.BindingConstants typeOfBinding;
240         try {
241             typeOfBinding = endpoint.getBinding().getTypeOfBinding();
242         } catch (ClassCastException e) {
243             //            .ElementNSImpl instead of JAXBElement
244             // MK - TODO: remove default binding hack, caused by not recognizing Policy element,
245             //            which gets parsed into com.sun.org.apache.xerces.internal.dom
246             typeOfBinding = getSoapDefaultBindingTypeByNS(serviceQName, portQName, description);
247         }
248         return clientBuilderFactory.getClientBuilder()
249                 .service(serviceQName)
250                 .port(portQName)
251                 .wsdl(wsdlUrl)
252                 .operation(operationQName)
253                 .requestMessageBody(requestMessageBody)
254                 .responseMessageBody(responseMessageBody)
255                 .soapAction(bindingOperation.getSoapAction())
256                 .operationType(type)
257                 .address(endpoint.getAddress())
258                 .typeOfBinding(typeOfBinding.toString());
259     }

262             QName portQName, Description description) {
263         ServiceImpl service = (ServiceImpl) description.getService(serviceQName);
264         TService tService = service.getModel();
265         if (tService == null) {
266             return AbsItfBinding.BindingConstants.SOAP11_BINDING4WSDL11;
267         }
268         AbsItfBinding.BindingConstants typeOfBinding = null;
269         List<TPort> ports = tService.getPort();
270         for (TPort port : ports) {
271             for (Object obj : port.getAny()) {
272                 QName binding = port.getBinding();
273                 if (binding != null && binding.equals(portQName)) {
274                     if (obj instanceof JAXBElement<?>) {
275                         JAXBElement jaxbElement = (JAXBElement) obj;
276                         if (jaxbElement.getName().getNamespaceURI()
277                                 .equalsIgnoreCase("http://schemas.xmlsoap.org/wsdl/soap12/")) {
278                             typeOfBinding = AbsItfBinding.BindingConstants.SOAP12_BINDING4WSDL11;
279                             break;
280                         }
281                     }
282                 }
283             }
284         }
285         if (typeOfBinding == null) {
286             return AbsItfBinding.BindingConstants.SOAP11_BINDING4WSDL11;
287         } else {
288             return typeOfBinding;
289         }

308                     }

310                         Collection<Attachment> attachmentList = new LinkedList<>();
311                         for (Map<String, Object> attachmentMap : attachments) {
312                             Map<String, Object> contentTypeMap = (Map<String, Object>)
313                                     (attachmentMap.get(Messages.ATTACHMENT_CONTENTTYPE_PROP));
314                             Map<String, Object> contentBinaryMap = (Map<String, Object>)
315                                     (attachmentMap.get(Messages.ATTACHMENT_BINARY_PROP));
316                             Map<String, Object> contentIdMap = (Map<String, Object>)
317                                     (attachmentMap.get(Messages.ATTACHMENT_CONTENTID_PROP));

319                             String contentId = (String) contentIdMap.get(VALUE);

323                             DataHandler dataHandler = new DataHandler(

331                     return execCall(clientBuilderImpl, dispatch, requestMessage,

333                 } catch (IOException e) {
334                     throw new ExecutionException(e, IO_ERROR).withReason(IO_ERROR_REASON)
335                             .withResolutionAsDefect();

338                 throwBindingError(dispatch);
339             }

374     @Interceptable
375     protected SOAPMessage execCall(ClientBuilderImpl clientBuilderImpl,
376             Dispatch<SOAPMessage> dispatch, SOAPMessage requestMessage,
377             boolean outputHttpResponseHeaders) {
378         /*
379         * Apache CXF library supports one-way and two-way SOAP calls.
380         * One-way call is used to invoke those operations that do not return any response.
381         *  The respective method is invokeOneWay of {@link org.apache.cxf.jaxws.DispatchImpl}.
382         * And two-way call is used to invoke those operations that return response.
383         *  The respective method is invoke of {@link org.apache.cxf.jaxws.DispatchImpl}.
384         *
385         * However, CXF library has a drawback when it comes to one-way calls.  In case of failures,
386         *  it simply ignores the SOAP error message returned by server and throws an instance of
387         *  {@link org.apache.cxf.transport.http.HTTPException} with the error message
388         *  '500: Internal Server Error' when communicating with <endpoint URL>.
389         * This leads to confusion as to what could be the original cause of the error.
390         *
391         * Its best to always invoke two-way call (works for both one-way and two-way operations).
392         * It captures the SOAP error message in case of failures and provides the FAULT_STRING
393         *  while displaying the error message citing the exact reason for the failure.
394         * And, in case of successful operation invocation, it simply returns NULL for one-way
395         *  operations and the SOAP response message for two-way operations.
396         */
397         HTTPConduit conduit = (HTTPConduit) ((DispatchImpl<SOAPMessage>)dispatch).getClient()
398                 .getConduit();
399         RequestClientUtil.setProxyConfig(conduit);
400         Object soapMessage = dispatch.invoke(requestMessage);
401         SOAPMessage responseSOAPMessage = (SOAPMessage) soapMessage;
402 
403         if (outputHttpResponseHeaders) {
404             responseSOAPMessage = includeHttpResponseHeadersInSOAPMessage(dispatch,
405                     responseSOAPMessage);
406         }
407 
408         return responseSOAPMessage;
409     }

411     protected SOAPMessage includeHttpResponseHeadersInSOAPMessage(Dispatch<SOAPMessage> dispatch,
412             SOAPMessage responseSOAPMessage) {
413         Map<String, Object> responseContext = dispatch.getResponseContext();
414         if (responseContext != null) {
415             // get the HTTP headers returned from the response
416             Object responseHeaders = responseContext.get(Message.PROTOCOL_HEADERS);
417             if (responseHeaders instanceof Map) {
418                 TreeMap<String, ArrayList<String>> httpResponseHeaders =
419                         (TreeMap<String, ArrayList<String>>) responseHeaders;
420 
421                 MimeHeaders responseSOAPMessageMimeHeaders = responseSOAPMessage.getMimeHeaders();
422                 MimeHeaders mergedMimeHeaders = new MimeHeaders();
423 
424                 /*
425                 For each HTTP header in the response, check if the SOAPMessage already has such a
426                 HTTP header, and, if not, add it to the SOAPMessage's HTTP headers
427                  */
428                 for (Map.Entry<String, ArrayList<String>> httpResponseHeader :
429                         httpResponseHeaders.entrySet()) {
430                     String[] valuesForHeaderInSOAPMessage = responseSOAPMessageMimeHeaders
431                             .getHeader(httpResponseHeader.getKey());
432                     boolean httpHeaderIsAlreadyPresentInSOAPMessage =
433                             (valuesForHeaderInSOAPMessage != null &&
434                                     valuesForHeaderInSOAPMessage.length > 0);
435 
436                     /*
437                     If the SOAPMessage contains this HTTP header already, don't add the
438                     HTTP header from the response context as the interceptor chain
439                     may have been responsible for adding/removing/modifying this
440                      */
441                     if (httpHeaderIsAlreadyPresentInSOAPMessage) {
442                         for (String soapMessageHeaderValue : valuesForHeaderInSOAPMessage) {
443                             mergedMimeHeaders.addHeader(httpResponseHeader.getKey(),
444                                     soapMessageHeaderValue);
445                         }
446                     } else {
447                         for (String httpResponseHeaderValue : httpResponseHeader.getValue()) {
448                             mergedMimeHeaders.addHeader(httpResponseHeader.getKey(),
449                                     httpResponseHeaderValue);
450                         }
451                     }
452                 }
453 
454                 // clear out the existing HTTP headers from the SOAPMessage
455                 responseSOAPMessage.getMimeHeaders().removeAllHeaders();
456 
457                 // add each mergedMimeHeader
458                 Iterator allHeaders = mergedMimeHeaders.getAllHeaders();
459                 while (allHeaders.hasNext()) {
460                     MimeHeader header = (MimeHeader) allHeaders.next();
461                     responseSOAPMessage.getMimeHeaders().addHeader(header.getName(), header
462                             .getValue());
463                 }
464             }
465         }
466 
467         return responseSOAPMessage;
468     }

502     private void rewriteBodyTextAsDocument(SOAPMessage soapMessage, SOAPBody soapBody)
503             throws SOAPException, ParserConfigurationException {
504         DocumentBuilder documentBuilder = documentBuilderFactory.newDocumentBuilder();
505         org.w3c.dom.Document document = null;
506         try {
507             document = documentBuilder.parse(new ByteArrayInputStream(
508                     soapBody.getTextContent().getBytes()));
509         } catch (SAXException | IOException | DOMException e) {
510             // SOAP Body cannot be converted into a valid xml document. Ignore this and treat the
511             // body as plain text.
512             log.warn(SOAP_BODY_IS_NOT_SERIALIZABLE, soapBody.getTextContent());
513         }
514         soapBody.removeContents();
515         soapBody.addDocument(document);
516         soapMessage.saveChanges();
517     }
