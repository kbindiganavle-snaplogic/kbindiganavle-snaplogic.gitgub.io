20 /**

23  * @author mklumpp

25 public class XSDSchemaInfos implements Iterable<XSDSchemaInfo> {
26     private List<XSDSchemaInfo> infos = new ArrayList<>();
27 
28     /**
29      * Returns the infos which represent leafs in the template (aka variables)
30      *
31      * @return the list of leaf infos
32      */
33     public List<XSDSchemaInfo> getKeys() {
34         List<XSDSchemaInfo> newInfos = new ArrayList<>();
35         for (XSDSchemaInfo info : infos) {
36             if (info.getSnapType() != SnapType.COMPOSITE && info.getSnapType() != SnapType.TABLE) {
37                 XSDSchemaInfo newInfo = new XSDSchemaInfo(info);
38                 newInfos.add(newInfo);
39             }
40         }
41         return newInfos;
42     }
43 
44     /**

46      *
47      * @return the list of info elements
48      */
49     public List<XSDSchemaInfo> getAll() {

51     }

60     }
