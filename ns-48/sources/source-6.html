155     protected Object includeHttpResponseHeaders(SOAPMessage soapResponse, Object data) {
156         Preconditions.checkNotNull(soapResponse);
157         MimeHeaders mimeHeaders = soapResponse.getMimeHeaders();
158         if (mimeHeaders != null) {
159             Map<String, String> headers = new TreeMap<>();
160             Iterator allHeaders = mimeHeaders.getAllHeaders();
161             while (allHeaders.hasNext()) {
162                 MimeHeader header = (MimeHeader) allHeaders.next();
163                 String existingHeaderValue = headers.get(header.getName());
164                 if (existingHeaderValue != null) {
165                     headers.put(header.getName(), existingHeaderValue + "," + header.getValue());
166                 } else {
167                     headers.put(header.getName(), header.getValue());
168                 }
169             }
170             if (data == null) {
171                 data = new LinkedHashMap<String, Object>();
172             }
173             if (data instanceof Map) {
174                 ((Map<String, Object>) data).put(HTTP_RESPONSE_HEADERS, headers);
175             }
176         }
177         return data;
178     }

191     public void processSoapResponse(SOAPMessage soapResponse, Document document, String envelope,
192                                     boolean outputHttpResponseHeaders)
193             throws SnapDataException {
194         DOMSource domSource;
195         try {
196             domSource = (outputEnvelope) ?
197                     new DOMSource(soapResponse.getSOAPPart()) :
198                     new DOMSource(soapResponse.getSOAPBody());
199         } catch (SOAPException e) {
200             throw new SnapDataException(Messages.PROCESS_SOAP_RESPONSE_ERROR)
201                     .withReason(String.format(Messages.SOAP_EXCEPTION_RECEIVED_S), e.getMessage())
202                     .withResolution(Messages.PLEASE_REVIEW_MESSAGE_AND_ADDRESS);
203         }
204         if (outputEnvelope) {
205             SOAPHeader soapHeader;
206             try {
207                 soapHeader = soapResponse.getSOAPHeader();
208 
209                 Iterator<javax.xml.soap.Node> nodeIterator = soapHeader.getChildElements();
210                 if (nodeIterator != null) {
211 
212                     while (nodeIterator.hasNext()) {
213                         Node childNode = nodeIterator.next();
214                         if (childNode.getNodeType() != Node.ELEMENT_NODE) {
215                             continue;
216                         }
217                         SOAPElement node = (SOAPElement) childNode;
218                         Iterator<javax.xml.soap.Node> innerNodeIterator =
219                                 node.getChildElements();
220                         if (innerNodeIterator != null) {
221                             while (innerNodeIterator.hasNext()) {
222                                 Node innerChild = innerNodeIterator.next();
223                                 if (innerChild.getNodeType() != Node.ELEMENT_NODE) {
224                                     continue;
225                                 }
226                                 SOAPElement innerElement = (SOAPElement) innerChild;
227                                 if (innerElement == null || !StringUtils.equals(
228                                         innerElement.getLocalName(), USERNAME_TOKEN) ||
229                                         innerElement.getChildNodes() == null) {
230                                     continue;
231                                 }
232                                 NodeList innerList = innerElement.getChildNodes();
233                                 if (innerList != null) {
234                                     for (int j = 0; j < innerList.getLength(); j++) {
235                                         Node innerNode = innerList.item(j);
236                                         if (innerNode != null && StringUtils.equals(
237                                                 innerNode.getLocalName(), PASSWORD)) {
238                                             innerNode.setTextContent(MASKED_PASSWORD);
239                                             break;
240                                         }
241                                     }
242                                 }
243                             }
244                         }
245                     }
246                 }
247             } catch (SOAPException e) {
248                 LOGGER.warn("SOAP get SOAP header failed for " + envelope);
249             }
250         }
251 
252         try {
253             data = xmlUtils.convertToJson(xmlInputFactory, jsonOutputFactory,
254                     domSource);
255         } catch (XMLStreamException e) {
256             throw new SnapDataException(Messages.PROCESS_SOAP_RESPONSE_ERROR)
257                     .withReason(String.format(CONVERSION_FROM_XML_TO_JSON_FAILED_S), e.getMessage())
258                     .withResolution(Messages.PLEASE_REVIEW_MESSAGE_AND_ADDRESS);
259         }
260 
261         if (outputHttpResponseHeaders) {
262             data = includeHttpResponseHeaders(soapResponse, data);
263         }
264 
265         boolean callSucceededCondition = true;
266         if (data != null) {
267             Document newDocument = documentUtility.newDocument(data);
268             // The SOAP call may fail even if no SOAP Fault is thrown.
269             // If the success criterion is set, check that it has occurred.
270             if (!responseSuccessExpression.isEmpty()) {
271                 Object successConditionEval =
272                         responseSuccessExpression.eval(newDocument, data);
273                 if (successConditionEval == null) {
274                     throw new SnapDataException(Messages.SOAP_NULL_SEARCH_RESULT)
275                             .withReason(Messages.SERVER_DID_NOT_RETURN_A_RESPONSE)
276                             .withResolution(Messages.PLEASE_VERIFY_SERVER_AND_RETRY);
277                 } else {
278                     callSucceededCondition =
279                             soapUtils.toBoolean(successConditionEval);
280 
281                     if (!callSucceededCondition) {
282                         LOGGER.warn("SOAP invocation failed for " + envelope);
283                         throw new SnapDataException(Messages.SOAP_NULL_RESPONSE)
284                                 .withReason(Messages.SERVER_DID_NOT_RETURN_A_RESPONSE)
285                                 .withResolution(Messages.PLEASE_VERIFY_SERVER_AND_RETRY);
286                     }
287                 }
288             }
289             writeToOutputView(newDocument, document);
290         }
291 
292         Iterator attachmentsIterator = soapResponse.getAttachments();
293         while (attachmentsIterator.hasNext()) {
294             Map<String, Object> outputMap = new LinkedHashMap<>();
295             AttachmentPart attachment =
296                     (AttachmentPart) attachmentsIterator.next();
297             Iterator<MimeHeader> mimeHeadersIterator =
298                     attachment.getAllMimeHeaders();
299             while (mimeHeadersIterator.hasNext()) {
300                 MimeHeader mimeHeader = mimeHeadersIterator.next();
301                 outputMap.put(mimeHeader.getName(), mimeHeader.getValue());
302             }
303             try {
304                 outputMap.put(CONTENT,
305                         IOUtils.toString(attachment.getBase64Content()));
306             } catch (IOException | SOAPException e) {
307                 throw new SnapDataException(Messages.SOAP_PROCESS_RESPONSE_ERROR)
308                         .withReason(String.format(Messages.EXCEPTION_GETTING_ATTACHMENT_S,
309                                 e.getMessage()))
310                         .withResolution(Messages.PLEASE_VERIFY_SERVER_AND_RETRY);
311             }
312             writeToOutputView(documentUtility.newDocument(outputMap), document);
313         }
314     }

320     protected String formatReason(Map<String, Object> errorMap) {
321         StringBuilder builder = new StringBuilder();
322         for (Map.Entry<String, Object> entry : errorMap.entrySet()) {
323             String value = (String) entry.getValue();
324             if (StringUtils.isNotBlank(value)) {
325                 String key = entry.getKey();
326                 builder.append(key);
327                 if (!StringUtils.endsWith(key, COLON_SPACE)) {
328                     builder.append(COLON_SPACE);
329                 }
330                 builder.append(value).append(COMMA_SPACE);
331             }
332         }
333         String reason = builder.toString();
334         return StringUtils.isBlank(reason) ? reason :
335                 reason.substring(0, reason.length() - COMMA_SPACE.length());
336     }

338     public void processNullSoapResponse(Document document, String nullResponseType) {
339         LOGGER.debug("null response type is " + nullResponseType);
340         switch (nullResponseType) {
341             case Messages.NULL_RESPONSE_LEGACY_LABEL:
342                 break;
343             case Messages.NULL_RESPONSE_EMPTY_LABEL:
344                 // This is not an error. No useful information to pass along,
345                 // so just return an empty document so that this path does not stop
346                 // the pipeline execution.
347                 Document newDocument = documentUtility.newDocument(
348                         new HashMap<String, String>());
349                 outputViews.write(newDocument, document);
350                 break;
351             case Messages.NULL_RESPONSE_ERROR_LABEL:
352                 throw new SnapDataException(Messages.SOAP_NULL_RESPONSE)
353                         .withReason(Messages.SERVER_DID_NOT_RETURN_A_RESPONSE)
354                         .withResolution(Messages.PLEASE_VERIFY_SERVER_AND_RETRY);
355         }

382     public void processException(Document document, String envelope,
383                                 List<Header> headerList,
384                                  ClientBuilder clientBuilder,
385                                  List<Interceptor<? extends Message>> outInterceptors,
386                                  RetryException exception) {
387         Throwable cause = exception.getCause();
388         if (cause instanceof SOAPFaultException) {
389             SOAPFault soapFault = ((SOAPFaultException) cause).getFault();
390             Map<String, Object> errorMap = Maps.newHashMap();
391             errorMap.put(Messages.SOAP_FAULT_EXCEPTION, exception.getMessage());
392             errorMap.put(Messages.SOAP_FAULT_CODE, soapFault.getFaultCode());
393             errorMap.put(Messages.SOAP_FAULT_STRING, soapFault.getFaultString());
394             errorMap.put(Messages.SOAP_FAULT_ACTOR, soapFault.getFaultActor());
395             Detail newDetail = soapFault.getDetail();
396             if (newDetail != null) {
397                 errorMap.put(Messages.SOAP_FAULT_DETAIL_ENTRY, newDetail.getTextContent());
398             }
399             Throwable rootCause = Throwables.getRootCause(exception);
400             SnapDataException ex;
401             if (standardizeErrorOutput) {
402                 ex = new SnapDataException(exception, Messages.EXCEPTION_OCCURRED)
403                         .withReason(formatReason(errorMap))
404                         .withResolution(Messages.VERIFY_THE_INPUT_DATA_IS_CORRECT);
405             } else {
406                 String reason = rootCause.getMessage();
407                 errorMap.put(KEY_ERROR, reason);
408                 errorMap.put(KEY_REASON, reason);
409                 ex = new SnapDataException(documentUtility.newDocument(errorMap), exception,
410                         Messages.EXCEPTION_OCCURRED)
411                         .withReason(exception.getMessage())
412                         .withResolution(Messages.VERIFY_THE_INPUT_DATA_IS_CORRECT);
413             }
414             writeToErrorView(ex, document);
415 
416         } else if (cause instanceof XMLStreamException) {
417             throw new ExecutionException(exception, Messages.XML_SERIALIZATION_FAILED)
418                     .withReason(exception.getMessage())
419                     .withResolutionAsDefect();
420 
421         } else {
422             Throwable rootCause = Throwables.getRootCause(exception);
423             String reason = rootCause.getMessage();
424             if (standardizeErrorOutput) {
425                 reason = cause == null ? reason : cause.getMessage();
426                 SnapDataException ex = new SnapDataException(exception, Messages.EXCEPTION_OCCURRED)
427                         .withReason(reason)
428                         .withResolution(Messages.SOAP_EXCEPTION_RESOLUTION);
429                 writeToErrorView(ex, document);
430                 return;
431             }
432             Map<String, Object> errorMap = Maps.newHashMap();
433             errorMap.put(KEY_ERROR, reason);
434             errorMap.put(STACKTRACE, Throwables.getStackTraceAsString(exception.getCause() != null ?
435                     exception.getCause() : exception));
436             SnapDataException dataException = new SnapDataException(documentUtility
437                     .newDocument(errorMap), exception, Messages.EXCEPTION_OCCURRED)
438                     .withReason(reason)
439                     .withResolution(Messages.SOAP_EXCEPTION_RESOLUTION);
440 
441             Map<String, Object> soapHeaders = Maps.newHashMap();
442             Dispatch<SOAPMessage> dispatchClient = clientBuilder.getDispatchClient();
443             Map<?, Object> requestContext = dispatchClient.getRequestContext();
444             if (requestContext != null) {
445                 soapUtils.serializeHeader(requestContext, soapHeaders);
446             }
447 
448             if (outInterceptors != null) {
449                 for (Interceptor<? extends Message> interceptor : outInterceptors) {
450                     if (interceptor instanceof SoapMessageSenderInterceptor) {
451                         SoapMessageSenderInterceptor soapMessageSenderInterceptor =
452                                 (SoapMessageSenderInterceptor) interceptor;
453                         Message message = soapMessageSenderInterceptor.getMessage();
454                         if (message != null) {
455                             // protocol headers are the ones passed into the http/s request.
456                             soapUtils.serializeHeader((Map<?, Object>)
457                                     message.get(Message.PROTOCOL_HEADERS), soapHeaders);
458                         }
459                     }
460                 }
461             }
462 
463             if (soapHeaders.containsKey(KEY_AUTHORIZATION)) {
464                 soapHeaders.put(KEY_AUTHORIZATION, MASKED_AUTH_HEADER);
465             }
466             errorMap.put(REQUEST_HEADERS, soapHeaders);
467             // serialize the envelope
468             if (envelope != null) {
469                 errorMap.put(KEY_ENVELOPE, parseXML2JSON(envelope, xmlUtils));
470             }
471             writeToErrorView(dataException, document);
472         }
473     }
