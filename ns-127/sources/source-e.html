251     public void buildDriveSuggestionsBasedOnType(SuggestionBuilder suggestionBuilder,
252                                        PropertyValues propertyValues, String nodeName) {
253         String driveType = "";
254         String group = "";
255         String user = "";
256         try {
257             driveType = propertyValues.get(DRIVE_TYPE);
258             if (ONEDRIVE.equalsIgnoreCase(driveType)) {
259                 user = propertyValues.get(USER_ID);
260             } else if (DOCUMENT_LIBRARY.equalsIgnoreCase(driveType)) {
261                 group = propertyValues.get(SHARED_LIBRARY);
262             } else if (ALL.equalsIgnoreCase(driveType)) {
263                 user = propertyValues.get(USER_ID);
264             }
265         } catch (SnapDataException ex) {
266             throw new ConfigurationException(FETCH_DRIVES_ERROR)
267                     .withReason(ex.getMessage())
268                     .withResolution(DRIVE_UPSTREAM_SUGGESTION_ERROR);
269         }
270         driveIdNameMapping = new LinkedHashMap<>();
271         switch (driveType) {
272             case ALL:
273                 driveIdNameMapping = buildDriveIDMap(user);
274                 break;
275             case DRIVE:
276                 driveIdNameMapping.put(DRIVE_ID_REQUIRED, DRIVE_ID_REQUIRED);
277                 break;
278             default:
279                 driveIdNameMapping = getDriveIdNameMap(driveType, group, user);
280         }
281         if (driveIdNameMapping != null && !driveIdNameMapping.isEmpty()) {
282             suggestionBuilder.node(nodeName)
283                     .suggestions(driveIdNameMapping.keySet().toArray(new String[0]));
284         } else {
285             suggestionBuilder.node(nodeName)
286                     .suggestions(NO_SUGGESTIONS_AVAILABLE);
287         }

395         String driveID = null;
396         if (DRIVE.equalsIgnoreCase(type)) {
397             if (StringUtils.isBlank(driveName)) {
398                 throw new SnapDataException(FETCH_DRIVES_ERROR)
399                         .withReason(DRIVE_ID_NULL)
400                         .withResolution(DRIVE_ID_NULL_RESOLUTION);
401             }
402             driveID =  driveName;
403         } else if (ALL.equalsIgnoreCase(type)) {
404             String userName = userIdExpr == null ? null : userIdExpr.eval(document);
405             driveID = oneDriveCommonUtil.evaluateFullDriveName(graphClient, grantType,
406                     driveName, userName);
407         } else if (DOCUMENT_LIBRARY.equalsIgnoreCase(type)) {
408             String group = groupIdExpr == null ? null : groupIdExpr.eval(document);
409             driveID =  getDriveIdNameMap(type, group, "").get(driveName);
410         } else {
411             String userName = userIdExpr == null ? null : userIdExpr.eval(document);
412             driveID = getDriveIdNameMap(type, "", userName).get(driveName);
413         }
414         if (StringUtils.isBlank(driveID)) {
415             throw new SnapDataException(INVALID_SOURCE_DRIVE)
416                     .withReason(INVALID_SOURCE_DRIVE)
417                     .withResolution(INVALID_DRIVE_RESOLUTION);
418         }

543     public RetryPolicy createRetryPolicy(Document document) {
544         BigInteger retryVal = expNumRetry.eval(document);
545         if (Objects.nonNull(retryVal) && retryVal.intValue() >= MIN_RETRY) {
546             numRetry = retryVal.intValue();
547         }
548         BigInteger retryIntVal = expRetryInt.eval(document);
549         if (Objects.nonNull(retryIntVal) && retryIntVal.intValue() >= MIN_RETRY_INT) {
550             retryInt = retryIntVal.intValue();
551         }
552         return new RetryPolicy()
553                 .withMaxRetries(numRetry)
554                 .withDelay(retryInt, TimeUnit.SECONDS)
555                 .retryOn(Exception.class)
556                 .abortOn(SuggestViewAbortException.class);
