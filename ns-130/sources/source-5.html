359         if (channel == null || !channel.isOpen()) {
360             connectAndCreateChannel(header);
361         }
362         if (connection != null && !connection.isOpen()) {
363             if (isConnErrToErrorViewEnabled) {
364                 SnapDataException snapDataException =
365                         new SnapDataException(ERROR_WHILE_PREPARING_DESTINATION)
366                                 .withReason(ERR_RECONNECT_CHANNEL)
367                                 .withResolution(RES_VERIFY_INPUT_DATA);
368                 errorViews.write(snapDataException, header);
369             } else {
370                 throw new ExecutionException(ERROR_WHILE_PREPARING_DESTINATION)
371                         .withReason(ERR_RECONNECT_CHANNEL)
372                         .withResolution(RES_VERIFY_INPUT_DATA);
373             }
374             return false;
375         }
376         queueName = checkEmpty(getValue(queueNameExp, header), StringUtils.EMPTY);
377         exchangeName = checkEmpty(getValue(exchangeNameExp, header),
378                 String.format(DEFAULT_PREFIX_EXCHANGE_TYPE, exchangeType));
379         routingKey = checkEmpty(getValue(routingKeyExp, header), queueName);
380         if (StringUtils.isBlank(queueName) && !dynamicQueueCreated) {
381             dynamicQueueCreated = true;
382             channel.queueDeclare();
383         } else if (StringUtils.isNotBlank(queueName)) {
384             Map argumentMap =  MessageUtils.populateArgumentProperties(argumentProperties, header);
385             channel.queueDeclare(queueName, true, false, autoDelete, argumentMap);
386         }

388         if (headerMessageProperties != null) {
389             headers = MessageUtils.populateHeaderProperties(headerMessageProperties,
390                     header, xmatch);
391         }
392         channel.queueBind(queueName, exchangeName, routingKey, headers);

396     private void connectAndCreateChannel(Document header) {
397         int attemptCount = 0;
398         while (attemptCount < maxConnectionAttempts) {
399             try {
400                 attemptCount++;
401                 if (connection == null || !connection.isOpen()) {
402                     connection = account.connect();
403                 }
404                 if (channel == null || !channel.isOpen()) {
405                     channel = connection.createChannel();
406                     if (StringUtils.equals(processingMode, ASYNCHRONOUS)) {
407                         if (ackType == AcknowledgmentType.PIPELINE_CONTROL) {
408                             channel.basicQos(minPrefetchCount);
409                         } else {
410                             channel.basicQos(maxPrefetchCount);
411                         }
412                     }
413                 }
414                 LOGGER.debug(RETRY_SUCC_LOG_MSG, attemptCount);
415                 return;
416             } catch (Exception e) {
417                 LOGGER.error(RETRY_FAILED_LOG_MSG, attemptCount, e.getMessage());
418                 if (attemptCount >= maxConnectionAttempts) {
419                     acknowledgeHeader(header);
420                     Throwable cause = e.getCause() != null ? e.getCause() : e;
421                     throw new ExecutionException(cause, ERROR_CREATING_CONNECTION)
422                             .withReason(cause.getMessage())
423                             .withResolution(ERROR_CREATING_CONNECTION_RES);
424                 }
425                 try {
426                     Thread.sleep(retryInterval);
427                 } catch (InterruptedException ie) {
428                     //no op
429                 }
430             }
431         }
432     }

483     protected void process(final Document document, final String s) {
484         try {
485             if (prepareDestination(document)) {
486                 acknowledgeHeader(document);
487                 switch (processingMode) {
488                     case SYNCHRONOUS:
489                         consumeSynchronously(document);
490                         break;
491                     case ASYNCHRONOUS:
492                         consumeAsynchronously(document);
493                 }
494             }
495         } catch (Exception e) {
496             if (!isConnErrToErrorViewEnabled) {
497                 if (e instanceof SnapDataException) {
498                     throw (SnapDataException) e;
499                 } else {
500                     throw MessageUtils.handleQueueTypeException(e, queueName, false,
501                             ERROR_WHILE_PREPARING_DESTINATION, RES_VERIFY_INPUT_DATA);
502                 }
503             } else {
504                 if (e instanceof SnapDataException) {
505                     errorViews.write((SnapDataException) e, document);
506                 }
507                 else {
508                     SnapDataException snapDataException;
509                     snapDataException = (SnapDataException) MessageUtils.handleQueueTypeException(e, queueName,
510                             true, ERROR_WHILE_PREPARING_DESTINATION, RES_VERIFY_INPUT_DATA);
511                     errorViews.write(snapDataException, document);
512                 }
513             }
514         }
515     }

524      * Consumes messages synchronously from RabbitMQ
525      */
526     private void consumeSynchronously(Document document) {
527         try {
528             while (!stopFlag && !Thread.interrupted()) {
529                 if (channel == null || !channel.isOpen()) {
530                     connectAndCreateChannel(document);
531                 }
532                 if (ackType == AcknowledgmentType.PIPELINE_CONTROL) {
533                     GetResponse response = channel.basicGet(queueName, false);
534                     if (response != null) {
535                         synchronized (channel) {
536                             processSyncMessage(response, document);
537                             try {
538                                 channel.wait(ACK_TIMEOUT);
539                             } catch (InterruptedException e) {
540                                 LOGGER.error(ACKNOWLEDGE_LISTENER_THREAD_INTERRUPTED, e);
541                             }
542                         }
543                     } else {
544                         if (maxMessageCount == 0) {
545                             setStopFlag();
546                         } else {
547                             //wait when there are no messages in the queue before
548                             //making next request
549                             Thread.sleep(SLEEP_TIME_INTERVAL_IN_MS);
550                         }
551                     }
552                 } else {
553                     GetResponse response = channel.basicGet(queueName, false);
554                     if (response != null) {
555                         processSyncMessage(response, document);
556                     } else {
557                         if (maxMessageCount == 0) {
558                             setStopFlag();
559                         } else {
560                             //wait when there are no messages in the queue before
561                             //making next request
562                             Thread.sleep(SLEEP_TIME_INTERVAL_IN_MS);
563                         }
564                     }
565                 }
566             }
567         } catch (IOException | AlreadyClosedException e) {
568             handleException(e, document);
569         } catch (InterruptedException e) {
570             LOGGER.error(WAITING_FOR_SYNC_MESSAGE_INTERRUPTED, e);
571         }
572     }

578         try {
579             if (channel == null || !channel.isOpen()) {
580                 connectAndCreateChannel(document);
581             }
582             Consumer consumer = new DefaultConsumer(channel) {
583                 @Override
584                 public void handleDelivery(String consumerTag, Envelope envelope,
585                                            AMQP.BasicProperties properties, byte[] body) {
586                     if (maxMessageCount == 0 && !stopListening) {
587                         stopFlag = false;
588                     }
589                     if (!stopFlag) {

593                                     processMessage(body, properties, document, envelope);
594                                     channel.wait(ACK_TIMEOUT);
595                                 }
596                             } catch (InterruptedException ie) {
597                                 LOGGER.error(ACKNOWLEDGE_LISTENER_THREAD_INTERRUPTED, ie);
598                             }
599                         } else {
600                             processMessage(body, properties, document, envelope);
601                         }
602                     }
603                 }
604             };
605             channel.basicConsume(queueName, false, consumer);
606             while (!stopFlag) {
607                 if (maxMessageCount == 0) {
608                     stopFlag = true;
609                 }
610                 Thread.sleep(SLEEP_TIME_INTERVAL_IN_MS);
611             }
612         } catch (IOException | AlreadyClosedException e) {
613             handleException(e, document);
614         } catch (InterruptedException e) {
615             LOGGER.error(CONSUMER_MSG_LISTENER_THREAD_INTERRUPTED, e);

642      * Writes the message read from to output view
643      *
644      * @param  byte message body
645      * @param  message properties
646      */
647     private void processMessage(final byte[] body, final AMQP.BasicProperties props, Document
648             header, Envelope envelope) {
649         outputViews.write(new BinaryOutput() {
650             @Override
651             public Document getHeader() {
652                 try {
653                     if (props != null) {
654                         return documentUtility.newDocumentFor(header,
655                                 MessageUtils.getHeaderMap(props, envelope));
656                     } else {
657                         return null;
658                     }
659                 } catch (SnapException e) {
660                     errorViews.write(MessageUtils.getSnapDataException(e, header, documentUtility));
661                     return null;
662                 }
663             }
664 
665             @Override
666             public void write(WritableByteChannel writeChannel)
667                     throws IOException {
668                 try {
669                     IOUtils.write(body, Channels.newOutputStream(writeChannel));
670                 } catch (SnapException e) {
671                     errorViews.write(MessageUtils.getSnapDataException(e, header, documentUtility));
672                 }
673             }
674         });
675         messageCount.incrementAndGet();
676         if (ackType == AcknowledgmentType.AUTO_ACKNOWLEDGE) {
677             try {
678                 switch (messageAction) {
679                     case RECOVER_MESSAGE_PROP:
680                         channel.basicReject(envelope.getDeliveryTag(), true);
681                         break;
682                     case REJECT_MESSAGE_PROP:
683                         channel.basicReject(envelope.getDeliveryTag(), false);
684                         break;
685                     case ACK_MESSAGE_PROP:
686                         channel.basicAck(envelope.getDeliveryTag(), false);
687                         break;
688                 }
689             } catch (IOException | AlreadyClosedException e) {
690                 handleException(e, header);
691             }
692         }
693         if ((maxMessageCount > 0) && (messageCount.get() >= maxMessageCount) &&
694                 (ackType != AcknowledgmentType.PIPELINE_CONTROL)) {
695             setStopFlag();
696         }
697     }

701         // Hence notification should be made on the session object.
702         if (channel != null) {

710         }

724     @Override
725     public void cleanup() throws ExecutionException {
726         stopFlag = true;
727         stopListening = true;
728         if (acknowledgeListener != null) {
729             synchronized (ackThreadWaitObj) {
730                 ackThreadWaitObj.notifyAll();
731             }
732             //When JMS Consumer is run in PIPELINE_CONTROL mode wait is called on
733             //session object in consumeSynchronously method. It is better to
734             //notify it before stopping the consumer.
735             if (channel != null) {
736                 synchronized (channel) {
737                     channel.notifyAll();
738                 }
739             }
740             if (!RUUID_TO_ACK_LISTENER.isEmpty()) {
741                 RUUID_TO_ACK_LISTENER.remove(ackThreadName);
742             }
743             acknowledgeListener = null;
744         }
745         /**
746          * Use of the isOpen() method of channel and connection objects is not recommended,
747          * because the value returned by the method is dependent on the existence
748          * of the shutdown cause
749          * */
750         if (channel != null) {
751             try {
752                 channel.close(320, FORCE_CLOSE);
753             } catch (IOException | AlreadyClosedException | TimeoutException e) {
754                 try {
755                     channel.abort(320, FORCE_CLOSE);
756                 } catch (IOException | AlreadyClosedException ex) {
757                     throw new ExecutionException(e, ERROR_WHILE_CLOSING_CHANNEL)
758                             .withReason(e.getMessage())
759                             .withResolution(RES_CHECK_NETWORK_CONNECTION);
760                 }
761             } finally {
762                 if (account != null) {
763                     account.disconnect();
764                 }
765             }
766             channel = null;
767         }
768     }

810      * Observer thread

812     class MessageAckListener extends Thread implements Observer {
813 
814         public MessageAckListener(final ThreadGroup group, final String name) {
815             super(group, name);
816         }
817 
818         @Override
819         public void run() {
820             while (!stopFlag) {
821                 synchronized (ackThreadWaitObj) {
822                     try {
823                         ackThreadWaitObj.wait(ACK_TIMEOUT);
824                     } catch (InterruptedException e) {
825                         LOGGER.error(ACKNOWLEDGE_LISTENER_THREAD_INTERRUPTED, e);
826                         break;
827                     }
828                 }
829             }
830         }
831 
832         @Override
833         public void update(Observable observable, Object arg) {
834             if (arg != null) {
835                 rabbitmqMessage = (RabbitmqMessage) arg;
836                 if (channel == null) {
837                     throw new ConfigurationException(ERROR_MULTIPLE_CONSUMERS);
838                 }
839                 synchronized (channel) {
840                     try {
841                         long deliveryTag = verifyMsgDeliveryTag(rabbitmqMessage.getDeliveryTag());
842                         channel.txSelect();
843                         switch (rabbitmqMessage.getMessageAction()) {
844                             case MessageConstants.RECOVER_MESSAGE_PREFIX:
845                                 channel.basicReject(deliveryTag, true);
846                                 break;
847                             case MessageConstants.REJECT_MESSAGE_PREFIX:
848                                 channel.basicReject(deliveryTag, false);
849                                 break;
850                             case MessageConstants.ACKNOWLEDGE_MESSAGE_PREFIX:
851                                 channel.basicAck(deliveryTag, false);
852                                 break;
853                         }
854                         channel.txCommit();
855                     } catch (IOException | AlreadyClosedException e) {
856                         SnapDataException se;
857                         if (e.getCause() instanceof ShutdownSignalException) {
858                             se = new SnapDataException(e.getCause(), e.getCause().getMessage());
859                             setStopFlag();
860                         } else {
861                             se = new SnapDataException(e, String.format(ERROR_ACKNOWLEDGING_MESSAGE,
862                                     rabbitmqMessage.getDeliveryTag()));
863                         }
864                         se.withResolution(RESOLUTION_MESSAGE_ACK_FAILURE);
865                         errorViews.write(se);
866                     }
867                     if ((maxMessageCount > 0) && (messageCount.get() >= maxMessageCount)) {
868                         setStopFlag();
869                         try {
870                             channel.close();
871                         } catch (IOException | TimeoutException | AlreadyClosedException e) {
872                             SnapDataException se = new SnapDataException(e, ERROR_CLOSING_CONSUMER)
873                                     .withReason(e.getMessage())
874                                     .withResolution(RES_CHECK_NETWORK_CONNECTION);
875                             errorViews.write(se);
876                         }
877                     }
878                     channel.notifyAll();
879                 }
880             }
881         }

884             if (!(msgDeliveryTag instanceof Number) || ((Number) msgDeliveryTag).longValue() < 0) {

887                         rabbitmqMessage.getDeliveryTag()));
