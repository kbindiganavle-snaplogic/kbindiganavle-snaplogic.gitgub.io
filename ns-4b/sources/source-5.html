251      * Starts templatizing a SOAP request
252      *
253      * @param namespaceURI the namespace URI of the XSD node
254      * @param localPart    the local part (or local name) of the XSD node
255      * @param nodeType     the type of the XSD node
256      */
257     private void startTemplatizing(String namespaceURI, String localPart,
258             TopLevelNodeType nodeType) {
259         LOGGER.debug(String.format ("Start Building the Template now with namespace: %s, " +
260                 "localpart: %s, nodeType: %s", namespaceURI, localPart, nodeType));
261         XSComponent xsComponent = types.getXSComponent(namespaceURI, localPart, nodeType);
262         if (xsComponent == null) {
263             throw new ExecutionException(NO_COMPONENT_FOUND_ERROR)
264                     .withReason(String.format(NO_COMPONENT_EXISTS, localPart, namespaceURI))
265                     .withResolution(String.format(PLEASE_CHECK_FOR_COMPONENT, localPart,
266                             namespaceURI));
267         }
268         switch (nodeType) {
269             case ELEMENT:
270                 XSElementDecl xsElementDecl = XSElementDecl.class.cast(xsComponent);
271                 elementDecl(xsElementDecl);
272                 break;
273             case SIMPLE_TYPE:
274                 XSSimpleType xsSimpleType = XSSimpleType.class.cast(xsComponent);
275                 simpleType(xsSimpleType);
276                 break;
277             case COMPLEX_TYPE:
278                 XSComplexType xsComplexType = XSComplexType.class.cast(xsComponent);
279                 complexType(xsComplexType);
280                 break;
281             case ATTRIBUTE_DECL:
282                 XSAttributeDecl xsAttributeDecl = XSAttributeDecl.class.cast(xsComponent);
283                 attributeDecl(xsAttributeDecl);
284                 break;
285             default:
286                 throw new IllegalArgumentException();
287         }
288     }

325     public void templatize(List<Input> inputs, SoapProtocol soapProtocol) {
326         String protocol;
327         switch (soapProtocol) {
328             case SOAP_11:
329                 protocol = SOAPConstants.SOAP_1_1_PROTOCOL;
330                 break;
331             case SOAP_12:
332                 protocol = SOAPConstants.SOAP_1_2_PROTOCOL;
333                 break;
334             default:
335                 throw new IllegalArgumentException();
336         }
337         if (translationContext.isDisposed()) {
338             //isDisposed() returns true when translationContext object state is cleared.
339             translationContext = new TranslationContext(types, new DefaultXSComplexTypeResolver(
340                     schemaVisitor.getResolvedTypesMapHolder()));
341         }
342         SOAPMessage soapMessage = SOAPMessageFactory.createMessage(protocol);
343         SOAPMessageHolder soapMessageHolder = new SOAPMessageHolder(soapMessage);
344         translationContext.setSoapMessageHolder(soapMessageHolder);
345         int inputSize = inputs.size();
346         for (int i = 0; i < inputSize; i++) {
347             if (i > 0) {
348                 translationContext.pushPath(JsonPathBuilder.ROOT_ELEMENT);
349             }
350             translationContext.setLocation(inputs.get(i).getLocation());
351             Input input = inputs.get(i);
352             if (input.isDocumentStyleInput()) {
353                 DocumentStyleInput documentStyleInput = input.asDocumentStyleInput();
354                 QName elementQName = documentStyleInput.elementQName();
355                 startTemplatizing(elementQName.getNamespaceURI(), elementQName.getLocalPart(),
356                         TopLevelNodeType.ELEMENT);
357             } else if (input.isRPCStyleInput()) {
358                 RPCStyleInput rpcStyleInput = input.asRPCStyleInput();
359                 templatizeRPCInput(rpcStyleInput, i, inputSize);
360             } else {
361                 throw new IllegalArgumentException();
362             }
363         }
364         namespacePrefixMap = translationContext.unmodifiableNamespaceMapper();
365     }

367     private void templatizeRPCInput(RPCStyleInput rpcStyleInput, int i, int inputSize) {
368         QName operationQName = rpcStyleInput.operationQName();
369         QName partQName = rpcStyleInput.partQName();
370         QName typeQName = rpcStyleInput.typeQName();
371         try {
372             SOAPElement operationElement;
373             if (i == 0) {
374                 operationElement = attachAndReturnRPCOperationElement(operationQName);
375                 translationContext.pushElement(operationElement);
376             } else {
377                 if (translationContext.peekElement() == null) {
378                     // Sample WSDL : https://rctus9.enablon.com/8.8/Tyson.RCT/rpc_literal.wsdl
379                     LOGGER.info("RPC Style input was not the first element (i != 0)");
380                     LOGGER.info("Using the Alternative fix for mixed input style of Document and " +
381                             "RPC");
382                     LOGGER.info("Operation: {} ; partName : {} ; typeName: {}", operationQName
383                             .getLocalPart(), partQName.getLocalPart(), typeQName.getLocalPart());
384                     operationElement = attachAndReturnRPCOperationElement(operationQName);
385                     translationContext.pushElement(operationElement);
386                 }
387                 operationElement = translationContext.peekElement();
388             }
389             String localPart = partQName.getLocalPart();
390             String normalizedLocalPart = StringUtil.normalizeLocalPart(localPart);
391             translationContext.pushPath(translationContext.peekPath()
392                     .concat(normalizedLocalPart)
393                     .concat(JsonPathBuilder.CURRENT_ELEMENT));
394             SOAPElement partElement = operationElement.addChildElement(localPart);
395             translationContext.pushElement(partElement);
396             TopLevelNodeType topLevelNodeType;
397             if (rpcStyleInput.isSimpleType()) {
398                 topLevelNodeType = TopLevelNodeType.SIMPLE_TYPE;
399             } else if (rpcStyleInput.isComplexType()) {
400                 topLevelNodeType = TopLevelNodeType.COMPLEX_TYPE;
401             } else {
402                 throw new IllegalArgumentException();
403             }
404             startTemplatizing(typeQName.getNamespaceURI(), typeQName.getLocalPart(),
405                     topLevelNodeType);
406             translationContext.popElement();
407             translationContext.popPath();
408             if (i == inputSize - 1) {
409                 translationContext.popElement();
410             }
411         } catch (SOAPException e) {
412             throw new ExecutionException(e, COULD_NOT_TRANSLATE_RPC_STYLE_INPUT)
413                     .withReason(Throwables.getRootCause(e).getMessage())
414                     .withResolutionAsDefect();
415         }
416     }

418     private SOAPElement attachAndReturnRPCOperationElement(QName operationQName)
419             throws SOAPException {
420         SOAPElement bodyElement = translationContext.getSoapMessage().getSOAPBody();
421         String tns = operationQName.getNamespaceURI();
422         String prefixForTns = translationContext.addNamespaceDeclIfAbsent(tns);
423         SOAPElement operationElement = bodyElement.addChildElement(operationQName.getLocalPart(),
424                 prefixForTns);
425         operationElement.addAttribute(new QName(SOAP_ENV_NAMESPACE,
426                 ENCODING_STYLE, SOAP_ENV_PREFIX), SOAP_ENCODING_URL);
427         return operationElement;
428     }

445     public void collectSchema(List<Input> inputs, boolean includeRootElement) {
446         if (inputs.size() == 1) {
447             Input input = inputs.get(0);
448             if (input.isDocumentStyleInput()) {
449                 DocumentStyleInput documentStyleInput = input.asDocumentStyleInput();
450                 QName elementQName = documentStyleInput.elementQName();
451                 String localPart = elementQName.getLocalPart();
452                 String namespaceURI = elementQName.getNamespaceURI();
453                 XSComponent xsComponent = types.getXSComponent(namespaceURI, localPart,
454                         TopLevelNodeType.ELEMENT);
455                 if (xsComponent == null) {
456                     throw new ExecutionException(NO_COMPONENT_FOUND_ERROR)
457                             .withReason(String.format(NO_COMPONENT_EXISTS, localPart, namespaceURI))
458                             .withResolution(String.format(PLEASE_CHECK_FOR_COMPONENT,
459                                     localPart, namespaceURI));
460                 }
461                 XSElementDecl xsElementDecl = XSElementDecl.class.cast(xsComponent);
462                 // NOTE: We don't want to show the type and the name of the top level element if
463                 // there is only one part in wsdl message tag
464                 // i.e. <wsdl:part name="..." element="tns:abc"/>
465                 // Here we want to start collecting schema from the type of "abc" element
466                 XSType xsType = xsElementDecl.getType();
467                 if (xsType.isComplexType()) {
468                     if (includeRootElement) {
469                         schemaVisitor.visit(xsElementDecl);
470                     } else {
471                         schemaVisitor.visit(xsType.asComplexType());
472                     }
473                 }
474             } else if (input.isRPCStyleInput()) {
475                 RPCStyleInput rpcStyleInput = input.asRPCStyleInput();
476                 collectSchemaFromRPCInput(rpcStyleInput);
477             } else {
478                 throw new IllegalArgumentException();
479             }
480         } else {
481             for (Input input : inputs) {
482                 if (input.isDocumentStyleInput()) {
483                     DocumentStyleInput documentStyleInput = input.asDocumentStyleInput();
484                     QName elementQName = documentStyleInput.elementQName();
485                     startCollectingSchema(elementQName.getNamespaceURI(),
486                             elementQName.getLocalPart(), TopLevelNodeType.ELEMENT);
487                 } else if (input.isRPCStyleInput()) {
488                     RPCStyleInput rpcStyleInput = input.asRPCStyleInput();
489                     collectSchemaFromRPCInput(rpcStyleInput);
490                 } else {
491                     throw new IllegalArgumentException();
492                 }
493             }
494         }
495     }

502     private void collectSchemaFromRPCInput(RPCStyleInput rpcStyleInput) {
503         SchemaCollectionContext schemaCollectionContext = schemaVisitor.getContext();
504         String currentPath = schemaCollectionContext.peekPath();
505         String localPart = rpcStyleInput.partQName().getLocalPart();
506         String normalizedLocalPart = StringUtil.normalizeLocalPart(localPart);
507         String newPath = currentPath
508                 .concat(normalizedLocalPart);
509         QName typeQName = rpcStyleInput.typeQName();
510         TopLevelNodeType topLevelNodeType;
511         if (rpcStyleInput.isComplexType()) {
512             topLevelNodeType = TopLevelNodeType.COMPLEX_TYPE;
513         } else if (rpcStyleInput.isSimpleType()) {
514             topLevelNodeType = TopLevelNodeType.SIMPLE_TYPE;
515         } else {
516             throw new IllegalArgumentException();
517         }
518         XSType xsType = XSType.class.cast(types.getXSComponent(typeQName.getNamespaceURI(),
519                 typeQName.getLocalPart(), topLevelNodeType));
520         schemaVisitor.getCollector().collect(newPath, xsType, false);
521         schemaCollectionContext.pushPath(newPath.concat(JsonPathBuilder.CURRENT_ELEMENT));
522         if (xsType != null && xsType.isComplexType()) {
523             schemaVisitor.visit(xsType.asComplexType());
524         }
525         schemaCollectionContext.popPath();
526     }
