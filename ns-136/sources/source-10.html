338                 DEFAULT_CONN_TIMEOUT_IN_SEC : connTimeoutValue.intValue();
339 
340         List<Map<String, String>> params = propertyValues.get(QUERY_PARAMS_PROP);
341         if (CollectionUtils.isNotEmpty(params)) {

343             for (Map<String, String> param : params) {
344                 String queryParam = JsonPath.read(param, QUERY_PARAM_PATH);
345                 ExpressionProperty queryParamValue = propertyValues.getExpressionPropertyFor(param,
346                         QUERY_PARAM_VALUE_PROP);
347                 queryParams.add(Pair.of(queryParam, queryParamValue));
348             }
349         }
350         // Gather the header expression properties so we can evaluate them for each input document.
351         List<Map<String, Object>> httpHeader = propertyValues.get(HTTP_HEADER_PROP);
352         if (httpHeader != null) {
353             for (Map<String, Object> item : httpHeader) {

358                 httpHeaders.add(Pair.of(keyProp, valueProp));

360         }

362         processArray = propertyValues.getBoolean(PROCESS_ARRAY_PROP, false);

366             throw new ConfigurationException(UNSUPPORTED_SNAP_CONFIGURATION)

439         Map<String, Object> documentData = documentUtility.getAsMap(document, errorViews);

444                 documentData, new BasicHeader(HttpHeaders.ACCEPT, MediaType.WILDCARD));
445 
446         if (account != null) {
447             Header authHeader = account.connect();
448             if (authHeader != null) {
449                 if (account instanceof RestAWSV4Account) {
450                     addAWSV4SignHeaders(document, headerList, authHeader);
451                 } else {
452                     headerList.add(authHeader);
453                 }
454             }
455         }
456         // create the final header array
457         return headerList.toArray(new Header[0]);
458     }

460     private TreeMap<String, String> buildParamsMap(
461             List<Pair<String, ExpressionProperty>> queryParams, Document document) {
462         TreeMap<String, String> qParamMap = new TreeMap<>();
463         for (Pair<String, ExpressionProperty> paramPair : queryParams) {
464             // left is the param key, right is the value
465             String paramKey = paramPair.getLeft();
466             if (paramKey == null) {
467                 continue;
468             }
469             ExpressionProperty paramValueProp = paramPair.getRight();
470             String paramValue = eval(document, null, paramValueProp);
471             qParamMap.put(paramKey, paramValue);
472         }
473         return qParamMap;
474     }

664     protected final URLResolver createURLResolver() {
665         final Map<String, Object> authProperties = createOAuth2Properties(account);
666 
667         if (urlResolver == null) {
668             urlResolver = new URLResolver() {
669                 @Override
670                 public String resolveUrl(final Document document) {
671                     if (authProperties != null) {
672                         Map<String, Object> content = documentUtility.getAsMap(document,
673                                 errorViews);
674                         if (content != null) {
675                             authProperties.putAll(content);
676                         }
677                     }
678                     String url = eval(document, authProperties, serviceUrlProperty);
679                     try {
680                         URIBuilder uriBuilder = new URIBuilder(url);
681                         if (queryParams != null) {
682                             for (Pair<String, ExpressionProperty> paramPair : queryParams) {
683                                 // left is the param key, right is the value
684                                 String paramKey = paramPair.getLeft();
685                                 if (paramKey == null) {
686                                     continue;
687                                 }
688                                 ExpressionProperty paramValueProp = paramPair.getRight();
689                                 String paramValue = eval(document, authProperties, paramValueProp);
690                                 uriBuilder.addParameter(paramKey, paramValue);
691                             }
692                         }
693                         return uriBuilder.build().toString();
694                     } catch (URISyntaxException e) {
695                         throw new SnapDataException(e, INVALID_URI)
696                                 .formatWith(url)
697                                 .withResolution(INVALID_URI_RESOLUTION);
698                     }
699                 }
700             };
701         }
702         return urlResolver;
703     }

752     boolean retryImmediately(HttpResponse response) {
753         checkNotNull(response, "response is null");
754         if (account instanceof Oauth2Provider) {
755             int sc = response.getStatusLine().getStatusCode();
756             if (sc == HttpStatus.SC_FORBIDDEN || sc == HttpStatus.SC_UNAUTHORIZED) {
757                 logger.warn("Rest call returned {} - reauthorization in progress", sc);
758                 if (account instanceof InMemoryOAuth2TokenAccount) {
759                     if (((InMemoryOAuth2TokenAccount<Header>) account).reloadOauth2Token()) {
760                         logger.warn("rest_InMemoryOauth2 - cache reloaded; will retry");
761                         updateHeaders();
762                         return true;
763                     } else {
764                         logger.warn("rest_InMemoryOauth2 - failed to reload cache and could" +
765                                 " not get new access token");
766                     }
767                 } else {
768                     if (snapControl.reloadSnapAccount()) {
769                         logger.warn("rest_oauth2 - account loaded from control plane; will retry");
770                         updateHeaders();
771                         return true;
772                     }
773                     logger.warn("Account still facing issues after sync with control plane " +
774                             "sleeping before attempted re-auth to prevent potential infinite loop");
775                     restRequestExecutor.sleepFor(DEFAULT_RETRY_INTERVAL_IN_SEC);
776                     if ((snapControl.reloadSnapAccount() || snapControl.reloadSnapAccount(true))) {
777                         logger.warn("rest_oauth2 - account reloaded; will retry");
778                         updateHeaders();
779                         return true;
780                     } else {
781                         logger.warn("rest_oauth2 - failed to reload account and " +
782                                 "could not get new access token");
783                     }
784                 }
785             }
786         }
787         return false;
788     }
