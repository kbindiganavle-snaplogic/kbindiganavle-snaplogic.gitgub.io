173     protected void process(Document document, String inputViewName) {
174         deleteDir = toBoolean(deleteDirExp.eval(document, null, new NullValueHandler()));
175         Object objDir = dirProp.eval(document);
176         String dir;
177         if (!(objDir instanceof String) || StringUtils.isEmpty((String) objDir)) {
178             String filePath = fileProp.eval(document);
179             if (filePath != null && UrlUtils.getProtocolPattern().matcher(filePath).find()) {
180                 dir = StringUtils.EMPTY;
181             } else {
182                 dir = DEFAULT_DIR;
183             }
184         } else {
185             dir = (String) objDir;
186             if (!dir.endsWith(URL_SEPARATOR)) {
187                 dir = dir.concat(URL_SEPARATOR);
188             }
189         }
190         if (fileProp.isEmpty()) {
191             FileSystem fileSystem = null;
192             try {
193                 Set<String> files;
194                 if (StringUtils.startsWithIgnoreCase(dir, PROTOCOL_ABFS)) {
195                     initializeAbfsCallableRestUtility();
196                     if (deleteDir) {
197                         // Delete the folder and its content recursively
198                         deleteFiles(document, dir);
199                         return;
200                     }
201                     // Lookup for files which are matching with regex pattern
202                     List<String> filesFound =
203                             abfsCallableRestUtility.executeListFiles(retryInterval, retries,
204                                     dir, true, filter, true, document);
205                     deleteFiles(document, filesFound.toArray(new String[0]));
206                 } else {
207                     String dirWithUserInfo = urlUtils.insertAuthority(dir, accountType, userInfo,
208                             UrlUtils.getProtocolPattern(), abfsRestUtility);
209                     URI pathUri = URI.create(dirWithUserInfo);
210                     if (StringUtils.isBlank(pathUri.getScheme())) {
211                         SnapDataException ex = new SnapDataException(ERR_INVALID_DIR)
212                                 .formatWith(dir)
213                                 .withReason(REASON_EMPTY_PROTOCOL)
214                                 .withResolution(RESOLUTION_INVALID_DIR);
215                         errorViews.write(ex, document);
216                         return;
217                     }
218                     String pattern;
219                     if (UrlUtils.URL_PROTOCOL_FILE.equals(pathUri.getScheme()) ||
220                             UrlUtils.URL_PROTOCOL_HDFS.equals(pathUri.getScheme())) {
221                         pattern = StringUtils.join(GLOB_PATTERN, pathUri.getPath(), filter);
222                     } else {
223                         pattern = StringUtils.join(GLOB_PATTERN, dirWithUserInfo, deleteDir ?
224                                 DOUBLE_STARS : filter);
225                     }
226                     PathMatcher matcher = FileSystems.getDefault().getPathMatcher(pattern);
227                     fileSystem = getFileSystem(dir);
228                     if (fileSystem == null) {
229                         SnapDataException ex = new SnapDataException(ERR_DIR_NOT_FOUND)
230                                 .formatWith(dir)
231                                 .withReason(REASON_DIR_NOT_FOUND)
232                                 .withResolution(RESOLUTION_DIR_NOT_FOUND);
233                         errorViews.write(ex, document);
234                         return;
235                     }
236                     if (dir.startsWith(UrlUtils.URL_PROTOCOL_HDFS)) {
237                         if (deleteDir) {
238                             deleteFiles(document, fileSystem, dir);
239                             return;
240                         }
241                         // Lookup for files which are matching with regex pattern
242                         files = findHadoopFiles(fileSystem, pathUri, matcher, false,
243                                 true, true);
244                         deleteFiles(document, fileSystem, files.toArray(new String[0]));
245                     } else {
246                         //wasb:, file:
247                         Path dirPath = FileSystemUtils.createPath(fileSystem, pathUri, false);
248                         // Lookup for files which are matching with regex pattern including
249                         // subdirectories
250                         files = findFiles(dirPath, matcher, false, true, deleteDir ?
251                                 true : false);
252                         if (deleteDir) {
253                             deleteDirectory(document, fileSystem, dir,
254                                     files.toArray(new String[0]));
255                             return;
256                         }
257                         deleteFiles(document, fileSystem, files.toArray(new String[0]));
258                     }
259                 }
260             } catch (SnapDataException e) {
261                 errorViews.write(e, document);
262             } catch (UnsupportedEncodingException e) {
263                 SnapDataException ex = new SnapDataException(e, ERROR_INVALID_ENCODING)
264                         .withReason(e.getMessage())
265                         .withResolution(RESOLUTION_ADDRESS_ISSUE);
266                 errorViews.write(ex, document);
267             } finally {
268                 FileSystemUtils.closeFileSystem(fileSystem, dir);
269             }
270         } else {
271             // Handles single file deletion when file property is set with filename
272             String file = fileProp.eval(document);
273             if (file != null) {
274                 String filePath = dir.concat(file);
275                 if (StringUtils.startsWithIgnoreCase(filePath, PROTOCOL_ABFS)) {
276                     // Handle file delete for ABFS
277                     initializeAbfsCallableRestUtility();
278                     deleteFiles(document, filePath);
279                 } else {
280                     // Handle file delete for Hdfs and wasb
281                     FileSystem fileSystem = getFileSystem(filePath);
282                     deleteFiles(document, fileSystem, filePath);
283                 }
284             } else {
285                 SnapDataException ex = new SnapDataException(ERR_FILE_NOT_FOUND)
286                         .withReason(REASON_FILEEXP_NULL)
287                         .withResolution(RESOLUTION_FILEEXP_NULL);
288                 errorViews.write(ex, document);
289             }
290         }
291     }

362                 if (isExists(filePath)) {

379                 errorViews.write(e, document);

388     /**
389      * Delete all the files in Windows Azure Storage Blob recursively.
390      * @param document
391      * @param fileSystem
392      * @param dir
393      * @param files
394      */
395     private void deleteDirectory(Document document, FileSystem fileSystem,
396                                  String dir, String... files) {
397         try {
398             /*
399                 For WASB, we cannot perform a directory delete operation using java.nio; it
400                 will throw an IOException if we attempt to do so. To support directory deletion,
401                 WASB provides a solution whereby the empty directory will automatically be
402                 removed. Therefore, to delete the directory, we should first remove its content.
403                 That means, in the hierarchical tree structure, we need to purge files from the
404                 leaf node to the root node. So, make sure to have a list of files in descending
405                 order.
406             */
407             for (String path : files) {
408                 deleteWithRetry(getPath(fileSystem, path));
409             }
410             Map data = new LinkedHashMap<String, Object>();
411             data.put(FILE_PATH_LABEL.toLowerCase(), dir);
412             data.put(STATUS, STATUS_200_OK);
413             outputViews.write(documentUtility.newDocumentFor(document, data), document);
414         } catch (SuggestViewAbortException e) {
415             throw e;
416         } catch (SnapDataException e) {
417             errorViews.write(e, document);
418         } catch (Throwable t) {
419             errorViews.write((SnapDataException) new SnapDataException(t, ERR_DELETE)
420                     .formatWith(dir)
421                     .withResolution(RESOLUTION_ADDRESS_ISSUE), document);
422         }
423     }

446      * Deletes file and does support retry on failure cases.
447      *
448      * @param path
449      * @return
450      * @throws Throwable
451      */
452     protected boolean deleteWithRetry(Path path) throws Throwable {
453         try {
454             ContextualCallable callable = context -> {
455                 return Files.deleteIfExists(path);
456             };
457             CheckedConsumer<Throwable> failedAttemptListener = e -> LOGGER.warn(
458                     "Failed to delete file: {}, {}", filePathURI, e.getMessage());
459             return failsafeExecute(callable, failedAttemptListener, retries, retryInterval);
460         } catch (Throwable e) {
461             throw e instanceof FailsafeException ? e.getCause() : e;
462         }
463     }
