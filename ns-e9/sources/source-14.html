299     protected void doWork() {
300         if (!inputViews.isEmpty()) {
301             final String[] columnNames = columnDefinitions.keySet().toArray(new String[0]);
302             final int columnSize = columnDefinitions.size();
303             Writer writer = null;
304             Path path = new Path(filename);
305             try {
306                 if (account instanceof FileSystemAccount) {
307                     FileConfig fileConfig = ((FileSystemAccount) account).buildFileConfig(
308                             filenameNoAuth,
309                             configuration);
310                     path = new Path(fileConfig.getFilepath());
311                     configuration = fileConfig.getConfig();
312                 }
313                 OrcRowInspector orcRowInspector = getOrcRowInspector(columnDefinitions);
314                 for (Document document : inputViews.documents(inputViews.get())) {
315                     Map<String, Object> dataMap = document.get(Map.class);
316                     Map<String, Object> rowData = loadRowData(orcRowInspector,
317                             dataMap, columnSize, columnNames);
318                     OrcRow record = new OrcRow(columnSize);
319                     int i = 0;
320                     for (Map.Entry<String, Object> entry : rowData.entrySet()) {
321                         record.setFieldValue(i++, entry.getValue());
322                     }
323                     if (writer == null) {
324                         try {
325                             writer = getORCFileWriter(orcRowInspector, path);
326                         } catch (FileAlreadyExistsException e) {
327                             handleFileAlreadyExistsException(e, document);
328                             continue;
329                         } catch (AWSBadRequestException e) {
330                             SnapDataException ex = new SnapDataException(e, BAD_REQUEST_EXCEPTION)
331                                     .withReason(e.getMessage())
332                                     .withResolution(BAD_REQUEST_RESOLUTION);
333                             errorViews.write(ex, document);
334                             continue;
335                         } catch (SnapDataException e) {
336                             errorViews.write(e, document);
337                             continue;
338                         }
339                     }
340                     writer.addRow(record);
341                 }
342             } catch (SerDeException e) {
343                 SnapDataException ex = new SnapDataException(e, SERDE_INITIALIZATION_ERROR)
344                         .withReason(e.getMessage())
345                         .withResolution(RESOLUTION_ADDRESS_ISSUE);
346                 errorViews.write(ex);
347             } catch (Exception e) {
348                 SnapDataException ex = new SnapDataException(e, UNABLE_TO_READ_INPUT_STREAM)
349                         .withReason(e.getMessage())
350                         .withResolution(RESOLUTION_ADDRESS_ISSUE);
351                 errorViews.write(ex);
352             } finally {
353                 for (Document document : inputViews.documents(inputViews.get())) {
354                     LOG.info("The discarded documents : " + document);
355                 }
356                 if (writer != null) {
357                     // This will close all the underlying streams -
358                     // outputStream and fsDataOutputStream
359                     try {
360                         writer.close();
361                         // If ABFS, writer writes to a temp file and we use the temp file to
362                         // write to ABFS using REST API
363                         if (StringUtils.startsWith(filenameNoAuth, PROTOCOL_ABFS)) {
364                             handleParquetOrcWriteToAbfs();
365                             deleteTmpFile(tempFile, DELETE_TEMP_FILE_MESSAGE_ORC);
366                             setAbfsPosixPermissions();
367                             /* Orc Writer writes to the output view here in the finally block,
368                              with more useful information like the raw data size etc. The
369                              handle() method in SimpleWriter does not write again to the output
370                              view as it fails at the check bytesRead > -1 which is true because
371                              the SimpleWriter's write methods are not invoked by Orc writer.
372                              But ABFS uses SimpleWriter's writeDataToAbfs, and bytesRead are
373                              updated to 0 at the end of a successful ABFS write. So there are
374                              two writes to the output view 1 from Orc and 1 from SimpleWriter.
375                              To skip the SimpleWriter writing to output view, reset bytesRead
376                              to -1 here. */
377                             bytesRead = -1;
378                         }
379                         //On success, write the number of bytes written to outputview.
380                         if (writer.getNumberOfRows() > 0 && !isBadConfig) {
381                             Map<String, String> data = new LinkedHashMap<>(4);
382                             data.put(FILENAME, filenameNoAuth);
383                             data.put(RESULT, SUCCESS);
384                             data.put(RESULT_BYTES, String.valueOf(writer.getRawDataSize()));
385                             data.put(RESULT_ROWS, String.valueOf(writer.getNumberOfRows()));
386                             outputViews.write(documentUtility.newDocument(data));
387                         }
388                     } catch (FileAlreadyExistsException e) {
389                         handleFileAlreadyExistsException(e, null);
390                     } catch (IOException e) {
391                         SnapDataException ex =
392                                 new SnapDataException(e, GENERIC_FILE_PARSE_EXCEPTION)
393                                         .withReason(e.getMessage())
394                                         .withResolutionAsDefect();
395                         errorViews.write(ex);
396                     } catch (Exception e) {
397                         SnapDataException ex = new SnapDataException(e, ERR_CANT_WRITE)
398                                 .withResolution(RESOLUTION_ADDRESS_ISSUE)
399                                 .withReason(e.getMessage());
400                         errorViews.write(ex);
401                     }
402                 }
403             }
404         }
405     }

407     private void handleFileAlreadyExistsException(FileAlreadyExistsException e, Document document) {
408         String fileNameToBeWrittenToOutputView;
409         String message;
410         // In ABFS case, filename should be abfs:// instead of the temp
411         // file name
412         if (StringUtils.startsWithIgnoreCase(filenameNoAuth, PROTOCOL_ABFS)) {
413             fileNameToBeWrittenToOutputView = filenameNoAuth;
414             message = String.format(ERR_ABFS_FILE_EXISTS,
415                     fileNameToBeWrittenToOutputView);
416         } else {
417             fileNameToBeWrittenToOutputView = filename;
418             message = e.getMessage();
419         }
420         switch (fileAction) {
421             case IGNORE:
422                 Map<String, String> map = new LinkedHashMap<>(2);
423                 map.put(RESULT, IGNORED);
424                 map.put(FILENAME, fileNameToBeWrittenToOutputView);
425                 outputViews.write(documentUtility.newDocument(map));
426                 break;
427             case ERROR:
428                 SnapDataException ex = new SnapDataException(e, ERR_FILE_EXISTS)
429                         .formatWith(fileNameToBeWrittenToOutputView)
430                         .withReason(message)
431                         .withResolution(RESOLUTION_ADDRESS_ISSUE);
432                 errorViews.write(ex, document);
433         }
434     }

440     private Writer getORCFileWriter(StructObjectInspector inspector, Path path)
441             throws IOException {
442         if (isExecuteDuringSuggest) {
443             if (fileAction == FileAction.OVERWRITE) {
444                 final FileSystem fileSystem = StringUtils.startsWith(filename, FILE_PROTOCOL) ?
445                         FileSystem.get(new Configuration()) : FileSystem.get(configuration);
446                 if (fileSystem.exists(path)) {
447                     try {
448                         UserGroupInformation.getCurrentUser().doAs(
449                                 new PrivilegedExceptionAction<Void>() {
450                                     @Override
451                                     public Void run() throws Exception {
452                                         fileSystem.delete(path, true);
453                                         return null;
454                                     }
455                                 });
456                     } catch (InterruptedException e) {
457                         LOG.error("Interrupted while deleting file");
458                         Thread.currentThread().interrupt();
459                     } catch (IOException e) {
460                         SnapDataException ex = new SnapDataException(e, UNABLE_TO_DELETE_FILE)
461                                 .withReason(e.getMessage())
462                                 .withResolution(RESOLUTION_ADDRESS_ISSUE);
463                         errorViews.write(ex);
464                     }
465                 }
466             }
467         }
468         OrcFile.WriterOptions writerOptions = OrcFile.writerOptions(configuration).
469                 inspector(inspector).compress(compressionKind);
470         return OrcFile.createWriter(path, writerOptions);
471     }

475         try {
476             Map<String, Object> rowData = new LinkedHashMap<>(dataMap.size());
477             for (int i = 0; i < columnSize; i++) {

494     private Object processTypes(ObjectInspector objectInspector, Object columnData) {
495         String type = objectInspector.getTypeName();
496         if (objectInspector instanceof StandardListObjectInspector) {
497             List<Object> returnList = new ArrayList<>(((List) columnData).size());
498             ObjectInspector elementInspector = ((StandardListObjectInspector) objectInspector)
499                     .getListElementObjectInspector();
500             for (Object element : (List) columnData) {
501                 returnList.add(processTypes(elementInspector, element));
502             }
503             return returnList;
504         } else if (objectInspector instanceof StandardMapObjectInspector) {
505             StandardMapObjectInspector mapObjectInspector
