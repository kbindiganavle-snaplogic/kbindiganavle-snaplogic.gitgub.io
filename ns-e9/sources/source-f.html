305     /**
306      * Process each input document. This is similar to {@link SimpleWriter#doWork() but
307      * there's no need to call initFilename (to insert auth info into URL) since we handle
308      * that differently. (Note: is this true with ADL?)
309      *
310      * NOTE: if there are multiple input views later documents may overwrite the first.
311      */
312     @Override
313     protected void doWork() {
314         try {
315             // write zipfile within privileged block.
316             doPrivilegedWork(filename);
317         } catch (IOException | InterruptedException e) {
318             // error creating zip file - individual failures are handled in writeData.
319             writeResult = WriteResult.ERROR;
320             handleException(e, ERROR_WRITING_ZIP_FILE, filenameNoAuth, e.getMessage(),
321                     RESOLUTION_ADDRESS_ISSUE, documentUtility, errorViews, EMPTY_DOC);
322         } catch (ExecutionException e) {
323             writeResult = WriteResult.ERROR;
324             handleException(e, ERROR_WRITING_ZIP_FILE, filenameNoAuth, e.getMessage(),
325                     RESOLUTION_ADDRESS_ISSUE, documentUtility, errorViews, EMPTY_DOC);
326         } catch (UndeclaredThrowableException e) {
327             writeResult = WriteResult.ERROR;
328             // error creating zip file - individual failures are handled in writeData.
329             // catching this here allows us to unwrap the undeclared exception.
330             handleException(e, ERROR_WRITING_ZIP_FILE, filenameNoAuth,
331                     e.getUndeclaredThrowable().getMessage(), RESOLUTION_ADDRESS_ISSUE,
332                     documentUtility, errorViews, EMPTY_DOC);
333         } catch (Exception e) {
334             // error creating zip file - individual failures are handled in writeData.
335             // this is a catch-all.
336             writeResult = WriteResult.ERROR;
337             handleException(e, ERROR_WRITING_ZIP_FILE, filenameNoAuth, e.getMessage(),
338                     RESOLUTION_ADDRESS_ISSUE, documentUtility, errorViews, EMPTY_DOC);
339         }
340     }

349     Map<String, Object> writeDocument(final ZipFileWriter w, final BinaryInput binaryData,
350             final FilenameGenerator filenameGenerator) {
351         final Document header = binaryData.getHeader();
352         Map<String, Object> metadata = new LinkedHashMap<>();
353         try {
354             metadata = w.writeDocument(binaryData, filenameGenerator);
355         } catch (SnapDataException e) {
356             errorViews.write(e);
357             writeResult = WriteResult.ERROR;
358         } catch (IOException e) {
359             handleException(e, ERROR_WRITING_ZIP_FILE, filenameNoAuth, e.getMessage(),
360                     RESOLUTION_ADDRESS_ISSUE, documentUtility, errorViews, header);
361             writeResult = WriteResult.ERROR;
362         }
363 
364         return metadata;
365     }

391     /**
392      * Do privileged work
393      *
394      * @param filename
395      * @throws IOException
396      * @throws InterruptedException
397      */
398     public void doPrivilegedWork(String filename) throws IOException, ExecutionException, InterruptedException {
399         // write zipfile within privileged block.
400         user.doAs(new PrivilegedExceptionAction<Void>() {
401             public Void run() throws Exception {
402                 try {
403                     LOG.debug("current user: {}", KerberosUtilities.getCurrentUser());
404 
405                     // one output file for all documents in input view.
406                     // may throw exception if output file exists.
407                     // this will create an empty archive if there are no input documents.
408                     initOutputStream();
409 
410                     // perform actual write.
411                     statusReporter.reset();
412                     statusReporter.notify(OPEN_FILE, filenameNoAuth);
413 
414                     // create a single output file
415                     try (final ZipFileWriter w = new ZipFileWriterImpl(filename, outputStream,
416                             statusReporter, WRITE_FILE)) {
417 
418                         // process each input view.
419                         long counter = 0;
420                         for (final InputView inputView : inputViews) {
421                             // individual FilenameGenerator using correct name and label.
422                             final FilenameGenerator filenameGenerator =
423                                     new DefaultFilenameGenerator(zipfileConfig.isUseInputViewLabel(),
424                                             zipfileConfig.getBaseDirectory(), inputView.getName(),
425                                             inputView.getLabel());
426 
427                             // process each input document
428                             for (final BinaryInput binaryData : inputViews.binaryInputs(inputView)) {
429                                 // limit number of entries during preview
430                                 if (isPreview && (counter++ >= maxSuggestValue)) {
431                                     break;
432                                 }
433 
434                                 outputMetadata.add(writeDocument(w, binaryData, filenameGenerator));
435                             }
436                         }
437                     }
438 
439                     // change file permissions. The output stream is closed first since some
440                     // filesystems have problems changing file permissions if the file is open.
441                     try {
442                         outputStream.close();
443                     } finally {
444                         // If the abfs scheme is there then need to write the data using rest call
445                         // before setting permission and then set permission accordingly
446                         if (StringUtils.startsWith(filenameNoAuth, PROTOCOL_ABFS)) {
447                             initInputStream();
448                             writeDataToAbfs(inputStream);
449                             inputStream.close();
450                             if (fileAction == FileAction.OVERWRITE || isCreated) {
451                                 setAbfsPosixPermissions();
452                             }
453                             if (isCreated) {
454                                 writeResult = WriteResult.CREATED;
455                             }
456                         } else {
457                             uri = new URI(filename);
458                             statusReporter.notify(SET_PERMISSIONS, filenameNoAuth);
459                             setPosixPermissions(uri, filenameNoAuth);
460                         }
461                     }
462                 } catch (Exception e) {
463                     // explicitly dump the stacktrace here since it's otherwise lost.
464                     ErrorUtils.dumpException(e);
465                     throw e;
466                 } finally {
467                     statusReporter.done();
468                 }
469                 return null;
470             }
471         });
472     }
