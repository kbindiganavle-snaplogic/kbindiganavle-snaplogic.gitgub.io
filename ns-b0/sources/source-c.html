105         if (MapUtils.isEmpty(queryParams)) {//no params provided

108         StringBuilder tags = new StringBuilder("");
109         for (Map.Entry entry : queryParams.entrySet()) {
110             tags.append(String.format(ANDED_CRITERIA_TAG, entry.getValue(), entry.getKey()));
111         }
112         String fetchXML = String.format(PAGING_REQUEST_WITH_FILTER_CRITERIA,
113                 Integer.toString(startPage),
114                 Integer.toString(entitiesPerPage),
115                 entityType);
116         fetchXML = fetchXML.replace(INJECTION_POINT, tags.toString());
117         fetchXML = loadFetchXmlWithPagingCookie(fetchXML, pagingCookie);
118         return fetchXML;

121     public static  String loadFetchXmlWithPagingCookie(String fetchXML, String pagingCookie) {
122         if (StringUtils.isNotBlank(pagingCookie)) {
123             String sanitizedCookie = microSoftlySanitizeCookie(pagingCookie);
124             String cookieAsAttribute =
125                     String.format(FETCH_XML_PAGING_COOKIE_ATTRIBUTE, sanitizedCookie);
126             fetchXML = fetchXML.replace(COOKIE_INJECTION_POINT, cookieAsAttribute);
127         } else {
128             fetchXML = fetchXML.replace(COOKIE_INJECTION_POINT, "");
129         }
130         return fetchXML;
131     }

133     public static String microSoftlySanitizeCookie(String cookiePackage) {

140     public static String urlDecodeCookie(String pagingCookie) {

142         pagingCookie = pagingCookie.replaceAll(DOUBLE_QUOTE_ENCODED, DOUBLE_QUOTE_DECODED);
143         pagingCookie = pagingCookie.replaceAll(SLASH_ENCODED, SLASH_DECODED);
144         pagingCookie = pagingCookie.replaceAll(OPEN_ANGLE_ENCODED, OPEN_ANGLE_DECODED);
145         pagingCookie = pagingCookie.replaceAll(EQUALS_ENCODED, EQUALS_DECODED);
146         pagingCookie = pagingCookie.replaceAll(CLOSE_ANGLE_ENCODED, CLOSE_ANGLE_DECODED);
147         pagingCookie = pagingCookie.replaceAll(OPEN_CURLY_ENCODED, OPEN_CURLY_DECODED);
148         pagingCookie = pagingCookie.replaceAll(CLOSE_CURLY_ENCODED, CLOSE_CURLY_DECODED);
149         pagingCookie = pagingCookie.replaceAll(DOUBLE_QUOTE_DECODED, SINGLE_QUOTE_DECODED);

153     public static String extractDecodedCookie(String urlDecoded) {
154         String returnValue = urlDecoded;
155         int pagingCookieStart = urlDecoded.indexOf(INNER_COOKIE_OPEN_TAG);
156         int pagingCookueStop = urlDecoded.indexOf(INNER_COOKIE_CLOSE_TAG);
157         if (pagingCookieStart > 0 && pagingCookueStop > 0) {
158             String extractedCookie = urlDecoded.substring(pagingCookieStart, pagingCookueStop);
159             extractedCookie += INNER_COOKIE_CLOSE_TAG;
160             returnValue = extractedCookie;
161         }
162         return returnValue;
163     }
