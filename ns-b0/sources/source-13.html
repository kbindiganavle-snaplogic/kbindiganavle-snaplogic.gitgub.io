270     public static HttpRequestBase getRequestObject(URIBuilder builder,
271             URI rawURIAlternative,
272             final RestMethodType crudSnapType,
273             StringEntity optionalPayload) {
274         HttpRequestBase request = null;
275         try {
276             URI finalizedURI = null != rawURIAlternative ? rawURIAlternative : builder.build();
277             switch (crudSnapType) {
278                 case GET:
279                     request = new HttpGet(finalizedURI);
280                     break;
281                 case POST:
282                     HttpPost post = new HttpPost(finalizedURI);
283                     if (optionalPayload != null) {
284                         post.setEntity(optionalPayload);
285                     }
286                     request = post;
287                     break;
288                 case PUT:// I don't believe we will need PUT for WebAPI, but just in case...
289                     HttpPut put = new HttpPut(finalizedURI);
290                     if (optionalPayload != null) {
291                         put.setEntity(optionalPayload);
292                     }
293                     request = put;
294                     break;
295                 case PATCH:
296                     HttpPatch patch = new HttpPatch(finalizedURI);
297                     if (optionalPayload != null) {
298                         patch.setEntity(optionalPayload);
299                     }
300                     request = patch;
301                     break;
302                 case DELETE:
303                     request = new HttpDelete(finalizedURI);
304                     break;
305                 default:
306                     throw new SnapDataException(String.format(Messages.UNSUPPORTED_HTTP_METHOD,
307                             crudSnapType.name()));
308             }
309 
310         } catch (URISyntaxException e) {
311             handleUriSyntaxException(e);
312         }
313         return request;
314     }

571             if (param.charAt(character) == HEX_VALUE_DECIMAL) {
572                 // continue the analysis
573                 decimalCounts++;
574                 continue;
575             } else if (param.charAt(character) < Constants.HEX_VALUE_ZERO ||
576                     param.charAt(character) > Constants.HEX_VALUE_NINE) {
577                 //out of range for numeric value
578                 isNumeric = false;
579                 break;
580             }
581         }
582         if (decimalCounts > 1) {//some number String or something with multiple periods
583             isNumeric = false;
584         }
