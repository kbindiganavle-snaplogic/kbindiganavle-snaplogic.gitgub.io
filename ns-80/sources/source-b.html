264     @Override
265     protected void process(Document inputDocument, String inputViewName) {
266         distinguishedName = util.escapeSpecialCharacters(
267                 distinguishedNameExp.eval(inputDocument));
268         if (distinguishedName == null) {
269             SnapDataException dataException = new SnapDataException(ERR_DN_NULL_MSG)
270                     .withReason(ERR_DN_NULL_REASON)
271                     .withResolution(ERR_DN_NULL_RESOLUTION);
272             errorViews.write(dataException);
273             return;
274         }
275         StringBuffer filter = new StringBuffer().append(APPEND_CRITERIA);
276         if (pairRoot == null) {
277             filter.append(RIGHT_PARENTHESES).toString();
278             performSearch(filter.toString(), inputDocument, true);
279         } else {
280             for (Pair<ExpressionProperty, ExpressionProperty> routeEntry : pairRoot) {
281                 String key = routeEntry.getLeft().eval(inputDocument);
282                 Object value = routeEntry.getRight().eval(inputDocument);
283                 if (DN_ATTRBUTES.contains(key) && value != null) {
284                     value = util.escapeSpecialCharactersForSearch(value.toString());
285                 }
286                 if (key != null && value != null) {
287                     if (key.contains(NOT)) {
288                         filter.append(LEFT_PARENTHESES)
289                                 .append(NOT)
290                                 .append(LEFT_PARENTHESES)
291                                 .append(key.substring(1))
292                                 .append(EQUALS)
293                                 .append(value.toString())
294                                 .append(RIGHT_PARENTHESES)
295                                 .append(RIGHT_PARENTHESES);
296                     } else {
297                         filter.append(LEFT_PARENTHESES)
298                                 .append(key)
299                                 .append(EQUALS)
300                                 .append(value.toString())
301                                 .append(RIGHT_PARENTHESES);
302                     }
303                 }
304             }
305             filter.append(RIGHT_PARENTHESES).toString();
306             performSearch(filter.toString(), inputDocument, true);
307         }
308     }

366     private void performSearch(String filterValue, Document inputDocument, boolean retry) {
367         if (ldapContext == null) {
368             ldapContext = account.connect();
369         }
370         searchControls = createSearchControls(scopeValue);
371         searchControls.setReturningAttributes(getReturningAttributes(inputDocument));
372         byte[] cookie;
373         String reason;
374         try {
375             if (ldapContextToFetchAttributes == null) {
376                 ldapContextToFetchAttributes = account.connect();
377             }
378             NamingEnumeration<SearchResult> answer;
379             // Request the paged results control
380             util.setControls(pageSize, ldapContext);
381             List<Map<String, Object>> results = new ArrayList<>();
382             Map<String, Object> output = new LinkedHashMap<>();
383             int count = 0;
384             do {
385                 if (pairRoot != null && !filterValue.equals(NO_FILTERS)) {
386                     answer = ldapContext.search(distinguishedName, filterValue, searchControls);
387                     reason = filterValue;
388                 } else {
389                     answer = ldapContext.search(distinguishedName, FILTER, searchControls);
390                     reason = FILTER;
391                 }
392                 if (!answer.hasMore()) {
393                     SnapDataException dataException = new SnapDataException(ERR_NO_SEARCH_RESULTS_MSG)
394                             .withReason(String.format(ERR_NO_SEARCH_RESULTS_REASON, reason))
395                             .withResolution(ERR_NO_SEARCH_RESULTS_RESOLUTION);
396                     errorViews.write(dataException);
397                     return;
398                 }
399                 try {
400                     while (answer.hasMore()) {
401                         Map<String, Object> recordEntry = new LinkedHashMap<>();
402                         SearchResult result = answer.next();
403                         Enumeration<? extends Attribute> attributes = result.getAttributes()
404                                 .getAll();
405                         while (attributes.hasMoreElements()) {
406                             Attribute attribute = attributes.nextElement();
407                             ADUtilities.putAttributeToMap(recordEntry, attribute);
408                         }
409                         Set<String> unassignedAttributes = getUnAssignedAttributes(
410                                 recordEntry.keySet(), util.getAllPossibleAttributes(
411                                         ldapContextToFetchAttributes, result.getNameInNamespace(),
412                                         util, cachedAttributes));
413                         for (String unassignedAttribute : unassignedAttributes) {
414                             recordEntry.put(unassignedAttribute, StringUtils.EMPTY);
415                         }
416                         if (!groupResult) {
417                             Document outDoc = documentUtility.newDocumentFor(inputDocument, recordEntry);
418                             outputViews.write(outDoc, passThrough ? inputDocument : null);
419                         } else {
420                             results.add(recordEntry);
421                         }
422                         count++;
423                         if (limit == count) {
424                             break;
425                         }
426                     }
427                 } catch (PartialResultException e) {
428                     // Workaround for AD servers not handling referrals correctly.
429                     // Inspired by Spring LDAP: https://docs.spring.io/spring-ldap/docs/1.1.2/api/org/springframework/ldap/LdapTemplate.html
430                     LOGGER.debug("PartialResultException encountered and ignored", e);
431                 }
432                 if (limit == count) {
433                     break;
434                 }
435                 cookie = util.parseControls(ldapContext);
436                 util.setRequestControls(ldapContext, cookie, pageSize);
437             } while (cookie != null && cookie.length != 0);
438             if (groupResult) {
439                 output.put("result", results);
440                 Document outDoc = documentUtility.newDocumentFor(inputDocument, output);
441                 outputViews.write(outDoc, passThrough ? inputDocument : null);
442             }
443         } catch (NamingException | IOException e) {
444             if (retry && groupResult) {
445                 ldapContextToFetchAttributes = null;
446                 ldapContext = null;
447                 performSearch(filterValue, inputDocument, false);
448             } else {
449                 SnapDataException snapEx = new SnapDataException(e, SEARCH_ERROR)
450                         .formatWith(distinguishedName)
451                         .withReason(e.getMessage())
452                         .withResolution(SEARCH_ERROR_RESOLUTION);
453                 errorViews.write(snapEx, inputDocument);
454             }
455         }
456     }
