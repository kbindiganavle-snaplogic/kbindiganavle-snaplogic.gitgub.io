208      * Creates a BinaryData object for each selected file and writes it to the output view. Then,
209      * reads data from the selected file and writes data to the piped output stream of the
210      * BinaryOutput object.
211      */
212     @Override
213     protected void process(Document document, String inputViewName) {
214         int numOfRetries = evalIntProperty(propertyValues.getAsExpression(RETRIES_PROP), document,
215                 0, 0, errorViews);
216         int retryInterval = evalIntProperty(propertyValues.getAsExpression(RETRY_INTERVAL_PROP),
217                 document, 1, 1, errorViews);
218         fileFinder.setRetries(numOfRetries, retryInterval);
219         if (numOfRetries > 0) {
220             tmpDir = JfsUtils.getTmpDir(expressionUtils, propertyValues, errorViews);
221         }
222         Map<String, String> env = getSasEnvProperties(advPropMap, document);
223         BinaryUtils.embedAuthenticationConfigIfNeeded(env, account);
224         String userInfo = null;
225         List<Path> fileOrFolders = null;
226         try {
227             userInfo = account.connect();
228             AccountType accountType = binaryUtils.getAccountType(account);
229             fileOrFolders = fileFinder.populate(propertyValues, document, accountType, userInfo,
230                     errorViews, env);
231         } catch (SuggestViewAbortException e) {
232             throw e;
233         } catch (SnapDataException e) {
234             errorViews.write(e, document);
235         } catch (SnapException e) {
236             writeErrorView(e, e.getMessage(), e.getReason(), e.getResolution(), documentUtility,
237                     errorViews, 0, document);
238         } catch (Throwable e) {
239             if (e.getCause() instanceof SuggestViewAbortException) {
240                 throw (SuggestViewAbortException) e.getCause();
241             }
242             String reason = e.getCause() != null ? e.getCause().getMessage() : null;
243             reason = reason == null ? ERR_POPULATE_FILE_LIST : reason;
244             writeErrorView(e, e.getMessage(), reason, RESOLUTION_ADDRESS_ISSUE, documentUtility,
245                     errorViews, 0, document);
246         }
247         try {
248             for (URI selectedFile : fileFinder.getResult()) {
249                 String filenameNoAuth = getFileNoAuth(selectedFile);
250                 // perform the file read with retries
251                 try {
252                     final String userinfoFinal = userInfo;
253                     ContextualRunnable runnable = context ->
254                             readFile(selectedFile, filenameNoAuth, userinfoFinal, document,
255                                     numOfRetries, retryInterval, env);
256                     CheckedConsumer<Throwable> failedAttemptListener = e -> LOGGER.warn(
257                             "{} - Failed to read file: {}, {}",
258                             RETRY_SIGNATURE, filenameNoAuth, e.getMessage());
259                     failsafeExecute(runnable, failedAttemptListener, numOfRetries, retryInterval);
260                 } catch (Throwable e) {
261                     Throwable ex = e instanceof FailsafeException ? e.getCause() : e;
262                     if (ex instanceof SuggestViewAbortException) {
263                         throw (SuggestViewAbortException) ex;
264                     } else if (ex.getCause() instanceof SuggestViewAbortException) {
265                         throw (SuggestViewAbortException) ex.getCause();
266                     } else if (ex instanceof SnapDataException) {
267                         SnapDataException ex2 = (SnapDataException) ex;
268                         ex2.withReason(addSelectedFiles(ex2.getReason()));
269                         errorViews.write(ex2, document);
270                     } else if (ex instanceof SnapException) {
271                         String reason = addSelectedFiles(((SnapException) ex).getReason());
272                         SnapDataException ex2 = new SnapDataException(ex, ex.getMessage())
273                                 .withReason(reason)
274                                 .withResolution(((SnapException) ex).getResolution());
275                         errorViews.write(ex2, document);
276                     } else {
277                         SnapDataException ex2 = new SnapDataException(ex, ERR_CANT_READ)
278                                 .formatWith(filenameNoAuth)
279                                 .withReason(addSelectedFiles(ex.getMessage()))
280                                 .withResolution(RESOLUTION_ADDRESS_ISSUE);
281                         errorViews.write(ex2, document);
282                     }
283                 }
284             }
285         } finally {
286             if (fileOrFolders != null) {
287                 for (Path path : fileOrFolders) {
288                     if (path instanceof S3Path) {
289                         ((S3Path) path).shutdown();
290                     }
291                 }
292             }
293         }
294     }

297     String addSelectedFiles(String reason) {
298         List<String> files = new ArrayList<>();
299         List<URI> uris = fileFinder.getResult();
300         for (URI uri : uris) {
301             if (files.size() > MAX_FILES_IN_ERROR) {
302                 files.add(String.format("...(%,d more)", uris.size() - MAX_FILES_IN_ERROR));
303                 break;
304             }
305             files.add(URLUtils.getFileNoAuth(uri));
306         }
307         return String.format(FORMAT_SELECTED_FILES, reason,
308                 StringUtils.join(files, JOIN_SEPARATOR));
309     }

312      * Reads a file.
313      *
314      * @param uri            - full poath URI
315      * @param filenameNoAuth - string URI without username/password
316      * @param userInfo       - username:password
317      * @param document       - input document
318      * @param numOfRetries   - max number of retries
319      * @throws IOException if it fails to read a file
320      */
321     protected void readFile(URI uri, String filenameNoAuth, String userInfo, Document document,
322             int numOfRetries, int retryInterval, Map<String, String> env) throws IOException {
323         URLConnection urlConnection = null;
324         File tmpFile = null;
325         try {
326             statusReporter.notify(OPEN_FILE, filenameNoAuth);
327             urlConnection = jfsUtils.openURLConnection(uri);
328             if (urlConnection == null) {
329                 writeErrorView(null, String.format(ERR_URL_CONNECT, filenameNoAuth),
330                         REASON_URL_CONNECT, RESOLUTION_CHECK_URL, documentUtility, errorViews,
331                         0, document);
332             }
333             if (account instanceof UrlAccount) {
334                 urlConnection = ((UrlAccount) account).withHeaders(urlConnection);
335             }
336             final long limit = maxSuggestValue == null ? Long.MAX_VALUE :
337                     maxSuggestValue.longValue();
338             // for some schemes, userInfo should be set to the request property
339             setRequestPropertyForAuthorization(uri.getScheme(), userInfo, urlConnection);
340             setFilesystemProperties(urlConnection, account, numOfRetries, retryInterval, env);
341             urlConnection.connect();
342             statusReporter.notify(READ_FILE, filenameNoAuth);
343             LOGGER.debug(DEBUG_URL_CONNECTED, filenameNoAuth);
344             final Document header;
345             // get input stream
346             try (final InputStream inputStream = urlConnection.getInputStream()) {
347                 header = documentUtility.newDocument(
348                         binaryUtils.createHeader(uri, urlConnection));
349                 final long contentLength = urlConnection.getContentLengthLong();
350                 if (contentLength >= 0) {
351                     statusReporter.begin(contentLength, "bytes");
352                 } else {
353                     statusReporter.beginUnknown();
354                 }
355                 if (numOfRetries > 0 && tmpDir != null) {
356                     tmpFile = File.createTempFile(this.getClass().getSimpleName(), null,
357                             tmpDir);
358                     try (OutputStream outputStream =
359                                  new GZIPOutputStream(new FileOutputStream(tmpFile))) {
360                         StatusReportingUtils.copy(statusReporter, inputStream, outputStream,
361                                 limit);
362                     }
363                     LOGGER.debug(DEBUG_COPY_IO_STREAM, tmpFile.toString());
364                 } else {
365                     LOGGER.debug(DEBUG_INPUTSTREAM_OPEN, filenameNoAuth);
366                     writeBinaryOutput(header, inputStream, limit, filenameNoAuth);
367                 }
368             }
369             if (tmpFile != null) {
370                 try (InputStream inputStream =
371                              new GZIPInputStream(new FileInputStream(tmpFile))) {
372                     writeBinaryOutput(header, inputStream, limit, tmpFile.toString());
373                 }
374             }
375         } finally {
376             statusReporter.reset();
377             urlUtil.disconnect(urlConnection);
378             if (tmpFile != null) {
379                 try {
380                     tmpFile.delete();
381                 } catch (Exception e) {
382                     // delete tmp file quietly
383                 }
384             }
385             LOGGER.debug(DEBUG_DISCONNECTED, filenameNoAuth);
386         }
387     }
