223     @Override
224     protected void setPermissions(final URI uri, final String filenameNoAuth) {
225         if (MapUtils.isNotEmpty(userFilePermissionsExpr)) {
226             setGivenPermissions();
227         }
228         if (userFilePermissions != null) {
229             String scheme = uri.getScheme();
230             if (PATTERN_PERMISSION_PROTOCOL.matcher(uri.toString()).find()) {
231                 FileSystem fileSystem = null;
232                 Path path = null;
233                 try {
234                     if (account instanceof FileSystemPropertiesProvider) {
235                         fileSystem = fileFinder.getFileSystemFor(uri, filenameNoAuth,
236                                 ((FileSystemPropertiesProvider) account).getRequestProperties());
237                     } else {
238                         fileSystem = fileFinder.getFileSystemFor(uri, filenameNoAuth);
239                     }
240                     if (URL_PROTOCOL_FILE.equalsIgnoreCase(scheme)) {
241                         path = fileSystem.getPath(uri.getPath());
242                     } else {
243                         path = fileSystem.getPath(uri.toString());
244                     }
245                     try {
246                         if (exists(path, filenameNoAuth)) {
247                             setPermissionsFailSafe(path);
248                         }
249                     } catch (IOException e) {
250                         errorHandlingOnSFTPFileOperation(e, filenameNoAuth);
251                     } catch (SuggestViewAbortException e) {
252                         throw e;
253                     } catch (SnapDataException e) {
254                         errorViews.write(e);
255                     } catch (Throwable e) {
256                         SnapDataException ex = new SnapDataException(e, ERR_SET_PERMISSIONS)
257                                 .formatWith(filenameNoAuth)
258                                 .withReason(e.getMessage())
259                                 .withResolution(RESOLUTION_ADDRESS_ISSUE);
260                         errorViews.write(ex);
261                     }
262                 } finally {
263                     fileFinder.closeFileSystem(fileSystem, filenameNoAuth);
264                     if (path instanceof S3Path) {
265                         ((S3Path) path).shutdown();
266                     }
267                 }
268             } else {
269                 SnapDataException ex =
270                         new SnapDataException(UNSUPPORTED_PROTOCOL_FOR_PERMISSIONS_ERR_MSG)
271                                 .withReason(UNSUPPORTED_PROTOCOL_FOR_PERMISSIONS_ERR_REASON)
272                                 .withResolution(
273                                         String.format(UNSUPPORTED_PROTOCOL_FOR_PERMS_ERR_RESOLUTION,
274                                                 SUPPORTED_PROTOCOLS));
275                 errorViews.write(ex);
276             }
277         }
278     }

281      * Checks if the file exists.
282      * Some remote file servers intermittently return false for the file-exists test even if the
283      * file has been written successfully in the previous step. It may be necessary to wait a
284      * short while and try again.
285      *
286      * @param path           - Path object of the file
287      * @param filenameNoAuth - file path without user credential
288      *
289      * @return true if the file exists
290      * @throws IOException when it fails to check if the file exists
291      */
292     @VisibleForTesting
293     boolean exists(Path path, String filenameNoAuth) throws IOException {
294         if (!binaryUtils.existsWithRetry(path, filenameNoAuth, retries, retryInterval, LOG)) {
295             LOG.warn("Target file not found unexpectedly, trying again for: {}",
296                     filenameNoAuth);
297             JfsUtils.sleepNoInterrupt(retryInterval, TimeUnit.SECONDS, LOG);
298             if (!binaryUtils.existsWithRetry(path, filenameNoAuth, retries, retryInterval, LOG)) {
299                 LOG.warn("Target file still not found, cannot set permissions for: {}",
300                         filenameNoAuth);
301                 return false;
302             }
303         }
304         return true;
305     }

307     private void errorHandlingOnSFTPFileOperation(IOException ioe, String filenameNoAuth) {
308         // Java's instanceof will return false when the library loaded
309         // are of different versions across snap and platform. Hence, checking
310         // the classname of the Exception class, com.jcraft.jsch.SftpException
311         if (Throwables.getRootCause(ioe).getClass().toString()
312                 .equalsIgnoreCase(SftpException.class.toString()) &&
313                 String.valueOf(Throwables.getRootCause(ioe)).contains(NO_SUCH_FILE)) {
314             LOG.info("File is not present while setting the permission: "
315                     + Throwables.getRootCause(ioe).getMessage());
316         } else {
317             SnapDataException ex = new SnapDataException(ioe, ERR_SET_PERMISSIONS)
318                     .formatWith(filenameNoAuth)
319                     .withReason(Throwables.getRootCause(ioe).getMessage())
320                     .withResolution(RESOLUTION_ADDRESS_ISSUE);
321             errorViews.write(ex);
322         }
323     }

326      * Creates a set of posix file permissions on the file referred to by uri variable.
327      * Presently only following protocols are supported: file, ftp, ftps and sftp

329     protected Set<PosixFilePermission> createPosixPermissionSetOnFile() {
330         Set<PosixFilePermission> posixFilePermissions = new HashSet<>();
331         for (UserFilePermissions object : userFilePermissions) {
332             String user = object.user;
333             String permissions[] = object.permissions.split(PLUS_SYMBOL);
334             boolean isPermissionsInvalid = false;
335             switch (user) {
336                 case USER_OWNER:
337                     for (String permission : permissions) {
338                         permission = permission.trim();
339                         switch (permission) {
340                             case FILE_PERMISSIONS_READ:
341                                 posixFilePermissions.add(PosixFilePermission.OWNER_READ);
342                                 break;
343                             case FILE_PERMISSIONS_WRITE:
344                                 posixFilePermissions.add(PosixFilePermission.OWNER_WRITE);
345                                 break;
346                             case FILE_PERMISSIONS_EXECUTE:
347                                 posixFilePermissions.add(PosixFilePermission.OWNER_EXECUTE);
348                                 break;
349                             default:
350                                 isPermissionsInvalid = true;
351                                 break;
352                         }
353                     }
354                     break;
355                 case USER_GROUP:
356                     for (String permission : permissions) {
357                         permission = permission.trim();
358                         switch (permission) {
359                             case FILE_PERMISSIONS_READ:
360                                 posixFilePermissions.add(PosixFilePermission.GROUP_READ);
361                                 break;
362                             case FILE_PERMISSIONS_WRITE:
363                                 posixFilePermissions.add(PosixFilePermission.GROUP_WRITE);
364                                 break;
365                             case FILE_PERMISSIONS_EXECUTE:
366                                 posixFilePermissions.add(PosixFilePermission.GROUP_EXECUTE);
367                                 break;
368                             default:
369                                 isPermissionsInvalid = true;
370                                 break;
371                         }
372                     }
373                     break;
374                 case USER_OTHERS:
375                     for (String permission : permissions) {
376                         permission = permission.trim();
377                         switch (permission) {
378                             case FILE_PERMISSIONS_READ:
379                                 posixFilePermissions.add(PosixFilePermission.OTHERS_READ);
380                                 break;
381                             case FILE_PERMISSIONS_WRITE:
382                                 posixFilePermissions.add(PosixFilePermission.OTHERS_WRITE);
383                                 break;
384                             case FILE_PERMISSIONS_EXECUTE:
385                                 posixFilePermissions.add(PosixFilePermission.OTHERS_EXECUTE);
386                                 break;
387                             default:
388                                 isPermissionsInvalid = true;
389                                 break;
390                         }
391                     }
392                     break;
393                 default:
394                     throw new SnapDataException(INVALID_USER_ERROR_MSG)
395                             .withReason(INVALID_USER_ERROR_REASON)
396                             .withResolution(String.format(INVALID_USER_ERROR_RESOLUTION, USERS));
397             }
398             if (isPermissionsInvalid) {
399                 throw new SnapDataException(INVALID_PERMISSIONS_ERROR_MSG)
400                         .withReason(INVALID_PERMISSIONS_ERROR_REASON)
401                         .withResolution(String.format(INVALID_PERMISSIONS_ERROR_RESOLUTION,
402                                 FILE_PERMISSIONS));
403             }
404         }
405         return posixFilePermissions;
406     }

409      * This method reads and caches the file permissions entered by the user.

411     private void setGivenPermissions() {
412         userFilePermissions = new UserFilePermissions[userFilePermissionsExpr.size()];
413         Set<String> users = new HashSet<>();
414         int index = 0;
415         for (Map.Entry<ExpressionProperty, ExpressionProperty> entry :
416                 userFilePermissionsExpr.entrySet()) {
417             String user = entry.getKey().eval(header);
418             String permissions = entry.getValue().eval(header);
419             if (StringUtils.isEmpty(user) || StringUtils.isEmpty(permissions)) {
420                 throw new ConfigurationException(INVALID_ENTRY_ERROR_MSG)
421                         .withReason(INVALID_ENTRY_ERROR_REASON)
422                         .withResolution(INVALID_ENTRY_ERROR_RESOLUTION);
423             }
424             user = user.toLowerCase().trim();
425             if (users.contains(user)) {
426                 throw new ConfigurationException(USER_PERMISSIONS_ERR_MSG)
427                         .withReason(USER_PERMISSIONS_ERR_REASON)
428                         .withResolution(USER_PERMISSIONS_ERR_RESOLUTION);
429             }
430             users.add(user);
431             userFilePermissions[index++] = new UserFilePermissions(user,
432                     permissions.toLowerCase());
433         }
434     }

436     private void setPermissionsFailSafe(Path path) throws Throwable {
437         userProvidedFilePermissions = createPosixPermissionSetOnFile();
438         try {
439             ContextualRunnable runnable = context ->
440                     Files.getFileAttributeView(path, PosixFileAttributeView.class)
441                             .setPermissions(userProvidedFilePermissions);
442             CheckedConsumer<Throwable> failedAttemptListener = e -> LOG.warn(
443                     "{} - Failed to set file permission, {}",
444                     RETRY_SIGNATURE, e.getMessage());
445             failsafeExecute(runnable, failedAttemptListener, retries, retryInterval);
446         } catch (Throwable t) {
447             throw t instanceof FailsafeException ? t.getCause() : t;
448         }
