162     @Override
163     protected void process(final Document document, final String inputViewName) {
164         if (connectFailed) {
165             return;
166         }
167         // evaluate expression properties
168         String query = queryExpression.eval(document);
169         String earliestTime = earliestTimeExpression.eval(document);
170         String latestTime = latestTimeExpression.eval(document);
171 
172         // process the Last property to get earliest/latest times
173         boolean isLastValueInvalid = lastExpression.isEmpty();
174         if (!isLastValueInvalid && presetSearchValue.equals(PRESET_SEARCH_DEFAULT)) {
175             BigInteger last = lastExpression.eval(document);
176             if (last == null) {
177                 isLastValueInvalid = true;
178             } else {
179                 int lastValue = last.intValue();
180                 switch (unit) {
181                     case SECONDS:
182                         timeRange = getRelativeTimeRangeBasedOnLastUnit(lastValue,
183                                 SplunkTimeMetrics.SECOND);
184                         earliestTime = timeRange[0];
185                         latestTime = timeRange[1];
186                         break;
187                     case MINUTES:
188                         timeRange = getRelativeTimeRangeBasedOnLastUnit(lastValue,
189                                 SplunkTimeMetrics.MINUTE);
190                         earliestTime = timeRange[0];
191                         latestTime = timeRange[1];
192                         break;
193                     case HOURS:
194                         timeRange = getRelativeTimeRangeBasedOnLastUnit(lastValue,
195                                 SplunkTimeMetrics.HOUR);
196                         earliestTime = timeRange[0];
197                         latestTime = timeRange[1];
198                         break;
199                     case WEEKS:
200                         timeRange = getRelativeTimeRangeBasedOnLastUnit(lastValue,
201                                 SplunkTimeMetrics.WEEK);
202                         earliestTime = timeRange[0];
203                         latestTime = timeRange[1];
204                         break;
205                     case DAYS:
206                         timeRange = getRelativeTimeRangeBasedOnLastUnit(lastValue,
207                                 SplunkTimeMetrics.DAY);
208                         earliestTime = timeRange[0];
209                         latestTime = timeRange[1];
210                         break;
211                     case MONTHS:
212                         timeRange = getRelativeTimeRangeBasedOnLastUnit(lastValue,
213                                 SplunkTimeMetrics.MONTH);
214                         earliestTime = timeRange[0];
215                         latestTime = timeRange[1];
216                         break;
217                     case QUARTERS:
218                         timeRange = getRelativeTimeRangeBasedOnLastUnit(lastValue,
219                                 SplunkTimeMetrics.QUARTER);
220                         earliestTime = timeRange[0];
221                         latestTime = timeRange[1];
222                         break;
223                     case YEARS:
224                         timeRange = getRelativeTimeRangeBasedOnLastUnit(lastValue,
225                                 SplunkTimeMetrics.YEAR);
226                         earliestTime = timeRange[0];
227                         latestTime = timeRange[1];
228                         break;
229                     default:
230                         SnapDataException ex = new SnapDataException(ERR_UNSUPPORTED_UNIT)
231                                 .formatWith(unit.toString())
232                                 .withReason(REASON_UNSUPPORTED_UNIT)
233                                 .withResolutionAsDefect();
234                         errorViews.write(ex, document);
235                         return;
236                 }
237             }
238         } else {
239             // get the time range based on preset search value
240             timeRange = getTimeRangeBasedOnPresetValue(earliestTime, latestTime);
241             earliestTime = timeRange[0];
242             latestTime = timeRange[1];
243         }
244         if (isLastValueInvalid &&
245                 (StringUtils.isBlank(earliestTime) || StringUtils.isBlank(latestTime))) {
246             SnapDataException ex = new SnapDataException(ERR_EMPTY_TIME)
247                     .withReason(REASON_EMPTY_TIME)
248                     .withResolution(RESOLUTION_EMPTY_TIME);
249             errorViews.write(ex, document);
250             return;
251         }
252         // Set the parameters for the search:
253         JobExportArgs args = new JobExportArgs();
254         args.put(EARLIEST_TIME_PROP, earliestTime);
255         args.put(LATEST_TIME_PROP, latestTime);
256         args.setSearchMode(JobExportArgs.SearchMode.NORMAL);
257         args.setOutputMode(outputMode);
258         MultiResultsReader<? extends SearchResults> multiResultsReader = null;
259         Map<String,Object> result = null;
260         int offset = 0;
261         try (InputStream inputStream = service.export(query, args)) {
262             if (JobExportArgs.OutputMode.JSON.equals(outputMode)) {
263                 multiResultsReader = new MultiResultsReaderJson(inputStream);
264             } else {
265                 multiResultsReader = new MultiResultsReaderXml(inputStream);
266             }
267             for (SearchResults searchResults : multiResultsReader) {
268                 for (Event event : searchResults) {
269                     result = Maps.newLinkedHashMap();
270                     result.put(PREVIEW, searchResults.isPreview());
271                     result.put(OFFSET, offset);
272                     result.putAll(event);
273                     outputViews.write(documentUtility.newDocument(result));
274                     offset++;
275                 }
276                 offset = 0;
277             }
278         } catch (SuggestViewAbortException e) {
279             throw e;
280         } catch (Exception e) {
281             String reason = improveErrorMessage(e.getMessage());
282             SnapDataException ex = new SnapDataException(e, ERR_GET_SEARCH_RESULT)
283                     .withReason(reason)
284                     .withResolution(RESOLUTION_ADDRESS_ISSUE);
285             errorViews.write(ex, document);
286         } finally {
287             try {
288                 if (multiResultsReader != null) {
289                     multiResultsReader.close();
290                 }
291             } catch (Exception e) {
292                 // close quietly
293                 LOGGER.info("Failed to close MultiResultsReader", e);
294             }
295         }
296     }

298     /**
299      * This method calculates the earliest time and latest time based on last & unit property
300      * which relates to the 'Relative Search' of Splunk and metrics are
301      * calculated based on Splunk API doc (Running an Export Job). Refer to
302      * https://dev.splunk.com/view/java-sdk/SP-CAAAEHQ#codeexamples for reference
303      *
304      * @param last
305      * @param timeMetric
306      * @return calculated earliest and latest time in an array of strings
307      */
308     private String[] getRelativeTimeRangeBasedOnLastUnit(final Integer last,
309             final SplunkTimeMetrics timeMetric) {
310         String latestTime;
311         String earliestTime;
312         if (StringUtils.equals(latestRelative, LATEST_RELATIVE_NOW)) {
313             latestTime = NOW;
314         } else {
315             latestTime = String.format(searchFormat, timeMetric);
316         }
317         String earliestTimeDefault = "-" + last + timeMetric;
318         if (StringUtils.equals(earliestRelative, EARLIEST_RELATIVE_NO_SNAP_TO)) {
319             earliestTime = earliestTimeDefault;
320         } else {
321             earliestTime = String.format(searchFormatWithPrefix, earliestTimeDefault, timeMetric);
322         }
323 
324         return new String[]{earliestTime, latestTime};
325     }
