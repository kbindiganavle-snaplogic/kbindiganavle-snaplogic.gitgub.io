542         for (Map<String, Object> entry : entries) {
543             String schema = (String) entry.get(SCHEMA_NAME);
544             List<Map> fields = (List<Map>) entry.get(FIELDS);
545             //ignore if either schema or fields are null
546             if (fields == null || schema == null) {
547                 continue;
548             }
549             Map attributes = new HashMap();
550             for (Map<String, Object> attr : fields) {
551                 if (attr != null) {
552                     attributes.put(attr.get(FIELD_NAME), attr.get(FIELD_VALUE));
553                 }
554             }
555             custom_attributes.put(schema, attributes);
556         }
557         user.set(CUSTOM_SCHEMAS, custom_attributes);
558     }

560     private String getMD5(String value) throws NoSuchAlgorithmException {
561         MessageDigest md = MessageDigest.getInstance(HASH_FUNCTION);
562         byte[] array = md.digest(value.getBytes());
563         StringBuilder sb = new StringBuilder();
564         for (byte elem : array) {
565             sb.append(Integer.toHexString((elem & 0xFF) | 0x100).substring(1, 3));
566         }
567         return sb.toString();
568     }
