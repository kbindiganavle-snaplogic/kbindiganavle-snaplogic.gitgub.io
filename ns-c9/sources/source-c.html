262     protected final String castToTypeForAllCDWs(String value, String targetColumn) {
263         switch (eltSCD2.getName()) {
264             case SNOWFLAKE:
265                 // snowflake is case sensitive when it comes to columns
266                 targetColumnNameToDataType.computeIfAbsent(targetColumn, missing -> {
267                     throw new ExecutionException("Unable to fetch data type for: %s")
268                                 .formatWith(targetColumn)
269                             .withReason("Metadata for column not found. Snowflake column names " +
270                                         "are case sensitive")
271                             .withResolution("Ensure the column names are configured in the same " +
272                                             "case as they are stored in Snowflake metadata");
273                 });
274                 return String.format(CAST_AS_TYPE_FMT, value,
275                         skipConstraints(targetColumnNameToDataType.get(targetColumn)));
276             default:
277                 // tgt column needs to be converted to lowercase before checking in map as we have
278                 // all keys in map converted to lowercase in the initial fetch target columns step
279                 targetColumnNameToDataType.computeIfAbsent(targetColumn.toLowerCase(), missing -> {
280                    throw new ExecutionException("Unable to fetch data type for: %s")
281                                 .formatWith(targetColumn)
282                            .withReason("Metadata for column not found")
283                            .withResolution("Ensure target table columns are correctly configured");
284                 });
285 
286                 String dataType = targetColumnNameToDataType.get(targetColumn.toLowerCase());
287 
288                 if (eltSCD2.isBigQueryDb()) {
289                     // BigQuery errors when casting to a parameterized data type eg. decimal(38,
290                     // 9) or string(1). In such cases, we leave it to the CDW to do implicit
291                     // casting and exclude from explicit casting done with this method
292                     if (dataType.contains("(") && dataType.contains(")")) {
293                         return value;
294                     }
295                 }
296                 return String.format(CAST_AS_TYPE_FMT, value, dataType);
297         }
298     }

400         List<String> insertColList = new ArrayList<>();
401         List<String> insertValueList = new ArrayList<>();
402         for (String naturalKey : naturalKeys) {
403             if (!scdMeaningToColumnName.containsValue(naturalKey)) {
404                 insertColList.add(naturalKey);
405                 insertValueList.add(remapAliasToStagedUpdateAlias(sourceAlias,
406                                                                  targetToSourceColumnName
407                                                                          .get(naturalKey)));
408             }
409         }
410 
411         for (String causeField : causeFields) {
412             if (!scdMeaningToColumnName.containsValue(causeField)) {
413                 insertColList.add(causeField);
414                 insertValueList.add(remapAliasToStagedUpdateAlias(sourceAlias,
415                                                                  targetToSourceColumnName
416                                                                          .get(causeField)));
417             }
418         }
419 
420         for (Map.Entry<String, String> entry : scdMeaningToColumnName.entrySet()) {
421             String scdFieldMeaning = entry.getKey();
422             String scdFieldColName = entry.getValue();
423             if (scdFieldMeaning.equalsIgnoreCase(HISTORICAL_ROW_LABEL)) {
424                 continue; // Historical row and current row point to the same target table
425                 // column, so using just one is sufficient for inserting.
426             } else if (scdFieldMeaning.equalsIgnoreCase(CURRENT_ROW_LABEL)) {
427                 insertColList.add(scdFieldColName);
428                 insertValueList.add(getCurrentRowInsertValue());
429             } else if (scdFieldMeaning.equalsIgnoreCase(END_DATE_LABEL) ||
430                        scdFieldMeaning.equalsIgnoreCase(END_DATE_HIST_LABEL)) {
431                 insertColList.add(scdFieldColName);
432                 insertValueList.add(getEndDateInsertValue());
433             } else if (scdFieldMeaning.equalsIgnoreCase(START_DATE_LABEL)) {
434                 insertColList.add(scdFieldColName);
435                 insertValueList.add(getStartDateInsertValue());
436             } else if (scdFieldMeaning.equalsIgnoreCase(INVALID_ROW_FLAG_LABEL)) {
437                 insertColList.add(scdFieldColName);
438                 insertValueList.add(getInvalidRowFlagInsertValue());
439             }
440         }
441 
442         for (Map.Entry<String, String> targetSourceMapColEntry :
443                                                 targetToSourceColumnName.entrySet()) {
444             String targetCol = targetSourceMapColEntry.getKey();
445             String sourceCol = targetSourceMapColEntry.getValue();
446             // if natural keys or cause fields or scd fields are mapped, they are already handled
447             // in the code above, only take care of the rest of mapped columns
448             if (!insertColList.contains(targetCol)) {
449                 insertColList.add(targetCol);
450                 insertValueList.add(eltSCD2.isAzureSynapseDb() ?
451                         addStagedUpdatesPrefix(castInsertValuesSynapse.get(targetCol)) :
452                         castToTypeForAllCDWs(
453                                 remapAliasToStagedUpdateAlias(sourceAlias, sourceCol), targetCol));
454             }
455         }
456 
457         List<List<String>> retLists = new ArrayList<>();
458         retLists.add(insertColList);
459         retLists.add(insertValueList);
460 
461         return retLists;
462     }
