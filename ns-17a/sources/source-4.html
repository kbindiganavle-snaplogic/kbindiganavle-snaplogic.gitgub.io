451     public void configure(PropertyValues propertyValues) throws ConfigurationException {
452         serviceName = propertyValues.get(SERVICE_KEY);
453         if (workdayAccount == null) {
454             throw new ConfigurationException(ERR_INITIALIZE_ACCOUNT)
455                     .withReason(REASON_EMPTY_ACCOUNT)

457         }
458         String wsdl = workdayAccount.getWsdlURL(serviceName);
459         if (StringUtils.isBlank(wsdl)) {
460             throw new ConfigurationException(INVALID_PROPERTY_VALUE_CONFIGURATION)

462                     .withResolution(PLEASE_UPDATE_THE_ACCOUNT_WSDLS);
463         }

465         Map<String, QName> defaultSettingValues = getDefaultSettingValues(serviceName);
466         QName serviceQName = defaultSettingValues.get(WsdlPropertiesFactory.SERVICE_PROP);
467         QName portQName = defaultSettingValues.get(WsdlPropertiesFactory.ENDPOINT_PROP);
468         String namespace = serviceQName.getNamespaceURI();
469         String localPart = objectName;
470         QName operationQName = new QName(namespace, localPart);
471         timeout = (propertyValues.get(PROP_TIMEOUT) != null &&
472                 propertyValues.get(PROP_TIMEOUT) instanceof  BigInteger) ?
473                         ((BigInteger) propertyValues.get(PROP_TIMEOUT)).intValue() : 0;
474 
475         if ((StringUtils.isBlank(serviceQName.toString()) ||
476                 StringUtils.isBlank(portQName.toString()) ||
477                 StringUtils.isBlank(operationQName.toString()))) {
478             throw new ConfigurationException(INVALID_PROPERTY_VALUE_CONFIGURATION)
479                     .withReason(MISSING_EITHER_WSDL_URL_PORT_AND_OR_SERVICE_VALUES)
480                     .withResolution(PLEASE_SET_THE_PROPERTY_VALUES_AND_RETRY);
481         }

488             if (e instanceof  ExecutionException) {
489                 /**

492                 ExecutionException ee = (ExecutionException) e;

498             }
499             /**
500              * throwing config error which are not related to WSDL parse issues
501              */
502             throw new ConfigurationException(e, String.format(ERR_INVALID_SERVICE_AND_OBJECT,
503                     serviceName, objectName))
504                     .withReason(e.getMessage())
505                     .withResolution(GENERIC_RESOLUTION);
506         }
507 
508         List<Handler> handlerChain = new ArrayList<>();

514         if (isValidateOnlyLoad) {

798     @SuppressWarnings("unchecked")
799     protected Object simplifyOutput(Object data) {
800         if (data instanceof Map) {
801             Map<String, Object> map = (Map<String, Object>) data;
802             Map<String, Object> newMap = new LinkedHashMap<>(map.size());
803             for (Map.Entry<String, Object> entry : map.entrySet()) {
804                 String key = entry.getKey();
805                 Object value = entry.getValue();
806                 String newKey;
807                 Object newValue = value;
808                 if (key.equals(WD_COLON_ID)) {
809                     // simplify ID data
810                     newKey = ID;
811                     if (value instanceof List) {
812                         newValue = new LinkedHashMap<>();
813                         String idKey = null;
814                         String idValue = null;
815                         for (Map<String, String> item : (List<Map<String, String>>) value) {
816                             for (Map.Entry<String, String> idEntry : item.entrySet()) {
817                                 if (DOLLAR.equals(idEntry.getKey())) {
818                                     idValue = idEntry.getValue();
819                                 } else {
820                                     idKey = idEntry.getValue();
821                                 }
822                             }
823                             if (idKey != null) {
824                                 ((Map<String, String>) newValue).put(idKey, idValue);
825                             } else {
826                                 idKey = idKey;
827                             }
828                         }
829                     }
830                     newMap.put(newKey, newValue);
831                 } else {
832                     if (key.startsWith(WD_COLON)) {
833                         newKey = key.substring(WD_COLON.length());
834                     } else if (key.startsWith(AT_WD_COLON)) {
835                         newKey = key.substring(AT_WD_COLON.length());
836                     } else if (key.startsWith(AT_XMLNS_COLON)) {
837                         continue;
838                     } else {
839                         newKey = key;
840                     }
841                     newValue = simplifyOutput(value);
842                     if (newKey != null) {
843                         newMap.put(newKey, newValue);
844                     } else {
845                         newKey = newKey;
846                     }
847                 }
848             }
849             return newMap;
850         } else if (data instanceof List) {
851             List<Object> dataList = (List<Object>) data;
852             List<Object> newList = new ArrayList<>(dataList.size());
853             for (Object item : (List<Object>) data) {
854                 newList.add(simplifyOutput(item));
855             }
856             return newList;
857         }
858         return data;
859     }
