491     protected String getContent(HttpEntity entity, @Nullable Map<String, Object> headers) throws
492             IOException {
493         String contentEncoding = null;
494         String contentType = null;
495         if (headers != null) {
496             contentEncoding = getHeader(headers, HttpHeaders.CONTENT_ENCODING);
497             contentType = getHeader(headers, HttpHeaders.CONTENT_TYPE);
498         }
499         // If the HTTP response does not explicitly mention the charset then we
500         // are assuming JCC's default charset to be used for Windows/ Linux
501         String charset = Charset.forName(UTF8).name();
502         if (contentType != null) {
503             int charsetEqualToIndex = contentType.toLowerCase().indexOf(CHARSET_EQUAL_TO);
504             if (charsetEqualToIndex > -1) {
505                 charset = contentType.substring(charsetEqualToIndex + CHARSET_EQUAL_TO.length());
506                 // if the value of the content-type in the headers is a map, charset may
507                 // ends with '}'
508                 charset = substringTo(charset, '}');
509                 // the content-type may be of pattern "...;charset=utf-8;..."
510                 charset = substringTo(charset, ';');
511                 // the content-type may be of pattern "..., charset=utf-8, ..."
512                 charset = substringTo(charset, ',');
513             }
514         }
515 
516         InputStream is = null;
517         try {
518             is = entity.getContent();
519             if (is == null) {
520                 return null;
521             }
522             if (StringUtils.containsIgnoreCase(contentEncoding, GZIP)) {
523                 is = new GZIPInputStream(is);
524                 LOG.warn("Detected gzip encoded entity");
525             }
526             String entityString = null;
527             try {
528                 return IOUtils.toString(is, charset);
529             } catch (IllegalCharsetNameException | UnsupportedEncodingException e) {
530                 return IOUtils.toString(is, StandardCharsets.UTF_8);
531             }
532         } finally {
533             IOUtils.closeQuietly(is);
534         }
535     }

542     private String getHeader(Map<String, Object> headers, String headerName) {
543         Object value = headers.get(headerName);
544         if (value == null) {
545             value = headers.get(StringUtils.lowerCase(headerName));
546         }
547         String result = null;
548         if (value instanceof String) {
549             result = (String) value;
550         } else if (value instanceof Iterable) {
551             result = StringUtils.join((Iterable<?>) value, ';');
552             LOG.warn("{}: {}, {}", headerName, result, value.getClass().getSimpleName());
553         } else if (value == null) {
554             LOG.trace("{}: null", headerName);
555         } else {
556             result = value.toString();
557             LOG.warn("{}: {}, {}", headerName, result, value.getClass().getSimpleName());
558         }
559         return result;
560     }

724      * merge cookies into headers list
725      * @param headers
726      */
727     private void addAllCookies(Map headers) {
728         if (cookieList == null || cookieList.isEmpty()) {
729             return;
730         }
731         if (headers == null) {
732             headers = new HashMap();
733         }
734         List<Object> cookieHeaders = new ArrayList<>();
735         for (Cookie cookie : cookieList) {
736             Map cookieMap = OBJECT_MAPPER.convertValue(cookie, Map.class);
737             cookieMap.put(cookieMap.get(COOKIE_NAME_FIELD), cookieMap.get(COOKIE_VALUE_FIELD));
738             cookieMap.remove(COOKIE_NAME_FIELD);
739             cookieMap.remove(COOKIE_VALUE_FIELD);
740             cookieMap.values().removeAll(Collections.singleton(null));
741             cookieHeaders.add(cookieMap);
742         }
743         String cookieHeaderName = SET_COOKIE_LOWERCASE;
744         if (headers.get(SET_COOKIE_CAMELCASE) != null) {
745             cookieHeaderName = SET_COOKIE_CAMELCASE;
746             cookieHeaders.addAll((Collection<?>) headers.get(SET_COOKIE_CAMELCASE));
747         } else if (headers.get(SET_COOKIE_LOWERCASE) != null) {
748             cookieHeaders.addAll((Collection<?>) headers.get(SET_COOKIE_LOWERCASE));
749         }
750         headers.put(cookieHeaderName, cookieHeaders.stream()
751                 .distinct()
752                 .collect(Collectors.toList()));
753     }

755     /**
756      * Attempts to parse the header value if its format is "key1=value1; key2=value2; ..."
757      *
758      * @param value - header value string
759      *
760      * @return parsed map data if the format is as expected, or else the value itself
761      */
762     protected Object parseHeaderValue(Object value) {
763         if (value instanceof String) {
764             String[] items = ((String) value).split(SEMI_COLON);
765             if (items.length > 1) {
766                 Map<String, String> map = new LinkedHashMap<>(items.length);
767                 for (String item : items) {
768                     if (StringUtils.isNotBlank(item)) {
769                         String[] pair = item.split(EQUAL);
770                         if (pair.length == 2) {
771                             map.put(StringUtils.trim(pair[0]), StringUtils.trim(pair[1]));
772                         } else {
773                             // This item is not in the format "key=value" and is not parsed
774                             String key = VALUE;
775                             int i = 1;
776                             while (map.containsKey(key)) {
777                                 key = String.format(FORMAT_VALUE, i++);
778                             }
779                             map.put(key, item);
780                         }
781                     }
782                 }
783                 return map;
784             }
785         }
786         return value;
787     }
