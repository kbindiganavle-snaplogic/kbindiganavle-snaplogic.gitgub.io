299     protected void initSessionsForExecute() {
300         JMSAccount jmsAccount = ((JMSAccount) account);
301 
302         switch (ackType) {
303             case AUTO_ACKNOWLEDGE:
304                 jmsAccount.setConnectionSessionProps(false, Session.AUTO_ACKNOWLEDGE,
305                         clientId, isDurableSubscriber, controlSessionRequired, asyncMode);
306                 session = account.connect();
307                 break;
308             case CLIENT_ACKNOWLEDGE:
309                 if (account instanceof SQSAccount) {
310                     //Amazon SQS doesn't have support for transacted session instead it provides
311                     //an extra acknowledge mode called UNORDERED_ACKNOWLEDGE which enables
312                     //acknowledging specific message consumed without getting all the consumed
313                     //messages auto acknowledged
314                     jmsAccount.setConnectionSessionProps(false,
315                             SQSSession.UNORDERED_ACKNOWLEDGE, clientId, isDurableSubscriber,
316                             controlSessionRequired, asyncMode);
317                     session = account.connect();
318                 } else {
319                     jmsAccount.setConnectionSessionProps(true,
320                             Session.CLIENT_ACKNOWLEDGE, clientId, isDurableSubscriber,
321                             controlSessionRequired, asyncMode);
322                     session = account.connect();
323                 }
324                 break;
325             case DUPS_OK_ACKNOWLEDGE:
326                 jmsAccount.setConnectionSessionProps(false,
327                         Session.DUPS_OK_ACKNOWLEDGE, clientId, isDurableSubscriber,
328                         controlSessionRequired, asyncMode);
329                 session = account.connect();
330                 break;
331             default:
332                 // Should never reach here
333                 throw new ConfigurationException(String.format(INVALID_ACKNOWLEDGE_MODE,
334                         ackType.name()));
335         }
336         if (controlSessionRequired) {
337             controlSession = jmsAccount.getControlSession();
338             controlDestination = JMSUtils.createDestination(controlSession, destinationType,
339                     controlDestinationName);
340         }

366     @Override
367     public void execute() throws ExecutionException {
368         errorMap = new ConcurrentHashMap<>();
369 
370         try {
371             Failsafe
372                     .with(retryPolicy)
373                     .get(context -> {
374                         Execution execution = (Execution) context;
375                         if (execution.getAttemptCount() > 0) {
376                             // its a retry; close prev connection/sessions
377                             account.disconnect();
378                         }
379                         initSessionsForExecute();
380                         if (controlDestination != null) {
381                             startControlMessageListener();
382                         }
383                         return null;
384                     });
385         } catch (Exception e1) {
386             Throwable cause = e1 instanceof FailsafeException ? e1.getCause() : e1;
387 
388             if (cause instanceof JMSException || cause instanceof SnapDataException) {
389                 throw new ExecutionException(e1, JMS_CONSUMER_CONTROL_DESTINATION_ERROR)
390                         .withResolution(JMS_CONSUMER_CONTROL_DESTINATION_ERROR_RES);
391             }
392         }
393 
394         if (!inputViews.isEmpty()) {
395             //If the user passes messageSelector from upstream snap then we create one consumer
396             //for each input document and terminate it after specified number of messages read
397             //If user doesn't provide message count value the default it to 1
398             if (maxMessageCount <= 0) {
399                 maxMessageCount = 1;
400             }
401             for (Document document : inputViews.documents(inputViews.get())) {
402                 process(document);
403             }
404         } else {
405             process(null);
406         }
407     }

409     private void process(Document document) throws SnapDataException {
410         messageSelector = (messageSelectorExp != null) ?
411                 (String) messageSelectorExp.eval(document) :
412                 null;
413         setMessageCount(document);
414 
415         try {
416             // startProcessing() throws JMSException or
417             //  SnapDataException/SnapException which in turn wraps JMSException
418             startProcessing(document);
419             stopFlag = false;
420             messageCount.set(0);
421         } catch (SnapDataException e1) {
422             errorViews.write(e1, document);
423         } catch (SnapException e2) {
424             SnapDataException sde = new SnapDataException(
425                     Throwables.getRootCause(e2),
426                     e2.getMessage())
427                     .withReason(e2.getReason())
428                     .withResolution(e2.getResolution());
429             errorViews.write(sde, document);
430         } catch (Exception e3) {
431             SnapDataException sde = new SnapDataException(
432                     Throwables.getRootCause(e3),
433                     e3.getMessage())
434                     .withReason(PLEASE_SEE_ERROR_MESSAGE_FOR_DETAILS)
435                     .withResolution(PLEASE_ADDRESS_ERROR_MESSAGE_AND_RETRY);
436             errorViews.write(sde, document);
437         } finally {
438             if (consumer != null) {
439                 try {
440                     consumer.close();
441                 } catch (JMSException je) {
442                     throw new ExecutionException(je, ERROR_CLOSING_CONSUMER)
443                             .withResolutionAsDefect();
444                 }
445             }
446         }
447     }

481     private void startProcessing(Document document) {
482         if (!asyncMode) {
483             // consumeSynchronously() throws JMSException or SnapException which in turn wraps
484             //   JMSException
485             consumeSynchronously(document);
486         } else {
487             boolean firstIteration = true;
488 
489             while (!stopFlag) {
490                 if (firstIteration) {
491                     firstIteration = false;
492                     try {
493                         // For the first time, we will try with all the retries
494                         Failsafe
495                                 .with(retryPolicy)
496                                 .get(context -> {
497                                     Execution execution = (Execution) context;
498                                     if (execution.getAttemptCount() > 1) {
499                                         // its a retry
500                                         // close previous sessions/connection
501                                         account.disconnect();
502                                         // reconnect and reinitialise
503                                         reconnectToJmsServer();
504                                         initControlMessageListener();
505                                         initConsumer(document);
506                                         setMessageAndExceptionListeners();
507                                     } else {
508                                         initConsumer(document);
509                                         setMessageAndExceptionListeners();
510                                     }
511                                     return null;
512                                 });
513                     } catch (Exception e1) {
514                         Throwable cause = e1 instanceof FailsafeException ? e1.getCause() : e1;
515 
516                         throw new SnapDataException(cause,
517                                 ERROR_ASYNC_MODE_CONSUME_MSG)
518                                 .withResolution(ERROR_ASYNC_MODE_CONSUME_RESOLUTION);
519                     }
520                 }
521                 if (maxMessageCount == 0) {
522                     setStopFlag();
523                 }
524                 try {
525                     // Waiting after the if condition as the consumer needs to wait
526                     //  until the specified interval before it checks for the exit (either no
527                     //  messages in the broker or the force stop invokes).
528                     synchronized (session) {
529                         session.wait(SLEEP_TIME_INTERVAL_IN_MS);
530                     }
531                 } catch (InterruptedException e1) {
532                     LOG.error(CONSUMER_SNAP_THREAD_INTERRUPTED, e1);
533                     break;
534                 }
535                 // If an exception occurs while consuming in async mode
536                 //   (broker instance down, network connection disrupted, etc) then that is
537                 //   handled here
538                 if (asyncJmsException != null) {

540                     // will try n retries if n > 0
541                     boolean throwException = (retryLimit == 0);
542                     final AtomicBoolean firstPass = new AtomicBoolean(true);
543                     try {
544                         if (!throwException) {

546                                     .with(retryPolicy)
547                                     .run(() -> {

549                                         if (firstPass.get()) {
550                                             firstPass.set(false);
551                                             throw (Exception) asyncJmsException;
552                                         } else {
553                                             // its a retry
554                                             // close the previous sessions/connection
555                                             account.disconnect();
556                                             // reconnect and reinitialise
557                                             reconnectToJmsServer();
558                                             initControlMessageListener();

561                                             //   SnapDataException or SnapException

563                                             setMessageAndExceptionListeners();
564                                         }

566                         }
567                     } catch (Exception e1) {
568                         throwException = true;
569                     } finally {
570                         if (throwException) {
571                             throw new SnapDataException(asyncJmsException,
572                                     ERROR_ASYNC_MODE_CONSUME_MSG)
573                                     .withResolution(ERROR_ASYNC_MODE_CONSUME_RESOLUTION);

576                 }

861                     if (message != null) {

864                             //  JMSException

969                             .withResolutionAsDefect();

971             }
972             setStopFlag();
973         }
974         if (ackType == AcknowledgmentType.CLIENT_ACKNOWLEDGE
975                 && account instanceof SQSAccount) {
976             try {
977                 acknowledgeMessageMap.put(message.getJMSMessageID(), message);
978             } catch (JMSException e) {
979                 throw new ExecutionException(e, ERROR_WHILE_PROCESSING_MESSAGE)
980                         .withResolutionAsDefect();
981             }
982         }
983         for (String messageID : errorMap.keySet()) {
984             errorViews.write(errorMap.get(messageID));

1053             final AtomicBoolean recoverError = new AtomicBoolean();
1054             final AtomicBoolean ackError = new AtomicBoolean();
1055             final AtomicBoolean closeError = new AtomicBoolean();
1056             final StringBuffer messageIDToBeDisplayed = new StringBuffer();
1057 
1058             RetryPolicy retryPolicy = (RetryPolicy) new RetryPolicy()
1059                     .withDelay(Duration.ofMillis(retryIntervalMillis))

1066                                     failure.getMessage(), failure);
1067                         } else if (ackError.get()) {
1068                             LOG.warn("Error doing commit on the session or " +
1069                                             "acknowledge on the message {}",
1070                                     failure.getMessage(),
1071                                     failure);
1072                         } else if (closeError.get()) {
1073                             LOG.warn("Error closing the consumer {}",
1074                                     failure.getMessage(),
1075                                     failure);
1076                         }
1077                     })
1078                     .onRetry(e -> {
1079                         // log the retry as it seems to be a network/connection failure
1080                         ExecutionAttemptedEvent event = (ExecutionAttemptedEvent) e;
1081                         Throwable failure = event.getLastFailure();
1082                         LOG.warn("Retrying {}th time as new session and connection " +
1083                                 "might be obtained next", event.getAttemptCount());
1084                     })
1085                     .handleIf(failure -> isRetryRequired((Throwable) failure));
1086 
1087             if (arg != null) {
1088                 try {
1089                     Failsafe
1090                             .with(retryPolicy)
1091                             .run(() -> {
1092                                 synchronized (session) {
1093                                     try {
1094                                         recoverError.set(false);
1095                                         ackError.set(false);
1096                                         closeError.set(false);
1097

1099                                         if (StringUtils.startsWith(messageID,
1100                                                 MessageHandler.RECOVER_MESSAGE_PREFIX)) {
1101                                             recoverError.set(true);
1102                                             messageID = StringUtils.substringAfter(messageID,
1103                                                     COLON);
1104                                             messageIDToBeDisplayed.delete(0,
1105                                                     messageIDToBeDisplayed.length());
1106                                             messageIDToBeDisplayed.append(messageID);
1107                                             if (StringUtils.isNotBlank(messageID)
1108                                                     && acknowledgeMessageMap.size() > 0) {
1109                                                 acknowledgeMessageMap.remove(messageID);
1110                                             } else {
1111                                                 session.rollback();
1112                                             }
1113                                         } else if (StringUtils.startsWith(messageID,
1114                                                 MessageHandler.ACKNOWLEDGE_MESSAGE_PREFIX)) {
1115                                             ackError.set(true);
1116                                             messageID = StringUtils.substringAfter(messageID,
1117                                                     COLON);
1118                                             messageIDToBeDisplayed.delete(0,
1119                                                     messageIDToBeDisplayed.length());
1120                                             messageIDToBeDisplayed.append(messageID);
1121                                             if (StringUtils.isNotBlank(messageID)
1122                                                     && acknowledgeMessageMap.size() > 0) {
1123                                                 Message message = acknowledgeMessageMap
1124                                                         .get(messageID);
1125                                                 if (message != null) {
1126                                                     message.acknowledge();
1127                                                 }
1128                                             } else {
1129                                                 session.commit();
1130                                             }
1131                                         } else {
1132                                             messageIDToBeDisplayed.delete(0,
1133                                                     messageIDToBeDisplayed.length());
1134                                             messageIDToBeDisplayed.append(messageID);
1135                                         }
1136                                         if ((maxMessageCount > 0) &&
1137                                                 (messageCount.get() >= maxMessageCount)) {
1138                                             closeError.set(true);

1140                                             setStopFlag();
