230     private void processBatch(Document document, Map<String, Object> record) {
231         if (isFirstDocument) {
232             int columnCount = record.size();
233             if (columnCount == 0) {
234                 return;
235             }
236             initWorksheetV4();
237 
238             keys = record.keySet();
239             if (writeHeaderRow) {
240                 updatePayloadSize(keys);
241                 buffer.add(new ArrayList<>(keys));
242             }
243             isFirstDocument = false;
244         }
245         RowValues rowValues = getCellValues(record);
246             // The purpose of the following test is to make sure the payload would not exceed
247             // the 10MB limit. If the average size of cell values are large, MAX_PAYLOAD_BYTES will
248             // kick in first while, if cell values are small, rowCountPerBatch will be used to
249             // determine when the batch should be sent.
250         if (payloadBytes > MAX_PAYLOAD_BYTES || buffer.size() >= DEFAULT_ROWCOUNT_PER_BATCH) {
251             writeBatch();
252             payloadBytes = rowValues.getSize();
253         }
254         buffer.add(rowValues.getValues());
255         document.acknowledge();
256     }

345     private Spreadsheet deleteSheet(int sheetId) {
346         DeleteSheetRequest deleteSheetRequest = new DeleteSheetRequest().setSheetId(sheetId);
347         Request request = new Request().setDeleteSheet(deleteSheetRequest);
348         BatchUpdateSpreadsheetRequest requestBody = new BatchUpdateSpreadsheetRequest()
349                 .setRequests(ImmutableList.of(request));
350         requestBody.setIncludeSpreadsheetInResponse(true);
351         try {
352             return Failsafe.with(getRetryPolicy(numOfRetries, retryInterval))
353                     .onFailedAttempt(e -> LOG.warn("Failed to delete worksheet: {} in spreadsheet" +
354                     ": {} for overwriting, {}", worksheetName, spreadsheetNameOrId, e.getMessage()))
355                     .get(context -> {
356                         logRetryAttempt(context, LOG);
357                         return deleteSheet(sheetId, requestBody);
358                     });
359         } catch (Throwable t) {
360             Throwable t2 =  t instanceof FailsafeException ? t.getCause() : t;
361             if (t2 instanceof IOException &&
362                     StringUtils.containsIgnoreCase(t2.getMessage(), NO_SHEET)) {
363                 // Worksheet does not exists. Return quietly.
364                 LOG.warn("Failed to delete worksheet: {} in spreadsheet: {} for overwriting, {}",
365                         worksheetName, spreadsheetNameOrId, t2.getMessage());
366                 return null;
367             }
368             throw new SnapDataException(t2, ERR_DELETE_WORKSHEET)
369                     .formatWith(worksheetName, spreadsheetNameOrId)
370                     .withResolution(RESOLUTION_ADDRESS_ISSUE);
371         }
372     }

374     private Spreadsheet deleteSheet(int sheetId, BatchUpdateSpreadsheetRequest requestBody)
375             throws IOException {
376         BatchUpdateSpreadsheetResponse response = sheets.spreadsheets()
377                 .batchUpdate(spreadsheetId, requestBody)
378                 .execute();
379         Spreadsheet updatedSpreadsheet = response.getUpdatedSpreadsheet();
380         if (updatedSpreadsheet == null) {
381             throw new IOException(String.format(ERR_DELETE_WORKSHEET_NO_ERROR, worksheetName));
382         }
383         // Check if successfully deleted since intermittently the delete request completes without
384         // error, but the sheet still exists.
385         for (Sheet sheet : updatedSpreadsheet.getSheets()) {

387                 // throw exception so that delete can be retried

389             }
390         }
391         return updatedSpreadsheet;

395      * Writes a batch of rows into a worksheet,
396      */
397     protected void writeBatch() {
398         try {
399             ValueRange valueRange = new ValueRange().setValues(buffer)
400                     .setMajorDimension(ROWS);
401 
402             if (writeMode.equals(WRITE_MODE_OPTION_CREATE) || StringUtils.isBlank(cellReference)) {
403                 if (isFirstBatchWritten) {
404                     writeRange = String.format(WRITE_RANGE, worksheetName, validRecordCount + 1,
405                             validRecordCount + buffer.size());
406                 } else {
407                     writeRange = worksheetName;
408                     isFirstBatchWritten = true;
409                 }
410             } else {
411                 int endRow = 0;
412 
413                 if (!isFirstBatchWritten) {
414                     // For the first batch, calculate column range and appropriate row values from
415                     // cell ref start point (cell ref serves as top left cell of written data)
416                     Pattern a1CellGroupsPattern = Pattern.compile(PATTERN_A1_CELL_GROUPS);
417                     Matcher matcher = a1CellGroupsPattern.matcher(cellReference);
418                     matcher.find();
419                     String rangeStartColumnCode = matcher.group(1).toUpperCase();
420 
421                     columnWriteRange = computeColumnRange(rangeStartColumnCode,
422                             buffer.get(0).size());
423 
424                     matcher.find();
425                     int startRowNum = Integer.parseInt(matcher.group(2));
426                     lastRowWritten = startRowNum - 1;
427                     // end row # = cell ref row # + # of rows in buffer - 1
428                     // (-1 because start row is inclusive)
429                     endRow = startRowNum + buffer.size() - 1;
430 
431                     isFirstBatchWritten = true;
432                 } else {
433                     endRow = lastRowWritten + buffer.size();
434                 }
435 
436                 writeRange = String.format(columnWriteRange, worksheetName,
437                         lastRowWritten + 1, endRow);
438 
439                 lastRowWritten = endRow;
440             }
441 
442             if (StringUtils.isBlank(cellReference) || writeMode.equals(WRITE_MODE_OPTION_CREATE)) {
443                 // Default #append behavior is to automatically add data following the last row of
444                 // content of the last logical table in the sheet
445                 // (see https://developers.google.com/sheets/api/guides/values#appending_values)
446                 AppendValuesResponse resp = sheets.spreadsheets().values()
447                         .append(spreadsheetId, writeRange, valueRange)
448                         .setValueInputOption(valueInputOption)
449                         .execute();
450                 validRecordCount += resp.getUpdates().getUpdatedRows();
451             } else {
452                 UpdateValuesResponse resp = sheets.spreadsheets().values()
453                         .update(spreadsheetId, writeRange, valueRange)
454                         .setValueInputOption(valueInputOption)
455                         .execute();
456                 validRecordCount += resp.getUpdatedRows();
457             }
458         } catch (SocketTimeoutException e) {
459             // Intermittently, socket timeout may occur even if data is written successfully.
460             LOG.warn("Timed out while a batch of data is written to the worksheet '{}' " +
461                     "in the spreadsheet '{}'", worksheetName, spreadsheetNameOrId, e);
462         } catch (IOException|SnapDataException e) {
463             // there is no per-row error list from server
464             // all data in the buffer should be routed to the error view
465             int failedRecords = buffer.size();
466             invalidRecordCount += failedRecords;
467             SnapDataException ex;
468             if (e instanceof SnapDataException) {
469                 ex = (SnapDataException) e;
470             } else {
471                 ex = new SnapDataException(e, ERR_BATCH_WRITE)
472                         .formatWith(failedRecords, worksheetName)
473                         .withReason(e.getMessage())
474                         .withResolution(RESOLUTION_ADDRESS_ISSUE);
475             }
476             boolean isHeader = writeHeaderRow;
477             for (List<Object> row : buffer) {
478                 if (isHeader) {
479                     isHeader = false;
480                     continue;
481                 }
482                 Iterator<Object> it = row.iterator();
483                 Map<String, Object> data = new LinkedHashMap<>(keys.size());
484                 for (String key : keys) {
485                     data.put(key, it.hasNext() ? it.next() : StringUtils.EMPTY);
486                 }
487                 errorViews.write(ex, data);
488             }
489         } finally {
490             buffer.clear();
491         }
492     }

620     /**
621      * Computes the column range for a range of data of a specified width, starting from a specified
622      * column, and creates a range-format string partially completed with the column range.
623      *
624      * @param startColumnCode the alphabetical code for the column at which the computed range
625      *                        should start, inclusive
626      * @param writeRangeWidth the number of columns the computed range should span, inclusive of
627      *                        the start column
628      * @return a range-format String with the start and end column of the computed range filled in
629      */
630     protected String computeColumnRange(String startColumnCode, int writeRangeWidth) {
631         // convert start col code to start column #
632         // For column code with max three letters:
633         // [(26^2) * charVal][26 * charVal][charVal]
634         int startColumnNum = 0;
635         for (int i = 0, power = startColumnCode.length() - 1; i < startColumnCode.length(); i++, power--) {
636             int charVal = ((int) startColumnCode.charAt(i)) - 'A' + 1;
637             int colVal = (int) (Math.pow(26, power) * charVal);
638             startColumnNum += colVal;
639         }
640 
641         // end col # = start col # + row size - 1 (-1 because start col is inclusive)
642         int endColumnNum = startColumnNum + writeRangeWidth - 1;
643         if (endColumnNum > GOOGLE_SHEETS_MAX_COLUMN_COUNT) {
644             throw new SnapDataException(String.format(ERR_WRITE_RANGE_INVALID, writeRangeWidth, cellReference))
645                     .withReason(ERR_WRITE_RANGE_INVALID_MAX_COLUMN_REASON)
646                     .withResolution(ERR_WRITE_RANGE_INVALID_MAX_COLUMN_RESOLUTION);
647         }
648 
649         // convert end col # to end col code
650         // Since letters of the start column were converted to a 1-indexed numeric value,
651         // subtract 1 from the end column to shift to 0-indexing for each letter calculation
652         StringBuilder endColumnCodeBuilder = new StringBuilder();
653         while (endColumnNum-- > 26) {
654             int columnCharOffset = endColumnNum % 26;
655             endColumnCodeBuilder.insert(0, (char) (columnCharOffset + 'A'));
656             endColumnNum = endColumnNum / 26;
657         }
658         endColumnCodeBuilder.insert(0, (char) (endColumnNum + 'A'));
659         String endColumnCode = String.valueOf(endColumnCodeBuilder);
660 
661         return String.format("%%s!%s%%d:%s%%d", startColumnCode, endColumnCode);
662     }

666         // get the spreadsheet and worksheet names
667         try {
668             evalSheetNames(null);

677             String spreadsheetId = getSpreadsheetId(googleDrive, spreadsheetNameOrId);
678             values = readWorksheet(spreadsheetId, worksheetName);
679         } catch (Throwable t) {
680             LOG.warn("Spreadsheet/Worksheet does not exist, or cannot be read", t);
681             return;
682         }
683 
684         if (CollectionUtils.isEmpty(values)) {
685              // There is no data in the sheet.
686              return;
687         }
688 
689         List<RowData> allRows = values.get(0).getRowData();
690         if (allRows == null) {
691             // Probably empty worksheet
692             return;
693         }
694

696         List<CellData> headerColumns = headerRow.getValues();
697         if (headerColumns == null) {
698             LOG.warn("Worksheet header row is empty");
699             return;
700         }
701         List<String> header = new ArrayList<>(headerColumns.size());
702 
703         // Get the header columns
704         for (CellData headerColumn : headerColumns) {
705             String headerName = headerColumn.getFormattedValue();
706             if (StringUtils.isBlank(headerName)) {
707                 // Empty header name
708                 return;
709             }
710             if (header.contains(headerName)) {
711                 // Duplicate header names
712                 return;
713             }
714             header.add(headerName);
715         }
716 
717         if (CollectionUtils.isEmpty(header)) {
718             // Empty header
719             return;
720         }
721 
722         // Get the first data row
723         Map<Object, Object> firstDataRow;
724         try {
725             firstDataRow = getRecord(allRows.get(1).getValues(), header);
726         } catch (Throwable t) {
727             LOG.warn("Error while reading the first data row from the worksheet", t);
728             return;
729         }
730         // Build schema
731         for (String inputViewName : provider.getRegisteredViewNames()) {
732             SchemaBuilder schemaBuilder = provider.getSchemaBuilder(inputViewName);
733             schemaBuilder.type(SnapType.COMPOSITE);
734             for (String headerCol : header) {

736                         headerCol);

738             }
