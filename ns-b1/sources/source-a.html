131     private void createTableSchema(List<AttributeDefinition> attributeDefinitions) {
132         tableSchema = new LinkedHashMap<>();
133         columnSchema = new LinkedHashMap<>();
134         for (AttributeDefinition entry : attributeDefinitions) {
135             Map<String, Object> columnTypeMap = new LinkedHashMap<>();
136             String attrName = entry.getAttributeName();
137             String attrType = entry.getAttributeType();
138             switch (attrType) {
139                 case BOOL:
140                     columnTypeMap.put(attrType, false);
141                     break;
142                 case NUMBER:
143                     columnTypeMap.put(attrType, BigDecimal.ZERO);
144                     break;
145                 default:
146                     columnTypeMap.put(attrType, StringUtils.EMPTY);
147                     break;
148             }
149             columnSchema.put(attrName, columnTypeMap);
150         }
151         Map<String, Object> itemMap = new LinkedHashMap<>();
152         itemMap.put(ITEM, columnSchema);
153         tableSchema.put(PUT_REQUEST, itemMap);
154     }

505             DocumentMapData mapData = schemaData.toDocumentSchemaData();
506             Schema schema = mapData.getSchema();

511     /**
512      * Evaluates expression attributes given by the user
513      */
514     protected void evaluateExprAttrParams(final Document document) {
515         if (exprAttrParams != null) {
516             nameMap = new NameMap();
517             valueMap = new ValueMap();
518             Object attrValue;
519             for (Map<String, ExpressionProperty> exprDef : exprAttrParams) {
520                 ExpressionProperty attrNameProp = exprDef.get(EXPR_ATTR_NAME_PROP);
521                 String attrName = attrNameProp.eval(document);
522                 ExpressionProperty attrTypeProp = exprDef.get(EXPR_ATTR_TYPE_PROP);
523                 String attrType = attrTypeProp.eval(document);
524                 ExpressionProperty attrValueProp = exprDef.get(EXPR_ATTR_VALUE_PROP);
525                 attrValue = attrValueProp.eval(document);
526                 if (StringUtils.startsWith(attrName, POUND)) {
527                     nameMap.with(attrName, (String) attrValue);
528                 } else {
529                     Object attributeValue = convertValue(attrType, attrValue);
530                     valueMap.with(attrName, attributeValue);
531                 }
532             }
533         }
534     }
