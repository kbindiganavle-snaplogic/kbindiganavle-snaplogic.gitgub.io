71     // 'create or replace table "' + (_catalog == '' ? '' : _catalog + '"."') +
72     // _schema + '"."' + _table + '" (\n' + jsonPath($, '$srcDef[*].columnDefinition').join('\n,') +
73     // (jsonPath($, "$srcDef[?(@.isPrimaryKeyColumn)].column") != [] ? '\n, primary key
74     // (AUTOSYNC_PRIMARYKEY, AUTOSYNC_EFFECTIVEBEGINTIME)' : '') + '\n) comment=\'Auto-generated
75     // by SnapLogic pipeline ' + pipe.projectPath + '/' + pipe.label + ' at ' +

77     @Override
78     public String buildCreateTableSql(String projectId, String catalog, String schema, String table,
79                                       List<Map<String, Object>> columns) {
80         StringBuilder sb = new StringBuilder("create or replace table \"");
81         sb.append(getQualifiedTableName(SNOWFLAKE, null, catalog, schema, table));
82         sb.append(" (").append(NEWLINE);
83 
84         sb.append(columns.stream()
85                 .map((column) -> (String)(column.get("columnDefinition")))
86                 .peek(def -> {
87                     if (StringUtils.isBlank(def)) {
88                         throw new SnapDataException("columnDefinition can not be null or empty");
89                     }
90                 })
91                 .map(StringUtils::trim)
92                 .collect(Collectors.joining(NEWLINE + ",")));
93 
94         boolean hasPrimaryKey = columns.stream()
95                 .map((column) -> (Boolean)(column.get("isPrimaryKeyColumn")))
96                 // We are checking nullity here to prevent NPE
97                 // when the "isPrimaryKeyColumn" value of any column entry is null.
98                 .filter((isPrimaryKeyColumn) -> isPrimaryKeyColumn != null &&
99                         (boolean) isPrimaryKeyColumn)
100                 .findAny()
101                 .orElse(Boolean.FALSE);
102         if (hasPrimaryKey) {
103             sb.append(NEWLINE)
104                     .append(
105                             ", primary key (AUTOSYNC_PRIMARYKEY, AUTOSYNC_EFFECTIVEBEGINTIME)");
106         }
107         sb.append(NEWLINE).append(")");
108 
109         return sb.toString();
110     }
