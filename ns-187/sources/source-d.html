154             Map<Object, Object> output = new HashMap();
155             for (Map.Entry<Writable, Writable> entry : input.entrySet()) {
156                 Writable key = entry.getKey();
157                 Writable value = entry.getValue();
158                 Converter keyConverter = this.parent.getConverterFor(key);
159                 Converter valueConverter = this.parent.getConverterFor(value);
160                 output.put(keyConverter.convert(key), valueConverter.convert(value));

163         }

200             List<Writable> writables = new ArrayList<>(output.size());
201             Class<? extends Writable> valueClass = null;
202             for (Object object : output) {
203                 Converter converter = this.parent.getReverseConverterFor(object);
204                 Writable reverseValue = converter.reverse(object);

206                 if (valueClass == null) {

209             }
210             return new ArrayWritable(valueClass, writables.toArray(new Writable[0]));
211         }
