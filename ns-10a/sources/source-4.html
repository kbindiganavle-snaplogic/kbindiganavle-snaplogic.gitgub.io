264     /**
265      * Create H2O Frame from list of input documents.
266      */
267     public void createFrame() {
268 
269         Map<String, Object> data = dataset.getData();
270 
271         List<Vec> vecs = new ArrayList<>();
272         vecGroup = new Vec.VectorGroup();
273 
274         String[] fieldNames = dataset.getFields().toArray(new String[fields.size()]);
275         for (String field : fieldNames) {
276             // Field type is derived from the first input document.
277             String fieldType = dataset.getType(field) == DataFieldType.NUMERIC ?
278                     NUMERIC : CATEGORICAL;
279             Vec v;
280 
281             // Numeric field.
282             if (fieldType.equals(NUMERIC)) {
283                 double[] vals = Arrays.copyOf((double[]) data.get(field), dataset.getSize());
284                 if (field.equals(labelFieldName)) {
285                     labelList = vals;
286                 }
287                 v = Vec.makeVec(vals, vecGroup.addVec());
288                 vecs.add(v);
289             }
290 
291             // Categorical field.
292             else {
293                 String[] vals = Arrays.copyOf((String[]) data.get(field), dataset.getSize());
294                 if (field.equals(labelFieldName)) {
295                     labelList = vals;
296                 }
297                 v = Vec.makeVec(vals, vecGroup.addVec());
298                 // In H2O, String and categorical are different.
299                 vecs.add(v.toCategoricalVec());
300                 v.remove();
301             }
302 
303             // The following code is from H2O. This key is created during the process and store
304             // as private member which we cannot access. So, we need to manually generate the
305             // same key here to be used during the cleaning process.
306             if (espcKey == null) {
307                 byte[] bits = v._key._kb.clone();
308                 bits[0] = 6;
309                 UnsafeUtils.set4(bits, 2, -1);
310                 UnsafeUtils.set4(bits, 6, -2);
311                 espcKey = Key.make(bits);
312             }
313         }
314 
315         Vec[] vecArr = vecs.toArray(new Vec[vecs.size()]);
316 
317         // Construct frame.
318         Key<Frame> frameKey = Key.make(uniqueId + "_frame");
319         dataframe = new Frame(frameKey, fieldNames, vecArr);
320 
321         // Add to DKV which is distributed key-value storage. H2O stores everything in DKV which
322         // is global and accessible by all H2O jobs.
323         DKV.put(frameKey, dataframe);
324 
325         dataset.clear();
326     }

329      * Start AutoML.
330      */
331     public void build() throws InterruptedException, IOException {
332         // Lock to prevent other AutoML Snaps to do full clean up.
333         lock.readLock().lock();
334         try {
335 
336             // Have a model in the second input view.
337             if (bestModelMap != null) {
338                 try {
339                     Map<String, Object> metadataMap = (Map<String, Object>)
340                             Serializer.deserialize(String.valueOf(bestModelMap.get(MODEL_METADATA)));
341 
342                     // Use seed from the second input if have one.
343                     if (metadataMap.containsKey(MODEL_SUMMARY_SEED)) {
344                         seed = Long.parseLong(String.valueOf(
345                                 metadataMap.get(MODEL_SUMMARY_SEED)));
346                     }
347 
348                     // Restore model parameters from the best model.
349                     String paramsString = String.valueOf(metadataMap.get(MODEL_BEST));
350                     byte[] modelBytesArr = Base64.getDecoder().decode(paramsString);
351                     FileUtils.writeByteArrayToFile(new File(modelTmpPath), modelBytesArr);
352                     Model bestModel = Model.importBinaryModel(modelTmpPath);
353                     bestParams = bestModel._parms;
354 
355                     bestModelTrainMetrics = (Map<String, Object>) metadataMap.get(MODEL_TRAIN_METRICS);
356                     bestModelCVMetrics = (Map<String, Object>) metadataMap.get(MODEL_CV_METRICS);
357 
358                     // Set fold.
359                     fold = bestParams._nfolds;
360                 } catch (ClassNotFoundException e) {
361                     throw new SnapDataException(ERR_LOAD_BEST_MODEL)
362                             .withReason(ERR_LOAD_BEST_MODEL_REASON)
363                             .withResolution(ERR_LOAD_BEST_MODEL_RESO);
364                 }
365             }
366 
367             // Do not run AutoML (rebuild the model in the second input view only) if there is a
368             // model in the second input view and the model limit is set to 1.
369             needAutoML = !(bestModelMap != null && modelLimit == 1);
370 
371             // Generate header (field names and types).
372             for (String field : dataset.getData().keySet()) {
373                 String fieldType = dataset.getType(field) == DataFieldType.NUMERIC ?
374                         NUMERIC : CATEGORICAL;
375                 fields.put(field, fieldType);
376             }
377 
378             // Create data frame.
379             createFrame();
380 
381             // Create spec.
382             AutoMLBuildSpec autoMLBuildSpec = new AutoMLBuildSpec();
383             autoMLBuildSpec.input_spec.training_frame = dataframe._key;
384             autoMLBuildSpec.input_spec.response_column = labelFieldName;
385 
386             // Set time limit.
387             if (!needAutoML) {
388                 timeLimit = 1;
389             } else if(timeLimit == 0) {
390                 timeLimit = Integer.MAX_VALUE;
391             }
392             autoMLBuildSpec.build_control.stopping_criteria.set_max_runtime_secs(timeLimit);
393 
394             autoMLBuildSpec.build_models.exclude_algos = AlgorithmGroup.getH2OExcludedAlgorithm(algorithmGroupSet);
395 
396             // Set maximum number of models to be built. Building model based on best parameters
397             // is count as 1 in the model limit.
398             if (bestParams != null) {
399                 modelLimit--;
400             }
401             autoMLBuildSpec.build_control.stopping_criteria.set_max_models(modelLimit);
402 
403             // Set random seed.
404             if (applySeed) {
405                 autoMLBuildSpec.build_control.stopping_criteria.set_seed(seed);
406             }
407 
408             autoMLBuildSpec.build_control.keep_cross_validation_models = false;
409             autoMLBuildSpec.build_control.keep_cross_validation_predictions = false;
410             autoMLBuildSpec.build_control.keep_cross_validation_fold_assignment = false;
411             autoMLBuildSpec.build_control.nfolds = fold;
412             autoMLBuildSpec.build_control.project_name = uniqueId + "_project";
413 
414             // H2O uses start time as part of the keys of models and other objects. This needs to be
415             // unique for each AutoML job.
416             Date startTime;
417             synchronized (H2O.STORE) {
418                 startTime = new Date();
419                 String timeStamp = startTimeStamp = timestampFormatForKeys.format(startTime);
420                 while (timestampFormatForKeys.format(startTime).equals(timeStamp)) {
421                     Thread.sleep(1100);
422                     startTime = new Date();
423                     startTimeStamp = timestampFormatForKeys.format(startTime);
424                 }
425             }
426 
427             log.info("H2O: Start AutoML at " + startTimeStamp);
428             this.startTime = new DateTime();
429             task = AutoML.makeAutoML(Key.make(autoMLBuildSpec.build_control.project_name),
430                     startTime, autoMLBuildSpec);
431             DKV.put(task);
432 
433             // Start the job.
434             AutoML.startAutoML(task);
435 
436             // If the pipeline is manually stopped, we force the job to stop.
437             if (stopped.get()) {
438                 task.stop();
439                 this.endTime = new DateTime();
440                 return;
441             }
442 
443             // Probe the status of the job every 3 seconds.
444             while (true) {
445                 if (task.job().isStopped()) {
446                     this.endTime = new DateTime();
447                     break;
448                 } else {
449                     Thread.sleep(3000);
450                 }
451             }
452             log.info("H2O: AutoML " + startTimeStamp + " is completed");
453 
454             if (bestParams != null) {
455                 bestParams.setTrain(dataframe._key);
456                 bestModel = ModelBuilder.trainModelNested(task.job(),
457                         Key.make(task._key.toString() + BEST_MODEL_SUFFIX), bestParams, null);
458                 this.endTime = new DateTime();
459                 Map<String, Object> trainMetrics = getMetrics(bestModel, 1, true, false, true);
460                 boolean identical = true;
461                 for (Map.Entry<String, Object> entry : trainMetrics.entrySet()) {
462                     String key = entry.getKey();
463                     double value = Double.parseDouble(String.valueOf(entry.getValue()));
464                     double oldBestValue =
465                             Double.parseDouble(String.valueOf(bestModelTrainMetrics.get(key)));
466                     if (Math.abs(value - oldBestValue) > EPSILON) {
467                         identical = false;
468                         break;
469                     }
470                 }
471                 bestIdentical = identical;
472             }
473 
474             // Serialize the best model.
475             modelMap = serialize();
476             leaderboard = computeLeaderboard();
477             report = computeReport(MAX_NO_OF_REPORT);
478 
479         } finally {
480             lock.readLock().unlock();
481         }
482     }

487     private double compareModel(Map<String, Object> aMetric, Map<String, Object> bMetric) {
488         boolean moreBetter;
489         double aVal, bVal;
490         if (aMetric.containsKey(AUC)) {
491             aVal = ((BigDecimal) aMetric.get(AUC)).doubleValue();
492             bVal = ((BigDecimal) bMetric.get(AUC)).doubleValue();
493             moreBetter = true;
494         } else if (aMetric.containsKey(DEVIANCE)) {
495             aVal = ((BigDecimal) aMetric.get(DEVIANCE)).doubleValue();
496             bVal = ((BigDecimal) bMetric.get(DEVIANCE)).doubleValue();
497             moreBetter = false;
498         } else {
499             aVal = ((BigDecimal) aMetric.get(MEAN_PER_CLASS_ERROR)).doubleValue();
500             bVal = ((BigDecimal) bMetric.get(MEAN_PER_CLASS_ERROR)).doubleValue();
501             moreBetter = false;
502         }
503         if (moreBetter) {
504             return aVal - bVal;
505         } else {
506             return bVal - aVal;
507         }
508     }

511      * Serialize metadata and model so that we can write to the Snap's output view.
512      */
513     public Map<String, Object> serialize() throws IOException {
514         Model leader = needAutoML ? task.leader() : null;
515 
516         // Leader is the best model, it will be null if the time limit is not enough to build a
517         // model.
518         if (needAutoML && leader == null) {
519             throw new SnapDataException(ERR_BUILD_MODEL)
520                     .withReason(ERR_NOT_ENOUGH_TIME_REASON)
521                     .withResolution(ERR_NOT_ENOUGH_TIME_RESO);
522         }
523 
524         if (!needAutoML) {
525             leader = bestModel;
526             stillBest = true;
527         } else if (bestModel != null) {
528             double diff = compareModel(
529                     bestIdentical ? bestModelCVMetrics : getMetrics(bestModel, 0, true, false, false),
530                     getMetrics(leader, 0, true, false, false));
531             if (diff >= 0) {
532                 leader = bestModel;
533                 stillBest = true;
534             }
535         }
536 
537         ByteArrayOutputStream modelStream = new ByteArrayOutputStream();
538         leader.getMojo().writeTo(modelStream);
539         Base64.Encoder enc = Base64.getEncoder();
540         byte[] encoded = enc.encode(modelStream.toByteArray());
541         String modelStr = new String(encoded, StandardCharsets.UTF_8);
542 
543         Map<String, Object> metadata = new LinkedHashMap<>();
544         metadata.put(MODEL_MODEL_TYPE, dataset.getDatasetType().name().toLowerCase());
545         metadata.put(MODEL_ENGINE, AutoMLEngine.H2O_ENGINE.getDisplayName());
546         metadata.put(MODEL_FIELDS, fields);
547         metadata.put(MODEL_LABEL_FIELD, labelFieldName);
548 
549         leader.exportBinaryModel(modelTmpPath, true);
550         File modelTmpFile = new File(modelTmpPath);
551         FileInputStream modelTmpFileStream = new FileInputStream(modelTmpFile);
552         byte modelBytesArr[] = new byte[(int) modelTmpFile.length()];
553         modelTmpFileStream.read(modelBytesArr);
554         modelTmpFile.delete();
555         String modelBase64 = Base64.getEncoder().encodeToString(modelBytesArr);
556 
557         metadata.put(MODEL_BEST, modelBase64);
558         metadata.put(MODEL_PARAMS_CLASS, leader._parms.getClass().getName());
559         metadata.put(MODEL_TRAIN_METRICS, getMetrics(leader, 0, true, false, true));
560         if (bestIdentical && stillBest) {
561             metadata.put(MODEL_CV_METRICS, bestModelCVMetrics);
562         } else {
563             metadata.put(MODEL_CV_METRICS, getMetrics(leader, 0, false, true, false));
564         }
565 
566         Map<String, Object> summary = new LinkedHashMap<>();
567         summary.put(MODEL_SUMMARY_ENGINE, AutoMLEngine.H2O_ENGINE.getDisplayName());
568         summary.put(MODEL_SUMMARY_MODEL_TYPE, dataset.getDatasetType().name().toLowerCase());
569         summary.put(MODEL_SUMMARY_ALGORITHM, leader._parms.fullName());
570         summary.put(MODEL_SUMMARY_START_TIME, this.startTime);
571         summary.put(MODEL_SUMMARY_END_TIME, this.endTime);
572         summary.put(MODEL_SUMMARY_DURATION,
573                 BigInteger.valueOf((this.endTime.getMillis() - this.startTime.getMillis()) / 1000));
574 
575         Map<String, Object> metrics;
576         if (bestIdentical && stillBest) {
577             metrics = this.getStatOnly(bestModelCVMetrics);
578         } else {
579             metrics = getMetrics(
580                     leader, 0, true, false, false);
581         }
582 
583         // Remove log loss and deviance.
584         if (metrics.containsKey(MULTICNOMIAL_CLS_METRICS_LOG_LOSS)) {
585             metrics.remove(MULTICNOMIAL_CLS_METRICS_LOG_LOSS);
586         }
587         if (metrics.containsKey(REGRESSION_METRICS_DEVIANCE)) {
588             metrics.remove(REGRESSION_METRICS_DEVIANCE);
589         }
590 
591         summary.put(MODEL_SUMMARY_STATISTICS, metrics);
592         if (this.applySeed) {
593             if (bestParams == null) {
594                 summary.put(MODEL_SUMMARY_SEED, BigInteger.valueOf(this.seed));
595             } else {
596                 summary.put(MODEL_SUMMARY_SEED, BigInteger.valueOf(this.bestParams._seed));
597             }
598         }
599 
600         Map<String, Object> modelMap = new LinkedHashMap<>();
601         modelMap.put(MODEL_SUMMARY_SUMMARY, summary);
602         if (isReadable) {
603             modelMap.put(MODEL_READABLE, leader.toString());
604         }
605         modelMap.put(MODEL_METADATA, Serializer.serialize(new LinkedHashMap<>(metadata)));
606         modelMap.put(MODEL_MODEL, modelStr);
607 
608         return modelMap;
609     }

631     /**
632      * Sort leaderboard.
633      */
634     private void sortLeaderboard(List<Map<String, Object>> leaderboard) {
635         Collections.sort(leaderboard, new Comparator<Map<String, Object>>() {
636             @Override
637             public int compare(Map<String, Object> o1, Map<String, Object> o2) {
638                 double diff = -compareModel(o1, o2);
639                 return diff > 0 ? 1 : (diff < 0 ? -1 : 0);
640             }
641         });
642         if (leaderboard.get(0).containsKey(RANK)) {
643             for (int i = 0; i < leaderboard.size(); i++) {
644                 leaderboard.get(i).put(RANK, BigInteger.valueOf(i + 1));
645             }
646         }
647         // Remove log loss and deviance.
648         for (int i = 0; i < leaderboard.size(); i++) {
649             if (leaderboard.get(i).containsKey(REGRESSION_METRICS_DEVIANCE)) {
650                 leaderboard.get(i).remove(REGRESSION_METRICS_DEVIANCE);
651             }
652             if (leaderboard.get(i).containsKey(MULTICNOMIAL_CLS_METRICS_LOG_LOSS)) {
653                 leaderboard.get(i).remove(MULTICNOMIAL_CLS_METRICS_LOG_LOSS);
654             }
655         }

793             log.error("H2O: Failed to get leaderboard.", e);
794             return new LinkedHashMap<>();
795         }

798     /**
799      * Get cross validation metrics of model.
800      */
801     private Map<String, Object> getMetrics(Model model, int rank, boolean statOnly,
802                                            boolean confusionMatrix, boolean trainMetric) {
803         Map<String, Object> modelMap = new LinkedHashMap<>();
804         if (!statOnly) {
805             modelMap.put(RANK, BigInteger.valueOf(rank));
806             modelMap.put(ALGORITHM, model._parms.fullName());
807         }
808         ModelMetrics metrics;
809         if (trainMetric) {
810             metrics = model._output._training_metrics;
811         } else {
812             metrics = model._output._cross_validation_metrics;
813         }
814 
815         // Binary Classification
816         if (model._output.isBinomialClassifier()) {
817             ModelMetricsBinomial metricsBinomial = (ModelMetricsBinomial) metrics;
818             modelMap.put(BINARY_CLS_METRICS_ACCURACY,
819                     ReportUtil.createBigDecimal(metricsBinomial.cm().accuracy(), 0));
820             modelMap.put(BINARY_CLS_METRICS_AUC, BigDecimal.valueOf(metricsBinomial.auc()));
821             modelMap.put(BINARY_CLS_METRICS_F1,
822                     ReportUtil.createBigDecimal(metricsBinomial.cm().f1(), 0));
823             modelMap.put(BINARY_CLS_METRICS_PRECISION,
824                     ReportUtil.createBigDecimal(metricsBinomial.cm().precision(), 0));
825             modelMap.put(BINARY_CLS_METRICS_RECALL,
826                     ReportUtil.createBigDecimal(metricsBinomial.cm().recall(), 0));
827             modelMap.put(BINARY_CLS_METRICS_LOG_LOSS, BigDecimal.valueOf(metricsBinomial.logloss()));
828             modelMap.put(BINARY_CLS_METRICS_AVG_CLASS_ERROR, BigDecimal.valueOf(metricsBinomial.mean_per_class_error()));
829             modelMap.put(BINARY_CLS_METRICS_RMSE,
830                     BigDecimal.valueOf(metricsBinomial.rmse()));
831             modelMap.put(BINARY_CLS_METRICS_MSE, BigDecimal.valueOf(metricsBinomial.mse()));
832         }
833 
834         // Multiclass Classification
835         else if (model._output.isMultinomialClassifier()) {
836             ModelMetricsMultinomial metricsMultinomial = (ModelMetricsMultinomial) metrics;
837 
838             ConfusionMatrix cm = metricsMultinomial.cm();
839             double[][] cmTable = cm._cm;
840 
841             double[] tp = new double[cmTable.length];
842             double[] tn = new double[cmTable.length];
843             double[] fp = new double[cmTable.length];
844             double[] fn = new double[cmTable.length];
845             double[] precision = new double[cmTable.length];
846             double[] recall = new double[cmTable.length];
847             double[] f1 = new double[cmTable.length];
848 
849             double sumtp = 0, sumfp = 0, sumfn = 0;
850 
851             double microPrecision = 0, microRecall = 0, microF1 = 0;
852             double macroPrecision = 0, macroRecall = 0, macroF1 = 0;
853 
854             for (int i = 0; i < cmTable.length; i++) {
855                 for (int j = 0; j < cmTable.length; j++) {
856                     for (int k = 0; k < cmTable.length; k++) {
857                         boolean trueVal = i == j;
858                         boolean predVal = i == k;
859                         if (trueVal && predVal) {
860                             tp[i] += cmTable[j][k];
861                         } else if (trueVal && !predVal) {
862                             fn[i] += cmTable[j][k];
863                         } else if (!trueVal && predVal) {
864                             fp[i] += cmTable[j][k];
865                         } else {
866                             tn[i] += cmTable[j][k];
867                         }
868                     }
869                 }
870             }
871 
872             for (int i = 0; i < f1.length; i++) {
873                 precision[i] = tp[i] != 0 ? tp[i] / (tp[i] + fp[i]) : 0;
874                 recall[i] = tp[i] != 0 ? tp[i] / (tp[i] + fn[i]) : 0;
875                 f1[i] = (precision[i] != 0 || recall[i] != 0) ?
876                         2 * precision[i] * recall[i] / (precision[i] + recall[i])
877                         : 0;
878                 macroPrecision += precision[i];
879                 macroRecall += recall[i];
880                 sumtp += tp[i];
881                 sumfp += fp[i];
882                 sumfn += fn[i];
883             }
884 
885             macroPrecision /= precision.length;
886             macroRecall /= recall.length;
887             macroF1 = (macroPrecision != 0 || macroRecall != 0) ?
888                     2 * macroPrecision * macroRecall / (macroPrecision + macroRecall) : 0;
889 
890             microPrecision = sumtp != 0 ? sumtp / (sumtp + sumfp) : 0;
891             microRecall = sumtp != 0 ? sumtp / (sumtp + sumfn) : 0;
892             microF1 = (microPrecision != 0 || microRecall != 0) ?
893                     2 * microPrecision * microRecall / (microPrecision + microRecall) : 0;
894 
895             modelMap.put(MULTICNOMIAL_CLS_METRICS_ACCURACY,
896                     ReportUtil.createBigDecimal(metricsMultinomial.cm().accuracy(), 0));
897             modelMap.put(MULTICNOMIAL_CLS_METRICS_AVG_CLASS_ERROR,
898                     BigDecimal.valueOf(metricsMultinomial.mean_per_class_error()));
899             modelMap.put(MULTICNOMIAL_CLS_METRICS_F1_MICRO,
900                     ReportUtil.createBigDecimal(microF1, 0));
901             modelMap.put(MULTICNOMIAL_CLS_METRICS_F1_MACRO,
902                     ReportUtil.createBigDecimal(macroF1, 0));
903             modelMap.put(MULTICNOMIAL_CLS_METRICS_PRECISION_MACRO,
904                     ReportUtil.createBigDecimal(macroPrecision, 0));
905             modelMap.put(MULTICNOMIAL_CLS_METRICS_RECALL_MACRO,
906                     ReportUtil.createBigDecimal(macroRecall, 0));
907             modelMap.put(MULTICNOMIAL_CLS_METRICS_LOG_LOSS,
908                     BigDecimal.valueOf(metricsMultinomial.logloss()));
909             modelMap.put(MULTICNOMIAL_CLS_METRICS_RMSE,
910                     BigDecimal.valueOf(metricsMultinomial.rmse()));
911             modelMap.put(MULTICNOMIAL_CLS_METRICS_MSE,
912                     BigDecimal.valueOf(metricsMultinomial.mse()));
913 
914         }
915 
916         // Regression
917         else {
918             ModelMetricsRegression metricsRegression = (ModelMetricsRegression) metrics;
919             modelMap.put(REGRESSION_METRICS_DEVIANCE,
920                     BigDecimal.valueOf(metricsRegression.mean_residual_deviance()));
921             modelMap.put(REGRESSION_METRICS_RMSE,
922                     BigDecimal.valueOf(metricsRegression.rmse()));
923             modelMap.put(REGRESSION_METRICS_MSE,
924                     BigDecimal.valueOf(metricsRegression.mse()));
925             modelMap.put(REGRESSION_METRICS_MAE,
926                     BigDecimal.valueOf(metricsRegression.mae()));
927 
928             modelMap.put(REGRESSION_METRICS_RMSLE,
929                     ReportUtil.createBigDecimal(metricsRegression.rmsle(), Double.MAX_VALUE));
930         }
931 
932         if (model._output.isClassifier() && confusionMatrix) {
933             modelMap.put(CLASSES,
934                     Arrays.asList(metrics.cm()._domain));
935             List<List<BigDecimal>> cm = new ArrayList<>();
936             double[][] cmTable = metrics.cm()._cm;
937             for (int i = 0; i < cmTable.length; i++) {
938                 List<BigDecimal> row = new ArrayList<>();
939                 for (int j = 0; j < cmTable[i].length; j++) {
940                     row.add(BigDecimal.valueOf(cmTable[i][j]));
941                 }
942                 cm.add(row);
943             }
944             modelMap.put(CONFUSION_MATRIX, cm);
945         }
946 
947         return modelMap;
948     }

982     /**
983      * Clean up.
984      * If there is no other Snaps using H2O (we can acquire write lock), delete all values in DKV.
985      * If there is at least one Snap using H2O, we will try to clean things as much as we can.
986      */
987     public void clean() {
988         synchronized (H2O.STORE) {
989 
990             // If we can acquire write lock, we can simply wipe everything in DKV.
991             if (lock.writeLock().tryLock()) {
992                 try {
993                     // Wipe everything in DKV.
994                     H2O.raw_clear();
995                     // Invoke gabage collector.
996                     H2O.gc();
997                     log.info("H2O: Clean everything.");
998                 } finally {
999                     lock.writeLock().unlock();
1000                 }
1001             }
1002 
1003             // If we cannot acquire write lock, we will try to clean things up as much as we can.
1004             else {
1005                 try {
1006                     if (task == null) {
1007                         return;
1008                     }
1009 
1010                     Set<Model> modelSet = new HashSet<>();
1011 
1012                     // Try to get all models from the leaderboard.
1013                     Leaderboard leaderboard = task.leaderboard();
1014                     if (task.leader() != null && leaderboard != null) {
1015                         TwoDimTable twoDimTable = null;
1016                         try {
1017                             twoDimTable = leaderboard.toTwoDimTable();
1018                         } catch (Exception e) {
1019                             log.error("H2O: Failed to get leaderboard table.");
1020                         }
1021                         int numModel = twoDimTable.getRowDim();
1022                         for (int i = 0; i < numModel; i++) {
1023                             Key modelKey = Key.make((String) twoDimTable.get(i, 0));
1024                             Model model = DKV.get(modelKey).get();
1025                             modelSet.add(model);
1026                         }
1027                     }
1028 
1029                     // Remove grid. Grid is used during hyperparameter tuning.
1030                     for (Key key : H2O.localKeySet()) {
1031                         if (DKV.get(key) != null && (DKV.get(key).get() instanceof Grid)) {
1032                             Grid grid = DKV.get(key).get();
1033                             if (grid.getModelCount() > 0 && modelSet.contains(grid.getModels()[0])) {
1034                                 key.remove();
1035                             }
1036                         }
