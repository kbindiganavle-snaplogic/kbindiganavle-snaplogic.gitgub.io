278     /**
279      * Try to build the configuration for the extended SQS client. The configuration should
280      * include an S3 client and a dedicated S3 bucket. The bucket would be used to store messages
281      * that is larger than 256K
282      *
283      * @param sqsCredential The credentials for SQS access.
284      * @return The configuration for the extended SQS client
285      */
286     public ExtendedClientConfiguration getExtendedClientConfiguration(AWSCredentialsProvider sqsCredential) {
287         if (!supportLargeMessage) {
288             return null;
289         }
290         AWSCredentialsProvider credentialsProvider;
291         if (StringUtils.isEmpty(s3Region)) {
292             Region region = Regions.getCurrentRegion();
293             s3Region = region == null ? Regions.DEFAULT_REGION.getName() : region.getName();
294         }
295         String apnUserAgentHeader = SQSUtils.getApnUserAgentHeader();
296         if (USE_SQS_CREDENTIALS.equals(s3CredentialType)) {
297             credentialsProvider = sqsCredential;
298         } else {
299             if (s3UseEC2Role) {
300                 credentialsProvider = new InstanceProfileCredentialsProvider(false);
301             } else {
302                 if (StringUtils.isBlank(s3AccessKeyID) || StringUtils.isBlank(s3SecretKey)) {
303                     throw new ConfigurationException(ERROR_ACCOUNT_CREDENTIALS_MISSING)
304                             .withReason(REASON_ACCOUNT_CREDENTIALS_MISSING)
305                             .withResolution(RESOLUTION_ADDRESS_ISSUE);
306                 }
307                 AWSCredentials credentials =
308                         StringUtils.isEmpty(s3SecurityToken) ?
309                                 new BasicAWSCredentials(s3AccessKeyID, s3SecretKey) :
310                                 new BasicSessionCredentials(s3AccessKeyID, s3SecretKey,
311                                         s3SecurityToken);
312                 credentialsProvider =
313                         new AWSStaticCredentialsProvider(credentials);
314             }
315             if (StringUtils.isNotEmpty(s3AssumeRoleArn)) {
316                 ClientConfiguration configuration = new ClientConfiguration();
317                 configuration.setUserAgentPrefix(apnUserAgentHeader);
318                 AWSSecurityTokenService sts = AWSSecurityTokenServiceClientBuilder
319                         .standard()
320                         .withRegion(s3Region)
321                         .withCredentials(credentialsProvider)
322                         .withClientConfiguration(configuration)
323                         .build();
324                 String sessionName = String.format(ASSUME_ROLE_SESSION_NAME_FMT, this.hashCode(),
325                         System.nanoTime());
326                 AssumeRoleRequest request = new AssumeRoleRequest()
327                         .withRoleArn(s3AssumeRoleArn)
328                         .withRoleSessionName(sessionName);
329                 if (StringUtils.isNotEmpty(s3AssumeRoleExtID)) {
330                     request.setExternalId(s3AssumeRoleExtID);
331                 }
332                 AssumeRoleResult result = sts.assumeRole(request);
333                 credentialsProvider = new AWSStaticCredentialsProvider(
334                         new BasicSessionCredentials(result.getCredentials().getAccessKeyId(),
335                                 result.getCredentials().getSecretAccessKey(),
336                                 result.getCredentials().getSessionToken()));
337             }
338         }
339         // Wrap the client up with the aws sqs extended library.
340         // The new client can handle large payload up to 2GB,
341         // but it lost the asynchronous interfaces
342         ClientConfiguration s3Config = new ClientConfiguration();
343         s3Config.setUserAgentPrefix(apnUserAgentHeader);
344         AmazonS3 s3Client =
345                 AmazonS3ClientBuilder.standard()
346                         .withCredentials(credentialsProvider)
347                         .withRegion(s3Region)
348                         .withClientConfiguration(s3Config)
349                         .build();
350         return new ExtendedClientConfiguration()
351                 .withPayloadSupportEnabled(s3Client, s3Bucket);
352     }

354     @Override
355     public AWSCredentialsProvider connect() throws ExecutionException {
356         AWSCredentialsProvider awsCredentialsProvider;
357         String accessKeyId = accessKeyIdExpr.eval(null);
358         String secretKey = secretKeyExpr.eval(null);
359         securityToken = securityTokenExpr.eval(null);
360         if (isIAMRole) {
361             //  From: http://docs.aws.amazon.com/sdk-for-java/v1/developer-guide/java-dg-roles.html
362             //  the IAM Role specific provider is InstanceProfileCredentialProvider, which would
363             //  get credentials from the EC2 instance
364             awsCredentialsProvider = new InstanceProfileCredentialsProvider(false);
365         } else if (StringUtils.isNotEmpty(securityToken)) {
366             AWSCredentials credentials = new BasicSessionCredentials(accessKeyId, secretKey,
367                     securityToken);
368             awsCredentialsProvider = new AWSStaticCredentialsProvider(credentials);
369         } else {
370             if (StringUtils.isBlank(accessKeyId) || StringUtils.isBlank(secretKey)) {
371                 throw new ConfigurationException(ERROR_ACCOUNT_CREDENTIALS_MISSING)
372                         .withReason(REASON_ACCOUNT_CREDENTIALS_MISSING)
373                         .withResolution(RESOLUTION_ADDRESS_ISSUE);
374             }
375             AWSCredentials credentials = new BasicAWSCredentials(accessKeyId, secretKey);
376             awsCredentialsProvider = new AWSStaticCredentialsProvider(credentials);
377         }
378         if (StringUtils.isNotBlank(arnCrossAccountIamRole)) {
379             Region defaultRegion = Regions.getCurrentRegion();
380             if (StringUtils.isBlank(region)) {
381                 region = defaultRegion == null ?
382                         Regions.DEFAULT_REGION.getName() :
383                         defaultRegion.getName();
384             }
385             ClientConfiguration config = new ClientConfiguration();
386             config.setUserAgentPrefix(SQSUtils.getApnUserAgentHeader());
387             AWSSecurityTokenService sts =
388                     AWSSecurityTokenServiceClientBuilder
389                             .standard()
390                             .withRegion(region)
391                             .withCredentials(awsCredentialsProvider)
392                             .withClientConfiguration(config)
393                             .build();
394             String sessionName = String.format(ASSUME_ROLE_SESSION_NAME_FMT, this.hashCode(),
395                     System.nanoTime());
396             AssumeRoleRequest assumeRequest = new AssumeRoleRequest()
397                     .withRoleArn(arnCrossAccountIamRole)
398                     .withRoleSessionName(sessionName);
399             if (StringUtils.isNotEmpty(externalID)) {
400                 assumeRequest = assumeRequest.withExternalId(externalID);
401             }
402             AssumeRoleResult assumeRoleResult = sts.assumeRole(assumeRequest);
403             Credentials credentials = assumeRoleResult.getCredentials();
404             return new AWSStaticCredentialsProvider(new BasicSessionCredentials(
405                     credentials.getAccessKeyId(), credentials.getSecretAccessKey(),
406                     credentials.getSessionToken()));
407         } else {
408             return awsCredentialsProvider;
409         }
410     }
