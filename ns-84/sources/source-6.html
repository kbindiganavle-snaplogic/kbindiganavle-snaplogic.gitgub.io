264     protected void process(final Document header, final ReadableByteChannel readChannel) {
265         try (InputStream is = Channels.newInputStream(readChannel)) {
266             delaySecond = getIntValue(delaySecondExpr.eval(header), DEFAULT_DELAY_SECONDS);
267             messageGroupId = messageGroupIdExpr.eval(header);
268 
269             // Only evaluate the queue properties for each document if batching is disabled.
270             // If client side buffering is enabled it means batching is disabled.
271             if (clientSideBuffering) {
272                 region = regionExpr.eval(header);
273                 queueName = queueNameExpr.eval(header);
274                 queueUrl = queueUrlExpr.eval(header);
275 
276                 isFIFO = isFifo();
277                 if (isFIFO) {
278                     throw new SnapDataException(ERR_BUFFER_NOT_SUPPORT_FIFO)
279                             .withReason(ERR_BUFFER_NOT_SUPPORT_FIFO_REASON)
280                             .withResolution(ERR_BUFFER_NOT_SUPPORT_FIFO_RESOLUTION);
281                 }
282             }
283 
284             if (isFIFO && StringUtils.isBlank(messageGroupId)) {
285                 // If FIFO queue, then the GroupID is required
286                 if (StringUtils.isBlank(messageGroupId)) {
287                     throw new SnapDataException(ERROR_MISSING_GROUP_ID)
288                             .withReason(ERROR_MISSING_GROUP_ID_REASON)
289                             .withResolution(RESOLUTION_ADDRESS_ISSUE);
290                 }
291             }
292             deduplicateId = deduplicateIdExpr.eval(header);
293             queueUrl = updateQueueUrl(region, queueName, queueUrl);
294             Map<String, MessageAttributeValue> attributes = Maps.newHashMap();
295             attributeExpressions.keySet().forEach(key -> {
296                 String name = key.eval(header);
297                 Object value = attributeExpressions.get(key).eval(header);
298                 MessageAttributeValue attrValue = new MessageAttributeValue();
299                 if (value instanceof Number) {
300                     attrValue.setDataType(BASIC_ATTR_TYPE_NUMBER);
301                 } else {
302                     attrValue.setDataType(BASIC_ATTR_TYPE_STRING);
303                 }
304                 attrValue.setStringValue(value != null ? value.toString() : null);
305                 attributes.put(name, attrValue);
306             });
307 
308 
309             String message = IOUtils.toString(is, Charsets.UTF_8);
310 
311             // If it is using clientSideBuffering, we wouldn't use the batching process, since
312             // the bufferClient is only buffering requests for non-batched operations.
313             if (clientSideBuffering) {
314                 sendMessage(queueUrl, message, attributes, header);
315             } else {
316                 sendMessageBatch(queueUrl, message, attributes, header);
317             }
318         } catch (SnapDataException e) {
319             errorViews.write(e, header);
320         } catch (java.util.concurrent.ExecutionException | InterruptedException | IOException |
321                 AmazonServiceException e) {
322             SnapDataException dataException = new SnapDataException(e, ERR_FAIL_TO_PROCESS_MESSAGE)
323                     .withReason(e.getMessage())
324                     .withResolution(RESOLUTION_ADDRESS_ISSUE);
325             errorViews.write(dataException, header);
326         }
327     }

432             MessageAttributeValue> attributes, Document originalHeader) throws java.util.concurrent.ExecutionException, InterruptedException {
433         SendMessageBatchRequestEntry requestEntry = new SendMessageBatchRequestEntry()
434                 .withMessageAttributes(attributes)
435                 .withMessageBody(messageBody);
436         if (delaySecond != null && delaySecond > 0) {
437             requestEntry.withDelaySeconds(delaySecond.intValue());
438         }
439         if (isFIFO) {
440             requestEntry.withMessageGroupId(messageGroupId);
441             if (StringUtils.isNotBlank(deduplicateId)) {
442                 requestEntry.withMessageDeduplicationId(deduplicateId);
443             }
444         }
445         // The ID of the message is the index in the buffer
446         String batchID = String.format(BATCH_ID_FMT, batchCount, messageBuffer.size());
447         requestEntry.setId(batchID);
448         messageBuffer.add(requestEntry);
449         originalHeaderBuffer.put(batchID, originalHeader);
450         if (messageBuffer.size() == batchSize) {
451             flushMessageBufferFor(queueUrl, Maps.newHashMap(originalHeaderBuffer));
452             messageBuffer.clear();
453             originalHeaderBuffer.clear();
454         }

459      * Flush the message buffer for the queue
460      *
461      * @param queueUrl  the target queue the buffer is flushing to
462      * @param originals the original headers for each of the request in the batch
463      */
464     private void flushMessageBufferFor(String queueUrl, Map<String, Document> originals) throws java.util.concurrent.ExecutionException, InterruptedException {
465         AmazonSQS client = getSqsClient();
466         SendMessageBatchRequest request = new SendMessageBatchRequest()
467                 .withEntries(Lists.newArrayList(messageBuffer))
468                 .withQueueUrl(queueUrl);
469         String requestID = UUID.randomUUID().toString();
470         inFlightRequestIDs.add(requestID);
471         if(client instanceof AmazonSQSAsync) {
472             Future<SendMessageBatchResult> futureResult =
473                     ((AmazonSQSAsync) client).sendMessageBatchAsync(request,
474                     new AsyncHandler<SendMessageBatchRequest, SendMessageBatchResult>() {
475                         @Override
476                         public void onError(final Exception exception) {
477                             synchronized (errorViews) {
478                                 inFlightRequestIDs.remove(requestID);
479                                 SnapDataException dataException = new SnapDataException(exception,
480                                         ERR_FAIL_TO_PROCESS_MESSAGE)
481                                         .withReason(exception.getMessage())
482                                         .withResolution(RESOLUTION_ADDRESS_ISSUE);
483                                 errorViews.write(dataException);
484                             }
485 
486                         }
487 
488                         @Override
489                         public void onSuccess(final SendMessageBatchRequest request,
490                                               final SendMessageBatchResult sendMessageBatchResult) {
491                             List<SendMessageBatchResultEntry> sucessfulList =
492                                     sendMessageBatchResult.getSuccessful();
493                             List<BatchResultErrorEntry> failedList = sendMessageBatchResult.getFailed();
494                             failedList.forEach(batchResultErrorEntry ->
495                                     handleFailureMessage(batchResultErrorEntry, request, originals));
496                             sucessfulList.forEach(sendMessageBatchResultEntry ->
497                                     handleSuccessfulMessage(sendMessageBatchResultEntry, originals));
498                             inFlightRequestIDs.remove(requestID);
499                         }
500                     });
501             if (isFIFO) {
502                 //For FIFO we need to guarantee the order of the message, so we need to do it in
503                 // synchronise
504                 futureResult.get();
505             }
506         } else {
507             try {
508                 SendMessageBatchResult result = client.sendMessageBatch(request);
509                 List<SendMessageBatchResultEntry> successfulList =
510                         result.getSuccessful();
511                 List<BatchResultErrorEntry> failedList = result.getFailed();
512                 failedList.forEach(batchResultErrorEntry ->
513                         handleFailureMessage(batchResultErrorEntry, request, originals));
514                 successfulList.forEach(sendMessageBatchResultEntry ->
515                         handleSuccessfulMessage(sendMessageBatchResultEntry, originals));
516             } catch (Exception exception) {
517                 SnapDataException dataException = new SnapDataException(exception,
518                         ERR_FAIL_TO_PROCESS_MESSAGE)
519                         .withReason(exception.getMessage())
520                         .withResolution(RESOLUTION_ADDRESS_ISSUE);
521                 errorViews.write(dataException);
522             } finally {
523                 inFlightRequestIDs.remove(requestID);
524             }
525         }
526         ++batchCount;
527     }
