615                     //            basically, once the handler for a protocol is registered,

676                                 .withResolution(snapException.getResolution());

681                             MuteExpectedStackTraces.unmute();
682                         }
683                     }
684                 } catch (AssertionError ae) {
685                     throw ae.initCause(ex);
686                 }
687             }
688         } else if (ex != null) {
689             throw ex;
690         }
691

693         final List<AssertionError> errors = new ArrayList<>();

695                 testResult.getOutputViews(), mapper, expectedOutputMapper);
696         checkExpectedOutput(testFixture.expectedOutputPath(), outputViewResult, OUT_SUFFIX, errors);
697         final OutputResult errorViewResult = OutputResult.buildWithOutput(
698                 testResult.getErrorViews(), mapper, expectedOutputMapper);

700         final OutputResult binaryOutputResult = OutputResult.buildWithBinary(
701                 testResult.getBinaryOutputViews(), mapper, expectedOutputMapper);

703                 errors);
704         final OutputResult binaryErrorResult = OutputResult.buildWithBinary(
705                 testResult.getBinaryErrorViews(), mapper, expectedOutputMapper);
706         checkExpectedOutput(testFixture.expectedErrorPath(), binaryErrorResult, ERR_BIN_SUFFIX,
707                 errors);
708 
709         checkExpectedSchema(testResult, errors);
710 
711         // report all errors.
712         if (!errors.isEmpty()) {
713             // @formatter:off
714             String expected = (errors.stream()
715                     .map(e -> (ComparisonFailure) e)
716                     .map(ComparisonFailure::getExpected)
717                     .collect(Collectors.joining("\n---\n")));
718             String actual = (errors.stream()
719                     .map(e -> (ComparisonFailure) e)
720                     .map(ComparisonFailure::getActual)
721                     .collect(Collectors.joining("\n---\n")));
722             // @formatter:on
723 
724             throw new ComparisonFailure("", expected, actual);
725         }
726 
727         for (InputView inputView : inputViews) {
728             TestInputProvider inputProvider = (TestInputProvider) inputView;
729             inputProvider.verify(checkLineage);

821      */
822     private void executeTest() throws Exception {
823         log.debug("executing method {}::{} with {}={}",
824                 testClass.getJavaClass().getSimpleName(), testMethod.getName(),
825                 STRICT_MODE_KEY, strictMode);
826         int numParams = testMethod.getParameterTypes().length;
827         String suggestProperty = testFixture.suggestProperty();
828         if (StringUtils.isNotBlank(suggestProperty)) {
829             DocumentUtility documentUtility = new DocumentUtilityImpl();
830             Map<String, Object> values = testSetup.suggest(suggestProperty);
831             testSetup.getTestResult().getOutputViewByName("out").write(documentUtility
832                     .newDocument(values));

834             testSetup.createProperties();
835             callback(LifecycleEvent.CONFIGURE);
836             testSetup.configure();
837             checkLineage = enforceLineage
838                     && testFixture.preservesLineage()
839                     && testFixture.exception() == TestFixture.DefaultException.class
840                     && !reconfigDetails.singleLineage().isPresent()
841                     && !inputViews.isEmpty();
842             Stream.concat(outputViews.views().stream(), errorViews.views().stream())
843                     .forEach(view -> {
844                         TestOutputRecorder output = (TestOutputRecorder) view;
845                         output.setExpectingLineage(checkLineage);
846                         inputViews.views().forEach(inp -> ((TestInputProvider) inp)

850             testSetup.execute();
851             MuteExpectedStackTraces.mute();
852             try {
853                 snap.cleanup();
854             } finally {
855                 MuteExpectedStackTraces.unmute();
856             }
857             testSetup.propagateSchema(createTreeFromFile(testFixture.upstreamSchema()),
858                     createTreeFromFile(testFixture.downstreamSchema()));
859             if (numParams > 0) {
860                 testMethod.invoke(testInstance, testSetup.getTestResult());
861             } else {
862                 testMethod.invoke(testInstance);
863             }
864         } else if (isTestSetupParameter(testMethod)) {
865             testMethod.invoke(testInstance, testSetup);
866         } else {
867             throw new IllegalArgumentException(String.format(
868                     PARAMETER_TYPE_IS_NOT_SUPPORTED,
869                     testMethod.getParameterTypes()[0].getName()));
870         }
871     }

905                                       String inputFilePath,
906                                       Set<String> inputViewNames,
907                                       ViewPauses[] inputViewPauses) throws IOException {
908         ClassLoader classLoader = Thread.currentThread().getContextClassLoader();
909         try (InputStream inputStream = classLoader.getResourceAsStream(inputFilePath)) {
910             Map<String, Object> inputViews = mapper.readValue(inputStream, Map.class);
911 
912             if (inputViewNames == null || inputViewNames.isEmpty()) {
913                 inputViewNames = inputViews.keySet();
914             }
915 
916             for (String inputViewName : inputViewNames) {
917                 Object inputView = inputViews.get(inputViewName);
918                 ViewType viewType =
919                         capabilities.get(Capability.INPUT_VIEW_ALLOW_DOCUMENT) == Boolean.TRUE ?
920                         ViewType.DOCUMENT : ViewType.BINARY;
921                 List<Object> inputViewData;
922 
923                 if (inputView instanceof Map) {
924                     Map<String, Object> inputViewMap = (Map<String, Object>) inputView;
925                     String typeName = (String) inputViewMap.get("type");
926 
927                     if (StringUtils.isNotBlank(typeName)) {
928                         viewType = ViewType.typeFor(typeName);
929                     }
930                     inputViewData = (List<Object>) inputViewMap.get("docs");
931                 } else {
932                     inputViewData = (List<Object>) inputView;
933                 }
934 
935                 if (inputViewData == null) {
936                     throw new ExecutionException(NO_DATA_DEFINED_FOR_INPUT_VIEW,
937                             inputViewName);
938                 } else {
939                     Pause[] pauses = findPausesForView(inputViewPauses, inputViewName);
940                     switch (viewType) {
941                         case DOCUMENT:
942                             createAndAddInputProvider(inputViewName, inputViewData, pauses);
943                             break;
944                         case BINARY:
945                             createAndAddBinaryInputProvider(inputViewName, inputViewData, pauses);
946                             break;
947                     }
948                 }
949             }
950         }
951     }

971     private void createAndAddBinaryInputProvider(String viewName,
972                                                  List<?> viewData,
973                                                  Pause[] pauses) {
974         /* Each viewData element can be one of two things:
975          *   just a content filename:
976          *     "data/files/json_input1.json"
977          *   an object specifying the content filename and the header for that file:
978          *     {
979          *       "header": {
980          *         "content-type" : "application/json"
981          *       },
982          *       "content": "data/files/json_input1.json"
983          *     }
984          */
985         List<String> dataFiles = new ArrayList<>(viewData.size());
986         List<Map<String, String>> headerMaps = new ArrayList<>(viewData.size());
987         for (Object viewDatum: viewData) {
988             String dataFile;
989             Map<String, String> headerMap;
990             if (viewDatum instanceof String) {
991                 dataFile = (String) viewDatum;
992                 headerMap = new LinkedHashMap<>();
993             } else if (viewDatum instanceof Map) {
994                 Map<String, ?> datumMap = (Map<String, ?>) viewDatum;
995                 dataFile = (String) datumMap.get("content");
996                 headerMap = (Map<String, String>) datumMap.get("header");
997                 if (dataFile == null || headerMap == null || datumMap.size() != 2) {
998                     throw new IllegalArgumentException(
999                             "each array element must have keys: header, content");
1000                 }
1001             } else {
1002                 throw new IllegalArgumentException("expected a String or Map: " + viewDatum);
1003             }
1004             dataFiles.add(dataFile);
1005             headerMaps.add(headerMap);
1006         }
1007 
1008         BinaryInputProvider inputView = new BinaryInputProvider(viewName);
1009         inputView.record(dataFiles, headerMaps);
1010         inputView.initPauses(pauses);
1011         testSetup.addBinaryInputView(inputView);
1012     }
