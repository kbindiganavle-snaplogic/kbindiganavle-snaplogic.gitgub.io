212     public Map<String, Object> suggest(String propertyName) {
213         SuggestionBuilderImpl suggestionBuilder = new SuggestionBuilderImpl();
214         propertyBuilder.fromInstance(snap);
215         snap.defineProperties(propertyBuilder);
216         Suggestions suggestions = propertyBuilder.getSuggestion(propertyName);
217         Map<String, Object> data = Maps.newLinkedHashMap();
218         if (suggestions != null) {
219             suggestions.suggest(suggestionBuilder, propertyValues);
220         } else {
221             Class<? extends Suggester> cl = propertyBuilder.getSuggesterClass(propertyName);
222 
223             if (cl != null) {
224                 String names[] = propertyName.split("\\.");
225                 final SuggestionBuilder.SuggestionSetter node = suggestionBuilder.node(names[0]);
226                 for (int lpc = 1; lpc < names.length; lpc++) {
227                     node.over(names[lpc]);
228                 }
229                 if (account != null && accountPropertyValues != null) {
230                     accountPropertyValues.applyTo(account);
231                     account.configure(accountPropertyValues);
232                 }
233                 Suggester suggester = injector.getInstance(cl);
234                 testResult.getInjector().injectMembers(suggester);
235                 propertyValues.applyTo(suggester);
236                 try {
237                     suggester.configure(propertyValues);
238                     List<?> result = suggester.execute();
239                     if (result.isEmpty()) {
240                         node.suggestions("No suggestions available");
241                     } else {
242                         node.suggestions(result.toArray(new String[0]));
243                     }
244                 } catch (SnapException e) {
245                     node.suggestions(e.getMessage());
246                 } finally {
247                     suggester.cleanup();
248                 }
249             }
250         }
251         Map<String, Object> suggestion = suggestionBuilder.get();
252         JsonPath jsonPath = JsonPathImpl.compileStatic(propertyName);
253         final Object suggestedValue = jsonPath.readStatic(suggestion);
254         try {
255             jsonPath.write(data, suggestedValue);
256         } catch (InvalidPathException e) {
257             throw new RuntimeException(e);
258         }
259         return data;
260     }

332     public void propagateSchema(final ObjectNode upstreamSchemas, final ObjectNode
333             downstreamSchemas) {
334         if (!(snap instanceof SchemaPropagator)) {
335             return;
336         }
337 
338         SchemaPropagator schemaPropagator = ((SchemaPropagator) snap);
339         SchemaProvider schemaProvider = testResult.getSchemaProvider();
340 
341         schemaProvider.setViewDirection(ViewType.ViewDirection.OUTPUT);
342         for (String outputViewName : schemaProvider.getRegisteredViewNames()) {
343             for (Map.Entry<String, JsonNode> child : (Iterable<Map.Entry<String, JsonNode>>)
344                     upstreamSchemas::fields) {
345                 ObjectSchema upstreamSchema = schemaProvider.createSchema().fromTree(child
346                         .getValue());
347                 ObjectSchema outputSchema = (ObjectSchema) schemaProvider
348                         .getSchemaForViewName(outputViewName);
349 
350                 if (outputSchema == null) {
351                     outputSchema = injector.getInstance(ObjectSchemaNode.class);
352                     schemaProvider.setProvidedSchemaForViewName(outputViewName, outputSchema);
353                 }
354 
355                 schemaPropagator.propagateUpstreamOutputSchema(schemaProvider, child.getKey(),
356                         upstreamSchema, outputViewName, outputSchema);
357 
358                 // Make sure the neighbor's schema isn't modified
359                 Assert.assertEquals(schemaProvider.createSchema().fromTree(child.getValue()),
360                         upstreamSchema);
361             }
362         }
363         schemaProvider.setViewDirection(ViewType.ViewDirection.INPUT);
364         for (String inputViewName : schemaProvider.getRegisteredViewNames()) {
365             for (Map.Entry<String, JsonNode> child : (Iterable<Map.Entry<String, JsonNode>>)
366                     downstreamSchemas::fields) {
367                 ObjectSchema downstreamSchema = schemaProvider.createSchema().fromTree(child
368                         .getValue());
369                 ObjectSchema inputSchema = (ObjectSchema) schemaProvider
370                         .getSchemaForViewName(inputViewName);
371 
372                 if (inputSchema == null) {
373                     inputSchema = injector.getInstance(ObjectSchemaNode.class);
374                     schemaProvider.setProvidedSchemaForViewName(inputViewName, inputSchema);
375                 }
376                 schemaPropagator.pullDownstreamInputSchema(schemaProvider, child.getKey(),
377                         downstreamSchema, inputViewName, inputSchema);
378 
379                 // Make sure the neighbor's schema isn't modified.
380                 // NB: The downstreamSchema is regenerated since there is occasionally some
381                 // differences like requireds/titles.
382                 Assert.assertEquals(schemaProvider.createSchema().fromTree(child.getValue()),
383                         schemaProvider.createSchema().fromTree((JsonNode) downstreamSchema));
384             }
385         }
386     }

458     private void validateInputViews(final Map<Capability, Object> capabilityMap) throws Throwable {
459         if (!testResult.getInputViews().isEmpty()) {
460             int inputViewSize = testResult.getInputViews().size();
461             Integer minInputs = (Integer) capabilityMap.get(Capability.INPUT_VIEW_LOWER_LIMIT);
462             Integer maxInputs = (Integer) capabilityMap.get(Capability.INPUT_VIEW_UPPER_LIMIT);
463             if (inputViewSize < minInputs || inputViewSize > maxInputs) {
464                 throw new Throwable(SNAP_MIN_MAX_INPUT_VIEW_CAPABILITIES);
465             }
466         }
467     }

469     private void validateOutputViews(final Map<Capability, Object> capabilityMap) throws Throwable {
470         if (!testResult.getOutputViews().isEmpty() ||
471                 !testResult.getBinaryOutputViews().isEmpty()) {
472             int outputViewSize = testResult.getOutputViews().size() +
473                     testResult.getBinaryOutputViews().size();

475             Integer maxOutputs = (Integer) capabilityMap.get(Capability.OUTPUT_VIEW_UPPER_LIMIT);
476             if ((minOutputs != null && outputViewSize < minOutputs) ||
477                     (maxOutputs != null && outputViewSize > maxOutputs)) {
478                 throw new Throwable(SNAP_MIN_MAX_OUTPUT_VIEW_CAPABILITIES);
479             }
