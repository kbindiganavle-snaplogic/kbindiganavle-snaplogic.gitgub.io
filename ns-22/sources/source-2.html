237     /*
238         This will check if request referer is in list of allowed request origins
239         This is required in case if request doesnt contain origin header
240         Few examples where origin header is null is given here
241         https://wiki.mozilla.org/Security/Origin#Privacy-Sensitive_Contexts
242     */
243     private boolean checkValidReferer(String requestReferer) {
244         String requestOrigin = "";
245         try {
246             URL url = new URL(requestReferer);
247             requestOrigin = url.getProtocol() + "://" + url.getHost();
248             if (url.getPort() != -1) {
249                 requestOrigin = requestOrigin + ":" +  url.getPort();
250             }
251         } catch (MalformedURLException e) {
252             return false;
253         }
254 
255         return checkValidOrigin(requestOrigin);
256     }

318     private boolean checkCorsSafeHeaders(String requestHeader) {
319         for (String corsHeader : CORS_SAFE_HEADERS) {
320             if (requestHeader.equalsIgnoreCase(corsHeader)) {
321                 return true;
322             }

326     /*
327         This will check sequentially valid requests.
328         1. Check for valid origin in list of origins
329         2. Check for valid method in list of methods
330         3. Check if every header is in list of allowed headers
331      */
332     private boolean checkValidRequest(ContainerRequestContext requestContext){
333         if (!checkValidOrigin(requestContext.getHeaderString(ORIGIN)) &&
334                 !checkValidReferer(requestContext.getHeaderString(REFERER))) {
335             return false;
336         }
337         if (!checkValidMethod(requestContext.getMethod())) {
338             return false;
339         }
340 
341         MultivaluedMap<String, String> requestHeaders = requestContext.getHeaders();
342         for (String requestHeader: requestHeaders.keySet()) {
343             if (!checkCorsSafeHeaders(requestHeader)
344                     && !checkValidHeader(requestHeader)){
345                 return false;
346             }
347         }
348         return true;
349     }

351     /*
352         This will get the response headers for valid request.
353         This will set the following response headers
354         1. Access-Control-Allow-Origin 2. Access-Control-Allow-Methods
355         3. Access-Control-Allow-Headers 4. Access-Control-Expose-Headers
356      */
357     private void updateValidRequestResponseHeaders(
358             ContainerRequestContext requestContext, boolean options) {
359 
360         ACCESS_CONTROL_ALLOW_ORIGIN_HEADER.putInContext(requestContext,
361                 requestContext.getHeaderString(ORIGIN));
362 
363         String allowedMethods = getAllowedMethods();
364         if (!allowedMethods.isEmpty()) {
365             if (options) {
366                 ACCESS_CONTROL_ALLOW_METHODS_HEADER.putInContext(requestContext,
367                         allowedMethods);
368                 ACCESS_CONTROL_MAX_AGE_HEADER.putInContext(requestContext,
369                         getMaxAge());
370             } else {
371                 ACCESS_CONTROL_ALLOW_METHODS_HEADER.putInContext(requestContext,
372                         requestContext.getMethod());
373             }
374         }
375 
376         String allowedHeaders = getAllowedHeaders();
377         if (!allowedHeaders.isEmpty()) {
378             ACCESS_CONTROL_ALLOW_HEADERS_HEADER.putInContext(requestContext,
379                     allowedHeaders);
380         }
381 
382         String exposedHeaders = getExposedHeaders();
383         if (!exposedHeaders.isEmpty()) {
384             ACCESS_CONTROL_EXPOSE_HEADERS_HEADER.putInContext(requestContext,
385                     exposedHeaders);
386         }
387         if (accessControlAllowCredentials) {
388             ACCESS_CONTROL_ALLOW_CREDENTIALS_HEADER.putInContext(requestContext,
389                     true);
390         } else {
391             ACCESS_CONTROL_ALLOW_CREDENTIALS_HEADER.putInContext(requestContext,
392                     false);
393         }
394     }

424     /*
425         This will update response headers iff CORS policy is configured
426         1. If the request method is OPTIONS this add the appropriate response headers
427            iff the request domain is in list of allowed domains and return request with response 204
428         2. If the request method is not OPTIONS this will check if the request context is valid
429            by checking origin for the allowed origins, request headers in allowed request headers
430            request method in allowed methods
431         3. If the valid request check fails treat it as invalid request and set the default
432            response headers
433      */
434     @Override
435     public void filter(final ContainerRequestContext requestContext) throws IOException {
436         if (!allowedRequestOrigins.isEmpty()) {
437             if (HttpMethod.OPTIONS.equals(requestContext.getMethod())) {
438                 if (checkValidReferer(requestContext.getHeaderString(REFERER)) ||
439                         checkValidOrigin(requestContext.getHeaderString(ORIGIN))) {
440                     updateValidRequestResponseHeaders(requestContext, true);
441                 }
442                 requestContext.abortWith(Response.status(Response.Status.OK).build());
443                 return;
444             }
445             if (checkValidRequest(requestContext)) {
446                 updateValidRequestResponseHeaders(requestContext, false);
447             } else {
448                 updateInvalidRequestResponseHeaders(requestContext);
449             }
450         }
451     }
