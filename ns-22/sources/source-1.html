146      * Base for properties used to store the number of milliseconds until the token bucket is
147      * refilled
148      */
149     public static class RefillAfterBase implements ContextProperty<Long> {
150         @Override
151         public void addResponseHeader(final ContainerRequestContext requestContext,
152                 final MultivaluedMap<String, Object> headers) {
153             // The rate limit refill time is computed before the API is executed and, therefore,
154             // does not take into account how long it took the API to run.  This method will be
155             // called when building the response, so we can check the current time and return a
156             // meaningful refill time.
157             getFromContext(requestContext).ifPresent(refillTime -> {
158                 long now = System.currentTimeMillis();
159                 long after = 0;
160 
161                 if (now < refillTime) {
162                     after = refillTime - now;
163                 }
164                 headers.add(getHeaderName(), Math.max(1, TimeUnit.MILLISECONDS.toSeconds(after)));
165             });
166         }
167     }

240         public void configureTier(final int attempts, final long retryDelay) {
241             RateLimitQuota longTerm, peak;
242             long quotaValue = limit.eval(null);
243 
244             longTerm = RateLimitQuota.of(quotaValue).per(1, periodUnit.unit);
245             // Try to derive the peak limit from the long-term limit.  This isn't final...  I'll
246             // update later and allow it to be configured via feature flags.
247 
248             Map<String, Object> doc = ImmutableMap.of(
249                     "condition", condition.getExpression(),
250                     "periodUnit", periodUnit.toString(),
251                     "quota", quotaValue
252             );
253 
254             Map<String, Object> quotaMap = PEAK_QUOTA_EXPR.eval(doc);
255             peak = MAPPER.convertValue(quotaMap, RateLimitQuota.class);
256             policy = RateLimitPolicy.from(longTerm, peak);
257             log.debug("rate-limit policy: (condition: {}) {}", condition.getExpression(), policy);
258             PropertyValidation.forceExpression(this.condition);
259             retryPolicy = new RetryPolicy<RateLimitLocalService.AcquireResult>()
260                     .handleResultIf(result -> result.recommendation ==
261                             RateLimitLocalService.AcquireRecommendation.RETRY)
262                     .abortIf(result -> result.recommendation ==
263                             RateLimitLocalService.AcquireRecommendation.ALLOW ||
264                             result.recommendation ==
265                             RateLimitLocalService.AcquireRecommendation.BLOCK)
266                     .withMaxAttempts(attempts)
267                     .withDelay(Duration.ofMillis(retryDelay));
268         }

321     @Override
322     public void filter(final ContainerRequestContext requestContext) throws IOException {
323         RateLimitEntity entity = new RateLimitEntity();
324         Optional<SecurityContext> security =
325                 StandardContexts.SECURITY.getFromContext(requestContext);
326         entity.instanceId = instanceId;
327         entity.instanceVersion = instanceVersion;
328         // can safely call get() here since this requires a security context to be set.
329         entity.clientId = security.get().getUserPrincipal().getName();
330         // A request authenticated/authorized using a Task bearer token won't have a Principal
331         // name, so the Client IP is used instead.
332         if (StringUtils.isBlank(entity.clientId)) {
333             entity.clientId = StandardContexts.CLIENT_IP.getFromContext(requestContext).get();
334         }
335 
336         Pair<Integer, Tier> foundTier = IntStream.range(0, tiers.size())
337                 .mapToObj(index -> Pair.of(index, tiers.get(index)))
338                 .filter(pair -> ObjectType.toBoolean(pair.getValue().condition.eval(null)))
339                 .findFirst()
340                 .orElse(Pair.of(tiers.size(), defaultTier));
341         RetryPolicy<RateLimitLocalService.AcquireResult> retryPolicy =
342                 foundTier.getValue().retryPolicy;
343         RateLimitContext context = new RateLimitContext();
344         context.taskPath = StandardContexts.ASSET_PATH.getFromContext(requestContext).get();
345         context.proxy = StandardContexts.PROXY.getFromContext(requestContext).get();
346         entity.tier = foundTier.getKey();
347 
348         RateLimitLocalService.AcquireResult acquireResult = Failsafe.with(retryPolicy)
349                 .get(() -> rateLimitLocalService.acquire(context, entity,
350                         retryPolicy.getDelay().toMillis()));
351 
352         SERVICE_TIER.putInContext(requestContext, String.format("index=%s; %s", entity.tier,
353                 foundTier.getValue().policy));
354         if (acquireResult.recommendation == RateLimitLocalService.AcquireRecommendation.ALLOW) {
355             REFILL_AFTER.putInContext(requestContext, acquireResult.refillTime);
356         } else {
357             RETRY_AFTER.putInContext(requestContext, acquireResult.refillTime);
358             Response response = ResponseGenerator.toResponse(Response.Status.TOO_MANY_REQUESTS,
359                     "Too Many Requests");
360             requestContext.abortWith(response);
361         }
362     }
