120         public void checkConstraintConfig(JsonPath path) {
121             if ((pattern != null || minimumLength != null || maximumLength != null) &&
122                     (minimumValue != null || maximumValue != null)) {
123                 throw new ConfigurationException(ERR_INCOMPATIBLE_CONSTRAINTS)
124                         .formatWith(path)
125                         .withReason(REASON_INCOMPATIBLE_CONSTRAINTS)
126                         .withResolution(RESOLUTION_INCOMPATIBLE_CONSTRAINTS);
127             }
128             if (minimumLength != null && maximumLength != null && minimumLength > maximumLength) {
129                 throw new ConfigurationException(ERR_MIN_LENGTH_GREATER_THAN_MAX)
130                         .formatWith(minimumLength, maximumLength, path)
131                         .withReason(REASON_MIN_LENGTH_GREATER_THAN_MAX)
132                         .withResolution(RESOLUTION_MIN_LENGTH_GREATER_THAN_MAX);
133             }
134             if (minimumValue != null && maximumValue != null && minimumValue
135                     .compareTo(maximumValue) > 0) {
136                 throw new ConfigurationException(ERR_MIN_VALUE_GREATER_THAN_MAX)
137                         .formatWith(minimumValue, maximumValue, path)
138                         .withReason(REASON_MIN_VALUE_GREATER_THAN_MAX)
139                         .withResolution(RESOLUTION_MIN_VALUE_GREATER_THAN_MAX);
140             }
141         }

143         public void doValidation(WalkResult walkResult, Map<String, String> errors) {
144             Object value = walkResult.getValue();
145             String sourcePath = walkResult.getActualPath().toString();
146             if (type != null) {
147                 try {
148                     value = TransformUtils.validate(type, value, false);
149                 } catch (ClassCastException | NumberFormatException e) {
150                     String format = type.getJsonFormat();
151                     errors.put(sourcePath, String.format(ERR_DATA_TYPE_MISMATCH,
152                             sourcePath, StringUtils.isEmpty(format) ? type : format,
153                             value.getClass().getSimpleName(), value));
154                     return;
155                 }
156             }
157             if (pattern != null) {
158                 if (value == null || !pattern.matcher(String.valueOf(value)).find()) {
159                     errors.put(sourcePath, String.format(ERR_PATTERN_MISMATCH,
160                             sourcePath, pattern, value));
161                     return;
162                 }
163             }
164             if (minimumLength != null) {
165                 if (value == null || value.toString().length() < minimumLength) {
166                     errors.put(sourcePath, String.format(ERR_BELOW_MIN_LENGTH,
167                             sourcePath, minimumLength));
168                     return;
169                 }
170             }
171             if (maximumLength != null) {
172                 if (value != null && value.toString().length() > maximumLength) {
173                     errors.put(sourcePath, String.format(ERR_ABOVE_MAX_LENGTH,
174                             sourcePath, maximumLength));
175                     return;
176                 }
177             }
178             if (minimumValue != null) {
179                 value = ObjectType.attemptToConvertToBigDecimal(value);
180                 if (!(value instanceof BigDecimal)) {
181                     errors.put(sourcePath, String.format(ERR_CANNOT_CONVERT_TO_NUMBER, value));
182                     return;
183                 }
184                 if (minimumValue.compareTo((BigDecimal) value) > 0) {
185                     errors.put(sourcePath, String.format(ERR_BELOW_MIN_VALUE,
186                             sourcePath, minimumValue));
187                     return;
188                 }
189             }
190             if (maximumValue != null) {
191                 value = ObjectType.attemptToConvertToBigDecimal(value);
192                 if (!(value instanceof BigDecimal)) {
193                     errors.put(sourcePath, String.format(ERR_CANNOT_CONVERT_TO_NUMBER, value));
194                     return;
195                 }
196                 if (maximumValue.compareTo((BigDecimal) value) < 0) {
197                     errors.put(sourcePath, String.format(ERR_ABOVE_MAX_VALUE,
198                             sourcePath, maximumValue));
199                     return;
200                 }
201             }
202             for (JsonPath field : requiredProperties) {
203                 try {
204                     field.read(value);
205                 } catch (InvalidPathException e) {
206                     String path = walkResult.getActualPath().join(field).toString();
207                     errors.put(path, String.format(ERR_MISSING_PROPERTY, path));
208                 }
209             }
210         }

287     @Override
288     public void configure(final PropertyValues propertyValues) throws ConfigurationException {
289         List<Map<String, Object>> constraintTable = propertyValues.get(CONSTRAINT_TABLE_PATH);
290 
291         for (Map<String, Object> constraint : constraintTable) {
292             JsonPath srcPath = propertyValues.getJsonPathFor(constraint, SOURCE_PATH_PROP);
293             String constraintType = jsonPathUtil.nullableRead(CONSTRAINT_PROP_PATH, constraint);
294             String constraintValue = StringUtils.defaultString(
295                     jsonPathUtil.nullableRead(CONSTRAINT_VALUE_PROP_PATH, constraint));
296             PropertyValidator propertyValidator;
297 
298             if (REQUIRED_CONSTRAINT.equals(constraintType)) {
299                 propertyValidator = getValidatorForPath(srcPath.parent());
300             } else {
301                 propertyValidator = getValidatorForPath(srcPath);
302             }
303 
304             propertyValidator.srcPath = srcPath;
305             propertyValidator.constraintType = constraintType;
306             propertyValidator.constraintValue = constraintValue;
307 
308             switch (constraintType) {
309                 case PATTERN_CONSTRAINT:
310                     if (propertyValidator.pattern == null) {
311                         propertyValidator.type = SnapType.STRING;
312                         try {
313                             propertyValidator.pattern = Pattern.compile(constraintValue);
314                         } catch (PatternSyntaxException e) {
315                             throw new ConfigurationException(e, ERR_INVALID_PATTERN)
316                                     .formatWith(srcPath, constraintValue)
317                                     .withResolution(RESOLUTION_INVALID_PATTERN);
318                         }
319                     }
320                     break;
321                 case TYPE_CONSTRAINT:
322                     if (propertyValidator.type == null) {
323                         SnapType snapType;
324                         if (MAP.equalsIgnoreCase(constraintValue)) {
325                             snapType = SnapType.COMPOSITE;
326                         } else {
327                             snapType = SnapType.parse(constraintValue);
328                         }
329                         //https://mysnaplogic.atlassian.net/browse/SNAP-10767
330                         if (snapType == SnapType.UNKNOWN) {
331                             throw new ConfigurationException(ERR_INVALID_TYPE)
332                                     .formatWith(srcPath, constraintValue)
333                                     .withReason(REASON_INVALID_TYPE)
334                                     .withResolution(String.format(RESOLUTION_INVALID_TYPE,
335                                             SUPPORTED_TYPES));
336                         }
337                         propertyValidator.type = snapType;
338                     }
339                     break;
340                 case MINIMUM_LENGTH_CONSTRAINT:
341                     if (propertyValidator.minimumLength == null) {
342                         propertyValidator.type = SnapType.STRING;
343                         propertyValidator.minimumLength = convertToLong(srcPath, constraintValue);
344                     }
345                     break;
346                 case MAXIMUM_LENGTH_CONSTRAINT:
347                     if (propertyValidator.maximumLength == null) {
348                         propertyValidator.type = SnapType.STRING;
349                         propertyValidator.maximumLength = convertToLong(srcPath, constraintValue);
350                     }
351                     break;
352                 case MINIMUM_CONSTRAINT:
353                     if (propertyValidator.minimumValue == null) {
354                         propertyValidator.type = SnapType.NUMBER;
355                         propertyValidator.minimumValue = convertToBigDecimal(srcPath,
356                                 constraintValue);
357                     }
358                     break;
359                 case MAXIMUM_CONSTRAINT:
360                     if (propertyValidator.maximumValue == null) {
361                         propertyValidator.type = SnapType.NUMBER;
362                         propertyValidator.maximumValue = convertToBigDecimal(srcPath,
363                                 constraintValue);
364                     }
365                     break;
366                 case REQUIRED_CONSTRAINT:
367                     propertyValidator.requiredProperties.add(srcPath.basename());
368                     break;
369             }
370         }
371 
372         for (Map.Entry<JsonPath, PropertyValidator> entry : validators.entrySet()) {
373             entry.getValue().checkConstraintConfig(entry.getKey());
374         }
375         standardizeErrorOutput = Boolean.TRUE.equals(
376                 propertyValues.get(STANDARDIZE_ERROR_OUTPUT_PROP));
377         validateJsonPath = Boolean.TRUE.equals(propertyValues.get(VALIDATE_JSON_PATH_PROP));
378     }

380     @Override
381     protected void process(final Document document, final String inputViewName) {
382         Object data = document.get();
383         Map<String, String> errors = new LinkedHashMap<>();
384 
385         for (Map.Entry<JsonPath, PropertyValidator> entry : validators.entrySet()) {
386             JsonPath path = entry.getKey();
387             PropertyValidator validator = entry.getValue();
388             try {
389                 for (WalkResult walkResult : path.walk(data)) {
390                     validator.doValidation(walkResult, errors);
391                 }
392             } catch (PathNotFoundException e) {
393                 if (validateJsonPath) {
394                     errors.put(path.toString(), String.format(ERR_PATH_NOT_FOUND, e.getPath(),
395                             validator.srcPath, validator.constraintType,
396                             validator.constraintValue));
397                 }
398             } catch (InvalidPathException e) {
399                 errors.put(path.toString(), e.getMessage());
400             }
401         }
402 
403         if (MapUtils.isNotEmpty(errors)) {
404             // No resolution since this is expected.
405             errorViews.write(getException(errors), document);
406         } else {
407             outputViews.write(document);
408         }
409     }

411     private SnapDataException getException(Map<String, String> errors) {
412         SnapDataException ex;
413         // It is possible to have more than one error, but we only display the first error
414         String reason = errors.values().iterator().next();
415         if (standardizeErrorOutput) {
416             // Exception in standard format
417             ex = new SnapDataException(DATA_VALIDATION_ERROR)
418                     .withReason(reason)
419                     .withResolution(RESOLUTION_ADDRESS_ISSUE);
420         } else {
421             // Exception with an error map. We still keep this for backward compatibility (SWAT-618)
422             Map<String, Object> errorMap = Maps.newHashMapWithExpectedSize(2);
423             errorMap.put(ERROR, errors);
424             ex = new SnapDataException(documentUtility.newDocument(errorMap),
425                     DATA_VALIDATION_ERROR)
426                     .withReason(reason)
427                     .withResolution(RESOLUTION_ADDRESS_ISSUE);
428         }
429         return ex;
