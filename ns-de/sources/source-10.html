247     @Override
248     protected void process(final Document document, final String inputViewName) {
249         String taskPath = taskExpr.eval(document);
250         if (httpMethod == null) {
251             try {
252                 Map<String, Object> taskInfo = getTaskInfo(taskFacade, taskPath);
253                 if (taskInfo == null) {
254                     SnapDataException snapDataException = new SnapDataException(TASK_PATH_NOT_SET)
255                             .withReason(TASK_PATH_NOT_SET)
256                             .withResolution(TASK_PATH_NOT_SET_RESOLUTION);
257                     errorViews.write(snapDataException, document);
258                     return;
259                 }
260                 if (isLocal) {
261                     url = (String) taskInfo.get(LOCAL_RUN_URL);
262                     if (url == null) {
263                         SnapDataException snapDataException = new SnapDataException(ERROR_CONFIGURE)
264                                 .withReason(String.format(NO_LOCAL_URL, taskPath))
265                                 .withResolution(NO_LOCAL_URL_REASON);
266                         errorViews.write(snapDataException, document);
267                         return;
268                     }
269                 } else {
270                     url = (String) taskInfo.get(EXTERNAL_RUN_URL);
271                     if (url == null) {
272                         SnapDataException snapDataException = new SnapDataException(ERROR_CONFIGURE)
273                                 .withReason(String.format(NO_EXTERNAL_URL, taskPath))
274                                 .withResolutionAsDefect();
275                         errorViews.write(snapDataException, document);
276                         return;
277                     }
278                 }
279                 httpMethod = getHttpMethod(taskInfo, taskPath);
280             } catch (SnapDataException snapDataException) {
281                 errorViews.write(snapDataException, document);
282                 return;
283             } catch (SuggestViewAbortException e) {
284                 throw e;
285             } catch (Throwable e) {
286                 SnapDataException snapDataException = new SnapDataException(document,
287                         e.getMessage())
288                         .withResolution(TASK_ERR_GENERIC_RESOLUTION);
289                 errorViews.write(snapDataException, document);
290                 return;
291             }
292         }
293         if (batch != null && batch.size() == batchSize && document != null) {
294             writeBatch(document);
295             batch.clear();
296         }
297         if (batch != null && batch.size() < batchSize && document != null) {
298             lastBatchDocument = document;
299             batch.add(document.get());
300         } else {
301             processResponse(document, taskFacade.execute(resolveUrl(document), document,
302                     httpMethod));
303         }
304     }

357     /**
358      * Determines if the http method is a POST or a GET based on the link_map/snap_map
359      * information.
360      *
361      * @param taskInfo the task info
362      * @param taskPath the task path
363      *
364      * @return POST or GET
365      */
366     private String getHttpMethod(Map<String, Object> taskInfo, String taskPath) {
367         if (StringUtils.isEmpty(taskPath)) {
368             throw new ConfigurationException(TASK_PATH_NOT_SET);
369         }
370         Map<String, String> usedInputViews = Maps.newHashMap();
371         Map<String, Object> pipeInfo = getPipeInfo(taskInfo, taskPath);
372         Map<String, Map<String, Object>> linkMap = LINK_MAP.readStatic(pipeInfo, Maps.newHashMap());
373         for (Map<String, Object> linkInfo : linkMap.values()) {
374             Object destId = linkInfo.get(DST_ID);
375             Object viewId = linkInfo.get(DST_VIEW_ID);
376             if (destId != null && viewId != null) {
377                 usedInputViews.put((String) destId, (String) viewId);
378             }
379         }
380 
381         boolean hasUnlinkedInputView = false;
382         Map<String, Map<String, Object>> snapMap = SNAP_MAP.readStatic(pipeInfo);
383         for (Map.Entry<String, Map<String, Object>> snapInfo : snapMap.entrySet()) {
384             String snapId = snapInfo.getKey();
385             Map<String, Object> snapDef = snapInfo.getValue();
386             Map<String, Object> viewMap = INPUT_VIEW_MAP.readStatic(snapDef, Maps.newHashMap());
387             for (String viewId : viewMap.keySet()) {
388                 String usedViewId = usedInputViews.get(snapId);
389                 if (!viewId.equals(usedViewId)) {
390                     hasUnlinkedInputView = true;
391                 }
392             }
393         }
394 
395         return hasUnlinkedInputView ? HttpPost.METHOD_NAME : HttpGet.METHOD_NAME;
396     }

448      * Processes the response of the request. Writes the result to the output or error view
449      * depending on the status of the response.
450      *
451      * @param document as the incoming document
452      * @param response the response
453      */
454     private void processResponse(final Document document, Response response) {
455         int statusCode = response.getStatusCode();
456         try {
457             if (statusCode >= SC_OK && statusCode < SC_MULTIPLE_CHOICES) {
458                 writeToOutput(document, response);
459             } else {
460                 StringWriter writer = new StringWriter();
461                 IOUtils.copy(response.getResponseBodyAsStream(), writer);
462                 String error = writer.toString();
463                 String reason = String.format(ERR_EXECUTE_REASON,
464                         statusCode, response.getStatusText());
465                 Map<String, Object> errorMap = createErrorMap(reason, statusCode);
466                 if (error != null) {
467                     errorMap.put(ERROR_ENTITY, error);
468                 }
469                 errorMap.put(KEY_HEADERS, normalizeHeaders(response.getHeaders()));
470                 SnapDataException ex = new SnapDataException(
471                         documentUtility.newDocumentFor(document, errorMap), reason)
472                         .withReason(reason)
473                         .withResolution(ERR_EXECUTE_RESOLUTION);
474                 errorViews.write(ex);
475             }
476         } catch (SuggestViewAbortException e) {
477             throw e;
478         } catch (IOException e) {
479             String reason = String.format(IO_ERROR_EXEC, response.getStatusText(), statusCode,
480                     e.getMessage());
481             throw new ExecutionException(e, reason)
482                     .withReason(reason)
483                     .withResolution(ERR_EXECUTE_RESOLUTION);
484         }
485     }

488      * Writes the content to the output view.
489      *
490      * @param response the response
491      */
492     private void writeToOutput(Document document, Response response) throws IOException {
493         try {
494             List<String> encodings = response.getHeaders(HttpHeaders.CONTENT_ENCODING);
495             InputStream inputStream = response.getResponseBodyAsStream();
496             if (encodings != null && encodings.contains(SdkConstants.GZIP_CONTENT_ENCODING)) {
497                 inputStream = new GZIPInputStream(inputStream);
498             }
499             Object content = objectMapper.readValue(inputStream, Object.class);
500             if (content instanceof List) {
501                 for (Object item : (List) content) {
502                     outputViews.write(documentUtility.newDocumentFor(document, item));
503                 }
504             } else if (content != null) {
505                 outputViews.write(documentUtility.newDocumentFor(document, content));
506             }
507         } catch (SuggestViewAbortException e) {
508             throw e;
509         } catch (IOException e) {
510             // can't process the stream as json. currently only documents are supported as output
511             // of the task pipeline
512             throw new ExecutionException(e, PARSE_ERR_REASON)
513                     .withReason(PARSE_ERR_REASON)
514                     .withResolution(PARSE_ERR_RESOLUTION);
515         }
516     }

519      * Normalizes the response header, the fluent header values are represented as list
520      * values. We only expect one value per header.
521      *
522      * @param headers the original headers
523      *
524      * @return the normalized headers
525      */
526     private Map<String, String> normalizeHeaders(FluentCaseInsensitiveStringsMap headers) {
527         Map<String, String> normalizedHeaders = Maps.newHashMapWithExpectedSize(headers.size());
528         for (Map.Entry<String, List<String>> entry : headers.entrySet()) {
529             String key = entry.getKey();
530             List<String> values = entry.getValue();
531             String value = values == null ? null : values.iterator().next();
532             if (value != null) {
533                 normalizedHeaders.put(key, value);
534             }
535         }
536         return normalizedHeaders;
537     }
