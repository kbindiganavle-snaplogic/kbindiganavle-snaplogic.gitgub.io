493     @Override
494     public void execute() {
495         startCompletionReceiver();
496         startInputReader();
497         do {
498             try {
499                 final ExecEvent execEvent = execEvents.poll(1, TimeUnit.SECONDS);
500 
501                 if (execEvent != null) {
502                     switch (execEvent.getEventType()) {
503                         case DOCUMENT:
504                             LOG.trace("ExecEvent: DOCUMENT activityCount={}, execEvent={}", activityCount, execEvent);
505                             processSafely(execEvent);
506                             inputConsumedSemaphore.release();
507                             break;
508                         case SOURCE_IDLE:
509                             break;
510                         case PRIMARY_SOURCE_CLOSE:
511                             LOG.trace("ExecEvent: PRIMARY_SOURCE_CLOSE activityCount={}, execEvent={}", activityCount, execEvent);
512                             activityCount -= 1;
513                             if (reuse && loopDocExpression.isEmpty()) {
514                                 for (ExecutionTracker executionTracker : executions.values()) {
515                                     executionTracker.closeInputViews();
516                                 }
517                             }
518                             // catching the last element of the incomplete batch due to the end of the stream.
519                             final PipelineRun currentPipelineRun = lastBatchPipelineRun.get();
520                             if (batchSize > 1 && currentPipelineRun != null && execEvents.isEmpty()) {
521                                 if (linkedPipelineFacade == null)
522                                     throw new IllegalStateException("linkedPipelineFacade should be initialized, but isn't");
523                                 linkedPipelineFacade.getChildUnlinkedInputView(currentPipelineRun)
524                                         .ifPresent(this::flush);
525                                 lastBatchPipelineRun.set(null); // the last pipeline has been "closed"
526                             }
527 
528                             break;
529                         case SOURCE_CLOSE:
530                             LOG.trace("ExecEvent: SOURCE_CLOSE activityCount={}, execEvent={}", activityCount, execEvent);
531                             activityCount -= 1;
532                             break;
533                         case EXECUTION_FINISHED: {
534                             LOG.trace("ExecEvent: EXECUTION_FINISHED activityCount={}, execEvent={}", activityCount, execEvent);
535                             execEvent.getCompletionHandler().retryIfNeeded();
536 
537                             activityCount -= 1;
538                             break;
539                         }
540                         default:
541                             throw new IllegalStateException("Unreachable branch execEvent.type=" + execEvent.getEventType());
542                     }
543                 }
544 
545                 if ((reuse || batchSize > 1) && !loopDocExpression.isEmpty() && execEvents.isEmpty()) {
546                     for (ExecutionTracker executionTracker : executions.values()) {
547                         if (executionTracker.updateActiveCount() == 0) {
548                             executionTracker.closeInputViews();
549                         }
550                     }
551                 }
552 
553                 if ((reuse || batchSize > 1) && (executions.size() > poolLowWatermark)) {
554                     for (ExecutionTracker executionTracker : executions.values()) {
555                         if (executionTracker.expired()) {
556                             LOG.info("Reusable execution expired, closing it down: {}",
557                                     executionTracker);
558                             executionTracker.closeInputViews();
559                             // Break to limit how many executions we shut down at once.
560                             break;
561                         }
562                     }
563                 }
564             } catch (InterruptedException e) {
565                 LOG.info("Main snap loop thread interrupted", e);
566                 looping = false;
567                 break;
568             }
569 
570             // There's nothing active and there are no more events to process.
571             if (activityCount == 0 && execEvents.isEmpty()) {
572                 LOG.info("All activity completed and there are no more pending events");
573                 looping = false;
574             }
575         } while (looping && !aborted);
576 
577         if (aborted) {
578             LOG.info("Main thread aborted");
579         }
580 
581         Throwable th = exceptionQueue.poll();
582         if (th != null) {
583             LOG.error("Received an exception from a helper thread", th);
584             if (th instanceof RuntimeException) {
585                 throw (RuntimeException) th;
586             } else {
587                 throw new ExecutionException(th, "Internal error")
588                         .withResolutionAsDefect();
589             }
590         }
591     }

664     protected void process(@NotNull final ExecEvent original) {
665         documentsCounter.inc();
666         final Document doc = original.getDocumentOrHeader();
667         final long retryLimitValue = reuse || batchSize > 1 || isBinaryInputView() ?
668                 0 : retryLimit.eval(doc).orElse(0L);
669         final long retryIntervalValue = reuse || batchSize > 1 ? 0 : retryInterval.eval(doc).orElse(0L);
670         final Number poolSize = this.poolSize.eval(doc);
671 
672         if (batchCounter.get() == 0 && batchSize > 1) batchesCounter.inc();
673 
674         if (!schemaExec) createPool(poolSize);
675 
676         final String pipelinePath = pipeline.eval(doc);
677         if (linkedPipelineFacade == null) {
678             linkedPipelineFacade = new LinkedPipelineFacade(pipelinePath, projectPath, reuse, batchSize);
679         }
680         linkedPipelineFacade.tryToUpdatePipelinePath(pipelinePath, batchCounter.get());
681         linkedPipelineFacade.setSchemaExec(schemaExec);
682         linkedPipelineFacade.tryToInitEnvMap(getEnvMap(doc), batchCounter.get());
683         linkedPipelineFacade.tryToLoad(pipelineFacade, batchCounter.get());
684         linkedPipelineFacade.validateViews(doc, inputViews, outputViews, batchCounter.get());
685 
686         String label = execLabel.eval(doc).orElse(StringUtils.EMPTY);
687         PipelineExecutionBuilder builder = linkedPipelineFacade.createPipelineBuilder(label);
688         PipelineRun pipelineRun;
689 
690         evaluatedSnaplexPath = Optional.empty();
691         switch (executeOn) {
692             case LOCAL_NODE:
693                 break;
694 
695             case LOCAL_SNAPLEX:
696                 evaluatedSnaplexPath = Optional.ofNullable(localSnaplexPath);
697                 break;
698 
699             case SNAPLEX_WITH_PATH:
700                 evaluatedSnaplexPath = snaplex.eval(doc);
701                 break;
702         }
703         evaluatedSnaplexPath.ifPresent(builder::onSnaplex);
704 
705         PipeCompletionHandler completionHandler = null;
706         final boolean newPipelineStarted;
707         if (reuse) {
708             builder.isQueueable(false);
709 
710             // First, get a sorted list of existing runs so that we can find the least loaded one.
711             List<ExecutionTracker> existingRuns = new ArrayList<>();
712             boolean tryNewExecution = false;
713 
714             for (ExecutionTracker executionTracker : executions.values()) {
715                 if (executionTracker.isClosed()) {
716                     continue;
717                 }
718                 if (!evaluatedSnaplexPath.isPresent()) {
719                     executionTracker.updateActiveCount();
720                 }
721                 existingRuns.add(executionTracker);
722             }
723             // Since it's non-trivial to figure out whether a child execution is actively working
724             // on documents when it's non-local or we're not in immediate mode, we shuffle the
725             // existing runs and randomly pick one.
726             if (!runningAsUltraTask || evaluatedSnaplexPath.isPresent()) {
727                 Collections.shuffle(existingRuns);
728             } else {
729                 Collections.sort(existingRuns);
730             }
731 
732             if (existingRuns.isEmpty() || evaluatedSnaplexPath.isPresent() || !runningAsUltraTask) {
733                 // Nothing is running yet, need to start up something.
734                 tryNewExecution = true;
735             } else if (existingRuns.get(0).getLastActiveCount() > 0) {
736                 // The least-loaded run is still processing docs, so start up a new one.
737                 tryNewExecution = true;
738             }
739 
740             // Try to start up a new execution.  If we're already at our max pool size, the
741             // pool.tryAcquire() should return false and we'll just reuse an existing execution.
742             if (tryNewExecution && pool.tryAcquire()) {
743                 boolean needsImmediateMode = runningAsUltraTask;
744 
745                 if (!loopDocExpression.isEmpty()) {
746                     needsImmediateMode = true;
747                 }
748                 int childIndex = childCount;
749                 childCount += 1;
750                 completionHandler = new PipeCompletionHandler(
751                         builder.withImmediateMode(needsImmediateMode), null, linkedPipelineFacade,
752                         retryLimitValue, retryIntervalValue, childIndex, batchSize);
753                 pipelineRun = executePipeline(completionHandler, existingRuns, batchSize);
754 
755                 if (pipelineRun == null) {
756                     pipelineRun = existingRuns.get(0).useExecution();
757                     newPipelineStarted = false;
758                 } else {
759                     newPipelineStarted = true;
760                 }
761             } else {
762                 pipelineRun = existingRuns.get(0).useExecution();
763                 statusReporter.reset();
764                 newPipelineStarted = false;
765             }
766         } else if (batchSize > 1) {
767             builder.isQueueable(false);
768             List<ExecutionTracker> existingRuns = new ArrayList<>();
769 
770             for (ExecutionTracker executionTracker : executions.values()) {
771                 if (executionTracker.isClosed()) {
772                     continue;
773                 }
774                 if (!evaluatedSnaplexPath.isPresent()) {
775                     executionTracker.updateActiveCount();
776                 }
777                 existingRuns.add(executionTracker);
778             }
779             if (batchCounter.get() == 0) {
780                 // need to start new pipeline instance
781                 LOG.info("Waiting for execution slot: {}", pool.availablePermits());
782                 try {
783                     pool.acquire();
784                 } catch (InterruptedException e) {
785                     LOG.debug("Interrupted while trying to acquire pool semaphore", e);
786                     looping = false;
787                     return;
788                 }
789 
790                 final int childIndex = childCount++;
791 
792                 completionHandler = new PipeCompletionHandler(
793                         builder, null, linkedPipelineFacade,
794                         retryLimitValue, retryIntervalValue, childIndex, batchSize);
795                 pipelineRun = executePipeline(completionHandler, existingRuns, batchSize);
796                 if (pipelineRun != null) {
797                     lastBatchPipelineRun.set(pipelineRun);
798                     newPipelineStarted = true;
799                 } else {
800                     // the current batch pipeline is already flushed so no way we can reuse it.
801                     // Fail the execution.
802                     throw new SnapException("Can't start a pipeline for the new batch");
803                 }
804             } else {
805                 // keep sending to last pipeline
806                 final PipelineRun active = lastBatchPipelineRun.get();
807                 if (active == null) {
808                     throw new IllegalStateException("There is no pipeline to stream to. " +
809                             "(existingRuns=" + existingRuns + ", executions=" + executions +
810                             ", childCount=" + childCount + ")");
811                 }
812                 pipelineRun = active;
813                 newPipelineStarted = false;
814             }
815         } else {
816             builder.withTimeout(timeout.eval(doc).orElse(0L));
817             LOG.info("Waiting for execution slot: {}", pool.availablePermits());
818             try {
819                 pool.acquire();
820             } catch (InterruptedException e) {
821                 LOG.debug("Interrupted while trying to acquire pool semaphore", e);
822                 looping = false;
823                 return;
824             }
825 
826             LOG.info("Acquired execution slot: {}", pool.availablePermits());
827             int childIndex = childCount;
828             childCount += 1;
829             completionHandler = new PipeCompletionHandler(builder, doc, linkedPipelineFacade,
830                     retryLimitValue, retryIntervalValue, childIndex, batchSize);
831             pipelineRun = executePipeline(completionHandler, Collections.emptyList(), 1);
832 
833             newPipelineStarted = true;
834         }
835 
836         if (linkedPipelineFacade.needsRepeater(newPipelineStarted)) {
837             LOG.trace("linkedPipelineFacade.needsRepeater: activityCount={}", activityCount);
838             // Increase the count because we'll have a repeater active.
839             activityCount += 1;
840             Thread writerThread = spawnHelperThread("PipeExec OutputRepeater", new
841                     OutputRepeater(pipelineRun, reuse || batchSize > 1 ? null : doc, completionHandler));
842             writers.put(pipelineRun.getId(), writerThread);
843             writerThread.start();
844         }
845         if (doc != null) {
846             linkedPipelineFacade.getChildUnlinkedInputView(pipelineRun)
847                     .ifPresent(view -> writeDocumentToChild(original, doc, view, batchSize));
848         }
849     }

851     private void writeDocumentToChild(ExecEvent original, Document doc,
852                                       OutputViews outputViews, int batchSize) {
853         try {
854             switch (outputViews.get().getType()) {
855                 case BINARY: {
856                     outputViews.write(new BinaryOutput() {
857                         @Override
858                         public Document getHeader() {
859                             if (reuse || batchSize > 1) {
860                                 return doc;
861                             } else {
862                                 Object orgCopy = DeepUtils.copy(doc.get());
863                                 return documentUtility.newDocument(orgCopy);
864                             }
865                         }
866 
867                         @Override
868                         public void write(final WritableByteChannel writeChannel)
869                                 throws IOException {
870                             BinaryInput binaryInput = original.getBinaryInput();
871 
872                             try (ReadableByteChannel inChannel = binaryInput.getChannel()) {
873                                 statusReporter.notify(TRANSFERRING_BIN);
874                                 StatusReportingUtils.copy(statusReporter,
875                                         Channels.newInputStream(inChannel),
876                                         Channels.newOutputStream(writeChannel),
877                                         -1);
878                                 statusReporter.done();
879                             }
880                         }
881                     });
882                     break;
883                 }
884                 case DOCUMENT: {
885                     final Document toSend;
886 
887                     if (reuse || batchSize > 1) {
888                         toSend = doc;
889                     } else {
890                         Object orgCopy = DeepUtils.copy(doc.get());
891                         toSend = documentUtility.newDocument(orgCopy);
892                     }
893 
894                     outputViews.write(toSend);
895                     break;
896                 }
897             }
898         } finally {
899             // for standard mode flush every document, for batch mode flush only at the end
900             // of the batch, for reuse mode never flush.
901             if (!reuse && batchCounter.incrementAndGet() >= batchSize) {
902                 flush(outputViews);
903             }
904         }
905     }

1020     /**
1021      * Execute a pipeline as specified by the builder.
1022      *
1023      * @param completionHandler The handler to be invoked when the pipeline finishes.
1024      * @param existingRuns
1025      * @return
1026      */
1027     private PipelineRun executePipeline(PipeCompletionHandler completionHandler,
1028                                         final List<ExecutionTracker> existingRuns,
1029                                         int batchSize
1030     ) {
1031         final CountDownLatch pipelineInitialized = new CountDownLatch(1);
1032         PipelineExecutionBuilder builder = completionHandler.getBuilder()
1033                 .withInitializationHandler(new AbstractActivityHandler() {
1034                     @Override
1035                     public void run() {
1036                         executions.put(pipelineRun.getId(), new ExecutionTracker(pipelineRun));
1037                         pipelineInitialized.countDown();
1038                     }
1039                 })
1040                 .withCompletionHandler(completionHandler);
1041 
1042         final long overloadRetryDeadline = System.currentTimeMillis() + OVERLOAD_RETRY_TIMEOUT;
1043         final long childExecuteDeadline = System.currentTimeMillis() + CHILD_EXECUTE_RETRY_TIMEOUT;
1044 
1045         var retry = false;
1046         while (true) {
1047             try {
1048                 LOG.trace("Pipeline executed, activityCount={}, existingRuns={}, batchSize={}",
1049                         activityCount, existingRuns, batchSize);
1050                 final PipelineRun pipelineRun = retry ? builder.retry() : builder.execute();
1051 
1052                 // Increase the count to reflect the execution
1053                 activityCount += 1;
1054 
1055                 LOG.info("Executed pipeline: {}", pipelineRun.getId());
1056                 if (reuse || batchSize > 1) {
1057                     try {
1058                         // When reusing executions, we expect the 'executions' map to be updated,
1059                         // so we need to wait until the initialization handler has been run before
1060                         // we can return.
1061                         int attempts = 20;
1062                         while (!(shuttingDown || aborted) &&
1063                                 !pipelineInitialized.await(3, TimeUnit.SECONDS)) {
1064                             // to avoid hanging while awaiting for too long while the snap got
1065                             // interrupted for any reason, make attempts with small steps.
1066                             if (--attempts == 0) {
1067                                 break;
1068                             }
1069                         }
1070                         if (attempts == 0) {
1071                             throw new ExecutionException("The pipeline "
1072                                     + pipelineRun.getCurrentInfo().getRuuid()
1073                                     + " didn't get initialized")
1074                                     .withReason(pipelineRun.getCurrentInfo().getErrors().stream()
1075                                             .flatMap(error -> error.entrySet().stream())
1076                                             .map(e -> String.format("%s: %s", e.getKey(), e.getValue()))
1077                                             .collect(Collectors.joining("\n"))
1078                                     )
1079                                     .withResolution("Check the child pipelines for the errors, " +
1080                                             "contact support if the problem persists");
1081                         }
1082                     } catch (InterruptedException e) {
1083                         LOG.debug("Interrupted while waiting for new pipeline execution ID", e);
1084                     }
1085                 }
1086                 return pipelineRun;
1087             } catch (NoCCSnapiException e) {
1088                 LOG.warn("Plex is overloaded", e);
1089 
1090                 if ((reuse || batchSize > 1) && !existingRuns.isEmpty()) {
1091                     LOG.info("Unable to start a new execution while below pool size, need to " +
1092                             "reuse an existing execution");
1093                     statusReporter.notify(AT_CAPACITY);
1094                     pool.release();
1095                     return null;
1096                 }
1097 
1098                 if (System.currentTimeMillis() >= overloadRetryDeadline) {
1099                     pool.release();
1100                     throw new SnapDataException(e, "Unable to execute pipeline on overloaded " +
1101                             "Snaplex")
1102                             .withResolution("Add more resources to the given Snaplex");
1103                 } else {
1104                     try {
1105                         statusReporter.notify(OVERLOAD_DELAY);
1106                         Thread.sleep(OVERLOAD_RETRY_DELAY);
1107 
1108                         statusReporter.reset();
1109                     } catch (InterruptedException e1) {
1110                         throw new ExecutionException(e1, "Interrupted while attempting to execute" +
1111                                 " pipeline");
1112                     }
1113                 }
1114             } catch (RuntimeException e) {
1115                 if (e.getCause() instanceof TooManyRequestsIOException
1116                         || e.getCause() instanceof SocketTimeoutException) {
1117                     // prepare routine throws that exception is if it's still in progress, or times
1118                     // out on the first attempt, need to keep trying for a little while
1119                     if (System.currentTimeMillis() < childExecuteDeadline) {
1120                         LOG.debug("Can't execute the child pipeline, will retry in "
1121                                 + CHILD_EXECUTE_RETRY_DELAY + "ms");
1122                         try {
1123                             Date deadline = Date.from(Instant.ofEpochMilli(childExecuteDeadline));
1124                             statusReporter.notify(CHILD_EXECUTE_ATTEMPT_DELAY, deadline);
1125                             Thread.sleep(CHILD_EXECUTE_RETRY_DELAY);
1126                             statusReporter.reset();
1127                             retry = true;
1128                         } catch (InterruptedException interruptedException) {
1129                             throw new ExecutionException(interruptedException,
1130                                     "Interrupted while attempting to execute a child pipeline");
1131                         }
1132                     } else {
1133                         pool.release();
1134                         throw new ExecutionException("Can't execute the child pipeline")
1135                                 .withReason("Child pipeline prepare takes too long")
1136                                 .withResolution("Increase `com.snaplogic.snaps.flow.PipeExec." +
1137                                         "CHILD_EXECUTE_RETRY_TIMEOUT` feature flag value, " +
1138                                         "or contact support if problem persists");
1139                     }
1140                 } else {
1141                     LOG.error("Failed to execute pipeline", e);
1142                     pool.release();
