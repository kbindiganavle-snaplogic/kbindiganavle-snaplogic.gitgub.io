23  * Abstract base class for a Deserializer which will decode a String encoded with a given
24  * Charset, or throw a SerializationException if the data isn't valid for that Charset.
25  */
26 public abstract class CharsetDeserializer implements Deserializer<String> {
27 
28     private final Charset charset;
29 
30     CharsetDeserializer(Charset charset) {
31         this.charset = charset;
32     }
33 
34     @Override
35     public String deserialize(String topic, byte[] data) {
36         if (data == null) {
37             return null;
38         }
39         try {
40             CharsetDecoder decoder = charset.newDecoder();
41             ByteBuffer byteBuffer = ByteBuffer.wrap(data);
42             CharBuffer charBuffer = decoder.decode(byteBuffer);
43             return charBuffer.toString();
44         } catch (CharacterCodingException e) {
45             String msg = "Error deserializing as String using " + charset + " encoding";
46             throw new SerializationException(msg, e);
47         }
48     }
