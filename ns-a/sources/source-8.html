90     @Override
91     public ReconciliationResult processReconciliation(TableConfig tableConfig,
92                                                       EndpointService endpointService,
93                                                       Endpoint targetEndpoint,
94                                                       DetectionResult detectionResult,
95                                                       String path,
96                                                       NameHandler nameHandler) {
97         if (detectionResult == null || detectionResult.isEmpty()) {
98             LOGGER.error("PipelineSchemaReconciliator: given schema drift deteciton" +
99                     " is null or empty");
100             throw new IllegalArgumentException("Schema drift detection can not be null or empty");
101         }
102 
103         LOGGER.debug("PipelineSchemaReconciliator: tableConfig {}, detectionResult {}, path {}",
104                 prettyPrint(tableConfig), prettyPrint(detectionResult), path);
105         // We will have only one DriftUnitType and one action to execute
106         // for DriftUnitType.SCHEMA and DriftUnitType.TABLE
107         ReconciliationResult reconResult = new ReconciliationResult(tableConfig);
108         List<TargetAction> actions = new ArrayList<>();
109         List<String> sqlList = new ArrayList<>();
110         for (SourceDriftInfo driftInfo : detectionResult.getDriftInfoMap().values()) {
111             LOGGER.debug("PipelineSchemaReconciliator: begin working on driftInfo {}", driftInfo);
112 
113             TargetAction action = TargetAction.NONE;
114             List<Map<String, Object>> srcDefValuesForCurrentDrift =
115                     new ArrayList<Map<String, Object>>(driftInfo.getColumnSchemaInfoSet());
116 
117             // Find target action for the current "driftInfo"
118             if (driftInfo.getUnitType().equals(DriftUnitType.SCHEMA)) {
119                 action = TargetAction.CREATE_SCHEMA;
120             } else if (driftInfo.getUnitType().equals(DriftUnitType.TABLE)) {
121                 action = TargetAction.ADD_TABLE;
122             } else if (driftInfo.getUnitType().equals(DriftUnitType.COLUMN)) {
123                 if (driftInfo.getDrift() == ADD_COLUMN) {
124                     action = TargetAction.ADD_COLUMN;
125                 } else if (driftInfo.getDrift() == DROP_COLUMN) {
126                     // If hard drop column reconciliation is supported,
127                     // we can create a ALTER TABLE DROP COLUMN SQL here.
128                     // But soft drop column does not need any action except for
129                     // dropping NOT NULL constraint in AutoSync.
130                     LOGGER.debug("Skipping DROP COLUMN drift on: {}",
131                             StringUtils.join(driftInfo.getColumnSchemaInfoSet()));
132                     // If to-be-dropped column has NOT NULL constraint, then drop
133                     // that constraint so as for null value to be added in subsequent data
134                     // loading.
135                     Map<String, Object> columnSchema = srcDefValuesForCurrentDrift.get(0);
136 
137                     Boolean targetIsNullable = getValueAsBoolean(columnSchema
138                             .get(KEY_TARGET_IS_NULLABLE));
139                     if (Boolean.FALSE.equals(targetIsNullable)) {
140                         action = TargetAction.DROP_NOT_NULL;
141                         updateColumnSchemaForDropNotNullAction(columnSchema);
142                         LOGGER.info("PipelineSchemaReconciliator: starting action {} for " +
143                                 "taskId {}, target table {}, column {}",
144                                 action, tableConfig.getTaskId(), tableConfig.getTableName(),
145                                 columnSchema.get(KEY_COLUMN));
146                     } else {
147                         continue;
148                     }
149                 } else if (driftInfo.getDrift() == CHANGE_COLUMNN_TYPE) {
150                     action = TargetAction.CHANGE_COLUMNN_TYPE;
151                 } else if (driftInfo.getDrift() == CHANGE_COLUMNN_LENGTH) {
152                     action = TargetAction.CHANGE_COLUMNN_LENGTH;
153                 } else {
154                     throwUnsupportedColumnDriftException(driftInfo);
155                 }
156             } else {
157                 throwUnsupportedDriftUnitTypeException(driftInfo);
158             }
159 
160             // Process target action
161             processTargetAction(driftInfo,
162                     tableConfig,
163                     endpointService,
164                     targetEndpoint,
165                     action,
166                     path,
167                     srcDefValuesForCurrentDrift,
168                     sqlList,
169                     nameHandler,
170                     reconResult);
171 
172             actions.add(action);
173         } // end of for loop block
174         reconResult.setActions(actions);
175         reconResult.setReturnStatus(ReturnStatus.SUCCESS);
176         reconResult.setSqls(sqlList);
177         LOGGER.info("PipelineSchemaReconciliator: ReconciliationResult {}",
178                 prettyPrint(reconResult));
179         return reconResult;
180     }

182     private void updateColumnSchemaForDropNotNullAction(Map<String, Object> columnSchema) {
183         // Update "column" and other values at the columnSchema with target_ values
184         // to be used at target actions pipelines.
185         columnSchema.put(KEY_COLUMN, columnSchema.get(KEY_TARGET_COLUMN));
186         columnSchema.put(KEY_DATA_TYPE, columnSchema.get(KEY_TARGET_DATA_TYPE));
187         columnSchema.put(KEY_PRECISION, columnSchema.get(KEY_TARGET_PRECISION));
188 
189         // Update "columnDefinition" with "target_columnDefinition after removing "not"
190         // from "not null" for Azure Synapse and SQL Server target actions pipelines.
191         columnSchema.put(KEY_COLUMN_DEFINITION,
192                 StringUtils.replace((String)columnSchema.get(KEY_TARGET_COLUMN_DEFINITION),
193                         "not", ""));
194     }

196     private void processTargetAction(
197             SourceDriftInfo driftInfo, TableConfig tableConfig,
198             EndpointService endpointService, Endpoint targetEndpoint,
199             TargetAction action, String path,
200             List<Map<String, Object>> srcDefValuesForCurrentDrift,
201             List<String> sqlList,
202             NameHandler nameHandler, ReconciliationResult reconResult) {
203         Map<String, Object> actionResults = null;
204 
205         if (driftInfo.getUnitType() != DriftUnitType.COLUMN) {
206             // Non-COLUMN drift cases: SCHEMA, or TABLE
207             LOGGER.debug("Before executing {}: {}",
208                     action.toString(), srcDefValuesForCurrentDrift);
209 
210             actionResults = endpointService.actions(targetEndpoint, action.toString(),
211                     tableConfig, srcDefValuesForCurrentDrift,
212                     null, "target",
213                     EMPTY_STRING, path, null, action, nameHandler);
214             if (actionResults.isEmpty()) {
215                 throw new SnapDataException("No actionResults returned");
216             }
217         } else {
218             // COLUMN drift cases
219             if (driftInfo.getDrift() == CHANGE_COLUMNN_TYPE) {
220                 processActionForChangeDataTypeWithCdt(action, tableConfig, endpointService,
221                         targetEndpoint, path, srcDefValuesForCurrentDrift,
222                         nameHandler, reconResult);
223             } else if (driftInfo.getDrift() == CHANGE_COLUMNN_LENGTH) {
224                 // Build sql for all the columns with a change column length drift
225                 String sql = reconcileDatabase
226                         .buildChangeColumnSql(tableConfig.getTgtProjectId(),
227                                 tableConfig.getTgtCatalog(), tableConfig.getTgtSchema(),
228                                 tableConfig.getTargetTableName(),
229                                 srcDefValuesForCurrentDrift);
230                 LOGGER.debug("Before executing CHANGE_COLUMN_LENGTH SQL: {}", sql);
231                 if (StringUtils.isNotBlank(sql)) {
232                     actionResults = endpointService.actions(targetEndpoint, "Execute",
233                             tableConfig, null,
234                             null, "target",
235                             null, path, sql, action, nameHandler);
236                     if (actionResults.isEmpty()) {
237                         throw new SnapDataException("No actionResults returned");
238                     }
239                     LOGGER.debug("PipelineSchemaReconciliator: " +
240                                     "CHANGE_COLUMN_LENGTH action results: {}", actionResults);
241                     sqlList.add(sql);
242                 }
243             } else {
244                 // Loop over columns for the current column drift
245                 for (Map<String, Object> columnSchema : srcDefValuesForCurrentDrift) {
246                     if (MapUtils.isEmpty(columnSchema)) {
247                         throw new SnapDataException("column schema " +
248                                 "can not be null or empty");
249                     }
250                     String columnDefinition = (String) columnSchema.get("columnDefinition");
251                     LOGGER.debug("Before executing {}: {}",
252                             action.toString(), columnSchema);
253                     actionResults = endpointService.actions(targetEndpoint, action.toString(),
254                             tableConfig, srcDefValuesForCurrentDrift,
255                             columnSchema, "target",
256                             columnDefinition, path, null, action, nameHandler);
257                     if (actionResults.isEmpty()) {
258                         throw new SnapDataException("No actionResults returned");
259                     }
260                     LOGGER.debug("PipelineSchemaReconciliator: {} action results: {}", action,
261                             actionResults);
262                 }
263             }
264         }
265     }

267     private void processActionForChangeDataTypeWithCdt(
268             TargetAction action,
269             TableConfig tableConfig, EndpointService endpointService,
270             Endpoint targetEndpoint, String path,
271             List<Map<String, Object>> srcDefValuesForCurrentDrift,
272             NameHandler nameHandler,
273             ReconciliationResult reconResult) {
274         ChangeDataTypeHandler typeHandler =
275                 new CdtChangeDataTypeHandler(targetCDM);
276         LOGGER.debug("processActionForChangeDataTypeWithCdt: start for action {}," +
277                         " current drift {} ",
278                 action, prettyPrint(srcDefValuesForCurrentDrift));
279 
280         // Loop over columns for the current change data type drifts
281         for (Map<String, Object> columnSchema : srcDefValuesForCurrentDrift) {
282             // Retrieve srcDriftColumnDataType and tgtExistingColumnDatatype from
283             // "columnSchema"
284             String srcDriftColumnDataType = getValueAsString(columnSchema.get("dataType"));
285             String cdtTypeForSourceDriftType = getValueAsString(columnSchema.get("cdt"));
286             String tgtExistingColumnDatatype =
287                     getValueAsString(columnSchema.get("target_dataType"));
288             String cdtTypeForTargetExistingType = typeHandler
289                     .getCdtTypeFromTargetDataType(tgtExistingColumnDatatype);
290             BigInteger precisionForDrift =
291                     getValueAsBigInteger(columnSchema.get("precision")).orElse(null);
292             BigInteger scaleForDrift =
293                     getValueAsBigInteger(columnSchema.get("scale")).orElse(null);
294             BigInteger existingTgtPrecision =
295                     getValueAsBigInteger(columnSchema.get("target_precision")).orElse(null);
296             BigInteger existingTgtScale = getValueAsBigInteger(columnSchema.get("target_scale"))
297                     .orElse(null);
298 
299             LOGGER.debug("processActionForChangeDataTypeWithCdt: before findActionPlan" +
300                             " for {} {} -> {} {}",
301                     cdtTypeForTargetExistingType, existingTgtPrecision, cdtTypeForSourceDriftType,
302                     precisionForDrift);
303 
304             // Find Change Column Type action type
305             ChangeColumnTypeActionPlan columTypeActionPlan =
306                     typeHandler.findActionPlanWithCdt(cdtTypeForSourceDriftType,
307                             cdtTypeForTargetExistingType,
308                             srcDriftColumnDataType, tgtExistingColumnDatatype,
309                             precisionForDrift, scaleForDrift, existingTgtPrecision,
310                             existingTgtScale,
311                             isNotNullClauseRequired(columnSchema));
312             if (columTypeActionPlan.getActionType() == NOOP) {
313                 continue;
314             } else if (columTypeActionPlan.getActionType() == NOOP_SIZE_NARROWING) {
315                 // The system does not apply shrinking source column type change drifts.
316                 // @TODO enhance 4 precision/scale containing column data types as func
317                 logWarnMessage("Ignoring size narrowing source-data-type-change " +
318                         " schema drift on %s:%s from [%s(%d)] to [%s(%d)]",
319                         tableConfig, columnSchema, tgtExistingColumnDatatype, existingTgtPrecision,
320                         srcDriftColumnDataType, precisionForDrift);
321                 continue;
322             } else if (columTypeActionPlan.getActionType() == UNSUPPORTED) {
323                 // The system does not apply unsupported data type conversion by
324                 // source column type change drifts.
325                 // @TODO enhance 4 precision/scale containing column data types as func
326                 logWarnMessage("Ignoring UNSUPPORTED column data type change" +
327                                 " by source-data-type-change schema drift on %s:%s " +
328                                 "from [%s(%d)] to [%s(%d)]",
329                         tableConfig, columnSchema, tgtExistingColumnDatatype, existingTgtPrecision,
330                         srcDriftColumnDataType, precisionForDrift);
331                 continue;
332             }
333 
334             // Make columTypeActionPlan executable by setting up execution variables.
335             String columnName = getValueAsString(columnSchema.get("column"));
336             String columnDefinition = getValueAsString(columnSchema.get("columnDefinition"));
337             String tmpColumnName = columnName + "_temp" + getTimestampString();
338             String[] columnAndType = getEntriesFromColumnDefinitionString(columnDefinition);
339 
340             String conversionTargetColumnDefinition =
341                     String.format("%s %s", columnAndType[0].replace(columnName, tmpColumnName),
342                             columTypeActionPlan.getTargetFullColumnType());
343             columTypeActionPlan.setColumn(columnName);
344             columTypeActionPlan.setTmpColumn(tmpColumnName);
345             columTypeActionPlan.setTmpColumnDefinition(conversionTargetColumnDefinition);
346 
347             LOGGER.info("Before executing {}: {}", action.toString(), columTypeActionPlan);
348 
349             // Execute the columTypeActionPlan
350             executeActionsForChangeDataType(columTypeActionPlan,
351                     columnSchema,
352                     tableConfig,
353                     endpointService,
354                     targetEndpoint, path,
355                     srcDefValuesForCurrentDrift,
356                     nameHandler);
357             reconResult.getChangeColumnTypePlans().add(columTypeActionPlan.toString());
358         }
359     }

361     /**
362      * Execute the given "columnTypeActionPlan".
363      *
364      * One example is, given a Snowflake table BIGDATAQA.TEST_DATA.num2_test
365      * with NUMBER column c1,
366      * to build and return a list of 4 SQL statement strings
367      * to change the column data type from NUMBER to VARCHAR(30):
368      *
369      * 1. "ALTER TABLE \"public.num2_test\" ADD COLUMN \"c1_temp\" VARCHAR(30)"
370      * 2. "UPDATE \"public.num2_test\" SET c1_temp = CAST(\"c1\" AS VARCHAR(30))"
371      * 3. "ALTER TABLE \"num2_test\" DROP COLUMN \"c1\""
372      * 4. "ALTER TABLE \"num2_test\" RENAME COLUMN \"c1_temp\" TO \"c1\""
373      */
374     @VisibleForTesting
375     void executeActionsForChangeDataType(ChangeColumnTypeActionPlan columTypeActionPlan,
376                                        Map<String, Object> columnSchema,
377                                        TableConfig tableConfig,
378                                        EndpointService endpointService,
379                                        Endpoint targetEndpoint, String path,
380                                        List<Map<String, Object>> srcDefValuesForCurrentDrift,
381                                        NameHandler nameHandler) {
382         tableConfig.setTmpColumn(columTypeActionPlan.getTmpColumn());
383         tableConfig.setTmpColumnType(columTypeActionPlan.getTargetFullColumnType());
384 
385         // Build a column schema map for tmpColumn for the first ADD_COLUMN action.
386         // BigQuery target actions pipeline "Add Column" action expects a map for the column
387         // to add instead of columnDefinition. Thus we create a column schema map
388         // and pass to target actions pipeline along with columnDefinition for the tmpColumn.
389         Map<String, Object> tmpColumnSchema = new LinkedHashMap<String, Object>();
390         tmpColumnSchema.putAll(columnSchema);
391         tmpColumnSchema.put("column", columTypeActionPlan.getTmpColumn());
392         tmpColumnSchema.put("dataType", columTypeActionPlan.getLcaDatatype());
393         tmpColumnSchema.put("precision", columTypeActionPlan.getLcaPrecision());
394         tmpColumnSchema.put("srcPrecision", columTypeActionPlan.getLcaPrecision());
395         tmpColumnSchema.put("scale", columTypeActionPlan.getLcaScale());
396         tmpColumnSchema.put("columnDefinition", columTypeActionPlan.getTmpColumnDefinition());
397         Map<String, Object> tmpColumnSrcDefMap = new LinkedHashMap<>();
398         tmpColumnSrcDefMap.putAll(srcDefValuesForCurrentDrift.get(0));
399         tmpColumnSrcDefMap.putAll(tmpColumnSchema);
400         List<Map<String, Object>> tmpSrcDef = new ArrayList<>();
401         tmpSrcDef.add(tmpColumnSrcDefMap);
402 
403         int index = 0;
404         for (TargetAction targetAction : ChangeColumnTypeTargetActions) {
405             Map<String, Object> actionResults =
406                     endpointService.actions(targetEndpoint, targetAction.toString(),
407                     tableConfig,
408                     index == 0 ? tmpSrcDef : srcDefValuesForCurrentDrift,
409                     index == 0 ? tmpColumnSchema : columnSchema, "target",
410                     columTypeActionPlan.getTmpColumnDefinition(), path, null,
411                             targetAction, nameHandler);
412             index++;
413             if (actionResults.isEmpty()) {
414                 throw new SnapDataException("No actionResults returned");
415             }
416         }
417     }
