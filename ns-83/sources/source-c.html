25 public class SNSUtils {
26     private final static Logger LOGGER = LoggerFactory.getLogger(SNSUtils.class);
27     private final static String APN_TRACKING_FORMAT = "APN/1.0 %s/1.0 %s/%s";
28     private final static String INSTANCE_REGION;
29     static {
30         //Attempt to get the region information from the current EC2 instance. Set to null if the
31         //current machine is not a EC2 instance,or the instance has no metadata service
32         //installed. Putting it into a static block since the method below can be really slow, and
33         //the region is unlikely to be changed during the jcc process.
34         String region = null;
35         try {
36             region = EC2MetadataUtils.getEC2InstanceRegion();
37         } catch (Exception e) {
38             LOGGER.debug("Failed to fetch the region from the instance. ", e);
39         }
40         INSTANCE_REGION = region;
41     }
42 
43     public static String getApnUserAgentHeader() {
44         return String.format(APN_TRACKING_FORMAT,
45                 COMPANY_NAME,
46                 PROJECT_NAME,
47                 VERSION);
48     }
49 
50     /**

52      * current machine is not a EC2 instance,or the instance has no metadata service installed
53      * @return  the id of the region, or null if error happens.
54      */
55     public static String getRegionFromInstanceProfile() {
56         return INSTANCE_REGION;
57     }
