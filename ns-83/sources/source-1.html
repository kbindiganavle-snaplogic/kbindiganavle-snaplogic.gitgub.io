149         if (!clients.containsKey(region.id())) {
150             LOGGER.debug("Creating new client for region " + region);
151             AwsCredentialsProvider provider = account.connect();
152             SnsClientBuilder snsClientBuilder = account.snsBuilder(provider);
153             SnsClient snsClient = snsClientBuilder.region(region).build();
154 
155             if (account.largeMessageSupportEnabled()) {

158             }
159             clients.put(region.id(), snsClient);
160         }
161         return clients.get(region.id());

165      *  Attempt to evaluate the region from the following resources in order:
166      *  1. The Region property in the Snap
167      *  2. The region that appears in the topic arn
168      *  3. The region that appears in the subscription arn
169      *  4. The default region in the EC2 profile if the JCC is running on an EC2 instance
170      *  5. the default region given by the Snap code (currently US-WEST-2)
171      * @param document  The document we are evaluating with
172      * @return  the region object
173      */
174     protected Region evaluateRegion(Document document) {
175         String regionStr = regionExpr.eval(document);
176         if (StringUtils.isNotBlank(regionStr)) {
177             return Region.of(regionStr);
178         } else {
179             String instanceRegion = SNSUtils.getRegionFromInstanceProfile();
180             Region topicRegion = StringUtils.isEmpty(instanceRegion) ?
181                     AmazonSNSBaseAccount.DEFAULT_REGION : Region.of(instanceRegion);
182             String arnStr = StringUtils.isNotEmpty(topicArn) ? topicArn :
183                     StringUtils.isNotEmpty(subscriptionArn) ? subscriptionArn : StringUtils.EMPTY;
184             if (StringUtils.isNotEmpty(arnStr)) {
185                 try {
186                     Arn arn = Arn.fromString(arnStr);
187                     if (arn.region().isPresent()) {
188                         topicRegion = Region.of(arn.region().get());
189                     }
190                 } catch (IllegalArgumentException e) {
191                     throw new SnapDataException(e, ERROR_MALFORMED_ARN)
192                             .withReason(e.getMessage()).withResolution(ERR_INPUT_DATA_RESOLUTION);
193                 }
194             }
195             return topicRegion;
196         }
197     }

211         Map<String, Object> outMap = Maps.newLinkedHashMap();
212         outMap.put(REQUEST_ID, response.responseMetadata().requestId());
213         outMap.put(IS_SUCCESSFUL, response.sdkHttpResponse().isSuccessful());

215         outMap.put(STATUS_CODE, response.sdkHttpResponse().statusCode());
216         response.sdkFields().forEach(field ->

329     protected List<String> subscriptionsList(SnsClient snsClient, String topicArn) {
330         String nextToken = null;
331         List<String> subscriptionsList = Lists.newArrayList();
332         if (StringUtils.isEmpty(topicArn)) {
333             do {
334                 ListSubscriptionsRequest.Builder requestBuilder = ListSubscriptionsRequest.builder();
335                 if(StringUtils.isNotEmpty(nextToken)) {
336                     requestBuilder.nextToken(nextToken);
337                 }
338                 LOGGER.debug("Getting all subscriptions");
339                 ListSubscriptionsResponse response = snsClient.listSubscriptions(requestBuilder.build());
340                 subscriptionsList.addAll(toSubscriptionArnList(response.subscriptions()));
341                 nextToken = response.nextToken();
342                 LOGGER.debug("nextToken::" + nextToken);
343             } while (nextToken != null);
344         } else {
345             do {
346                 ListSubscriptionsByTopicRequest.Builder requestBuilder =
347                         ListSubscriptionsByTopicRequest.builder().topicArn(topicArn);
348                 LOGGER.debug("Getting subscriptions based on topicArn");
349                 if (StringUtils.isNotEmpty(nextToken)) {
350                    requestBuilder.nextToken(nextToken);
351                 }
352                 try {
353                     ListSubscriptionsByTopicResponse response =
354                             snsClient.listSubscriptionsByTopic(requestBuilder.build());
355                     subscriptionsList.addAll(toSubscriptionArnList(response.subscriptions()));
356                     nextToken = response.nextToken();
357                     LOGGER.debug("nextToken::" + nextToken);
358                 } catch (InvalidParameterException e) {
359                     nextToken = null; // Make sure loop doesn't continue after exception
360                     subscriptionsList.clear();
361                     subscriptionsList.add(ERROR_LOAD_SUBSCRIPTIONS);
362                 }
363             } while (nextToken != null);
364         }
365         return subscriptionsList;
366     }
