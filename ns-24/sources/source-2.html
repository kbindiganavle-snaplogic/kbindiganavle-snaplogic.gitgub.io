343      * @param doc Input document
344      * @param container Linked HashMap of headers provided by user
345      * @throws URISyntaxException Exception in trying to build URI
346      * @throws IgnoredRequestException Exception when user hasn't provided all the information
347      * we need

349      * @throws SnapDataException Exception when user tries to execute an unsupported HTTP method
350      * @throws UnsupportedEncodingException Exception when trying to convert users

355     public void execute(final Document doc, final Map<String, Object> container)
356             throws URISyntaxException,
357             IgnoredRequestException,
358             InvalidPathException,
359             SnapDataException,
360             UnsupportedEncodingException,
361             JsonProcessingException {

368         URI uri = new URIBuilder(optUrl.get())

377         for (HeaderProperty header : headers) {
378             if (header.toHeader(doc).isPresent()){
379                 httpRequestWrapper.addHeader(header.toHeader(doc).get());
380             }
381         }
382         if (httpRequestWrapper.getFirstHeader(HttpHeaders.ACCEPT) == null) {
383             httpRequestWrapper.addHeader(HttpHeaders.ACCEPT, MediaType.APPLICATION_JSON);
384         }
385

387         try {
388             TraceRecorder.record("request", httpRequestWrapper, RequestTrace::from);
389             HttpResponse urlResponse = httpUtils.executeRequest(httpClient, httpRequestWrapper);
390             TraceRecorder.record("response", urlResponse, ResponseTrace::from);
391             populateResponseMap(optUrl, responseMap, urlResponse);
392         } catch (IOException e) {

395                     .withResolution("Ensure request properties are valid");

402     protected HttpRequestWrapper getHttpRequestWrapper(
403             HttpMethods httpMethod,
404             URI uri,
405             TypedExpressionProperty<Optional<Object>> requestBody,
406             Document doc
407     ) throws SnapDataException, UnsupportedEncodingException, JsonProcessingException {
408 
409         String serializedRequestBody = null;
410 
411         if (!requestBody.isEmpty() && requestBody.eval(doc).isPresent()) {
412             Object evaluatedDoc = requestBody.eval(doc).get();
413             serializedRequestBody = objectMapper.writeValueAsString(evaluatedDoc);
414         }
415 
416         switch (httpMethod){
417             case GET:
418                 return HttpRequestWrapper.wrap(new HttpGet(uri));
419             case POST:
420                 HttpPost httpPost = new HttpPost(uri);
421                 if (serializedRequestBody != null) {
422                     httpPost.setEntity(new StringEntity(serializedRequestBody));
423                 }
424                 return HttpRequestWrapper.wrap(httpPost);
425             case PUT:
426                 HttpPut httpPut = new HttpPut(uri);
427                 if (serializedRequestBody != null) {
428                     httpPut.setEntity(new StringEntity(serializedRequestBody));
429                 }
430                 return HttpRequestWrapper.wrap(httpPut);
431             case PATCH:
432                 HttpPatch httpPatch = new HttpPatch(uri);
433                 if (serializedRequestBody != null) {
434                     httpPatch.setEntity(new StringEntity(serializedRequestBody));
435                 }
436                 return HttpRequestWrapper.wrap(httpPatch);
437             default:
438                 throw new SnapDataException("Invalid http method, only GET, " +
439                         "POST, PUT and PATCH are supported");

443     protected void validateResult(Map<String, Object> result,
444                                   Map<String, Object> container, Optional<String> optUrl,
445                                   Map<String, Object> responseMap)
446             throws InvalidPathException, IgnoredRequestException {
447         // Make sure the result exists and is well-formed
448         ValidationContext validationContext = new ValidationContext();
449         RESULT_VALIDATOR.validate(validationContext, result);
450         if (validationContext.hasErrors()) {
451             throw new SnapDataException("Invalid result from response handler")
452                     .withReason(validationContext.getErrors().entrySet().stream()
453                             .map(entry -> String.format("%s: %s", entry.getKey(),
454                                     String.join(", ", entry.getValue())))
455                             .collect(Collectors.joining("; ")));
456         }
457         switch ((String) result.get("variant")) {
458             case "OK":
459                 targetPath.get().write(container, result.get("value"));
460                 break;
461             case "IGNORE":
462                 throw new IgnoredRequestException();
463             case "ERROR":
464                 throw new SnapDataException("Request to '%s' failed with: %s")
465                         .formatWith(optUrl.get(), result.get("reason"))
466                         .withReason(ObjectType.toString(responseMap.get("entity")))
467                         .withResolution("Ensure the request is configured correctly and the " +
468                                 "endpoint is working");
469             case "CONTINUE":
470                 targetPath.get().write(container, result.get("value"));
471                 throw new IgnoredRequestException();
472         }
473     }

475     protected Map<String, Object> handleScopeAndGetResult(Map<String, Object> container,
476                                               Map<String, Object> responseMap, Document doc) {
477         Map<String, Object> result;
478         Scope scope = new ScopeBuilder()
479                 .addVariable("Result", RESULT_EXPR.eval(container))
480                 .addVariable("response", responseMap)
481                 .build();
482 
483         if (responseHandler.isEmpty()) {
484             // The user did not provide a custom response handler, so use the default
485             ScopeStack scopes = new ScopeStack();
486 
487             scopes.push(LanguageService.getGlobalScope());
488             scopes.push(scope);
489             result = DEFAULT_RESPONSE_HANDLER_EXPR.eval(container, scopes);
490         } else {
491             try (ExpressionProperty exprWithResponse = responseHandler.withScope(scope)) {
492                 result = exprWithResponse.eval(doc);
493             } catch (Exception e) {
494                 throw new SnapDataException(e,
495                         "Encountered an error while processing the response");
496             }
497         }
498 
499         return result;

501     protected void populateResponseMap(Optional<String> optUrl, Map<String,
502             Object> responseMap, HttpResponse urlResponse) throws IOException {
503 
504         // We need to pass the response, headers and all, to the response handling expression.
505         // We'll leverage the trace object to convert it into a Map that the e-lang can work
506         // with.
507         responseMap.putAll(MAPPER.convertValue(ResponseTrace.from(urlResponse), Map.class));
508         // Note: The NamedProxyStream is used here so that a more informative name shows up
509         // in error messages.
510 
511         try (InputStream inputStream = new NamedProxyStream(optUrl.get(),
512                 urlResponse.getEntity().getContent())) {
513             Header contentTypeHeader = urlResponse.getFirstHeader(HttpHeaders.CONTENT_TYPE);
514             MediaType contentType;
515 
516             // Assume JSON if the request was successful and no Content-Type was set.
517             // Otherwise, assume text/plain
518             if (urlResponse.getStatusLine().getStatusCode() ==
519                     Response.Status.OK.getStatusCode()) {
520                 contentType = MediaType.APPLICATION_JSON_TYPE;
521             } else {
522                 contentType = MediaType.TEXT_PLAIN_TYPE;
523             }
524             if (contentTypeHeader != null) {
525                 try {
526                     contentType = MediaType.valueOf(contentTypeHeader.getValue());
527                 } catch (IllegalArgumentException e) {
528                     throw new SnapDataException(e, "Invalid Content-Type header from URL: %s")
529                             .formatWith(optUrl.get())
530                             .withResolution("Ensure the endpoint is responding correctly");
531                 }
532             }
533             for (Map.Entry<MediaType, BiFunction<String, InputStream, Object>> entry :
534                     CONTENT_PARSERS.entrySet()) {
535                 if (!contentType.isCompatible(MediaType.APPLICATION_JSON_TYPE)) {
536                     continue;
537                 }
538                 responseMap.put("entity", entry.getValue().apply(optUrl.get(), inputStream));
539             }
540             if (!responseMap.containsKey("entity")) {
541                 // No parser was found, so just convert the response into a string
542                 responseMap.put("entity", IOUtils.toString(inputStream,
543                         StandardCharsets.UTF_8));
544             }
545         }
546     }
