74 /**
75  * Perform a bulk upload to AlloyDB using the COPY command. The external pg_restore
76  * does not support uploading standard SQL and there's no benefits to using psql \copy instead
77  * of the COPY command.
78  *
79  * This snap sidesteps jdbcOperationsImpl in a few places since the Binary input view
80  * allows the customer to specify data types that we don't support ourselves.
81  *
82  * FREEZE
83  *
84  * Support for 'freeze' is... complicated. We can do it in two ways - either do it as
85  * part of the copy command or do it in a subsequent jdbc call. In the former case we must
86  * also perform a 'truncate' (or 'create table') in the same "subtransaction". Due to our
87  * jdbc connection model that means the truncate needs to be done within this snap. It also
88  * means that we can't be running it in auto-commit (iirc).
89  *
90  * We can't use 'vacuum freeze' after the 'copy' unless we're in auto-commit.
91  *
92  * VACUUM, ANALYZE, and ERROR RECOVERY
93  *
94  * The user might want to run 'analyze' after the data is loaded. We don't want to do
95  * this ourselves (without user consent) since that could introduce a sizeable delay.
96  *
97  * If there is an error the data is written to the table, not visible, but <b>not</b>
98  * removed until a 'vacuum' has been performed. Again we don't want to do this ourselves
99  * (without user consent) since it could introduce a sizeable delay. We need to clearly
100  * document this.
101  *
102  * For more information: https://www.postgresql.org/docs/current/sql-copy.html
103  * For more information: https://wiki.postgresql.org/wiki/COPY
104  */

318     public void execute() {
319         if (!hasData()) {
320             if (!account.getDatabaseConfig().isAutoCommit()) {
321                 // required since jdbcOperationsImpl#commit() will always be called.
322                 jdbcOperations.acquireConnection(account);
323             }
324             return;
325         }
326         Connection conn = jdbcOperations.acquireConnection(account);
327         if (Boolean.FALSE.equals(autoCreateTable) &&
328                 !jdbcOperations.tableExists(conn,
329                         account.getDatabaseConfig().getDatabaseName(), tableName)) {
330             errorViews.write((SnapDataException) new SnapDataException(ERR_TABLE_DOESNT_EXIST)
331                     .withReason(ERR_TABLE_DOESNT_EXIST_REASON)
332                     .withResolution(ERR_TABLE_DOESNT_EXIST_RESOLUTION));
333 
334             // this is an error condition so *don't* write document to output view.
335             return;
336         }
337 
338         // capture schema, if provided
339         final InputView schemaView = inputViews.get(schemaViewName);
340         if (schemaView != null) {
341              final Iterator<Document> iter = inputViews.getDocumentsFrom(schemaView);
342              // Only the first document of the 2nd input view is used to setup the schema
343              //  Must loop through all of the documents in the view so close method won't throw
344              //  a Not all documents consumed error
345              while (iter.hasNext()) {
346                  Document document = iter.next();
347                  if (schema == null) {
348                      schema = document;
349                      columnTranslation = databaseUtils.buildColumnTranslation(schema);
350                      // Commit the table creation if Binary input view
351                      for (InputView iv : inputViews) {
352                          if (ViewType.BINARY.equals(iv.getType()) &&
353                              Boolean.TRUE.equals(autoCreateTable) && createTableIfAbsent(null)) {
354                              if (!account.getDatabaseConfig().getAutoCommit()) {
355                                  jdbcOperations.commit();
356                              }
357                          }
358                      }
359                  }
360              }
361         }
362 
363         final InputView dataView = inputViews.get(rowViewName);
364 
365         final long start = System.currentTimeMillis();
366         statusReporter.beginUnknown();
367 
368         final Map<String, String> columnMap = copyUtils.getColumnMap(tableName);
369 
370         // DOCUMENTS are formatted internally and always uploaded in CSV format.
371         // BINARY documents can be in any supported format so we have to ask the user.
372         final String formatStr = ViewType.DOCUMENT.equals(dataView.getType()) ? CSV_FORMAT : format;
373 
374         final String headerStr = ViewType.DOCUMENT.equals(dataView.getType()) ? ""
375                 : (Boolean.TRUE.equals(header) ? "header true," : "");
376 
377         // Note: this will be handled as SWAT stories - we need to identify dependencies
378         // on jdbc driver version.
379         // Generate SQL command.
380         // could add 'freeze', 'quote', 'escape', 'null', but see above
381         final String columns = copyUtils.getColumns(columnList, columnMap, header);
382         final String forceNotNullColumns = copyUtils.getForceNotNullColumns(columnList, columnMap);
383         final String forceNullColumns = copyUtils.getForceNullColumns(columnList, columnMap);
384 
385         final String sql = String.format(
386                     "COPY %s%s FROM stdin (FORMAT %s,%s%s%s ENCODING '%s')",
387                     tableName, columns, formatStr, headerStr, forceNotNullColumns,
388                     forceNullColumns, encoding);
389 
390         LOG.debug("Copying data : {}", sql);
391 
392         long count = 0;
393         try {
394             switch (dataView.getType()) {
395                 case DOCUMENT:
396                     count = processDocuments(sql, dataView);
397                     break;
398 
399                 case BINARY:
400                     count = processBinaryDocuments(sql, dataView);
401                     break;
402             }
403 
404             // write output document
405             if (count != 0) {
406                 final Map<String, Object> data = new LinkedHashMap<>();
407                 data.put("count", count);
408                 outputViews.write(documentUtility.newDocument(data));
409             }
410         } catch (SuggestViewAbortException e) {
411             throw e;
412         } catch (SnapDataException e) {
413             errorViews.write(e);
414             // safe to fall through if 'ignore' is true.
415         } catch (DataAccessException e) {
416             Throwable t = e.getCause();
417             if (t instanceof SnapDataException) {
418                 errorViews.write((SnapDataException) t);
419             } else if (t != null) {
420                 errorViews.write((SnapDataException)
421                         new SnapDataException(t, ERR_UPLOAD_EXCEPTION_MSG)
422                                 .withResolution(ERR_UPLOAD_EXCEPTION_MSG));
423             } else {
424                 errorViews.write((SnapDataException)
425                         new SnapDataException(e, ERR_UPLOAD_EXCEPTION_MSG)
426                                 .withResolution(ERR_UPLOAD_EXCEPTION_MSG));
427             }
428         } catch (Exception e) {
429             // wraps SQLException, etc.
430             SnapDataException sde = new SnapDataException(e, ERR_UPLOAD_EXCEPTION_MSG)
431                 .withReason(e.getMessage())
432                 .withResolution(ERR_UPLOAD_EXCEPTION_MSG);
433             errorViews.write(sde);
434         } finally {
435             long elapsed = System.currentTimeMillis() - start;
436             statusReporter.notify(UPLOADED_COUNT, count, elapsed / 1000.0);
437             statusReporter.worked(count);
438             statusReporter.done();
439         }
440     }

443      * Process documents
444      *
445      * @param sql
446      * @param dataView
447      * @return
448      */
449     @VisibleForTesting
450     protected long processDocuments(String sql, InputView dataView) {
451 
452         // do we have any input documents?
453         final Iterator<Document> documents = inputViews.getDocumentsFrom(dataView);
454         if (!documents.hasNext()) {
455             return 0;
456         }
457 
458         final ExecutorService executor = Executors.newSingleThreadExecutor(threadFactory);
459 
460         long count = 0;
461 
462         Document document = documents.next();
463 
464         // Create table if absent for Document inputView
465         if (Boolean.TRUE.equals(autoCreateTable) && createTableIfAbsent(document)) {
466             if (!account.getDatabaseConfig().getAutoCommit()) {
467                 jdbcOperations.commit();
468             }
469         }
470 
471         // get list of all columns if the user didn't specify any.
472         if (columnList.isEmpty()) {
473             columnList = copyUtils.getDefaultColumns(tableName);
474         }
475 
476         // the child thread is responsible for its own connection in order to
477         // eliminate any risk of simultaneous access by both parent and child
478         // threads.
479         try {
480             final Pipe pipe = Pipe.open();
481             final PostgresOperations.DocumentCopyInThread uploader =
482                     new PostgresOperations.DocumentCopyInThread(account, sql, pipe);
483 
484             final Future<Long> future = executor.submit(uploader);
485 
486             try (SinkChannel sc = pipe.sink()) {
487                 final CsvBulkUploadConverter converter = new CsvBulkUploadConverter(sc, columnList);
488 
489                 converter.writeHeader();
490                 converter.formatOneDocument(document);
491 
492                 while (documents.hasNext()) {
493                     document = documents.next();
494                     converter.formatOneDocument(document);
495                 }
496 
497                 converter.writeFooter();
498                 LOG.trace("Process Documents loop finished.");
499             }
500 
501             count = future.get();
502         } catch (IOException | SQLException e) {
503             // any error here is probably still related to the PipedReader/Writer
504             // but we can't associate it with a document.
505             throw new SnapDataException(e, ERR_UPLOAD_EXCEPTION_MSG)
506                 .withReason(e.getMessage())
507                 .withResolution(ERR_UPLOAD_EXCEPTION_RESOLUTION);
508         } catch (ExecutionException e) {
509             if (e.getCause() instanceof SnapException) {
510                 throw (SnapException) e.getCause();
511             } else {
512                 throw new SnapDataException(e.getCause(), ERR_UPLOAD_EXCEPTION_MSG)
513                     .withReason(e.getMessage())
514                     .withResolution(ERR_UPLOAD_EXCEPTION_RESOLUTION);
515             }
516         } catch (InterruptedException e) {
517             Thread.currentThread().interrupt();
518         } finally {
519             if (executor != null) {
520                 executor.shutdown();
521             }
522         }
523 
524         return count;
525     }
