20 /**

25 public class VersionSuggestionBuilder implements Suggestions {

27     private final String node;
28     private final String parentNode;
29 
30     public VersionSuggestionBuilder(WorkdayRestExecuteService workdayRestExecuteService,
31             String node, String parentNode) {

33         this.node = node;
34         this.parentNode = parentNode;
35     }
36 
37     @Override
38     public void suggest(SuggestionBuilder suggestionBuilder, PropertyValues propertyValues) {
39         String service = propertyValues.get(parentNode);
40         try {
41             suggestionBuilder.node(node)
42                     .suggestions(workdayRestExecuteService.getVersionAvailableForService(service)
43                             .toArray(new String[0]));
44         } catch (Exception e) {
45             suggestionBuilder.node(node)
46                     .suggestions(e.getMessage());
47         }
48     }
49 }
