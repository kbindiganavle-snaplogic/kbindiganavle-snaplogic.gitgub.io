69      * Builds list of alter table sqls for BigQuery
70      *
71      * @param columns list of columns
72      * @param dataType list of data types corresponding to the list
73      * @param alterColumnModifiers list of alter column modifiers corresponding to columns
74      * @param targetTablePath table to be modified
75      *
76      * @return list of alter table sqls
77      */
78     public static List<String> buildBigQueryAlterTableAddDropColumnsSqls(
79             List<String> columns,
80             List<String> dataType,
81             List<String> alterColumnModifiers,
82             String targetTablePath) {
83         List<String> alterTableAddColumns = new ArrayList<>();
84         List<String> alterTableDropColumns = new ArrayList<>();
85         List<String> alterTableAddDropColumnsSqls = new ArrayList<>();
86 
87         for (int indx = 0; indx < alterColumnModifiers.size(); indx++) {
88             if (StringUtils.isBlank(columns.get(indx))) {
89                 continue;
90             }
91 
92             if (StringUtils.equals(alterColumnModifiers.get(indx), ADD)) {
93                 if (StringUtils.isBlank(dataType.get(indx))) {
94                     throwInvalidDataTypeException();
95                 }
96                 alterTableAddColumns.add(
97                         String.format(ADD_COLUMN_NAME_DATA_TYPE_FMT,
98                                 columns.get(indx),
99                                 dataType.get(indx)));
100             } else {
101                 alterTableDropColumns.add(
102                         String.format(DROP_COLUMN_NAME_FMT, columns.get(indx)));
103             }
104         }
105 
106         if (alterTableAddColumns.isEmpty() && alterTableDropColumns.isEmpty()) {
107             throw new ConfigurationException(ERR_BLANK_ADD_OR_DROP_COLUMNS)
108                     .withReason(REASON_BLANK_ADD_OR_DROP_COLUMNS)
109                     .withResolution(RESOLUTION_BLANK_ADD_OR_DROP_COLUMNS);
110         }
111 
112         if (CollectionUtils.isNotEmpty(alterTableDropColumns)) {
113             alterTableAddDropColumnsSqls.add(
114                     String.format(ALTER_TABLE_DROP_COLUMNS_SQL_BIGQUERY_FMT,
115                             targetTablePath,
116                             String.join(COMMA_SPACE, alterTableDropColumns)));
117         }
118 
119         if (CollectionUtils.isNotEmpty(alterTableAddColumns)) {
120             alterTableAddDropColumnsSqls.add(
121                     String.format(ALTER_TABLE_ADD_COLUMNS_SQL_BIGQUERY_FMT,
122                             targetTablePath,
123                             String.join(COMMA_SPACE, alterTableAddColumns)));
124         }
125 
126         return alterTableAddDropColumnsSqls;
127     }

470      * Builds credentials from BQ Acct. Required to do auth with GCP for BigQuery Services
471      *
472      * @param eltDatabaseAccount ELT Database Acct
473      *
474      * @return Credentials GCP credentials object
475      */
476     public static Credentials getCredentialsFromBQAcct(EltDatabaseAccount eltDatabaseAccount) {
477         String bigQueryAuthType = eltDatabaseAccount.getBigQueryAuthType();
478         DatabaseConfig databaseConfig = eltDatabaseAccount.getDatabaseConfig();
479 
480         switch (bigQueryAuthType) {
481             case BIGQUERY_SERVICE_ACCOUNT_AUTH: {
482                 try {
483                     // Get the service account key file
484                     // Keyfile contains details needed to get authenticated/authorized to access
485                     // backend Google Cloud services through Cloud APIs.
486                     // Keyfile is downloaded and kept at predetermined location path at:
487                     // $SL_ROOT/run/jcc/snapjars/keys/<case16370-7f7c54a1c19b.json>
488                     String serviceAccountKeyPath = databaseConfig.getServiceAccountKeyPath();
489 
490                     GoogleCredentials credentials = ServiceAccountCredentials.fromStream(
491                             new FileInputStream(serviceAccountKeyPath));
492 
493                     if (credentials.createScopedRequired()) {
494                         credentials = credentials.createScoped(BigqueryScopes.all());
495                     }
496 
497                     return credentials;
498                 } catch (IOException e) {
499                     throw new ConfigurationException(e, ERR_SERVICE_ACC_FAILED_TO_LOAD)
500                             .withReason(e.getMessage());
501                 }
502             }
503 
504             case BIGQUERY_OAUTH2_ACCESS_TOKEN_AUTH: {
505                 String oAuth2AccessToken = databaseConfig.getOAuth2AccessToken();
506 
507                 Credentials credentials = GoogleCredentials.newBuilder()
508                         .setAccessToken(
509                                 new AccessToken(oAuth2AccessToken, null))
510                         .build();
511 
512                 return credentials;
513             }
514 
515             case BIGQUERY_OAUTH2_REFRESH_TOKEN_AUTH: {
516                 String oAuth2RefreshToken = databaseConfig.getOAuth2RefreshToken();
517                 String clientId = databaseConfig.getClientId();
518                 String clientSecret = databaseConfig.getClientSecret();
519 
520                 Credentials credentials = UserCredentials.newBuilder()
521                         .setClientId(clientId)
522                         .setClientSecret(clientSecret)
523                         .setRefreshToken(oAuth2RefreshToken)
524                         .build();
525 
526                 return credentials;
527             }
528 
529             default:
530                 throw new ConfigurationException("Invalid Auth Type: " + bigQueryAuthType);
531         }
