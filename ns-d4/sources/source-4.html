85     private static String getTableOptionsDLP(EltDatabaseAccount eltDatabaseAccount,
86                                              List<String> tableOptionList) {
87         StringBuilder tableOptionsBuilder = new StringBuilder();
88         StringBuilder tblpropertiesBuilder = new StringBuilder();
89 
90         boolean levelOneFirst = true;
91         for (String tableOption : tableOptionList) {
92             tableOption = tableOption.trim();
93 
94             // Merge DLP account properties and TBLPROPERTIES table option if
95             // there is an intersection between them.
96             // The precedence is in TBLPROPERTIES table option.
97             if (tableOption.startsWith(DLP_TABLE_OPTION_KEY_TBLPROPERTIES)) {
98                 // Strip start and end to get properties string.
99                 int startIndex = tableOption.indexOf(LEFT_PARENTHESIS);
100                 if (startIndex < 0) {
101                     throw new ConfigurationException(ERR_INVALID_TBLPROPERTIES_FORMAT)
102                             .withReason(String.format(
103                                     REASON_INVALID_TBLPROPERTIES_LEFT_FOUND, tableOption))
104                             .withResolution(RESOLUTION_INVALID_TBLPROPERTIES_FOUND);
105                 }
106                 int endIndex = tableOption.indexOf(RIGHT_PARENTHESIS);
107                 if (endIndex < 0) {
108                     throw new ConfigurationException(ERR_INVALID_TBLPROPERTIES_FORMAT)
109                             .withReason(String.format(
110                                     REASON_INVALID_TBLPROPERTIES_RIGH_FOUND, tableOption))
111                             .withResolution(RESOLUTION_INVALID_TBLPROPERTIES_FOUND
112                             );
113                 }
114                 tableOption = tableOption.substring(startIndex + 1, endIndex);
115 
116                 // Build TBLPROPERTIES properties with tblpropertiesBuilder
117                 tblpropertiesBuilder.append(tableOption);
118 
119                 if (StringUtils.containsIgnoreCase(tableOption,
120                         PREFIX_DELTA_AUTO_OPTIMIZE)) {
121                     // Merge DLP account properties and TBLPROPERTIES table option.
122 
123                     // Append DLP account properties to 'tblPropertiesBuilder'
124                     // only if not in the current "tableOption".
125                     if (!StringUtils.containsIgnoreCase(
126                             tableOption, DLP_TBLPROPERTY_KEY_OPTIMIZEWRITE)) {
127                         tblpropertiesBuilder.append(COMMA_SPACE);
128                         tblpropertiesBuilder.append(
129                                 String.format(DLP_TBLPROPERTY_OPTIMIZEWRITE_SET,
130                                         eltDatabaseAccount.getOptimizeWrite()));
131                     }
132                     if (!StringUtils.containsIgnoreCase(
133                             tableOption, DLP_TBLPROPERTY_KEY_AUTOCOMPACT)) {
134                         tblpropertiesBuilder.append(COMMA_SPACE);
135                         tblpropertiesBuilder.append(
136                                 String.format(DLP_TBLPROPERTY_AUTOCOMPACT_SET,
137                                         eltDatabaseAccount.getAutoCompact()));
138                     }
139                 } else {
140                     // Append delta.autoOptimize property values from the DLP
141                     // account since TBLPROPERTIES has no such properties.
142                     tblpropertiesBuilder.append(COMMA_SPACE);
143                     tblpropertiesBuilder.append(
144                             String.format(DLP_AUTO_OPTIMIZE_TBLPROPERTIES_FORMAT,
145                                     eltDatabaseAccount.getOptimizeWrite(),
146                                     eltDatabaseAccount.getAzureAccountName()));
147                 }
148             } else {
149                 if (levelOneFirst) {
150                     levelOneFirst = false;
151                 } else {
152                     tableOptionsBuilder.append(SPACE);
153                 }
154                 tableOptionsBuilder.append(tableOption);
155             }
156         }
157 
158         // Append TBLPROPERTIES table option to 'tableOptionsBuilder'.
159         if (StringUtils.isBlank(tblpropertiesBuilder)) {
160             tblpropertiesBuilder.append(
161                     String.format(DLP_AUTO_OPTIMIZE_TBLPROPERTIES_FORMAT,
162                     eltDatabaseAccount.getOptimizeWrite(),
163                             eltDatabaseAccount.getAzureAccountName()));
164         }
165         if (!levelOneFirst) {
166             tableOptionsBuilder.append(SPACE);
167         }
168         tableOptionsBuilder.append(
169                 String.format(DLP_TBLPROPERTIES_FORMAT, tblpropertiesBuilder.toString()));
170 
171         return tableOptionsBuilder.toString();
172     }

187      * @param columns list of columns
188      * @param dataType list of data types corresponding to the list
189      * @param alterColumnModifiers list of alter column modifiers corresponding to columns
190      * @param targetTablePath table to be modified
191      * @param alterTableAddSqlFmt alter table add column sql format
192      * @return alter table sql
193      */
194     public static List<String> buildDLPAlterTableSqls(List<String> columns, List<String> dataType,
195                                             List<String> alterColumnModifiers,
196                                             String targetTablePath,
197                                             String alterTableAddSqlFmt) {
198         List<String> alterTableAddColumns = new ArrayList<>();
199         List<String> alterTableDropColumns = new ArrayList<>();
200         List<String> alterTableSqlList = new ArrayList<>();
201 
202         for (int indx = 0; indx < alterColumnModifiers.size(); indx++) {
203             if (StringUtils.isBlank(columns.get(indx))) {
204                 continue;
205             }
206 
207             if (StringUtils.equals(alterColumnModifiers.get(indx), ADD)) {
208                 if (StringUtils.isBlank(dataType.get(indx))) {
209                     throwInvalidDataTypeException();
210                 }
211                 alterTableAddColumns.add(String.format(
212                         COLUMN_DATA_TYPE_FMT, columns.get(indx), dataType.get(indx)));
213             } else {
214                 alterTableDropColumns.add(columns.get(indx));
215             }
216         }
217 
218         if (alterTableAddColumns.isEmpty() && alterTableDropColumns.isEmpty()) {
219             throw new ConfigurationException(ERR_BLANK_ADD_OR_DROP_COLUMNS)
220                     .withReason(REASON_BLANK_ADD_OR_DROP_COLUMNS)
221                     .withResolution(RESOLUTION_BLANK_ADD_OR_DROP_COLUMNS);
222         }
223 
224         if (CollectionUtils.isNotEmpty(alterTableAddColumns)) {
225             alterTableSqlList.add(getAlterTableSql(targetTablePath, alterTableAddColumns,
226                                                alterTableAddSqlFmt));
227         }
228 
229         return alterTableSqlList;
230     }

298         // build create table sql with new cols to be added and without cols to be dropped
299         List<String> createTableColumns = Lists.newArrayList();
300         for (Map.Entry colEntry : colsAndDataTypesMap.entrySet()) {

332     private static void validateDropColumnWithPartitions(List<String> dropColumnList,
333                                                 List<Map<String, Object>> describeTableResultList,
334                                                 int indexOfPartitionInfo) {
335         if (indexOfPartitionInfo < 0) {
336             return;
337         }
338 
339         // index points to header, so +1 for info
340         int startIndexOfPartitionCols = indexOfPartitionInfo + 1;
341         String key = StringUtils.EMPTY;
342         String value = StringUtils.EMPTY;
343 
344         do {
345             Map<String, Object> partitionInfoMap =
346                     describeTableResultList.get(startIndexOfPartitionCols);
347             for (Map.Entry<String, Object> partitionEntry : partitionInfoMap.entrySet()) {
348                 key = partitionEntry.getKey().trim();
349                 value = partitionEntry.getValue().toString().trim();
350 
351                 // Eg: no partitioning case
352                 // # Partitioning
353                 // Not partitioned
354 
355                 if (key.equalsIgnoreCase(DLP_NOT_PARTITIONED_HEADER)) {
356                     return;
357                 } else {
358                     // Eg: with partitions
359                     // # Partitioning
360                     // Part 0	a1
361                     // Part 1	b1
362                     if (dropColumnList.contains(value)) {
363                         throwInvalidDropColumnException(value, DLP_DELTA_PARTITIONING_INFO_HEADER);
364                     }
365                 }
366             }
367             startIndexOfPartitionCols++;
368         } while (StringUtils.isNotBlank(key) &&
369                  startIndexOfPartitionCols < describeTableResultList.size());
370     }

372     private static void validateDropColumnWithSortAndClusterInfo(List<String> dropColumnList,
373                                                  List<Map<String, Object>> describeTableResultList,
374                                                  int indexOfDetailedInfo) {
375         if (indexOfDetailedInfo < 0) {
376             return;
377         }
378 
379         String clusterColumnsStr = StringUtils.EMPTY;
380         String sortColumnsStr = StringUtils.EMPTY;
381         List<String> clusterColList = Lists.newArrayList();
382         List<String> sortColList = Lists.newArrayList();
383         String key = StringUtils.EMPTY;
384         String value = StringUtils.EMPTY;
385         boolean isSortChecked = false;
386         boolean isCLusterChecked = false;
387 
388         // index points to header, so +1 in the above line
389         int startIndexOfDetails = indexOfDetailedInfo + 1;
390 
391         do {
392             Map<String, Object> detailedInfoMap =
393                     describeTableResultList.get(startIndexOfDetails);
394             for (Map.Entry<String, Object> detailedInfoEntry : detailedInfoMap.entrySet()) {
395                 key = detailedInfoEntry.getKey().trim();
396                 value = detailedInfoEntry.getValue().toString().trim();
397                 if (key.equalsIgnoreCase(DLP_BUCKET_COLUMNS_HEADER)) {
398                     clusterColumnsStr = value;
399                     isCLusterChecked = true;
400                 } else if (key.equalsIgnoreCase(DLP_SORT_COLUMNS_HEADER)) {
401                     sortColumnsStr = value;
402                     isSortChecked = true;
403                 }
404             }
405             startIndexOfDetails++;
406             if (isSortChecked && isCLusterChecked) {
407                 break;
408             }
409         } while (startIndexOfDetails < describeTableResultList.size());
410 
411         clusterColList = convertStringToList(clusterColumnsStr);
412         sortColList = convertStringToList(sortColumnsStr);
413 
414         for (String clusterCol : clusterColList) {
415             if (dropColumnList.contains(clusterCol)) {
416                 throwInvalidDropColumnException(clusterCol, DLP_CLUSTERING);
417             }
418         }
419 
420         for (String sortCol : sortColList) {
421             if (dropColumnList.contains(sortCol)) {
422                 throwInvalidDropColumnException(sortCol, DLP_SORTING);
423             }
424         }
425     }

427     private static List<String> convertStringToList(String commaSeparatedString) {
428         if (StringUtils.isBlank(commaSeparatedString)) {
429             return Lists.newArrayList();
430         }
431         commaSeparatedString = StringUtils.stripStart(commaSeparatedString, LEFT_SQUARE_BRACKET);
432         commaSeparatedString = StringUtils.stripEnd(commaSeparatedString, RIGHT_SQUARE_BRACKET);
433         return Stream.of(commaSeparatedString.split(COMMA_SPACE, -1))
434                 .map(s -> StringUtils.strip(s, BACK_TICK))
435                 .filter(s -> !s.trim().isEmpty())
436                 .collect(Collectors.toList());
437     }

449     /**
450      * Converts a non boolean value to boolean compatible value by comparing the scd fields ui
451      * values with the target column data type
452      * @param scdColValueMap - the column values to be converted
453      * @param scdColNameMap - the column name map
454      * @param colNameDataTypeMap - the column names and data types map for the table
455      * @param sourceTableAlias - the alias reference to check for in the column value
456      */
457     public static void convertDLPSCDValuesNonBooleanToBoolean(
458             final Map<String, String> scdColValueMap, final Map<String, String> scdColNameMap,
459             final Map<String, String> colNameDataTypeMap, final String sourceTableAlias) {
460         for (Map.Entry<String, String> scdEntry : scdColValueMap.entrySet()) {
461             String key = scdEntry.getKey();
462             String value = scdEntry.getValue();
463             String dataTypeOfColumn = colNameDataTypeMap.get(scdColNameMap.get(key));
464             if (StringUtils.isBlank(dataTypeOfColumn) || StringUtils.isBlank(value)) {
465                 continue;
466             }
467 
468             if (dataTypeOfColumn.equalsIgnoreCase(BOOLEAN_STR)) {
469                 // first handle quoted and back tick enclosed strings
470                 if (value.startsWith(SINGLE_QUOTE) && value.endsWith(SINGLE_QUOTE)) {
471                     value = StringUtils.strip(value, SINGLE_QUOTE);
472                 } else if (value.startsWith(DOUBLE_QUOTES) && value.endsWith(DOUBLE_QUOTES)) {
473                     value = StringUtils.strip(value, DOUBLE_QUOTES);
474                 } else if (value.startsWith(BACK_TICK) && value.endsWith(BACK_TICK)) {
475                     value = StringUtils.strip(value, BACK_TICK);
476                 }
477 
478                 // check for boolean values
479                 if (value.equalsIgnoreCase(TRUE_BOOLEAN_STR) ||
480                     value.equalsIgnoreCase(FALSE_BOOLEAN_STR)) {
481                     // No-op, valid boolean values
482                 } else if (value.contains(sourceTableAlias + DOT)) {
483                     // No-op as the value is a reference to source table column
484                 } else {
485                     // if numeric values present, string to bigdecimal will work for both floats
486                     // and ints and the bigdecimal value is compared with 0. If value > 0, then
487                     // convert to true, else to false
488                     try {
489                         BigDecimal bdValueOfStr = new BigDecimal(value);
490                         BigDecimal zeroValue = new BigDecimal("0");
491                         if (bdValueOfStr.compareTo(zeroValue) > 0) {
492                             // first value is greater than second value in the above comparison.
493                             // ie. > 0
494                             value = TRUE_BOOLEAN_STR;
495                         } else {
496                             value = FALSE_BOOLEAN_STR;
497                         }
498                         scdColValueMap.replace(key, value);
499                     } catch (Exception e) {
500                         // if conversion to integer failed, it could be alphanumeric or an invalid
501                         // value for boolean column type
502                         throwInvalidValueForBooleanTypeColumnDLP();
503                     }
504                 }
505             }
506         }
507     }
