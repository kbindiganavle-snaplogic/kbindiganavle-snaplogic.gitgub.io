494     /**
495      * Builds alter table sql
496      * @param columns list of columns
497      * @param dataType list of data types corresponding to the list
498      * @param alterColumnModifiers list of alter column modifiers corresponding to columns
499      * @param targetTablePath table to be modified
500      * @param alterTableAddSqlFmt alter table add column sql format
501      * @param alterTableDropSqlFmt alter table drop column sql format
502      * @return alter table sql
503      */
504     public static String buildAlterTableSql(List<String> columns, List<String> dataType,
505                                             List<String> alterColumnModifiers,
506                                             String targetTablePath,
507                                             String alterTableAddSqlFmt,
508                                             String alterTableDropSqlFmt) {
509         List<String> alterTableAddColumns = new ArrayList<>();
510         List<String> alterTableDropColumns = new ArrayList<>();
511 
512         for (int indx = 0; indx < alterColumnModifiers.size(); indx++) {
513             if (StringUtils.isBlank(columns.get(indx))) {
514                 continue;
515             }
516 
517             if (StringUtils.equals(alterColumnModifiers.get(indx), ADD)) {
518                 if (StringUtils.isBlank(dataType.get(indx))) {
519                     throwInvalidDataTypeException();
520                 }
521                 alterTableAddColumns.add(String.format(
522                         COLUMN_DATA_TYPE_FMT, columns.get(indx), dataType.get(indx)));
523             } else {
524                 alterTableDropColumns.add(columns.get(indx));
525             }
526         }
527 
528         if (alterTableAddColumns.isEmpty() && alterTableDropColumns.isEmpty()) {
529             throw new ConfigurationException(ERR_BLANK_ADD_OR_DROP_COLUMNS)
530                     .withReason(REASON_BLANK_ADD_OR_DROP_COLUMNS)
531                     .withResolution(RESOLUTION_BLANK_ADD_OR_DROP_COLUMNS);
532         } else if (alterTableAddColumns.isEmpty()) {
533             return getAlterTableSql(targetTablePath, alterTableDropColumns, alterTableDropSqlFmt);
534         } else if (alterTableDropColumns.isEmpty()) {
535             return getAlterTableSql(targetTablePath, alterTableAddColumns, alterTableAddSqlFmt);
536         } else {
537             return getAlterTableSql(targetTablePath, alterTableAddColumns, alterTableAddSqlFmt)
538                     .concat(getAlterTableSql(targetTablePath, alterTableDropColumns,
539                             alterTableDropSqlFmt));
540         }
541     }

760     public static String copyFileToSnaplex(URL url) {
761         // resolve the urls
762         if (url == null) {
763             throw new ExecutionException(URLS_MUST_BE_SET).withResolutionAsDefect();
764         }
765         Path fileNameWithPath = null;
766         int retry = 3;
767         IOException ioException = null;
768         while (--retry >= 0) {
769               try {
770 
771                   Path snapPath = IOUtils.buildPath(DIR_PATH, KEYS_DIR);
772                   Path versionPath = Files.createDirectories(snapPath);
773                   Path markerFile = versionPath.resolve(MARKER_NAME);
774                   String fileName = url.getPath();
775                   int ridx = fileName.lastIndexOf(SLASH);
776                   if (ridx >= 0) {
777                       fileName = fileName.substring(ridx + 1);
778                   }
779                   LOG.info("File downloaded into plex: {}", fileName);
780                   fileNameWithPath = IOUtils.buildPath(snapPath.toString(), fileName);
781                   if (!Files.exists(markerFile) || !Files.exists(fileNameWithPath)) {
782                       URLConnection urlConnection = url.openConnection();
783                       urlConnection.setUseCaches(true);
784 
785                       try (InputStream fileStream = urlConnection.getInputStream()) {
786                              fileNameWithPath = downloadFile(versionPath, fileStream,
787                                      markerFile, fileName);
788                       }
789                   }
790                   retry = -1;
791                   ioException = null;
792               } catch (IOException e) {
793                     // ignore until we exceed the retries
794                   try {
795                         Thread.sleep(SEC_1);
796                   } catch (InterruptedException ex) {
797                         LOG.warn(INTERRUPTED_WHILE_SLEEPING, Thread.interrupted(), e);
798                   }
799                   ioException = e;
800                   LOG.warn(String.format(ERR_DOWNLOADING_FILE, url.toString()), e);
801               }
802         }
803         if (ioException != null) {
804             String msg = String.format(ERR_DOWNLOADING_FILE, url.toString());
805             throw new ExecutionException(ioException, msg)
806                      .withReason(msg)
807                      .withResolution(RESOLUTION_ERR_DOWNLOADING_FILE);
808         }
809 
810         return fileNameWithPath.toString();
811 
812     }

814     private static Path downloadFile(Path snapDirPath,
815                                      InputStream inputStream,
816                                      final Path markerFile,
817                                      final String fileName) throws IOException {
818         Path tmpFilePath = Files.createTempFile(snapDirPath, fileName, DOT_TMP);
819         Path filePath = snapDirPath.resolve(fileName);
820         try {
821             LOG.info("Downloading file into Snaplex: {}", tmpFilePath);
822             Files.copy(inputStream, tmpFilePath, StandardCopyOption.REPLACE_EXISTING);
823             // In case there are multiple threads downloading the same file, we need to make
824             // sure only one of them will actually do the final install.  Creating the marker file
825             // using createFile is atomic and it should throw an exception if the file already
826             // exists.
827             Files.createFile(markerFile);
828             LOG.info("Copying file for use: {}", filePath);
829             try {
830                 Files.move(tmpFilePath, filePath, StandardCopyOption.ATOMIC_MOVE);
831             } catch (IOException e) {
832                 LOG.error("Unable to copy file: {}", filePath, e);
833                 try {
834                     Files.delete(markerFile);
835                 } catch (IOException e2) {
836                     LOG.trace("Unable to delete marker: {}", tmpFilePath, e2);
837                 }
838                 // Throw the exception so the caller will do a retry.
839                 throw e;
840             }
841         } catch (FileAlreadyExistsException e) {
842             LOG.info("Another thread is installing the JAR, waiting: {}", filePath);
843             for (int attempt = 0; attempt < 5; attempt++) {
844                 if (!Files.exists(filePath)) {
845                     break;
846                 } else {
847                     ThreadUtils.sleep(100);
848                 }
849             }
850             if (!Files.exists(filePath)) {
851                 // If it's not there yet, it's probably never going to be, so we delete the marker
852                 // file.
853                 try {
854                     LOG.warn("Deleting old marker file to try and recover from broken download: " +
855                             "{}", markerFile);
856                     Files.delete(markerFile);
857                 } catch (IOException e2) {
858                     LOG.trace("Unable to delete marker file: {}", markerFile, e2);
859                 }
860                 // Throw the exception so the caller will do a retry
861                 throw e;
862             }
863             LOG.info("The file has been copied and is ready for use: {}", filePath);
864         } finally {
865             try {
866                 Files.delete(tmpFilePath);
867             } catch (IOException e) {
868                 LOG.trace("Unable to delete temporary: {}", tmpFilePath, e);
869             }
870         }
871 
872         return filePath;
873     }
