43      * Return a list [map key value, set key value] for the given parameters.
44      *
45      * For example, if the given parameters have "CORR" as "functionName",
46      * and "y" and "x" as "argumentField1" and "argumentField2" respectively,
47      * this function sorts the fields (since CORR is commutative)
48      * and return a list ["CORR.x.y" ,"x.y"].
49      *
50      * If the function is not commutative, for example, "REGR_SXY" and
51      * is given "y", "x" as "argumentField1" and "argumentField2" respectively,
52      * this function does not sort the fields and return <"REGR_SXY.y.x", "y.x">.
53      */

55             String functionName, String argumentField1, String argumentField2,
56             EltWindowFunctionsList eltWindowFunctionsList) {
57         List<String> keyList = new ArrayList<>();
58         StringBuilder mapKeyBuilder = new StringBuilder();
59 
60         if (isCommutative(functionName, eltWindowFunctionsList)) {
61             mapKeyBuilder.append(makeSetKeyForCommutativeFunction(argumentField1, argumentField2));
62         } else {
63             mapKeyBuilder.append(
64                     String.format(STRING_PERIOD_STRING, argumentField1, argumentField2));
65         }
66 
67         keyList.add(functionName + "." + mapKeyBuilder.toString());
68         keyList.add(mapKeyBuilder.toString());

89      * Build and return the argument pair map for the given "functions" list.
90      *
91      * The map has one entry per function list with the same argument pair.
92      * For example, {"u,v": [CORR(u, v), REGR_R2(u, v)]}.
93      *
94      * If the function is commutative, argument ordering can be ignored, so that,
95      * for example, for the case of commutative "CORR" function, CORR(x, y) and
96      * CORR(y, x) has the same results and only one entry
97      * ["x,y", [CORR(x, y), CORR(y, x)] is created for CORR(x, y) and CORR(y, x)
98      * to the map.
99      */
100     public static Map<String, List<RewriterFunction>> buildArgumentPairMap(
101             List<RewriterFunction> functions, EltWindowFunctionsList eltWindowFunctionsList) {
102         Map<String, List<RewriterFunction>> pairMap = new LinkedHashMap<>();
103         for (RewriterFunction function : functions) {
104             String setKey;
105             if (isCommutative(function.getFunctionName(), eltWindowFunctionsList)) {
106                 setKey = makeSetKeyForCommutativeFunction(
107                         function.getField1Str(), function.getField2Str());
108             } else {
109                 setKey = String.format(STRING_PERIOD_STRING,
110                         function.getField1Str(), function.getField2Str());
111             }
112             function.setArgumentPairMapKey(setKey);
113             if (!pairMap.containsKey(setKey)) {
114                 pairMap.put(setKey, new ArrayList<>());
115             }
116             pairMap.get(setKey).add(function);
117         }
118         return pairMap;
119     }

152      * Rewrite the given "sqlQuery" to use subqueries instead of CTEs and
153      * return the rewritten SQL statement.
154      *
155      * Example 1. Let's assume that we have
156      *  "WITH cte1 (col1, col2)                 -- cte name, cte column list
157      *  AS (SELECT c1, c2 FROM t1)              -- cte query
158      *  SELECT cte1.col1, cte1.col2 FROM cte1"  -- main outer query
159      *  as the input parameter 'sqlQuery'.
160      *  Then this function returns
161      *  "SELECT cte1.col1, cte1.col2 FROM (SELECT c1 AS col1, c2 AS col2 FROM t1) AS cte1".
162      *  Here we substitute subqueries for CTEs with additional aliases.
163      *
164      * Example 2. Let's assume we have
165      *  "WITH cte1 (col1, col2) AS
166      *      (SELECT c1, c2 FROM t1),
167      *   cte2 (col1, col2) AS
168      *      (SELECT c1, c2 FROM t2)
169      *  SELECT cte1.col1, cte2.col2 FROM cte1, cte2" as input 'sqlQuery'.
170      *  Then we return
171      *  "SELECT cte1.col1, cte2.col2 FROM (SELECT c1 AS col1, c2 AS col2 FROM t1) AS cte1,
172      * 	    (SELECT c1 AS col1, c2 AS col2 FROM t2) AS cte2" after substitution.
173      *
174      * Example 3 with a format without a CTE column list:
175      *
176      * WITH cte1 AS                	                  -- cte name 1
177      *    (SELECT c1, c2 FROM t1)                     -- cte query 1
178      * SELECT c1, c2+100 FROM cte1	                  -- main outer query
179      *
180      * @param sqlQuery  input SELECT SQL statemment that uses CTEs
181      * @return  output SELECT SQL statement that uses subqueries and
182      *              is equivalent to the input in the results.
183      */
184     public static String rewriteCteToSubquery(String sqlQuery) {
185         String sqlQueryString = sqlQuery.trim();
186         String[] parts = sqlQueryString.split(SPLIT_BY_AS_PATTERN);
187         if (parts.length < 2) {
188             throw new ConfigurationException(ERR_INVALID_SELECT_WITH_CLAUSE_FOUND)
189                     .withReason(String.format(
190                             REASON_INVALID_SELECT_WITH_CLAUSE_FOUND,
191                             sqlQueryString))
192                     .withResolution(RESOLUTION_INVALID_SELECT_WITH_CLAUSE_FOUND);
193         }
194 
195         int cteIndex = 0;
196         Set<String> cteWithColumnListSet = new HashSet<>();
197         String cteName = EMPTY_STRING;
198         String cteColumnList = EMPTY_STRING;
199         String cteQuery = EMPTY_STRING;
200         Map<String, String> cteNameToCteQueryMap = new LinkedHashMap<>();
201         // The next is a map of maps (a map from the cte column list entries,
202         // e.g., "col1, col2" in Example 1, to the cte query column list entries,
203         // e.g., "c1, c2" in Example 1).
204         Map<String, Map<String, String>>
205                 cteColumnsToCteQueryColumnsMaps = new LinkedHashMap<>();
206         String mainOuterQuery = EMPTY_STRING;
207 
208         // 1. Process first part:
209         // For Example 1, the first part is "WITH cte1 (col1, col2)" and
210         // we want to capture the CTE expression name. "cte1" and
211         // its CTE column list, "col1, col2.
212         // These have CTE_NAME, CTE_COLUMN_LIST as capture group names.
213         //
214         // For Example 3, the first part is "WITH cte1".
215         // We want to capture "cte1". This has CTE_NAME as capture group name.
216         String firstPart = parts[0].trim();
217         Matcher matcher = AZURE_SYNAPSE_CTE_NAME_COLUMNLIST_PATTERN.matcher(firstPart);
218         if (matcher.find()) {
219             if (matcher.start(INDEX_FOR_CTE_NAME_GROUP_AT_PATTERN1) > 0) {
220                 cteName = matcher.group(CTE_NAME);
221                 cteColumnList = matcher.group(CTE_COLUMN_LIST);
222                 cteWithColumnListSet.add(cteName);
223             } else if (matcher.start(INDEX_FOR_CTE_NAME_ALT_GROUP_AT_PATTERN1) > 0) {
224                 cteName = matcher.group(CTE_NAME_ALT);
225             } else {
226                 throwExceptionIfInvalidNameOrColumnList(sqlQueryString);
227             }
228             cteIndex++;
229         } else {
230             throwExceptionIfInvalidNameOrColumnList(sqlQueryString);
231         }
232 
233         // 2. Process middle parts
234         // From Example 2, the middle part is
235         // "(SELECT c1, c2 FROM t1), cte2 (col1, col2)".
236         // We want to capture the CTE query "(SELECT c1, c2 FROM t1)"
237         // (for the previous CTE "cte1"), the next CTE name, "cte2",
238         // and its CTE column list, "col1, col2".
239         if (parts.length > 2) {
240             for ( ; cteIndex < parts.length - 1; cteIndex++) {
241                 String currentPart = parts[cteIndex].trim();
242                 matcher = AZURE_SYNAPSE_CTE_QUERY_NAME_COLUMNLIST_PATTERN
243                         .matcher(currentPart);
244                 if (matcher.find()) {
245                     if (matcher.start(INDEX_FOR_CTE_QUERY_GROUP_AT_PATTERN2) > 0) {
246                         // Process the current CTE
247                         cteQuery = matcher.group(CTE_QUERY);
248 
249                         // Add a map entry to both maps for the current CTE.
250                         addEntryToMaps(cteName, cteQuery, cteColumnList, sqlQueryString,
251                                 cteNameToCteQueryMap, cteColumnsToCteQueryColumnsMaps,
252                                 cteWithColumnListSet.contains(cteName));
253 
254                         // Capture objects for the next CTE
255                         cteName = matcher.group(CTE_NAME);
256                         cteColumnList = matcher.group(CTE_COLUMN_LIST);
257                         cteWithColumnListSet.add(cteName);
258                     } else if (matcher.start(
259                             INDEX_FOR_CTE_QUERY_ALT_GROUP_AT_PATTERN2) > 0) {
260                         // Process the current CTE
261                         cteQuery = matcher.group(CTE_QUERY_ALT);
262                         // Add a map entry to both maps for the current CTE.
263                         addEntryToMaps(cteName, cteQuery, cteColumnList, sqlQueryString,
264                                 cteNameToCteQueryMap, cteColumnsToCteQueryColumnsMaps,
265                                 cteWithColumnListSet.contains(cteName));
266 
267                         // Capture objects for the next CTE
268                         cteName = matcher.group(CTE_NAME_ALT);
269                     } else {
270                         throwExceptionIfInvalidCteQueryNameOrList(sqlQueryString);
271                     }
272                 } else {
273                     throwExceptionIfInvalidCteQueryNameOrList(sqlQueryString);
274                 }
275             }
276         }
277 
278         // 3. Process last part
279         // From Example 1, the last part is
280         // "(SELECT c1, c2 FROM t1) SELECT cte1.col1, cte1.col2 FROM cte1"
281         // and we want to capture the CTE query "(SELECT c1, c2 FROM t1)" and
282         // the main SELECT outer query "SELECT cte1.col1, cte1.col2 FROM cte1".
283         String lastPart = parts[cteIndex].trim();
284         matcher = CTEQUERY_OUTERQUERY_PATTERN.matcher(lastPart);
285         if (matcher.find()) {
286             cteQuery = matcher.group(CTE_QUERY);
287             mainOuterQuery = matcher.group(MAIN_OUTER_QUERY);
288             // Add a map entry to both maps for the current CTE.
289             addEntryToMaps(cteName, cteQuery, cteColumnList, sqlQueryString,
290                     cteNameToCteQueryMap, cteColumnsToCteQueryColumnsMaps,
291                     cteWithColumnListSet.contains(cteName));
292         } else {
293             throw new ConfigurationException(ERR_INVALID_SELECT_WITH_CLAUSE_FOUND)
294                     .withReason(String.format(
295                             REASON_CTE_OR_MAIN_SELECT_NOT_CORRECT,
296                             sqlQueryString))
297                     .withResolution(RESOLUTION_INVALID_SELECT_WITH_CLAUSE_FOUND);
298         }
299 
300         // 4. Rewrite the given query to build a query using subquery and return.
301         return replaceCteWithSubquery(mainOuterQuery,
302                 cteNameToCteQueryMap,
303                 cteColumnsToCteQueryColumnsMaps,
304                 cteWithColumnListSet);
305     }

460     /**
461      * Rewrite the given "mainOuterQueryStr" to use subqueries
462      * in place of CTEs.
463      *
464      * For example, for Example 1 input,
465      *  "WITH cte1 (col1, col2)                 -- cte name, cte column list
466      *  AS (SELECT c1, c2 FROM t1)              -- cte query
467      *  SELECT cte1.col1, cte1.col2 FROM cte1"  -- main outer query
468      *
469      *  this functio returns
470      *  "SELECT cte1.col1, cte1.col2 FROM (SELECT c1 AS col1, c2 AS col2 FROM t1) AS cte1".
471      *
472      *  after
473      *  1. Modify the cte query, only if its name is not contained at cteWithColumnListSet,
474      *     "(SELECT c1, c2 FROM t1)" to
475      *     "(SELECT c1 AS col1, c2 AS col2 FROM t1)" by appending AS clauses and
476      *     updating if AS clauses already exist.
477      *  2. Update the main outer query by replacing cteName references
478      *     (in FROM clause) with its cte query definition and appending the cteName.
479      *
480      * @param mainOuterQueryStr         main outer query string
481      * @param cteNameToCteQueryMap      map with <cteName, cteQuery> entries.
482      * @param cteColumnsToCteQueryColumnsMaps
483      *                          map: cteQuery column list entries -> cteColumnList entries.
484      * @param cteWithColumnListSet        Set to contain names of the CTEs with column list.
485      * @return  Rewritten query.
486      */
487     private static String replaceCteWithSubquery(
488             String mainOuterQueryStr,
489             Map<String, String> cteNameToCteQueryMap,
490             Map<String, Map<String, String>> cteColumnsToCteQueryColumnsMaps,
491             Set<String> cteWithColumnListSet) {
492         StringBuilder cteQuerySelectClauseBuilder = new StringBuilder();
493 
494         // 1. Modify each cte query from the cteNameToCteQueryMap.
495         for (Map.Entry<String, String> ent : cteNameToCteQueryMap.entrySet()) {
496             String cteName = ent.getKey();
497             // Skip the modification if the current CTE does not have column list.
498             if (!cteWithColumnListSet.contains(cteName)) {
499                 continue;
500             }
501             String cteQuery = ent.getValue();
502             Map<String, String> colummMap = cteColumnsToCteQueryColumnsMaps.get(cteName);
503 
504             // Partition the cteQuery into SELECT clause and the rest of it.
505             // Note that parts[] gets one entry when a CTE query is "SELECT 2, 3, 4".
506             String[] parts = cteQuery.split(SPLIT_BY_FROM_KEYWORD_PATTERN);
507 
508             cteQuerySelectClauseBuilder.setLength(0);
509             cteQuerySelectClauseBuilder = new StringBuilder(parts[0]);
510 
511             for (Map.Entry<String, String> columnMapEnt : colummMap.entrySet()) {
512                 String key = columnMapEnt.getKey();
513                 String value = columnMapEnt.getValue();
514 
515                 if (cteQuerySelectClauseBuilder.indexOf(key) >= 0) {
516                     String replacement = EMPTY_STRING;
517                     if (key.matches(COLUMN_AS_PATTERN_STR)) {
518                         replacement = String.format(COLUMN_REPLACEMENT_FORMAT,
519                                 key.substring(0, StringUtils.indexOfIgnoreCase(key, AS_STRING)),
520                                 value);
521                     } else {
522                         replacement = String.format(COLUMN_REPLACEMENT_FORMAT, key, value);
523                     }
524                     cteQuerySelectClauseBuilder.replace(
525                             cteQuerySelectClauseBuilder.indexOf(key),
526                             cteQuerySelectClauseBuilder.indexOf(key) + key.length(),
527                             replacement);
528                 }
529                 Pattern pattern = Pattern.compile(key);
530                 Matcher matcher = pattern.matcher(cteQuerySelectClauseBuilder);
531             }
532             if (parts.length == 2) {
533                 cteNameToCteQueryMap.put(cteName,
534                         String.format(CTE_QUERY_BUILD_FORMAT,
535                                 cteQuerySelectClauseBuilder.toString(),
536                                 parts[1]));
537             } else if (parts.length == 1) {
538                 cteNameToCteQueryMap.put(cteName,
539                         String.format(CTE_QUERY_BUILD_ALT_FORMAT,
540                                 cteQuerySelectClauseBuilder.toString()));
541             } else {
542                 throw new ConfigurationException(ERR_INVALID_CTE_SELECT_QUERY_FOUND)
543                         .withReason(String.format(
544                                 REASON_SELECT_NOT_CORRECT,
545                                 mainOuterQueryStr))
546                         .withResolution(RESOLUTION_INVALID_SELECT_QUERY_FOUND);
547             }
548         }
549 
550         // 2. Modify the given "mainOuterQueryStr" by
551         //    replacing "cteName" with cteQuery string in a backward direction
552         //    rom the end to the start of "mainOuterQueryStr".
553         StringBuilder mainQueryBuilder = new StringBuilder(mainOuterQueryStr);
554         int start = 0;
555         int end = 0;
556 
557         // One multi CTE example A is
558         // "WITH cte1 AS (SELECT c1, c2 FROM t1) SELECT * FROM cte1 alias1 " +
559         // "WHERE alias1.c1 < (SELECT Avg(c1) * 10 FROM cte1 alias2 WHERE c1 > 0)".
560         // Here, mainOuterQueryStr is "SELECT * FROM cte1 alias1 " +
561         // "WHERE alias1.c1 < (SELECT Avg(c1) * 10 FROM cte1 alias2 WHERE c1 > 0)".
562         start = StringUtils.lastIndexOfIgnoreCase(mainQueryBuilder, FROM_STRING);
563         // Error out if FROM clause is missing from the main outer query.
564         if (start < 0) {
565             throw new ConfigurationException(ERR_INVALID_MAIN_OUTER_SELECT_QUERY_FOUND)
566                     .withReason(String.format(
567                             REASON_FROM_CLAUSE_MISSING,
568                             mainOuterQueryStr))
569                     .withResolution(RESOLUTION_INVALID_SELECT_QUERY_FOUND);
570         }
571         end = mainQueryBuilder.length();
572         // For the example A, start will be 65 and end 95.
573 
574         while ( start >= 0) {
575             // If the current start location is wrapped by whitespaces,
576             // then update the mainQueryBuilder.
577             if (isWrappedByWhitespace(mainQueryBuilder, FROM_STRING, start)) {
578                 updateFromClause(mainQueryBuilder, start, end, cteNameToCteQueryMap);
579             }
580             end = start;
581             start = StringUtils.lastIndexOfIgnoreCase(
582                     mainQueryBuilder.substring(0, end), FROM_STRING);
583             // For the first span, end (previous start) is 65, start is 9.
584         }
585 
586         return mainQueryBuilder.toString();
587     }

597      * Update the FROM clause from the given substring mainQueryBuilder[begin:finish].
598      * Note that the substring has only one FROM clause.
599      *
600      * For example, given "SELECT br_id2, COUNT(k2) AS total_sales, sales_year2" +
601      * " FROM cte1 GROUP BY sales_year2, br_id2" as mainQueryBuilder,
602      * "FROM cte1 GROUP BY sales_year2, br_id2" as mainQueryBuilder[begin:finish],
603      * this function takes "cte1" as FROM list, and replace "cte1" with "cte1"'s query.
604      *
605      * @param mainQueryBuilder      main query string
606      * @param begin                 begin index of the substring to update
607      * @param finish                finish index of the substring to update
608      * @param cteNameToCteQueryMap  map with {cteName: cteQuery, ..}
609      */
610     private static void updateFromClause(StringBuilder mainQueryBuilder,
611                                          int begin, int finish,
612                                          Map<String, String> cteNameToCteQueryMap) {
613         int start = 0;
614         int end = 0;
615 
616         // Take FROM list from mainQueryBuilder[begin:finish].
617         String[] strings =
618                 mainQueryBuilder.substring(begin, finish).split(SPLIT_FROM_LIST_PATTERN);
619         end = strings[0].length();
620 
621         String replacement = updateAndReturnFromList(
622                 mainQueryBuilder.substring(begin + start, begin + end), cteNameToCteQueryMap);
623         mainQueryBuilder.replace(begin + start, begin + end, replacement);
624     }

626     /**
627      * Update the given "fromListString" by replacing cteName with cteQuery
628      * for each CTE from the given "ctenameToCteQueryMap",
629      * update its alias if necessary and return the updated string.
630      *
631      * INput "fromListString" examples include
632      * "FROM cte1 alias2",
633      * "FROM t2 alias1 LEFT OUTER JOIN cte1 alias2 ON alias1.c1 = alias2.c1"
634      * "ROM t2 alias2 INNER JOIN t3 alias3 ON alias2.c1 = alias3.c1 " +
635      *      "INNER JOIN cte1 ON cte1.c1 = alias2.c1"
636      * @param fromListString         FROM list of a query
637      * @param cteNameToCteQueryMap   map from cteName to cteQuery
638      * @return                       Updated string
639      */
640     private static String updateAndReturnFromList(String fromListString,
641                                                     Map<String, String> cteNameToCteQueryMap) {
642         StringBuilder fromListStringBuilder =
643                 new StringBuilder(fromListString);
644 
645         for (Map.Entry<String, String> ent : cteNameToCteQueryMap.entrySet()) {
646             String cteName = ent.getKey();
647             String cteQuery = ent.getValue();
648 
649             if (fromListStringBuilder.indexOf(cteName) >= 0) {
650                 int start = fromListStringBuilder.indexOf(cteName);
651                 int aliasStart = -1;
652 
653                 // Check if JOIN keyword follows cteName
654                 Pattern joinKeywordPattern =
655                         Pattern.compile(String.format(
656                                 JOIN_OR_ON_KEYWORD_FOLLOWS_CTE_NAME_PATTERN_STR,
657                                 cteName),
658                                 CASE_INSENSITIVE);
659                 Matcher matcher = joinKeywordPattern.matcher(fromListStringBuilder);
660                 if (!matcher.find()) {
661                     // Check if an alias follows cteName
662                     Pattern aliasPattern =
663                             Pattern.compile(String.format(
664                                     NAME_AS_ALIAS_PATTERN_STR,
665                                     cteName, ALIAS_GROUP_NAME,
666                                     cteName, ALIAS_ALT_GROUP_NAME),
667                                     CASE_INSENSITIVE);
668                     matcher = aliasPattern.matcher(fromListStringBuilder);
669                     if (matcher.find()) {
670                         if (matcher.start(INDEX_FOR_ALIAS_GROUP) > 0) {
671                             aliasStart = matcher.start(INDEX_FOR_ALIAS_GROUP);
672                         } else if (matcher.start(INDEX_FOR_ALIAS_ALT_GROUP) > 0) {
673                             aliasStart = matcher.start(INDEX_FOR_ALIAS_ALT_GROUP);
674                         } else {
675                             throw new ConfigurationException(ERR_INVALID_FROM_CLAUSE_FOUND)
676                                     .withReason(String.format(
677                                             REASON_INVALID_FROM_CLAUSE_FOUND,
678                                             fromListString))
679                                     .withResolution(RESOLUTION_INVALID_FROM_CLAUSE_FOUND);
680                         }
681                     }
682                 }
683 
684                 // Replace cteNAme with cteQuery at fromListStringBuilder.
685                 if (aliasStart >= 0) {
686                     String replacement = String.format(
687                             REPLACEMENT_FOR_CTE_REFERENCE_WITH_ALIAS_AT_FROM_CLAUSE,
688                             cteQuery);
689                     fromListStringBuilder.replace(
690                             start, start + cteName.length(), replacement);
691                 } else {
692                     fromListStringBuilder.replace(
693                             start, start + cteName.length(),
694                             String.format(REPLACEMENT_FOR_CTE_REFERENCE_AT_FROM_CLAUSE,
695                                     cteQuery, cteName));
696                 }
697             }
698         }
699         return fromListStringBuilder.toString();
700     }
