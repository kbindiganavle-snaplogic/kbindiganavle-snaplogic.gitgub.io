285

292             }

304                     .formatWith(tableName)

306                     .withResolution(e.getMessage());
307         } finally {
308             JDBCUtils.safeClose(tblMetaData);
309         }

358     private Field findSqlTypeField(int type) {
359         Class cl = Types.class;
360 
361         // XXX Pre-convert types from the driver that are not found in the Types class to
362         // a value that is defined.
363         if (sqlTypeMapping.containsKey(type)) {
364             type = sqlTypeMapping.get(type);
365         }
366 
367         for (Field field : cl.getFields()) {
368             if (!Modifier.isStatic(field.getModifiers()) ||
369                     !Modifier.isPublic(field.getModifiers())) {
370                 continue;
371             }
372 
373             try {
374                 if (field.getInt(null) == type) {
375                     return field;
376                 }
377             } catch (IllegalAccessException e) {
378                 continue;
379             }
380         }
381 
382         return null;
383     }

412     /**
413      * Convert the ResultSet from a getColumns() call into a Map.  To make
414      * the data a little easier to digest, we convert some of the enumerated
415      * values, like the java.sql.Type, into a string value.
416      *
417      * @param columnResult The getColumns() ResultSet to copy.
418      * @return A Map object with the converted contents of the ResultSet.
419      * @throws SQLException If there was a problem reading from the ResultSet.
420      */
421     public Map<String, Object> readColumnSchema(ResultSet columnResult) throws SQLException {
422         Map<String, Object> retval = convertResultToMap(columnResult, metadataColumnAliases);
423         int dataType = columnResult.getInt(DATA_TYPE);
424         Field typeField = findSqlTypeField(dataType);
425         String typeName = (String) retval.get(TYPE_NAME);
426 
427         if (typeField == null) {
428             // The driver is returning a bogus data type value, so we'll just change
429             // it to VARCHAR(32700)
430             log.warn(UNABLE_TO_FIND_JAVA_SQL_TYPE, retval);
431             retval.put(DATA_TYPE, VARCHAR);
432             retval.put(CHAR_OCTET_LENGTH, 32700);
433         } else {
434             retval.put(DATA_TYPE, typeField.getName());
435         }
436         switch (columnResult.getInt(NULLABLE)) {
437             case java.sql.DatabaseMetaData.columnNoNulls:
438                 retval.put(NULLABLE, COLUMN_NO_NULLS);
439                 break;
440             case java.sql.DatabaseMetaData.columnNullable:
441                 retval.put(NULLABLE, COLUMN_NULLABLE);
442                 break;
443             case java.sql.DatabaseMetaData.columnNullableUnknown:
444                 retval.put(NULLABLE, COLUMN_NULLABLE_UNKNOWN);
445                 break;
446         }
447         for (Map.Entry<String, Object> entry : METADATA_COLUMN_DEFAULTS.entrySet()) {
448             if (!retval.containsKey(entry.getKey())) {
449                 retval.put(entry.getKey(), entry.getValue());
450             }
451         }
452         // Workaround drivers that return a CHAR_OCTET_LENGTH when they should not.
453         switch (dataType) {
454             case Types.CHAR:
455             case Types.VARCHAR:
456             case Types.LONGNVARCHAR:
457             case Types.LONGVARCHAR:
458             case Types.NCHAR:
459             case Types.NVARCHAR:
460                 break;
461             case Types.OTHER:
462                 if (typeName.equals(NCHAR) || typeName.equals(NVARCHAR_2)) {
463                     // Oracle classifies these types as OTHER for some reason and the reported
464                     // length is double the original value.
465                     Number charOctetLength = (Number) retval.get(CHAR_OCTET_LENGTH);
466                     switch (typeName) {
467                         case NVARCHAR_2:
468                             retval.put(DATA_TYPE, NVARCHAR);
469                             break;
470                         default:
471                             retval.put(DATA_TYPE, typeName);
472                             break;
473                     }
474                     retval.put(CHAR_OCTET_LENGTH, charOctetLength.intValue() / 2);
475                     break;
476                 }
477             default:
478                 retval.put(CHAR_OCTET_LENGTH, null);
479                 break;
480         }
481         switch (dataType) {
482             case Types.BIT:
483             case Types.BINARY:
484             case Types.VARBINARY:
485             case Types.LONGVARBINARY:
486                 retval.put(SL_BINARY_LENGTH, retval.get(COLUMN_SIZE));
487                 break;
488         }
489 
490         return retval;
491     }

494      * Convert the ResultSet from an getIndexInfo() call into a Map.
495      *
496      * @param indexInfo   The Map to populate with index information.
497      * @param indexResult The ResultSet from a getIndexInfo() call.
498      * @throws SQLException If there was a problem reading from the ResultSet.
499      */
500     private void convertIndexInfo(final Map<String, Object> indexInfo,
501                                   final ResultSet indexResult) throws SQLException {
502         while (indexResult.next()) {
503             String indexName = indexResult.getString(INDEX_NAME);
504             List<Object> indexKeys;
505 
506             // We can't convert the map to JSON with a null key, so use an empty string.
507             if (indexName == null) {
508                 indexName = "";
509             }
510             if (primaryIndexName.contains(indexName)) {
511                 continue;
512             }
513             if (indexResult.getString(COLUMN_NAME) == null) {
514                 // Oracle returns an index with a null COLUMN_NAME for some reason.
515                 continue;
516             }
517             // We group the indexes by name so they're easier to process.
518             indexKeys = (List<Object>) indexInfo.get(indexName);
519             if (indexKeys == null) {
520                 indexKeys = new ArrayList<>();
521                 indexInfo.put(indexName, indexKeys);
522             }
523             indexKeys.add(convertResultToMap(indexResult, metadataIndexAliases));
524         }
525     }

611     /**
612      * Read information about a table from the DatabaseMetaData into a Map.
613      *
614      * @param metaData    The DatabaseMetaData to use when reading information about a table.
615      * @param tableResult The ResultSet from a getTables() call.
616      * @return A map with the table metadata.
617      * @throws SQLException If there was a problem reading the metadata.
618      */
619     private Map<String, Object> readTableSchema(final java.sql.DatabaseMetaData metaData,
620                                                 ResultSet tableResult, String tableName)
621             throws SQLException {
622         String tableCatalog = tableResult.getString(TABLE_CAT);
623         String tableSchema = tableResult.getString(TABLE_SCHEM);
624         Map<String, Object> columns = new LinkedHashMap<>();
625         Map<String, Object> primaryKeyColumns = new LinkedHashMap<>();
626         Map<String, Object> indexInfo = new LinkedHashMap<>();
627         Map<String, Object> importedKeys = new LinkedHashMap<>();
628         Map<String, Object> retval = new LinkedHashMap<>(2);
629         Map<String, Object> declaration = convertResultToMap(tableResult, metadataTableAliases);
630         retval.put(DECLARATION, declaration);
631         retval.put(COLUMNS, columns);
632         retval.put(PRIMARY_KEY_COLUMNS, primaryKeyColumns);
633         retval.put(INDEX_INFO, indexInfo);
634         retval.put(IMPORTED_KEYS, importedKeys);
635         // Unfortunately, getColumns() does not always give us all of the metadata that
636         // might be available, so we do a select to get "display size", "scale", and "precision".
637         try (Statement stmt = metaData.getConnection().createStatement()) {
638             log.debug(GETTING_COLUMN_INFO_FOR_TABLE, tableName);
639             if (SYNONYM.equals(declaration.get(TABLE_TYPE))) {
640                 // if column name is a synonym, get its real column name and get the column meta
641                 // data
642                 getColumnsMetaFromSynonym(columns, stmt, tableSchema, tableName, metaData);
643             } else {
644                 getColumnsMetaData(columns, tableCatalog, tableSchema, tableName, metaData);
645             }
646             log.debug(GETTING_RESULT_SET_METADATA_FOR_TABLE, tableName);
647             GenContext context = new GenContext(getGrammarBindingFor(quotationHandler));
648             ResultSet rows = null;
649             try {
650                 String metaDataSelectQuery;
651                 //Select query with where condition of 1 = 0 is not being supported for
652                 // Cassandra db, and using limit 1.
653                 String productName = metaData.getDatabaseProductName().toLowerCase();
654                 switch (productName) {
655                     case CASSANDRA:
656                         metaDataSelectQuery = String.format(SELECT_FOR_CASSANDRA_METADATA,
657                                 tableName);
658                         break;
659                     default:
660                         metaDataSelectQuery = String.format(SELECT_FOR_METADATA,
661                                 context.generate(BasicSqlGrammar.FQ_TABLE_NAME, retval));
662                         break;
663                 }
664                 rows = stmt.executeQuery(metaDataSelectQuery);
665                 ResultSetMetaData resultSetMetaData = rows.getMetaData();
666                 for (int i = 1; i <= resultSetMetaData.getColumnCount(); i++) {
667                     Object object = columns.get(resultSetMetaData.getColumnName(i));
668                     if (object instanceof Map) {
669                         @SuppressWarnings("unchecked")
670                         Map<String, Object> columnInfo = (Map<String, Object>) object;
671                         switch (resultSetMetaData.getColumnType(i)) {
672                             case Types.TINYINT:
673                             case Types.SMALLINT:
674                             case Types.INTEGER:
675                             case Types.BIGINT:
676                                 // scale and precision don't really apply to ints.
677                                 columnInfo.put(SL_DISPLAY_SIZE,
678                                         resultSetMetaData.getColumnDisplaySize(i));
679                                 break;
680                             case Types.REAL:
681                             case Types.DECIMAL:
682                             case Types.NUMERIC:
683                             case Types.FLOAT:
684                             case Types.DOUBLE:
685                                 columnInfo.put(SL_DISPLAY_SIZE,
686                                         resultSetMetaData.getColumnDisplaySize(i));
687                                 columnInfo.put(SL_SCALE, resultSetMetaData.getScale(i));
688                                 columnInfo.put(SL_PRECISION, resultSetMetaData.getPrecision(i));
689                                 break;
690                             default:
691                                 break;
692                         }
693                     }
694                 }
695             } catch (ConsumeException e) {
696                 throw new SnapException(e, UNABLE_TO_GENERATE_FULLY_QUALIFIED_TABLE_NAME)
697                         .withReason(e.getMessage())
698                         .withResolutionAsDefect();
699             } finally {
700                 JDBCUtils.safeClose(rows);
701             }
702         }
703         log.debug(GETTING_PRIMARY_KEYS_FOR_TABLE, tableName);
704         try (ResultSet rs = metaData.getPrimaryKeys(tableCatalog, tableSchema, tableName)) {
705             while (rs.next()) {
706                 primaryKeyColumns.put(rs.getString(COLUMN_NAME), convertResultToMap(rs,
707                         metadataKeyAliases));
708             }
709         }
710 
711         log.debug(GETTING_INDEX_INFORMATION_FOR_TABLE, tableName);
712         // NB: We don't care about the index stats, so we pass approximate=true here.  This
713         // also fixes issues with this call potentially taking a long time to complete.
714         try (ResultSet rs = getIndexInfoResultSet(metaData, tableCatalog, tableSchema, tableName,
715                 false, true)) {
716             convertIndexInfo(indexInfo, rs);
717         } catch (SQLException e) {
718             // XXX Oracle's driver doesn't seem to handle table names that need to be quoted for
719             // this query...
720             // sometimes it just fails (corrupted database?) like with SNAP-3770. Since this
721             // information is rarely used and isn't critical it's better to provide incomplete
722             // information than to cause the snap to fail.
723             log.warn(WARN_NO_INDEX_INFO, e);
724             if (!metaData.getConnection().isClosed() &&
725                     !metaData.getConnection().getAutoCommit()) {
726                 metaData.getConnection().rollback();
727             }
728         }
729 
730         if (supportsImportedKeys()) {
731             // XXX Unfortunately, the Redshift server returns an error when we try to make this
732             // call, so we have to leave it out for now.
733             log.debug(GETTING_IMPORTED_KEYS_FOR_TABLE, tableName);
734             try (ResultSet rs = metaData.getImportedKeys(tableCatalog, tableSchema, tableName)) {
735                 readForeignKeys(importedKeys, rs);
736             } catch (SQLException e) {
737                 // sometimes metadata queries just fail (corrupted database?) like with SNAP-3770.
738                 // Since this information is rarely used and isn't critical it's better to provide
739                 // incomplete than to cause the snap to fail.
740                 log.warn(WARN_NO_IMPORTED_KEYS_INFO, e);
741             }
742         } else {
743             log.warn(DATABASE_DOES_NOT_SUPPORT_GET_IMPORTED_KEYS_METHOD, tableName);
744         }
745 
746         // Note: We don't do getExportedKeys because it can be really slow.  I was seeing 30+
747         // second run times on the dev-test MySQL database which had 250+ tables.
748 
749         retval.put(DATABASE_PRODUCT_NAME, metaData.getDatabaseProductName());
750         retval.put(DATABASE_PRODUCT_VERSION, metaData.getDatabaseProductVersion());
751         retval.put(DATABASE_MAJOR_VERSION, metaData.getDatabaseMajorVersion());
752         retval.put(DATABASE_MINOR_VERSION, metaData.getDatabaseMinorVersion());
753 
754         return retval;
755     }

800                                                           final String dbName,
801                                                           final String tableName) {
802         if (StringUtils.isEmpty(tableName)) {
803             return new LinkedHashMap<>();
804         }

806             List<String> tableParts = parseTableName(dbName, tableName);
807             String tableNamePart = tableParts.get(2);

820                     if (StringUtils.equalsIgnoreCase(tableMeta.getString(TABLE_NAME),

882                                                final Document document, final Document schema) {
883         // TODO tstack - Remove this older code to generate the 'CREATE TABLE' statement when
884         // the newer stuff is ready-to-use
885         StringBuilder buffer = new StringBuilder();
886         buffer.append(CREATE_TABLE_COMMAND);
887 
888         if (useUpperCaseIdentifiers()) {
889             buffer.append(tableName);
890         } else {
891             buffer.append(quoteIdentifier(tableName, 0));
892         }
893         databaseProductName = getDatabaseProductName(conn);
894         buffer.append(SPACE_CHAR);
895         buffer.append(BEGIN_PARENTHESES);

900             data = document.get(Map.class);
901         } else {

905         for (String field : data.keySet()) {

929

934

967     /**
968      * Ensures that the columns in the schema have types that are understood by the target
969      * database implementation.
970      *
971      * @param conn
972      * @param schemaData
973      * @param columns    The columns in the database.
974      * @return The column names.
975      */
976     private Set<Object> checkSchemaTypes(final Connection conn,
977                                          final Map<String, Object> schemaData,
978                                          final Map<String, Map<String, Object>> columns) {
979         String targetDatabaseName;
980         Object srcDatabaseName;
981         Set<Object> retval = new HashSet<>();
982 
983         srcDatabaseName = schemaData.get(DATABASE_PRODUCT_NAME);
984         try {
985             targetDatabaseName = conn.getMetaData().getDatabaseProductName();
986         } catch (SQLException e) {
987             throw new ExecutionException(e, UNABLE_TO_GET_DATABASE_METADATA)
988                     .withReason(JdbcOperationsImpl.getSQLErrorMessage(e))
989                     .withResolution(e.getMessage());
990         }
991 
992         try {
993             // Do simple conversions of the java.sql.Types for columns.  For example,
994             // databases that don't support TINYINT probably support SMALLINT.
995             for (WalkResult walkResult : COLUMN_DATA_TYPES.walk(columns)) {
996                 String altJavaType = javaSqlTypeMap.get(walkResult.getValue());
997 
998                 if (altJavaType == null) {
999                     continue;
1000                 }
1001 
1002                 walkResult.setValue(altJavaType);
1003             }
1004         } catch (InvalidPathException e) {
1005             throw new ExecutionException(e, UNABLE_TO_WALK_COLUMN_DATA_TYPES)
1006                     .withResolutionAsDefect();
1007         }
1008 
1009         // Make sure the column types from the source database are valid in the target.
1010         for (Map<String, Object> column : columns.values()) {
1011             String typeName = (String) column.get(TYPE_NAME);
1012 
1013             retval.add(column.get(COLUMN_NAME));
1014             if (typeName == null || !typeInfo.containsKey(typeName.toLowerCase())) {
1015                 // We should be able to use the corresponding JDBC type in the target.
1016                 String dataType = (String) column.get(DATA_TYPE);
1017                 Object columnName = column.get(COLUMN_NAME);
1018                 log.debug(COLUMN_HAS_UNKNOWN_TYPE, columnName, typeName, dataType);
1019                 if (targetDatabaseName.equals(srcDatabaseName)) {
1020                     log.debug(SRC_AND_TARGET_ARE_THE_SAME);
1021                 } else if (UNSUPPORTED_JAVA_SQL_TYPES.contains(dataType.toUpperCase()) ||
1022                         unsupportedJavaSqlTypes.contains(dataType.toUpperCase())) {
1023                     throw new ExecutionException(UNABLE_TO_CONVERT_COLUMN)
1024                             .formatWith(typeName, columnName, dataType)
1025                             .withReason(TARGET_DATABASE_DOES_NOT_SUPPORT_THE_GIVEN_DATA_TYPE)
1026                             .withResolution(UNABLE_TO_CONVERT_RESOLUTION);
1027                 } else {
1028                     column.put(TYPE_NAME, dataType);
1029                 }
1030             }
1031         }
1032         return retval;
1033     }

1058     /**
1059      * Adds fields mentioned in a document to the schema for a to-be-created table.
1060      * The most basic flow for replicating a table just connects a Select snap to
1061      * an Insert snap.  However, if the user adds new fields to the rows that are
1062      * being inserted, then we should automatically update the schema to accommodate
1063      * the new fields.
1064      *
1065      * @param tableParts  The table category, schema, and name.
1066      * @param data        The document data.
1067      * @param columns     The columns in the schema.
1068      * @param columnNames The columnNames in the schema.
1069      */
1070     private void addDocFieldsToSchema(final List<String> tableParts, final Map<String, Object> data,
1071                                       final Map<String, Map<String, Object>> columns,
1072                                       final Set<Object> columnNames) {
1073         // Check for fields that are in the doc, but not in the schema.
1074         for (String field : data.keySet()) {
1075             if (columnNames.contains(field)) {
1076                 continue;
1077             }
1078 
1079             // TODO tstack - Can we get the input view schema and use that to determine
1080             // the types instead of an actual value?
1081             Object value = data.get(field);
1082             ColumnInfoBuilder columnInfo = getColumnInfo(value);
1083             if (columnInfo != null) {
