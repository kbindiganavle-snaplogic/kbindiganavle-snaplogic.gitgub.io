100     public final List<String> getOptionArguments(Object keyValue) {
101         List<String> arguments = new ArrayList<>(1);
102         // no Tableau requirement for command option types other than string or boolean.
103         if (keyValue instanceof String) {
104             String stringKeyValue = (String) keyValue;
105             if (!stringKeyValue.isEmpty()) {
106                 arguments.add(TableauCommand.ARG_KEY_IDENTIFIER + argumentString);
107                 // set to configured snap property value
108                 arguments.add(stringKeyValue);
109             }
110         } else if (keyValue instanceof Boolean) {
111             // boolean snap key values used as toggle for the command option
112             Boolean booleanKeyValue = (Boolean) keyValue;
113             if (booleanKeyValue) {
114                 arguments.add(TableauCommand.ARG_KEY_IDENTIFIER + argumentString);
115             }
116         }
117         return arguments;
118     }
