93                 Object> statusLine, Map<String, Object> headersResult) {
94             Document refinedDoc = removeCredentials(document);
95             int code = (int) statusLine.get(STATUS_CODE);
96             Map<String, Object> response = new HashMap<>(3);
97             response.put(KEY_STATUS_LINE, statusLine);
98             response.put(KEY_ENTITY, content);
99             response.put(KEY_HEADERS, headersResult);
100             if (code == HttpStatus.SC_OK) {
101                 if (!isErrorResponse(content)) {

106             }
107             errorViews.write(documentUtility.newDocumentFor(document,
108                     getErrorResponse(statusLine, content, headersResult)), refinedDoc);
109         }

345     protected boolean isErrorResponse(Object content) {
346         try {
347             Map<String, Object> responseMap = null;
348             if (content instanceof Map) {
349                 responseMap = (Map) content;
350             } else if (content instanceof String) {
351                 responseMap = objectMapper.readValue((String) content, MAP_TYPE_REFERENCE);
352             } else {
353                 throw new ExecutionException(UNEXPECTED_RESPONSE_TYPE)
354                         .formatWith(content == null ? EMPTY_RESPONSE : content.getClass().getName())
355                         .withResolutionAsDefect();
356             }
357             if (responseMap != null) {
358                 Object attr = responseMap.get(RESP_ATTR_CODE);
359                 int code = ((attr != null) ? ((BigInteger) attr).intValue() : 0);
360                 if (code >= RESP_CODE_CLIENT_ERROR) {
361                     return true;
362                 }
363             }
364         } catch (IOException e) {
365             LOGGER.debug(ERROR_PARSING_RESPONSE);
366             return false;
367         }
368         return false;
369     }
