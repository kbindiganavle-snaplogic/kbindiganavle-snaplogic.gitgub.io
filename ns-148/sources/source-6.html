274     @Override
275     protected void process(final Document document, final String inputViewName) {
276         if (isStopped) {
277             // STOP button is pressed, do not submit the PySpark job
278             return;
279         }
280         this.document = document;
281         String sparkHome = spartHomeExpr.eval(document);
282         String sparkSubmitArgs = evalExpressionProperty(sparkSubmitArgsExpr, document);
283         KerberosConfig conf = account.connect();
284         if (conf != null) {
285             sparkSubmitArgs += " " + String.format(KRB_CONFIG_ARGS_FMT,
286                     conf.getKeytabFile(), conf.getClientPrincipal());
287         }
288         String scriptArgs = evalExpressionProperty(scriptArgsExpr, document);
289         String sparkSubmitPath = getSparkSubmitPath(sparkHome);
290         yarnRM = evalExpressionProperty(yarnRmExpr, document);
291         Optional<Long> timeoutOptional = evalTimeoutProperty(timeoutSecExpr, document);
292         long timeoutSec = timeoutOptional.isPresent() ? timeoutOptional.get() : -1;
293         try {
294             createScriptFile();
295             if (sparkSubmitPath != null && scriptFile != null) {
296                 // execute spark-submit
297                 String sparkSubmitCommand = String.format(SPARK_SUBMIT_FORMAT, sparkSubmitPath,
298                         sparkSubmitArgs, scriptPath, scriptArgs);
299                 String cmdForLog = sparkSubmitCommand;
300                 if(conf != null) {
301                     // Mask the keytab file path
302                     cmdForLog = cmdForLog.replaceAll(conf.getKeytabFile(), KEY_TAB_MASK);
303                 }
304                 LOG.debug("Command being executed: {}", cmdForLog);
305 
306                 process = executeSparkSubmit(sparkSubmitCommand);
307                 isSubmitted = true;
308                 // handle the spark-submit result
309                 if (process != null) {
310                     Integer exitCode;
311                     try {
312                         if (timeoutSec < 0) {
313                             exitCode = process.waitFor();
314                         } else {
315                             if (process.waitFor(timeoutSec, TimeUnit.SECONDS)) {
316                                 exitCode = process.exitValue();
317                             } else {
318                                 errorViews.write((SnapDataException)
319                                         new SnapDataException(ERR_TIMEOUT)
320                                                 .formatWith(timeoutSec)
321                                                 .withReason(REASON_TIMEOUT)
322                                                 .withResolution(RESOLUTION_ADDRESS_ISSUE), document);
323                                 return;
324                             }
325                         }
326                     } catch (InterruptedException e) {
327                         // STOP button is pressed
328                         LOG.debug("Interrupted while waiting for PySpark job to complete", e);
329                         return;
330                     }
331                     String standardOutput = getStandardOutput(process);
332                     if (exitCode == 0) {
333                         Map<String, Object> output = new LinkedHashMap<>();
334                         output.put(STATUS, COMPLETED);
335                         putAppMetadata(output);
336                         if (StringUtils.isNotBlank(standardOutput)) {
337                             output.put(STANDARD_OUTPUT, standardOutput);
338                         }
339                         outputViews.write(documentUtility.newDocument(output), document);
340                         return;
341                     }
342                     if (isStopped) {
343                         // If STOP button has been pressed while waiting for the process, the error
344                         // document should not be written to the error view
345                         return;
346                     }
347                     // PySpark script failed
348                     Map<String, Object> errData = new LinkedHashMap<>();
349                     putAppMetadata(errData);
350                     errData.put(ERROR, ERR_PYSPARK_SCRIPT);
351                     String reason = String.format(EXIT_CODE_FORMAT, exitCode);
352                     errData.put(REASON, reason);
353                     errData.put(RESOLUTION, RESOLUTION_ADDRESS_ISSUE);
354                     String errorOutput = getErrorOutput(process);
355                     if (StringUtils.isNotBlank(errorOutput)) {
356                         errData.put(ERROR_OUTPUT, errorOutput);
357                     }
358                     if (StringUtils.isNotBlank(standardOutput)) {
359                         errData.put(STANDARD_OUTPUT, standardOutput);
360                     }
361                     errData.put(SPARK_SUBMIT_COMMAND_KEY, cmdForLog);
362                     Document errDoc = documentUtility.newDocument(errData);
363                     SnapDataException ex = new SnapDataException(errDoc, ERR_PYSPARK_SCRIPT);
364                     errData.put(STACKTRACE, Throwables.getStackTraceAsString(ex));
365                     errorViews.write(ex, document);
366                 } else {
367                     errorViews.write((SnapDataException) new SnapDataException(ERR_PYSPARK_SCRIPT)
368                             .withReason(REASON_PYSPARK_SUBMIT)
369                             .withResolution(RESOLUTION_ADDRESS_ISSUE), document);
370                 }
371             }
372         } catch (SnapDataException sde) {
373             errorViews.write(sde);
374         } finally {
375             cleanupPySpark();
376         }
377     }

388     public void handle(final LifecycleEvent event) {
389         switch (event) {
390             case STOP:
391                 isStopped = true;
392                 if (isSubmitted) {
393                     if (StringUtils.isNotBlank(yarnRM)) {
394                         try {
395                             Optional<String> appIdOptional;
396                             int countDown = 7; // 3-second interval x 7 = total 21 seconds
397                             do {
398                                 // there is a delay for a submitted PySpart job to appear as an
399                                 // application in YARN resource manager
400                                 sleep(3, TimeUnit.SECONDS);
401                                 appIdOptional = getApplicationId();
402                                 if (--countDown == 0) {
403                                     LOG.info("Could not find the application ID for the " +
404                                             "submitted PySpark job");
405                                     return;
406                                 }
407                             } while (!appIdOptional.isPresent());
408                             String applicationId = appIdOptional.get();
409                             if (StringUtils.isNotBlank(applicationId)) {
410                                 stopPySaprkJob(httpClient, applicationId);
411                             }
412                         } catch (IOException e){
413                             errorViews.write((SnapDataException)
414                                     new SnapDataException(e, ERR_STOP_PYSPARK)
415                                             .withResolution(RESOLUTION_ADDRESS_ISSUE), document);
416                         } finally {
417                             cleanupPySpark();
418                         }
419 
420                     } else {
421                         statusReporter.notify(NO_RESOURCE_MANAGER);
422                     }
423                 }
424         }
425     }

456     protected Process executeSparkSubmit(String command) {
457         try {
458             if (StringUtils.isNotBlank(command) && StringUtils.isNotBlank(scriptPath)) {
459                 Process process =  Runtime.getRuntime().exec(command);
460                 stdoutListener = new StandardOutputListener(process,
461                         StandardOutputListener.OutputType.STD_OUT,
462                         Thread.currentThread().getName());
463                 stderrListener = new StandardOutputListener(process,
464                         StandardOutputListener.OutputType.STD_ERR,
465                         Thread.currentThread().getName());
466                 stderrListener.start();
467                 stdoutListener.start();
468                 return process;
469             }
470         } catch (SuggestViewAbortException e) {
471             throw e;
472         } catch (Exception e) {
473             throw new SnapDataException(e, ERR_SUBMIT_PYSPARK)
474                     .withResolution(RESOLUTION_ADDRESS_ISSUE);
475         }
476         return null;
477     }

565     /**
566      * Since spark-submit does not give the application ID of the submitted PySpark job, this
567      * method finds it with the name of the script file from YARN. The HTTP GET request can
568      * retrieve the list of application metadata. The 'name' field is the script file name and the
569      * 'id' field is the application ID. With this ID value, the Snap can stop the job.
570      *
571      * @return Optional.empty() if it needs to retry,
572      * Optional.of(empty string) if host:port is  invalid or the job has been stopped or compeleted,
573      * Optional.of(valid ID string) if it finds the matching name field.
574      *
575      * @throws IOException if it fails during HTTP request processing or parsing the response JSON
576      */
577     private Optional<String> getApplicationId() throws IOException {
578         Optional<Map<String, Object>> optionalApp = getAppMetadata();
579         if (optionalApp.isPresent()) {
580             Map<String, Object> application = optionalApp.get();
581             if (application.isEmpty()) {
582                 return Optional.of(StringUtils.EMPTY);
583             }
584             switch ((String) application.get(STATE)) {
585                 case FINISHED:
586                 case FAILED:
587                 case KILLED:
588                     // already done, no need to stop
589                     return Optional.of(StringUtils.EMPTY);
590                 default:
591                     // application ID found
592                     return Optional.of((String) application.get(ID));
593             }
594         }
595         return Optional.empty();
596     }

598     private Optional<Map<String, Object>> getAppMetadata() {
599         try {
600             if (httpClient == null) {
601                 httpClient = HttpClients.createSystem();
602             }
603             // execute REST Get to YARN
604             HttpGet request = new HttpGet(String.format(URL_YARN_APPS, yarnRM));
605             request.addHeader(ACCEPT, APPLICATION_JSON);
606             HttpResponse httpResponse = httpClient.execute(request);
607             StatusLine statusLine = httpResponse.getStatusLine();
608             int statusCode = statusLine.getStatusCode();
609             if (statusCode != HttpStatus.SC_OK) {
610                 String reason = String.format(FORMAT_HTTP_ERROR, statusLine.getReasonPhrase(),
611                         statusCode, getEntityString(httpResponse));
612                 errorViews.write((SnapDataException)
613                         new SnapDataException(ERR_RESOURCE_MANAGER_APPS)
614                                 .formatWith(yarnRM)
615                                 .withReason(reason)
616                                 .withResolution(RESOLUTION_ADDRESS_ISSUE), document);
617                 return Optional.of(new HashMap<>());
618             }
619             try (InputStream inputStream = httpResponse.getEntity().getContent()) {
620                 // parse JSON to get the list of applications in Resource Manager
621                 Map<String, Object> result = MAPPER.readValue(inputStream,
622                         new TypeReference<Map<String, Object>>() {});
623                 List<Map<String, Object>> apps = JsonPaths.compileStatic(JSON_PATH_YARN_APPS)
624                         .readStatic(result);
625                 for (Map<String, Object> appliaction : apps) {
626                     if (scriptFile.getName().equals(appliaction.get(NAME))) {
627                         return Optional.of(appliaction);
628                     }
629                 }
630             }
631             return Optional.empty();
632         } catch (IOException e) {
633             errorViews.write((SnapDataException)
634                     new SnapDataException(e, ERR_RESOURCE_MANAGER_APPS)
635                             .formatWith(yarnRM)
636                             .withResolution(RESOLUTION_ADDRESS_ISSUE), document);
637             return Optional.of(new HashMap<>());
638         }
639     }
