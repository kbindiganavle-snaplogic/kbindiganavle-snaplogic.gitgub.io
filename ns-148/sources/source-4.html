19 /**

25 public class LineageTracker {

27     private final DocumentUtility documentUtility;
28     private final ScriptHookInput scriptHookInput;
29 
30     public LineageTracker(DocumentUtility documentUtility, ScriptHookInput scriptHookInput) {
31         this.documentUtility = documentUtility;
32         this.scriptHookInput = scriptHookInput;
33     }
34 
35     /**
36      * Returns the output document with the lineage entry if possible.
37      *
38      * @param data
39      * @return outputDocument
40      */
41     public Document output(Object data) {
42         final Document document;
43         Document incomingDocument = scriptHookInput.getInputDocument();
44         if (incomingDocument != null) {
45             // Pass the lineage
46             document = documentUtility.newDocumentFor(incomingDocument, data);
47         } else {
48             document = documentUtility.newDocument(data);
49         }
50         return document;
51     }
52 
53     /**
54      * Returns the header document that should be sent along with the binary output objects.
55      *
56      * @return header
57      */
58     public Document getHeaderDocument() {
59         // We return a empty map as the header content. This mainly transfers the lineage
60         // information to the header document.
61         return documentUtility.newDocumentFor(scriptHookInput.getInputDocument(), new HashMap());
62     }
