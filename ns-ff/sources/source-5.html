196     @Override
197     protected void process(Document document, String s) {
198         try {
199             teamID = utils.extractID(expTeamID.eval(document));
200             if (!LIST_TEAM.equals(operation)) {
201                 utils.validateReqField(TEAMID_LABEL, teamID);
202             }
203             switch (operation) {
204                 case ADD_MEMBER:
205                     invokeAddMember(document, ERR_ADD_T_MEMBER, T_NOTHING_TO_ADD_RES,
206                             T_ADD_MEMBER_MSG, teamID);
207                     break;
208                 case REMOVE_MEMBER:
209                     invokeRemoveMember(document, ERR_REMOVE_T_MEMBER, T_NOTHING_TO_REMOVE_RES,
210                             T_REM_MEMBER_MSG,teamID);
211                     break;
212                 case LIST_MEMBER:
213                     invokeListMember(document, T_ERR_LIST_MEMBER);
214                     break;
215                 case LIST_TEAM:
216                     invokeListTeam(document);
217                     break;
218                 case LIST_CHANNEL:
219                     invokeListChannel(document);
220                     break;
221                 case RENAME_TEAM:
222                     invokeRenameTeam(document);
223                     break;
224                 case DELETE_TEAM:
225                     Failsafe.with(createRetryPolicy(document))
226                             .onFailedAttempt(e -> LOG.warn(ERR_DELETE_TEAM, e))
227                             .run(context -> {
228                                 logRetryAttempt(context);
229                                 invokeDeleteTeam(document);
230                             });
231                     break;
232                 case ARCHIVE_TEAM:
233                     Failsafe.with(createRetryPolicy(document))
234                             .onFailedAttempt(e -> LOG.warn(ERR_ARCHIVE_TEAM, e))
235                             .run(context -> {
236                                 logRetryAttempt(context);
237                                 invokeArchiveTeam(document);
238                             });
239                     break;
240                 case UNARCHIVE_TEAM:
241                     Failsafe.with(createRetryPolicy(document))
242                             .onFailedAttempt(e -> LOG.warn(ERR_UNARCHIVE_TEAM, e))
243                             .run(context -> {
244                                 logRetryAttempt(context);
245                                 invokeUnarchiveTeam(document);
246                             });
247                     break;
248                 default:
249                     // should not reach here
250                     throw new SnapDataException(NOT_SUPPORT_MSG)
251                             .withReason(NOT_SUPPORT_REASON)
252                             .withResolution(NOT_SUPPORT_RES);
253             }
254         } catch (SnapDataException snapEx) {
255             throw snapEx;
256         } catch (Throwable ex) {
257             LOG.error(String.format(ERR_GENERAL_MSG, operation), ex);
258             String reason = (ex.getCause() != null && ex.getCause().getMessage() != null) ?
259                     ex.getCause().getMessage() : ex.getMessage();
260             throw new SnapDataException(ex, String.format(ERR_GENERAL_MSG, operation))
261                     .withReason(reason)
262                     .withResolution(ERR_COMMON_RESOLUTION);
263         }
264     }
