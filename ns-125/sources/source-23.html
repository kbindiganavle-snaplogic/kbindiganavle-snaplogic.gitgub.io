773     private List<Map<String, Object>> getCustomFieldRecords(List<Map<String, Object>> customizationRefList,
774                                                             String customFieldType,
775                                                             NetSuiteBaseAccount account) {
776         // process getCustomizationId response to build the list to pass for the getList operation
777         final String cacheKeyPrefix = account.getAccountId() + ':' + customFieldType + ':';
778         List<Map<String, Object>> customFieldRecords = new ArrayList<>(customizationRefList.size());
779         List<Map<String, Object>> internalIds = new ArrayList<>(customizationRefList.size());
780         for (Map<String, Object> customizationRef : customizationRefList) {
781             String internalId = (String) customizationRef.get(RESP_INTERNAL_ID);
782             String cacheKey = cacheKeyPrefix + internalId;
783             Map<String, Object> customFieldRecord =
784                     CUSTOM_FIELD_RECORD_CACHE.getIfPresent(cacheKey);
785             if (customFieldRecord != null) {
786                 customFieldRecords.add(customFieldRecord);
787             } else {
788                 internalIds.add(ImmutableMap.of(
789                         INTERNAL_ID, internalId,
790                         TYPE, customFieldType));
791             }
792         }
793         if (!internalIds.isEmpty()) {
794             // call getList operation
795             List<Map<String, Object>> readResponseList = callGetList(internalIds, account);
796 
797             // process getList response
798             for (Map<String, Object> readResponse : readResponseList) {
799                 Map<String, Object> customFieldRecord = (Map<String, Object>) readResponse.get("record");
800                 checkNotNull(customFieldRecord, "customFieldRecord");
801                 customFieldRecords.add(customFieldRecord);
802                 String internalId = (String) customFieldRecord.get(RESP_INTERNAL_ID);
803                 checkNotNull(internalId, "internalId");
804                 String cacheKey = cacheKeyPrefix + internalId;
805                 CUSTOM_FIELD_RECORD_CACHE.put(cacheKey, customFieldRecord);
806             }
807         }
808         return customFieldRecords;
809     }

854                                              OperationType operationType,
855                                              String sublistName,
856                                              String sublistItemName) {
857         // needs a structure conforming to the object definition for the property
858         // {
859         //   // The "label" field defines the display name of the suggestion...
860         //   "label" : {
861         //     "value" : "some label"
862         //   },
863         //   // The remaining fields are hidden...
864         //   "internalId": {
865         //     "value": "some id"
866         //   },
867         //   "fieldType": {
868         //     "value": "some type name"
869         //   },
870         //   "sublistName": { // only added if sublistName != null
871         //     "value", "some name"
872         // }
873 
874         String label = customFieldDef.getScriptId();
875         if (sublistItemName != null) {
876             label = sublistItemName + '.' + label;
877         }
878 
879         Map<String, Object> suggestion = Maps.newHashMap();
880         PATH_LABEL.writeStatic(suggestion, label);
881         PATH_INTERNAL_ID.writeStatic(suggestion, customFieldDef.getInternalId());
882         PATH_FIELD_TYPE.writeStatic(suggestion, getValueType(customFieldDef, operationType));
883         if (sublistName != null) {
884             PATH_SUBLIST_NAME.writeStatic(suggestion, sublistName);
885         }
886         return suggestion;
