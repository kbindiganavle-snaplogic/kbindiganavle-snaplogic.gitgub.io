184     private WsdlVersion configureWsdlVersion(PropertyValues propertyValues) {
185         WsdlVersion minWsdlVersion = getMinWsdlVersion();
186         WsdlVersion wsdlVersion = new WsdlVersion();
187         Integer year = getStringAsInt(propertyValues, PROP_WSDL_VERSION_YEAR, WSDL_VERSION_YEAR_LABEL, minWsdlVersion.getYear());
188         Integer release = getStringAsInt(propertyValues, PROP_WSDL_VERSION_RELEASE, WSDL_VERSION_RELEASE_LABEL, 1);
189         Integer patch = getStringAsInt(propertyValues, PROP_WSDL_VERSION_PATCH, WSDL_VERSION_PATCH_LABEL, 0);
190         wsdlVersion.year(checkNotNull(year, PROP_WSDL_VERSION_YEAR));
191         // release and patch are new properties, so might be null if the account hasn't been updated
192         if (release != null) {
193             wsdlVersion.release(release);
194         }
195         if (patch != null) {
196             wsdlVersion.patch(patch);
197         }
198         if (wsdlVersion.compareTo(minWsdlVersion) < 0) {
199             throw new ConfigurationException(String.format(MIN_WSDL_VERSION_MSG, minWsdlVersion))
200                     .withReason(String.format(MIN_WSDL_VERSION_REASON, wsdlVersion))
201                     .withResolution(String.format(MIN_WSDL_VERSION_RESOLUTION, minWsdlVersion));
202         }
203         return wsdlVersion;
204     }

217     static void checkWsdlExists(String wsdlUrl) {
218         if (wsdlUrlCheckOkList.contains(wsdlUrl)) {
219             return;
220         }
221         HttpURLConnection connection = null;
222         IOException cause = null;
223         try {
224             URLConnection urlConn = new URL(wsdlUrl).openConnection();
225             if (!(urlConn instanceof HttpURLConnection)) {
226                 // In a TestFixture that doesn't specify a customStreamHandler, urlConn will be a
227                 // MockUrlConnection, so skip the rest.
228                 return;
229             }
230             connection = (HttpURLConnection) urlConn;
231             // our custom handlers for http/s implement HttpStreamHandler
232             if (!(connection instanceof HttpStreamHandler)) {
233                 // this will only work for the default handler; our custom handler doesn't support HEAD
234                 connection.setRequestMethod("HEAD");
235             }
236             int code = connection.getResponseCode();
237             String contentType = connection.getHeaderField("Content-Type");
238             // For NetSuite, it's important to check the response's content type, since currently,
239             // an invalid WSDL URL redirects to a "Page Not Found" HTML document with an OK status.
240             if (code == HttpStatus.SC_OK && contentType.contains("/xml")) {
241                 // The URL references a valid XML document.
242                 wsdlUrlCheckOkList.add(wsdlUrl);
243                 return;
244             }
245         } catch (IOException e) {
246             cause = e;
247         } finally {
248             if (connection != null) {
249                 connection.disconnect();
250             }
251         }
252         throw new ConfigurationException(cause, String.format(WSDL_NOT_FOUND_MSG, wsdlUrl))
253                 .withReason(String.format(WSDL_NOT_FOUND_REASON, wsdlUrl))
254                 .withResolution(WSDL_NOT_FOUND_RESOLUTION);
255     }

294     public boolean handleMessage(SOAPMessageContext context) {
295         Boolean outBound = (Boolean) context.get(MessageContext.MESSAGE_OUTBOUND_PROPERTY);
296         if (outBound != null && outBound) {
297             SOAPMessage message = context.getMessage();
298             try {
299                 SOAPPart soapPart = message.getSOAPPart();
300                 SOAPEnvelope envelope = soapPart.getEnvelope();
301                 SOAPHeader header = envelope.getHeader();
302                 if (header != null) {
303                     header.detachNode();
304                 }
305                 header = envelope.addHeader();
306                 buildSOAPHeader(header, envelope);
307                 soapHelper.logRequestMessage(message);
308             } catch (IOException | SOAPException e) {
309                 throw new SnapDataException(e, ERROR_WHILE_CONFIGURING_SOAP_HEADER)
310                         .withResolutionAsDefect();
311             }
312         }
313         return true;
314     }

420     public Void validate(Void arg) {
421         ClientBuilder clientBuilder = createClientBuilder(GET_SERVER_TIME);
422         try {
423             SOAPMessage requestMessage = buildRequestMessageForGetServerTime();
424             Document responseDoc = send(requestMessage, clientBuilder);
425             Node node = getNode(responseDoc, XPATH_FOR_STATUS);
426             if (node != null && TRUE.equals(node.getNodeValue())) {
427                 return null;
428             } else {
429                 node = getNode(responseDoc, XPATH_FOR_STATUS_DETAIL);
430                 throw new ExecutionException(INVALID_ACCOUNT)
431                         .withReason(node.getNodeValue())
432                         .withResolution(PLEASE_VERIFY_CREDENTIALS);
433             }
434         } catch (SOAPFaultException | SOAPException e) {
435             throw new ExecutionException(e, ERROR_WHILE_EXECUTING_SOAP_REQUEST)
436                     .withReason(Throwables.getRootCause(e).getMessage())
437                     .withResolution(INVALID_INPUT_DATA);

439             throw new ExecutionException(e, ERROR_WHILE_PARSING_RESPONSE_DOCUMENT)
440                     .withReason(Throwables.getRootCause(e).getMessage())
441                     .withResolution(INVALID_INPUT_DATA);

458                     .withReason(Throwables.getRootCause(e).getMessage())
459                     .withResolution(INVALID_INPUT_DATA);
