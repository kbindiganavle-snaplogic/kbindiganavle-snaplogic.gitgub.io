128                                   final DatabaseAccount account, boolean isPlainQuery
129                                     , int maxRetries, long retryInterval)
130                 throws BatchUpdateException {
131 
132         final List<String> dml =
133                 Arrays.asList("WITH", "INSERT", "UPDATE", "DELETE", "MERGE", "UPSERT", "CALL",
134                         "DESC", "SELECT");
135         final String[] words = statement.trim().toUpperCase().split(" ");
136         if (dml.contains(words[0])) {
137             return executeDMLStatement(statement, bindValues, account
138                 , isPlainQuery, maxRetries, retryInterval);
139         } else {
140             try {
141                 return new int[] { executeDDLStatement(statement, account) };
142             } catch (SQLException e) {
143                 throw new BatchUpdateException(e);
144             }
145         }

172     /**
173      * Execute DML statement. These statements can be batched.
174      *
175      * @param statement
176      * @param bindValues
177      * @param account
178      * @param isPlainQuery
179      * @param maxRetries
180      * @param retryInterval
181      * @return
182      * @throws BatchUpdateException
183      */
184     @SuppressWarnings("deprecation")
185     protected int[] executeDMLStatement(final String statement, final List<Object> bindValues,
186                                         final DatabaseAccount account, boolean isPlainQuery
187                                         , int maxRetries, long retryInterval)
188                             throws BatchUpdateException {
189 
190         final DatabaseConfig databaseConfig = account.getDatabaseConfig();
191         final SQLDialect dialect = databaseConfig.getSQLDialect();
192         final Settings settings = databaseConfig.getSettings();
193         if (SQLDialect.SQL99.equals(dialect)) {
194             settings.withStatementType(StatementType.STATIC_STATEMENT);
195         }
196         acquireConnection(account);
197         executeConfProperties(confProperties);
198         if (isPlainQuery) {
199                 final Query query;
200                 final DSLContext context = DSL.using(conn, dialect, settings);
201                 query = context.query(statement);
202                 final int[] returnArray = new int[1];
203                 returnArray[0] = query.execute();
204 
205                 // Statements return 0 when successfully executed with Snowflake, which would cause
206                 // problem later in DatabaseUtils.handleUpdateResult(). Apache Hive succeeds with
207                 // -1, so this won't affect it;
208                 returnArray[0] = returnArray[0] == 0 ? 1 : returnArray[0];
209                 return returnArray;
210         } else {
211             if (batch == null) {
212                 final DSLContext context = DSL.using(conn, dialect, settings);
213                 final Query query = context.query(statement, bindValues.toArray());
214                 batch = context.batch(query);
215             }
216             ((BatchBindStep) batch).bind(bindValues.toArray());
217             if (batch.size() == databaseConfig.getBatchSize()) {
218                 return flushBatch(account, maxRetries, retryInterval);
219             }
220         }
221         return null;
222     }

224     /**
225      * Set configuration properties.
226      */
227     protected void executeConfProperties(Map<String, String> confProperties) {
228         try (Statement stmt = conn.createStatement()) {
229             for (Map.Entry<String, String> entry : confProperties.entrySet()) {
230                 if (!entry.getKey().matches(PROP_NAME_PATTERN)) {
231                     throw new ConfigurationException(ERR_INVALID_PROPERTY_NAME);
232                 }
233                 String sql = String.format("set %s = %s", entry.getKey(), entry.getValue());
234                 if (entry.getValue().matches(".*\".")) {
235                     sql = String.format("set %s = \"%s\"", entry.getKey(),
236                              entry.getValue().replaceAll("\"", "\\\""));
237                 }
238                 try {
239                     stmt.execute(sql);
240                 } catch (SQLException e) {
241                     throw new ConfigurationException(e,
242                             String.format(ERR_CANT_SET_PROPERTY_FMT, entry.getKey()));
243                 }
244             }
245         } catch (SQLException e) {
246             throw new ConfigurationException(e, ERR_NO_CONNECTION);
247         }
248     }

261     public int[] executeStatement(final String statement, final List<Object> bindValues,
262             final DatabaseAccount account, boolean isPlainQuery,
263             Map<String, String> confProperties) {
264 
265         final DatabaseConfig databaseConfig = account.getDatabaseConfig();
266         final SQLDialect dialect = databaseConfig.getSQLDialect();
267         final Settings settings = databaseConfig.getSettings();
268         settings.withStatementType(StatementType.STATIC_STATEMENT);
269 
270         acquireConnection(account);
271         executeConfProperties(confProperties);
272 
273         final DSLContext context = DSL.using(conn, dialect, settings);
274         Query query;
275         if (isPlainQuery) {
276             query = context.query(statement);
277         } else {
278             query = context.query(statement, bindValues.toArray());
279         }
280 
281         // Apache Hive succeeds with -1.
282         final int[] returnArray = new int[1];
283         returnArray[0] = query.execute();
284         return returnArray;
285     }

288     public DatabaseCursor select(final SelectBuilder selectBuilder, final DatabaseAccount account,
289             int maxRetries, long retryInterval) {
290         try {
291             acquireConnection(account);
292             DatabaseConfig databaseConfig = account.getDatabaseConfig();
293             SQLDialect dialect = databaseConfig.getSQLDialect();
294             Settings settings = databaseConfig.getSettings();
295             DSLContext context = DSL.using(conn, dialect, settings);
296             // The standard configuration quotes identifiers with a double quote
297             // and that's interpreted as a string literal by hive.
298             context.configuration().settings().withRenderNameStyle(RenderNameStyle.AS_IS);
299             //selectQuery = selectBuilder.build(context.configuration()).toString();
300             //return executeQuery(selectQuery, null, account);
301             handleAutoCommit();
302             int fetchSize = databaseConfig.getFetchSize();
303             Cursor<Record> cursor = fetchLazy(context, selectBuilder.build(context.configuration()),
304                     fetchSize);
305             return getDatabaseCursor(conn, cursor, fetchSize);
306         } catch (ExecutionException | SnapDataException e) {
307             throw e;
308         } catch (DataAccessException e) {
309             throw new ExecutionException(EXECUTE_SELECT_FAILED).withReason(e.getMessage())
310                     .withResolution(CHECK_QUERY).withSuppressed(e);
311         } catch (Throwable e) {
312             throw new ExecutionException(UNEXPECTED_EXCEPTION_OCCURRED).withReason(e.getMessage())
313                     .withResolution(PLEASE_ADDRESS_REPORTED_ISSUE).withSuppressed(e);

323                         final DatabaseAccount account, final TableMetaData tableMetaData,
324                         final DuplicateKeyAction duplicateKeyAction,
325                         final boolean preserveCaseSensitiveness,
326                         final OutputViews outputViews, final ErrorViews errorViews,
327                         final DocumentUtility documentUtility, final List<Document> inputDocuments,
328                         final int maxRetries, final long retryInterval)
329             throws BatchUpdateException {
330         if (document != null) {
331             final DatabaseConfig databaseConfig = account.getDatabaseConfig();
332             final SQLDialect dialect = databaseConfig.getSQLDialect();
333             final Settings settings = databaseConfig.getSettings();
334             final Map<String, Object> data = document.get(Map.class);
335             if (MapUtils.isEmpty(data)) {
336                 return null;
337             }
338 
339             acquireConnection(account);
340             final Map<Field<?>, Object> fieldMap = convertValues(data, tableMetaData, dialect,
341                     preserveCaseSensitiveness);
342 
343             final DSLContext context = DSL.using(conn, dialect, settings);
344             final Query query = getInsertQuery(tableName, duplicateKeyAction, fieldMap, context);
345             // special handling for Hive DB.
346             try {
347                 final String sql = query.getSQL().replace(DOUBLE_QUOTE, BACK_TICK);
348                 return executeQueryDirectly(context, query, sql, maxRetries, retryInterval);
349             } catch (Throwable e) {
350                 DatabaseUtils.writeErrorView(errorViews, inputDocuments, e);
351                 return null;
352             }
353         }
354         return null;
355     }

417     public boolean tableExists(Connection conn, String dbName, String tableName) {
418         final List<String> tableParts = metadataProvider.parseTableName(dbName, tableName);
419         final String catalog = quotationHandler.unescape(tableParts.get(0));
420         final String schema = quotationHandler.unescape(tableParts.get(1));
421         final String table = quotationHandler.unescape(tableParts.get(2));
422 
423         ResultSet resultSet = null;
424         try {
425             final DatabaseMetaData dbMetaData = conn.getMetaData();
426 
427             resultSet = dbMetaData.getTables(catalog, schema, table, TABLE_TYPES);
428             return resultSet.next();
429         } catch (SQLException e) {
430             throw new ExecutionException(RETRIEVE_DATABASE_METADATA_FAILED)
431                     .withReason(getSQLErrorMessage(e)).withResolution(e.getMessage())
432                     .withSuppressed(e);
433         } finally {
434             JDBCUtils.safeClose(resultSet);
435         }
436     }
