155     @Override
156     public int[] executeStatement(final String statement, final List<Object> bindValues,
157                                   final DatabaseAccount account, boolean isPlainQuery,
158                                   final int maxRetries, final long retryInterval)
159             throws BatchUpdateException {
160         int[] result;
161         if (isDDLStatement(statement)) {
162             final DatabaseConfig databaseConfig = account.getDatabaseConfig();
163             if (batch != null && batch.size() > 0) {
164                 flushBatch(account, maxRetries, retryInterval);
165             }
166             final SQLDialect dialect = databaseConfig.getSQLDialect();
167             final Settings settings = databaseConfig.getSettings();
168             acquireConnection(account);
169             final DSLContext context = DSL.using(conn, dialect, settings);
170             List<Object> updatedValues = convertValues(bindValues, dialect);
171             final Query query = context.query(statement,
172                     updatedValues.toArray());
173             try {
174                 result = executeQueryDirectly(context, query, statement, maxRetries,
175                         retryInterval);
176             } catch (DataAccessException dataAccessException) {
177                 if (dataAccessException.getMessage().contains(JOOQ_NOT_SUPPORTED)) {
178                     return executeQueryDirectly(context, query, statement, maxRetries,
179                             retryInterval);
180                 }
181                 Throwable rootCause = Throwables.getRootCause(dataAccessException);
182                 throw new ExecutionException(rootCause, rootCause.getMessage())
183                         .withReason(rootCause.getMessage())
184                         .withResolution(PLEASE_ADDRESS_THE_ISSUE);
185             } catch (Throwable e) {
186                 Throwable cause = e.getCause();
187                 if (cause instanceof BatchUpdateException) {
188                     throw (BatchUpdateException) cause;
189                 }
190                 throw e;
191             }
192             return result;
193         } else {
194             return super.executeStatement(statement, bindValues, account, isPlainQuery,
195                 maxRetries, retryInterval);
196         }
197     }

209     public int[] insert(final String tableName, final Document document,
210                         final DatabaseAccount account, final TableMetaData tableMetaData,
211                         final DuplicateKeyAction duplicateKeyAction,
212                         final boolean preserveCaseSensitiveness,
213                         final OutputViews outputViews, final ErrorViews errorViews,
214                         final DocumentUtility documentUtility, final List<Document> inputDocuments,
215                         final int maxRetries, final long retryInterval)
216             throws BatchUpdateException {
217         Map<String, Object> docData = (Map<String, Object>) document.get();
218         semiStructuredColumns = Lists.newArrayList();
219         for (String key : docData.keySet()) {
220             // If any of Semi-Structured data type is there then we need to have a different
221             // Insert query and also need to keep track of key so we can use proper converter
222             if (StringUtils.equalsIgnoreCase(VARIANT, tableMetaData.getColumnTypeName(key)) ||
223                     StringUtils.equalsIgnoreCase(ARRAY, tableMetaData.getColumnTypeName(key)) ||
224                     StringUtils.equalsIgnoreCase(OBJECT, tableMetaData.getColumnTypeName(key))) {
225                 semiStructuredColumns.add(key);
226             }
227         }
228         if (!semiStructuredColumns.isEmpty()) {
229             return insertWithVariantColumns(tableName, document, account, tableMetaData,
230                     preserveCaseSensitiveness);
231         } else {
232             return super.insert(tableName, document, account, tableMetaData, duplicateKeyAction,
233                     preserveCaseSensitiveness, outputViews, errorViews, documentUtility,
234                     inputDocuments, maxRetries, retryInterval);
235         }

238     /**
239      * Insert the input document that contains Variant column into the target table.
240      *
241      * In Snowflake, the Variant column can only be inserted through :
242      * INSERT INTO table(c1)
243      * SELECT PARSE_JSON(column1) FROM VALUES('{"key":"value"}')
244      *
245      * The statement could not be batched in JOOQ, so we have to directly execute it one by one.
246      */
247     private int[] insertWithVariantColumns(final String tableName, final Document document,
248             final DatabaseAccount account, final TableMetaData tableMetaData,
249             final boolean preserveCaseSensitiveness) {
250         Map<String, Object> docData = (Map<String, Object>) document.get();
251         final DatabaseConfig databaseConfig = account.getDatabaseConfig();
252         final SQLDialect dialect = databaseConfig.getSQLDialect();
253         final Settings settings = databaseConfig.getSettings();
254         DSLContext context = DSL.using(conn, dialect, settings);
255         context.configuration()
256                 .set(new DefaultExecuteListenerProvider(new BatchExecuteListener()));
257         final Map<Field<?>, Object> fieldMap = convertValues(docData, tableMetaData, dialect,
258                 preserveCaseSensitiveness);
259         acquireConnection(account);
260         validateValues(fieldMap);
261         try {
262             Set<Field<?>> fields = fieldMap.keySet();
263             int colIndex = 0;
264             StringJoiner columnNameJoiner = new StringJoiner(COMMA);
265             StringJoiner columnPlaceholderJoiner = new StringJoiner(COMMA);
266             StringJoiner valuePlaceholderJoiner = new StringJoiner(COMMA);
267             for (Field field : fields) {
268                 ++colIndex;
269                 String columnPlaceHolder = String.format(COLUMN_PLACEHOLDER_FMT, colIndex);
270                 columnNameJoiner.add(field.getName());
271                 if (semiStructuredColumns.contains(field.getName())) {
272                     String dataType = tableMetaData.getColumnTypeName(field.getName());
273                     columnPlaceholderJoiner.add(convertColumnForSemiStructuredDataType
274                             (columnPlaceHolder, conn, fieldMap.get(field), dataType));
275                 } else {
276                     columnPlaceholderJoiner.add(columnPlaceHolder);
277                 }
278                 valuePlaceholderJoiner.add(VALUE_PLACEHOLDER);
279             }
280             String insert = String.format(INSERT_INTO_TABLE_FMT, quoteIdentifier(tableName),
281                     columnNameJoiner.toString());
282             String query = String.format(SELECT_FMT, insert, columnPlaceholderJoiner
283                     .toString(), valuePlaceholderJoiner.toString());
284             LOG.debug("Variant inserting query: {}", query);
285             if (Objects.nonNull(queryQueue)) {
286                 DatabaseUtils.submitQueryToThread(query, queryQueue);
287             }
288             final int[] result = {0};
289             context.connection(connection -> {
290                 try (PreparedStatement ps = conn.prepareStatement(query)) {
291                     int bindValueIndex = 0;
292                     for (Field field : fields) {
293                         bindValueToStatement(ps, ++bindValueIndex, fieldMap.get(field));
294                     }
295                     if (!ps.execute()) {
296                         result[0] = ps.getUpdateCount();
297                     }
298                 }
299             });
300             return result;
301         } catch (SQLException e) {
302             throw new SnapDataException(e, ERR_SNOWFLAKE_INSERT_VARIANT)
303                     .withReason(e.getMessage())
304                     .withResolution(RESOLUTION_ADDRESS_ISSUE);
305         }
306     }

309     /**
310      * Wrap the column place holder (e.g. column1, column2 ..) up with the Variant casting
311      * function (PARSE_JSON(), PARSE_XML() or TO_VARIANT()). The function to use would be decided
312      * by the actual value.
313      *
314      * @param columnPlaceHolder The column place holder used by the 'insert into select from
315      *                          values' query.
316      * @param conn              Corresponding database connection to Snowflake
317      * @param value             The actual value for the Variant column
318      * @param dataType          Snowflake Semi structured Data Type
319      *
320      * @return The new place holder with casting function wrapped up
321      * @throws SQLException
322      */
323     private String convertColumnForSemiStructuredDataType(String columnPlaceHolder,
324             Connection conn, Object value, String dataType) throws SQLException {
325         if (value instanceof String && StringUtils.equalsIgnoreCase(dataType, VARIANT)) {
326             switch (verifyVariantType(conn, (String) value)) {
327                 case JSON:
328                     return String.format(PARSE_JSON, columnPlaceHolder);
329                 case XML:
330                     return String.format(PARSE_XML, columnPlaceHolder);
331             }
332         } else if (StringUtils.equalsIgnoreCase(dataType, ARRAY) && ((value instanceof List) ||
333                 (isValueStringRepresentingList(value)))) {
334             return String.format(PARSE_JSON, columnPlaceHolder);
335         } else if (StringUtils.equalsIgnoreCase(dataType, OBJECT)) {
336             return String.format(TO_VARIANT, columnPlaceHolder);
337         }
338 
339         return String.format(TO_VARIANT, columnPlaceHolder);
340     }

373      * Verify the format of the variant value.
374      *
375      * The method is using CHECK_JSON() and CHECK_XML() query to make the decision. If the
376      * checking passed, the checking query would return a null from the result set.
377      *
378      * Refer: https://docs.snowflake.net/manuals/sql-reference/functions/check_json.html ,
379      * https://docs.snowflake.net/manuals/sql-reference/functions/check_xml.html
380      *
381      * @param conn  Database connection to Snowflake
382      * @param value Actual value for the Variant column
383      *
384      * @return The result of content check, it can be either JSON, XML or RAW
385      * @throws SQLException
386      */
387     private VariantFormats verifyVariantType(Connection conn, String value) throws
388             SQLException {
389         try (PreparedStatement ps = conn.prepareStatement(CHECK_TYPE_STATEMENT)) {
390             ps.setString(1, value);
391             ps.setString(2, value);
392             ResultSet results = ps.executeQuery();
393             if (results.next()) {
394                 if (results.getString(1) == null) {
395                     return VariantFormats.JSON;
396                 } else if (results.getString(2) == null) {
397                     return VariantFormats.XML;
398                 }
399             }
400             return VariantFormats.RAW;
401         }
402     }

411             Settings settings = databaseConfig.getSettings();
412             DSLContext context = DSL.using(conn, dialect, settings);
413             // Snowflake does not support bind values nor turning-off auto-commit
414             String selectQuery = selectBuilder.build(context.configuration()).toString();
415             if (Objects.nonNull(queryQueue)) {
416                 DatabaseUtils.submitQueryToThread(selectQuery, queryQueue);
417             }
418             dateTimeOption = ((SnowflakeRecordMapper) recordMapper)
419                     .getDateTimeOption();
420             if (dateTimeOption.equals(DateTimeOption.DefaultDateTime)) {
421                 return executeQuery(selectQuery, null,
422                         account, maxRetries, retryInterval);
423             } else {
424                 DatabaseCursor cursor = executeQueryWithoutjOOQ(account, context,
425                         context.resultQuery(selectQuery), maxRetries, retryInterval);
426                 ((SnowflakeRecordMapper) recordMapper).setStoreTimestampValues(
427                         storeTimestampValues);
428                 return cursor;
429             }
430         } catch (ExecutionException | SnapDataException e) {
431             throw e;
432         } catch (DataAccessException e) {
433             throw new ExecutionException(EXECUTE_SELECT_FAILED).withReason(e.getMessage())
434                     .withResolution(CHECK_QUERY).withSuppressed(e);
435         } catch (IndexOutOfBoundsException e) {
436             throw new SnapDataException(e, ERR_INDEX_OUT_OF_BOUND).formatWith(e.getMessage())
437                     .withReason(REASON_BUG_IN_JDBC_DRIVER)
