529     private List<Map<String, Object>> groupBy(
530             List<Map<String, Object>> inputDoc, String groupByfield,
531             String targetField) {
532         // 1. Create an intermediate map for output doc.
533         //  'groupByField' becomes the key to the map and its group becomes the value.
534         Map<String, List<Map<String, Object>>> intermediateMap =
535                 new LinkedHashMap<String, List<Map<String, Object>>>();
536 
537         // 2. Loop on the inputDoc: process one entry per span
538         for (Map<String, Object> inputEntry : inputDoc) {
539             String groupByKey = (String) inputEntry.get(groupByfield);
540 
541             // Add the current inputEntry to the intermediateMap
542             // Create outEntry map and add inputEntry to the map.

714                                                     List<Map<String, Object>> tgtDef,
715                                                     List<String> joinKeyForRight,

718         // {
719         //   "TestPerson__c": {
720         //      "groupBy" : {
721         //      ..

723         //   },

733         for (Map<String, Object> ent : srcDef) {
734             // joinKey will have "TestPerson__c" for the first left side map from the example.
735             String joinKey = getJoinKeyValue(ent, joinKeyForLeft);
736             if (joinKey == null) {
737                 throw new SnapDataException("Cdt prepreparation needs join key");
738             }
739             intermediateMap.put(joinKey, ent);

743         if (tgtDef != null && tgtDef.size() > 0) {
744             for (Map<String, Object> ent : tgtDef) {

746                 if (joinKey == null) {

748                 }

759         List<Map<String, Object>> outputList = new ArrayList<Map<String, Object>>();
760         for (Map.Entry<String, Map<String, Object>>  entry : intermediateMap.entrySet())  {

780     private String getJoinKeyValue(Map<String, Object> input, List<String> joinKey) {
781         if (MapUtils.isEmpty(input) || (joinKey == null || joinKey.size() == 0)) {
782             return null;
783         }
784 
785         Map<String, Object> current = input;
786         Object nextObj = null;
787         int i = 0;
788         for ( ; i < joinKey.size(); i++) {
789             nextObj = current.get(joinKey.get(i));
790             if (nextObj == null) {
791                 return null;
792             } else if (nextObj instanceof Map) {
793                 current = (Map<String, Object>) nextObj;
794             } else {
795                 break;
796             }
797         }
798 
799         // If nextObj is the leaf and is a String, then return the String value.
800         if (i == joinKey.size() - 1 && nextObj instanceof String) {
801             return (String) nextObj;
802         }
803         return null;

841      * Return target precision value after validating
842      * the given source 'precisionValue' (source precision value).
843      *
844      * Note that 'precisionValue' may be BigInteger, integer string, or
845      * even special string value like "max".
846      *
847      * @param precisionValue    source precision value
848      * @param dataTypeValue     target data type
849      * @param tgtDatasizeMap    target data size map with min and max per data type
850      * @return    target scale value
851      */
852     private BigInteger getTargetPrecisionValue(
853             Object precisionValue, String dataTypeValue, Map<String, Object> tgtDatasizeMap) {
854         // We may have a String value like "max" from precisionValue
855         boolean isPrecisionValueNonNumeric =
856                 precisionValue instanceof String &&
857                         !StringUtils.isNumeric((String) precisionValue);
858         BigInteger precisionValueBigInteger =
859                 getValueAsBigInteger(precisionValue).orElse(null);
860 
861         Map<String, Object> datasizeEntry =
862                 (Map<String, Object>) tgtDatasizeMap.get(dataTypeValue);
863         BigInteger maxValue = ((List<BigInteger>) datasizeEntry.get("max")).get(0);
864         BigInteger minValue = ((List<BigInteger>) datasizeEntry.get("min")).get(0);
865 
866         if (precisionValue == null || precisionValue.equals(BigInteger.ZERO)) {
867             return maxValue;
868         }
869 
870         // Validate and return precision value with minValue
871         if (minValue == null) {
872             return null;
873         }
874 
875         if (!isPrecisionValueNonNumeric &&
876                 precisionValueBigInteger != null) {
877 
878             if (precisionValueBigInteger.compareTo(BigInteger.ZERO) < 0) {
879                 return maxValue;
880             }
881 
882             if (precisionValueBigInteger.compareTo(minValue) < 0) {
883                 return minValue;
884             }
885         }
886 
887         // Validate and return precision value with maxValue
888         if (maxValue == null) {
889             return null;
890         }
891 
892         if (isPrecisionValueNonNumeric &&
893                 StringUtils.equals("max", (String) precisionValue)) {
894             return maxValue;
895         }
896 
897         if (!isPrecisionValueNonNumeric && precisionValueBigInteger != null &&
898                 precisionValueBigInteger.compareTo(maxValue) > 0) {
899             return maxValue;
900         }
901 
902         return precisionValueBigInteger;
903     }
