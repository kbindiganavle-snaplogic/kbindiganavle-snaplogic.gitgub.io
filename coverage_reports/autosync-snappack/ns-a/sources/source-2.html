13 /**
14  * Change column type reconciliation action types on the target table
15  */
16 public enum ChangeColumnTypeAction {
17     UNKNOWN,
18     /**
19      * No operation is required.
20      */
21     NOOP,
22     /**
23      * SINGLE_ALTER_TABLE action is chosen when the required
24      * data type conversion is supported with a single ALTER TABLE statement.
25      *
26      * For example, given a column <COLUMN_NAME> of an existing target table <TABLE_NAME>,
27      * ALTER TABLE <TABLE_NAME> ALTER COLUMN <COLUMN_NAME> SET DATA TYPE <DESTINATION_DATA_TYPE>;
28      */
29     SINGLE_ALTER_TABLE {
30         @Override
31         public String toString() {
32             return "Single ALTER TABLE";
33         }
34     },
35     /**
36      * ALTER_UPDATE_ALTER_ALTER action is chosen when the required
37      * data type conversion is not supported with SINGLE_ALTER_TABLE type.
38      *
39      * This action type takes a 4-step approach:
40      * Given a column <COLUMN_NAME> of an existing target table <TABLE_NAME>,
41      * and the required data type conversion being, for example, from NUMBER
42      * to VARCHAR, where VARCHAR is the source data type drift, and NUMBER is
43      * the existing data type of <COLUMN_NAME>,
44      *
45      * ALTER TABLE <TABLE_NAME> ADD COLUMN <COLUMN_NAME_TEMP> VARCHAR;
46      * UPDATE <TABLE_NAME> SET <COLUMN_NAME_TEMP> = CAST(<COLUMN_NAME> AS VARCHAR);
47      * ALTER TABLE <TABLE_NAME> DROP COLUMN <COLUMN_NAME>;
48      * ALTER TABLE <TABLE_NAME> RENAME COLUMN <COLUMN_NAME_TEMP> to <COLUMN_NAME>;
49      */
50     ALTER_UPDATE_ALTER_ALTER {
51         @Override
52         public String toString() {
53             return "ALTER-UPDATE-ALTER-ALTER";
54         }
55     },
56     /**
57      * ALTER_UPDATE_ALTER_ALTER_WITH_LCA action is chosen when the required
58      * data type conversion is not supported with SINGLE_ALTER_TABLE and
59      * ALTER_UPDATE_ALTER_ALTER. Then we find lowest level common supertype
60      * (of the two data types) from the data type tree and use that type as
61      * the destination column data type. This takes the same 4-step approach
62      * of ALTER_UPDATE_ALTER_ALTER except the supertype as casting type at
63      * the UPDATE statement.
64      */
65     ALTER_UPDATE_ALTER_ALTER_WITH_NON_DIRECT_LCA {
66         @Override
67         public String toString() {
68             return "ALTER-UPDATE-ALTER-ALTER with NON-DIRECT LCA";
69         }
70     },
71     /**
72      * UNSUPPORTED action is defined as unsupported data type conversion type.
73      */
74     UNSUPPORTED {
75         @Override
76         public String toString() {
77             return "Unsupported";
78         }
79     },
80     /**
81      * NOOP_SIZE_NARROWING action is defined as noop since the type conversion
82      * is size narrowing one.
83      *
84      * A data type size narrowing conversion like the one from INT32 to INT64 is
85      * one of the examples.
86      */
87     NOOP_SIZE_NARROWING {
88         @Override
89         public String toString() {
90             return "Noop-size-narrowing";
91         }
92     }
93 }
