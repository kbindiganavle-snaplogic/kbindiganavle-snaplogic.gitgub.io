184     private List<String> copyPath(Path source, Path target, CopyOption... options)
185             throws IOException {
186         boolean replaceExisting = Arrays.asList(options).contains(REPLACE_EXISTING);
187         S3Path s3Source = (S3Path) source;
188         String sourceBucket = s3Source.getBucketName();
189         String sourceKey = s3Source.getKey();
190         Map<String, Object> kmsProperties = s3Source.getKmsProperties();
191         S3Path s3Target = (S3Path) target;
192         String targetBucket = s3Target.getBucketName();
193         String targetKey = s3Target.getKey();
194         AmazonS3 amazonS3 = s3Source.getAmazonS3();
195         initS3Path(amazonS3, sourceBucket, sourceKey, s3Source);
196         initS3Path(amazonS3, targetBucket, targetKey, s3Target);
197         List<String> copiedKeys = new ArrayList<>();
198         if (s3Source.isRegularFile()) {
199             if (s3Target.isDirectory()) {
200                 // if source is a file and target is a directory, the source file should be
201                 // copied into the target directory
202                 int index = StringUtils.lastIndexOf(sourceKey, URL_SEPARATOR);
203                 String filename = (index > 0) ? sourceKey.substring(index) : sourceKey;
204                 targetKey = JfsUtils.concat(targetKey, filename);
205             }
206             if (!replaceExisting && exists(amazonS3, targetBucket, targetKey)) {
207                 throw new FileAlreadyExistsException(String.format(
208                         FORMAT_S3_PATH, targetBucket, targetKey), null, ERR_TARGET_EXISTS);
209             }
210             copyS3ObjectFailSafe(amazonS3, sourceBucket, sourceKey, targetBucket, targetKey,
211                     kmsProperties);
212             copiedKeys.add(sourceKey);
213         } else if (s3Source.isDirectory()) {
214             if (s3Target.isRegularFile() && !replaceExisting) {
215                 throw new FileAlreadyExistsException(target.toUri().getPath(), null,
216                         ERR_TARGET_EXISTS);
217             }
218             // copy all files in the source directory to the target directory
219             sourceKey = JfsUtils.endsWithUrlSeparator(sourceKey);
220             ListObjectsRequest listObjectsRequest = new ListObjectsRequest()
221                     .withBucketName(sourceBucket)
222                     .withPrefix(sourceKey);
223             ObjectListing objectListing = amazonS3.listObjects(listObjectsRequest);
224             boolean isTruncated;
225             do {
226                 for (S3ObjectSummary s3ObjectSummary : objectListing.getObjectSummaries()) {
227                     String childKey = s3ObjectSummary.getKey();
228                     if (StringUtils.equals(childKey, sourceKey)) {
229                         copiedKeys.add(childKey);
230                         continue;
231                     }
232                     String childTargetKey = childKey.equals(sourceKey) ?
233                             JfsUtils.endsWithUrlSeparator(targetKey) :
234                             JfsUtils.concat(targetKey, childKey.substring(sourceKey.length()));
235                     if (!replaceExisting && exists(amazonS3, targetBucket, childTargetKey)) {
236                         throw new FileAlreadyExistsException(targetBucket.concat(URL_SEPARATOR)
237                                 .concat(childTargetKey), null, ERR_TARGET_EXISTS);
238                     }
239                     copyS3ObjectFailSafe(amazonS3, sourceBucket, childKey, targetBucket,
240                             childTargetKey, kmsProperties);
241                     copiedKeys.add(childKey);
242                 }
243                 if (isTruncated = objectListing.isTruncated()) {
244                     objectListing = amazonS3.listNextBatchOfObjects(objectListing);
245                 }
246             } while (isTruncated);
247         } else {
248             throw new NoSuchFileException(source.toUri().getPath());
249         }
250         return copiedKeys;
251     }

275     // Validates if an S3 object exists, and retries if it does not exists. This method is called
276     // after copyS3Object() to make sure the target exist because AWS S3 fails silently and
277     // intermittently in both copyObject() and copyPart(). Also intermittently, the target may not
278     // be available immediately after it is copied.
279     private boolean existsFailSafe(AmazonS3 amazonS3, String bucketName, String key) {
280         try {
281             Failsafe.with(getRetryPolicy(3, 2))
282                     .onFailedAttempt(e -> LOGGER.warn("{} - S3 object not found, bucket: {}, " +
283                     "key: {}, detail: {}", RETRY_SIGNATURE, bucketName, key, e.getMessage()))
284                     .run(context -> {
285                         logRetryAttempt(context, LOGGER, RETRY_SIGNATURE);
286                         amazonS3.getObjectMetadata(bucketName, key);
287                     });
288         } catch (Throwable t) {
289             return false;
290         }
291         return true;
292     }

294     private void copyS3ObjectFailSafe(AmazonS3 amazonS3, String sourceBucket, String sourceKey,
295             String targetBucket, String targetKey, Map<String, Object> kmsProperties) {
296         String source = String.format(FORMAT_S3_PATH, sourceBucket, sourceKey);
297         String target = String.format(FORMAT_S3_PATH, targetBucket, targetKey);
298         try {
299             Failsafe.with(getRetryPolicy(3, 1))
300                     .onFailedAttempt(e -> LOGGER.warn("{} - AWS S3 failed to copy from {} " +
301                     "to {}, detail: {}", RETRY_SIGNATURE, source, target, e.getMessage()))
302                     .run(context -> {
303                         logRetryAttempt(context, LOGGER, RETRY_SIGNATURE);
304                         copyS3Object(amazonS3, sourceBucket, sourceKey, targetBucket, targetKey,
305                                 kmsProperties);
306                         if (!existsFailSafe(amazonS3, targetBucket, targetKey)) {
307                             throw new SnapDataException(ERR_COPY_FILE)
308                                     .formatWith(sourceBucket, sourceKey, targetBucket, targetKey)
309                                     .withReason(REASON_COPY_FAIL_SILENT)
310                                     .withResolution(RESOLUTION_ADDRESS_ISSUE);
311                         }
312                     });
313         } catch (Throwable t) {
314             Throwable t2 = t instanceof FailsafeException ? t.getCause() : t;
315             if (t2 instanceof SuggestViewAbortException) {
316                 throw (SuggestViewAbortException) t2;
317             } else if (t2 instanceof SnapDataException) {
318                 throw (SnapDataException) t2;
319             } else {
320                 throw new SnapDataException(t2, ERR_COPY_FILE)
321                         .formatWith(sourceBucket, sourceKey, targetBucket, targetKey)
322                         .withResolution(RESOLUTION_ADDRESS_ISSUE);
323             }
324         }
325     }

327     /**
328      * Copies S3 file in the same region. If the size is smaller than 4GB, use simple copyObject()
329      * method. Otherwise, use Multipart Upload API. For more details, refer to AWS S3 document:
330      * https://docs.aws.amazon.com/AmazonS3/latest/dev/CopyingObjctsUsingLLJavaMPUapi.html
331      * The threshold is actually 5GB in the above document, but we'll use 4GB to be safe since it
332      * is not clear if AWS document means to say 5*1024*1024*1024 or 5,000,000,000.
333      *
334      * @param amazonS3     - AmazonS3 service
335      * @param sourceBucket - source bucket
336      * @param sourceKey    - source key
337      * @param targetBucket - target bucket
338      * @param targetKey    - target key
339      */
340     protected void copyS3Object(AmazonS3 amazonS3, String sourceBucket, String sourceKey,
341             String targetBucket, String targetKey, Map<String, Object> kmsProperties) {
342         // Get the object size to track the end of the copy operation.
343         Long objectSize = getSize(amazonS3, sourceBucket, sourceKey);
344         if (objectSize == null) {
345             // AmazonS3 intermittently gives an encoded sourceKey which is a defect in AWS Java
346             // SDK aws-java-sdk, version 1.10.47. To work around, decode sourceKey and try again
347             String sourceKeyTemp = JfsUtils.decode(sourceKey, false);
348             objectSize = getSize(amazonS3, sourceBucket, sourceKeyTemp);
349             if (objectSize != null) {
350                 sourceKey = sourceKeyTemp;
351                 targetKey = JfsUtils.decode(targetKey);
352             } else {
353                 throw new SnapDataException(ERR_SOURCE_FILE_NOT_FOUND)
354                         .formatWith(sourceBucket, sourceKey)
355                         .withReason(REASON_OBJECT_METADATA)
356                         .withResolution(RESOLUTION_ADDRESS_ISSUE);
357             }
358         }
359         if (objectSize == 0L) {
360             try {
361                 amazonS3.copyObject(sourceBucket, sourceKey, targetBucket, targetKey);
362                 return;
363             } catch (AmazonServiceException e) {
364                 throw new SnapDataException(e, ERR_COPY_FILE)
365                         .formatWith(sourceBucket, sourceKey, targetBucket, targetKey)
366                         .withResolution(RESOLUTION_ADDRESS_ISSUE);
367             }
368         }
369         try {
370             InitiateMultipartUploadRequest initRequest = new InitiateMultipartUploadRequest(
371                     targetBucket, targetKey);
372             // Handle Server-side Encryption
373             Object value = MapUtils.getObject(kmsProperties, ENCRYPT_TYPE_KEY);
374             if (value instanceof S3EncryptionType) {
375                 switch ((S3EncryptionType) value) {
376                     case SSE_KMS:
377                         Object kmsKey = kmsProperties.get(KMS_KEY);
378                         if (kmsKey instanceof String) {
379                             initRequest.setSSEAwsKeyManagementParams(
380                                     new SSEAwsKeyManagementParams((String) kmsKey));
381                         }
382                         break;
383                     case SSE:
384                         ObjectMetadata objectMetadata = new ObjectMetadata();
385                         objectMetadata.setSSEAlgorithm(AES_256_SERVER_SIDE_ENCRYPTION);
386                         initRequest.setObjectMetadata(objectMetadata);
387                         break;
388                 }
389             }
390             // Initiate the multipart upload.
391             InitiateMultipartUploadResult initResult = amazonS3
392                     .initiateMultipartUpload(initRequest);
393 
394             // Copy the object using multipart upload API.
395             long partSize = GB;
396             long bytePosition = 0;
397             int partNum = 1;
398             List<CopyPartResult> copyResponses = new ArrayList<CopyPartResult>();
399             while (bytePosition < objectSize) {
400                 // The last part might be smaller than partSize, so check to make sure
401                 // that lastByte isn't beyond the end of the object.
402                 long lastByte = Math.min(bytePosition + partSize - 1, objectSize - 1);
403 
404                 // Copy this part.
405                 CopyPartRequest copyRequest = new CopyPartRequest()
406                         .withSourceBucketName(sourceBucket)
407                         .withSourceKey(sourceKey)
408                         .withDestinationBucketName(targetBucket)
409                         .withDestinationKey(targetKey)
410                         .withUploadId(initResult.getUploadId())
411                         .withFirstByte(bytePosition)
412                         .withLastByte(lastByte)
413                         .withPartNumber(partNum++);
414                 // TODO - psung: multi-thread can be used for copyPart() to copy file faster
415                 copyResponses.add(amazonS3.copyPart(copyRequest));
416                 bytePosition += partSize;
417             }
418 
419             // Complete the upload request to concatenate all uploaded parts and make the copied
420             // object available.
421             CompleteMultipartUploadRequest completeRequest = new CompleteMultipartUploadRequest(
422                     targetBucket, targetKey, initResult.getUploadId(), getETags(copyResponses));
423             amazonS3.completeMultipartUpload(completeRequest);
424             log("Multipart copy complete", sourceBucket, sourceKey, targetBucket, targetKey, null);
425         } catch (AmazonServiceException e) {
426             log(LOG_COPY_MULTIPART, sourceBucket, sourceKey, targetBucket, targetKey, e);
427             throw new SnapDataException(e, ERR_COPY_MULTIPART)
428                     .formatWith(sourceBucket, sourceKey, targetBucket, targetKey)
429                     .withReason(String.format(REASON_COPY_MULTIPART, e.getMessage()))
430                     .withResolution(RESOLUTION_ADDRESS_ISSUE);
431         } catch (Exception e) {
432             log(LOG_COPY_MULTIPART, sourceBucket, sourceKey, targetBucket, targetKey, e);
433             throw new SnapDataException(e, ERR_COPY_MULTIPART)
434                     .formatWith(sourceBucket, sourceKey, targetBucket, targetKey)
435                     .withReason(String.format(REASON_COPY_MULTIPART_2, e.getMessage()))
436                     .withResolution(RESOLUTION_ADDRESS_ISSUE);
437         }
438     }
