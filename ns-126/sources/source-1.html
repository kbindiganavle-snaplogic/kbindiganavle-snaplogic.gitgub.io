21 /**
22  * Generates cryptographically-secure random strings suitable for the nonce field
23  * of a TokenPassport element in a NetSuite Token-Based-Authentication request.
24  */
25 @Singleton
26 class NonceGenerator {
27 
28     // Number of random bytes obtained from the secure random number generator for each nonce.
29     private static final int NUM_BYTES = 15;
30 
31     /**
32      * The length of a generated nonce string.
33      */
34     static final int NONCE_LENGTH = NUM_BYTES / 3 * 4;
35 
36     private final Random random = new SecureRandom();
37 
38     /**
39      * Generates a random nonce value.
40      * @return a nonce value of length {@link #NONCE_LENGTH}.
41      */
42     String generate() {
43         byte[] bytes = new byte[NUM_BYTES];
44         random.nextBytes(bytes);
45         String nonce = Base64.encodeBase64URLSafeString(bytes);
46         return nonce;
47     }
48 }
