175     @SuppressWarnings("unchecked")
176     public void suggestTableName(SuggestionBuilder suggestionBuilder,
177                                  ServiceNowAccountBase account) {
178         CacheKey cacheKey = new CacheKey(account);
179         String[] tableSchemas = TABLE_CACHE.getIfPresent(cacheKey);
180         if (tableSchemas != null) {
181             suggestionBuilder.node(TABLE_NAME_PROP)
182                     .suggestions(tableSchemas);
183             return;
184         }
185         String uri = String.format(METADATA_FETCH_URL, account.getInstance(), SYS_DICT_LIST,
186                 TABLE_LIST_QUERY);
187         HttpResponse response = null;
188         try {
189             Digester digester = new Digester();
190             URI fullUri = new URI(uri);
191             HttpPost postRequest = new HttpPost(fullUri);
192             response = executeRequestAndGetResponse(postRequest, account);
193             StatusLine statusLine = response.getStatusLine();
194             int statusCode = statusLine.getStatusCode();
195             if (statusCode == HttpStatus.SC_UNAUTHORIZED) {
196                 throw new ExecutionException(INVALID_CREDENTIALS)
197                         .withResolution(INVALID_CREDENTIALS_RESOLUTION);
198             } else if (statusCode >= HttpStatus.SC_MULTIPLE_CHOICES) {
199                 throw new ExecutionException(ERR_WHILE_FETCHING_TABLE_LIST)
200                         .formatWith(statusCode)
201                         .withReason(statusLine.getReasonPhrase());
202             }
203             try (InputStream is = response.getEntity().getContent()) {
204                 digester.setValidating(false);
205                 digester.addObjectCreate(UNLOAD, HashSet.class);
206                 digester.addCallMethod(UNLOAD_SYS_DICTIONARY, ADD, 1);
207                 digester.addCallParam(UNLOAD_SYS_DICTIONARY_NAME, 0);
208                 Set<String> tableNames = (Set<String>) digester.parse(is);
209                 if (CollectionUtils.isEmpty(tableNames)) {
210                     suggestionBuilder.node(TABLE_NAME_PROP)
211                             .suggestions(ERR_NO_TABLE_NAMES);
212                 } else {
213                     tableNames.removeAll(ABSTRACT_TABLES);
214                     String[] tableNameValues = tableNames.toArray(new String[0]);
215                     Arrays.sort(tableNameValues);
216                     TABLE_CACHE.put(cacheKey, tableNameValues);
217                     suggestionBuilder.node(TABLE_NAME_PROP)
218                             .suggestions(tableNameValues);
219                 }
220             }
221         } catch (URISyntaxException e) {
222             serviceNowRequest.abort();
223             throw new ExecutionException(e, UNABLE_TO_CREATE_FULL_URI);
224         } catch (SAXException e) {
225             serviceNowRequest.abort();
226             throw new ExecutionException(e, PARSING_EXCEPTION)
227                     .withResolution(IO_EXCEPTION_RESOLUTION);
228         } catch (IOException e) {
229             serviceNowRequest.abort();
230             throw new ExecutionException(e, IO_EXCEPTION)
231                     .withReason(IO_EXCEPTION_REASON)
232                     .withResolution(IO_EXCEPTION_RESOLUTION);
233         } finally {
234             if (response != null) {
235                 try {
236                     EntityUtils.consume(response.getEntity());
237                 } catch (IOException e) {
238                     LOG.error(Throwables.getRootCause(e).getMessage(), e);
239                 }
240             }
241         }
242     }

257         String uri = String.format(TABLE_METADATA_FETCH_URL, account.getInstance(), tableName);
258         HttpResponse response = null;
259         try {
260             URI fullUri = new URI(uri);
261             HttpGet postRequest = new HttpGet(fullUri);
262             response = executeRequestAndGetResponse(postRequest, account);
263             StatusLine statusLine = response.getStatusLine();
264             int responseCode = statusLine.getStatusCode();
265             if (responseCode == HttpStatus.SC_UNAUTHORIZED) {
266                 throw new ExecutionException(INVALID_CREDENTIALS)
267                         .withResolution(INVALID_CREDENTIALS_RESOLUTION);
268             } else if (responseCode == HttpStatus.SC_FORBIDDEN) {
269                 LOG.warn(NOT_AUTHOURIZED_FOR_SCHEMA,tableName);
270                 return;
271             } else if (responseCode >= HttpStatus.SC_MULTIPLE_CHOICES) {
272                 throw new ExecutionException(ERR_WHILE_FETCHING_SCHEMA)
273                         .formatWith(tableName, responseCode)

275             }
276             HttpEntity entity = response.getEntity();
277             if (entity != null) {
278                 try (InputStream is = entity.getContent()) {
279                     String jsonBody = IOUtils.toString(is, Charset.forName(CHAR_SET_UTF8));
280                     List<String> elementsList = new ArrayList<>();
281 
282                     InputSource source = new InputSource(new StringReader(jsonBody));
283                     DocumentBuilderFactory dbf = SnapDocumentBuilderFactory.newInstance();
284                     DocumentBuilder db = dbf.newDocumentBuilder();
285                     Document document = db.parse(source);
286 
287                     NodeList nodes = document.getElementsByTagName(XML_ELEMENT);
288 
289                     for (int i = 0; i < nodes.getLength(); i++) {
290                         Element element = (Element) nodes.item(i);
291                         String name = element.getAttribute(XML_ATTRIBUTE_NAME);
292                         elementsList.add(name);
293                     }
294 
295 
296                     switch (schemaType) {
297                         case INPUT:
298                             for (String inputViewName : provider.getRegisteredInputViewNames()) {
299                                 SchemaBuilder schemaBuilder =
300                                         provider.getSchemaBuilder(inputViewName);
301                                 addFieldsToSchema(elementsList, schemaBuilder, provider,
302                                         schemaType, action);
303                                 schemaBuilder.build();
304                             }
305                             break;
306                         case OUTPUT:
307                             for (String viewName : provider.getRegisteredViewNames()) {
308                                 SchemaBuilder schemaBuilder = provider.getSchemaBuilder(viewName);
309                                 addFieldsToSchema(elementsList, schemaBuilder, provider,
310                                         schemaType, action);
311                                 schemaBuilder.build();
312                             }
313                             break;
314                     }
315                 }
316             }
317         } catch (URISyntaxException e) {
318             serviceNowRequest.abort();
319             throw new ExecutionException(e, UNABLE_TO_CREATE_FULL_URI);
320         } catch (ParserConfigurationException pe) {
321             serviceNowRequest.abort();
322             throw new ExecutionException(pe, ERROR_ON_XML_CONFIGURATI0N_PARSING)
323                     .withReason(pe.getLocalizedMessage())

328                     .withResolutionAsDefect();

332                     .withReason(IO_EXCEPTION_REASON)
333                     .withResolution(IO_EXCEPTION_RESOLUTION);
334         } finally {
335             if (response != null) {
336                 try {
337                     EntityUtils.consume(response.getEntity());
338                 } catch (IOException e) {
339                     LOG.error(Throwables.getRootCause(e).getMessage(), e);
340                 }

345     private HttpResponse executeRequestAndGetResponse(HttpUriRequest request,
346             ServiceNowAccountBase account) throws IOException {
347         boolean refreshed;
348         int oauthTokenRetryCount = 0;
349         HttpResponse response;
350         do {
351             refreshed = false;
352             serviceNowRequest.initialize(request, account);
353             response = serviceNowRequest.execute();
354             StatusLine statusLine = response.getStatusLine();
355             int responseCode = statusLine.getStatusCode();
356             if (responseCode == SC_UNAUTHORIZED &&
357                     (account instanceof ServiceNowOAuth2Account) &&
358                     (oauthTokenRetryCount++ < MAX_OAUTH_RETRY_COUNT)) {
359                 refreshed = ServiceNowHelper.refreshOAuth2Account(account, LOG);
360                 LOG.debug("Able to refresh oauth2 token after 401 error? {}", refreshed);
361             }
362             if (refreshed) {
363                 serviceNowRequest.updateHeaders(account);
364             }
365         } while (refreshed);
366 
367         return response;
368     }

375     private void addFieldsToSchema(List<String> elementsList, SchemaBuilder schemaBuilder,
376                                    SchemaProvider provider, SchemaType schemaType,
377                                    ServiceNowBaseWrite.Action action) {
378         for (String element : elementsList) {
379             // some elements have empty value; ignoring them
380             if (!StringUtils.isEmpty(element)) {
381                 switch (element) {
382                     case SYS_CLASS_NAME:
383                     case SYS_CREATED_BY:
384                     case SYS_CREATED_ON:
385                     case SYS_DOMAIN:
386                     case SYS_MOD_COUNT:
387                     case SYS_UPDATED_BY:
388                     case SYS_UPDATED_ON:
389                         switch (schemaType) {
390                             case INPUT:
391                                 continue;
392                             case OUTPUT:
393                                 break;
394                         }
395                     case SYS_ID:
396                         switch (action) {
397                             case INSERT:
398                                 continue;
399                             case UPDATE:
400                                 break;
401                         }
402                 }
403                 // All attributes are always of type string
404                 Schema childSchema = provider.createSchema(SnapType.STRING, element);
405                 schemaBuilder.withChildSchema(childSchema);
406             }
407         }
408     }
