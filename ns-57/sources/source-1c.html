59     /**
60      * Converts a DateTimeOffset object into a Datetime object.
61      *
62      * @param object as DateTimeOffset object
63      *
64      * @return the datetime object or NaN if invalid.
65      */
66     public Object fromDateTimeOffset(Object object) {
67         if (object != null) {
68             Class klass = object.getClass();
69             try {
70                 Method getMinutesOffsetMethod = klass.getMethod("getMinutesOffset", new Class[]{});
71                 Integer minutes = (Integer) getMinutesOffsetMethod.invoke(object, new Object[]{});
72                 Method getTimestampMethod = klass.getMethod("getTimestamp", new Class[]{});
73                 Timestamp timestamp = (Timestamp) getTimestampMethod.invoke(object, new Object[]{});
74                 long ts = timestamp.getTime();
75                 DateTime dateTime = new DateTime(ts, DateTimeZone.forOffsetMillis(
76                         minutes * SIXTYK_SEC));
77                 return verifyValidDateTime(dateTime);
78             } catch (ReflectiveOperationException e) {
79                 throw new ExecutionException(e, ERR_UNSUPPORTED_DATA_TYPE)
80                         .formatWith(object.getClass().getName())
81                         .withReason(REASON_UNSUPPORTED_DATA_TYPE)
82                         .withResolution(RESOLUTION_ADDRESS_ISSUE);
83             }
84         }
85         return object;
86     }

88     /**
89      * Verifies the corner case, where we get a Date such as 0001-01-01 00:00:00.0000000 and then
90      * apply the time zone on top of it, which can lead to a date of 0000-12-13-....,
91      * making it an invalid date representation.
92      *
93      * @param dateTime the datetime
94      *
95      * @return the incoming datetime if its valid, otherwise NAN
96      */
97     private static Object verifyValidDateTime(DateTime dateTime) {
98         if (dateTime.getCenturyOfEra() == 0 && dateTime.getYearOfCentury() == 0) {
99             return Double.NaN;
100         }
101         return dateTime;
102     }
