152     public HttpUriRequest unwrap() throws HttpCoreDataException {
153         HttpUriRequest request;
154         if ("GET".equals(method) && entity == null){
155             request = new HttpGet(uriWithParameters);
156         } else {
157             request = new HttpMethodWithEntity(method, uriWithParameters);
158             if (entity != null) {
159                 ((HttpMethodWithEntity) request).setEntity(entity.getEntity());
160             }
161         }
162         for (Pair<String, String> header : headers) {
163             if (StringUtils.isNotEmpty(header.getKey())) {
164                 request.addHeader(header.getKey(), header.getValue());
165             }
166         }
167         if (oAuth1Consumer != null) {
168             try {
169                 request = (HttpMethodWithEntity) oAuth1Consumer.sign(request).unwrap();
170             } catch (OAuthMessageSignerException | OAuthExpectationFailedException
171                     | OAuthCommunicationException e) {
172                 throw new HttpCoreDataException("Error signing the request for Oauth 1.0 " +
173                         "authentication", e);
174             }
175         }
176         return request;
177     }

232     public void attachAuthorization(AuthData authData, HttpClientContext context)
233             throws HttpCoreDataException {
234         if (authData == null) {
235             return;
236         }
237         if (authData.getCredentials() != null) {
238             CredentialsProvider credentialsProvider = context.getCredentialsProvider();
239             if (credentialsProvider == null) {
240                 // TODO try SystemDefaultCredentialsProvider
241                 credentialsProvider = new BasicCredentialsProvider();
242             }
243             credentialsProvider.setCredentials(AuthScope.ANY, authData.getCredentials());
244             updateProxyAuthCredentialsIfNeeded(credentialsProvider, context);
245 
246             Registry<AuthSchemeProvider> authSchemeProviderRegistry =
247                     RegistryBuilder.<AuthSchemeProvider>create()
248                             .register(AuthSchemes.NTLM, new NTLMSchemeFactory())
249                             .register(AuthSchemes.BASIC, new BasicSchemeFactory())
250                             .register(AuthSchemes.DIGEST, new DigestSchemeFactory())
251                             .build();
252             context.setAuthSchemeRegistry(authSchemeProviderRegistry);
253             context.setCredentialsProvider(credentialsProvider);
254             if (authData.isPreemptive()) {
255                 if (context.getAuthCache() == null) {
256                     context.setAuthCache(new BasicAuthCache());
257                 }
258                 context.getAuthCache().put(new HttpHost(uriWithParameters.getHost(),
259                                 uriWithParameters.getPort(), uriWithParameters.getScheme()),
260                         authData.getAuthScheme());
261             }
262         }
263         if (authData.getAuthHeader() != null) {
264             // Update the existing headers; If multiple headers with same key are given,
265             // then update the existing one(s) and append the remaining to the header list
266             Map<String, LinkedList<String>> authHeaders = new HashMap<>();
267             for (Pair<String, String> header : authData.getAuthHeader()) {
268                 authHeaders.putIfAbsent(header.getKey(), new LinkedList<>());
269                 authHeaders.get(header.getKey()).add(header.getValue());
270             }
271             // Have to create a new list since the Pair is immutable
272             List<Pair<String, String>> newHeaderList = new ArrayList<>();
273             for (Pair<String, String> header : headers) {
274                 if (authHeaders.containsKey(header.getKey())) {
275                     newHeaderList.add(Pair.of(header.getKey(),
276                             authHeaders.get(header.getKey()).pollFirst()));
277                     if (authHeaders.get(header.getKey()).isEmpty()) {
278                         authHeaders.remove(header.getKey());
279                     }
280                 } else {
281                     newHeaderList.add(header);
282                 }
283             }
284             for (String key : authHeaders.keySet()) {
285                 for (String val : authHeaders.get(key)) {
286                     newHeaderList.add(Pair.of(key, val));
287                 }
288             }
289             headers = newHeaderList;
290         }
291         if (authData.getAuthUrlParam() != null) {
292             URIBuilder uriBuilder = new URIBuilder(uriWithParameters);
293             authData.getAuthUrlParam().forEach(pair
294                     -> uriBuilder.addParameter(pair.getKey(), pair.getValue()));
295             try {
296                 uriWithParameters = uriBuilder.build();
297             } catch (URISyntaxException e) {
298                 throw new HttpCoreDataException("Error setting the oauth2 access_token parameter"
299                         , e);
300             }
301         }
302         if (authData.getAwsV4Signer() != null) {
303             HttpSnapAWSV4Signer signer = authData.getAwsV4Signer();
304             //removing headers that are added while AWS signV4 process
305             List<String> awsV4SecureHeaderKeys = Arrays.asList(X_AMZ_DATE, AUTHORIZATION,
306                     X_AMZ_CONTENT_SHA256, X_AMZ_SECURITY_TOKEN);
307             headers.removeIf(header -> awsV4SecureHeaderKeys.contains(header.getKey()));
308             // fill in the required information for doing signature
309             signer.withRequestDetails(method, uriWithParameters, headers, entity);
310             headers.addAll(signer.sign());
311         }
312         if (authData.getOAuth1Consumer() != null) {
313             oAuth1Consumer = authData.getOAuth1Consumer();
314         }
315     }

334      * Create the URI object with the given uri String plus an optional list of parameters.
335      * <p>
336      * The method is supposed to handle the URL encoding as well.
337      * <p>
338      * Note:
339      *  Apache HttpClient 5.x has changed the behavior of URIBuilder to encode
340      *  spaces in the query string with %20 (RFC 3986) instead of + (RFC 1866)
341      *  which is what HttpClient 4.x uses (see SWAT-3796).
342      *
343      *  https://issues.apache.org/jira/browse/HTTPCORE-628
344      *
345      *  After we upgrade to HttpClient 5.x, we'll still need separate paths to
346      *  build the URI depending on whether the input url is encoded or not.
347      * <p>
348      */
349     private URI buildURI(String uri, List<Pair<String, String>> parameters)
350             throws URISyntaxException, MalformedURLException, HttpCoreDataException {
351         if (StringUtils.isEmpty(uri)) {
352             throw new HttpCoreDataException("The URI cannot be null");
353         }
354         if (StringUtils.indexOf(uri, "://") < 0) {
355             uri = "http://" + uri;
356         }
357         return (requestProperties.isURLEncodingDisabled()) ?
358             buildURIWithoutEncoding(uri, parameters) : buildURIWithEncoding(uri, parameters);
359     }

361     private URI buildURIWithEncoding(String uri, List<Pair<String, String>> parameters)
362             throws URISyntaxException, MalformedURLException, HttpCoreDataException {
363         URL givenURL = new URL(uri);
364         // URI Builder here would mostly be responsible for URL encoding
365         // .setPath() and .setParameters() are expecting unescaped values. So try decode them first.
366         URIBuilder uriBuilder = new URIBuilder()
367                 .setScheme(givenURL.getProtocol())
368                 .setHost(givenURL.getHost())
369                 .setPort(givenURL.getPort())
370                 .setPath(tryURLDecode(givenURL.getPath()));
371         List<NameValuePair> params = URLEncodedUtils.parse(givenURL.getQuery(),
372                 StandardCharsets.UTF_8);
373         if (parameters != null) {
374             parameters.stream()
375                     .filter(pair -> StringUtils.isNotEmpty(pair.getKey())
376                             && pair.getValue() != null) // param value can be an empty string
377                     .forEach(pair -> params.add(new BasicNameValuePair(
378                             tryURLDecode(pair.getKey()),
379                             tryURLDecode(pair.getValue())))
380                     );
381         }
382         uriBuilder.setParameters(params);
383         URI finalUri = uriBuilder.build();
384 
385         // SWAT-3796: when we upgrade to HttpClient 5.x, we'll want to maintain
386         // backwards compatibility by converting percent-encoding (%20) to plus
387         // symbol if the "form encode space" property has been selected.
388         if (!requestProperties.isSpaceFormEncoded() && finalUri.toString().indexOf("?") != -1) {
389             String urlString = finalUri.toString();
390             int queryIndex = urlString.indexOf("?");
391             // only percent encode plus-es in query portion of url
392             String encodedUrl = urlString.substring(0, queryIndex)
393                 + urlString.substring(queryIndex).replaceAll("\\+", "%20");
394             // single-arg URI ctor doesn't escape illegal chars (ex: % -> %25)
395             finalUri = new URI(encodedUrl);
396         }
397         return finalUri;
398     }

400     private URI buildURIWithoutEncoding(String uri, List<Pair<String, String>> parameters)
401             throws URISyntaxException, MalformedURLException, HttpCoreDataException {
402 
403         // URIBuilder ctor accepts and preserves an encoded uri
404         StringBuilder uriBuilder = new StringBuilder(new URIBuilder(uri).toString());
405         StringJoiner extraParams = new StringJoiner("&");
406         if (parameters != null && parameters.size() > 0) {
407             parameters.stream()
408                 .forEach(pair -> extraParams
409                     .add(pair.getKey() + "=" + pair.getValue()));
410 
411             if (uri.indexOf("?") == -1) {
412                 uriBuilder.append("?");
413             } else if (!uri.endsWith("?")) {
414                 uriBuilder.append("&");
415             }
416             uriBuilder.append(extraParams.toString());
417         }
418         // single-arg URI ctor doesn't escape illegal chars (ex: % -> %25)
419         return new URI(uriBuilder.toString());
420     }

422     private String tryURLDecode(String val) {
423         try {
424             return URLDecoder.decode(val == null ? StringUtils.EMPTY : val, StandardCharsets.UTF_8);
425         } catch (IllegalArgumentException e) {
426             // in case that the decode doesn't succeed, we fall back to the original string
427             return val;
428         }
