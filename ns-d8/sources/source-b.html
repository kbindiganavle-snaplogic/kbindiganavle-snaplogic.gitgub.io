52     protected static final String MAIL_SMTP_TIMEOUT = "mail.smtp.timeout";
53     protected static final String MAIL_SMTP_WRITE_TIMEOUT = "mail.smtp.writetimeout";
54     protected static final String MAIL_SMTP_SSL_PROTOCOLS = "mail.smtp.ssl.protocols";

57     @Override
58     public Session connect() {
59         // Setup mail server
60         Properties properties = new Properties();
61         properties.setProperty(SMTP_HOST, serverDomain);
62         properties.put(SMTP_PORTT, port);
63         // if Email ID is blank, attempt anonymous connection
64         if (StringUtils.isBlank(emailId)) {
65             properties.put(SMTP_AUTH, FALSE);
66         } else {
67             properties.put(SMTP_AUTH, TRUE);
68         }
69 
70         properties.put(MAIL_SMTP_CONNECTION_TIMEOUT, String.valueOf(connectionTimeout));
71         properties.put(MAIL_SMTP_TIMEOUT, String.valueOf(readTimeout));
72         properties.put(MAIL_SMTP_WRITE_TIMEOUT, String.valueOf(writeTimeout));
73 
74         // Specify protocols that will be enabled for TLS connections
75         // - include TLSv1 and TLSv1.1 that deprecated in 2020 for backward-compatibility
76         properties.put(MAIL_SMTP_SSL_PROTOCOLS, SUPPORTED_PROTOCOLS);
77 
78         switch (secureConnection) {
79             case STARTTLS:
80                 properties.put(SMTP_TLS, TRUE);
81                 break;
82             case SSL:
83                 properties.put(SMTP_SSL, TRUE);
84                 break;
85             case NONE:
86                 break;
87             default:
88                 throw new SnapDataException(ERR_UNSUPPORTED_SECURE_CONNECTION)
89                         .formatWith(secureConnection.toString())
90                         .withReason(ERR_UNSUPPORTED_SECURE_CONNECTION_REASON)
91                         .withResolution(ERR_CUSTOMER_SUPPORT);
92         }
93         overrideProperties(mailAccountProperties, properties);
94         // return email Session object
95         if (StringUtils.isBlank(emailId)) {
96             return Session.getInstance(properties);
97         } else {
98             return Session.getInstance(properties,
99                     new Authenticator() {
100                         protected PasswordAuthentication getPasswordAuthentication() {
101                             return new PasswordAuthentication(emailId, password);
102                         }
103                     });
104         }
105     }

109         // do nothing
