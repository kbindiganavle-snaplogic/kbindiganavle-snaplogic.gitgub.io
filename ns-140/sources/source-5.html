135         private void convertType(String path, Object obj) {
136             if (obj instanceof String) {
137                 String type = (String) obj;
138                 convertType(path, type);
139             } else if (obj instanceof Map) {
140                 @SuppressWarnings("unchecked")
141                 Map<String, Object> map = (Map<String, Object>) obj;
142                 convertType(path, map);
143             } else if (obj instanceof Collection) {
144                 @SuppressWarnings("unchecked")
145                 Collection<Object> list = (Collection<Object>) obj;
146                 for (Object item : list) {
147                     convertType(path, item);
148                 }
149             } else {
150                 LOG.debug("unknown type of 'type': {}", obj.getClass().getName());
151                 oops("field - unknown class", this.path, obj.getClass().getName());
152             }
153         }

164         private void convertType(String path, Map<String, Object> map) {
165             SObjectSchema schema = new SObjectSchema(path, map);
166             String type = schema.getName();
167             if (!knownTypes.containsKey(type)) {
168                 types.put(type, schema);
169                 knownTypes.put(type, schema);
170             } else {
171                 LOG.debug("schema for {} contains duplicate type {}", name, type);
172                 oops("field - duplicate type", this.path, type);
173             }
174         }

269      * a collection of two or more strings.
270      *
271      * @param obj
272      */
273     private void convertType(Object obj) {
274         if (obj instanceof String) {
275             String type = (String) obj;
276             if (this.knownTypes.containsKey(type)) {
277                 this.types.put(type, knownTypes.get("type"));
278             } else {
279                 LOG.debug("schema for {} contains unknown type {}", name, type);
280                 oops("type - unknown type", this.path, type);
281             }
282         } else if (obj instanceof Collection) {
283             @SuppressWarnings("unchecked")
284             Collection<Object> coll = (Collection<Object>) obj;
285             for (Object o : coll) {
286                 convertType(o);
287             }
288         } else {
289             LOG.debug("unknown type of 'type': {}", obj.getClass().getName());
290             oops("type - unknown class", this.path, obj.getClass().getName());
291         }
292     }
