181     public void execute() {
182         emailUpdate("start");
183         updateStatus("Start");
184         jobStatusHandler.saveStartedStats(lastRunStatus);
185         List<TableConfig> tables = spec.getTables();
186         jobStatusHandler.setPropagateTableStatusToJobStatus(tables.size() == 1);
187         ArrayList<CompletableFuture<Void>> tasks = new ArrayList<>(tables.size());
188         ExecutorService executor = Executors.newFixedThreadPool(POOL_SIZE);
189         int idx = 0;
190 
191         for (TableConfig tableConfig : tables) {
192             int tableIdx = idx;
193             CompletableFuture<Void> task = CompletableFuture.runAsync(() ->
194                     processTable(tableConfig, tableIdx), executor).exceptionally(e -> {
195                 ErrorDTO errorDto;
196                 Throwable cause = e.getCause();
197                 if (cause instanceof PipelineExecutionException &&
198                         ((PipelineExecutionException) cause).getErrorDto() != null) {
199                     errorDto = ((PipelineExecutionException) cause).getErrorDto();
200                 } else {
201                     errorDto = new ErrorDTO();
202                     errorDto.setFailure("Failed to perform AutoSync for table " +
203                             tableConfig.getName());
204                     errorDto.setReason(AN_AUTO_SYNC_INTERNAL_ERROR_OCCURRED);
205                     errorDto.setResolution(CONTACT_SUPPORT);
206                 }
207                 LOGGER.error("Error processing table {} = {}, {}",
208                         tableConfig.getName(), errorDto, cause);
209                 jobStatusHandler.updateIntermediaryStatus(tableConfig.getTableName(),
210                         errorDto);
211                 return null;
212             });
213             idx++;
214             tasks.add(task);
215         }
216         tasks.forEach(CompletableFuture::join);
217         AutosyncAssetStatsDTO finalStats = jobStatusHandler.updateFinalStats();
218         this.integrationJobStats = finalStats.getTablesStats();
219         //NOTE: show compressed view of the final stats per table
220         emailUpdate("complete");
221         LOGGER.debug("Final stats: {}", finalStats);
222         if (jobStatusHandler.hasErrors()) {
223             this.errorMap = finalStats.getTablesStats().entrySet().stream()
224                     .filter(entry -> !JobStatusHandler.SUCCESS
225                             .equalsIgnoreCase(entry.getValue().getStatus()))
226                     .collect(Collectors.toMap(Map.Entry::getKey,
227                             e -> Optional.ofNullable(e.getValue().getError())
228                                     .orElseGet(AutoSyncUtils::getGenericErrorDto)));
229             LOGGER.debug("Execution Has errors = {}", errorMap);
230             outputViews.write(documentUtility.newDocument(Map.of("errors", errorMap)));
231             emailUpdate("error");
232             for (Map.Entry<String, ErrorDTO> errorDTOEntry : errorMap.entrySet()) {
233                 errorViews.write((SnapDataException) new SnapDataException(
234                         String.format("Integration failed for %s with message: %s",
235                                 errorDTOEntry.getKey(),
236                                 errorDTOEntry.getValue().getFailure()))
237                         .withReason(errorDTOEntry.getValue().getReason())
238                         .withResolution(errorDTOEntry.getValue().getResolution()));
239             }
240         }
241         executor.shutdown();
242     }

265         LOGGER.debug("AutoSyncExecutionJobHandler: begin processTable for {}",
266                 tableConfig.getTableName());
267 
268         delay(tableConfig, tableIdx);
269         var lastRunTblStats = lastRunStatus.getTablesStats();
270         String tableName = tableConfig.getName();
271         if (lastRunTblStats != null &&
272                 lastRunTblStats.containsKey(tableName)) {
273             setLastModifiedRecord(tableConfig, lastRunTblStats.get(tableName));
274         }
275         // Pipeline -- SyncTableDefinition
276         Endpoint src = spec.getSource();
277         Endpoint tgt = spec.getTarget();
278 
279         jobStatusHandler.updateIntermediaryStatus(tableName,
280                 "Schema Inference Started");
281 
282         LOGGER.trace("AutoSyncExecutionJobHandler: begin executing source definitions: src {}," +
283                 " tableConfig {}, path {}",
284                 prettyPrint(src), prettyPrint(tableConfig), spec.getPath());
285 
286         if (StringUtils.equalsIgnoreCase(CSV_FILE_READER, tableConfig.getSrcDbType())) {
287             addFieldsToExtraParamsForCSVFileReaderSource(tableConfig);
288         }
289         Map<String, Object> tgtTableDef = endpointService.definitions(
290                 tgt, tableConfig, "columns", spec.getPath(), "target");
291         
292         boolean isTgtTableExists = isTableExist(tgtTableDef);
293         LoadType loadType = fromString(tableConfig.getLoadMethod());
294         if (!isTgtTableExists) {
295             loadType = RELOAD;
296         }
297         setTableConfigFromLoadType(tableConfig, loadType);
298 
299         Map<String, Object> srcTableDef = endpointService.definitions(

301 
302         src.withPrimaryKey(getColumnNameByBooleanAttributeName(srcTableDef,
303                 IS_PRIMARY_KEY_COLUMN));

305                 IS_LAST_UPDATE_COLUMN));
306         tgt.withPrimaryKey(getColumnNameByBooleanAttributeName(tgtTableDef,
307                 IS_PRIMARY_KEY_COLUMN));
308         tableConfig.setLastUpdateColumn(src.getLastUpdateColumn());
309         tableConfig.setPrimaryKey(src.getPrimaryKey());
310 
311         DataMapper cdm = parseCDM(tgtTableDef);
312 
313         NameHandler nameHandler = new NameHandlerFactory()

317 
318         SchemaService schemaService = new SchemaService(jobStatusHandler, nameHandler);
319         schemaService.enquoteNames(tableConfig.isEnquoteNames());
320 
321         jobStatusHandler.updateIntermediaryStatus(tableName,
322                 "Schema Inference Complete");
323 
324         LOGGER.info("AutoSyncExecutionJobHandler: begin syncTable for {}",

327                 endpointService, srcTableDef, tgtTableDef, spec);
328         SyncTableActionStatus actionStatus =
329                 (SyncTableActionStatus) inputToSyncData.remove(1).get("actionStatus");
330 
331         jobStatusHandler.updateIntermediaryStatus(tableName,
332                 "Target Data Loading Started");
333 
334         loadType = RELOAD.equals(fromString(tableConfig.getLoadMethod())) ? RELOAD :
335                 getLoadMethod(isTgtTableExists, inputToSyncData, tgt, tableConfig, nameHandler);
336 
337         setTableConfigFromLoadType(tableConfig, loadType);
338 
339         // preprocess for syncTable

342 
343         // Get org name from path.

345 
346         if (tableConfig.isEnableErrorTable()) {

364         //emailUpdate("per_table_email");

366                 tableConfig.getTableName());
367 
368         if (tableConfig.isEnableErrorTable()) {
369             dropErrorIdTable(tgt, tableConfig, inputToSyncData, nameHandler);

371 
372         updateStatus((String) syncDataRes.get("status"));
373     }

550      * Add a delay before each execution to avoid the same plex scheduling issue.
551      *
552      * @param tableConfig {@link TableConfig} instance
553      * @param tableIdx index of the table
554      */
555     private void delay(TableConfig tableConfig, int tableIdx) {
556         String table = tableConfig.getTableName();
557         Duration delay = calculateDelay(tableIdx);
558         if (delay.isZero()) {
559             return;
560         }
561         try {
562             Thread.sleep(delay.toMillis());
563         } catch (InterruptedException e) {
564             throw new ExecutionException(e, "Interrupted before loading {}", table);
565         }
566         LOGGER.debug("Table {}-{} started after {}s delay", tableIdx, table, delay.getSeconds());
567     }
