393     private void startCompletionReceiver() {
394         execActivityThread = spawnHelperThread(() -> {
395                     while (looping) {
396                         try {
397                             LOG.debug("Waiting for completions");
398                             AbstractActivityHandler handler = executionActivity.poll(MAX_WAIT_TIME,
399                                     MAX_WAIT_UNIT);
400                             if (handler != null) {
401                                 LOG.debug("  Received completion: {}", handler);
402                                 handler.run();
403                             }
404                         } catch (InterruptedException e) {
405                             LOG.debug("Interrupted while waiting for pipeline completions!", e);
406                             return;
407                         } catch (Throwable th) {
408                             handleUnexpectedException(th);
409                         }
410                     }
411                 });
412         execActivityThread.start();
413     }

459     private void shutdown() {
460         Thread.interrupted(); // Clear the interrupt state
461         looping = false;
462         if (execActivityThread != null) {
463             try {
464                 execActivityThread.join(EXEC_ACTIVITY_THREAD_TIMEOUT_SECONDS * 1000);
465             } catch (InterruptedException e) {
466                 // no-op
467             }
468         }
469         if (inputThread != null) {
470             inputThread.stop();
471         }
472         LOG.info("Shutting down any remaining child executions");
473         for (ExecutionTrackerInterface executionTracker : executions.values()) {
474             executionTracker.stop();
475         }
476         executions.clear();
477         if (aborted) {
478             // On a failure, we want to encourage the writer threads to exit.
479             for (Thread thread : writers.values()) {
480                 thread.interrupt();
481             }
482         }
483     }

510     /**
511      * If the child is not executed during normal validation, we need to try and do it here.  Since
512      * we're doing this during the schema propagation stage, the platform should not fully execute
513      * the pipeline and stop after defining schema.
514      */
515     private void executeChildForSchema() {
516         if (schemaProvider != null && !definedSchema) {
517             try {
518                 schemaExec = true;
519                 looping = true;
520                 startCompletionReceiver();
521                 pool = new Semaphore(1);
522                 try {
523                     process(ExecEvent.nullDocument());
524                     for (ExecutionTrackerInterface tracker : executions.values()) {
525                         tracker.closeInputViews();
526                     }
527                     // Wait for the pipeline to finish.
528                     execEvents.take();
529                 } catch (SnapDataException e) {
530                     LOG.warn("Unable to get schema", e);
531                 }
532 
533                 looping = false;
534                 execActivityThread.interrupt();
535                 execActivityThread.join();
536             } catch (SnapDataException|InterruptedException e) {
537                 LOG.warn("Unable to get schema", e);
538             }
539         }
540     }

574     protected void handleErrors(PipelineRun pipeRun, TypeEnum type,
575                                    String source, String target, String action) {
576         PipelineRun.PipelineInfo pipelineInfo = pipeRun.getCurrentInfo();
577         Map<String, Object> snapMap =
578                 castToMap(pipelineInfo.getRawInfo().get("snap_map"));
579         List<Map<String, String>> pipeErrors = pipelineInfo.getErrors();
580         if (pipeErrors != null && !pipeErrors.isEmpty()) {
581             ErrorDTO errorDto = pipeErrors.stream()
582                     .map(error -> transformError(error, snapMap, type, source, target))
583                     .map(AutoSyncUtils::extractException)
584                     .findFirst().get();
585             errorDto.setFailedAction(action);
586             throw new PipelineExecutionException(errorDto);
587         }

594     private RuntimeException transformError(Map<String, String> error,
595                                             Map<String, Object> snapMap,
596                                             TypeEnum type, String source, String target) {
597         String pipeExecLabel = error.get("label");
598         String failureReason = error.get("reason");
599         String failureMessage = error.get("failure");
600         String failedSnapLabel = StringUtils.substringBetween(failureReason, "label=", ",");
601         if (StringUtils.isBlank(failedSnapLabel)){
602             failedSnapLabel = pipeExecLabel;
603         }
604         String snapClassFqId =
605                 getFailedSnapClassFqId(snapMap, failedSnapLabel);
606 
607         if (StringUtils.isBlank(snapClassFqId)) {
608             snapClassFqId =
609                     StringUtils.substringBetween(failureReason, "snap_fqid=", "}");
610         }
611 
612         if (StringUtils.isBlank(snapClassFqId)) {
613             snapClassFqId = getFailedSnapClassFqId(snapMap, pipeExecLabel);
614         }
615         ErrorHandler errorHandler =
616                 ErrorHandlerCreator.getErrorHandler(snapClassFqId);
617 
618         String stringError = failureMessage.contains(GENERIC_PIP_EXEC_ERROR_MESSAGE) ?
619                 failureReason : StringUtils.join(error);
620 
621         return errorHandler.handle(
622                 stringError, type, source, target);
623     }

627         return snapMap.values().stream()
628                 .map(BasePipeExecSnap::castToMap)
629                 .filter(val -> val.get("label").toString().equalsIgnoreCase(failedSnapLabel))
630                 .map(val -> val.get("snap_class_fqid").toString())
631                 .findFirst()
632                 .orElse("");

655      * Propagates the output from child pipeline's output view to the parent's output view
656      * @param pipeRun pipeline run object
657      */
658     protected List<Document> getDocumentsFromChild(PipelineRun pipeRun, TypeEnum type,
659                                                    String source, String target, String action) {
660         List<Document> documents = new ArrayList<>();
661         PipelineState state = pipeRun.getCurrentInfo().getPipelineState();
662         if (state != PipelineState.FAILED) {
663             // get data from child's output view
664             InputViews iVs = pipeRun.getPipelineOutputs();
665             try {
666                 if (iVs.size() >= 1) {
667                     for (Document document : (DocumentInputView) iVs.get()) {
668                         documents.add(document);
669                     }
670                 }
671             } catch (RuntimeException e) {
672                 LOG.error("Pipeline execution failed", e);
673                 handleErrors(pipeRun, type, source, target, action);
674                 throw e;
675             }
676         } else {
677             handleErrors(pipeRun, type, source, target, action);
678         }
679         return documents;
680     }
