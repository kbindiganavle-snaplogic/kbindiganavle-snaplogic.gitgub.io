316     public void configure(PropertyValues propertyValues) throws ConfigurationException {
317         // This Snap does not support Ultra Task.
318         if (propertyValues.inImmediateMode()) {
319             throw new ConfigurationException(INCOMPATIBLE_USAGE_OF_THIS_SNAP)
320                     .withReason(INCOMPATIBLE_ULTRA_SNAP)
321                     .withResolution(RESOLUTION_REMOVE_SNAP);
322         }
323 
324         // This Snap does not support hierarchical data.
325         try {
326             List<Object> jsonPathTokens = labelFieldJsonPath.resolvedPath(null);
327             if (jsonPathTokens.isEmpty()) {
328                 throw new ConfigurationException(ERR_LAB_INVALID)
329                         .withReason(ERR_LAB_INVALID_REASON)
330                         .withResolution(ERR_LAB_INVALID_RESO);
331             }
332             if (jsonPathTokens.size() > 1) {
333                 throw new ConfigurationException(ERR_LAB_X_NOT_VALID).formatWith(labelFieldJsonPath)
334                         .withReason(ERR_HIER_REASON)
335                         .withResolution(RESOLUTION_ADDRESS_ISSUE);
336             }
337             this.labelFieldName = String.valueOf(jsonPathTokens.get(0));
338         } catch (InvalidPathException e) {
339             throw new ConfigurationException(e, ERR_LAB_NOT_FOUND)
340                     .withResolution(RESOLUTION_ADDRESS_ISSUE);
341         }
342 
343         String engineName = engineExpression.eval(null);
344         Optional<AutoMLEngine> engineValue =
345                 AutoMLEngine.stream().filter(e -> e.getDisplayName().equals(engineName)).findFirst();
346         if (!engineValue.isPresent()) {
347             throw new ConfigurationException(ERR_ENGINE_INVALID).formatWith(engineName)
348                     .withReason(ERR_ENGINE_INVALID_REASON)
349                     .withResolution(ERR_ENGINE_INVALID_RESO);
350         }
351         this.engine = engineValue.get();
352 
353         this.timeLimit = timeLimitExpression.eval(null).longValue();
354         this.modelLimit = modelLimitExpression.eval(null).intValue();
355         this.fold = foldExpression.eval(null).intValue();
356         this.seed = seedExpression.eval(null).longValue();
357         this.title = repTitleExp.eval(null).orElse("");
358 
359         // Either time or number of models needs to be limited. 0 means unlimited.
360         if (this.timeLimit == 0 && this.modelLimit == 0) {
361             throw new ConfigurationException(ERR_TIME_MODEL_INVALID).formatWith(engineName)
362                     .withReason(ERR_TIME_MODEL_INVALID_REASON)
363                     .withResolution(ERR_TIME_MODEL_INVALID_RESO);
364         }
365 
366         // get and parse options
367         Optional<String> algorithmsGroupValue = this.algorithmsExpression.eval(null);
368 
369         if (algorithmsGroupValue.isPresent()) {
370 
371             algorithmGroupSet = new HashSet<>();
372             for (String algorithmsGroupStr : algorithmsGroupValue.get().split(",")) {
373                 Optional<AlgorithmGroup> algorithmGroup = AlgorithmGroup.getAlgorithmGroupFromDisplayName(algorithmsGroupStr.trim());
374 
375                 if (!algorithmGroup.isPresent()) {
376                     throw new ConfigurationException(ERR_ALGORITHMS_INVALID).formatWith(algorithmsGroupStr)
377                             .withReason(ERR_ALGORITHMS_INVALID_REASON)
378                             .withResolution(ERR_ALGORITHMS_INVALID_RESO);
379                 }
380 
381                 if (algorithmGroupSet.contains(algorithmGroup.get())) {
382                     throw new ConfigurationException(ERR_DUPLICATED_ALGORITHMS).formatWith(algorithmsGroupStr)
383                             .withReason(ERR_DUPLICATED_ALGORITHMS_REASON)
384                             .withResolution(ERR_DUPLICATED_ALGORITHMS_RESO);
385                 }
386 
387                 algorithmGroupSet.add(algorithmGroup.get());
388             }
389 
390             if (algorithmGroupSet.size() == 0) {
391                 throw new ConfigurationException(ERR_EMPTY_ALGORITHMS)
392                         .withReason(ERR_EMPTY_ALGORITHMS_REASON)
393                         .withResolution(ERR_EMPTY_ALGORITHMS_RESO);
394             }
395 
396         } else {
397             algorithmGroupSet = EnumSet.allOf(AlgorithmGroup.class);
398         }
399 
400         // Snap-level temporary directory.
401         File pipeTmpDir = JfsUtils.getTmpDir(expressionUtils, propertyValues, errorViews);
402         try {
403             snapTmpDir = Files.createTempDirectory(pipeTmpDir.toPath(), "").toFile();
404         } catch (IOException e) {
405             throw new ConfigurationException(ERR_CREATE_TEMP_DIR)
406                     .withReason(ERR_CREATE_TEMP_DIR_REASON)
407                     .withResolution(ERR_CREATE_TEMP_DIR_REASON_RESO);
408         }
409 
410         // H2O engine uses global storage. Each Snap will be assigned a unique value using
411         // runtime id and Snap id.
412         try {
413             this.uniqueId = expressionUtils.createExpressionProperty(
414                     propertyValues, true,
415                     "(pipe.ruuid + snap.instanceId).replaceAll(\"-\",\"\")")
416                     .eval(null);
417         } catch (SnapDataException e) {
418             this.uniqueId = UUID.randomUUID().toString();
419         }
420 
421         try {
422             // Pipeline validation.
423             isSuggest = expressionUtils.createExpressionProperty(
424                     propertyValues, true,
425                     "pipe.flags.is_suggest")
426                     .eval(null);
427         } catch (Exception e) {
428             isSuggest = false;
429         }
430 
431         // In the pipeline validation mode, time and number of models must be limited.
432         if (isSuggest) {

526                             List<Map<String, Object>> leaderboard = h2OManager.getLeaderboard();
527                             for (Map<String, Object> record : leaderboard) {
528                                 outputViews.getDocumentViewFor(outputViewNameList.get(1))
529                                         .write(documentUtility.newDocument(record));
530                             }
531                         }
532

534                         if (outputViewNameList.size() > 2) {

577      * Read checkpoint model if available, build the model using WEKA engine and output model,
578      * leaderboard, and report.
579      */
580     private void startWekaEngine(InputView checkpointInputView) throws IOException, ClassNotFoundException {
581 
582         AutoWEKAClassifierExternalModel externalModel = null;
583         int seedFromSecondInput = -1;
584 
585         // if summary data is given, create special configuration to build model
586         if (checkpointInputView != null) {
587             Iterator<Document> documentIterator = inputViews.getDocumentsFrom(checkpointInputView);
588             int count = 0;
589             while (documentIterator.hasNext()) {
590                 if (count == 1) {
591                     throw new SnapDataException(ERR_LOAD_BEST_MODEL)
592                             .withReason(ERR_MULTIPLE_BEST_MODEL_REASON)
593                             .withResolution(RESOLUTION_ADDRESS_ISSUE);
594                 }
595                 Document document = documentIterator.next();
596                 Map<String, Object> inputMap = document.get(Map.class);
597                 seedFromSecondInput = extractSeedFromCheckpointInputView(inputMap);
598                 externalModel = extractModelFromSecondInput(inputMap);
599                 count++;
600             }
601         }
602         AutoWekaParams params = new AutoWekaParams();
603         params.setSnapTmpFolder(snapTmpDir);
604         params.setDataInstances(dataset.getInstances());
605         params.setTimeLimit(timeLimit);
606         params.setFoldNo(fold);
607 
608         Set<String> allowedAlgorithmSet = AlgorithmGroup.getWekaAllowedAlgorithm(algorithmGroupSet);
609 
610         params.setAllowedAlgorithmSet(allowedAlgorithmSet);
611         int seed = (applySeed) ?
612                 (seedFromSecondInput >= 0 ? seedFromSecondInput : (int) this.seed) :
613                 (int) (Math.random() * Integer.MAX_VALUE);
614         params.setSeed(seed);
615         params.setModelLimit(modelLimit);
616         params.setExternalModel(externalModel);
617         if (dataset.getDatasetType() == DatasetType.CLASSIFICATION) {
618             params.setMetric(WEKA_CLS_METRIC);
619         } else {
620             params.setMetric(WEKA_REG_METRIC);
621         }
622 
623         AutoWekaWrapper wrapper = new AutoWekaWrapper(params);
624 
625         AutoWEKAResult autoWEKAResult = wrapper.run();
626 
627         if(autoWEKAResult.getEvaluation() == null) {
628             // Weka run successfully but don't generate any model result
629             // number of data input might be too low
630             throw new SnapDataException(ERR_CAANOT_GENERATE_MODEL)
631                     .withReason(ERR_CAANOT_GENERATE_MODEL_REASON)
632                     .withResolution(ERR_CAANOT_GENERATE_MODEL_RESO);
633         }
634 
635         createAutoWekaOutput(autoWEKAResult);
636 
637     }
