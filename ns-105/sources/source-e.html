314     public void processHelper() {
315         while (script != null && !pendingList.isEmpty()) {
316             Document document = pendingList.removeFirst();
317             if (document == null) {
318                 continue;
319             }
320             if (isCompleted.get()) {
321                 try {
322                     // Send ending message to Remote Python Executor.
323                     sockOut.newLine();
324                     sockOut.flush();
325                 } catch (IOException e) {
326                     errorViews.write((SnapDataException) new SnapDataException(ERR_SEND_DATA)
327                             .withReason(ERR_SEND_DATA_REASON)
328                             .withResolution(RESOLUTION_TRY_AGAIN));
329                     break;
330                 }
331                 waitForCompletion();
332                 continue;
333             }
334             try {
335                 docList.add(document);
336                 Map<String, String> documentMap = document.get(Map.class);
337                 Map<String, Object> messageMap = new HashMap<>();
338 
339                 // Piggyback python script with the first row.
340                 if (firstRow) {
341                     messageMap.put("script", script);
342                     messageMap.put("token", token);
343                     messageMap.put("pipe_info", pipeInfo);
344                 }
345                 hasInputDoc = true;
346                 firstRow = false;
347 
348                 messageMap.put("type", "data");
349                 messageMap.put("data", documentMap);
350                 String message = snapObjectMapper.writeValueAsString(messageMap);
351                 sockOut.write(message);
352                 sockOut.newLine();
353                 if (currentInFlight.incrementAndGet() == BATCH_SIZE) {
354                     sockOut.flush();
355                     currentInFlight.set(0);
356                 }
357             } catch (SnapDataException e) {
358                 errorViews.write(e, document);
359             } catch (JsonProcessingException e) {
360                 errorViews.write((SnapDataException) new SnapDataException(e, ERR_SERIALIZE)
361                         .withResolution(RESOLUTION_ADDRESS_ISSUE), document);
362             } catch (IOException e) {
363                 errorViews.write((SnapDataException) new SnapDataException(e, ERR_SEND_DATA)
364                         .withReason(ERR_SEND_DATA_REASON)
365                         .withResolution(RESOLUTION_TRY_AGAIN));
366             }
367         }
368     }

370     @Override
371     public void handle(final LifecycleEvent event) {
372         switch (event) {
373             case SUCCESS:
374                 try {
375                     if (script == null) {
376                         throw new SnapDataException(ERR_LOAD_SCRIPT)
377                                 .withReason(ERR_SCRIPT_NOT_AVAI_REASON)
378                                 .withResolution(RESOLUTION_ADDRESS_ISSUE);
379                     }
380                     processHelper();
381                     if (isCompleted.get()) {
382                         waitForCompletion();
383                         return;
384                     }
385                     Map<String, Object> messageMap = new HashMap<>();
386                     // Piggyback python script if there is no input from upstream snap.
387                     if (!hasInputDoc) {
388                         messageMap.put("script", script);
389                         messageMap.put("token", token);
390                         messageMap.put("pipe_info", pipeInfo);
391                     }
392                     messageMap.put("type", "finish");
393                     String finishScript = snapObjectMapper.writeValueAsString(messageMap);
394                     sockOut.write(finishScript);
395                     sockOut.newLine();
396                     sockOut.newLine(); // Indicate the end of stream.
397                     sockOut.flush();
398                     waitForCompletion();
399                 } catch (SnapDataException e) {
400                     errorViews.write(e);
401                 } catch (JsonProcessingException e) {
402                     errorViews.write(new SnapDataException(e, ERR_SERIALIZE));
403                 } catch (IOException e) {
404                     errorViews.write((SnapDataException) new SnapDataException(e, ERR_SEND_DATA)
405                             .withReason(ERR_SEND_DATA_REASON)
406                             .withResolution(RESOLUTION_TRY_AGAIN));

425     private void waitForCompletion() {
426         completeLock.lock();
427         try {
428             while (!isCompleted.get()) {
429                 try {
430                     notComplete.await();
431                 } catch (InterruptedException e) {
432                     break;
433                 }
434             }
435         } finally {
436             completeLock.unlock();
437         }
438         if (!exceptionList.isEmpty()) {
439             SnapDataException e = exceptionList.remove(0);
440             if (e.getMessage().equals(PYT_TOKEN_NOT_MATCH_ERR)) {

443                         .withResolution(ERR_TOKEN_NOT_MATCH_RESO);

446         }

496                     outputDocument(mapObject);

498                     exceptionList.add(e);
499                     break;
500                 } catch (JsonProcessingException e) {
501                     errorViews.write((SnapDataException) new SnapDataException(e, ERR_SERIALIZE)
502                             .withResolution(RESOLUTION_ADDRESS_ISSUE));
503                 } catch (IOException e) {
504                     errorViews.write((SnapDataException) new SnapDataException(e, ERR_SEND_DATA)
505                             .withReason(ERR_SEND_DATA_REASON)
506                             .withResolution(RESOLUTION_TRY_AGAIN));
507                 }
508             }
509             completeLock.lock();
510             try {
511                 isCompleted.set(true);
512                 notComplete.signalAll();
513             } finally {
514                 completeLock.unlock();
515             }
516         }

518         private void outputDocument(Map<String, Object> result) throws SnapDataException {
519             // Exception appears.
520             if (result.containsKey(PYT_EXE_ERROR)) {
521                 throw new SnapDataException(String.valueOf(result.get(PYT_EXE_ERROR)))
522                         .withReason(ERR_SCRIPT_SCRIPT_ERROR_REASON)
523                         .withResolution(ERR_SCRIPT_SCRIPT_ERROR_RESO);
524             }
525             // Do not output heartbeat message.
526             if (PYT_HEARTBEAT.equals(result.get(PYT_FLAG))) {
527                 return;
528             }
529             if (result.containsKey(PYT_LOG)) {
530                 // Log is whatever the user use print statement in python.
531                 Set<String> outputViewNames = outputViews.names();
532                 for (String outputViewName : outputViewNames) {
533                     if (!outputViewName.equals(OUTPUT_0)) {
534                         Map<String, Object> stdoutMap = new LinkedHashMap<>();
535                         stdoutMap.put(CONTENT_FIELD, result.get(PYT_LOG));
536                         outputViews.getDocumentViewFor(outputViewName).write(
537                                 documentUtility.newDocument(stdoutMap));
538                         break;
539                     }
540                 }
541                 return;
542             }
