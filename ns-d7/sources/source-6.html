63      *      ASC NULLS LAST ROWS BETWEEN UNBOUNDED PRECEDING AND UNBOUNDED FOLLOWING )
64      *      test_result_field1,

102         if (passThrough) {

104                     String.format(WINDOW_AND_PASS_THROUGH_SQL_TYPE_TWO_WITH_SUBQUERY,

107                     buildMainQueryOnClauseForTypeTwo(randomString, theFunction));
108         } else {

124         StringBuilder rewrittenStrBuilder = new StringBuilder();

144     class DefaultTypeTwoPerWindowFunctionProcessor implements PerWindowFunctionProcessor {
145         /**
146          * Builds final main SELLECT expression and subquery for the given "function"
147          * and return the main SELECT expression.
148          * @param function
149          * @param inputString
150          * @param randomString
151          * @return  main SELECT expression for the given "function"
152          */
153         @Override
154         public String processPerWindowFunction(
155                 RewriterFunction function, RewriterFunction first, String inputString,
156                 String randomString, boolean isFirstFunction, List<String> partitionByFields) {
157             if (function.getOriginalArrayIdx() != function.getCompressedArrayIdx()) {
158                 // Skip adding subquery if an equivalent function to the current function already
159                 // exists.
160                 return String.format(DUPLICATE_ITEM_TYPE_ONE_WITH_SUBQUERY_FMT,
161                         compressedMap.get(function.getKeyToCompressedMap()).getResultExprStr(),
162                         function.getResultExprStr());
163             }
164             String rewrittenSqlStr = rewriterDatabase.rewriteFormulae(
165                     function.getFunctionName(),
166                     function.getField1Str(),
167                     function.getField2Str(),
168                     inputString,
169                     randomString,
170                     partitionByFieldsList);
171             function.setUpContextForTypeTwoRewriteWithSubquery(partitionByFields);
172             function.setFunctionRewriteStr(rewrittenSqlStr);
173 
174             // Use first fuction's base subquery alias for this type two rewriter.
175             return String.format(TABLE_NAME_DOT_COLUMN_FMT, first.getBaseSubqueryAlias(),
176                     function.getResultExprStr());
177         }

180     /**
181      * Build the main query ON clause for rewriter type two and
182      * return the string.
183      * For example, in the example of process(), D+15 is an example ON clause.
184      */
185     protected String buildMainQueryOnClauseForTypeTwo(String randomString,
186                                                       RewriterFunction theFunction) {
187         StringBuilder onClauseStrBuilder = new StringBuilder(ON_IN_SPACES);
188 
189         boolean firstSubexpr = true;
190         for (int i = 0; i < this.partitionByFields.size(); i++) {
191             if (firstSubexpr) {
192                 firstSubexpr = false;
193             } else {
194                 onClauseStrBuilder.append(AND_IN_SPACES);
195             }
196             String currentExpr = partitionByFields.get(i);
197             if (!currentExpr.equals(this.partitionByColumnList.get(i))) {
198                 String columnName = this.partitionByColumnList.get(i);
199                 currentExpr = currentExpr.replaceAll(columnName,
200                         String.format(DERIVED_TABLE_NAME_DOT_COLUMN_FMT, randomString, columnName));
201                 onClauseStrBuilder.append(String.format(
202                         MAIN_QUERY_ON_CLAUSE_ENTRY_ALREADY_MODIFIED_FMT,
203                         currentExpr, theFunction.getSelectKeyFieldAliases().get(i)));
204             } else {
205                 onClauseStrBuilder.append(String.format(MAIN_QUERY_ON_CLAUSE_ENTRY_FMT,
206                         randomString, currentExpr, theFunction.getSelectKeyFieldAliases().get(i)));
207             }
208         }
209         return onClauseStrBuilder.toString();
210     }
