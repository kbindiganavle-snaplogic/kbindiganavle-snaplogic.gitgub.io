13 import java.util.LinkedHashMap;
14 import java.util.List;
15 import java.util.Map;

17 /**
18  * Schema handler utils

20 public class SchemaHandler {
21     /**
22      * Builds and returns source column name string or source column converting SQL
23      * expression string for all the source columns from the given 'convertedSchema'.
24      *
25      * For example, for SQL Server source endpoint, we create and save
26      * "convert(varchar, \"UpdatedAt\", 21) as \"UPDATEDAT\"" as selectColumn
27      * when cdm selectAs entry for the column type exists, target colName is "UPDATEDAT"
28      * and colDef.sourceColumn "UpdatedAt". Otherwise we save source column name as is.
29      *
30      * @param sourceMapper      cdm map
31      * @param convertedSchema   column schema list
32      * @return      columnDefinition map with selectColumn if necessary
33      */
34     public static Map<String, ColumnDefinition> prepareDefinitionForExtractValue(
35             DataMapper sourceMapper, List<ColumnDefinition> convertedSchema) {
36         Map<String, String> srcSelectAs = sourceMapper.getSelectAs();
37         Map<String, ColumnDefinition> srcColumnDefMap = new LinkedHashMap<>();
38         for (ColumnDefinition colDef : convertedSchema) {
39             String srcColName = colDef.getSourceColumn();
40             String sourceType = colDef.getSourceType();
41             String srcColumnNameStr = "\"" + srcColName + "\"";
42             String selectColumn = srcColumnNameStr;
43             if (srcSelectAs != null && srcSelectAs.containsKey(sourceType)) {
44                 selectColumn = srcSelectAs.get(sourceType)
45                         .replace("$column", srcColumnNameStr) + " as " + srcColumnNameStr;
46             }
47             colDef.setSelectColumn(selectColumn);
48             srcColumnDefMap.put(colDef.getColumn(), colDef);
49         }
50         return srcColumnDefMap;
51     }
