61     void defineInputSchema(SchemaProvider provider, Operation operation, CoreObject coreObject) {
62         String viewName = provider.getRegisteredViewNames().iterator().next();
63         SchemaBuilder builder = provider.getSchemaBuilder(viewName);
64         boolean isCreate = operation == Operation.CREATE;
65         if (operation == Operation.UPDATE) {
66             builder.withChildSchema(createSchema(provider, STRING, ID, true));
67         }
68         switch (coreObject) {
69             case ACCOUNTS:
70             case ACCOUNTS_2:
71                 builder.withChildSchema(provider.createSchema(STRING, NAME))
72                         .withChildSchema(provider.createSchema(STRING, ADDRESS_1))
73                         .withChildSchema(provider.createSchema(STRING, ADDRESS_2))
74                         .withChildSchema(provider.createSchema(STRING, ADDRESS_3))
75                         .withChildSchema(provider.createSchema(STRING, CITY))
76                         .withChildSchema(provider.createSchema(STRING, PROVINCE))
77                         .withChildSchema(provider.createSchema(STRING, POSTAL_CODE))
78                         .withChildSchema(provider.createSchema(STRING, COUNTRY))
79                         .withChildSchema(provider.createSchema(STRING, BUSINESS_PHONE));
80                 break;
81             case CONTACTS:
82             case CONTACTS_2:
83                 if (operation == Operation.UPDATE) {
84                     // for some reason, emailFormatPreference works on update, but not create
85                     builder.withChildSchema(provider.createSchema(STRING, EMAIL_FORMAT_PREFERENCE));
86                 }
87                 builder.withChildSchema(provider.createSchema(STRING, EMAIL_ADDRESS))
88                         .withChildSchema(provider.createSchema(STRING, FIRST_NAME))
89                         .withChildSchema(provider.createSchema(STRING, LAST_NAME))
90                         .withChildSchema(provider.createSchema(STRING, ACCOUNT_NAME))
91                         .withChildSchema(provider.createSchema(STRING, ADDRESS_1))
92                         .withChildSchema(provider.createSchema(STRING, ADDRESS_2))
93                         .withChildSchema(provider.createSchema(STRING, ADDRESS_3))
94                         .withChildSchema(provider.createSchema(STRING, CITY))
95                         .withChildSchema(provider.createSchema(STRING, PROVINCE))
96                         .withChildSchema(provider.createSchema(STRING, POSTAL_CODE))
97                         .withChildSchema(provider.createSchema(STRING, COUNTRY))
98                         .withChildSchema(provider.createSchema(STRING, BUSINESS_PHONE))
99                         .withChildSchema(provider.createSchema(STRING, MOBILE_PHONE))
100                         .withChildSchema(provider.createSchema(STRING, FAX))
101                         .withChildSchema(provider.createSchema(STRING, TITLE))
102                         .withChildSchema(provider.createSchema(STRING, SALES_PERSON))
103                         .withChildSchema(provider.createSchema(STRING, IS_SUBSCRIBED))
104                         .withChildSchema(provider.createSchema(STRING, IS_BOUNCEBACK));
105                 break;
106             case CUSTOM_OBJECT_DATA:
107                 // add a required array-type schema 'fieldValues'
108                 ObjectSchema fieldValues = createSchema(provider, TABLE, FIELD_VALUES, true);
109                 builder.withChildSchema(fieldValues);
110                 // add a map-type child-schema
111                 ObjectSchema map = createSchema(provider, COMPOSITE, null, false);
112                 fieldValues.addChild(map);
113                 // add a required string-type child-schema 'id'
114                 map.addChild(createSchema(provider, STRING, ID, true));
115                 // add a required string-type child-schema 'value'
116                 map.addChild(createSchema(provider, STRING, VALUE, true));
117                 break;
118             case CUSTOM_OBJECT_METADATA:
119             case CUSTOM_OBJECT_METADATA_2:
120                 // add a required string-type schema 'name'
121                 builder.withChildSchema(createSchema(provider, STRING, NAME, isCreate));
122                 // add a string-type schema 'description'
123                 builder.withChildSchema(createSchema(provider, STRING, DESCRIPTION, false));
124                 if (isCreate) {
125                     // add a required array-type schema 'fields'
126                     ObjectSchema fields = createSchema(provider, TABLE, FIELDS, true);
127                     builder.withChildSchema(fields);
128                     // add a map-type child-schema
129                     map = createSchema(provider, COMPOSITE, null, false);
130                     fields.addChild(map);
131                     // add a required string-type child-schema 'name'
132                     map.addChild(createSchema(provider, STRING, NAME, true));
133                     // add a required string-type child-schema 'dataType'
134                     map.addChild(createSchema(provider, STRING, DATA_TYPE, true));
135                     // add a string-type child-schema 'defaultValue'
136                     map.addChild(createSchema(provider, STRING, DEFAULT_VALUE, false));
137                     // add a string-type child-schema 'displayType'
138                     map.addChild(createSchema(provider, STRING, DISPLAY_TYPE,
139                             coreObject == CoreObject.CUSTOM_OBJECT_METADATA_2));
140                 }
141                 break;
142             default:
143                 // TODO: define more complete schemas for other types
144                 builder.withChildSchema(provider.createSchema(STRING, NAME));
145         }
146         builder.build();
147     }
