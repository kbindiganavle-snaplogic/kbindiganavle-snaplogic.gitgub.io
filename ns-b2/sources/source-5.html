178     protected void addAdditionalParams(URIBuilder builder, PropertyValues propertyValues) {
179         List<Map<String, Object>> additionalParameters = propertyValues.get(PARAMETER_TABLE);
180         if (additionalParameters != null) {
181             for (Map<String, Object> additionalParameter : additionalParameters) {
182                 String name = JsonPath.read(additionalParameter, PARAMETER_NAME_PATH);
183                 String value = JsonPath.read(additionalParameter, PARAMETER_VALUE_PATH);
184 
185                 if (StringUtils.isBlank(name)) {
186                     throw new ConfigurationException(ERROR_EMPTY_ADDITIONAL_PARAMETERS)
187                             .withReason(PARAMETER_NAME_LABEL + REASON_EMPTY_ADDITIONAL_PARAMETERS)
188                             .withResolution(RESOLUTION_EMPTY_ADDITIONAL_PARAMETER_NAME);
189                 }
190                 if (StringUtils.isBlank(value)) {
191                     throw new ConfigurationException(ERROR_EMPTY_ADDITIONAL_PARAMETERS)
192                             .withReason(PARAMETER_VALUE_LABEL + REASON_EMPTY_ADDITIONAL_PARAMETERS)
193                             .withResolution(RESOLUTION_EMPTY_ADDITIONAL_PARAMETER_VALUE);
194                 }
195                 // Compensate for anyone who was working around axod:8358 so that their workaround doesn't break things
196                 // now that the defect is fixed.  See the last line in the description for this defect.
197                 boolean decodeValue = false;
198                 if (name.endsWith("=")) {
199                     decodeValue = true;
200                     name = name.substring(0, name.length() - 1); // remove the '='
201                 } else if (value.startsWith("=")) {
202                     decodeValue = true;
203                     value = value.substring(1); // remove the '='
204                 }
205                 if (decodeValue) {
206                     try {
207                         value = URLDecoder.decode(value, StandardCharsets.UTF_8.name());
208                     } catch (UnsupportedEncodingException e) {
209                         // won't happen; UTF-8 is always supported
210                     }
211                 }

267         if (request instanceof HttpEntityEnclosingRequestBase) { // POST/PUT
268             try {
269                 ((HttpEntityEnclosingRequestBase)request).setEntity(toJsonEntity(document));

272                 errorViews.write(sde, document);

280     private HttpResponse executeRequest(HttpUriRequest request) {
281         log.debug("request: {}", request);
282         request.addHeader(HttpHeaders.ACCEPT, JSON_MIME_TYPE);
283         request.addHeader(HttpHeaders.AUTHORIZATION, account.getAuthorizationHeader(securityToken));
284         try {
285             return httpRequester.execute(request);
286         } catch (IOException e) {
287             request.abort();
288             throw new ExecutionException(e, IO_ERROR_REQUEST_EXEC)
289                     .withReason(e.getMessage())
290                     .withResolution(IO_ERROR_RES);
291         }
292     }

294     private List<?> processResponse(HttpResponse response, HttpUriRequest request, Document document) {
295         List<?> contentList = null;
296         StatusLine statusLine = response.getStatusLine();
297         int statusCode = statusLine.getStatusCode();
298         if (statusCode == SC_OK || statusCode == SC_CREATED) {
299             InputStream inputStream = fetchInputStream(response.getEntity());
300             try {
301                 contentList = getContentList(inputStream);
302             } finally {
303                 IOUtils.closeQuietly(inputStream);
304             }
305         } else {
306             contentList = getContentListForStatusCode(statusCode);
307             if (contentList == null) {
308                 String errorMessage = String.format(ERROR_REQUEST_FAILED, request.getMethod());
309                 String reason = modifyReason(statusLine.getReasonPhrase());
310                 Map<String, Object> errorDoc = Maps.newLinkedHashMap();
311                 errorDoc.put(ERROR, errorMessage);
312                 errorDoc.put(STATUS_CODE, statusCode);
313                 errorDoc.put(REASON, reason);
314                 errorDoc.put(URL, request.getURI().toString());
315                 if (statusCode == SC_BAD_REQUEST || statusCode == SC_CONFLICT) {
316                     // This typically means there was some sort of validation error with parameters,
317                     // like count > 1000 or startDate > endDate.
318                     // Eloqua returns a JSON list of one or more objects describing the errors.
319                     // Try to return them in the response if available.
320                     try {
321                         errorDoc.put(DETAILS, fetchListFromInputStream(fetchInputStream(response.getEntity())));
322                     } catch (Throwable t) {
323                         // oh well, we tried
324                         t.printStackTrace();
325                     }
326                 }
327                 Document newDoc = documentUtility.newDocumentFor(document, errorDoc);
328                 SnapDataException snapDataException = new SnapDataException(newDoc, errorMessage)
329                         .withReason(reason)
330                         .withResolution(RES_REQUEST_FAILED);
331                 errorViews.write(snapDataException, document);
332             }
333         }
334         return contentList;
335     }

349     // see http://docs.oracle.com/cloud/latest/marketingcs_gs/OMCAB/index.html#Developers/GettingStarted/Tutorials/determining-base-URLs.htm
350     private void initBaseURLsFromLoginEndpoint() {
351         URI endpoint = EloquaUtil.toURI(account.getLoginEndpoint());
352         log.debug(CONNECTING_TO_THE_ENDPOINT, endpoint);
353         HttpResponse response = executeGet(endpoint);
354         HttpEntity entity = response.getEntity();
355         InputStream is;
356         if (entity != null) {
357             is = fetchInputStream(entity);
358         } else {
359             throw new ExecutionException(UNABLE_TO_FETCH_BASE_URI).formatWith(endpoint)
360                     .withReason(RESPONSE_DOES_NOT_CONTAIN_DATA)
361                     .withResolutionAsDefect();
362         }
363 
364         //Getting the response as a byte array...
365         byte[] responseBytes;
366         try {
367             responseBytes = IOUtils.toByteArray(is);
368         } catch (IOException e) {
369             throw new ExecutionException(e, INPUT_STREAM_TO_BYTE_ARRAY_ERROR)
370                     .withReason(IO_ERROR)
371                     .withResolutionAsDefect();
372         } finally {
373             IOUtils.closeQuietly(is);
374         }
375 
376         // Checking if the response is a failure...
377         if (Arrays.equals(responseBytes, NOT_AUTHENTICATED.getBytes())) {
378             throw new ExecutionException(NOT_AUTHENTICATED_ERROR)
379                     .withReason(INVALID_CREDENTIALS)
380                     .withResolution(PLEASE_VERIFY_CREDENTIALS);
381         }
382 
383         int statusCode = response.getStatusLine().getStatusCode();
384         if (statusCode > 300) {
385             String message = fetchMessageFromBytes(responseBytes);
386             if (StringUtils.isEmpty(message)) {
387                 message = response.getStatusLine().getReasonPhrase();
388             }
389             throw new ExecutionException(REQUEST_FAILED).formatWith(statusCode, message)
390                     .withReason(String.format(REQUEST_FAILED, statusCode, message))
391                     .withResolution(REQUEST_FAILED_RESOLUTION);
392         }
393 
394         Map<String, Object> map = fetchMapFromBytes(responseBytes);
395 
396         baseRestURL = getBaseURL(map, REST_URL_PATH);
397         baseBulkURL = getBaseURL(map, BULK_URL_PATH);
398     }

400     private String getBaseURL(Map<String, Object> map, String path) {
401         String url = JsonPath.read(map, path);
402         if (!url.endsWith(BASE_URL_SUFFIX)) {
403             throw new ExecutionException("URL returned by Eloqua login endpoint doesn't end with %s as expected: %s")
404                     .formatWith(BASE_URL_SUFFIX, url)
405                     .withResolution(RESOLUTION_CUSTOMER_SUPPORT);
406         }
407         // strip the suffix
408         return url.substring(0, url.length() - BASE_URL_SUFFIX.length());
409     }
