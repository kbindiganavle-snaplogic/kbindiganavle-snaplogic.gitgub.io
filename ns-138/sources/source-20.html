398     // This method is called by the main thread in the process() method, and by reauthenticate().
399     // It initiates a handshake with the Salesforce Streaming API server, the result of which is
400     // notified asynchronously by calling handshakeCallback. On a successful handshake,
401     // handshakeCallback will call subscribe(). The Login is always fresh when calling this
402     // method, so in theory this handshake should always succeed. However, the same handshakeCallback
403     // may be called again later: see scenarios 2 and 3 below.
404     private void connect() {
405         LOG.debug("connect: handshake");
406         statusReporter.notify(HANDSHAKING_MESSAGE, now());
407         bayeuxClient.handshake(this::handshakeCallback); // handshakeCallback will call subscribe()
408         // handshake and subscribe are asynchronous, so this method returns before they are called
409     }

411     // This is called asynchronously by Cometd on a handshake event, basically one of 4 scenarios:
412     // 1. Momentarily after BayeuxClient.handshake() is called by connect().
413     //    Since connect() is only called right after obtaining fresh valid Login, the Message passed
414     //    to these calls will normally have successful=true, so subscribe() will be called.
415     // 2. Occasional event where an error occurs at the protocol layer such as "403::Unknown client"
416     //    that Cometd is able to successfully handshake again without requiring reauthentication.
417     //    In this case, it calls this callback with a successful=true Message to indicate a
418     //    successful handshake so subscribe() will be called again.
419     // 3. Occasional event where an error occurs such as "403::Handshake denied" that Cometd can't
420     //    handle itself because the request requires reauthentication. In this case, it calls this
421     //    callback with a successful=false Message. In response, we reauthenticate by obtaining
422     //    fresh Login and calling connect() again (which will trigger another event of type 1).
423     // 4. Some error where this callback is called with a Message that has successful=false but
424     //    the error/failureReason is not HANDSHAKE_DENIED/REAUTH_REQUIRED, like an IOException
425     //    due to the inability to authenticate with a proxy. In this case, we can't recover,
426     //    so we stop.
427     // Note that this callback does not get called on a *temporary* connection failure due to a
428     // network disruption. For those, connectFailureListener() will be called and log a warning,
429     // but the  transport will keep trying to reconnect transparently.
430     private void handshakeCallback(Message message) {
431         if (message.isSuccessful()) {
432             LOG.debug("handshakeCallback: message={}", message);
433             // On any successful handshake, we need to subscribe to the channel of interest.
434             subscribe();
435         } else {
436             LOG.debug("handshakeCallback: message={}", message);
437             String error = getError(message);
438             String failureReason = getFailureReason(message);
439             if (HANDSHAKE_DENIED.equals(error) && REAUTH_REQUIRED.equals(failureReason)) {
440                 // On this failure reason, we must reauthenticate by obtaining fresh
441                 // Login and then re-handshake and re-subscribe by calling connect().
442                 // Unfortunately, we can't do that directly inside this callback since the
443                 // BayeuxClient is still in a connected state so a call to handshake() will fail;
444                 // it's state won't become disconnected until this callback returns.
445                 // To deal with this, we schedule a task to perform the reauthenticate
446                 // in a separate thread a few seconds after this callback returns.
447                 LOG.debug("handshakeCallback: message={}, scheduling reauthenticate()", message);
448                 statusReporter.notify(HANDSHAKE_FAILED_MESSAGE, now(), getErrorReason(message));
449                 scheduler.schedule(this::reauthenticate, 3, TimeUnit.SECONDS);
450             } else {
451                 // On any other handshake failure, such as a proxy connection error, stop.
452                 stop(new SnapDataException(getException(message), UNABLE_TO_CONNECT)
453                         .withReason(getErrorReason(message))
454                         .withResolution(RESOLUTION_ADDRESS_ISSUE));
455             }
456         }
457     }

459     // This is called by handshakeCallback on a successful handshake, and by subscribeCallback
460     // on a failed subscription attempt due to a stale replayId; replayExtension handles the
461     // communication of the replayId itself.
462     private void subscribe() {
463         LOG.debug("subscribe: subscribing to topic={} with replayId={}", topic, replayId);
464         statusReporter.notify(SUBSCRIBING_MESSAGE, topic, replayId, now());
465         ClientSessionChannel channel = bayeuxClient.getChannel(topic);
466         if (!channel.subscribe(this::onTopicMessage, this::subscribeCallback)) {

470                     .withReason("channel.subscribe() returned false")
471                     .withResolution(ERR_CUSTOMER_SUPPORT));
472         }
473     }

475     // This is called asynchronously by Cometd momentarily after subscribe() to report whether
476     // the subscription attempt succeeded or failed.
477     private void subscribeCallback(Message message) {
478         if (message.isSuccessful()) {
479             LOG.debug("subscribeCallback: message={}", message);
480             statusReporter.notify(SUBSCRIBED_MESSAGE, topic, replayId, now());
481         } else {
482             LOG.warn("subscribeCallback: message={}", message);
483             String errorReason = getErrorReason(message);
484             statusReporter.notify(SUBSCRIBE_FAILED_MESSAGE, topic, replayId, now(), errorReason);
485             // Match errors like "400::The replayId {2717838} you provided was invalid. Please
486             // provide a valid ID, -2 to replay all events, or -1 to replay only new events."
487             // The exact message content might change in the future, so look only for the 400
488             // and the replayId value.
489             long lastReplayId = replayId.get();
490             String error = getError(message);
491             if (error != null && error.startsWith("400::") && lastReplayId > 0 &&
492                     error.contains(String.valueOf(lastReplayId))) {
493                 LOG.debug("subscribeCallback: retrying subscription with replayId=-2");
494                 INVALID_REPLAY_ID_LINT.report(() -> new Object[]{lastReplayId});
495                 replayId.set(REPLAY_FROM_EARLIEST);
496                 subscribe();
497             } else if (DAILY_LIMIT_EXCEEDED.equals(error)) {
498                 stopOnDailyLimitExceededError();
499             } else {
500                 // We were unable to subscribe for some other reason we're not able to
501                 // recover from, like a "400::The channel specified is not valid".
502                 stop(new SnapDataException(getException(message), UNABLE_TO_SUBSCRIBE).formatWith(topic)
503                         .withReason(errorReason)
504                         .withResolution(ERR_UNKNOWN_TOPIC_RESOLUTION));
505             }
506         }
507     }

518                 statusReporter.notify(REAUTHENTICATING_MESSAGE, now());

520                 if (loginException != null) {

532     // This listener is called when a new message for the subscribed topic is received.
533     private void onTopicMessage(ClientSessionChannel channel, Message message) {
534         // Messages received by this listener will not have a "successful" field,
535         // so message.isSuccessful() will always return false and is of no use here.
536         try {
537             // message payload may contain sensitive data so don't log unless trace is enabled
538             LOG.trace("onTopicMessage: message={}", message);
539             if (isStopped()) {
540                 return; // we've already stopped; don't output any more documents
541             }
542             // convert Message to an output Document
543             Map<String, Object> flatData = flattenData(message.getDataAsMap());
544             Object outDocData = convertData(flatData);
545             Document outDoc = documentUtility.newDocumentFor(inDoc, outDocData);
546             outputViews.write(outDoc, passThrough ? inDoc : null);
547             long count = currentMsgCount.incrementAndGet();
548             Long msgReplayId = getReplayId(message);
549             if (msgReplayId == null) {
550                 LOG.warn("onTopicMessage: message has no replayId! {}", message);
551             } else {
552                 replayId.set(msgReplayId);
553             }
554             statusReporter.notify(EVENT_MESSAGE, message.getChannel(), msgReplayId, now());
555             LOG.debug("onTopicMessage: wrote output doc {} with replayId={}", count, msgReplayId);
556             if (maxMsgCount != INFINITE_MSG_COUNT && count >= maxMsgCount) {
557                 LOG.debug("onTopicMessage: stopping because count >= maxMsgCount {}",
558                         maxMsgCount);
559                 stop(null);
560             } else if (maxSuggestValue != null && count >= maxSuggestValue) {
561                 LOG.debug("onTopicMessage: stopping because count >= maxSuggestValue {}",
562                         maxSuggestValue);
563                 stop(null);
564             }
565         } catch (Throwable t) {
566                 LOG.error("onTopicMessage: unexpected Throwable", t);
567         }
568     }
