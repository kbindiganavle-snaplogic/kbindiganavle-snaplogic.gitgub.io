317     protected void process(final Document document, final String inputViewName) {
318         initServiceVersion(serviceVersionExpr, document, options);
319         getServiceEndpoint();
320         edgemartAlias = edgemartAliasExpression.eval(document);
321         if (StringUtils.isBlank(edgemartAlias)) {
322             throwException(null, ERR_EMPTY_PROPERTY, EDGEMART_ALIAS_LABEL, REASON_EMPTY_PROPERTY,
323                     RESOLUTION_ADDRESS_ISSUE);
324         } else if (StringUtils.contains(edgemartAlias, '-')) {
325             throwException(null, ERR_DATASET_ALIAS, edgemartAlias, REASON_DASH_NOT_ALLOWED,
326                     RESOLUTION_ADDRESS_ISSUE);
327         }
328         if (StringUtils.contains(edgemartAlias, DASH)) {
329             throw new ConfigurationException(ERR_DATASET_ALIAS).formatWith(edgemartAlias)
330                     .withReason(REASON_DASH_NOT_ALLOWED)
331                     .withResolution(RESOLUTION_ADDRESS_ISSUE);
332         }
333         edgemartContainer = edgemartContainerExpression.eval(document);
334         switch (action) {
335             case UPLOAD_AND_PROCESS:
336             case UPLOAD:
337             case PREVIEW:
338                 if (isSampling) {
339                     documentSamples.add(document);
340                     if (documentSamples.size() == MAX_SAMPLE_SIZE) {
341                         fields = EdgeMartSchema.generateSchema(documentSamples,
342                                 (String) metadataObject.get(FULLY_QUALIFIED_NAME));
343                         isSampling = false;
344                         mergeCsvFields();
345                         for (Document sample : documentSamples) {
346                             formatCsvLine(sample);
347                         }
348                         loginInfo = getLoginInfo(false); // make sure access token is not expired
349                         if (loginException != null) {
350                             for (Document d : documentSamples) {
351                                 errorViews.write(loginException, d);
352                             }
353                             return;
354                         }
355                         insightsExternalDataID = createInsightsExternalData(edgemartAlias);
356                     }
357                 } else {
358                     formatCsvLine(document);
359                 }
360                 break;
361             case PROCESS:
362                 if (idExpression != null) {
363                     insightsExternalDataID = idExpression.eval(document);
364                     if (StringUtils.isNotBlank(insightsExternalDataID)) {
365                         loginInfo = getLoginInfo(false); // make sure access token is not expired
366                         if (loginException != null) {
367                             errorViews.write(loginException, document);
368                             return;
369                         }
370                         processUploadedData(document);
371                     } else {
372                         writeSnapDataException(ERR_EMPTY_ID, edgemartAlias, REASON_EMPTY_ID,
373                                 RESOLUTION_EMPTY_ID, document);
374                     }
375                 }
376                 break;
377             default:
378                 throwException(null, ERR_UNSUPPORTED_ACTION, action.toString(),
379                         REASON_UNSUPPORTED_ACTION, RESOLUTION_SNAP_PROPERTY);
380         }

383     @SuppressWarnings("unchecked")
384     private void mergeCsvFields() {
385         int index = 0;
386         Object fieldsObject = metadataObject.get(FIELDS);
387         if (fieldsObject instanceof List && !((List<Object>) fieldsObject).isEmpty()) {
388             List<Object> fieldsOriginal = (List<Object>) fieldsObject;
389             // The value of "fields" should be a list of Map data.
390             Object fieldObject = fieldsOriginal.get(0);
391             if (fieldObject instanceof Map) {
392                 Object object = ((Map<String, Object>) fieldObject).get(NAME);
393                 // The value of the "name" field should be non-empty string.
394                 if (object instanceof String && StringUtils.isNotBlank((String) object)) {
395                     // The value of "fields" from the Edit metadata property is valid.
396                     Map<String, Map<String, Object>> fieldsMap =
397                             Maps.newHashMapWithExpectedSize(fields.size());
398                     for (Map<String, Object> field : fields) {
399                         fieldsMap.put((String) field.get(NAME), field);
400                     }
401                     header = new String[fieldsOriginal.size()];
402                     for (Object value : fieldsOriginal) {
403                         Map<String, Object> fieldOriginal = (Map<String, Object>) value;
404                         String name = (String) fieldOriginal.get(NAME);
405                         header[index++] = name;
406                         Map<String, Object> field = fieldsMap.get(name);
407                         for (Map.Entry<String, Object> entry : fieldOriginal.entrySet()) {
408                             if (entry.getValue() == null) {
409                                 String key = entry.getKey();
410                                 entry.setValue(field.get(key));
411                             }
412                         }
413                     }
414                     return;
415                 }
416             }
417         }
418         // If the value of "fields from the Edit metadata property is empty or mal-formed, use
419         // automatically-generated CSV metadata.
420         header = new String[fields.size()];
421         for (Map<String, Object> field : fields) {
422             header[index++] = (String) field.get(NAME);
423         }
424         metadataObject.put(FIELDS, fields);
425     }

501         sObject.setField(METADATA_JSON, getMetadataAsBytes());
502         RestWriteResult restWriteResult = null;
503         String id = StringUtils.EMPTY;
504         if (action != Action.PREVIEW) {
505             try {
506                 syncWrite.initialize(options, loginInfo);
507                 restWriteResult = syncWrite.execute(sObject);

669     @SuppressWarnings("unchecked")
670     private void pollProcessStatus(Document document) {
671         String query = String.format(FORMAT_SOQL, insightsExternalDataID);
672         options.withOperation(Options.Operation.QUERY)
673                 .withSnapName(Options.SnapName.QUERY)
674                 .withQuery(query)
675                 .withServiceUrl(Query.getServiceEndpoint(versionedBaseServiceUrl, query, false));
676         SyncRead syncRead = injector.getInstance(SyncRead.class);
677         loginInfo = getLoginInfo(false); // make sure access token is not expired
678         if (loginException != null) {
679             errorViews.write(loginException, document);
680             return;
681         }
682         syncRead.initialize(options, loginInfo);
683         Object object;
684         int waitSec = 0;
685         Object status = null;
686         do {
687             if (waitSec >= pollingTimeoutSec) {
688                 writeSnapDataException(ERR_TIMEOUT, edgemartAlias, REASON_LOAD_EXTERNAL_DATA,
689                         RESOLUTION_POLL_TIMEOUT, document);
690                 return;
691             }
692             // delay
693             try {
694                 TimeUnit.MILLISECONDS.sleep(pollingIntervalSec * 1000);
695                 waitSec += pollingIntervalSec;
696             } catch (InterruptedException ex) {
697                 // NO-OP
698             }
699             object = checkStatus(syncRead, document);
700             if (object instanceof Map) {
701                 status = ((Map<String, Object>) object).get(STATUS);
702             }
703         } while (!COMPLETED.equals(status) && !FAILED.equals(status));
704         if (COMPLETED.equals(status)) {
705             insightsExternalDataMap.put(STATUS, COMPLETED);
706             outputViews.write(documentUtility.newDocumentFor(document, outputMap));
707         } else if (FAILED.equals(status)) {
708             insightsExternalDataMap.put(STATUS, FAILED);
709             writeSnapDataException(ERR_LOAD_EXTERNAL_DATA, edgemartAlias,
710                     REASON_LOAD_EXTERNAL_DATA, RESOLUTION_SNAP_PROPERTIES, document);
711         }
712     }
