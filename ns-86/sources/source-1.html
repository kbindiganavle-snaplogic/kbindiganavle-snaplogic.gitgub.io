403     public void configureGraphQLClient(PropertyValues propertyValues)
404             throws ConfigurationException {
405         baseUrl = getString(propertyValues.getAsExpression(BASE_URL));
406         authHeaders = new ArrayList<>();
407         if (!(account instanceof DefaultAccount)) {
408             AuthData authData;
409             try {
410                 authData = account.connect().generateAuthInfo();
411             } catch (HttpCoreDataException e) {
412                 throw new ConfigurationException(e, ERR_AUTH_INFO_GENERATE_FAILED)
413                         .withReason(ERR_AUTH_INFO_GENERATE_FAILED_REASON)
414                         .withResolution(ERR_AUTH_INFO_GENERATE_FAILED_RESOLUTION);
415             }
416             if (account instanceof HeaderAuthAccount) {
417                 if (authData.getAuthHeader() != null) {
418                     authHeaders.addAll(authData.getAuthHeader());
419                 }
420             } else if (account instanceof HttpOAuth2Account) {
421                 if (((HttpOAuth2Account) account).isHeaderAuthenticated()) {
422                     // Access token sent as Bearer token in Authorization header
423                     if (authData.getAuthHeader() != null) {
424                         authHeaders.addAll(authData.getAuthHeader());
425                     }
426                 } else {
427                     // Access token sent in URL params
428                     if (authData.getAuthUrlParam() != null &&
429                             !authData.getAuthUrlParam().isEmpty()) {
430                         Pair<String, String> authParamPair = authData.getAuthUrlParam().get(0);
431                         String accessTokenParamString = String.format("&%s=%s",
432                                 authParamPair.getKey(), authParamPair.getValue());
433                         baseUrl = baseUrl.contains("?") ? baseUrl.concat(accessTokenParamString) :
434                                 baseUrl.concat("?").concat(accessTokenParamString);
435                     }
436                 }
437             } else if (account instanceof BasicAuthAccount) {
438                 String usernamePassword = authData.getCredentials().getUserPrincipal().getName() +
439                         ":" + authData.getCredentials().getPassword();
440                 String basicAuthEncodedCredents = "Basic " +
441                         Base64.getEncoder().encodeToString(usernamePassword.getBytes());
442                 authHeaders.add(Pair.of("Authorization", basicAuthEncodedCredents));
443             }
444         }
445         trustAllCerts = propertyValues.getBoolean(
446                 compositePropertyName(CLIENT_PROPERTIES, TRUST_ALL_CERT_PROP), false);
447         followRedirects = propertyValues.getBoolean(
448                 compositePropertyName(CLIENT_PROPERTIES, FOLLOW_REDIRECT), true);
449         int maxRedirects = evalInt(propertyValues.getAsExpression(
450                 compositePropertyName(CLIENT_PROPERTIES, MAX_REDIRECT)),
451                 null, DEFAULT_MAX_REDIRECT);
452         readTimeout = propertyValues.getInt(
453                 compositePropertyName(CLIENT_PROPERTIES, READ_TIMEOUT));
454         connTimeout = propertyValues.getInt(
455                 compositePropertyName(CLIENT_PROPERTIES, CONN_TIMEOUT));
456 
457         graphqlOptions = GraphQLOptions.builder()
458                 .trustAllCerts(trustAllCerts)
459                 .followRedirects(followRedirects)
460                 .maxRedirects(maxRedirects)
461                 .readTimeoutInSeconds(readTimeout.intValue())
462                 .connectionTimeoutInSeconds(connTimeout.intValue())
463                 .build();
464 
465         initClient();
466     }

478     private List<Pair<ExpressionProperty, ExpressionProperty>> readTableProperty(
479             PropertyValues propertyValues, String tableProp, String keyProp, String valProp) {
480         List<Map<String, Object>> graphQLVarsList = propertyValues.get(tableProp);
481         List<Pair<ExpressionProperty, ExpressionProperty>> graphQLVarExprList = new ArrayList<>();
482         String VALUE = "value";
483         if (CollectionUtils.isNotEmpty(graphQLVarsList)) {
484             for (Map<String, Object> graphQLVar : graphQLVarsList) {
485                 if (((Map<String, Object>)graphQLVar.get(keyProp)).get(VALUE) == null ||
486                         ((Map<String, Object>)(graphQLVar.get(valProp))).get(VALUE) == null) {
487                     continue;
488                 }
489                 ExpressionProperty graphQLVarNameExpr =
490                         propertyValues.getExpressionPropertyFor(graphQLVar, keyProp);
491                 ExpressionProperty graphQLVarValueExpr =
492                         propertyValues.getExpressionPropertyFor(graphQLVar, valProp);
493                 graphQLVarExprList.add(Pair.of(graphQLVarNameExpr, graphQLVarValueExpr));
494             }
495         }
496         return graphQLVarExprList;
497     }

521 
522             if (paginationEnabled && maxPageCount != 1) {
523                 if (paginationType.equals(PAGINATION_TYPE_CURSOR)) {
524                     int pagesRetrieved = 0;
525                     boolean hasAdditionalPage = false;
526                     Object nextCursor = "";
527                     do {
528                         initClient();

530                                 client.execute(query, graphQLVariables) : client.execute(query);
531                         client.shutdown();
532                         if (response.hasError()){

534                                     ((SnapDataException) new SnapDataException(

537                                             .withResolution(
538                                                     ERR_GQL_SERVER_ERROR_RESPONSE_RESOLUTION));
539                             // Any error will stop page requests, as subsequent request depends on
540                             // page availability/next cursor data from a successful request
541                             break;
542                         } else {
543                             ObjectMapper mapper = new ObjectMapper()
544                                     .registerModule(new JsonFactoryModule())

546                                     .registerModule(new JavaTimeModule());

548                             writeOutput(outputViews,

566                         }

568                         // is available (no next/previous page); otherwise, stop when pages
569                         // retrieved reaches either the specified max page count or end-of-data.
570                     } while (hasAdditionalPage &&

576                 }

580                         client.execute(query, graphQLVariables) : client.execute(query);
581                 client.shutdown();
582                 if (response.hasError()) {
583                     errorViews.write
584                             ((SnapDataException) new SnapDataException(
585                                     response.getErrors().get(0))
586                                     .withReason(ERR_GQL_SERVER_ERROR_RESPONSE_REASON)
587                                     .withResolution(
588                                             ERR_GQL_SERVER_ERROR_RESPONSE_RESOLUTION));
589                 } else {

591                             .registerModule(new JsonFactoryModule())
592                             .registerModule(new JodaModule())
593                             .registerModule(new JavaTimeModule());

616     private Object decipherDataType(Object obj){
617         if (obj instanceof BigInteger) {
618             return Integer.valueOf(((BigInteger)obj).intValue());
619         } else if (obj instanceof BigDecimal) {
620             return Double.valueOf(((BigDecimal)obj).doubleValue());
621         }
622         if (!(obj instanceof String)) {
623             return obj;
624         }
625         try {
626             return Integer.parseInt((String)obj);
627         } catch (Exception e) {
628             try {
629                 return Double.parseDouble((String) obj);
630             } catch (Exception ex) {
631                 String value = (String) obj;
632                 if (value.equalsIgnoreCase(Boolean.TRUE.toString()) ||
633                         value.equalsIgnoreCase(Boolean.FALSE.toString())) {
634                     return Boolean.parseBoolean(value);
635                 }
636                 if (value.startsWith("\"") && value.endsWith("\"")) {
637                     return value.substring(1, value.length() - 1);
638                 }
639                 return value;
640             }
641         }
642     }

727     private int evalInt(ExpressionProperty expression, Document document, int defaultVal) {
728         Object val = expression.eval(document);
729         try {
730             if (val instanceof Number) {
731                 return ((Number) val).intValue();
732             } else if (val instanceof String) {
733                 String str = ((String) val).trim();
734                 if (str.isEmpty()) {
735                     return defaultVal;
736                 } else {

748     void validateCursorVariableName(String targetQuery, String variableName) {
749         if (variableName == null || variableName.isBlank()) {
750             throw new ConfigurationException(ERR_CURSOR_VARIABLE_NAME_MISSING)
751                     .withReason(ERR_CURSOR_VARIABLE_NAME_MISSING_REASON)
752                     .withResolution(ERR_CURSOR_VARIABLE_NAME_MISSING_RESOLUTION);
753         }
754         List<String> parsedQueryVariables = new ArrayList<>();
755         Pattern queryVarPattern = Pattern.compile(QUERY_VARIABLE_PATTERN);
756         Matcher queryVarMatcher = queryVarPattern.matcher(targetQuery);
757         while (queryVarMatcher.find()) {
758             // As of Java 9 we could actually check for \s* in the lookbehind regex, but
759             // it sounds like that support is still inefficient - so instead, let leading white
760             // space be included in the match and just strip it
761             String match = queryVarMatcher.group().strip();
762             parsedQueryVariables.add(match);
763         }
764 
765         if (!parsedQueryVariables.contains(variableName)) {
766             throw new ConfigurationException(ERR_CURSOR_VARIABLE_MATCH_NOT_FOUND)
767                     .withReason(ERR_CURSOR_VARIABLE_MATCH_NOT_FOUND_REASON)
768                     .withResolution(ERR_CURSOR_VARIABLE_MATCH_NOT_FOUND_RESOLUTION);
769         }
