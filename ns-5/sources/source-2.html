37 
38     private static final JsonPath PIPELINE_PARAMS_PATH = JsonPaths.compileStatic(
39             "$property_map.settings.param_table.value[*].key.value");
40 
41     @Inject
42     private SnapiService snapiService;
43 
44     @Property
45     private TypedExpressionProperty<String> pipeline;
46 
47     @Property
48     private final List<ChildParam> params = new ArrayList<>();
49 
50     @Override
51     public List<String> execute() {
52         try {
53             String pipelinePath = pipeline.eval(null);
54 
55             PipelineFacade pipelineFacade = snapiService.createPipelineFacade();
56             LinkedPipeline pipeInfo = pipelineFacade.linkAndLoad(pipelinePath);
57             List<String> pipelineParams = PIPELINE_PARAMS_PATH.readStatic(
58                     pipeInfo.getDefinition(), Collections.emptyList());
59 
60             pipelineParams.removeAll(params.stream()
61                     .map(childParam -> childParam.paramName.orElse(StringUtils.EMPTY))
62                     .collect(Collectors.toList()));
63 
64             return pipelineParams;
65         } catch (SnapDataException e) {
66             throw new SnapDataException(e, "Pipeline property evaluation failed: %s")
67                     .formatWith(e.getMessage())
68                     .withReason(e.getReason())
69                     .withResolution(e.getResolution());
70         } catch (SnapiException e) {
71             throw new SnapDataException(e, "Unable to retrieve pipeline: %s")
72                     .formatWith(e.getMessage())
73                     .withResolution("Make sure the pipeline property refers to a pipeline " +
74                             "that exists");
75         }
76     }
77 }
