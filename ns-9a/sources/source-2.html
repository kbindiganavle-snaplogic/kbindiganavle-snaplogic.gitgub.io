101             Set<String> suggestions = new TreeSet<>();
102             switch (entityType) {
103                 case ENTITY_TYPE_USER:

105                             isRequiredAttributesMarked);

107                     break;
108                 case ENTITY_TYPE_GROUP:
109                     addRequiredAttributes(suggestions, DEFAULT_GROUP_ATTRIBUTES_REQUIRED,

112                     break;
113                 default:
114                     suggestions = EMPTY_SET;
115                     break;
116             }
117             suggestionBuilder.node(PROP_ATTRIBUTES)
118                     .over(PROP_ATTRIBUTE_NAME)
119                     .suggestions(suggestions.toArray(new String[0]));

286      * Evaluates attribute value expressions using given input document
287      *
288      * @param inputDoc
289      * @return true if no data error, false otherwise
290      */
291     protected boolean evaluateAttributes(Document inputDoc) {
292         Object attributeValue;
293         String attributeName;
294         evaluatedAttributes = new TreeMap<>();
295         Set<Map.Entry<ExpressionProperty, ExpressionProperty>> entries = attributes.entrySet();
296         for (Map.Entry<ExpressionProperty, ExpressionProperty> entry : entries) {
297             attributeName = entry.getKey().eval(inputDoc);
298             if (StringUtils.isBlank(attributeName)) {
299                 SnapDataException sde = new SnapDataException(ERR_ATTRIBUTE_NAME_BLANK_MSG)
300                         .withReason(ERR_ATTRIBUTE_NAME_BLANK_REASON)
301                         .withResolution(ERR_ATTRIBUTE_NAME_OR_VALUE_BLANK_RESOLUTION);
302                 errorViews.write(sde, inputDoc);
303                 return false;
304             }
305             attributeValue = entry.getValue().eval(inputDoc);
306             if (attributeValue == null || StringUtils.isBlank(attributeValue.toString())) {
307                 SnapDataException dataException =
308                         new SnapDataException(ERR_ATTRIBUTE_VALUE_BLANK_MSG)
309                                 .withReason(String.format(ERR_ATTRIBUTE_VALUE_BLANK_REASON,
310                                         entry.getKey()))
311                                 .withResolution(ERR_ATTRIBUTE_NAME_OR_VALUE_BLANK_RESOLUTION);
312                 errorViews.write(dataException, inputDoc);
313                 return false;
314             }
315             if (evaluatedAttributes.get(attributeName.replace(ASTERISK, StringUtils.EMPTY))
316                     == null) {
317                 evaluatedAttributes.put(attributeName.replace(ASTERISK, StringUtils.EMPTY),
318                         attributeValue.toString());
319             } else {
320                 SnapDataException dataException = new SnapDataException(ERR_ATTRIBUTE_REPEATED_MSG)
321                         .withReason(String.format(ERR_ATTRIBUTE_REPEATED_REASON,
322                                 attributeName))
323                         .withResolution(ERR_ATTRIBUTE_REPEATED_RESOLUTION);
324                 errorViews.write(dataException, inputDoc);
325                 return false;
326             }
327             addCustomAttributes();
328         }
329         if (!isRequiredAttributesCheckNotNeeded) {
330             Set<String> requiredAttributes = getRequiredAttributes(entityType);
331             for (String requiredAttributeName : requiredAttributes) {
332                 if (!evaluatedAttributes.keySet().contains(requiredAttributeName)) {
333                     requiredAttributeName = requiredAttributeName.replace(ASTERISK,
334                             StringUtils.EMPTY);
335                     if (!evaluatedAttributes.keySet().contains(requiredAttributeName)) {
336                         SnapDataException sde =
337                                 new SnapDataException(ERR_REQUIRED_ATTRIBUTE_MISSING_MSG)
338                                         .withReason(String.format(
339                                                 ERR_REQUIRED_ATTRIBUTE_MISSING_REASON,
340                                                 requiredAttributeName))
341                                         .withResolution(String.format(
342                                                 ERR_REQUIRED_ATTRIBUTE_MISSING_RESOLUTION,
343                                                 requiredAttributeName));
344                         errorViews.write(sde, inputDoc);
345                         return false;
346                     }
347                 }
348             }
349         }
350         return true;
351     }

385      * Converts given set of attributes into a JSON object
386      *
387      * @param attributes
388      * @return JSON object representing the given attributes
389      */
390     protected JSONObject createJsonFromAttributes(Map<String, String> attributes) {
391         boolean forcePasswordChange = false;
392         JSONObject entity = new JSONObject();
393         Set<Map.Entry<String, String>> entries = attributes.entrySet();
394         for (Map.Entry<String, String> entry : entries) {
395             if (entry.getKey().equalsIgnoreCase(PASSWORD)) {
396                 JSONObject passwordProfileObject = new JSONObject();
397                 passwordProfileObject.put(PASSWORD, entry.getValue());
398                 passwordProfileObject.put(FORCE_CHANGE_PASSWORD_NEXT_SIGNIN, forcePasswordChange);
399                 entity.put(PASSWORD_PROFILE, passwordProfileObject);
400             } else {
401                 if (entry.getKey().equalsIgnoreCase(FORCE_CHANGE_PASSWORD_NEXT_SIGNIN)) {
402                     forcePasswordChange = Boolean.parseBoolean(entry.getValue());
403                 } else {
404                     entity.put(entry.getKey(), entry.getValue());
405                 }
406             }
407         }
408         return  entity;
409     }
