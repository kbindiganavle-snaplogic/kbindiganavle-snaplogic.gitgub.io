554      * Lists all the subfolders (if exist) along with their IDs for the given folder
555      *
556      * @param connection
557      * @param folderID
558      * @param builder
559      * @param propertyKey
560      *
561      * @throws BoxAPIException The ApiException will be thrown only when the account is
562      *                         unauthorized
563      */
564     public void suggestSubFolders(BoxAPIConnection connection, String folderID,
565             SuggestionBuilder builder, String propertyKey) {
566         if (StringUtils.isBlank(folderID)) {
567             builder.node(propertyKey).suggestions(new String[]{DEFAULT_FOLDER_ROOT});
568             return;
569         }
570         List<String> subfolders = new ArrayList<>();
571         String folderPath;
572         folderID = extractID(folderID);
573         try {
574             BoxFolder currentFolder = new BoxFolder(connection, folderID);
575             for (BoxItem.Info info : currentFolder) {
576                 if (info instanceof BoxFolder.Info) {
577                     folderPath = getFolderPath(new BoxFolder(connection, info.getID())
578                             .getInfo());
579                     if (StringUtils.isBlank(folderPath)) {
580                         folderPath = info.getName() + '/';
581                     }
582                     subfolders.add(String.format(FOLDER_SUGGESTION_FORMAT,
583                             ROOT_FOLDER, folderPath, info.getID()));
584                 }
585             }
586             if (subfolders.isEmpty()) {
587                 subfolders.add(FOLDER_SUGGESTIONS_NONE);
588             }
589             if (!folderID.equalsIgnoreCase(FOLDER_ID_ZERO)) {
590                 subfolders.add(0, String.format(PARENT_FOLDER_SUGGESTION,
591                         currentFolder.getInfo().getParent().getID()));
592             }
593             builder.node(propertyKey).suggestions(subfolders.toArray(
594                     new String[subfolders.size()]));
595         } catch (BoxAPIException e1) {
596             switch (e1.getResponseCode()) {
597                 case HttpURLConnection.HTTP_UNAUTHORIZED:
598                     throw e1;
599                 default:
600                     throw new ExecutionException(String.format(ERR_BROWSER_FOLDER_SUGGEST_MSG,
601                             e1.getMessage()));
602             }
603         }
604     }

617     public void suggestSourceId(BoxAPIConnection connection, String sourceId,
618             SuggestionBuilder builder, String propertyKey) {
619         if (StringUtils.isBlank(sourceId)) {
620             builder.node(propertyKey).suggestions(new String[]{DEFAULT_FOLDER_ROOT});
621             return;
622         }
623         List<String> suggestedItems = new ArrayList<>();
624         String sourcePath = extractPath(sourceId);
625         sourceId = extractID(sourceId);
626         String path;

630             try {

634                                 .getInfo());

636                             path = info.getName() + '/';

642                         suggestedItems.add(

644                     }
645                 }

647                     suggestedItems.add(FOLDER_SUGGESTIONS_NONE);

650                     suggestedItems.add(0, String.format(PARENT_FOLDER_SUGGESTION,

660                     case java.net.HttpURLConnection.HTTP_NOT_FOUND:

701     public String getFolderPath(BoxFolder.Info folderInfo) {
702         List<BoxFolder.Info> pathCollection = folderInfo.getPathCollection();
703         // NOTE: pathCollection will be null for root folder
704         // pathCollection is the list of parent folders starting with the root (All Files)
705         if (pathCollection == null || pathCollection.isEmpty()) {
706             // root folder
707             return StringUtils.EMPTY;
708         }
709         StringBuilder pathBuilder = new StringBuilder(512);
710         // append the name of all parents except for the root folder, for backward-compatibility
711         for (int i = 1; i < pathCollection.size(); i++) {
712             pathBuilder.append(pathCollection.get(i).getName()).append('/');
713         }
714         // append the name of this folder
715         pathBuilder.append(folderInfo.getName()).append('/');
716         return pathBuilder.toString();
717     }

822     public BoxFolder getCurrentFolderFromPath(BoxAPIConnection connection, String filePath) {
823         BoxFolder root = BoxFolder.getRootFolder(connection);
824         if (StringUtils.isBlank(filePath)) {
825             return root;
826         }
827 
828         List<String> pathElements = splitPathToList(filePath);
829         // Since we cannot get the nested folder by path name so we need to recursively traverse the
830         // folder.
831         Iterator<String> pathIterator;
832         if (pathElements.size() > MAX_TRAVERSE) {
833             pathIterator = pathElements.subList(0, MAX_TRAVERSE).iterator();
834         } else {
835             pathIterator = pathElements.subList(0, pathElements.size() - 1).iterator();
836         }
837         BoxFolder current = traverseRecursive(pathIterator, root);
838         if (current == null) {
839             return root;
840         }
841         return current;
842     }
843 
844     protected BoxFolder traverseRecursive(Iterator<String> pathIterator,
845                                           BoxFolder current) {
846         if (!pathIterator.hasNext()) {
847             return current;
848         }
849         String name = pathIterator.next();
850         BoxFolder.Info folderInfo = null;
851         for (BoxItem.Info info : current.getChildren(MIN_FIELDS)) {
852             if (info instanceof BoxFolder.Info && name.equals(info.getName())) {
853                 folderInfo = (BoxFolder.Info) info;
854                 break;
855             }
856         }
857         if (folderInfo != null) {
858             current = folderInfo.getResource();
859             return traverseRecursive(pathIterator, current);
860         }
861         return current;
862     }
