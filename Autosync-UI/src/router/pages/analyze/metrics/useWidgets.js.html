import { useState, useCallback, useContext, useEffect } from 'react';
import useUserSettings from 'GlobalCommon/lib/sessionContext/useUserSettings';
import useMutateUserSettings from 'GlobalCommon/lib/sessionContext/useMutateUserSettings';
import { SessionContext } from 'GlobalCommon/lib/sessionContext';
import { get } from 'lodash/object';
import { arrayMove } from '@dnd-kit/sortable';
 
import { defaultWidgetsPreference } from './consts';
 
const useWidget = () => {
  const { activeOrg } = useContext(SessionContext);
  const { data: userSettings, error } = useUserSettings({ appName: 'studio' });
  const { mutate } = useMutateUserSettings({ appName: 'studio' });
  const [widgets, setWidgets] = useState([]);
  const [selectedWidget, setSelectedWidget] = useState(null);
 
  const updateUserSettings = useCallback((_widgets) => {
    const _settings = get(userSettings, `studio.orgs.${activeOrg.name}`, {});
    mutate({ settings: { ..._settings, metricsPreference: _widgets } });
  }, [userSettings, activeOrg, mutate]);
 
  const reorderWidgets = useCallback((active, over) => {
    if (active.id !== over.id) {
      setWidgets((items) => {
        const oldIndex = items.findIndex(item => item.id === active.id);
        const newIndex = items.findIndex(item => item.id === over.id);
        const _widgets = arrayMove(items, oldIndex, newIndex);
        updateUserSettings(_widgets);
        return _widgets;
      });
    }
  }, [updateUserSettings]);
 
  const removeWidget = useCallback((id) => {
    setWidgets((items) => {
      const _widgets = [...items];
      _widgets.find(_widget => _widget.id === id).display = false;
      updateUserSettings(_widgets);
      return _widgets;
    });
  }, [updateUserSettings]);
 
  const addWidget = useCallback((id) => {
    setWidgets((items) => {
      const _widgets = [...items];
      let _widget = _widgets.find(_w => _w.id === id);
      if (_widget) {
        _widget.display = true;
      } else {
        _widget = defaultWidgetsPreference.find(_w => _w.id === id);
        _widget.display = true;
        _widgets.push(_widget);
      }
 
      updateUserSettings(_widgets);
      return _widgets;
    });
  }, [updateUserSettings]);
 
  const toggleWidget = useCallback(id => setSelectedWidget(prev => (prev !== id ? id : null)), []);
 
  useEffect(() => {
    if (userSettings && !widgets.length) {
      const _widgets = get(userSettings, `studio.orgs.${activeOrg.name}.metricsPreference`, []);
 
      if (!_widgets.length) {
        setWidgets(defaultWidgetsPreference);
      } else {
        setWidgets(_widgets.filter(_w => defaultWidgetsPreference.find(_dw => _dw.id === _w.id)));
      }
      updateUserSettings(_widgets);
    }
  }, [userSettings, updateUserSettings, activeOrg, widgets]);
 
  useEffect(() => {
    if (error && !widgets.length) {
      setWidgets(defaultWidgetsPreference);
    }
  }, [error, widgets]);
 
  return {
    widgets,
    selectedWidget,
    setSelectedWidget,
    addWidget,
    removeWidget,
    reorderWidgets,
    toggleWidget,
  };
};
 
export default useWidget;
