const AdvancedSearch = ({
  activeOrg,
  columns,
  setFiltersOpen,
  inputRef,
  isNodePushOut,
  style,
}, ref) => {
  const { advancedSearch, applyAdvancedSearch, aggregateType } = useContext(ExecutionContext);
  const {
    status, invoker, projectId, plexSearch, nodeSearch, userSearch,
  } = advancedSearch;
  const [stateSelected, setStateSelected] = useState(status.map(mapStateStringToObjects));
  const [invokerSelected, setInvokerSelected] = useState(invoker);
  const [projectSelected, setProjectSelected] = useState(projectId ? projectId.split(',') : []);
  const [plexSearchValue, setPlexSearchValue] = useState(plexSearch);
  const [nodeSearchValue, setNodeSearchValue] = useState(nodeSearch);
  const [userSearchValue, setUserSearchValue] = useState(userSearch);
  const [preference, setPreference] = useState([...columns]);
 
  const onToggleColumn = (name) => {
    setPreference(_columns => _columns.map((column) => {
      if (column.name === name) {
        return { ...column, display: !column.display };
      }
      return column;
    }));
  };
 
  const onSortEnd = ({ oldIndex, newIndex }) => {
    setPreference(prev => reorder(
      prev,
      oldIndex,
      newIndex,
    ));
  };
 
  const resetColumns = useCallback(() => {
    const originalColumns = (() => {
      if (aggregateType === -1) return runtimeColumns;
      if (aggregateType === 1) return taskColumns;
      return pipelineColumns;
    })();
    const sortedStrings = originalColumns.map(mapToName);
 
    setPreference(prev =>
      prev
        // apply original sorting
        .sort(
          ({ name: a }, { name: b }) =>
            sortedStrings.indexOf(a) - sortedStrings.indexOf(b),
        )
        // apply original visibility
        .map(col => ({
          ...col,
          ...originalColumns.find(({ name }) => name === col.name),
        })),
    );
  }, [aggregateType]);
 
  const onClearFilters = useCallback(() => {
    resetColumns();
    setInvokerSelected(null);
    setProjectSelected([]);
    setStateSelected([]);
    setUserSearchValue('');
    if (!isNodePushOut) {
      setPlexSearchValue('');
      setNodeSearchValue('');
    }
  }, [resetColumns]);
 
  const onApplyFilters = useCallback(() => {
    const hiddenColumns = preference.filter(({ display }) => !display).map(mapToName);
    const columnOrder = preference.map(mapToName);
 
    setFiltersOpen(false);
 
    applyAdvancedSearch({
      hiddenColumns,
      columnOrder,
      advancedSearch: {
        plexSearch: plexSearchValue.trim(),
        nodeSearch: nodeSearchValue.trim(),
        userSearch: userSearchValue.trim(),
        invoker: invokerSelected,
        status: stateSelected.map(mapToStrings),
        projectId: projectSelected.length ? projectSelected.join(',') : null,
      },
    });
  }, [
    preference,
    invokerSelected,
    stateSelected,
    nodeSearchValue,
    plexSearchValue,
    projectSelected,
    userSearchValue,
    applyAdvancedSearch,
  ]);
 
  const SortableItem = SortableElement(({ value }) => (
    <Chip
      label={value.filterLabel || value.label}
      selected={value.display}
      onDelete={() => onToggleColumn(value.name)}
      onAdd={() => onToggleColumn(value.name)}
      ariaLabel="column"
      className="sortable-chip-item"
    />
  ));
 
  const SortableList = SortableContainer(({ items }) => (
    <div className="table-preference-item">
      {items.map((item, index) => (
        <SortableItem key={`item-${item.label}`} index={index} value={item} />
      ))}
    </div>
  ));
 
  const renderInvocationListItemAriaLabel = item => `${item.name || 'all'}`;
  const renderStatusListItemAriaLabel = item => `${item.value || 'all'}`;
 
  const onInvocationSelect = (item) => {
    setInvokerSelected((prev) => {
      if (item.value === prev) {
        return null;
      }
      return item.value;
    });
  };
 
  const onProjectSelect = useCallback(items => setProjectSelected(items.map(i => i.key)), []);
  const onClearProject = useCallback(() => setProjectSelected([]), []);
 
  return (
    <OnBody>
      <FilterDropdown
        ref={ref}
        inputRef={inputRef}
        title={message.ANALYZE.EXECUTIONS.ADVANCED_FILTERING}
        qaId="analyze-execution-runtime-filters"
        onClear={onClearFilters}
        onApply={onApplyFilters}
        onCancel={() => setFiltersOpen(false)}
        style={{ height: 430 }}
        dataAnalytics="runtime table filters"
      >
        <div className="sl-analyze-execution-filters" style={style} ref={ref}>
          <div className="dropdowns">
            {aggregateType !== 0 && (
              <Dropdown
                items={invocationList}
                label={message.ANALYZE.EXECUTIONS.INVOCATION_TYPE}
                onSelect={onInvocationSelect}
                selected={invocationList.find(
                  item => item.value === invokerSelected,
                )}
                qaId="analyze-execution-runtime-filters-invoker"
                ariaLabel="execution table invocation type filter"
                renderListItemAriaLabel={renderInvocationListItemAriaLabel}
                insideLabel
              />
            )}
            {aggregateType === -1 && (
              <Fragment>
                <Dropdown
                  className="dsm-dropdown"
                  items={stateList}
                  label={message.ANALYZE.EXECUTIONS.STATUS}
                  multi
                  onSelect={setStateSelected}
                  placeholder="All"
                  chipLabelFrom="chipLabel"
                  selectedList={stateSelected}
                  qaId="analyze-execution-runtime-filters-state"
                  ariaLabel="execution table status filter"
                  renderListItemAriaLabel={renderStatusListItemAriaLabel}
                  insideLabel
                />
                <ProjectDropdown
                  className="dsm-dropdown"
                  dropdownPlaceholder="All"
                  orgName={activeOrg.name}
                  selectedList={projectSelected}
                  onSelect={onProjectSelect}
                  onClear={onClearProject}
                  qaId="analyze-execution-runtime-filters-project"
                  ariaLabel="execution table projects filter"
                />
              </Fragment>
            )}
            <TextField
              value={plexSearchValue}
              onChange={e => setPlexSearchValue(e.target.value)}
              label={message.ANALYZE.EXECUTIONS.PLEX_NAME}
              qaId="analyze-execution-runtime-filters-plex"
              insideLabel
              containerStyle={{ boxShadow: 'none' }}
              ariaLabel={`execution table snaplex filter ${plexSearchValue}`}
              disabled={isNodePushOut}
            />
            <TextField
              value={nodeSearchValue}
              onChange={e => setNodeSearchValue(e.target.value)}
              label={message.ANALYZE.EXECUTIONS.NODE_NAME}
              qaId="analyze-execution-runtime-filters-node"
              insideLabel
              containerStyle={{ boxShadow: 'none' }}
              ariaLabel={`execution table node filter ${nodeSearchValue}`}
              disabled={isNodePushOut}
            />
            <TextField
              value={userSearchValue}
              onChange={e => setUserSearchValue(e.target.value)}
              label={message.ANALYZE.EXECUTIONS.OWNER}
              qaId="analyze-execution-runtime-filters-user"
              insideLabel
              containerStyle={{ boxShadow: 'none' }}
              ariaLabel={`execution table owned by filter ${userSearchValue}`}
            />
          </div>
          <div className="columns">
            <div className="header">
              {message.ANALYZE.EXECUTIONS.COLUMN_PREFERENCES}
            </div>
            <SortableList
              items={preference}
              onSortEnd={onSortEnd}
              axis="xy"
              distance={8}
            />
          </div>
        </div>
      </FilterDropdown>
    </OnBody>
  );
};
 
export default forwardRef(AdvancedSearch);
