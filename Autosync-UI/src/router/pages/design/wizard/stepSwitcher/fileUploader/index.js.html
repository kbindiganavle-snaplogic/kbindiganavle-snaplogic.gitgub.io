import React, { useCallback, useMemo, useState } from 'react';
import cn from 'classnames';
import { useDropzone } from 'react-dropzone';
import message from '@/locales';
import { maxFileSizeString } from '@/common/utils';
import { camelCase } from 'lodash';
import FileErrorDialog from './fileErrorDialog';
import FileTable from './fileTable';
 
const {
  DESIGN: {
    DATA_PIPELINE: {
      FILE_UPLOAD: { CTA, MAX_FILE_COUNT },
    },
  },
} = message;
 
const mapDropzoneToError = ({ errors, file }) => ({ errorCode: camelCase(errors[0].code), fileName: file.path });
 
const FileUploader = ({
  qaId, selectedFiles = [], onSelect, disabled, accept, maxSize, maxFiles, isWizard, loadType,
}) => {
  const [errors, setErrors] = useState({});
  const acceptObj = useMemo(() => {
    const key = `text/${accept}`;
    const value = `.${accept}`;
 
    return {
      [key]: [value],
    };
  }, [accept]);
 
  // Filter out files that were uploaded previously
  const stagedFiles = selectedFiles.filter(({ _size }) => _size);
 
  const readAsText = useCallback(
    (files) => {
      const mapToPromise = file =>
        new Promise((resolve) => {
          const reader = new FileReader();
          const { path, size } = file;
          reader.onload = ({ target: { result } }) => {
            resolve({
              table: path.replace(`.${accept}`, `_${accept}`),
              load_type: loadType || 'reload',
              // need to upload separately and remove
              _content: result,
              _path: path,
              _size: size,
            });
          };
          reader.readAsText(file);
        });
 
      return Promise.all(files.map(mapToPromise));
    },
    [accept],
  );
 
  const { getRootProps, getInputProps } = useDropzone({
    accept: acceptObj,
    maxSize,
    maxFiles,
    onDrop: async (acceptedFiles, fileRejections) => {
      setErrors([]);
      const acceptedWithContentFiles = (acceptedFiles || []).filter(file => file.size > 0);
      const readFiles = await readAsText(acceptedWithContentFiles);
 
      // Remove duplicates
      const newFilesOnly = readFiles.filter(file => !stagedFiles.some(({ table }) => table === file.table));
 
      // Combine with previously selected files
      const combinedFiles = [...stagedFiles, ...newFilesOnly];
 
      // Limit to maxFiles
      const acceptedNumberOfFiles = combinedFiles.slice(0, maxFiles);
 
      // Select the files that were too many
      const rejectedBecauseTooMany = combinedFiles.slice(maxFiles);
 
      // If user selects more than maxFiles *at once*, React Dropzone will reject all of them
      // Grab whichever files we can, while it is less than the max
      const { moreFiles, _fileRejections } = fileRejections.reduce((acc, obj) => {
        const hasReachedMax = stagedFiles.length + acc.moreFiles.length === maxFiles;
        const isOtherError = obj.errors[0].code !== 'too-many-files';
        const isNew = !stagedFiles.some(({ _path }) => _path === obj.file.path);
 
        if (hasReachedMax || isOtherError) {
          acc._fileRejections.push(obj);
        } else if (isNew) {
          acc.moreFiles.push(obj.file);
        }
        return acc;
      }, { moreFiles: [], _fileRejections: [] });
 
      const moreFilesRead = await readAsText(moreFiles);
 
      const combinedErrors = [
        // Map React Dropzone errors to an error object for the dialog
        ..._fileRejections.map(mapDropzoneToError),
        // Map files to the same error object
        ...rejectedBecauseTooMany.map(({ _path }) => ({ errorCode: 'tooManyFiles', fileName: _path })),
      ].reduce((acc, { errorCode, fileName }) => {
        acc[errorCode].push(fileName);
        return acc;
      }, { tooManyFiles: [], fileInvalidType: [], fileTooLarge: [] });
 
      setErrors(combinedErrors);
 
      onSelect([...acceptedNumberOfFiles, ...moreFilesRead]);
    },
    disabled,
  });
 
  const onDelete = useCallback((table) => {
    onSelect(stagedFiles.filter(f => f.table !== table));
  }, [stagedFiles]);
 
  const onClose = () => setErrors({});
  const isEmpty = !stagedFiles.length;
 
  const { onClick, ...rootProps } = getRootProps({ className: cn({ 'autosync-file-uploader__container': isEmpty }) });
 
  return (
    <div className="autosync-file-uploader">
      <div {...rootProps} onClick={isEmpty ? onClick : null}>
        <input qa-id={qaId} {...getInputProps()} />
        {isEmpty ? (
          <React.Fragment>
            <p className="autosync-file-uploader__container__big">{CTA}</p>
            {maxSize && (
              <p className="autosync-file-uploader__container__small">{maxFileSizeString(maxSize)}</p>
            )}
            {maxFiles && (
              <p className="autosync-file-uploader__container__small">{String.format(MAX_FILE_COUNT, maxFiles)}</p>
            )}
          </React.Fragment>
        ) : (
          <FileTable files={stagedFiles} onDelete={onDelete} isWizard={isWizard} onAddMore={onClick} />
        )}
      </div>
      <FileErrorDialog maxFiles={maxFiles} maxSize={maxSize} onClose={onClose} {...errors} />
    </div>
  );
};
 
export default FileUploader;
