import React, { useState, useMemo, useCallback, useEffect } from 'react';
import { get } from 'lodash/object';
import { useCredentials, useSchemas, useMutateUpdateCredentials } from '@/hooks/autosync';
import Dropdown from 'SL_UI/v3/dropdown';
import message from '@/locales';
import LoadingOrError from '@/common/loadingOrError';
import { getAccountObjFromCredentials, getAutosyncAccountTag } from '@/common/utils';
 
import AccountFormContainer from '../../accountFormContainer';
 
import '../styles.scss';
 
const Credentials = ({ account, closeDialog, credentials: credentialsFromProps,
  credentialsLoading: loadingFromProps, loadingClassName, hideChildrenWhileLoading = false,
  clearOnClose = true, setAccount, shareSelected, runtimePath }) => {
  const {
    data: credentials = [],
    isPlaceholderData: credentialsLoading,
  } = useCredentials();
  const { mutate } = useMutateUpdateCredentials();
  const { data: schemas, isFetching: schemaLoading } = useSchemas();
  const [selected, setSelected] = useState(account);
 
  const onClose = useCallback((reason) => {
    if (closeDialog) {
      closeDialog();
    } else if (reason === 'delete' || reason === 'cancel') {
      setSelected(null);
    }
  }, [closeDialog]);
 
  const onSelect = useCallback(item => setSelected(item.id), []);
 
  const selectedAccount = useMemo(() => getAccountObjFromCredentials(selected, credentials), [selected, credentials]);
 
  useEffect(() => {
    setSelected(account);
  }, [account]);
 
  const filterSchema = (item) => {
    if (item.account_class_ids.indexOf(get(selectedAccount, 'data.class_id')) > -1) {
      const accountTags = get(selectedAccount, 'data.metadata.tags', []);
      return item.needAccountTag ? accountTags.includes(getAutosyncAccountTag(item.name)) : true;
    }
    return false;
  };
 
  const schema = useMemo(() => {
    if (!selectedAccount || !schemas.length) return null;
    return schemas.find(filterSchema);
  }, [selectedAccount, schemas]);
 
  return (<LoadingOrError
    isLoading={credentialsLoading || schemaLoading || loadingFromProps}
    hideChildrenWhileLoading={hideChildrenWhileLoading}
    className={loadingClassName}
  >
    <div className="autosync-settings-container">
      <Dropdown
        items={credentialsFromProps || credentials}
        onSelect={onSelect}
        selected={selectedAccount}
        label={message.DESIGN.DATA_PIPELINE.SETTINGS.SELECT_ACCOUNT}
        qaId="autosync-credentials-selection"
      />
      {
        !!schema && <AccountFormContainer
          accountInstance={selectedAccount.data}
          overwriteValue={schema.overwriteValue}
          visibleFieldsPath={schema.visibleFieldsPath}
          visibleButNotRequiredFieldsPath={schema.visibleButNotRequiredFieldsPath}
          isOAuth2Account={schema.is_oauth}
          saveBeforeValidate={schema.saveBeforeValidate}
          switchAccountMode={false}
          classId={schema.default_account_class_id}
          extraProperties={schema.extraProperties}
          autosyncAccountTag={schema.needAccountTag ? getAutosyncAccountTag(schema.name) : ''}
          onClose={onClose}
          isReadOnly={selectedAccount.isReadOnly}
          mutate={mutate}
          clearOnClose={clearOnClose}
          setAccount={setAccount}
          shareSelected={shareSelected}
          accountPath={selectedAccount.data?.path_id.slice(0, selectedAccount.data?.path_id.lastIndexOf('/'))}
          runtimePath={runtimePath}
        />
      }
    </div>
  </LoadingOrError>);
};
 
export default Credentials;
