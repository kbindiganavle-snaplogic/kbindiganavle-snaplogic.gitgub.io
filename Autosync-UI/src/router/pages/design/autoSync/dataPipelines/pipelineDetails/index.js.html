import React, { useMemo, useState, useCallback, useContext } from 'react';
import { Link } from 'react-router-dom';
import Button from 'SL_UI/v3/button';
import Tooltip from 'SL_UI/v3/tooltip';
import { get } from 'lodash/object';
import { defaultTo } from 'lodash/util';
import cn from 'classnames';
import { isEmpty } from 'lodash/lang';
import {
  formatDate,
  autosyncStringToIcon,
  getAbbreviatedInt,
} from '@/common/utils';
import PipelineSVG from '@/common/pipeline';
import LoadingOrError from '@/common/loadingOrError';
import message from '@/locales';
import {
  useSchemaMap,
  useMutateDataPipeline,
  useMutateDeleteAsset,
  useFallbackIcon,
  useDataPipelines,
  useSchemas,
} from '@/hooks';
import { useNextRuntimes, useMutateRunDataPipeline, usePermsBooleans } from '@/hooks/autosync';
import { SessionContext } from 'GlobalCommon/lib/sessionContext';
import { isOrgAdministrator } from 'GlobalCommon/lib/sessionContext/utils';
import { NotifyContext } from 'GlobalCommon/lib/notifyContext';
import { useListUser } from '@/hooks/user';
import PipelineNotification from '../pipelineNotification';
 
import './styles.scss';
 
import RunningIcon from '../RunningIcon';
import {
  getRefreshScheduleStr,
  mapAutosyncStatusToRuntimeStatus,
  areEndpointsConfigured,
  isCSVConfigured,
  getLastRunDescription,
  getLastRunElapsedTime,
  areSourcesContainCSV,
  getConflictTables,
} from '../../utils';
import DeleteConfirmDialog from '../deleteConfirmDialog';
import { pipelineDescription } from '../../../wizard/previewPushout/description';
import { qaIdPrefix } from '../../consts';
import EnableSwitch from '../enableSwitch';
 
const {
  COMMON: {
    EDIT, MOVE, CHOOSE_LOCATION,
  },
  DESIGN: {
    DATA_PIPELINE: {
      LAST_RUN, NEXT_RUN, REPEATS_EVERY, RUNNING, AUTO_SYNCHRONIZE, PIPELINE_OVERVIEW, ELAPSED, SCHEDULED_RUN_ONCE,
      CREATED_BY, RUN_NOW, TOTAL_TABLES_LOADED, RECORDS_PROCESSED, TOTAL_MONTHLY_RECORDS_PROCESSED,
      TOTAL, NEW, UPDATED, FAILED, EXECUTIONS, VIEW_EXECUTION_PAGE, SCHEDULED_EVERY, RUN_ONCE_DNR,
      FILE_UPLOAD: { CTA_SHORT },
      ON_RUN: { ERROR },
      CONFLICT_TABLES: { ERROR: CONFLICT_TABLES_ERROR },
    },
  },
} = message;
 
const PipelineDetails = ({ onClose, card, setShareDialogOpen, setSharePath,
  setShareInstance, instanceId, onUpload, onDeleteSuccess }) => {
  const [deleteConfirmOpen, setDeleteConfirmOpen] = useState(false);
  // const [showErrors, setShowErrors] = useState(true);
  const { data: schemaMap } = useSchemaMap();
  const isCSV = card?.sources.some(({ name }) => name.includes('CSV'));
 
  const mutateEditDataPipeline = useMutateDataPipeline(instanceId, true);
  const mutateDeleteAsset = useMutateDeleteAsset();
  const { mutate: runAutosync } = useMutateRunDataPipeline();
 
  const { user, activeOrg } = useContext(SessionContext);
  const { data: userList } = useListUser(activeOrg.name);
 
  const { data: dataPipeline } = useDataPipelines({ instance_id: instanceId });
 
  const owner = userList.find(u => u.username === dataPipeline?.owner) || {};
 
  const { canRun, canEdit } = usePermsBooleans(dataPipeline.perms);
 
  const isOwner = useMemo(() => user.username === dataPipeline?.owner, [dataPipeline]);
  const isOrgAdmin = isOrgAdministrator(user, activeOrg.name);
  const canShare = isOrgAdmin ? true : isOwner;
 
  const { data: schemas } = useSchemas();
  const conflictTables = useMemo(() => {
    const target = dataPipeline?.targets?.length ? dataPipeline.targets[0] : {};
    const schema = schemas?.find(s => s.endpoint_type === 'target' && s.name === target?.name);
    return getConflictTables(dataPipeline?.sources, target, schema);
  }, [dataPipeline, schemas]);
 
  const { error } = useContext(NotifyContext);
 
  const runtimeStatusClass = useMemo(() => {
    const lastRunStatus = get(dataPipeline, 'stats.last_run.status');
    if (lastRunStatus) {
      return mapAutosyncStatusToRuntimeStatus(lastRunStatus);
    }
    return '';
  }, [dataPipeline]);
 
  const showError = useMemo(() => runtimeStatusClass === 'failed', [runtimeStatusClass]);
 
  const {
    data: nextRt,
    isLoading: isRuntimesLoading,
  } = useNextRuntimes([], dataPipeline?.scheduled_task_snid);
 
  let nextRun = '-';
  if (isRuntimesLoading) {
    nextRun = '';
  } else if (nextRt) {
    nextRun = formatDate(nextRt, true);
  }
 
  const sources = useMemo(() => {
    if (!dataPipeline || !dataPipeline.sources) return [];
    return dataPipeline.sources.map(source => ({
      service: schemaMap && schemaMap[source.schema_id] && schemaMap[source.schema_id].icon,
      id: schemaMap && schemaMap[source.schema_id] && schemaMap[source.schema_id].name,
    }));
  }, [dataPipeline, schemaMap]);
 
  const target = useMemo(() => {
    if (!dataPipeline || !dataPipeline.targets || !schemaMap) return {};
    if (dataPipeline.targets.length > 0) {
      const schema_id = get(dataPipeline, 'targets[0].schema_id');
      const icon = get(schemaMap, `${schema_id}.icon`);
      const name = get(schemaMap, `${schema_id}.name`);
      if (!schema_id || !icon || !name) return {};
      return {
        service: icon,
        name,
      };
    }
    return {};
  }, [dataPipeline, schemaMap]);
 
  let refreshScheduleStr = '-';
  if (isEmpty(dataPipeline?.schedule)) {
    refreshScheduleStr = '-';
  } else if (isEmpty(dataPipeline?.schedule?.repeat)) {
    if (dataPipeline.enabled) {
      refreshScheduleStr = RUN_ONCE_DNR;
    } else {
      refreshScheduleStr = SCHEDULED_RUN_ONCE;
    }
  } else {
    refreshScheduleStr = String.format(
      dataPipeline.enabled ? REPEATS_EVERY : SCHEDULED_EVERY,
      getRefreshScheduleStr(dataPipeline.schedule),
    );
  }
 
  const description = useMemo(() => {
    if (!dataPipeline) return '';
    // if not enabled or no next runtime, hide refresh schedule.
    const hideInterval = !dataPipeline.enabled || !nextRt;
    return pipelineDescription(dataPipeline, hideInterval);
  }, [dataPipeline, nextRt]);
 
  const onToggle = () => {
    mutateEditDataPipeline.mutate({
      payload: {
        instance: {
          ...dataPipeline,
          enabled: !dataPipeline.enabled,
        },
        full_save: !dataPipeline.enabled,
      },
    });
  };
 
  const onRun = () => {
    if (conflictTables.length) {
      error({ message: CONFLICT_TABLES_ERROR });
    } else {
      runAutosync({
        dataPipeline,
        _onError: () => error({ message: String.format(ERROR, dataPipeline?.name) }),
      });
    }
  };
 
  const runNowDisabled = useMemo(() => {
    if (!dataPipeline || !schemaMap || !dataPipeline.sources || !dataPipeline.targets) return true;
    return areSourcesContainCSV(schemaMap, dataPipeline.sources)
        || !areEndpointsConfigured(schemaMap, dataPipeline.sources, dataPipeline.targets);
  }, [dataPipeline, schemaMap]);
 
  const closeConfirmDialog = useCallback(() => setDeleteConfirmOpen(false), []);
  const openDeleteConfirmDialog = () => {
    setDeleteConfirmOpen(true);
  };
 
  const handleDeletePipeline = () => {
    mutateDeleteAsset.mutate({
      closeAfter: true,
      onClose,
      snode_id: dataPipeline.snode_id,
      isAutosyncAsset: true,
      _onSuccess: onDeleteSuccess(dataPipeline.snode_id),
    });
  };
 
  const { data: purposeIcon } = useFallbackIcon(
    'img/common/autosync',
    autosyncStringToIcon(dataPipeline.metadata.purpose),
    dataPipeline.metadata._purposeIconFallback,
  );
 
  const isPipelineRunning = useMemo(() =>
    mapAutosyncStatusToRuntimeStatus(get(dataPipeline, 'stats.status', '')) === 'running', [dataPipeline]);
 
  const records_processed = useMemo(() => {
    const inserted_records = get(dataPipeline, 'stats.last_run.inserted_records');
    const updated_records = get(dataPipeline, 'stats.last_run.updated_records');
    if (!inserted_records || !updated_records
      || isNaN(inserted_records) || isNaN(updated_records)) return '-';
    return getAbbreviatedInt(parseInt(inserted_records, 10) + parseInt(updated_records, 10));
  }, [dataPipeline, isPipelineRunning]);
 
  const lastRunDescription = getLastRunDescription(dataPipeline);
  const elapsedTime = getLastRunElapsedTime(dataPipeline);
 
 
  const onShare = () => {
    setShareDialogOpen(true);
    setSharePath(card.path);
    setShareInstance(card);
  };
 
  const onClickUpload = useCallback(() => {
    let _card = card;
    // if instance_version from dataPipeline is more up-to-date, then use dataPipeline
    if (dataPipeline.instance_version > card.instance_version) {
      _card = dataPipeline;
    }
    onUpload(_card);
  }, [card, onUpload, dataPipeline]);
 
  return (
    <React.Fragment>
      <div style={{ height: '100%', overflow: 'auto' }}>
        <div className="sl-design-pipelines-detail-container">
          <div className="sl-design-pipelines-detail-header">
            <div style={{ width: '80%' }}>
              <div className="sl-f-16 sl-f-bold label" title={dataPipeline.name}>{dataPipeline.name}</div>
              <div className="sl-f-12 sl-f-bold sl-color-grey-2">
                {`${CREATED_BY} ${owner.first_name} ${owner.last_name}`}, {formatDate(dataPipeline.time_created, true)}
              </div>
            </div>
            <div style={{ display: 'flex', gap: 8 }}>
              <div
                className={cn('sl-design-pipelines-delete-icon', {
                  disabled: !canEdit,
                })}
                onClick={openDeleteConfirmDialog}
                qa-id={`${qaIdPrefix}-details-delete`}
                disabled={!canEdit}
              />
              {(!isCSV || canShare) && <div
                className={cn('sl-design-pipelines-share-icon', {
                  disabled: !canShare,
                })}
                disabled={!canShare}
                onClick={onShare}
              />}
              <div
                className="sl-design-pipelines-close-icon"
                onClick={onClose}
                qa-id={`${qaIdPrefix}-details-close`}
              />
            </div>
          </div>
          <div className="sl-design-pipelines-detail-description">
            {description}
          </div>
          <Link
            to={{
              pathname: `/design/datapipeline/edit/${instanceId}`,
            }}
            className={`pipeline-details-edit-link ${!canEdit ? 'disabled' : ''}`}
            disabled={!canEdit}
          >
            <Button
              qaId={`${qaIdPrefix}-details-edit`}
              label={EDIT}
              secondary
              className="sl-design-pipelines-button secondary"
              onClick={() => {}}
              disabled={!canEdit}
            />
          </Link>
        </div>
        <div className="divider" />
        <div className="sl-design-pipelines-detail-container">
          {isCSV ? (
            <Button
              qaId={`${qaIdPrefix}-details-upload-csv`}
              label={CTA_SHORT}
              className="sl-design-pipelines-button"
              onClick={onClickUpload}
              disabled={!(isCSVConfigured(schemaMap, card.sources, card.targets) && canEdit)}
            />
          ) : (
            <Button
              qaId={`${qaIdPrefix}-details-run-now`}
              label={RUN_NOW}
              className="sl-design-pipelines-button"
              onClick={onRun}
              disabled={runNowDisabled || !canRun}
            />
          )}
 
          {!isCSV && (
            <div className="sl-design-pipelines-detail-autosync">
              <div className="sl-design-pipelines-detail-autosync-run">
                <div>
                  <div className="sl-f-16 sl-f-bold">
                    {AUTO_SYNCHRONIZE}
                  </div>
                  <div className="sl-f-12 sl-f-bold sl-color-grey-2" qa-id={`${qaIdPrefix}-details-schedule-string`}>
                    {refreshScheduleStr}
                  </div>
                </div>
                <div>
                  <EnableSwitch nextRt={nextRt} {...dataPipeline} />
                </div>
              </div>
 
              <div className="sl-design-pipelines-detail-autosync-run">
                <div>
                  <div className="sl-f-12 sl-f-bold sl-color-grey-2">
                    {NEXT_RUN}
                  </div>
                  <div className="sl-f-14 sl-f-bold" qa-id={`${qaIdPrefix}-details-next-run`}>
                    {dataPipeline.enabled || isPipelineRunning ?
                      <LoadingOrError
                        iconSize={20}
                        data={nextRt}
                        isLoading={isRuntimesLoading}
                      >
                        {isPipelineRunning ? 'Running' : nextRun}
                      </LoadingOrError>
                      : '-'
                    }
                  </div>
                </div>
                {isPipelineRunning && <RunningIcon />}
              </div>
            </div>
          )}
          <div className="sl-design-pipelines-detail-autosync-run">
            <div>
              <div className={`sl-f-12 sl-f-bold sl-color-grey-2 sl-color-${runtimeStatusClass}`}>
                {LAST_RUN}
              </div>
              <div
                className={`sl-f-14 sl-f-bold sl-color-grey-1 sl-color-${runtimeStatusClass}`}
                qa-id={`${qaIdPrefix}-details-last-run`}
              >
                {lastRunDescription}
              </div>
            </div>
            <div className="elapsed sl-f-extra-bold">
              <Tooltip overlay={ELAPSED} placement="top">
                <div
                  className="autosync-icon elapsed-pushout"
                />
              </Tooltip>
              {elapsedTime}
            </div>
          </div>
          {showError &&
            <PipelineNotification stats={dataPipeline?.stats} pipelineName={dataPipeline?.name} />
          }
          <div className="sl-design-pipelines-process-details">
            <div>
              <div className="sl-f-10 sl-color-grey-2">
                {TOTAL_TABLES_LOADED}
              </div>
              <div className="sl-f-12 sl-color-grey-1" qa-id={`${qaIdPrefix}-details-table-loaded`}>
                {defaultTo(dataPipeline.stats?.last_run?.tables_loaded, '-')}
              </div>
            </div>
            <div className="records-processed-row">
              <div>
                <div className="sl-f-10 sl-color-grey-2">
                  {RECORDS_PROCESSED}
                </div>
                <div className="sl-f-12 sl-color-grey-1" qa-id={`${qaIdPrefix}-details-records-processed`}>
                  {records_processed}
                </div>
              </div>
              <div className="records-processed-row__details">
                <div>
                  <div className="sl-f-8 sl-f-normal sl-color-grey-3">{NEW}</div>
                  <div qa-id={`${qaIdPrefix}-details-new-records`}>
                    {getAbbreviatedInt(dataPipeline?.stats?.last_run?.inserted_records)}
                  </div>
                </div>
                <div>
                  <div className="sl-f-8 sl-f-normal sl-color-grey-3">{UPDATED}</div>
                  <div qa-id={`${qaIdPrefix}-details-updated-records`}>
                    {getAbbreviatedInt(dataPipeline?.stats?.updated_records)}
                  </div>
                </div>
                <div>
                  <div className="sl-f-8 sl-f-normal sl-color-grey-3">{FAILED}</div>
                  <div qa-id={`${qaIdPrefix}-details-failed-records`}>
                    {getAbbreviatedInt(dataPipeline?.stats?.failed_records)}
                  </div>
                </div>
              </div>
            </div>
            {/* Hidden for SKO/March  */}
            {/* <div style={{ display: 'flex', justifyContent: 'space-between' }}>
              <div>
                <div className="sl-f-10 sl-color-grey-2">
                  {TOTAL_MONTHLY_RECORDS_PROCESSED}
                </div>
                <div className="sl-f-12 sl-color-grey-1">
                  {getAbbreviatedInt(dataPipeline.stats.total_monthly_records_processed)}
                  {!isNaN(dataPipeline.stats.total_monthly_records_processed) &&
                    <span className="sl-f-10 sl-color-grey-3">{` ${TOTAL}`}</span>
                  }
                </div>
              </div>
              <div style={{ display: 'flex', gap: 8 }}>
                <div>
                  <div className="sl-f-8 sl-f-normal sl-color-grey-3">{NEW}</div>
                  <div>{getAbbreviatedInt(dataPipeline.stats.inserted_records)}</div>
                </div>
                <div>
                  <div className="sl-f-8 sl-f-normal sl-color-grey-3">{UPDATED}</div>
                  <div>{getAbbreviatedInt(dataPipeline.stats.updated_records)}</div>
                </div>
                <div>
                  <div className="sl-f-8 sl-f-normal sl-color-grey-3">{FAILED}</div>
                  <div>{getAbbreviatedInt(dataPipeline.stats.failed_records)}</div>
                </div>
              </div>
            </div> */}
          </div>
        </div>
        <div className="divider" />
        <div className="sl-design-pipelines-detail-container">
          <div className="sl-f-16 sl-f-bold">{PIPELINE_OVERVIEW}</div>
          <div className="sl-design-pipelines-detail-svg-container">
            <PipelineSVG
              width={260}
              height={210}
              sources={sources}
              destination={target.service}
              output={purposeIcon}
              showLabels={false}
            />
          </div>
        </div>
        {/* All the below will most likely be hidden for February release */}
        {/* <div className="divider" />
        <div className="sl-design-pipelines-detail-container space-between">
          <div className="sl-f-16 sl-f-bold">{EXECUTIONS}</div>
          <div className="temp-link">{VIEW_EXECUTION_PAGE}</div>
        </div>
        <div className="divider" />
        <div className="sl-design-pipelines-detail-container">
          <div className="sl-f-16 sl-f-bold">{MOVE}</div>
          <Button
            qaId="data-pipeline-choose-location"
            label={CHOOSE_LOCATION}
            secondary
            className="sl-design-pipelines-button secondary"
            onClick={() => {}}
          />
        </div> */}
      </div>
      <DeleteConfirmDialog
        open={deleteConfirmOpen}
        onConfirm={handleDeletePipeline}
        onClose={closeConfirmDialog}
        instanceToDelete={dataPipeline}
      />
    </React.Fragment>
  );
};
 
export default PipelineDetails;
