import React, { useState, useContext, useMemo } from 'react';
import Button from 'SL_UI/v3/button';
import cn from 'classnames';
import message from '@/locales';
import Title from '@/common/title';
import { ProvisionOrgContext } from '../contexts/provisionOrgContext';
import {
  IIP_APP_KEY,
  provisioningWrapper,
} from './actions';
import CustomerInfo from './components/customerInfo';
import CreateOrg from './components/createOrg';
import Features from './components/features';
import SnapPacks from './components/snappacks';
import Nodes from './components/nodes';
import Confirmation from './components/confirmation';
import Results from './components/results';
 
import './styles.scss';
 
const steps = ['Customer Info', 'Create Org', 'Features', 'Snap Packs', 'Nodes', 'Confirmation', 'Results'];
 
export const FULL_PACK_ACCESS_KEY = 'full_pack_access';
 
const ProvisionOrgModal = ({ onClose }) => {
  const {
    formData,
    snapPackList,
    featuresMap,
    appSubMap,
    activeStep,
    setActiveStep,
  } = useContext(ProvisionOrgContext);
  const { customer_info, organization, administrator_account, nodes } = formData;
  const [validatedSteps, setValidatedSteps] = useState(new Array(6).fill(false));
  const [showErrors, setShowErrors] = useState(false);
 
  // states for when API is called to determine result of provisioning the org.
  const [provisioningOrg, setProvisioningOrg] = useState(false);
  const [orgSuccess, setOrgSuccess] = useState(undefined); // org successfully created, init undefined until api called
  const [orgErrors, setOrgErrors] = useState([]);
  const [userSuccess, setUserSuccess] = useState(undefined); // user successfully created
  const [userErrors, setUserErrors] = useState([]);
  const [featureSuccess, setFeatureSuccess] = useState(undefined);
  const [snapPackSuccess, setSnapPackSuccess] = useState(undefined);
  const [nodeSuccess, setNodeSuccess] = useState(undefined);
  const [orgName, setOrgName] = useState(null);
  const [orgId, setOrgId] = useState(null);
 
  // set the validated step at index to true.
  const updateValidatedStep = (index) => {
    const _validatedSteps = [...validatedSteps];
    _validatedSteps[index] = true;
    setValidatedSteps(_validatedSteps);
  };
 
  const validateEmail = email => String(email)
    .toLowerCase()
    .match(
      // eslint-disable-next-line max-len
      /^(([^<>()[\]\\.,;:\s@"]+(\.[^<>()[\]\\.,;:\s@"]+)*)|(".+"))@((\[[0-9]{1,3}\.[0-9]{1,3}\.[0-9]{1,3}\.[0-9]{1,3}])|(([a-zA-Z\-0-9]+\.)+[a-zA-Z]{2,}))$/,
    );
 
  const isValidJson = (str) => {
    try {
      JSON.parse(str);
    } catch (e) {
      return false;
    }
    return true;
  };
 
  const errors = useMemo(() => {
    const _errors = {
      customer_info: {},
      organization: {},
      nodes: {},
      administrator_account: {},
    };
 
    // check for customer_info errors
    if (customer_info.customer_name.value === '') _errors.customer_info.customer_name = true;
    if (customer_info.customer_sfdc_id.value === '') _errors.customer_info.customer_sfdc_id = true;
    if (customer_info.org_type.value === '') _errors.customer_info.org_type = true;
 
    // check for create_org errors
    if (organization.org_name.value === '') _errors.organization.org_name = true;
    if (organization.org_name.value.indexOf(' ') >= 0) {
      _errors.organization.org_name = true;
      _errors.organization.org_name_message = message.DEVOPS.ORG.FORM.ERRORS.ORG_NAME_SPACE;
    }
    if (organization.total_api_limit.value === null) _errors.organization.total_api_limit = true;
    if (organization.concurrent_api_limit.value === null) _errors.organization.concurrent_api_limit = true;
 
    if (administrator_account.user_name.value === '') _errors.administrator_account.user_name = true;
    if (administrator_account.first_name.value === '') _errors.administrator_account.first_name = true;
    if (administrator_account.last_name.value === '') _errors.administrator_account.last_name = true;
    if (administrator_account.user_name.value.length > 0 && !validateEmail(administrator_account.user_name.value)) {
      _errors.administrator_account.user_name = true;
      _errors.administrator_account.user_name_message = message.DEVOPS.ORG.FORM.ERRORS.USER_NAME_EMAIL;
    }
 
    // check for nodes errors
    if (nodes.jcc_nodes.value === null) _errors.nodes.jcc_nodes = true;
    if (nodes.jcc_node_size.value === '' && nodes.jcc_nodes.value > 0) _errors.nodes.jcc_node_size = true;
    if (nodes.feedmaster_nodes.value === null) _errors.nodes.feedmaster_nodes = true;
    if (nodes.feedmaster_node_size.value === '' && nodes.jcc_nodes.value > 0) _errors.nodes.feedmaster_node_size = true;
 
    if (nodes.json_payload.value.length > 0) {
      if (!isValidJson(nodes.json_payload.value)) _errors.nodes.json_payload = true;
    }
 
    return _errors;
  }, [customer_info, organization, administrator_account, nodes]);
 
  const validatePage = (step) => {
    switch (step) {
      case 0:
        return Object.keys(errors.customer_info).length === 0;
      case 1:
        return Object.keys(errors.organization).length === 0;
      case 2:
        return true;
      case 3:
        return true;
      case 4:
        return Object.keys(errors.nodes).length === 0;
      case 5:
        return true;
      default:
        return true;
    }
  };
 
  const handleClickNext = () => {
    const is_valid = validatePage(activeStep);
    if (is_valid) {
      setShowErrors(false);
      updateValidatedStep(activeStep);
      setActiveStep(activeStep + 1);
    } else {
      setShowErrors(true);
    }
  };
 
  // handle clicking a step from the progress bar.
  // should only be able to click if validated already.
  const handleClickStep = (index) => {
    // if we are in the results page, should not be able to click back
    if (activeStep === 6) return;
    if (index > activeStep && !validatePage(activeStep)) return;
    if (validatedSteps[index] || (index > 0 && validatedSteps[index - 1])) {
      setActiveStep(index);
    }
  };
 
  // format org data to post to API.
  const formatOrgData = () => ({
    asset_type: 'Org',
    metadata: {
      aws_region: {
        region: organization.aws_region.value,
        name: organization.aws_region.name,
      },
      requestOrigin: 'ui',
    },
    path: organization.org_name.value,
  });
 
  // format admin data to post to API.
  const formatAdminData = () => ({
    username: administrator_account.user_name.value,
    first_name: administrator_account.first_name.value,
    last_name: administrator_account.last_name.value,
    administrator: true,
  });
 
  // format org limit to post to API.
  const formatOrgLimitData = () => ({
    limit_value: parseInt(organization.user_limit.value, 10) || 10,
  });
 
  // format customer data to post to API.
  const formatCustomerMetadata = () => ({
    Customer: customer_info.customer_name.value,
    Customer_ID: customer_info.customer_sfdc_id.value,
    Org_Type: customer_info.org_type.value,
  });
 
  // format meter data to post to API.
  const formatMeterData = () => ({
    calls_per_day: organization.total_api_limit.value,
    concurrent_calls: organization.concurrent_api_limit.value,
  });
 
  // format feature data to post to API.
  const formatFeatureData = () => {
    const featureData = {};
    Object.keys(featuresMap).map(feature => (featureData[featuresMap[feature].key] = featuresMap[feature].value));
    Object.keys(appSubMap).map(appSub => (featureData[appSubMap[appSub].key] = appSubMap[appSub].value));
    return featureData;
  };
 
  const updated_snapPackList = snapPackList
    .map(current_data => ({
      snap_pack_label: current_data.snap_pack_label, is_sub: current_data.is_sub,
    }));
 
  // format snap subscriptions data to post to API.
  const formatSnapSubscriptions = () => (
    updated_snapPackList
  );
 
  // format node data to post to API.
  const formatNodeData = () => {
    const nodeData = {
      regular_nodes: nodes.jcc_nodes.value,
      feedmaster_nodes: nodes.feedmaster_nodes.value,
      node_config: {
        node_size_jcc: nodes.jcc_node_size.value,
        node_size_fm: nodes.feedmaster_node_size.value,
        aws_region: nodes.aws_region.value,
      },
    };
    if (isValidJson(nodes.json_payload.value)) {
      nodeData.node_config.data_plane = JSON.parse(nodes.json_payload.value);
    }
    return nodeData;
  };
 
  // format org type data to post to API.
  const formatOrgType = () => ({
    org_type: customer_info.pricing.value.toLowerCase(),
  });
 
  // format org default app data to post to API.
  const formatOrgDefaultApp = () => ({
    app_access: {
      default_app: IIP_APP_KEY,
    },
  });
 
  // handle submitting form
  const handleSubmit = () => {
    setActiveStep(6);
    setProvisioningOrg(true);
 
    if (!orgSuccess) {
      if (!userSuccess) {
        let _featureSuccess,
          _snapPackSuccess,
          _nodeSuccess;
 
        const payloadData = {
          create_org: formatOrgData(),
          create_admin: [formatAdminData()],
          post_org_limit: formatOrgLimitData(),
          post_customer_metadata: formatCustomerMetadata(),
          post_meter_data: formatMeterData(),
          post_features: formatFeatureData(),
          post_snap_subscriptions: formatSnapSubscriptions(),
          configure_nodedata: formatNodeData(),
          configure_org_type: formatOrgType(),
          configure_org_default_app: formatOrgDefaultApp(),
        };
 
        provisioningWrapper(payloadData)
          .then((subRes) => {
            const wrapper_response = subRes.data.response_map.wrapper_response;
            const org = subRes.data.response_map.org;
 
            // check if create_org succeeded
            if (wrapper_response.create_org.status_code === 200) {
              setOrgSuccess(true);
              setOrgName(org.name);
              setOrgId(org.snode_id);
 
              // check if create_admin succeeded
              if (wrapper_response.create_admin.status_code === 200) {
                setUserSuccess(true);
              } else {
                setUserSuccess(false);
                setUserErrors(wrapper_response.create_admin.error_msg);
              }
              // if create_org not succeeded
            } else {
              setOrgSuccess(false);
              setOrgErrors(wrapper_response.create_org.error_msg);
              setUserSuccess(false);
              setUserErrors(wrapper_response.create_admin.error_msg);
            }
 
            setFeatureSuccess(wrapper_response.post_features.status_code);
            _featureSuccess = wrapper_response.post_features.status_code;
            setSnapPackSuccess(wrapper_response.post_snap_subscriptions.status_code);
            _snapPackSuccess = wrapper_response.post_snap_subscriptions.status_code;
            setNodeSuccess(wrapper_response.configure_nodedata.status_code);
            _nodeSuccess = wrapper_response.configure_nodedata.status_code;
 
            // check if post_features succeeded
            if (_featureSuccess >= 400) {
              setFeatureSuccess(false);
            } else {
              setFeatureSuccess(true);
            }
 
            // check if post_snap_subscriptions succeeded
            if (_snapPackSuccess >= 400) {
              setSnapPackSuccess(false);
            } else {
              setSnapPackSuccess(true);
            }
 
            // check if configure_nodedata succeeded
            if (_nodeSuccess >= 400) {
              setNodeSuccess(false);
            } else {
              setNodeSuccess(true);
            }
            setProvisioningOrg(false);
          },
          ).catch((err) => {
            const wrapper_response = err.response.data.response_map.wrapper_response;
            setUserSuccess(false);
            setOrgSuccess(false);
            setOrgErrors([...orgErrors, wrapper_response.create_org.error_msg]);
            setFeatureSuccess(false);
            setSnapPackSuccess(false);
            setNodeSuccess(false);
            setProvisioningOrg(false);
          });
      }
    }
  };
 
  return (
    <div className="sl-devops-provision-org-modal">
      <div className="sl-devlops-provision-org-content">
        <div className="sl-devops-provision-org-modal-header">
          <Title>Create New Organization</Title>
          <div className="sl-icon close-icon" onClick={provisioningOrg ? null : onClose} style={{ marginLeft: 15 }} />
        </div>
        <div className="sl-devops-provision-org-modal-body">
          <div className="form-steps">
            {steps.map((step, index) => (
              <div
                className={cn('provision-step',
                  { active: activeStep === index, filled: validatedSteps[index] && activeStep !== index })
                }
                onClick={() => handleClickStep(index)}
                key={index}
              >
                {step}
              </div>
            ))}
          </div>
 
          {activeStep === 0 && <CustomerInfo errors={errors} showErrors={showErrors} />}
          {activeStep === 1 && <CreateOrg errors={errors} showErrors={showErrors} />}
          {activeStep === 2 && <Features />}
          {activeStep === 3 && <SnapPacks />}
          {activeStep === 4 && <Nodes errors={errors} showErrors={showErrors} />}
          {activeStep === 5 && <Confirmation />}
          {activeStep === 6 &&
            <Results
              orgId={orgId}
              orgName={orgName}
              orgSuccess={orgSuccess}
              orgErrors={orgErrors}
              userSuccess={userSuccess}
              userErrors={userErrors}
              featureSuccess={featureSuccess}
              snapPackSuccess={snapPackSuccess}
              nodeSuccess={nodeSuccess}
            />
          }
        </div>
      </div>
      <div className={cn('sl-devops-provision-org-modal-footer', { withBackButton: activeStep > 0 && activeStep < 6 })}>
        {activeStep > 0 && activeStep < 6 &&
          <Button
            label="Back"
            secondary
            onClick={() => setActiveStep(activeStep - 1)}
            qaId="provision-org-back-button"
          />
        }
        <div>
          {activeStep !== 6 && <Button label="Cancel" secondary onClick={onClose} qaId="provision-org-cancel-button" />}
          {activeStep === 5 && <Button label="Submit" onClick={handleSubmit} qaId="provision-org-submit-button" />}
          {activeStep < 5 && <Button label="Next" onClick={handleClickNext} qaId="provision-org-next-step-button" />}
          {activeStep === 6 &&
            <Button
              label={provisioningOrg ? 'Creating org...' : 'Close'}
              disabled={provisioningOrg}
              onClick={onClose}
              qaId="provision-org-close-button"
            />
          }
        </div>
      </div>
    </div>
  );
};
 
export default ProvisionOrgModal;
