import React, { useState } from 'react';
import { useHistory } from 'react-router-dom';
import message from '@/locales';
import { useMutateEditUser } from '@/hooks/user';
import { useSessionContext, useBusyContext } from '@/hooks';
import { useSAML } from '@/hooks/sso';
import Page from '@/common/page';
import TextField from 'SL_UI/v3/textField';
import Button from 'SL_UI/v3/button';
import { isEqual, isEmpty } from 'lodash/lang';
import { get } from 'lodash/object';
 
import DeleteUserModal from '../../deleteUserModal';
import ErrorModal from '../../errorModal';
import RoleSection from '../roleSection';
import AppAccessSection from '../appAccessSection';
import DetailsFooter from '../../detailsFooter';
import PasswordAuthSection from '../passwordAuthSection';
 
const { MANAGER: { USERS_AND_GROUPS: { USERS } } } = message;
 
const UserDetails = ({ user, admins }) => {
  const history = useHistory();
  const { activeOrg } = useSessionContext();
  const { busy, unBusy } = useBusyContext();
 
  const { data: saml } = useSAML();
 
  const { mutate: mutateUser } = useMutateEditUser(user.username, false);
 
  const isUserLoginAllowed = user.disallowed_auth ? user.disallowed_auth.indexOf('basic') === -1 : true;
 
  const [firstname, setFirstname] = useState(user.first_name);
  const [lastname, setLastname] = useState(user.last_name);
  const [uiAccess, setUiAccess] = useState(user.ui_access);
  const [appAccess, setAppAccess] = useState(get(user, ['app_access', activeOrg.name], {}));
  const [isAdmin, setIsAdmin] = useState(!!admins.includes(user.username));
  const [isPwAuth, setIsPwAuth] = useState(isUserLoginAllowed);
  const [payload, setPayload] = useState({});
  const [deleteOpen, setDeleteOpen] = useState(false);
  const [errorOpen, setErrorOpen] = useState(false);
 
  const backToList = () => history.push('/manager/users');
 
  const onCancel = () => {
    setPayload({});
    setFirstname(user.first_name);
    setLastname(user.last_name);
    setUiAccess(user.ui_access);
    setAppAccess(user.app_access[activeOrg.name]);
    setIsAdmin(!!admins.includes(user.username));
  };
 
  const onUIAccessChange = (_uiAccess) => {
    setUiAccess(_uiAccess);
    setPayload(prev => ({ ...prev, ui_access: _uiAccess }));
  };
 
  const onAppCheck = (checked, app) => {
    const newAppAccess = {
      ...appAccess,
      [app]: checked,
    };
    if (Object.values(newAppAccess).every(v => v === false)) { // user need at least on app access
      return;
    }
    setAppAccess(newAppAccess);
  };
 
  const onChangePwAuth = () => {
    const newIsPwAuth = !isPwAuth;
    let disallowed_auth = [];
    Eif (!newIsPwAuth) {
      disallowed_auth = ['basic'];
    }
    setIsPwAuth(newIsPwAuth);
    setPayload(prev => ({ ...prev, disallowed_auth }));
  };
 
  const onDeleteDone = () => {
    setDeleteOpen(false);
    backToList();
  };
 
  const isDirty = isEmpty(payload) ? false : !Object.entries(payload).every(([key, value]) => value === user[key]);
  const isAppAccessDirty = !isEqual(appAccess, get(user, ['app_access', activeOrg.name], {}));
  const isRoleDirty = (isAdmin && !admins.includes(user.username)) || (!isAdmin && admins.includes(user.username));
 
  const onSave = () => {
    let newAdmins = null;
    if (isRoleDirty) {
      if (isAdmin) {
        newAdmins = [...admins, user.username];
      } else {
        newAdmins = [...admins];
        newAdmins.splice(admins.indexOf(user.username), 1);
      }
    }
 
    if (isDirty || isAppAccessDirty || isRoleDirty) {
      mutateUser({
        user: isDirty ? payload : null,
        appAccess: isAppAccessDirty ? {
          org_path: activeOrg.name,
          users: [{
            app_access: appAccess,
            username: user.username,
          }],
          send_email: true,
        } : null,
        role: newAdmins,
        _onMutate: () => busy('editUser'),
        _onSettled: (err) => {
          unBusy('editUser');
          if (err) {
            const errStatus = get(err, 'response.status');
            if (errStatus === 403) {
              setErrorOpen(true);
            }
          } else {
            backToList();
          }
        },
      });
    }
  };
 
  return (
    <React.Fragment>
      <Page.Header backTo="/manager/users" backToAlt="Back to user list">
        <div className="title">
          {`${user.first_name || ''} ${user.last_name || ''}`}
        </div>
      </Page.Header>
      <Page.Body className="manager-details">
        <div className="manager-details__column">
          <div className="container">
            <TextField
              qaId="manager-users-details-firstname"
              label={USERS.FIRST_NAME}
              value={firstname}
              onChange={(e) => {
                setFirstname(e.target.value);
                setPayload(prev => ({ ...prev, first_name: e.target.value }));
              }}
            />
            <TextField
              qaId="manager-users-details-lastname"
              label={USERS.LAST_NAME}
              value={lastname}
              onChange={(e) => {
                setLastname(e.target.value);
                setPayload(prev => ({ ...prev, last_name: e.target.value }));
              }}
            />
            <TextField
              qaId="manager-users-details-email"
              label={USERS.EMAIL}
              value={user.username}
              onChange={() => {}}
              disabled
            />
            <AppAccessSection
              uiAccess={uiAccess}
              appAccess={appAccess}
              onAppCheck={onAppCheck}
              onUIAccessChange={onUIAccessChange}
            />
            <RoleSection isAdmin={isAdmin} setIsAdmin={setIsAdmin} />
            { saml && saml.idp_entity_id &&
              <PasswordAuthSection isEnabled={isPwAuth} onChange={onChangePwAuth} />
            }
          </div>
          <div className="manager-details-split" />
          <div className="manager-details-section">
            <Button
              qaId="manager-users-details-delete"
              label={USERS.DELETE_USER}
              onClick={() => setDeleteOpen(true)}
              redHollow
            />
            {USERS.DELETE_WARNING}
          </div>
        </div>
        <DetailsFooter
          qaIdPrefix="manager-users-details"
          onSave={onSave}
          canSave={isDirty || isRoleDirty || isAppAccessDirty}
          onCancel={onCancel}
        />
      </Page.Body>
      <DeleteUserModal
        open={deleteOpen}
        selected={[user.username]}
        onDone={onDeleteDone}
        onClose={() => setDeleteOpen(false)}
      />
      <ErrorModal
        open={errorOpen}
        onClose={() => setErrorOpen(false)}
      />
    </React.Fragment>
  );
};
 
export default UserDetails;
