import React, { useState, useRef } from 'react';
import OnBody from 'SL_UI/v3/utils/onBody';
import { DropdownList } from 'SL_UI/v3/dropdown';
import { useClickAway } from '@/hooks';
import IconButton from 'SL_UI/v3/iconButton';
import KebobSrc from 'SL_UI/v3/icons/kebob.svg';
import KebobHoverSrc from 'SL_UI/v3/icons/kebob-hover.svg';
import KebobDisabledSrc from 'SL_UI/v3/icons/kebob-disabled.svg';
import message from '@/locales';
import { getDropDownPosition } from '@/common/utils';
import RenameConnectionModal from './actions/renameConnection';
import DeleteConnectionModal from './actions/deleteConnection';
 
const { MANAGER: { ENVIRONMENT_SETTINGS: { SNAPLEXES } } } = message;
const { COMMON } = message;
const deletedLabel = <div className="deleted-connections">{SNAPLEXES.ACTIONS.DELETE_CONNECTION}</div>;
 
function setItems(connectionData) {
  return [
    {
      key: 'submit-service-url',
      value: 'submit-service-url',
      label: SNAPLEXES.ACTIONS.SUBMIT_SERVICE_URL,
      labelText: SNAPLEXES.ACTIONS.SUBMIT_SERVICE_URL,
      disabled: connectionData.status === SNAPLEXES.STATUS_LIST.INITIALIZING
        || connectionData.status === SNAPLEXES.STATUS_LIST.PENDING_APPROVAL
        || connectionData.status === SNAPLEXES.STATUS_LIST.ACTIVE
        || connectionData.status === SNAPLEXES.STATUS_LIST.DELETED,
    },
    {
      key: 'view-status',
      value: 'view-status',
      label: SNAPLEXES.ACTIONS.VIEW_STATUS,
      labelText: SNAPLEXES.ACTIONS.VIEW_STATUS,
      disabled: connectionData.status === SNAPLEXES.STATUS_LIST.NOT_SUBMITTED
        || connectionData.status === SNAPLEXES.STATUS_LIST.DELETED,
    },
    {
      key: 'rename-connection',
      value: 'rename-connection',
      label: SNAPLEXES.ACTIONS.RENAME_CONNECTION,
      labelText: SNAPLEXES.ACTIONS.RENAME_CONNECTION,
      disabled: connectionData.status === SNAPLEXES.STATUS_LIST.DELETED },
    {
      key: 'delete-connection',
      value: 'delete-connection',
      label: deletedLabel,
      labelText: SNAPLEXES.ACTIONS.DELETE_CONNECTION,
      disabled: connectionData.status === SNAPLEXES.STATUS_LIST.DELETED },
  ];
}
 
const SnaplexActionMenu = ({ selected, qaId, setCurrentStep, setAddNewConnectionOpen, setStatusState,
  setShowButton, setSubmitButtonLabel, setConnectionId, setDetails }) => {
  const [renameConnectionOpen, setRenameConnectionOpen] = useState(false);
  const [deleteConnectionOpen, setDeleteConnectionOpen] = useState(false);
  const onSelect = (value) => {
    setConnectionId(selected.id);
    switch (value.label) {
      case SNAPLEXES.ACTIONS.SUBMIT_SERVICE_URL:
        setShowButton(true);
        setStatusState(selected.status);
        setCurrentStep(1);
        setAddNewConnectionOpen(true);
        break;
      case SNAPLEXES.ACTIONS.VIEW_STATUS:
 
        if (selected.status === SNAPLEXES.STATUS_LIST.APPROVED ||
            selected.status === SNAPLEXES.STATUS_LIST.ACTIVE) {
          setDetails({ dns: selected.dns_entries, name: selected.name });
          setShowButton(true);
          setSubmitButtonLabel(COMMON.DONE);
        }
 
        if (selected.status === SNAPLEXES.STATUS_LIST.NOT_SUBMITTED) {
          setDetails({});
          setShowButton(true);
          setSubmitButtonLabel(COMMON.NEXT);
        }
 
        if (selected.status === SNAPLEXES.STATUS_LIST.REJECTED ||
            selected.status === SNAPLEXES.STATUS_LIST.PENDING_APPROVAL ||
            selected.status === SNAPLEXES.STATUS_LIST.INITIALIZING ||
            selected.status === SNAPLEXES.STATUS_LIST.FAILED) {
          setDetails({});
          setShowButton(false);
        }
        setStatusState(selected.status);
        setCurrentStep(2);
        setAddNewConnectionOpen(true);
        break;
      case SNAPLEXES.ACTIONS.RENAME_CONNECTION:
        setRenameConnectionOpen(true);
        break;
      case deletedLabel:
        setDeleteConnectionOpen(true);
        break;
      default:
        break;
    }
  };
  const [show, setShow] = useState(false);
  const dropdownRef = useRef(null);
  const dropdownListRef = useRef();
  const iconRef = useRef(null);
 
  const showDropdownList = () => {
    setShow(!show);
  };
 
  const handleSelect = (value) => {
    setShow(false);
    onSelect(value);
  };
 
  useClickAway(() => setShow(false), dropdownRef, dropdownListRef);
 
  return (
    <div>
      <IconButton
        ref={iconRef}
        qaId={`${qaId}`}
        iconSrc={KebobSrc}
        hoverIconSrc={KebobHoverSrc}
        disabledIconSrc={KebobDisabledSrc}
        onClick={showDropdownList}
        noBackground
        alt="Open action menu"
        disabled={selected.status === SNAPLEXES.STATUS_LIST.DELETED}
      />
      {show ?
        <OnBody>
          <DropdownList
            ref={dropdownListRef}
            items={setItems(selected)}
            style={getDropDownPosition(iconRef)}
            selected={selected}
            onSelect={handleSelect}
            qaId={`${qaId}-list`}
          />
        </OnBody> : null
      }
      {renameConnectionOpen && (
        <RenameConnectionModal
          renameConnectionOpen={renameConnectionOpen}
          setRenameConnectionOpen={setRenameConnectionOpen}
          connection={selected}
        />
      )}
      {deleteConnectionOpen && (
        <DeleteConnectionModal
          deleteConnectionOpen={deleteConnectionOpen}
          setDeleteConnectionOpen={setDeleteConnectionOpen}
          connection={selected}
        />
      )}
    </div>
  );
};
export default SnaplexActionMenu;
