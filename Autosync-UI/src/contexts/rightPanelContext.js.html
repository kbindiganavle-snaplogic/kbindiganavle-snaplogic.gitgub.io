import React, { useCallback } from 'react';
 
const actions = {
  OPEN_PANEL: 'OPEN_PANEL',
  CLOSE_PANEL: 'CLOSE_PANEL',
  CLOSE_ALL_PANEL: 'CLOSE_ALL_PANEL',
  SET_WIDTH: 'SET_WIDTH',
  SET_MIN_WIDTH: 'SET_MIN_WIDTH',
};
 
const initialState = {
  list: [],
  open: false,
  mask: false,
  width: 0,
  minWidth: 1000,
};
 
function reducer(state, action) {
  let newList;
  switch (action.type) {
    case actions.OPEN_PANEL:
      newList = [...state.list, action.value];
      return {
        ...state,
        open: true,
        list: newList,
      };
    case actions.CLOSE_PANEL: {
      const toClose = action.value === undefined ? state.list.length - 1 : action.value;
      newList = state.list.slice(0, toClose);
      return { ...state, open: !!newList.length, list: newList };
    }
    case actions.CLOSE_ALL_PANEL:
      return { ...state, list: [], open: false };
    case actions.SET_WIDTH:
      return { ...state, width: action.width };
    case actions.SET_MIN_WIDTH:
      return { ...state, minWidth: action.minWidth };
    default:
      throw new Error();
  }
}
 
const RightPanelContext = React.createContext();
 
const RightPanelProvider = ({ children }) => {
  const [state, dispatch] = React.useReducer(reducer, initialState);
 
  const openPanel = useCallback(({ content, mask = false, style = {}, resizable = false, width, ...rest }) => {
    dispatch({ type: actions.OPEN_PANEL, value: { content, mask, style, resizable, width, ...rest } });
    return state.list.length - 1;
  }, [state]);
 
  const closePanel = useCallback(() => dispatch({ type: actions.CLOSE_PANEL }), []);
 
  const closeAllPanel = useCallback(() => dispatch({ type: actions.CLOSE_ALL_PANEL }), []);
 
  const setWidth = useCallback(width => dispatch({ type: actions.SET_WIDTH, width }), []);
 
  const setMinWidth = useCallback(minWidth => dispatch({ type: actions.SET_MIN_WIDTH, minWidth }), []);
 
  const activePanel = state.list[state.list.length - 1];
 
  const onResize = useCallback((event, { size }) => {
    if (
      (size.width <= state.minWidth && size.width < state.width) ||
      (size.width >= window.innerWidth * 0.8 && size.width > state.width)) {
      return;
    }
    setWidth(size.width);
  }, [state, setWidth]);
 
  const value = {
    list: state.list,
    open: state.open,
    width: state.width,
    setWidth,
    minWidth: state.minWidth,
    setMinWidth,
    openPanel,
    closePanel,
    closeAllPanel,
    activePanel,
    onResize,
    isMasked: state.open && state.width && activePanel.mask,
  };
 
  return (<RightPanelContext.Provider value={value}>
    { children }
  </RightPanelContext.Provider>);
};
 
export { RightPanelContext, RightPanelProvider };
