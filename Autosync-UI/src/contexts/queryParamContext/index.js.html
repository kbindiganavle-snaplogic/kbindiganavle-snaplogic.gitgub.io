import React, { createContext, useEffect, useMemo, useReducer, useCallback, useContext } from 'react';
import { useHistory, useLocation } from 'react-router-dom';
import { useQueryParams } from '@/hooks';
import { reducer, initialState, init, setQueryParam } from './reducer';
import parseQueryValue from './parseQueryValue';
 
const QueryParamContext = createContext();
 
const QueryParamProvider = ({
  children,
}) => {
  const history = useHistory();
  const query = useQueryParams();
  const location = useLocation();
 
  const [state, dispatch] = useReducer(reducer, initialState);
 
  // Whether or not we should sync query params
  const isEnabled = useMemo(() =>
    !!state[location.pathname],
  [state, location],
  );
 
  /**
  * programmatically consume query params onClick
  * used on <Link /> components
  * pages must call this once onload to enable it
  *
  * @param keysConfig - object describing which keys are allowed and what their values can be:
  * {
  *   someKey: {}, // all values are allowed
  *   someOtherKey: { allowedValues: ['hello', 'world'] } // only these are allowed
  * }
  * @param _query - URLSearchParams (optional)
  *
  * @return null
  */
  const consumeQueryParams = useCallback(
    (keysConfig = {}, _query = query) => {
      const queryParams = [..._query.keys()].reduce((acc, key) => {
        const config = keysConfig[key];
 
        if (config) {
          const value = parseQueryValue(_query.get(key));
 
          Eif (
            // any value is allowed
            !config.allowedValues ||
            // value is valid
            config.allowedValues.includes(value)
          ) {
            acc[key] = value;
          }
        }
        return acc;
      }, {});
      dispatch(init(queryParams, location.pathname));
    },
    [query, location],
  );
 
  const valuesAtPathname = useMemo(
    () => state[location.pathname]?.queryParams || {},
    [state, location],
  );
 
  const onSetQueryParam = useCallback(
    (key, value) => {
      dispatch(setQueryParam(key, value, location.pathname));
    },
    [location],
  );
 
  const hasConsumedQueryParams = useMemo(
    () => !!state[location.pathname]?.hasConsumedQueryParams,
    [state, location],
  );
 
  const queryString = useMemo(() => {
    if (!isEnabled) return '';
 
    const queryParams = new URLSearchParams();
    Object.entries(valuesAtPathname).forEach(([key, value]) => {
      queryParams.append(key, JSON.stringify(value));
    });
    return queryParams.toString();
  }, [isEnabled, valuesAtPathname]);
 
  // If this page is enabled, push state changes to the url
  useEffect(() => {
    if (isEnabled) {
      history.push({
        search: `?${queryString}`,
      });
    }
  }, [isEnabled, history, queryString]);
 
  const value = {
    state: valuesAtPathname,
    hasConsumedQueryParams,
    consumeQueryParams,
    onSetQueryParam,
    queryString,
  };
 
  return (
    <QueryParamContext.Provider value={value}>
      {children}
    </QueryParamContext.Provider>
  );
};
 
const useConfigQueryParams = (keysConfig) => {
  const { hasConsumedQueryParams, consumeQueryParams } = useContext(QueryParamContext);
  useEffect(() => {
    if (!hasConsumedQueryParams && keysConfig) {
      consumeQueryParams(keysConfig);
    }
  }, [hasConsumedQueryParams, consumeQueryParams, keysConfig]);
};
 
 
export { QueryParamContext, QueryParamProvider, useConfigQueryParams };
