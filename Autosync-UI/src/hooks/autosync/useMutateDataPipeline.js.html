import { useCallback, useContext, useEffect, useState } from 'react';
import message from '@/locales';
import { NotifyContext } from 'GlobalCommon/lib/notifyContext';
import { useMutation, useQueryClient } from '@tanstack/react-query';
import axios from 'axios';
import { setAutosyncEmptyValues } from '@/common/utils';
import * as queryKeys from '@/common/queryKeys';
import { parseErrorResponse } from '@/errorHandling';
import { get } from 'lodash/object';
import { SessionContext } from 'GlobalCommon/lib/sessionContext';
import useSchemas from './useSchemas';
import useMutateRunDataPipeline from './useMutateRunDataPipeline';
 
const { ON_CREATE, ON_EDIT, ON_SAVE_AND_RUN, ON_RUN } = message.DESIGN.DATA_PIPELINE;
 
const isFile = ({ _content, _path }) => !!_content && !!_path;
 
const useMutateDataPipeline = (instance_id = '', silent = true) => {
  const { activeOrg } = useContext(SessionContext);
  const { info, success, error } = useContext(NotifyContext);
  const queryClient = useQueryClient();
  const [editId, setEditId] = useState(instance_id);
  const multipleKey = queryKeys.dataPipelines(activeOrg);
  const singleKey = queryKeys.dataPipeline(activeOrg, editId);
  const { data: endpointSchemas } = useSchemas();
  const { mutate: runAutosync } = useMutateRunDataPipeline();
 
  const combineWithDefaultEndpointObject = useCallback(
    (instance) => {
      const combineEndpoint = (endpoint) => {
        const found = endpointSchemas.find(
          ({ id }) => id === endpoint.schema_id,
        );
 
        if (found) {
          const { class_map } = found;
 
          const defaultObj = Object.values(class_map).reduce(
            setAutosyncEmptyValues,
            {},
          );
 
          return {
            ...defaultObj,
            ...endpoint,
          };
        }
 
        return endpoint;
      };
 
      const removeFileProperties = (endpoint) => {
        if (endpoint.schema_selection && endpoint.schema_selection.tables) {
          return {
            ...endpoint,
            schema_selection: {
              ...endpoint.schema_selection,
              tables: endpoint.schema_selection.tables.map(file =>
                Object.entries(file).reduce((acc, [key, value]) => {
                  if (!key.startsWith('_')) {
                    acc = {
                      ...acc,
                      [key]: value,
                    };
                  }
                  return acc;
                }, {}),
              ),
            },
          };
        }
 
        return endpoint;
      };
 
      const isCSV = instance?.sources?.some(({ name }) => name.includes('CSV'));
 
      return {
        ...instance,
        sources: instance.sources.map(combineEndpoint).map(removeFileProperties),
        targets: instance.targets.map(combineEndpoint).map(removeFileProperties),
        schedule: isCSV ? {} : instance.schedule,
      };
    },
    [endpointSchemas],
  );
 
  useEffect(() => {
    setEditId(instance_id);
  }, [instance_id]);
 
  const updateMultipleCache = useCallback(
    (prev, newDataPipeline) => {
      const _editId = editId || newDataPipeline.instance_id;
      // update cache if it exists
      if (prev) {
        let found = false;
        const mapped = prev.map((_item) => {
          if (_item.instance_id === _editId) {
            found = true;
            const tmpItem = { ..._item };
            delete tmpItem.schedule;
            return {
              ...tmpItem,
              ...newDataPipeline,
              // time_updated: new Date().toISOString().replace('Z', '000+00:00'),
            };
          }
 
          return _item;
        });
        if (found) {
          return mapped;
        }
        mapped.push(newDataPipeline);
        return mapped;
      }
 
      // no cache yet
      // leave undefined
      return prev;
    },
    [editId],
  );
 
  const mutateDataPipeline = useCallback(
    async ({
      name,
      path,
      metadata,
      payload,
      schedule,
      sources,
      targets,
      plex_path,
      shouldSaveAndRun = false,
    }) => {
      // is editing
      if (editId) {
        const newInstance = get(payload, 'instance', {});
        // Check if it needs to upload files first
        return new Promise((resolve) => {
          const fileEndpoints = newInstance.sources.filter(e => e.name.includes('CSV'));
          // Not uploading anything
          if (!fileEndpoints.length) {
            resolve([]);
            return;
          }
 
          const files = fileEndpoints.reduce(
            (outerAcc, { schema_id, ...rest }) => {
              const { class_map } = endpointSchemas.find(
                ({ id }) => id === schema_id,
              );
              const found = Object.values(class_map).reduce((acc, { key }) => {
                const value = get(rest, key);
                const isArray = Array.isArray(value);
                // is file
                if (isArray && value.every(isFile)) {
                  return value;
                }
                return acc;
              }, []);
 
              return [...outerAcc, ...found];
            },
            [],
          );
 
          Promise.all(
            files.map((file) => {
              const formData = new FormData();
 
              formData.append('_method', 'put');
              const fileObj = new File([file._content], file._path, {
                type: 'text/csv',
              });
              formData.append('file', fileObj);
 
              return axios.post(
                `/api/1/rest/slfs/${newInstance.path}/${encodeURIComponent(file._path)}`,
                formData,
                {
                  headers: {
                    'Content-Type': 'multipart/form-data',
                  },
                },
              );
            }),
          ).then(resolve);
        }).then(uploadedFileResponses =>
          axios
            .put(`/api/2/${activeOrg.org_snode_id}/rest/autosync/${editId}`, {
              ...payload,
              instance: {
                ...combineWithDefaultEndpointObject(newInstance),
                // time_updated: new Date().toISOString().replace('Z', '000+00:00'),
              },
              use_as_endpoint: true,
            })
            .then((res) => {
              const instance = res.data.response_map.instance;
 
              const filesWereUploadedSuccessfully =
                !!uploadedFileResponses.length &&
                uploadedFileResponses.every(
                  r => (r.data.http_status_code = 200),
                );
 
              if (shouldSaveAndRun || filesWereUploadedSuccessfully) {
                runAutosync({
                  dataPipeline: instance,
                  _onError: () => error({ message: String.format(ON_RUN.ERROR, instance?.name) }),
                });
              }
 
              return instance;
            }),
        );
      }
      // is creating
      return axios
        .post(`/api/2/${activeOrg.org_snode_id}/rest/autosync/`, {
          name,
          path,
          metadata,
          plex_path,
        })
        .then(async ({ data }) => {
          const instance = combineWithDefaultEndpointObject({
            ...data.response_map.instance,
            targets,
            sources,
          });
 
          setEditId(instance.instance_id);
 
          // Also needs to edit this pipeline
          if (schedule) {
            return axios
              .put(
                `/api/2/${activeOrg.org_snode_id}/rest/autosync/${instance.instance_id}`,
                // when we first create the pipeline, enabled is false.
                { instance: { ...instance, schedule, enabled: false }, use_as_endpoint: true },
              )
              .then(res => res.data.response_map.instance);
          }
          return instance;
        });
    },
    [activeOrg, editId, combineWithDefaultEndpointObject, endpointSchemas, runAutosync],
  );
 
  const onMutate = useCallback(
    ({ name, payload, shouldSaveAndRun }) => {
      if (editId) {
        if (!silent) {
          if (shouldSaveAndRun) {
            info({ message: `${ON_SAVE_AND_RUN.INFO}: ${payload.instance.name}` });
          } else {
            info({ message: `${ON_EDIT.INFO}: ${payload.instance.name}` });
          }
        }
        // TODO: implement optimistic updates correctly
        // The below code only sort of works. A handler needs
        // to be passed so it navigates forward, too.
 
        // Cancel any outgoing refetches (so they don't overwrite our optimistic update)
        // await queryClient.cancelQueries(multipleKey);
        // await queryClient.cancelQueries(singleKey);
 
        // Snapshot the previous value
        // const previousDataPipelines = queryClient.getQueryData(multipleKey);
        // const previousDataPipeline = queryClient.getQueryData(singleKey);
 
        // Optimistically update to the new value
        // queryClient.setQueryData(
        //   multipleKey,
        //   updateMultipleCache(newDataPipeline),
        // );
        // queryClient.setQueryData(singleKey, prev => ({
        //   ...prev,
        //   ...newDataPipeline,
        // }));
 
        // Return a context with the previous and new todo
        // return { previousDataPipelines, previousDataPipeline, newDataPipeline };
      } else if (!silent) {
        info({ message: `${ON_CREATE.INFO}: ${name}` });
      }
      // Snapshot the previous value
      const previousDataPipelines = queryClient.getQueryData(multipleKey);
      const previousDataPipeline = queryClient.getQueryData(singleKey);
 
      return { previousDataPipelines, previousDataPipeline /* , newDataPipeline */ };
    },
    [editId, silent, multipleKey],
  );
 
  const onSuccess = useCallback(
    (data, { name, payload, updateState, shouldSaveAndRun, _onSuccess }) => {
      const _editId = editId || data.instance_id;
      if (_onSuccess) {
        _onSuccess();
      }
      if (_editId) {
        if (!silent) {
          if (shouldSaveAndRun) {
            success({ message: `${ON_SAVE_AND_RUN.SUCCESS}: ${payload.instance.name}` });
          } else {
            success({ message: `${ON_EDIT.SUCCESS}: ${payload.instance.name}` });
          }
        }
        queryClient.setQueryData(singleKey, (prev) => {
          if (prev) {
            delete prev.schedule;
            return {
              ...prev,
              ...data,
            };
          }
          return data;
        });
        queryClient.setQueryData(multipleKey, prev => updateMultipleCache(prev, data));
        if (updateState) updateState(data);
        const nextRuntimeQuery = queryKeys.nextRuntime(data.snode_id, data.scheduled_task_snid);
        queryClient.invalidateQueries(nextRuntimeQuery);
      } else {
        // Update multiple cache
        queryClient.setQueryData(multipleKey, (prev) => {
          // already cached
          if (prev) {
            return [data, ...prev];
          }
          // not yet cached
          // leave undefined
          return prev;
        });
 
        // Create single cache
        queryClient.setQueryData(singleKey, data);
 
        if (!silent) success({ message: `${ON_CREATE.SUCCESS}: ${name}` });
      }
    },
    [editId, silent, multipleKey],
  );
 
  const onError = useCallback(
    (err, { name, payload, shouldSaveAndRun, _onError }, { previousDataPipeline, previousDataPipelines }) => {
      if (_onError) {
        _onError();
      }
      if (err) {
        const errorList = parseErrorResponse(JSON.stringify(err.response.data));
        if (errorList && errorList[0] && errorList[0].message) {
          errorList.forEach(error);
        }
      }
      // TODO: Integrate with DataDog
      if (editId) {
        if (!silent) {
          if (shouldSaveAndRun) {
            error({ message: `${ON_SAVE_AND_RUN.ERROR}: ${payload.instance.name}` });
          } else {
            error({ message: `${ON_EDIT.ERROR}: ${payload.instance.name}` });
          }
        }
        // queryClient.setQueryData(multipleKey, previousDataPipelines);
        // queryClient.setQueryData(singleKey, previousDataPipeline);
      } else if (!silent) {
        error({ message: `${ON_CREATE.ERROR}: ${name}` });
      }
 
      // revert changes
      queryClient.setQueryData(multipleKey, previousDataPipelines);
      queryClient.setQueryData(singleKey, previousDataPipeline);
    },
    [editId, silent, multipleKey],
  );
 
  return useMutation(mutateDataPipeline, {
    onMutate,
    onSuccess,
    onError,
    onSettled: (data, err, { _onSettled }) => {
      if (_onSettled) {
        _onSettled();
      }
    },
  });
};
 
export default useMutateDataPipeline;
