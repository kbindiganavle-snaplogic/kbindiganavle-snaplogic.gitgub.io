import { useContext } from 'react';
import axios from 'axios';
import { useInfiniteQuery } from '@tanstack/react-query';
import { SessionContext } from 'GlobalCommon/lib/sessionContext';
import { ExecutionContext } from '@/contexts/executionContext';
import * as queryKeys from '@/common/queryKeys';
import { lastPeriodOrDateRange, getDisplayInvoker } from '@/common/utils';
import { MINUTE, SECOND } from '@/hooks/constants';
import { get } from 'lodash';
 
export const fetchRuntimes = ({ queryKey, pageParam = 0 }) => {
  const [
    ,
    org_snode_id,
    perPage,
    searchPayload,
    lastPeriod,
    startDate,
    endDate,
  ] = queryKey;
  const lastUpdated = new Date().valueOf();
 
  const promise = axios
    .post(
      `/api/2/${org_snode_id}/rest/pm/runtime?version=latest`,
      {
        offset: pageParam,
        limit: perPage + 1,
        // start_ts: lastPeriod
        //   ? sub(new Date(), lastPeriod).getTime()
        //   : startDate.getTime(),
        // end_ts: lastPeriod ? new Date().getTime() : endDate.getTime(),
        ...lastPeriodOrDateRange(lastPeriod, startDate, endDate),
        ...searchPayload,
        field_list: ['duration_ms', 'kickoff_timestamp', 'remote_user', 'remote_address'],
      },
    )
    .then(({ data }) => ({
      ...data.response_map,
      entries: get(data, 'response_map.entries', []).map(entry => ({
        ...entry,
        display_invoker: getDisplayInvoker(entry),
      })),
      nextCursor:
        data.response_map.entries.length === perPage + 1
          ? pageParam + perPage
          // nothing more to fetch
          : undefined,
      lastUpdated,
    }));
 
  return promise;
};
 
const useExecutionRuntimes = (searchPayload) => {
  const { activeOrg } = useContext(SessionContext);
  const { startDate, endDate, lastPeriod, perPage } = useContext(ExecutionContext);
  // const { startDate, endDate, lastPeriod } = useContext(ExecutionContext);
  // const perPage = 25;
 
  return useInfiniteQuery(
    // key
    queryKeys.executionRuntimes(
      activeOrg,
      perPage,
      searchPayload,
      lastPeriod,
      startDate,
      endDate,
    ),
    // fetchFn
    fetchRuntimes,
    // config
    {
      enabled: !!activeOrg.org_snode_id,
      staleTime: 2 * MINUTE,
      refetchInterval: 30 * SECOND,
      getNextPageParam: lastPage => lastPage.nextCursor,
    },
  );
};
 
export default useExecutionRuntimes;
