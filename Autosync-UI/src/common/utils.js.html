export function convertAssetTreeData(
  rowTreeData,
  getTreeNodeData = data => ({
    label: data.name,
    value: data.snode_id,
  }),
  filterEmptyAPI = false,
) {
  const convertAssetDataToPathTreeData = function (data) {
      if (!data.snode_id || !data.path || !data.name) {
        return null;
      }
      return getTreeNodeData(data);
    },
    sortTreeData = function (a, b) {
      if (a.label === 'shared') {
        return -1;
      } else if (b.label === 'shared') {
        return 1;
      }
      return a.label.toLowerCase() > b.label.toLowerCase() ? 1 : -1;
    },
    convertList = function (list) {
      const returnList = [];
      if (list && list.length > 0) {
        list.forEach((entry) => {
          const newEntry = convertAssetDataToPathTreeData(entry);
          if (entry.name !== 'shared' && entry.entries && entry.entries.length > 0) {
            newEntry.children = convertList(entry.entries);
          }
          const isEmptyAPI = entry.asset_type === 'Api' && newEntry.children && newEntry.children.length === 0;
          if (newEntry && !(filterEmptyAPI && isEmptyAPI)) {
            returnList.push(newEntry);
          }
        });
      }
      return returnList.sort(sortTreeData);
    },
    returnList = convertList(rowTreeData);
  return returnList;
}

/**
 * converts milliseconds to human readable format
 * i.e. 10230 => "10.23 s"
 * if above 1 second, precision up to 2 decimals
 * if under 1 second, precision up to 3 decimals
 */
export function msToTime(ms) {
  if (!ms || isNaN(ms)) return '-';
  const seconds = (ms / 1000).toFixed(2);
  const minutes = (ms / (1000 * 60)).toFixed(2);
  const hours = (ms / (1000 * 60 * 60)).toFixed(2);
  const days = (ms / (1000 * 60 * 60 * 24)).toFixed(2);
  if (ms < 1000) return `${ms} ms`;
  if (seconds < 60) return `${seconds} s`;
  else if (minutes < 60) return `${minutes} min`;
  else if (hours < 24) return `${hours} hrs`;
  return `${days} days`;
}

export function formatDuration(duration = 0, { inMs = true, colons = true } = {}) {
  const durationInMS = inMs ? duration : duration * 1000;
  const days = Math.floor(durationInMS / DAY);
  let hours;
  let minutes;
  let seconds;
 
  hours = Math.floor((durationInMS - (days * DAY)) / HOUR);
  minutes = Math.floor((durationInMS - ((days * DAY) + (hours * HOUR))) / MINUTE);
  seconds = Math.floor((durationInMS - ((days * DAY) + (hours * HOUR) + (minutes * MINUTE))) / SECOND);
 
  // if number is 0 - 9, make it 00 - 09
  if (colons) {
    hours += days * 24;
    hours = hours < 10 ? `0${hours}` : hours;
    minutes = minutes < 10 ? `0${minutes}` : minutes;
    seconds = seconds < 10 ? `0${seconds}` : seconds;
  }
 
  if (colons) {
    return `${hours}:${minutes}:${seconds}`;
  }
 
  return `${days}d ${hours}h ${minutes}m ${seconds}s`;
}
 
export function formatSize(size, unit) {
  if (size === undefined || size === null) {
    return `0${unit}`;
  }
  if (size > 1000000000) {
    return `${_customPrecision(size / 1000000000)}${message.common.G}${unit}`;
  } else if (size > 1000000) {
    return `${_customPrecision(size / 1000000)}${message.common.M}${unit}`;
  } else if (size > 1000) {
    return `${_customPrecision(size / 1000)}${message.common.K}${unit}`;
  }
  return `${_customPrecision(size)}${unit}`;
}
 
export function formatDate(date, time_first = false) {
  try {
    if (typeof date === 'string' || date instanceof String) {
      date = new Date(date);
    }
    if (time_first) {
      return format(date, 'hh:mm:ss a, MMM dd, yyyy');
    }
    return format(date, 'MMM dd, yyyy hh:mm:ss a');
  } catch (err) {
    return 'unknown';
  }
}

/**
 * arrayIgnoreOrderEquals
 * This helper function compares two arrays to determine if they are equal
 * and ignores the order of the elements.
 * @param {*} array1 - first array to compare
 * @param {*} array2 - second array to compare
 * @returns bool - true if equal independent of order.
 */
export const arrayIgnoreOrderEquals = (array1, array2) => {
  if (!isArray(array1) || !isArray(array2)) {
    return undefined;
  }
  if (array1.length !== array2.length) {
    return false;
  }
  const sortedArray1 = sortBy(array1);
  const sortedArray2 = sortBy(array2);
  return isEqual(sortedArray1, sortedArray2);
};
