export const getViews = (viewMap = {}, viewType, snapInstance) => {
  const views = [];
 
  Object.keys(viewMap).forEach((viewId) => {
    if (viewMap[viewId]) {
      const viewStats = viewMap[viewId];
 
      if (viewType === 'error_views' && viewStats.documents_count === 0
        && viewStats.rate_per_second === 0 && viewStats.seconds_io === 0) {
        return;
      }
 
      const bytes = viewStats.byte_count || 0,
        docs = viewStats.documents_count || 0,
        seconds = (viewStats.seconds_io > 1 ?
          `${viewStats.seconds_io.toFixed(0)} ${message.COMMON.SEC}` :
          `${Math.ceil(viewStats.seconds_io * 1000)} ${message.COMMON.MS}`),
        rate_per_second = viewStats.rate_per_second ? formatSize(viewStats.rate_per_second, '') : 0;
 
      let bytes_per_second = viewStats.bytes_per_second ?
        formatSize(viewStats.bytes_per_second, message.COMMON.B) : '';
 
      if (bytes_per_second) {
        bytes_per_second += message.RUNINFO.PER_SEC;
      }
 
      views.push({
        view_name: getViewName(snapInstance, viewId),
        view_type: viewType,
        seconds,
        byte_count: bytes,
        doc_count: docs,
        rate_per_second,
        bytes_per_second,
      });
    }
  });
 
  return views;
};
 
export function getViewName(snapInstance, viewID) {
  let IOType;
 
  if (snapInstance) {
    if (snapInstance.get_view_label) {
      return snapInstance.get_view_label(viewID);
    }
    if (viewID.match(/out.*/)) {
      IOType = 'output';
    } else if (viewID.match(/in.*/)) {
      IOType = 'input';
    } else if (viewID.match(/err.*/)) {
      IOType = 'error';
    } else {
      return viewID;
    }
    try {
      if (snapInstance.property_map) {
        return snapInstance.property_map[IOType][viewID].label.value;
      }
    } catch (err) {
      // for infer snap, there might be some
      // invalid view types. That's expected,
      // because they don't use formal snap schema.
      // swallow
      return viewID;
    }
  }
  return viewID;
}
 
/*
 * Adds '.0' if number is less than 10 else returns as is
 */
function _customPrecision(num) {
  return num < 10 ? num.toFixed(1) : num.toFixed(0);
}
 
/*
 * converts to Giga, mega or kilo and adds unit to the size
 */
function formatSize(size, unit) {
  if (size === undefined || size === null) {
    return `0${unit}`;
  }
  if (size > 1000000000) {
    return `${_customPrecision(size / 1000000000)}${message.COMMON.G}${unit}`;
  } else if (size > 1000000) {
    return `${_customPrecision(size / 1000000)}${message.COMMON.M}${unit}`;
  } else if (size > 1000) {
    return `${_customPrecision(size / 1000)}${message.COMMON.K}${unit}`;
  }
  return `${_customPrecision(size)}${unit}`;
}
 
 
const getDuration = (snapStats) => {
  if (!snapStats.end_time_ms) return '-';
  const totalTime = snapStats.end_time_ms - snapStats.start_time_ms;
  return msToTime(totalTime);
};
 
const RenderViewColumn = ({ view, type }) => {
  let content = '';
  switch (type) {
    case VIEWS.NAME:
      content = view.view_name;
      break;
    case VIEWS.BYTE_COUNT:
      content = view.byte_count.toString();
      break;
    case VIEWS.DOC_COUNT:
      content = view.doc_count;
      break;
    // eslint-disable-next-line no-case-declarations
    case VIEWS.RATE:
      const ratePerSec = view.view_name ? `${view.rate_per_second} ${message.RUNINFO.DOC_PER_SEC}` : '',
        bytesPerSec = `${view.bytes_per_second === '' ? '0B/s' : view.bytes_per_second}`;
      return (<span style={{ minHeight: 40 }}>
        <div className="rate-col-bytes">{bytesPerSec}</div>
        <div className="rate-col-docs">{ratePerSec}</div>
      </span>);
    default:
      break;
  }
  return <div>{content}</div>;
};
 
const renderEltExtraDetailCol = (value, col) => {

        <div className="rate-col-docs">n/a</div>
      </span>
    );
  }
  return <div style={{ minHeight: 40 }}>{value}</div>;
};
 
const renderViews = (views, eltExtraDetail) => {
  const cells = [];
  const { RUNINFO: { RECORDS_ADDED, RECORDS_UPDATED, RECORDS_DELETED } } = message;
  let eltCells = [];
  let eltExtraDetailValues = [];
  if (eltExtraDetail) {
    eltCells = [RECORDS_ADDED, RECORDS_UPDATED, RECORDS_DELETED];
    eltExtraDetailValues = [eltExtraDetail.totalNumberOfInserts,
      eltExtraDetail.totalNumberOfUpdates, eltExtraDetail.totalNumberOfDeletes];
  }
 
  cells.push(<TableCell key={VIEWS.NAME}>
    {views.map(view => <RenderViewColumn key={view.view_name} view={view} type={VIEWS.NAME} />)}
    {eltCells.map(title => renderEltExtraDetailCol(title))}
  </TableCell>);
  cells.push(<TableCell key={VIEWS.BYTE_COUNT}>
    {views.map(view => <RenderViewColumn key={view.view_name} view={view} type={VIEWS.BYTE_COUNT} />)}
    {eltCells.map(() => renderEltExtraDetailCol(0))}
  </TableCell>);
  cells.push(<TableCell key={VIEWS.DOC_COUNT}>
    {views.map(view => <RenderViewColumn key={view.view_name} view={view} type={VIEWS.DOC_COUNT} />)}
    {eltExtraDetailValues.map(val => renderEltExtraDetailCol(val || 0))}
  </TableCell>);
  cells.push(<TableCell key={VIEWS.RATE}>
    {views.map(view => <RenderViewColumn key={view.view_name} view={view} type={VIEWS.RATE} />)}
    {eltCells.map(() => renderEltExtraDetailCol('', VIEWS.RATE))}
  </TableCell>);
 
  return cells;
};
 
export const renderRow = (argMap) => {
  const { snapName, snapInstance, snapExeStats, idx, instanceId, eltExtraDetail } = argMap,
    { instance_id, state, statistics, failure, reason, resolution,
      has_warnings, has_errors, lint_list, threads,
    } = snapExeStats,
    snapStats = statistics.snap;
 
  const inputs = getViews(statistics.input_views, 'input_views', snapInstance);
  const outputs = getViews(statistics.output_views, 'output_views', snapInstance);
  const errors = getViews(statistics.error_views, 'error_views', snapInstance);
  const views = [...inputs, ...outputs, ...errors];
  const rows = [];
  const hasError = failure || reason || resolution;
 
  // get snap runtime state
  let snapState = state;
  if (state === 'Completed') {
    if (has_warnings) {
      snapState = message.RUNINFO.COMPLETED_WITH_WARNINGS;
    }
    if (has_errors) {
      snapState = message.RUNINFO.COMPLETED_WITH_ERRORS;
    }
  }
 
  const hasErrorDetails = () => hasError || (threads && threads.length > 0) || (lint_list && lint_list.length > 0);
 
  const downloadQueryDetails = (zippedQuery, ruuid) => {
    const buffer = Buffer.from(zippedQuery, 'base64');
    const blob = new Blob([buffer], { type: 'application/gzip' });
    downloadBlob(blob, `${ruuid}_snap_log.json.zip`);
  };
 
  rows.push(
    <TableRow
      key={`${instanceId}${idx}`}
      data-instance-id={instance_id}
      style={hasErrorDetails() || eltExtraDetail ? { borderBottom: '0' } : {}}
    >
      <TableCell
        title={snapState}
      >
        <span className="runinfo-statistics-snap-col">
          <div className={snapState.toLowerCase().replaceAll(' ', '_')} />
          <div title={snapName}>{snapName}</div>
        </span>
      </TableCell>
      <TableCell
        id={`snap-duration-${instance_id}`}
      >
        {snapStats && getDuration(snapStats)}
      </TableCell>
      {renderViews(views.length > 0 ? views : [{
        view_name: 'n/a',
        byte_count: '0',
        doc_count: '0',
        rate_per_second: '0',
        bytes_per_second: '' }], eltExtraDetail)
      }
    </TableRow>);
 
  if (eltExtraDetail) {
    rows.push(
      <TableRow
        key={'meh'}
        className="download-query-details-row"
      >
        <TableCell colSpan="2" className="download-query-details-col" />
        <TableCell
          colSpan="4"
          className="download-query-details-col"
        >
          <div onClick={() => { downloadQueryDetails(eltExtraDetail.zipped_query, instanceId); }}>
            Download Query Details
          </div>
        </TableCell>
      </TableRow>);
  }
 
  if (hasError || (threads && threads.length > 0)) {
    rows.push(<ErrorMessageRow key={`errorMsgRow${instanceId}`} snapExeStats={snapExeStats} state={snapState} />);
  }
 
  if (lint_list) {
    lint_list.forEach((lint) => {
      rows.push(
        <LintMessageRow key={`lintMsgRow${instanceId}`} instance_id={instance_id} lint={lint} state={snapState} />);
    });
  }
 
  return rows;
};
