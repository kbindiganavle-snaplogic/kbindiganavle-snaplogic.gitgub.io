136     private Iterator<Object> executeBulkRead(boolean isVerticalSlicing) throws IOException {
137         // create batch job
138         JobInfo createJobInfo = createJob(options);
139         if (!createJobInfo.getState().equals(State.OPEN.toString())) {
140             String error = String.format(ERR_CREATE_BULK_READ_JOB, serviceUrl, objectType);
141             throw new IOException(error);
142         }
143         // check job status
144         jobId = createJobInfo.getId();
145         LOG.debug("SFDC Bulk API job created, job ID: {}", jobId);
146         checkJobStatus(serviceUrl, jobId);
147         if (isVerticalSlicing) {
148             if (jobIdList == null) {
149                 jobIdList = new ArrayList<>();
150             }
151             jobIdList.add(jobId);
152         }
153         // add batch to the job
154         String batchUrl = String.format(FORMAT_BATCH_URL, serviceUrl, jobId);
155         BatchInfo batchInfo = addBatch(batchUrl, options.getQuery());
156         // if PK Chunking, need to get all batch ID's
157         batchId = batchInfo.getId();
158         Map<String, Map<String, String>> batchInfoMap = getBatchInfoList(batchUrl, jobId, batchId);
159         List<Object> bulkInfoList = Lists.newArrayList();
160         for (Map.Entry<String, Map<String, String>> entry : batchInfoMap.entrySet()) {
161             Map<String, String> batchStatus = entry.getValue();
162             if (batchStatus != null && COMPLETED.equalsIgnoreCase(batchStatus.get(STATE))) {
163                 Object value = batchStatus.get(NUMBER_RECORDS_PROCESSED);
164                 recordsProcessed += value instanceof BigInteger ? ((BigInteger) value).longValue() :
165                         Long.valueOf(batchStatus.get(NUMBER_RECORDS_PROCESSED));
166                 // retrieve all the batch result ID list for a batch ID
167                 String batchId = entry.getKey();
168                 List<String> resultIds;
169                 if (ContentType.JSON == options.getContentType()) {
170                     resultIds = retrieveBatchResultIdsJson(serviceUrl, jobId, batchId);
171                 } else {
172                     ResultList resultList = retrieveBatchResultIds(serviceUrl, jobId, batchId);
173                     resultIds = resultList.getResults();

175                 addBulkInfoList(bulkInfoList, jobId, batchId, resultIds);
176             }
177         }
178         return bulkResponse.getResponseIterator(bulkInfoList);

183      *

185      * @param matchDataType true if data types should be matched to those output by the Salesforce

191     public void execute(Map<String, String> dateTimeFields, boolean matchDataType,
192                         Map<String, String> nonStringFields)
193             throws IOException, InterruptedException, ExecutionException, TimeoutException {
194 
195         // if it is Read snap, get all readable fields from Force.com
196         if (options.getSnapName() == SnapName.READ) {
197             buildSoqlQuery();
198         }
199         int batchSize = options.getBatchSize();
200         httpRequestHandler.setPKChunking(batchSize);
201         LOG.debug("SFDC PK Chunking enabled, batch size: {}", batchSize);
202         // content type should be CSV
203         options.withContentType(Options.ContentType.CSV);
204         // create batch job
205         JobInfo createJobInfo = createJob(options);
206         if (!createJobInfo.getState().equals(State.OPEN.toString())) {
207             String error = String.format(ERR_CREATE_BULK_READ_JOB, serviceUrl, objectType);
208             throw new IOException(error);
209         }
210         // check job status
211         jobId = createJobInfo.getId();
212 
213         LOG.debug("SFDC Bulk API job created, job ID: {}", jobId);
214         checkJobStatus(serviceUrl, jobId);
215         // add batch to the job
216         String batchUrl = String.format(FORMAT_BATCH_URL, serviceUrl, jobId);
217         BatchInfo batchInfo = addBatch(batchUrl, options.getQuery());
218         batchId = batchInfo.getId();
219         try {
220             downloadBatchResults(batchUrl, jobId, dateTimeFields, matchDataType, nonStringFields,
221                     options.getSnapContext());
222         } finally {
223             if (executor != null) {
224                 executor.shutdown(); // Disable new tasks from being submitted
225                 try {
226                     // Wait a while for existing tasks to terminate
227                     if (!executor.awaitTermination(EXECUTOR_SHUTDOWN_TIMEOUT, TimeUnit.SECONDS)) {
228                         executor.shutdownNow(); // Cancel currently executing tasks
229                         // Wait a while for tasks to respond to being cancelled
230                         if (!executor.awaitTermination(EXECUTOR_SHUTDOWN_TIMEOUT,
231                                 TimeUnit.SECONDS)) {
232                             errorViews.write((SnapDataException)
233                                     new SnapDataException(ERR_TIMEOUT_RETRIEVING_RESPONSE)
234                                     .withReason(String.format(ERR_PK_CHUNKING_THREADS_REASON,
235                                             jobId, false)));
236                         }
237                     }
238                 } catch (InterruptedException ie) {
239                     // (Re-)Cancel if current thread also interrupted
240                     executor.shutdownNow();
241                     // Preserve interrupt status
242                     Thread.currentThread().interrupt();
243                 }
244             }
245         }
246     }

291     // Immediately after the bulk query request is submitted,
292     // the response for the job status can be a Map class object if SFDC has not finished
293     // creating a set of batches. When the response becomes a List class object,
294     // it may not contain a complete set of batches. If the queried table is small compared to
295     // the chunk size, SFDC may send response with one Map object or one List object with two Map
296     // objects in it. When it contains two Map objects, one of them is not processed and will
297     // never become completed. This method covers all cases.
298     @SuppressWarnings("unchecked")
299     @VisibleForTesting
300     protected Map<String, Map<String, String>> getBatchInfoList(String url, String jobId,
301             String firstBatchId) {
302         List<Map<String, String>> batchInfoList = null;
303         long startTimeMs = System.currentTimeMillis();
304         boolean allBatchesCompleted = false;
305         List<Map<String, String>> failedBatches = Lists.newArrayList();
306         do {
307             try {
308                 checkForTimeout(jobId, StringUtils.EMPTY, startTimeMs);
309                 TimeUnit.SECONDS.sleep(options.getPollingIntervalSec());
310                 String result = getBulkResponse(url, null, HttpGet.METHOD_NAME);
311                 Object object = null;
312                 if (contentType == ContentType.JSON) {
313                     Map<String, Object> data = OBJECT_MAPPER.readValue(result, MAP_OBJECT_REF);
314                     object = data.get(BATCHINFO);
315                 } else {
316                     Source source = new StreamSource(new StringReader(result));
317                     Map<String, Object> data = (Map<String, Object>) xmlUtils
318                             .convertToJson(xmlInputFactory, jsonOutputFactory, source);
319                     JsonPath path = JsonPathImpl.compile(JSON_PATH_BATCHINFO);
320                     object = path.read(data);
321                 }
322                 if (object instanceof List) {
323                     batchInfoList = (List<Map<String, String>>) object;
324                     allBatchesCompleted = true;
325                     for (Map<String, String> batchInfo : batchInfoList) {
326                         String state = batchInfo.get(STATE);
327                         // There are two separate cases where we decide that the Bulk API
328                         // job has not been completed and break out of the for loop.
329                         // The first case is when State is not 'Completed' nor 'Failed' and the
330                         // batch is the original batch before SFDC splits the job into multiple
331                         // batches. This original batch is not processed and should be ignored.
332                         // In this first case, all batches should be either 'Completed' or 'Failed'
333                         // state except the first batch when the entire Bulk API job is completed.
334                         // The second case is when the Bulk API job is for JSON content type
335                         // and a large query, SFDC returns the batch info as a map object
336                         // wrapped in a list instead of just one map data. In this second case,
337                         // the State would be 'InProgress' when it is not completed.
338                         if (!COMPLETED.equalsIgnoreCase(state) && !FAILED.equalsIgnoreCase(state) &&
339                                 !firstBatchId.equalsIgnoreCase(batchInfo.get(ID)) ||
340                                 IN_PROGRESS.equalsIgnoreCase(state) ||
341                                 QUEUED.equalsIgnoreCase(state)) {
342                             allBatchesCompleted = false;
343                             break;
344                         }
345                     }
346                 } else if (object instanceof Map) {
347                     Map<String, String> batchInfo = (Map<String, String>) object;
348                     String state = batchInfo.get(STATE);
349                     if (COMPLETED.equalsIgnoreCase(state) || FAILED.equalsIgnoreCase(state)) {
350                         allBatchesCompleted = true;
351                         batchInfoList = Lists.newArrayListWithExpectedSize(1);
352                         batchInfoList.add(batchInfo);
353                     }
354                 }
355             } catch (Exception e) {
356                 throw new SnapDataException(e, ERR_BATCH_INFO).formatWith(jobId, e.getMessage());
357             }
358         } while (!allBatchesCompleted);
359         Map<String, Map<String, String>> batchInfoMap = Maps.newLinkedHashMap();
360         for (Map<String, String> batchInfo : batchInfoList) {
361             String state = batchInfo.get(STATE);
362             if (FAILED.equalsIgnoreCase(state)) {
363                 failedBatches.add(batchInfo);
364             } else if (COMPLETED.equalsIgnoreCase(state)) {
365                 batchInfoMap.put(batchInfo.get(ID), batchInfo);
366             }
367         }
368         if (!failedBatches.isEmpty()) {
369             for (Map<String, String> batchInfo : failedBatches) {
370                 errorViews.write((SnapDataException) new SnapDataException(
371                         documentUtility.newDocument(batchInfo), ERR_BATCH_JOB)
372                         .withReason(String.format(REASON_BATCH_JOB,
373                                 batchInfo.get(STATE_MESSAGE)))
374                         .withResolution(RESOLUTION_ADDRESS_ISSUE),
375                         options.getSnapContext().getOriginal());
376             }
377         }
378         return batchInfoMap;
379     }

381     /**
382      * This method is used in PK Chunking to get the status of bulk query job.
383      * It launches a separate thread to download records for each completed batch.
384      *
385      * @param url             - Salesforce batch result URL
386      * @param jobId           - Job ID
387      * @param dateTimeFields  - date/time fields to be handled in each output document
388      * @param matchDataType   - true if data types should be matched to those output by the Salesforce
389      *                          REST API, false otherwise
390      * @param nonStringFields - fields with non-string actual data types
391      *
392      * @throws IOException, InterruptedException, ExecutionException, TimeoutException
393      */
394     @SuppressWarnings("unchecked")
395     private void downloadBatchResults(String url, String jobId,
396             Map<String, String> dateTimeFields, boolean matchDataType,
397             Map<String, String> nonStringFields, SnapContext snapContext)
398             throws IOException, InterruptedException, ExecutionException, TimeoutException {
399 
400         //start the timer to keep track of the timeout period
401         long startTimeMs = System.currentTimeMillis();
402 
403         // get advanced properties
404         Map<String, Object> advancedProperties = options.getAdvancedProperties();
405         boolean useTempFiles = false;
406         int maxThreads = DEFAULT_MAX_THREADS;
407         if (MapUtils.isNotEmpty(advancedProperties)) {
408             Object value = advancedProperties.get(NUMBER_OF_THREADS_IN_PK_CHUNKING);
409             if (value != null) {
410                 maxThreads = (Integer) value;
411             }
412             value = advancedProperties.get(USE_TEMP_FILES_IN_PK_CHUNKING);
413             if (value != null) {
414                 useTempFiles = (Boolean) value;
415             }
416         }
417 
418         //Monitor the job progress. When each job in the batch is ready to be downloaded,
419         //a thread is started to download and process the records in the batch.
420         List<Map<String, String>> batchInfoList;
421         boolean allBatchesCompleted;
422         Set<String> completedBatches = new HashSet<>();
423         Map<String, Map<String, String>> failedBatches = new HashMap<>();
424         if (executor == null) {
425             executor = Executors.newFixedThreadPool(maxThreads);
426         }
427         Map<String, Future<Integer>> threads = new HashMap<>();
428         Set<String> completedThreads = new HashSet<>();
429         do {
430             try {
431                 checkForTimeout(jobId, StringUtils.EMPTY, startTimeMs);
432                 LOG.debug("PK Chunking download download batch wait/count time: {}/{} : #{}#",
433                         options.getPollingIntervalSec(), threads.size(), threads.keySet());
434                 TimeUnit.SECONDS.sleep(options.getPollingIntervalSec());
435                 // check for completed download threads
436                 for (Map.Entry<String, Future<Integer>> entry : threads.entrySet()) {
437                     Future<Integer> thread = entry.getValue();
438                     String batchId = entry.getKey();
439                     if (thread.isDone()) {
440                         LOG.debug("PK Chunking download thread completed for batch ID: {}",
441                                 batchId);
442                         //Since get() is called after checking that the thread is done,
443                         //the call should not take long. The get() timeout overload is used
444                         //just to be safe and avoid a situation where this hangs forever.
445                         int count = thread.get(options.getPollingIntervalSec(), TimeUnit.SECONDS);
446                         completedThreads.add(batchId);
447                     }
448                 }
449                 // remove completed threads from the thread list
450                 for (String batchId : completedThreads) {
451                     threads.remove(batchId);
452                 }
453                 completedThreads.clear();
454                 if (threads.size() >= maxThreads) {
455                     allBatchesCompleted = false;
456                     continue;
457                 }
458                 String result = getBulkResponse(url, null, HttpGet.METHOD_NAME);
459                 Map<String, Object> data = (Map<String, Object>) parseXml(result);
460                 JsonPath path = JsonPathImpl.compile(JSON_PATH_BATCHINFO);
461                 Object object = path.read(data);
462                 allBatchesCompleted = true;
463                 if (object instanceof List) {
464                     batchInfoList = (List<Map<String, String>>) object;
465 
466                     //Check each batch to see if it is completed.
467                     //If so, start a thread to download the results.
468                     for (Map<String, String> batchInfo : batchInfoList) {
469                         String batchId = batchInfo.get(ID);
470                         switch (batchInfo.get(STATE)) {
471                             case COMPLETED:
472                                 if (!completedBatches.contains(batchId) && Integer.valueOf(
473                                         batchInfo.get(NUMBER_RECORDS_PROCESSED)) > 0) {
474                                     if (threads.size() < maxThreads) {
475                                         Future<Integer> thread = executor.submit(
476                                                 new PKChunkingDownload(batchInfo,
477                                                 (BulkReadResponse) bulkResponse, this, useTempFiles,
478                                                 dateTimeFields, matchDataType, nonStringFields,
479                                                         options));
480                                         threads.put(batchId, thread);
481                                         completedBatches.add(batchId);
482                                         LOG.debug("PK Chunking download thread started for batch " +
483                                                 "ID: {}", batchId);
484                                     } else {
485                                         allBatchesCompleted = false;
486                                         LOG.debug("PK Chunking download allBatchesCompleted for " +
487                                                 "batch ID: {}. Waiting for available thread to " +
488                                                 "download results.", batchId);
489                                     }
490                                 }
491                                 break;
492                             case FAILED:
493                                 LOG.debug("PK Chunking download failed batch ID: {}", batchId);
494                                 failedBatches.put(batchId, batchInfo);
495                                 break;
496                             case NOT_PROCESSED:
497                                 LOG.debug("PK Chunking download not processed batch ID: {}",
498                                         batchId);
499                                 break;
500                             default:
501                                 LOG.debug("PK Chunking download default batch ID: {}", batchId);
502                                 allBatchesCompleted = false;
503                         }
504                     }
505                 } else if (object instanceof Map) {
506                     Map<String, String> batchInfo = (Map<String, String>) object;
507                     String batchId = batchInfo.get(ID);
508                     switch (batchInfo.get(STATE)) {
509                         case COMPLETED:
510                             Integer numRecsProcessed = Integer.valueOf(
511                                     batchInfo.get(NUMBER_RECORDS_PROCESSED));
512                             LOG.debug(String.format("PK Chunking Map download numRecsProcessed %d",
513                                     numRecsProcessed));
514                             Future<Integer> thread = executor.submit(new PKChunkingDownload(
515                                     batchInfo, (BulkReadResponse) bulkResponse, this, useTempFiles,
516                                     dateTimeFields, matchDataType, nonStringFields, options));
517                             threads.put(batchId, thread);
518                             LOG.debug("PK Chunking Map download default batch ID: {}", batchId);
519                             break;
520                         case FAILED:
521                             LOG.debug("PK Chunking Map download failed batch ID: {}", batchId);
522                             failedBatches.put(batchId, batchInfo);
523                             break;
524                         default:
525                             LOG.debug("PK Chunking Map download default batch ID: {}", batchId);
526                             allBatchesCompleted = false;
527                     }
528                 } else {
529                     LOG.debug("PK Chunking, JSONPath to batchinfo response neither list or map");
530                 }
531             } catch (SuggestViewAbortException e) {
532                 throw e;
533             } catch (Throwable e) {
534                 throw new SnapDataException(e, ERR_BATCH_INFO).formatWith(jobId, e.getMessage());
535             }
536         } while (!allBatchesCompleted);
537         if (!failedBatches.isEmpty()) {
538             for (Map<String, String> batchInfo : failedBatches.values()) {
539                 errorViews.write((SnapDataException) new SnapDataException(
540                         documentUtility.newDocument(batchInfo), ERR_FAILED_BATCHES)
541                         .withReason(String.format(REASON_FAILED_BATCHES,
542                                 batchInfo.get(STATE_MESSAGE)))
543                         .withResolution(RESOLUTION_ADDRESS_ISSUE));
544             }
545         }
546         // get the total number of records preocessed in the job
547         int totalRecordsProcessed;
548         try {
549             Map<String, String> jobInfoMap = getJobInfoMap(serviceUrl, jobId);
550             String recordCount = jobInfoMap.get(NUMBER_RECORDS_PROCESSED);
551             if (StringUtils.isNotBlank(recordCount)) {
552                 totalRecordsProcessed = Integer.valueOf(recordCount);
553             } else {
554                 throw new IOException(String.format(ERR_TOTAL_NUMBER_RECORDS_PROCESSED, jobId));
555             }
556         } catch (Exception e) {
557             throw new IOException(String.format(ERR_TOTAL_NUMBER_RECORDS_PROCESSED, jobId));
558         }
559         LOG.debug("PK Chunking download threads have been invoked and being monitored for their" +
560                 " completion.");
561         // monitor all threads with timeout
562         Map<String, Future<Integer>> threadInProgress = waitForCompletion(threads);
563         //If there are any batches of records that have not been processed, cancel them.
564         //This is considered a timeout for the threads.
565         if (threadInProgress != null && threadInProgress.size() > 0) {
566             cancelDownloadThreads(threadInProgress, executor);
567         }
568     }

571      * Cancels the threads downloading the PK Chunking batches.
572      * This is called when the threads have not completed in the expected timeout period.
573      *
574      * @param threadInProgress The threads that have not completed downloading and processing the
575      *                         batch of records.
576      * @param executor         The executor that the threads are associated with.
577      *
578      * @throws InterruptedException
579      */
580     private void cancelDownloadThreads(
581             Map<String, Future<Integer>> threadInProgress,
582             ExecutorService executor) throws InterruptedException {
583         // cancel any thread still in progress
584         for (Map.Entry<String, Future<Integer>> entry: threadInProgress.entrySet()) {
585             Future<Integer> eachThread = entry.getValue();
586             if (eachThread != null) {
587                 eachThread.cancel(true);
588                 LOG.info("Canceled thread for batch ID: {}", entry.getKey());
589             }
590         }
591     }

593     private Map<String, Future<Integer>> waitForCompletion(Map<String, Future<Integer>> threads)
594             throws InterruptedException, ExecutionException, TimeoutException, IOException {
595         int threadsSize = threads.size();
596         List<String> doneOrCanceled = new ArrayList<>(threadsSize);
597         boolean inProgress;
598         Map<String, Future<Integer>> threadsInProgress = new HashMap<>(threadsSize);
599         //start the timer to keep track of the timeout period
600         long startTimeMs = System.currentTimeMillis();
601         // monitor the states of threads with timeout
602         do {
603             inProgress = false;
604             sleep(options.getPollingIntervalSec(), TimeUnit.SECONDS);
605             for (Map.Entry<String, Future<Integer>> entry : threads.entrySet()) {
606                 Future<Integer> thread = entry.getValue();
607                 String batchId = entry.getKey();
608                 LOG.debug("PK Chunking download thread waiting for batch " +
609                         "ID: {}", batchId);
610                 if (!doneOrCanceled.contains(batchId)) {
611                     if (thread.isCancelled() || thread.isDone()) {
612                         //This get() call is required to bring any exceptions
613                         //thrown in the callable thread back to the main snap execution thread.
614                         //The most common use case is ExecutionExceptions thrown when writing
615                         //to the error views. The error views object throws an ExecutionException
616                         //when the snap is configured to stop execution if an error is encountered.
617                         //
618                         //Since get() is called after checking that the thread is done or cancelled,
619                         //the call should not take long. The get() timeout overload is used
620                         //just to be safe and avoid a situation where this hangs forever.
621                         thread.get(options.getPollingIntervalSec(), TimeUnit.SECONDS);
622                         doneOrCanceled.add(batchId);
623                         threadsInProgress.remove(batchId);
624                     } else {
625                         inProgress = true;
626                         threadsInProgress.put(batchId, entry.getValue());
627 
628                     }
629                 }
630             }
631         }
632         //we don't use checkForTimeout() because there could be a scenario where the download
633         // would have completed and sleep(options.getPollingIntervalSec(), TimeUnit.SECONDS),
634         // causing it go over the timeout interval and this would throw an exception in the
635         // checkForTimeout()
636         while (inProgress && (System.currentTimeMillis() - startTimeMs) < options
637                 .getPollingTimeoutSec() * 1000);
638         return threadsInProgress;
639     }

746     private List<Iterator<Object>> processVerticalSlicing(List<String> fieldList)
747             throws IOException {
748         List<Iterator<Object>> iteratorList = new ArrayList<>(fieldList.size());
749         long lastRecordsProcessed = 0;
750         for (String fields : fieldList) {
751             String soql = options.toSOQL(fields, ORDER_BY_ID);
752             options.withQuery(soql);
753             iteratorList.add(executeBulkRead(true));
754             if (lastRecordsProcessed == 0) {
755                 lastRecordsProcessed = recordsProcessed;
756             } else if (lastRecordsProcessed != recordsProcessed) {
757                 throw new SnapDataException(ERR_RECORDS_PROCESSED)
758                         .formatWith(lastRecordsProcessed, recordsProcessed)
759                         .withReason(ERR_INTERNAL_ERROR)
760                         .withResolutionAsDefect();
761             }
762             recordsProcessed = 0;
763         }
764         this.iteratorList = iteratorList;
765         return iteratorList;
766     }
