20 /**
21  * Attempts to convert any object to a BigDecimal object.
22  *
23  * @since May, 2017
24  */
25 public class BigDecimalConverter implements  Converter<BigDecimal> {
26     private Integer scale;
27 
28     // if scale is not null, convert() method will set scale to the output BigDecimal object
29     public BigDecimalConverter(Integer scale) {
30         this.scale = scale;
31     }
32 
33     @Override
34     public BigDecimal convert(Object object) {
35         if (!(object instanceof BigDecimal)) {
36             try {
37                 object = new BigDecimal(object.toString());
38             } catch (NumberFormatException e) {
39                 throw new SnapDataException(e, ERR_CONVERT)
40                         .formatWith(object, BigDecimal.class.getSimpleName())
41                         .withResolution(RESOLUTION_ADDRESS_ISSUE);
42             }
43         }
44         return scale == null ? (BigDecimal) object :
45                 ((BigDecimal) object).setScale(scale, BigDecimal.ROUND_HALF_UP);
46     }
47 }
