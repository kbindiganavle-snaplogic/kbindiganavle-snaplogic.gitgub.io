34     private RuntimeException handleCsvParserError(String message, TypeEnum type, String source,
35                                            String target) {
36         String errorMessage = getErrorMessage(type, source, target);
37 
38         if (message.contains(CSV_PARSE_EXCEPTION)) {
39             return new SnapDataException(getErrorMessage(type, source, target))
40                     .withReason("CSV file is invalid or corrupted.")
41                     .withResolution("Please make sure that file is valid.");
42         }
43 
44         return new SnapDataException(errorMessage)
45                 .withReason(AUTO_SYNC_JSON_PARSER_GENERIC_ERROR)
46                 .withResolution(CONTACT_AUTO_SYNC_TECHNICAL_SUPPORT);
