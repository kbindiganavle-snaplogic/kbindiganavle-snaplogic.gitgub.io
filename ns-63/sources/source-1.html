77     private static class ByteChannelAppendable implements Appendable, Closeable {
78         private final CharsetEncoder charsetEncoder;
79         private final WritableByteChannel outChannel;
80         private final char[] inBacking = new char[8 * 1024];
81         private final CharBuffer bufferIn = CharBuffer.wrap(inBacking);
82         private final byte[] outBacking = new byte[inBacking.length * 8];
83         private final ByteBuffer bufferOut = ByteBuffer.wrap(outBacking);
84

87             this.charsetEncoder = charsetEncoder;
88             this.outChannel = outChannel;
89         }

96         @Override
97         public Appendable append(final CharSequence csq, int start, final int end) throws
98                 IOException {
99             int remaining = end - start;
100             while (remaining > 0) {
101                 int subrange = Math.min(bufferIn.capacity() - bufferIn.position(), remaining);
102                 // if we're not at the end of the string, check to see if the character we're ending
103                 // at is a multi-byte character, we have to check from before the last character to
104                 // one after the last character specified to know for sure. We also have to ensure
105                 // that the character before the extension is less than 3 bytes, or we have a false
106                 // positive.
107                 if (start + subrange < csq.length() &&
108                         // see if character is split in half and causes a potential issue
109                         csq.subSequence(start + subrange - 1, start + subrange + 1)
110                                 .toString().getBytes(StandardCharsets.UTF_8).length == 4 &&
111                         // ensure that last character is less than 3 characters as if we're before
112                         // a delimiter or quote character, this will give us a false positive
113                         csq.subSequence(start + subrange - 1, start + subrange)
114                                 .toString().getBytes(StandardCharsets.UTF_8).length < 3) {
115                     subrange--;
116                 }
117                 bufferIn.append(csq, start, start + subrange);
118                 encodeBufferIn();
119                 start += subrange;
120                 remaining -= subrange;
121             }
122             return this;
123         }
124 
125         @Override
126         public Appendable append(final char c) throws IOException {
127             bufferIn.append(c);
128             encodeBufferIn();
129             return this;
130         }

132         private void encodeBufferIn() throws IOException {
133             bufferIn.flip();
134             try {
135                 while (bufferIn.hasRemaining()) {
136                     CoderResult result = charsetEncoder.encode(bufferIn, bufferOut, false);
137                     if (result.isError()) {
138                         bufferOut.clear();
139                         try {
140                             result.throwException();
141                         } catch (CharacterCodingException e) {
142                             throw new ColumnEncodingException(bufferIn.position(), e);
143                         }
144                     }
145                     if (result.isOverflow()) {
146                         flush();
147                     }
148                 }
149             } finally {
150                 bufferIn.clear();
151             }
152         }
153 
154         private void flush() throws IOException {
155             bufferOut.flip();
156             outChannel.write(bufferOut);
157             bufferOut.clear();
158         }
159 
160         @Override
161         public void close() throws IOException {
162             try {
163                 flush();
164             } finally {
165                 outChannel.close();
166             }
167         }

259      * Writes the document as CSV to the {@link WritableByteChannel}.
260      *
261      * @param document the document
262      * @throws SnapDataException for any formatting errors that might occur.
263      */
264     @SuppressWarnings("unchecked")
265     public void write(Document document) throws SnapDataException {
266         Object object = document.get();
267         // The Snap expects the document data to be in a flat map structure. If the data is not
268         // a map, then it is either ignored or the pipeline is failed.
269         if (object instanceof Map) {
270             Map<String, Object> data = (Map<String, Object>) object;
271             if (isFirst) {
272                 initialize(document, data);
273                 isFirst = false;
274             }
275             try {
276                 writeDocument(data);
277             } catch (IOException e) {
278                 throw new SnapDataException(e, IO_ERROR_OCCURRED)
279                         .withReason(e.getMessage())
280                         .withResolution(RESOLUTION_ADDRESS_ISSUE);
281             } catch (InvalidPathException e) {
282                 throw new SnapDataException(e, "Unable to read value")
283                         .withResolution(e.getResolution());
284             }
285         } else {
286             invalidDataFormat(object);
287         }
288     }

290     private void initialize(Document document, Map<String, Object> data) {
291         try {
292             if (writeBOM) {
293                 // write Byte Order Mark in the beginning of output stream
294                 ByteBuffer bomByteBuffer = ByteBuffer.wrap(
295                         TransformUtils.getByteOrderMark(charset).getBytes());
296                 writableByteChannel.write(bomByteBuffer);
297                 if (!UTF_8.equals(charset)) {
298                     // convert 8-bit delimiter and quote character to a given char set
299                     delimiter = TransformUtils.convertString(delimiter, charset);
300                     quoteCharacter = (TransformUtils.convertString(String.valueOf(quoteCharacter)
301                             , charset).charAt(0));
302                 }
303             }
304             if (quoteCharacter == UNICODE_NULL) {
305                 quoteMode = QuoteMode.NONE;
306             }
307             this.writer = new ByteChannelAppendable(Charset.forName(charset).newEncoder(),
308                     writableByteChannel);
309             try {
310                 csvPrinter = new CSVPrinter(writer, CSVFormat.Builder.create()
311                         .setDelimiter(delimiter)
312                         .setQuote(quoteCharacter)
313                         .setRecordSeparator(newline)
314                         .setEscape(DEFAULT_ESCAPE_CHARACTER)
315                         .setQuoteMode(quoteMode)
316                         .setEscape(escapeChar).build());
317 
318             } catch (Exception e) {
319                 throw new SnapDataException(e, ERR_CANNOT_FORMAT_CSV)
320                         .withReason(e.getMessage())
321                         .withResolution(RESOLUTION_ADDRESS_ISSUE);
322             }
323             if (useHeader) {
324                 header = Sets.newLinkedHashSet(colNames);
325             } else {
326                 Set<String> keys = data.keySet();
327                 if (CollectionUtils.isEmpty(keys)) {
328                     data.put(ERROR, ERR_NO_HEADER_IN_DATA);
329                     throw new SnapDataException(ERR_NO_HEADER_IN_DATA)
330                             .withReason(ERR_NO_HEADER_IN_DATA)
331                             .withResolution(ERR_NO_HEADER_IN_DATA_RESOLUTION);
332                 }
333                 header = Sets.newLinkedHashSet(keys.stream()
334                         .map(key -> JsonPaths.compile((Object) key))
335                         .collect(Collectors.toCollection(LinkedHashSet::new)));
336             }
337             if (writeHeader) {
338                 csvPrinter.printRecord(header.stream()
339                         .map(Document2CSVConverter::pathToName)
340                         .collect(Collectors.toList()));
341             }
342         } catch (IOException e) {
343             throw new ConfigurationException(e, e.getMessage())
344                     .withResolutionAsDefect();
345         }
346     }

348     private void writeDocument(Map<String, Object> data) throws IOException, InvalidPathException {
349         int numMissingEntries = 0;
350         for (JsonPath headerPath : header) {
351             Object value = headerPath.read(data, null);
352             // If the value is null, it means there was no value for the key.
353             // This solves the situation where you have recorded the header as a,b,c but you
354             // receive a document with headers b,c,d. The code would check known headers a,b,c
355             // for values. It wouldn't find anything for "a" so it would just write and empty
356             // string for that column, then process b and c. In the end, this would only
357             // "process" 2 keys (b,c) since "a" was not found. However, there were 3 keys
358             // (b,c,d) so there was a header that was missed and will need to be processed.
359             // This helps detect new keys and cut down on the expensive call to addNewHeaders.
360             if (value != null) {
361                 if ((value instanceof Map) || (value instanceof List)) {
362                     // the value should not be of a Map or List type
363                     invalidDataFormat(value);
364                 }
365                 if (!(quoteMode == QuoteMode.NON_NUMERIC && value instanceof Number)) {
366                     value = ObjectType.toString(value);
367                 }
368             } else {
369                 value = DEFAULT_COL_VALUE;
370                 if (!headerPath.existsIn(data)) {
371                     // A counter for missing kay/value entries in an input document is needed to
372                     // acutately determine if there are additional columns.
373                     numMissingEntries++;
374                 }
375             }
376             try {
377                 csvPrinter.print(value);
378             } catch (ColumnEncodingException e) {
379                 // Do a println here so the CSVPrinter resets its state to think it's at the start
380                 // of a new record.  Otherwise, we end up with an extra delimiter being added to
381                 // the output on the next record.  Hopefully, this leaves the output in a good
382                 // state, the blank line should not bother and CSV parsers.
383                 csvPrinter.println();
384                 writer.flush();
385                 throw new IOException(String.format(VALUE_CANNOT_BE_ENCODED,
386                         pathToName(headerPath), e.getCause() instanceof MalformedInputException ?
387                                 MALFORMED : UNMAPPABLE, e.getPosition()));
388             }
389         }
390         // If additional key/value pairs are found in the input documents, handle the case
391         // according to the header size error policy. If a user defined header is being
392         // used, then extra columns are ignored.
393         try {
394             if (header.size() < data.size() + numMissingEntries && !useHeader) {
395                 switch (errorPolicy) {
396                     case ERROR:
397                         throw new SnapDataException(ERR_EXTRA_COLUMN)
398                                 .withReason(REASON_EXTRA_COLUMN)
399                                 .withResolution(RESOLUTION_ADDRESS_ISSUE);
400                     case APPEND:
401                         for (Map.Entry<String, Object> entry : data.entrySet()) {
402                             if (!header.contains(JsonPaths.compile((Object) entry.getKey()))) {
403                                 header.add(JsonPaths.compile((Object) entry.getKey()));
404                                 Object value = entry.getValue();
405                                 String strVal = value == null ? DEFAULT_COL_VALUE :
406                                         ObjectType.toString(value);
407                                 csvPrinter.print(strVal);
408                             }
409                         }
410                         break;
411                     case IGNORE:
412                     default:
413                         // No-Op
414                 }
415             }
416         } finally {
417             csvPrinter.println();
418             writer.flush();
419         }
420     }
