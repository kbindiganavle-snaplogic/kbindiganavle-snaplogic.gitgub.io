43 @General(title = THROTTLE_BOT_LABEL, purpose = THROTTLE_BOT_PURPOSE)
44 @Category(snap = SnapCategory.WRITE)
45 @Inputs(min = 1, max = 1, accepts = {ViewType.DOCUMENT})
46 @Outputs(min = 1, max = 1, offers = {ViewType.DOCUMENT})
47 @Version(snap = 1)
48 public class ThrottleBot extends SimpleSnap {
49     private static final String PROP_THROTTLE_TYPE = "throttle_type";
50     private static final String PROP_SPEED = "speed";
51     private static final int MAX_WAIT_TIME = 5000;
52     private final Set<String> types = Sets.newHashSet(
53             ThrottleType.RANDOM.toString(),
54             ThrottleType.FIXED.toString(),
55             ThrottleType.EXPONENTIAL.toString(),

87     //---------------- Private class and enums ----------------------------------------
88     private class ThrottlePolicy {
89         private final Random random = new Random(System.currentTimeMillis());
90         private final ThrottleType throttleType;
91         private final int throttleSpeed;
92         private int counter = 0;
93 
94         private ThrottlePolicy(final ThrottleType throttleType, final int throttleSpeed) {
95             this.throttleType = throttleType;
96             if (throttleType == ThrottleType.RANDOM && throttleSpeed <= 0) {
97                 this.throttleSpeed = MAX_WAIT_TIME;
98             } else {
99                 this.throttleSpeed = throttleSpeed;
100             }
101         }
102 
103         public void throttle() {
104             int waitTimeInMs = calculateThrottleTime();
105             try {
106                 Thread.sleep(waitTimeInMs);
107             } catch (InterruptedException e) {
108                 throw new ExecutionException(e, INTERRUPTED_DURING_DOCUMENT_THROTTLING)
109                         .withReason(SNAP_INTERRUPTED_WHILE_WAITING);
110             }
111         }
112 
113         private int calculateThrottleTime() {
114             int waitTimeInMs;
115             switch (throttleType) {
116                 case RANDOM:
117                     waitTimeInMs = Math.abs(random.nextInt(throttleSpeed));
118                     break;
119                 case EXPONENTIAL:
120                     waitTimeInMs = throttleSpeed * (int) Math.round(Math.exp(++counter));
121                     break;
122                 case LOGARITHMIC:
123                     waitTimeInMs = throttleSpeed * (int) Math.round(Math.log(++counter));
124                     break;
125                 case FIXED:
126                 default:
127                     // Fixed throttling
128                     waitTimeInMs = throttleSpeed;
129             }
130             return waitTimeInMs;
131         }
132     }
133     private enum ThrottleType {
134         RANDOM("Random"),
135         FIXED("Fixed"),
136         EXPONENTIAL("Exponential"),
137         LOGARITHMIC("Logarithmic");
