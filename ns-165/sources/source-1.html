156     @Override
157     protected void process(final Document document, final String inputViewName) {
158 
159         computeExpectedData(document);
160         ByteArrayOutputStream baos = new ByteArrayOutputStream();
161         Document actualDocument;
162         try {
163             // de/serialize the incoming document to guarantee we have the same type
164             // representations.
165             writeObjectMapper.writeValue(baos, document.get());
166             Object actual = readObjectMapper.readValue(baos.toByteArray(), Object.class);
167             actualDocument = documentUtility.newDocument(actual);
168         } catch (IOException e) {
169             SnapDataException dataException = new SnapDataException(e, e.getMessage())
170                     .withResolution(PROVIDED_INPUT_DOCUMENT_IS_NOT_SERIALIZABLE);
171             errorViews.write(dataException, document);
172             return;
173         }
174         BigInteger actualMd5 = new BigInteger(actualDocument.getMd5());
175         Object expected = expectedData.isEmpty() ? null : expectedData.remove(0);
176         Document expectedDocument = documentUtility.newDocument(expected);
177         BigInteger expectedMd5 = new BigInteger(expectedDocument.getMd5());
178         if (expectedMd5.equals(actualMd5)) {
179             outputViews.write(document);
180         } else {
181             Map<String, Object> errorMap = Maps.newHashMapWithExpectedSize(3);
182             errorMap.put(ERROR, REASON_NOT_EQUAL);
183             errorMap.put(EXPECTED, expected);
184             SnapDataException ex = new SnapDataException(
185                     documentUtility.newDocument(errorMap), ERR_NOT_EQUAL)
186                     .formatWith(expected, document.get())
187                     .withReason(REASON_NOT_EQUAL)
188                     .withResolution(RESOLUTION_NOT_EQUAL);
189             errorViews.write(ex, document);
190         }
191     }

193     private List<Object> computeExpectedData(Document document) {
194         String expectedEditorData = evaluateExpressions ? editorProperty.eval(document)
195                 : editorProperty.eval(null);
196         if (expectedData == null) {
197             try {
198                 expectedData = MAPPER.readValue(expectedEditorData, TYPE_REF);
199             } catch (JsonMappingException e) {
200                 throwException(e, REASON_NOT_LIST +
201                                 StringUtils.SPACE +
202                                 Throwables.getRootCause(e).getMessage(),
203                         expectedEditorData);
204             } catch (JsonParseException e) {
205                 throwException(e, REASON_INVALID_JSON +
206                                 StringUtils.SPACE +
207                                 Throwables.getRootCause(e).getMessage(),
208                         expectedEditorData);
209             } catch (Exception e) {
210                 throwException(e, Throwables.getRootCause(e).getMessage(), expectedEditorData);
211             }
212             expectedSize = expectedData.size();
213         }
214         return expectedData;
215     }

226     private boolean isDataCleanedUp(List<Object> data) {
227         if (CollectionUtils.isEmpty(data)) {
228             return true;
229         } else if (data.size() == 1 && data.get(0) instanceof Map

233         return false;
