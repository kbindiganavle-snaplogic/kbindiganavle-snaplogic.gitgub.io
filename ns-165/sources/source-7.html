103     @Override
104     protected void processDocument(Document document, String inputViewName) {
105         // For every document received, leak one connection
106         Connection conn = jdbcOperations.acquireConnection(account);
107         document.acknowledge();
108         switch (documentCount++) {
109             case 2:
110                 try {
111                     conn.createStatement();
112                 } catch (SQLException e) {
113                     LOG.debug("(+) Should have closed a connection on 2nd document");
114                 }
115                 break;
116             case 5:
117                 throw new RuntimeException("(+) Leaking w/Runtime Exception " +
118                     "documentCount = " + documentCount);
119             default:
120                 break;
121         }
122     }
