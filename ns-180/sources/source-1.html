329     @Override
330     public void execute() {
331         SnapDataException oneTimeException = null;
332         if (inputViews.isEmpty()) {
333             return;
334         }
335         // Generate schema from existing table in v2, or from existing schema view otherwise
336         String schemaViewName = getSchemaViewName();
337         try {
338             datasetId = getDatasetID(datasetName);
339             if (StringUtils.isNotEmpty(datasetId) && VERSION_2.equals(getAPIVersion())) {
340                 // For the V2 API (or Table API), we always take the existing table
341                 // schema as the highest priority. So we use the remote schema when the
342                 // table exists and API version is 2
343                 schemaMap = getSchemaFromRemoteTable(datasetId);
344             } else if (schemaViewName != null) {
345                 // generate schema from schema view
346                 schemaMap = getSchemaFromView(inputViews.get(schemaViewName));
347             }
348             initializeClient();
349         } catch (SnapDataException sde) {
350             oneTimeException = sde;
351         } finally {
352             if (schemaViewName != null) {
353                 // Done with the schema view, and need to consume it fully to make sure pipeline
354                 // complete correctly
355                 inputViews.getDocumentsFrom(inputViews.get(schemaViewName))
356                         .forEachRemaining(document ->
357                                 LOG.info("The schema document is ignored, because either the v2 " +
358                                         "table exists or there are errors when generating schema"));
359             }
360         }
361         // Process Data
362         InputView dataView = inputViews.get(DATA_VIEW_NAME);
363         if (dataView == null) {
364             if(oneTimeException != null) {
365                 errorViews.write(oneTimeException);
366             }
367             return;
368         }
369         boolean firstDocument = true;
370         for (Document document : inputViews.documents(dataView)) {
371             if (oneTimeException != null) {
372                 errorViews.write(oneTimeException, document);
373                 continue;
374             }
375             if (firstDocument) {
376                 try {
377                     // For the first Document, we need to consider getting schema, create a
378                     // dataset/table if needed, and create a bucket for the load
379                     if (MapUtils.isEmpty(schemaMap)) {
380                         // The schemaMap would be empty here if the table doesn't exists in v2
381                         // api, or the schemaView is either not provided or no incoming data. In
382                         // this case, we generate schema from the first input document.
383                         LOG.debug("No schema generated from Schema view. Generating a schema" +
384                                     " from input Document");
385                         schemaMap = getSchemaFromDocument(document);
386                     }
387                     if (datasetId == null) {
388                         if (autoCreateDataset) {
389                             statusReporter.notify(CREATING_DATASET, datasetName);
390                             datasetId = createDataset(datasetName, document);
391                         } else {
392                             throw new SnapDataException(ERROR_DATASET_NOT_FOUND)
393                                     .formatWith(datasetName)
394                                     .withReason(REASON_DATASET_NOT_FOUND)
395                                     .withResolution(RESOLUTION_DATASET_NOT_FOUND);
396                         }
397                     }
398 
399                     // create a bucket
400                     statusReporter.notify(CREATING_BUCKET);
401                     bucketId = createBucket(datasetId, document, fileInsertMode);
402                     firstDocument = false;
403                     statusReporter.notify(ACCEPTING_DATA);
404                 } catch (SnapDataException sde) {
405                     oneTimeException = sde;
406                     errorViews.write(oneTimeException, document);
407                     continue;
408                 }
409             }
410             try {
411                 process(document, DATA_VIEW_NAME);
412             } catch (SnapDataException sde) {
413                 // Catch the exception here for per-document operations.
414                 errorViews.write(sde, document);
415             }
416         }
417         if (document2CSVConverter != null) {
418             // Flush the file output stream so that data would be written to the disk
419             IOUtils.closeQuietly(document2CSVConverter);
420         }
421         if (tmpFileReport != null) {
422             tmpFileReport.done();
423         }
424 
425         // Finalize wBucket
426         try {
427             if (bucketId != null) {
428                 doUpload(tempFile, bucketId);
429                 for (Future result : uploadResults) {
430                     result.get();
431                 }
432                 uploadThreadPool.shutdown();
433                 uploadThreadPool.awaitTermination(bucketCompletionTimeout, TimeUnit.MILLISECONDS);
434                 statusReporter.notify(COMPLETING_BUCKET);
435                 completeBucket(bucketId, null);
436                 pollStatus(bucketId);
437             }
438             statusReporter.done();
439         } catch (SnapDataException e) {
440             errorViews.write(e);
441         } catch (java.util.concurrent.ExecutionException | InterruptedException e) {
442             SnapDataException sde = new SnapDataException(e, ERROR_UPLOAD_FILES)
443                     .withReason(Throwables.getRootCause(e).getMessage())
444                     .withResolution(RESOLUTION_ADDRESS_ISSUE);
445             errorViews.write(sde);
446         }
447

449     }

451     @Override
452     protected void process(Document document, String inputViewName) {
453         if (tempFile == null) {
454             createTempFileRetry();
455         }
456         long curFileSize = tempFile.length();
457         tmpFileReport.worked(curFileSize);
458         if (curFileSize >= MAX_FILE_SIZE_IN_BYTES) {
459             LOG.debug("File size ({}) exceed maximum ({}). wrapping up current file and recreate " +
460                     "another one", curFileSize, MAX_FILE_SIZE_IN_BYTES);
461             try {
462                 document2CSVConverter.close();
463             } catch (IOException e) {
464                 LOG.warn("Error happened when closing the Csv Converter: {}", e.getMessage());
465             }
466             doUpload(tempFile, bucketId);
467             createTempFileRetry();
468             statusReporter.notify(ACCEPTING_DATA);
469         }
470         Document modifiedDocument = applyDatetimeFormat(document);
471         Failsafe.with(getRetryPolicy())
472                 .onRetry(throwable -> {
473                     LOG.warn(RETRY_FAILURE_MESSAGE, SNAPPACK_NAME, "accept data",
474                             SNAPPACK_NAME, throwable.getClass().getName(), throwable.getMessage());
475                 })
476                 .run(cntx -> {
477                     LOG.warn(RETRY_MESSAGE, SNAPPACK_NAME, cntx.getExecutions());
478                     document2CSVConverter.write(modifiedDocument);
479                 });

483     /**
484      * Generate a new document that has all datetime columns transformed to string according to
485      * their parseFormat defined in schema
486      */
487     private Document applyDatetimeFormat(Document document) {
488         // Make a copy of the document here since we don't want to modify the original document
489         Document modifiedDocument = document.copy();
490         Map<String, Object> data = documentUtility.getAsMap(modifiedDocument, errorViews);
491         for(String key: dateColumnFormats.keySet()) {
492             if(data.containsKey(key)) {
493                 Object value = data.get(key);
494                 if (value instanceof DateTime || value instanceof LocalDateTime
495                         || value instanceof LocalDate) {
496                     // The classes above all have toString(pattern) method to format the
497                     // object into a string base on a timestamp pattern. but they don't share a
498                     // common class. So call it through reflection
499                     try {
500                         Method toStringWithPatten = value.getClass().getMethod("toString",
501                                 String.class);
502                         data.put(key, toStringWithPatten.invoke(value, dateColumnFormats.get(key)));
503                     } catch (NoSuchMethodException | IllegalAccessException
504                             | InvocationTargetException e) {
505                         LOG.warn("Failed to invoke toString(pattern) method on value in type: {}," +
506                                         " reason: {}. Skipping", value.getClass(), e);
507                     }
508                 }
509             }
510         }
511         return modifiedDocument;
512     }

751     /**
752      * Extract the field defination from the response of describeDataset request. This is
753      * currently used by API v2 (Table API) only, since table is supposed to be schema pre-defined
754      *
755      */
756     private Map<String, Object> getSchemaFromRemoteTable(String tableID) {
757         Map<String, Object> tableMetaData = describeDataset(tableID);
758         String fieldsPathStr =
759                 new JsonPathBuilder(DATA_KEY).appendListIndex(0).appendCurrentElement(FIELDS_KEY).build();
760         JsonPath fieldsPath = JsonPathImpl.compileStatic(fieldsPathStr);
761         try {
762             List<Map<String, Object>> bucketSchema = Lists.newArrayList();
763             List<Map<String, Object>> fields = fieldsPath.read(tableMetaData);
764             // Use an array to preserve the order of fields.
765             JsonPath[] orderedFieldPath = new JsonPath[fields.size()];
766             int realFieldsSize = 0;
767             if(CollectionUtils.isNotEmpty(fields)) {
768                 for (Map<String, Object> fieldMetadata : fields) {
769                     String fieldName = (String) fieldMetadata.get(NAME_SCHEMA_KEY);
770                     Number ordinal = (Number) fieldMetadata.get(ORDINAL_SCHEMA_KEY);
771                     if(fieldName != null && !fieldName.startsWith(PRISM_RESERVED_NAME_PREFIX)) {
772                         // WPA_ is a reserved prefix for Prism to store some internal metadata
773                         Map<String, Object> filteredSchema = fieldMetadata.entrySet()
774                                 .stream()
775                                 .filter(stringObjectEntry ->
776                                         FIELDS_DEFINATION_KEYS.contains(stringObjectEntry.getKey()))
777                                 .collect(Collectors.toMap(Map.Entry::getKey, Map.Entry::getValue));
778                         // record the format of the datetime field
779                         if(filteredSchema.containsKey(PARSE_FORMAT_SCHEMA_KEY)) {
780                             dateColumnFormats.put(fieldName,
781                                     (String)filteredSchema.get(PARSE_FORMAT_SCHEMA_KEY));
782                         }
783                         bucketSchema.add(filteredSchema);
784                         // Ordinal value starts from 1
785                         orderedFieldPath[ordinal.intValue() - 1] = JsonPaths.compile(fieldName);
786                         ++ realFieldsSize;
787                     }
788                 }
789                 for (int i = 0; i < realFieldsSize; i++) {
790                     columnNames.add(orderedFieldPath[i]);
791                 }
792 
793             }
794             Map<String, Object> schemaMap = Maps.newLinkedHashMap();
795             schemaMap.put(FIELDS_KEY, bucketSchema);
796             return schemaMap;
797         } catch (InvalidPathException e) {
798             // Ideally the path should be valid, unless there are some changes in the Prism's
799             // response structure.
800             throw new SnapDataException(e, ERR_SCHEMA_FROM_REMOTE_TABLE)
801                     .withResolutionAsDefect();
802         }
803     }

824     private Map<String, Object> getSchemaFromView(final InputView schemaView) {
825         statusReporter.notify(GENERATING_SCHEMA);
826         List<Map<String, Object>> fieldsList = Lists.newArrayList();
827         SnapDataException schemaException = null;
828         int ordinal = 1;
829         for (Document document : inputViews.documents(schemaView)) {
830             if (schemaException != null) {
831                 continue;
832             }
833             try {
834                 Map<String, Object> data = document.get(Map.class);
835                 // Required field
836                 String name = NAME_SCHEMA_PATH.read(data);
837                 String typeLiteral = TYPE_LITERAL_SCHEMA_PATH.read(data);
838                 if (name == null || typeLiteral == null) {
839                     schemaException = new SnapDataException(ERROR_GENERATING_SCHMEA_FROM_SCHEMA_VIEW)
840                             .withReason(REASON_MISSING_MANDATORY_FIELD_SCHEMA)
841                             .withResolution(RESOLUTION_FOLLOW_SCHEMA_FORMAT);
842                 }
843                 columnNames.add(JsonPaths.compile((Object) name));
844 
845                 // Optionals
846                 String description = DESCRIPTION_SCHEMA_PATH.read(data, StringUtils.EMPTY);
847                 String parseFormat = PARSE_FORMAT_PATH.read(data, StringUtils.EMPTY);
848                 Number precision = PRECISION_SCHEMA_PATH.read(data, DEFAULT_NUMBER_PRECISION);
849                 Number scale = SCALE_SCHEMA_PATH.read(data, DEFAULT_NUMBER_SCALE);
850                 fieldsList.add(createSchemaEntry(ordinal, name, description, typeLiteral,
851                         parseFormat, precision.intValue(), scale.intValue()));
852                 if(StringUtils.isNotEmpty(parseFormat)) {
853                     dateColumnFormats.put(name, parseFormat);
854                 }
855                 ++ordinal;
856             } catch (FieldNotFoundException e) {
857                 schemaException = new SnapDataException(e, ERROR_GENERATING_SCHMEA_FROM_SCHEMA_VIEW)
858                         .withReason(String.format(REASON_MISSING_KEY_IN_SCHEMA, e.getField()))
859                         .withResolution(RESOLUTION_FOLLOW_SCHEMA_FORMAT);
860             } catch (InvalidPathException e) {
861                 schemaException = new SnapDataException(e, ERROR_GENERATING_SCHMEA_FROM_SCHEMA_VIEW)
862                         .withReason(e.getMessage())
863                         .withResolution(RESOLUTION_FOLLOW_SCHEMA_FORMAT);
864 
865             }
866         }
867         if (schemaException != null) {
868             throw schemaException;
869         }
870         if (fieldsList.isEmpty()) {
871             // If no document is provided, then we consider no schema provided here
872             return null;
873         }
874         Map<String, Object> schemaMap = Maps.newLinkedHashMap();
875         schemaMap.put(PARSE_OPTIONS_KEY, getParseOptions());
876         schemaMap.put(FIELDS_KEY, fieldsList);
877         return schemaMap;
878     }

881         PrismDataType type = getPrismType(value);
882         String typeLiteral = String.format(TYPE_ID_LITERAL_FMT, type);

895                     format = DEFAULT_DATE_FORMAT;

898                         format, 0, 0);
899             case BOOLEAN:
900             case TEXT:
