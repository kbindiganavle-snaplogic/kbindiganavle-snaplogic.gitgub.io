557     /**
558      * Process the response received from executing the HTTP request
559      */
560     private Map<String, String> processResponse(Map<String, Object> response,
561             ApiMethod apiMethod) {
562         Map<String, String> responseMap = Maps.newLinkedHashMap();
563         Map<String, String> errorMap = Maps.newLinkedHashMap();
564         if (response == null) {
565             return responseMap;
566         }
567         switch (apiMethod) {
568             case CREATE_DATASET:
569                 // Retrieve id of the dataset created
570                 if (response.containsKey(ID)) {
571                     responseMap.put(ID, response.get(ID).toString());
572                     responseMap.put(NAME, response.get(NAME).toString());
573                     return responseMap;
574                 } else if (response.containsKey(ERROR)) {
575                     responseMap.put(ERROR, response.get(ERROR).toString());
576                     ArrayList<Map<String, Object>> errorList =
577                             (ArrayList<Map<String, Object>>) response.get(ERRORS);
578                     for (Map<String, Object> data : errorList) {
579                         errorMap.put(ERROR_MSG, data.get(ERROR).toString());
580                     }
581                     return errorMap;
582                 }
583             case GET_BUCKETS:
584                 if (response.containsKey(ID)) {
585                     responseMap.put(ID, response.get(ID).toString());
586                     responseMap.put(NAME, response.get(NAME).toString());
587                     Map<String, Object> statsMap = (Map<String, Object>) response.get(STATS);
588                     if (MapUtils.isNotEmpty(statsMap)) {
589                         responseMap.put(TOTAL_ROW_COUNT,
590                                 String.valueOf(statsMap.get(TOTAL_ROW_COUNT)));
591                         responseMap.put(BUCKET_SIZE, statsMap.get(BUCKET_SIZE).toString());
592                         responseMap.put(PROCESSED_FILES, statsMap.get(PROCESSED_FILES).toString());
593                         if(statsMap.containsKey(ERROR_ROW_COUNT)) {
594                             responseMap.put(ERROR_ROW_COUNT,
595                                     statsMap.get(ERROR_ROW_COUNT).toString());
596                         }
597                     }
598                     Map<String, Object> stateMap = (Map<String, Object>) response.get(STATE);
599                     String state = stateMap.get(DESCRIPTOR).toString();
600                     responseMap.put(STATE, state);
601                     if(!StringUtils.equalsIgnoreCase(SUCCESS, state)
602                             && response.containsKey(ERROR_MSG)) {
603                         // The Success response also contains a errorMessage key, which doesn't
604                         // seem right but does exist. So need to specifically exclude Success
605                         // response
606                         responseMap.put(ERROR_MSG, String.valueOf(response.get(ERROR_MSG)));
607                     }
608                     return responseMap;
609                 } else if (response.containsKey(ERROR)) {
610                     ArrayList<Map<String, Object>> errorList =
611                             (ArrayList<Map<String, Object>>) response.get(ERRORS);
612                     for (Map<String, Object> data : errorList) {
613                         errorMap.put(ERROR_MSG, data.get(ERROR).toString());
614                     }
615                     return errorMap;
616                 }
617             case CREATE_BUCKET:
618                 if (response.containsKey(ID)) {
619                     responseMap.put(ID, response.get(ID).toString());
620                     responseMap.put(NAME, response.get(NAME).toString());
621                     return responseMap;
622                 } else if (response.containsKey(ERROR)) {
623                     responseMap.put(ERROR, response.get(ERROR).toString());
624                     ArrayList<Map<String, Object>> errorList =
625                             (ArrayList<Map<String, Object>>) response.get(ERRORS);
626                     for (Map<String, Object> data : errorList) {
627                         responseMap.put(ERROR_MSG, data.get(ERROR).toString());
628                         if (data.containsKey(FIELD)) {
629                             responseMap.put(FIELD, data.get(FIELD).toString());
630                         }
631                         if (data.containsKey(LOCATION)) {
632                             responseMap.put(LOCATION, data.get(LOCATION).toString());
633                         }
634                     }
635                     return errorMap;
636                 }
637             case GET_DATASETS:
638             case UPLOAD_FILE:
639             case COMPLETE_BUCKET:
640             default:
641                 return responseMap;
642         }

750     public class PrismResponseProcessor extends RestResponseProcessor {
751         private static final String STATUS_CODE = "statusCode";
752 
753         public PrismResponseProcessor(final XmlUtils xmlUtils, final String method,
754                 final ErrorViews errorViews, final DocumentUtility documentUtility) {
755             super(xmlUtils, method, null, errorViews, documentUtility);
756         }
757 
758         /**
759          * Simplified version of processing response. Throwing error out instead of writing to
760          * error view
761          *
762          * @param httpResponse   HTTP response received from REST API service endpoint
763          * @param document       as the incoming document
764          * @param requestHeaders as the originating request headers
765          * @param url
766          *
767          * @return
768          */
769         @Override
770         public Object process(final HttpResponse httpResponse, final Document document,
771                 final Header[] requestHeaders, final String url) {
772             Map<String, Object> statusLineMap = getStatusLineMap(httpResponse);
773             Map<String, Object> headers = getHeaderMap(httpResponse.getAllHeaders());
774             HttpEntity entity = httpResponse.getEntity();
775             String content = null;
776             int statusCode = httpResponse.getStatusLine().getStatusCode();
777             try {
778                 if (statusCode >= SC_OK && statusCode <= SC_TEMPORARY_REDIRECT) {
779                     try {
780                         Object result = processEntity(entity, document, statusLineMap, headers,
781                                 requestHeaders);
782                         if (document != null && isWriter) {
783                             document.acknowledge();
784                         }
785                         return result;
786                     } catch (SuggestViewAbortException e) {
787                         // should abort entity processing gracefully when Suggest reaches the max number
788                         // of documents
789                         return null;
790                     } catch (SnapDataException e) {
791                         throw e;
792                     } catch (Exception e) {
793                         throw new SnapDataException(e, ERROR_PROCESSING_PRISM_RESP)
794                                 .withReason(e.getMessage())
795                                 .withResolution(RESOLUTION_ADDRESS_ISSUE);
796                     }
797                 } else {
798                     String error = EntityUtils.toString(entity);
799                     String reason = String.format(ERR_REST_API_REASON_PHRASE,
800                             httpResponse.getStatusLine().getReasonPhrase(), error);
801                     Map<String, Object> errorMap = createErrorMap(reason, statusCode);
802                     if (error != null) {
803                         errorMap.put(ERROR_ENTITY, error);
804                         reason = reason.concat(ERR_ERROR_ENTITY);
805                     }
806                     if (requestHeaders != null) {
807                         errorMap.put(REQUEST_HEADERS, maskHeaders(Arrays.asList(getHeaderMap
808                                 (requestHeaders))));
809                     }
810                     errorMap.put(KEY_HEADERS, headers);
811                     throw new SnapDataException(
812                             documentUtility.newDocument(errorMap), ERR_REST_API)
813                             .formatWith(statusCode)
814                             .withReason(reason)
815                             .withResolution(RESOLUTION_ADDRESS_ISSUE);
816                 }
817             } catch (SnapDataException e) {
818                 throw e;
819             } catch (Exception e) {
820                 StatusLine statusLine = httpResponse.getStatusLine();
821                 String reason = Throwables.getRootCause(e).getMessage();
822                 SnapDataException ex;
823                 if (!isWriter && StringUtils.isNotBlank(content)) {
824                     Map<String, Object> errorMap = Maps.newHashMap();
825                     errorMap.put(KEY_ERROR, ERROR_PROCESS_PRISM_REQ);
826                     errorMap.put(KEY_ENTITY, content);
827                     errorMap.put(REASON, reason);
828                     errorMap.put(STATUS_CODE, statusLine.getStatusCode());
829                     errorMap.put(KEY_HEADERS, headers);
830                     errorMap.put(KEY_SERVICE_URL, url);
831                     errorMap.put(KEY_STATUS_LINE, statusLineMap);
832                     ex = new SnapDataException(documentUtility.newDocument(errorMap), e,
833                             ERROR_PROCESS_PRISM_REQ)
834                             .withReason(reason)
835                             .withResolution(RESOLUTION_ADDRESS_ISSUE);
836                 } else {
837                     ex = new SnapDataException(e, ERROR_PROCESS_PRISM_REQ)
838                             .withReason(reason)
839                             .withResolution(RESOLUTION_ADDRESS_ISSUE);
840                 }
841                 throw ex;
842             }
843         }
