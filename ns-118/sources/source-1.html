13 /**
14  * Match Condition that predefined regular expression for credit card
15  */
16 public class CreditCardMatcher extends RegexMatcher {
17 
18     private static String CREDITC_CARD_PATTERN = "^" +
19             "(?:4[0-9]{12}(?:[0-9]{3})?" +     // Visa
20             "|(?:5[1-5][0-9]{2}|222[1-9]|22[3-9][0-9]|2[3-6][0-9]{2}|27[01][0-9]|2720)[0-9]{12}" + // Master Card
21             "|3[47][0-9]{13}" +   //American Express
22             "|3(?:0[0-5]|[68][0-9])[0-9]{11}" +  // Diners Club
23             "|6(?:011|5[0-9]{2})[0-9]{12}" +     // Discover
24             "|(?:2131|1800|35\\d{3})\\d{11}" +   // JCB
25             ")$";
26 
27     public CreditCardMatcher() {
28         addPattern(CREDITC_CARD_PATTERN);
29     }
30 
31     @Override
32     protected String transformInput(Object value) {
33         String matchValue = super.transformInput(value).trim();
34 
35         String[] noGroups = matchValue.split("[-]|[ ]");
36         boolean keepGroupOf4 = true;
37         for (String noGroup : noGroups) {
38 
39             // If group of number that not equal 4 should be found only once at the end
40             // if found again, transform to null string which ensure that match will fail
41             if(!keepGroupOf4)
42                 return "";
43 
44             if(noGroup.length() != 4){
45                 keepGroupOf4 = false;
46             }
47         }
48 
49         return matchValue.replaceAll("[-]|[ ]", "");
50     }
