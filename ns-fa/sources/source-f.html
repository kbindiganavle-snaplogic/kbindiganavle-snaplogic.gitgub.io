271         super.process(document, s);
272         this.document = document;
273         Object object = expBody.eval(document);
274         if (object instanceof String) {
275             body = (String) object;
276         } else {
277             try {
278                 body = MAPPER.writeValueAsString(object);
279             } catch (IOException e) {
280                 String reason = String.format(ERR_JSON_REASON, e.getMessage());
281                 throw new ExecutionException(ERR_JSON)
282                         .withReason(reason)
283                         .withResolution(ERR_CUSTOMER_SUPPORT);
284             }
285         }
286 
287         if (document != null && HTML_TABLE.equals(type)) {
288             Object data = JsonPath.read(document.get(), tableJsonPath);
289             if (data instanceof Map) {
290                 tableDataList.add((Map<String, Object>) data);
291                 rows++;
292             } else if (data instanceof List) {
293                 for (Object value : (List) data) {
294                     if (value instanceof Map) {
295                         tableDataList.add((Map<String, Object>) value);
296                         rows++;
297                     } else {
298                         writeErrorView(document, value);
299                         return;
300                     }
301                 }
302             } else {
303                 writeErrorView(document, data);
304                 return;
305             }
306             if (rows < batchSize) {
307                 return;
308             }
309             prepareTable();
310             rows = 0;
311         }
312         sendMail(document);
313     }

364                             EmailAddress emailAddress = new EmailAddress();
365                             emailAddress.address = email.trim();

369             }
370             if (StringUtils.isNotBlank(bcc)) {
371                 bcc = trimSquareBrackets(bcc);
372                 message.bccRecipients = Stream.of(bcc.split(COMMA))
373                         .map(email -> {
374                             Recipient recipient = new Recipient();
375                             EmailAddress emailAddress = new EmailAddress();
376                             emailAddress.address = email.trim();
377                             recipient.emailAddress = emailAddress;
378                             return recipient;
379                         }).collect(Collectors.toCollection(LinkedList::new));
380             }
381             try {
382                 Failsafe.with(createRetryPolicy(document))
383                         .onFailedAttempt(e -> LOG.warn(ERR_SEND_MSG, e))
384                         .run(context -> {
385                             logRetryAttempt(context);
386                             invokeSendEmailAPI(document);
387                         });
388             } catch (Exception e) {
389                 throw e instanceof FailsafeException ? e.getCause() : e;
390             }
391         } catch (SnapDataException snapEx) {
392             throw snapEx;
393         } catch (Throwable ex) {
394             LOG.error(ERR_SEND_MSG, ex);
395             String reason = (ex.getCause() != null && ex.getCause().getMessage() != null) ?
396                     ex.getCause().getMessage() : ex.getMessage();
397             throw new SnapDataException(ex, ERR_SEND_MSG)
398                     .withReason(reason)
399                     .withResolution(ERR_COMMON_RESOLUTION);
400         }
401     }

763     private String trimSquareBrackets(String emailAddressList) {
764         if (Objects.isNull(emailAddressList)) {
765             return emailAddressList;
766         }
