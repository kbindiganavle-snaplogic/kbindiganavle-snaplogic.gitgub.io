366     @Override
367     protected void doWork() {
368         Map<String, Object> headerMap
369                 = ImmutableMap.<String, Object>builder()
370                 .put(SdkConstants.CONTENT_TYPE.toLowerCase(),
371                         CONTENT_TYPE_PREFIX + outputCharacterSet.toLowerCase())
372                 .build();
373         if (StringUtils.isNotEmpty(rootElementName)) {
374             // The user wants a single binary document with all of the input documents collected
375             // into it.
376             doCollectingWork(headerMap);
377         } else {
378             // The user wants separate binary documents for each input document.
379             for (final Document document : inputViews.documents(inputViews.get())) {
380                 final Document header = documentUtility.newDocumentFor(document,
381                         new LinkedHashMap<>(headerMap));
382                 log.info(OUTPUTTING_DOCUMENT_WITH_ID, document.getMetadata().getId());
383                 outputViews.write(new BinaryOutput() {
384                     @Override
385                     public Document getHeader() {
386                         return header;
387                     }
388                     @Override
389                     public void write(final WritableByteChannel writeChannel) throws IOException {
390                         XMLEventWriter xmlWriter;
391 
392                         // Add optional canonical xml output support,
393                         // based on isCanonicalXmlOutput property.
394                         OutputStream os = createOutputStream(writeChannel);
395 
396                         xmlWriter = getXmlEventWriter(os);
397                         if (xmlWriter != null) {
398                             try {
399                                 log.info(WRITING_DOCUMENT_START);
400                                 Map<String, Object> data = (Map<String, Object>)document.get();
401                                 data = (Map<String, Object>)handleAttributesInData(data);
402                                 xmlWriter.add(eventFactory.createStartDocument(outputCharacterSet));
403                                 handleDocument(xmlWriter, documentUtility.newDocument(data));
404                                 log.info(WRITING_DOCUMENT_END);
405                                 xmlWriter.add(eventFactory.createEndDocument());
406                                 log.info(CLOSING_XML_WRITER_STREAM);
407                             } catch (SnapDataException sde) {
408                                 xmlWriter = handleSnapDataException(xmlWriter, sde, document);
409                             } catch (XMLStreamException | RuntimeException e) {
410                                 SnapDataException sde = new SnapDataException(e,
411                                         COULD_NOT_CONVERT_DOCUMENT)
412                                         .withReason(Throwables.getRootCause(e).getMessage())
413                                         .withResolution(PLEASE_VERIFY_THE_DOCUMENT);
414                                 xmlWriter = handleSnapDataException(xmlWriter, sde, document);
415                             } catch (Exception e) {
416                                 String msg = e.getMessage();
417                                 String error = msg == null ? e.toString() : msg;
418                                 SnapDataException sde = new SnapDataException(e,
419                                         EXCEPTION_WRITING_XML)
420                                         .withReason(error)
421                                         .withResolution(EXCEPTION_WRITING_XML_REASON);
422                                 xmlWriter = handleSnapDataException(xmlWriter, sde, document);
423                             }
424                             closeStreamWriteXml(writeChannel, xmlWriter, os);
425                         }
426                     }
427                 });
428             }
429         }
430     }

474     private void doCollectingWork(Map<String, Object> headerMap) {
475         final Iterator<Document> documentIterator = inputViews.getDocumentsFrom(inputViews.get());
476         final boolean hasDocuments = documentIterator.hasNext();
477 
478         if (ignoreEmptyStream && !hasDocuments) {
479             return;
480         }
481         log.debug(CREATING_A_XML_OUTPUT_STREAM_WRITER);
482         // This is the first document. Write a binary output.
483         outputViews.write(new BinaryOutput() {
484             @Override
485             public Document getHeader() {
486                 return documentUtility.newDocument(new LinkedHashMap<>(headerMap));
487             }
488 
489             @Override
490             public void write(final WritableByteChannel writeChannel) throws IOException {
491                 XMLEventWriter xmlWriter;
492 
493                 // Add optional canonical xml output support,
494                 // based on isCanonicalXmlOutput property.
495                 OutputStream os = createOutputStream(writeChannel);
496 
497                 xmlWriter = getXmlEventWriter(os);
498                 Document document = null;
499                 if (xmlWriter != null) {
500                     try {
501                         log.info(WRITING_DOCUMENT_START);
502                         // first document is added according to selected outputCharacterSet
503                         xmlWriter.add(eventFactory.createStartDocument(outputCharacterSet));
504                         log.info(WRITING_START_ELEMENT, rootElementName);
505                         xmlWriter.add(eventFactory
506                                 .createStartElement(EMPTY, EMPTY, rootElementName));
507                         for ( int i = 0; i < intermediateNames.size(); i++) {
508                             xmlWriter.add(eventFactory.createStartElement(EMPTY,
509                                     EMPTY, intermediateNames.get(i)));
510                         }
511                         if (hasDocuments) {
512                             do {
513                                 document = documentIterator.next();
514                                 if (document != null) {
515                                     document.getMetadata().getId();
516                                     Map<String, Object> data = (Map<String, Object>)document.get();
517                                     data = (Map<String, Object>)handleAttributesInData(data);

707     private void initSchema() {
708         if (StringUtils.isNotBlank(schemaFileAbsolutePath) && strictXsdOutput) {
709             jsonSchema = xsd2JsonSchema(jsonSchema, schemaFileAbsolutePath, urlEncoder,
710                     typesFactory);
711             List<JsonSchemaElement> jsonSchemaElements = jsonSchema.getLeafSchemaElements();
712             if (CollectionUtils.isNotEmpty(jsonSchemaElements)) {
713                 // find a json path with maximum depth
714                 Integer max = 0;
715                 List<String> paths = new ArrayList<>();
716                 String maxPath = StringUtils.EMPTY;
717                 for (JsonSchemaElement jsonSchemaElement : jsonSchemaElements) {
718                     String elementPath = jsonSchemaElement.getPath();
719                     List<PathToken> elementTokens = getTokensFrom(elementPath);
720                     paths.add(elementPath);
721                     int size = elementTokens.size();
722                     if (size > max) {
723                         max = size;
724                         maxPath = elementPath;
725                     }
726                 }
727                 int min = max;
728                 String minPath = maxPath;
729                 // make maxPath a root element json path
730                 maxPath = maxPath.substring(0, maxPath.indexOf('.', 2));
731                 // find a json path of minimum depth(This only includes the common path)
732                 // and starting with root element
733                 for (String path : paths) {
734                     if (path.startsWith(maxPath)) {
735                         List<PathToken> elementTokens = getTokensFrom(path);
736                         if (elementTokens != null && min > elementTokens.size()) {
737                             min = elementTokens.size();
738                             minPath = path;
739                         }
740                     } else {
741                         minPath = maxPath;
742                         min = getTokensFrom(minPath).size();
743                         break;
744                     }
745                 }
746                 List<PathToken> minPathTokens = getTokensFrom(minPath);
747                 boolean isMinPathFound = false;
748                 for (String path : paths) {
749                     if (path.startsWith(maxPath)) {
750                         List<PathToken> elementTokens = getTokensFrom(path);
751                         for (int i = 0; i < minPathTokens.size() - 1; i++) {
752                             if (!minPathTokens.get(i).equals(elementTokens.get(i))) {
753                                 minPath = path;
754                                 min = i + 1;
755                                 isMinPathFound = true;
756                                 break;
757                             }
758                         }
759                     }
760                     if (isMinPathFound) {
761                         break;
762                     }
763                 }
764 
765                 // get rootElementName, wrapperName and intermediateNames from minPath
766                 List<PathToken> tokens = getTokensFrom(minPath);
767                 if (min == 3) {
768                     wrapperName = tokens.get(1).toString();
769                 } else if (min > 3) {
770                     rootElementName = tokens.get(1).toString();
771                     for (int i = 2; i < min - 1; i++) {
772                         String name = tokens.get(i).toString();
773                         intermediateNames.add(name);
774                     }
775                     if (wrapperName == null && !intermediateNames.isEmpty()) {
776                         wrapperName =
777                                 intermediateNames.remove(intermediateNames.size() - 1);
778                     }
779                 }
780             }
781         }
782     }

803                 schemaVisitor.setPathDepth(maxSchemaLevels);

805                 Iterator<XSElementDecl> iterator = xmlSchemaSet.iterateElementDecls();
806                 while (iterator.hasNext()) {
807                     XSElementDecl xsElementDecl = iterator.next();

809                 }

815             }
816         }

843      * Method is used to sort attributes to the start of the parent element.
844      * @param data
845      * @return
846      */
847     public Object handleAttributesInData(Object data) {
848         if (data instanceof Map) {
849             Map<String, Object> mapData = (Map<String, Object>) data;
850             Map<String, Object> tempData = new LinkedHashMap<>();
851             tempData.putAll(mapData);
852             Map<String, Object> resultData = new LinkedHashMap<>();
853             mapData.entrySet().stream()
854                     .forEach(entry -> {
855                         if (entry.getKey().startsWith("@")) {
856                             resultData.put(entry.getKey(), entry.getValue());
857                             tempData.remove(entry.getKey());
858                         } else {
859                             Object sorted = handleAttributesInData(entry.getValue());
860                             tempData.put(entry.getKey(), sorted);
861                         }
862                     });
863             resultData.putAll(tempData);
864             return resultData;
865         } else if (data instanceof List) {
866             List<Object> listData = (List<Object>) data;
867             List<Object> resultList = new ArrayList<>();
868             listData.stream().forEach(item -> {
869                 Object sorted = handleAttributesInData(item);
870                 resultList.add(sorted);
871             });
872             return resultList;
873         }
874         return data;
875     }
