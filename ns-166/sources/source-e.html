76  * Snap that transforms incoming data using expressions. The snap takes in a two column table (an
77  * expression column and a target column). The snap will evaluate the expression and map it to
78  * the target path. Source values will be deleted after all the expressions have been evaluated.
79  * To preserve original value, it should be copied previously in a {@link StructuralTransform}.

168     @Override
169     public void configure(final PropertyValues propertyValues) throws ConfigurationException {
170         mappingRoot = propertyValues.getAsJsonPath(MAPPING_ROOT_PATH);
171         if (mappingRoot == null) {
172             // Be nice and assume the root.
173             mappingRoot = JsonPaths.root();
174         }
175         mappingTable = propertyValues.get(MAPPING_TABLE_PATH);
176         nullSafeAccess = Boolean.TRUE.equals(propertyValues.get(NULL_SAFE_ACCESS));
177         isPassThrough = Boolean.TRUE.equals(propertyValues.get(PASS_THROUGH_PROP));
178         mappings = Lists.newArrayListWithExpectedSize(mappingTable.size());
179         Set<JsonPath> knownTargets = new HashSet<>();
180         for (Map<String, Object> mapping : mappingTable) {
181             final JsonPath targetPath = propertyValues.getJsonPathFor(mapping, TARGET_PATH);
182             if (targetPath == null) {
183                 final JsonPath deletePath = propertyValues.getJsonPathFor(mapping, EXPRESSION_PROP);
184                 if (deletions.contains(deletePath)) {
185                     throw new ConfigurationException(PATH_TO_DELETE_ALREADY_EXISTS)
186                             .formatWith(deletePath.toString())
187                             .withReason(CANNOT_DELETE_A_PATH_MORE_THAN_ONCE)
188                             .withResolution(REMOVE_THE_DUPLICATE_PATH);
189                 }
190                 deletions.add(deletePath);
191             } else {
192                 ExpressionProperty prop = propertyValues.getExpressionPropertyFor(mapping,
193                         EXPRESSION_PROP);
194                 if (!knownTargets.add(targetPath)) {
195                     throw new ConfigurationException(TARGET_MAPPING_EXISTS)
196                             .formatWith(targetPath)
197                             .withReason(TARGET_MAPPING_REASON)
198                             .withResolution(TARGET_MAPPING_EXISTS_RES);
199                 }
200                 mappings.add(Pair.of(prop, targetPath));
201             }
202         }
203         results = Lists.newArrayListWithExpectedSize(mappings.size());
204         hasBinaryOutput = outputViews.get().getType() == ViewType.BINARY;
205 
206         if (hasBinaryOutput) {
207             // Check if the user wants to rewrite the binary stream.
208             writesContent = mappings.stream()
209                     .anyMatch(pair -> CONTENT_PATH.equals(pair.getRight()) ||
210                             ENTITY_PATH.equals(pair.getRight()));
211         }
212     }

219     @Override
220     protected void processBinary(Document document, String inputViewName,
221             ReadableByteChannel channel) {
222         Object data = null;
223         if (document != null) {
224             data = document.get();
225             if (data == null) {
226                 throw new SnapDataException(DATA_TRANSFORM_CANNOT_PROCESS_NULL_DATA)
227                         .withReason(NULL_DATA)
228                         .withResolution(PLEASE_CHECK_YOUR_INPUT_DATA);
229             }
230             outputRoot = data;
231         }
232         try {
233             // If there is no binary output or content is being rewritten, we need to read in the
234             // content to a byte array.
235             if ((!hasBinaryOutput || writesContent) && channel != null) {
236                 Map<String, Object> map;
237 
238                 if (data == null) {
239                     data = map = new LinkedHashMap<>();
240                 } else {
241                     map = (Map<String, Object>) data;
242                 }
243 
244                 try {
245                     byte[] bits = ByteStreams.toByteArray(Channels.newInputStream(channel));
246 
247                     map.put("content", bits);
248                 } catch (IOException e) {
249                     throw new SnapDataException(e, "Unable to read binary data from upstream snap");
250                 }
251             }
252 
253             // This is the object we'll write out, it can still be overridden if the mapping root
254             // and target path of an expression is '$'.
255             // Walk the mapping root path looking for values to map.
256             for (WalkResult walkResult : mappingRoot.walk(data)) {
257                 final Object element = walkResult.getValue();
258                 // We want the expressions to operate on the original data only, so we need
259                 // to make the changes in a second pass.
260                 try {
261                     for (Pair<ExpressionProperty, JsonPath> mapping : mappings) {
262                         ExpressionProperty expressionProperty = mapping.getLeft();
263                         Object result = expressionProperty.eval(document, element,
264                                 nullSafeAccess ? NULL_VALUE_HANDLER : null);
265                         results.add(result);
266                     }
267                     // The results of the expressions will be written into target.
268                     final Object target;
269                     if (!isPassThrough && document != null) {
270                         target = Maps.newLinkedHashMap();
271                     } else if (element != null) {
272                         target = element;
273                     } else {
274                         target = Maps.newLinkedHashMap();
275                     }
276                     // Replace the original value in the tree with the new target.
277                     outputRoot = walkResult.setValue(target);
278                     populateAndDeleteTarget(walkResult, target);
279                 } finally {
280                     results.clear();
281                 }
282             }
283             if (!(outputRoot instanceof List || outputRoot instanceof Map)) {
284                 throw new SnapDataException(THE_OUTPUT_DOCUMENT_IS_A_PRIMITIVE_VALUE)
285                         .formatWith(outputRoot)
286                         .withReason(THE_OUTPUT_DOCUMENT_MUST_BE_AN_ARRAY_OR_OBJECT)
287                         .withResolution(CHECK_FOR_TARGET_PATHS_THAT_WRITE_TO_THE_ROOT);
288             }
289             if (document == null) {
290                 document = documentUtility.newDocument();
291             }
292             if (outputRoot == null) {
293                 outputRoot = Maps.newLinkedHashMap();
294             }
295             document.set(outputRoot);
296             doWrite(document, channel);
297         } catch (InvalidPathException e) {
298             if (nullSafeAccess && !mappingRoot.isRoot()) {
299                 // Just write out the original document if we're null-safe and the mapping
300                 // root cannot be found.
301                 LOG.debug(LOG_INVALID_MAPPING_ROOT, mappingRoot.toString(), e);
302                 if (document == null) {
303                     document = documentUtility.newDocument(Maps.newLinkedHashMap());
304                 }
305                 doWrite(document, channel);
306             } else {
307                 throw new SnapDataException(e, INVALID_MAPPING_ROOT)
308                         .formatWith(e.getPath())
309                         .withReason(e.getMessage())
310                         .withResolution(e.getResolution());
311             }
312         } finally {
313             outputRoot = null;
314         }
315     }

345     /**
346      * Populates the target object with the results of the expressions and performs any deletions.
347      *
348      * @param walkResult The current result of the mapping root walk that we working on.  This is
349      *                   only needed if a target path is '$'.
350      * @param data the data being populated with the results of the mappings.
351      */
352     private void populateAndDeleteTarget(final WalkResult walkResult, Object data) {
353         int resultSize = results.size();
354         for (int index = 0; index < resultSize; index++) {
355             try {
356                 JsonPath path = mappings.get(index).getRight();
357                 Object value = results.get(index);
358 
359                 // Check if we're overwriting the 'data' value itself, in which case
360                 // we overwrite the original value.
361                 if (path.isRoot()) {
362                     outputRoot = walkResult.setValue(value);
363                 } else {
364                     if (value == data && isPassThrough) {
365                         throw new SnapDataException(
366                                 "Cannot write the document into itself")
367                                 .withReason("Creating a loop in a document is not supported")
368                                 .withResolution("Delete this mapping, disable Passthrough, or " +
369                                         "create a copy of the input document");
370                     }
371                     path.write(data, value);
372                 }
373             } catch (InvalidPathException e) {
374                 throw new SnapDataException(e, UNABLE_TO_WRITE_TO_JSON_PATH)
375                         .formatWith(e.getPath())
376                         .withReason(e.getMessage())
377                         .withResolution(e.getResolution());
378             }
379         }
380         for (JsonPath path: deletions) {
381             try {
382                 path.delete(data);
383             } catch (InvalidPathException e) {
384                 if (!nullSafeAccess) {
385                     throw new SnapDataException(e, Messages.UNABLE_TO_DELETE_JSON_PATH)
386                             .formatWith(e.getPath())
387                             .withReason(e.getMessage())
388                             .withResolution(e.getResolution());
389                 }
390             }
391         }
392     }

437     public void defineOutputSchema(final SchemaProvider provider) {
438         ObjectNode outSchema = new ObjectNode(JsonNodeFactory.instance);
439 
440         for (Pair<ExpressionProperty, JsonPath> pair : mappings) {
441             if (hasBinaryOutput && (CONTENT_PATH.equals(pair.getRight()) ||
442                     ENTITY_PATH.equals(pair.getRight()))) {
443                 // We don't want $content showing up in the output schema when there is a binary
444                 // output
445                 continue;
446             }
447 
448             try {
449                 ExpressionProperty expr = pair.getLeft();
450                 ObjectNode schema = new ObjectNode(JsonNodeFactory.instance);
451 
452                 if (expr.isExpression()) {
453                     schema = ExpressionUtil.generateSchema(schema, expr.getExpression());
454                 } else {
455                     schema.with("properties").put("type", "string");
456                 }
457 
458                 JsonPathSchemaVisitor visitor = new JsonPathSchemaVisitor(outSchema);
459 
460                 try {
461                     List<PathToken> tokens = PathParser.parse(pair.getRight().toString(), new
462                             JsonPathExpressionService());
463 
464                     visitor.visit(tokens.subList(1, tokens.size()));
465                     visitor.getSchema().setAll(schema.with("properties").with("#"));
466                 } catch (InvalidPathException e) {
467                     LOG.warn("Unable to parse path: {}", pair.getRight(), e);
468                 }
469             } catch (Throwable th) {
470                 LOG.warn("Unable to generate schema", th);
471             }
472         }
473 
474         try {
475             ExpressionUtil.finalizeSchema(outSchema, false);
476             ObjectSchema objectSchema = provider.createSchema().fromTree(mappingRoot, outSchema);
477             for (String viewName : provider.getRegisteredViewNames()) {
478                 provider.setProvidedSchemaForViewName(viewName, objectSchema);
479             }
480         } catch (Throwable th) {
481             LOG.warn("Unable to convert schema", th);
482         }
483     }

485     @Override
486     public void pullDownstreamInputSchema(final SchemaProvider schemaProvider, final String name,
487             final ObjectSchema downstreamSchema, final String view, final ObjectSchema
488             inputSchema) {
489         try {
490             // Convert the snaplogic Schema type into regular Jackson JsonNodes used by the
491             // JsonPathSchemaVisitor.
492             String schemaStr = downstreamSchema.toString();
493             ObjectNode schemaToModify = (ObjectNode) snapObjectMapper.readTree(schemaStr);
494 
495             // We need to remove properties under the mapping root since the Mapper will be
496             // responsible for writing to those properties.
497             if (isPassThrough) {
498                 // When PassThrough is enabled, we only remove properties that are targeted by
499                 // this mapper.  The remaining can be passed upstream.
500                 for (Pair<ExpressionProperty, JsonPath> mapping : mappings) {
501                     removePropertyAtPath(schemaToModify, mapping.getRight());
502                 }
503             } else {
504                 // Without PassThrough, we need to remove all the properties since the upstream
505                 // won't be able to write to them.
506                 removePropertyAtPath(schemaToModify, JsonPaths.all());
507             }
508 
509             // Replace the schema with our modified one
510             ObjectSchema modifiedSchema = schemaProvider.createSchema();
511             modifiedSchema.fromString(schemaToModify.toString());
512             inputSchema.merge(modifiedSchema);
513         } catch (Throwable th) {
514             LOG.error("Unable to propagate schema", th);
515         }
516     }

518     @Override
519     public void propagateUpstreamOutputSchema(final SchemaProvider schemaProvider, final String name,
520             final ObjectSchema upstreamSchema, final String view, final ObjectSchema outputSchema) {
521         if (isPassThrough) {
522             outputSchema.merge(upstreamSchema);
523 
524             String schemaStr = outputSchema.toString();
525             try {
526                 ObjectNode schemaToModify = (ObjectNode) snapObjectMapper.readTree(schemaStr);
527                 // With passthrough turned on, we need to remove any properties in the deletions
528                 for (JsonPath deleter : deletions) {
529                     removePropertyAtPath(schemaToModify, deleter);
530                 }
531 
532                 // Replace the schema with our modified one
533                 outputSchema.clear();
534                 outputSchema.fromString(schemaToModify.toString());
535             } catch (Throwable th) {
536                 LOG.error("Unable to propagate schema", th);
537             }
538         } else if (!mappingRoot.isRoot()) {
539             // If we have a mapping root, we need to pass through the upper-level schema that is not
540             // affected by the mapping root.
541             String schemaStr = upstreamSchema.toString();
542             try {
543                 ObjectSchema modifiedSchema = schemaProvider.createSchema();
544                 ObjectNode schemaToModify = (ObjectNode) snapObjectMapper.readTree(schemaStr);
545                 JsonPathSchemaVisitor visitor = new JsonPathSchemaVisitor(schemaToModify);
546                 List<PathToken> tokens = PathParser.parse(mappingRoot.toString(), new
547                         JsonPathExpressionService());
548 
549                 visitor.visit(tokens.subList(1, tokens.size()));
550                 Stack<ObjectNode> schemaStack = visitor.getSchemaStack();
551 
552                 schemaStack.pop();
553                 schemaStack.peek().removeAll();
554 
555                 // Replace the schema with our modified one
556                 modifiedSchema.fromString(schemaToModify.toString());
557                 outputSchema.merge(modifiedSchema);
558             } catch (Throwable th) {
559                 LOG.error("Unable to propagate schema", th);
560             }
561         }
562     }

564     private void removePropertyAtPath(final ObjectNode schemaToModify, final JsonPath right)
565             throws InvalidPathException {
566         JsonPath fullDeleter = mappingRoot.join(right);
567         JsonPathSchemaVisitor visitor = new JsonPathSchemaVisitor(schemaToModify);
568         List<PathToken> tokens = PathParser.parse(fullDeleter.toString(), new
569                 JsonPathExpressionService());
570 
571         visitor.visit(tokens.subList(1, tokens.size()));
572         Stack<ObjectNode> schemaStack = visitor.getSchemaStack();
573 
574         schemaStack.pop();
