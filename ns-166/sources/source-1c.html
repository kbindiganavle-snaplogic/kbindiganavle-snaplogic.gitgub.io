275     protected void process(final Document document, final String inputViewName) {
276         checkMemoryCount++;
277         totalDocumentCount++;
278         if (checkMemoryCount > CHECK_MEMORY_INTERVAL) {
279             checkMemoryCount = 0;
280             checkMemory(minMemory, outOfResourceTimeoutMillis, statusReporter, STATUS_MESSAGE,
281                     LOG, totalDocumentCount);
282         }
283         Object data = document.get();
284         if (!(data instanceof Map)) {
285             errorViews.write((SnapDataException) new SnapDataException(ERR_INVALID_INPUT_DATA)
286                     .withReason(REASON_NOT_MAP_DATA)
287                     .withResolution(RESOLUTION_ADDRESS_ISSUE), document);
288             return;
289         }
290         Map<String, Object> map = (Map<String, Object>) data;
291         currentGroupByValue.clear();
292         for (JsonPath field : fields) {
293             Object o;
294             try {
295                 o = field.read(map);
296                 field.write(currentGroupByValue, o);
297             } catch (InvalidPathException e) {
298                 errorViews.write((SnapDataException)
299                         new SnapDataException(e, ERR_INVALID_PATH).formatWith(field.toString())
300                                 .withReason(e.getMessage())
301                                 .withResolution(RESOLUTION_ADDRESS_ISSUE), document);
302                 continue;
303             }
304         }
305         // if this is the first document through, we haven't yet established a group-by document
306         if (initialDocument) {
307             // add doc to current batch
308             group.add(map);
309             initialDocument = false;
310             // and establish the group-by document
311             lastGroupByValue = Maps.newHashMap(currentGroupByValue);
312         } else if (currentGroupByValue.equals(lastGroupByValue)) {
313             // add to batch
314             group.add(map);
315         } else {
316             // batch changed, write it out then add current doc to new batch
317             // write() flushes the batch
318             write(group, lastGroupByValue);
319             group.add(map);
320             // and establish the new group-by document
321             lastGroupByValue = Maps.newHashMap(currentGroupByValue);
322         }
323         // in any case, we're done with this doc
324         document.acknowledge();
325     }

352     private Map<String, Object> toMap(StatsAccumulator accum, boolean intType) {
353         Map<String, Object> map = new LinkedHashMap<>();
354         if (intType) {
355             map.put("min", roundToBigInteger(accum.min(), UNNECESSARY));
356             map.put("max", roundToBigInteger(accum.max(), UNNECESSARY));
357         } else {
358             map.put("min", BigDecimal.valueOf(accum.min()).setScale(2, HALF_UP));
359             map.put("max", BigDecimal.valueOf(accum.max()).setScale(2, HALF_UP));
360         }
361         map.put("mean", BigDecimal.valueOf(accum.mean()).setScale(2, HALF_UP));
362         map.put("std", BigDecimal.valueOf(accum.populationStandardDeviation()).setScale(2, HALF_UP));
363         map.put("count", BigInteger.valueOf(accum.count()));
364         if (intType) {

370     private void write(final List<Object> objects, Map<String, Object> currentGroupedValue) {
371         try {
372             int groupSize = objects.size();
373             groupSizeAccum.add(groupSize);
374             List<Object> listCopy = new ArrayList<>(objects);
375             if (dynamicMode) {
376                 // The key metric used here: the fraction of max memory that's available.
377                 // The value ranges from 0 to 1.
378                 availMem = memoryUtil.availableMemoryFraction();
379                 availMemAccum.add(availMem);
380 
381                 // Calculate the moving average (MA) of availMem using memSamples, an EvictingQueue which
382                 // stores the most recent NUM_MEM_SAMPLES values of availMem. Current value is added below.
383                 double availMemMA = availMemSamples.isEmpty() ? availMem : meanOf(availMemSamples);
384 
385                 // Calculate how much the current value deviates from the moving average.
386                 double availMemDelta = availMem - availMemMA;
387 
388                 // Now that we've already used memSamples to calculate the MA, add the new sample to it.
389                 availMemSamples.add(availMem);
390 
391                 // Check whether we should trigger a garbage collection.
392                 if (memoryUtil.requestGC(availMem)) {
393                     gcCount++;
394                 }
395 
396                 // Calculate the sigmoid function for the partial value for the groupSizeFraction.
397                 // This value also ranges from 0 to 1.
398                 double sigmoidResult = sigmoidFunc.applyAsDouble(availMem);
399 
400                 // Calculate the groupSizeFraction by adding two components:
401                 // sigmoidResult, based on the current available memory, and
402                 // availMemDelta, how much the current value deviates from the moving average.
403                 // Including the availMemDelta helps to dampen spikes (rapid increases/decreases).
404                 double groupSizeFraction = sigmoidResult + (availMemDelta * DELTA_WEIGHT);
405 
406                 // Set maxPartSize to the group size mean plus SD_WEIGHT standard deviations.
407                 double mean = groupSizeAccum.mean();
408                 double sd = groupSizeAccum.populationStandardDeviation();
409                 int maxPartSize = roundToInt(mean + (sd * SD_WEIGHT), DOWN);
410 
411                 // Multiply maxPartSize by sigmoidResult to get sizeLimit.
412                 int sizeLimit = roundToInt(maxPartSize * groupSizeFraction, HALF_UP);
413 
414                 // Constrain sizeLimit to the range [minPartSize..maxPartSize].
415                 sizeLimit = max(minPartSize, min(sizeLimit, maxPartSize));
416 
417                 int numParts = roundToInt(groupSize / (double) sizeLimit, UP);
418                 numPartsAccum.add(numParts);
419 
420                 int partSize = numParts == 1 ? groupSize : max(minPartSize, groupSize / numParts);
421                 partSizeAccum.add(partSize);
422 
423                 for (int iPart = 1; iPart <= numParts; iPart++) {
424                     int fromIndex = (iPart - 1) * partSize;
425                     int toIndex = (iPart == numParts ? groupSize : fromIndex + partSize);
426                     List<Object> part = listCopy.subList(fromIndex, toIndex);
427                     Map<String, Object> data = new LinkedHashMap<>();
428                     data.put(GROUP_BY, currentGroupedValue);
429                     Map<String, Object> partInfo = new LinkedHashMap<>();
430                     partInfo.put(PART_INDEX, BigInteger.valueOf(iPart));
431                     partInfo.put(NUM_PARTS, BigInteger.valueOf(numParts));
432                     partInfo.put(PART_SIZE, BigInteger.valueOf(part.size()));
433                     partInfo.put(GROUP_SIZE, BigInteger.valueOf(groupSize));
434                     data.put(PART_INFO, partInfo);
435                     targetField.write(data, part);
436                     mainView.write(documentUtility.newDocument(data));
437                 }
438             } else {
439                 Map<String, Object> data = new LinkedHashMap<>();
440                 data.put(GROUP_BY, currentGroupedValue);
441                 targetField.write(data, listCopy);
442                 mainView.write(documentUtility.newDocument(data));
443             }
444             if (groupSizeAccum.count() % UPDATE_STATUS_FREQUENCY == 0) {
445                 updateStatus();
446             }
447             objects.clear();
448         } catch (InvalidPathException e) {
449             LOG.error(ERR_WRITE_GROUP, e);
450             errorViews.write((SnapDataException) new SnapDataException(e, ERR_WRITE_GROUP)
451                     .withReason(e.getMessage())
452                     .withResolution(RESOLUTION_ADDRESS_ISSUE));
453         }
454     }
