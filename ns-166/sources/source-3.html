203     public void configure(PropertyValues propertyValues) throws ConfigurationException {
204         try {
205             isSuggest = expressionUtils.createExpressionProperty(
206                             propertyValues, true,
207                             "pipe.flags.is_suggest")
208                     .eval(null);
209         } catch (Exception e) {
210             isSuggest = false;
211         }
212 
213         this.profiler = new Profiler();
214         this.index = 0L;

216         mappingRoot = JsonPaths.root();
217         mappingTable = propertyValues.get(MAPPING_TABLE_PATH);
218         mappings = Lists.newArrayListWithExpectedSize(mappingTable.size());
219 
220         Set<JsonPath> knownTargets = new HashSet<>();
221         for (Map<String, Object> mapping : mappingTable) {
222             final JsonPath targetPath = propertyValues.getJsonPathFor(mapping, TARGET_PATH);
223             if (targetPath == null) {
224                 final JsonPath deletePath = propertyValues.getJsonPathFor(mapping, EXPRESSION_PROP);
225                 if (deletions.contains(deletePath)) {
226                     throw new ConfigurationException(PATH_TO_DELETE_ALREADY_EXISTS)
227                             .formatWith(deletePath.toString())
228                             .withReason(CANNOT_DELETE_A_PATH_MORE_THAN_ONCE)
229                             .withResolution(REMOVE_THE_DUPLICATE_PATH);
230                 }
231                 deletions.add(deletePath);
232             } else {
233                 ExpressionProperty prop = propertyValues.getExpressionPropertyFor(mapping,
234                         EXPRESSION_PROP);
235                 if (!knownTargets.add(targetPath)) {
236                     throw new ConfigurationException(TARGET_MAPPING_EXISTS)
237                             .formatWith(targetPath)
238                             .withReason(TARGET_MAPPING_REASON)
239                             .withResolution(TARGET_MAPPING_EXISTS_RES);
240                 }
241                 mappings.add(Pair.of(prop, targetPath));
242             }
243         }
244         results = Lists.newArrayListWithExpectedSize(mappings.size());
245         hasBinaryOutput = outputViews.get().getType() == ViewType.BINARY;
246 
247         if (hasBinaryOutput) {
248             // Check if the user wants to rewrite the binary stream.
249             writesContent = mappings.stream()
250                     .anyMatch(pair -> CONTENT_PATH.equals(pair.getRight()) ||
251                             ENTITY_PATH.equals(pair.getRight()));
252         }

257         // profile will only be created when pipe is in suggest mode
258         try {
259             if (isSuggest) {
260                 index++;

262             }
263 
264             // We need to check if the mapping table is filled
265             if (isMappingTableModified()) {
266                 processBinary(document, inputViewName, null);
267                 return;

269             outputViews.write(document);
270         } catch (Exception e) {

273                     .withResolution(RESOLUTION_ADDRESS_ISSUE);

278     @Override
279     protected void processBinary(Document document, String inputViewName,
280                                  ReadableByteChannel channel) {
281         Object data = null;
282         if (document != null) {
283             data = document.get();
284             if (data == null) {
285                 throw new SnapDataException(DATA_TRANSFORM_CANNOT_PROCESS_NULL_DATA)
286                         .withReason(NULL_DATA)
287                         .withResolution(PLEASE_CHECK_YOUR_INPUT_DATA);
288             }
289             outputRoot = data;
290         }
291         try {
292             // If there is no binary output or content is being rewritten, we need to read in the
293             // content to a byte array.
294             if ((!hasBinaryOutput || writesContent) && channel != null) {
295                 Map<String, Object> map;
296 
297                 if (data == null) {
298                     data = map = new LinkedHashMap<>();
299                 } else {
300                     map = (Map<String, Object>) data;
301                 }
302 
303                 try {
304                     byte[] bits = ByteStreams.toByteArray(Channels.newInputStream(channel));
305 
306                     map.put("content", bits);
307                 } catch (IOException e) {
308                     throw new SnapDataException(e, "Unable to read binary data from upstream snap");
309                 }
310             }
311 
312             // This is the object we'll write out, it can still be overridden if the mapping root
313             // and target path of an expression is '$'.
314             // Walk the mapping root path looking for values to map.
315             for (WalkResult walkResult : mappingRoot.walk(data)) {
316                 final Object element = walkResult.getValue();
317                 // We want the expressions to operate on the original data only, so we need
318                 // to make the changes in a second pass.
319                 try {
320                     for (Pair<ExpressionProperty, JsonPath> mapping : mappings) {
321                         ExpressionProperty expressionProperty = mapping.getLeft();
322                         Object result = expressionProperty.eval(document, element,
323                                 NULL_VALUE_HANDLER);
324                         results.add(result);
325                     }
326                     // The results of the expressions will be written into target.
327                     final Object target;
328                     if (!isPassThrough && document != null) {
329                         target = Maps.newLinkedHashMap();
330                     } else if (element != null) {
331                         target = element;
332                     } else {
333                         target = Maps.newLinkedHashMap();
334                     }
335                     // Replace the original value in the tree with the new target.
336                     outputRoot = walkResult.setValue(target);
337                     populateAndDeleteTarget(walkResult, target);
338                 } finally {
339                     results.clear();
340                 }
341             }
342             if (!(outputRoot instanceof List || outputRoot instanceof Map)) {
343                 throw new SnapDataException(THE_OUTPUT_DOCUMENT_IS_A_PRIMITIVE_VALUE)
344                         .formatWith(outputRoot)
345                         .withReason(THE_OUTPUT_DOCUMENT_MUST_BE_AN_ARRAY_OR_OBJECT)
346                         .withResolution(CHECK_FOR_TARGET_PATHS_THAT_WRITE_TO_THE_ROOT);
347             }
348             if (document == null) {
349                 document = documentUtility.newDocument();
350             }
351             if (outputRoot == null) {
352                 outputRoot = Maps.newLinkedHashMap();
353             }
354             document.set(outputRoot);
355             doWrite(document, channel);
356         } catch (InvalidPathException e) {
357             if (nullSafeAccess && !mappingRoot.isRoot()) {
358                 // Just write out the original document if we're null-safe and the mapping
359                 // root cannot be found.
360                 LOG.debug(LOG_INVALID_MAPPING_ROOT, mappingRoot.toString(), e);
361                 if (document == null) {
362                     document = documentUtility.newDocument(Maps.newLinkedHashMap());
363                 }
364                 doWrite(document, channel);
365             } else {
366                 throw new SnapDataException(e, INVALID_MAPPING_ROOT)
367                         .formatWith(e.getPath())
368                         .withReason(e.getMessage())
369                         .withResolution(e.getResolution());
370             }
371         } finally {
372             outputRoot = null;
373         }
374     }

376     private void doWrite(Document document, ReadableByteChannel channel) {
377         if (hasBinaryOutput && channel != null && !writesContent) {
378             outputViews.write(new BinaryOutput() {
379                 @Override
380                 public Document getHeader() {
381                     return document;
382                 }
383 
384                 @Override
385                 public void write(final WritableByteChannel writeChannel) throws IOException {
386                     ByteStreams.copy(channel, writeChannel);
387                 }
388             });
389         } else {
390             outputViews.write(document);
391         }
392     }

394     /**
395      * Populates the target object with the results of the expressions and performs any deletions.
396      *
397      * @param walkResult The current result of the mapping root walk that we working on.  This is
398      *                   only needed if a target path is '$'.
399      * @param data the data being populated with the results of the mappings.
400      */
401     private void populateAndDeleteTarget(final WalkResult walkResult, Object data) {
402         int resultSize = results.size();
403         for (int index = 0; index < resultSize; index++) {
404             try {
405                 JsonPath path = mappings.get(index).getRight();
406                 Object value = results.get(index);
407 
408                 // Check if we're overwriting the 'data' value itself, in which case
409                 // we overwrite the original value.
410                 if (path.isRoot()) {
411                     outputRoot = walkResult.setValue(value);
412                 } else {
413                     if (value == data && isPassThrough) {
414                         throw new SnapDataException(
415                                 "Cannot write the document into itself")
416                                 .withReason("Creating a loop in a document is not supported")
417                                 .withResolution("Delete this mapping, disable Passthrough, or " +
418                                         "create a copy of the input document");
419                     }
420                     path.write(data, value);
421                 }
422             } catch (InvalidPathException ignored) {
423             }
424         }
425         for (JsonPath path: deletions) {
426             try {
427                 path.delete(data);
428             } catch (InvalidPathException e) {
429                 if (!nullSafeAccess) {
430                     throw new SnapDataException(e, Messages.UNABLE_TO_DELETE_JSON_PATH)
431                             .formatWith(e.getPath())
432                             .withReason(e.getMessage())
433                             .withResolution(e.getResolution());
434                 }
435             }
436         }
437     }

439     private boolean isMappingTableModified() {
440         @SuppressWarnings("unchecked")
441         Object expressionValue = ((Map<String, Object>) mappingTable.get(0).get(EXPRESSION_PROP))
442                 .get(VALUE_PROP);
443         boolean checkExpression = (expressionValue == null) ||
444                 (expressionValue == StringUtils.EMPTY);
445         boolean checkTargetPath = mappingTable.get(0).get(TARGET_PATH_PROP)
446                 .equals(ROW1_TARGET_PATH_ENTRY);
447         boolean isDefault = checkExpression && checkTargetPath;
448 
449         return mappingTable.size() >= 1 && !isDefault;
450     }
