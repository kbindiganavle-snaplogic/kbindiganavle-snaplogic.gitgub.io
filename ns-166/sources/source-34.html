203         if (xsltFile != null) {

248     @Override
249     protected void doWork() {
250         // do nothing if input view is not connected
251         if (inputViews.isEmpty()) {
252             return;
253         }
254         for (BinaryInput binaryInput : inputViews.binaryInputs(inputViews.get())) {
255             final Document header = binaryInput.getHeader();
256             final Document headerCopy = header.copy();
257             incomingDocCounter.inc();
258             try (BufferedInputStream resettableInputStream = new BufferedInputStream(
259                     new BOMInputStream(Channels.newInputStream(binaryInput.getChannel())))) {
260                 if (shouldValidate && schemaValidator != null) {
261                     // Stream has to be opened again as the schema validator will close the
262                     // stream when the xml data is validated.
263                     try {
264                         resettableInputStream.mark(Integer.MAX_VALUE);
265                         InputStream uncloseableInputStream = new CloseShieldInputStream
266                                 (resettableInputStream);
267                         xmlUtils.validate(schemaValidator, uncloseableInputStream);
268                         resettableInputStream.reset();
269                     } catch (SAXException e) {
270                         log.warn(IGNORING_XML_DATA_AS_IT_DOES_NOT_CONFORM_TO_THE_SCHEMA);
271                         // Writes the validation error to the error view.
272                         errorViews.write(Utils.createDataExceptionForInvalidXML(e), header);
273                         continue;
274                     }
275                 }
276 
277                 outputViews.write(new BinaryOutput() {
278                     @Override
279                     public Document getHeader() {
280                         return headerCopy;
281                     }
282 
283                     @Override
284                     public void write(final WritableByteChannel writeChannel) throws IOException {
285                         try (PushbackInputStream pushbackInputStream = new PushbackInputStream(
286                                 resettableInputStream)) {
287                             if (Utils.isNotEmpty(pushbackInputStream)) {
288                                 // make each binary header available as an XSL parameter
289                                 setParametersFromHeaders(transformer, getHeader());
290                                 if (shouldCanonicalize) {
291                                     final ByteArrayOutputStream tempXMLOutputStream =
292                                             new ByteArrayOutputStream();
293                                     transformer.transform(new StreamSource(pushbackInputStream),
294                                             new StreamResult(tempXMLOutputStream));
295                                     byte[] canonicalXML =
296                                             createC14NXML(tempXMLOutputStream.toByteArray());
297                                     try (OutputStream outputStream =
298                                                  Channels.newOutputStream(writeChannel)) {
299                                         outputStream.write(canonicalXML);
300                                         outputErrors(header);
301                                     }
302                                 } else {
303                                     try (OutputStream outputStream =
304                                                  Channels.newOutputStream(writeChannel)) {
305                                         transformer.transform(new StreamSource(pushbackInputStream),
306                                                 new StreamResult(outputStream));
307                                         outputErrors(header);
308                                     }
309                                 }
310                             } else {
311                                 // Encountered empty document.
312                                 log.debug(SKIPPING_EMPTY_DOCUMENT);
313                             }
314                             header.acknowledge();
315                         } catch (final TransformerException e) {
316                             SnapDataException snapEx =
317                                     new SnapDataException(e, e.getLocalizedMessage())
318                                             .withResolution(PLEASE_CHECK_YOUR_INPUT_DATA);
319                             errorViews.write(snapEx, header);
320                         }
321                     }
322 
323                     /**
324                      * Make the binary input headers available as parameters within the XSL template
325                      * @param transformer instance responsible for doing the XSL transformation
326                      * @param header the Document that captures the header properties from the
327                      *               Binary input view
328                      * @return
329                      */
330                     protected void setParametersFromHeaders(Transformer transformer,
331                             Document header) {
332                         if (header != null) {
333                             Map<String, Object> headers = header.get(Map.class);
334                             if (headers != null) {
335                                 for (Map.Entry<String, Object> entry : headers.entrySet()) {
336                                     String key = entry.getKey();
337                                     Object value = entry.getValue();
338                                     log.trace("Setting XSL Parameter for header with key {} and " +
339                                             "value {}", key, value);
340                                     try {
341                                         // when delegating to a 3rd party library, IMO it is best to
342                                         // make the attempt, catch any exception, and retry with
343                                         // sanitized input. Then, if the library ever changes this
344                                         // behavior, unnecessary checks are no longer executed
345                                         transformer.setParameter(key, value);
346                                     } catch (NullPointerException npe) {
347                                         // not recoverable, wrap and throw
348                                         if (key == null) {
349                                             throw new SnapDataException(header, npe,
350                                                     "The binary header contains a null key");
351                                         }
352                                         if (value == null) {
353                                             log.debug("The {} binary header's value was null; " +
354                                                     "retrying with blank string value instead",
355                                                     key);
356                                             value = "";
357                                         }
358                                         transformer.setParameter(key, value);
359                                     }
360                                 }
361                             }
362                         }
363                     }
364                 });
365             } catch (IOException e) {
366                 SnapDataException snapEx =
367                         new SnapDataException(e, e.getLocalizedMessage())
368                                 .withResolution(PLEASE_CHECK_YOUR_INPUT_DATA);
369                 errorViews.write(snapEx, header);
370             }
371         }
372     }

390     static byte[] createC14NXML(byte[] bytes) {
391         try (ByteArrayOutputStream result = new ByteArrayOutputStream()) {
392             // This method must be called before any calls to the Canonicalizer.
393             // It creates the mapping of Canonicalization and Transform algorithms.
394             org.apache.xml.security.Init.init();
395 
396             // Choosing an algorithm that keeps comments
397             // http://www.w3.org/TR/2001/REC-xml-c14n-20010315#WithComments
398             Canonicalizer.getInstance(Canonicalizer.ALGO_ID_C14N_WITH_COMMENTS)
399                     .canonicalize(bytes, result, false);
400             return result.toByteArray();
401         } catch (Exception e) {
402             throw new SnapDataException(e, e.getLocalizedMessage())
403                     .withResolution(PLEASE_CHECK_YOUR_INPUT_DATA);
404         }
405     }
