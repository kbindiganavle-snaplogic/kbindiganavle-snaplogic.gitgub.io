306     @Override
307     public void execute() {
308         if (groupBy && sortOrder == UNSORTED) {
309             // init ascending sorter
310             List<ExpressionProperty> sortPathExpressions = new ArrayList(groupByFields.keySet());
311             SortComparator comparator = new SortComparator(ASCENDING, List.of(), false)
312                     .withExpressionProperties(sortPathExpressions)
313                     .withNullSafeAccess(true);
314             ExternalMergeSorter sorter = new ExternalMergeSorter(comparator, threadFactory,
315                     SORT_BUFFER_SIZE, tmpDir);
316             sorter.setStatusReporter(statusReporter, STATUS_MESSAGE);
317             InputView inputView = inputViews.get();
318             Iterator<Document> iterator = inputViews.getDocumentsFrom(inputView);
319             Document document = null;
320             try {
321                 // pre-sort all input documents
322                 while (iterator.hasNext()) {
323                     document = iterator.next();
324                     sorter.add(document.get());
325                     document.acknowledge();
326                 }
327                 sortOrder = ASCENDING;
328                 // process all sorted input documents
329                 Scanner scanner = sorter.sort();
330                 Object value = scanner.next();
331                 while (value instanceof Map) {
332                     Map<String, Object> data = (Map<String, Object>) value;
333                     Document sortedDoc = documentUtility.newDocument(data);
334                     try {
335                         process(sortedDoc, null);
336                     } catch (SnapDataException e) {
337                         errorViews.write(e, sortedDoc);
338                     }
339                     value = scanner.next();
340                 }
341             } catch (NoSuchElementException | IOException e) {
342                 errorViews.write((SnapDataException)
343                         new SnapDataException(e, ERR_PRESORT)
344                                 .withResolution(RESOLUTION_ADDRESS_ISSUE), document);
345             }
346         } else {
347             super.execute();
348         }
349     }

351     @Override
352     @SuppressWarnings("unchecked")
353     protected void process(final Document document, final String inputViewName) {
354         totalDocumentCount++;
355         checkMemoryDiskResources();
356         if (!firstDataReceived && document != null && document.get() != null) {
357             firstDataReceived = true;
358         }
359         AggregateKey aggregateKey = null;
360         AggregateCompute[] results;
361         if (groupBy) {
362             Object[] keyValues = new Object[groupByFields.size()];
363             int index = 0;
364             for (ExpressionProperty path : groupByFields.keySet()) {
365                 Object data = path.eval(document);
366                 if (data == null) {
367                     writeErrorView(document, ERR_INVALID_INPUT_DATA,
368                             String.format(ERR_NULL_VALUE, path.getExpression()),
369                             String.format(RESOLUTION_VALUE, path.getExpression()));
370                     return;
371                 }
372                 keyValues[index++] = data;
373             }
374             aggregateKey = new AggregateKey(keyValues);
375             results = aggrMap.get(aggregateKey);
376         } else {
377             results = noGroupByComputes;
378         }
379 
380         boolean newResults = false;
381         if (results == null) {
382             // no matching AggregateCompute object list found in the cache
383             if (groupBy && previousAggregateKey != null && !UNSORTED.equals(sortOrder)) {
384                 checkSortOrder(aggregateKey);
385                 if (!aggrMap.isEmpty()) {
386                     if (aggrMap.size() > 1) {
387                         LOGGER.debug(LOG_UNEXPECTED_AGGREGATE_MAP_SIZE, aggrMap.size());
388                     }
389                     writeResult();
390                     aggrMap.clear();
391                 }
392             }
393             results = new AggregateCompute[aggregateFields.size()];
394             newResults = true;
395         }
396 
397         int index = 0;
398         for (Map<String, Object> aggregateFieldMap : aggregateFields) {
399             AggregateFunction aggrFunction =
400                     (AggregateFunction) aggregateFieldMap.get(FUNCTION_PROP);
401             ExpressionProperty aggrField = (ExpressionProperty) aggregateFieldMap.get(FIELD_PROP);
402             AggregateCompute result;
403             if (newResults) {
404                 // no matching AggregateCompute object list found in the cache
405                 result = AggregateUtils.newAggregateCompute(aggrFunction);
406                 results[index++] = result;
407             } else {
408                 result = results[index++];
409             }
410             Object aggrValue = aggrField.eval(document);
411             if (aggrValue == null) {
412                 continue; // ignore null values
413             }
414             // if aggregate value is non-BigDecimal Number object, convert to BigDecimal
415             if ((aggrValue instanceof Number) && !(aggrValue instanceof BigDecimal)) {
416                 try {
417                     aggrValue = new BigDecimal(aggrValue.toString());
418                 } catch (NumberFormatException e) {
419                     // Upstream snap can cause a Number to be represented as Double.NaN.
420                     writeErrorView(document, ERR_NUMBER_TYPE, e.getMessage(),
421                             String.format(RESOLUTION_NUMBER, aggrField.getExpression()));
422                     return;
423                 }
424             }
425             // compute aggregate value
426             try {
427                 result.computeValue(aggrValue);
428             } catch (ExecutionException e) {
429                 writeErrorView(document, String.format(ERR_AGGREGATE_COMPUTE,
430                         aggrField.getExpression()), e.getMessage(), RESOLUTION_VALUE_TYPE);
431                 return;
432             }
433         }
434         if (newResults) {
435             aggrMap.put(aggregateKey, results);
436         }
437         previousAggregateKey = aggregateKey;
438     }

485     private void checkSortOrder(AggregateKey aggregateKey) {
486         int result = aggregateKey.compareTo(previousAggregateKey);
487         switch (sortOrder) {
488             case ASCENDING:
489                 if (result >= 0) {
490                     return;
491                 }
492                 break;
493             case DESCENDING:
494                 if  (result <= 0) {
495                     return;
496                 }
497                 break;
498             default:
499                 // No-Op
500         }
501         throw new ExecutionException(ERR_SORT_ORDER).formatWith(sortOrder.getType())
502                 .withResolution(RESOLUTION_SORT_ORDER);
503     }

505     private void writeResult() {
506         for (Map.Entry<AggregateKey, AggregateCompute[]> entry : aggrMap.entrySet()) {
507             // prepare an output map data for GROUP BY fields
508             Map<String, Object> data = new LinkedHashMap<>(groupByFields.size() + 1);
509             Object[] groupByFieldValues = entry.getKey().getValues();
510             if (groupByFieldValues.length != groupByFields.size()) {
511                 throw new ExecutionException(ERR_GROUP_BY_FIELD_VALUE)
512                         .withReason(ERR_UNKNOWN_REASON)
513                         .withResolution(ERR_TECH_SUPPORT_RESOLUTION);
514             }
515             // iterate over GROUP BY field names, which are used as keys in the output
516             // map data
517             Iterator<String> itValues = groupByFields.values().iterator();
518             for (Object value : groupByFieldValues) {
