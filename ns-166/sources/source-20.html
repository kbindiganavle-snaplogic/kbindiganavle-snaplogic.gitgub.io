306     @Override
307     protected void doWork() {
308         // do nothing if input view is not connected
309         if (inputViews.isEmpty()) {
310             return;
311         }
312         jsonFactory = new JsonFactory().setCodec(supportTypeExts ? writeObjectMapper : objectMapper);
313         Iterable<Document> documents = inputViews.documents(inputViews.get());
314         // The name formatSingleDocument is confusing!
315         // If true, each input document is written to a separate binary output stream.
316         // If false, a single output stream is written, combining all input documents.
317         if (formatSingleDocument) {
318             writeEachToSeparateOutput(documents);
319         } else {
320             writeAllToOneOutput(documents);
321         }
322     }

324     @SuppressWarnings("unchecked")
325     private void process(Document document, OutputStream outputStream) {
326         if (jsonGenerator == null) {
327             initJsonGenerator(document, outputStream);
328         }
329         try {
330             if (!docWritingStarted && !formatSingleDocument && !jsonLines) {
331                 jsonGenerator.writeStartArray();
332                 docWritingStarted = true;
333             }
334         } catch (IOException e) {
335             throw new ExecutionException(e, JSON_WRITE_ERROR)
336                     .withReason(JSON_WRITE_ERROR)
337                     .withResolution(ERR_CUSTOMER_SUPPORT);
338         }
339         Object data = document.get();
340         try {
341             if (contentPath.isPresent()) {
342                 try {
343                     data = contentPath.get().read(data);
344                 } catch (InvalidPathException e) {
345                     throw new SnapDataException(e, "Unable to get content from document: %s")
346                             .formatWith(e.getMessage())
347                             .withResolution(e.getResolution());
348                 }
349             }
350             if (jsonLines && formatSingleDocument && (data instanceof List)) {
351                 for (Object elem : (List<?>) data) {
352                     jsonGenerator.writeObject(elem);
353                     jsonGenerator.writeRaw("\n");
354                 }
355             } else {
356                 jsonGenerator.writeObject(data);
357                 if (jsonLines) {
358                     jsonGenerator.writeRaw("\n");
359                 }
360             }
361             documentCounter.inc();
362         } catch (ClosedChannelException e) {
363             // NO OP
364         } catch (IOException e) {
365             String error = String.format(ERR_CANNOT_CONVERT_TO_JSON, e.getMessage());
366             if (data instanceof Map) {
367                 ((Map<String, Object>) data).put(ERROR, error);
368             }
369             SnapDataException snapDataException = new SnapDataException(document, e, error)
370                     .withReason(error)
371                     .withResolution(ERR_CUSTOMER_SUPPORT);
372             errorViews.write(snapDataException, document);
373         }
374     }

376     private void writeAllToOneOutput(Iterable<Document> documents) {
377         final Iterator<Document> documentIterator = documentSampleBuffer != null ?
378                 documentSampleBuffer : documents.iterator();
379         final boolean hasNext = documentIterator.hasNext();
380         if (!ignoreEmptyStream || hasNext) {
381             // write all documents into a single output stream
382             final Document firstDocument = hasNext ? documentIterator.next() : null;
383             final Document header = createHeaderDocument(firstDocument, derviedSchema);
384             outputViews.write(new BinaryOutput() {
385                 @Override
386                 public Document getHeader() {
387                     return header;
388                 }
389 
390                 @Override
391                 public void write(final WritableByteChannel writeChannel) throws IOException {
392                     try (OutputStream outputStream = Channels.newOutputStream(writeChannel)) {
393                         // next() should not be called if hasNext() is false
394                         Document document = firstDocument;
395                         while (document != null) {
396                             process(document, outputStream);
397                             // Ack the document here since we're dropping it on the floor.
398                             document.acknowledge();
399                             document = null;
400                             if (documentIterator.hasNext()) {
401                                 document = checkNotNull(documentIterator.next(),
402                                         ITERATOR_RETURNED_NULL_DOCUMENT);
403                             }
404                         }
405                         onEndOfData(outputStream);
406                     }
407                 }
408             });
409         }
410     }

412     private void writeEachToSeparateOutput(Iterable<Document> documents) {
413         // write one document per output stream
414         for (final Document document : documents) {
415             try {
416                 // We create the header here so that the lineage is established right away.  No ack
417                 // is needed since the new doc is linked to the input one.
418                 Schema schema = null;
419 
420                 if (deriveSchemaLimit != DeriveSchemaLimit.None) {
421                     ObjectSchema objectSchema = schemaDeriver.deriveSchemaFromMap(
422                             documentUtility.getAsMap(document, errorViews));
423 
424                     schema = objectSchema.descendTo(contentPath.orElse(JsonPaths.root())).peek();
425                 }
426                 final Document header = createHeaderDocument(document, schema);
427                 outputViews.write(new BinaryOutput() {
428                     @Override
429                     public Document getHeader() {
430                         return header;
431                     }
432 
433                     @Override
434                     public void write(final WritableByteChannel writeChannel) throws IOException {
435                         try (OutputStream outputStream = Channels.newOutputStream(writeChannel)) {
436                             process(document, outputStream);
437                             onEndOfData(outputStream);
438                         }
439                     }
440                 });
441             } catch (SnapDataException e) {
442                 errorViews.write(e, document);
443             }
444         }
445     }

447     private Document createHeaderDocument(Document original, final Schema schema) {
448         Map<String, Object> headerMap = new LinkedHashMap<>(1);
449         headerMap.put(SdkConstants.CONTENT_TYPE.toLowerCase(), MediaType.JSON_UTF_8.toString());
450         if (original != null) {
451             for (HeaderProperty headerProperty : header) {
452                 headerProperty.populate(original, headerMap);
453             }
454         }
455         if (schema != null) {
456             headerMap.put("schema", schema.toMap());
457         }
458         return documentUtility.newDocumentFor(formatSingleDocument ? original : null, headerMap);

461     private void initJsonGenerator(Document document, OutputStream outputStream) {
462         /* Snaps with document input and binary output should now be designed to just output with UTF-8 charset and
463          * then connect to a Transcoder snap if a different charset is needed.  The following few lines use the
464          * obsolete document metadata facility to set the charset based on the metadata of the first inbound document.
465              * This is arguably a bad idea and we're only keeping this in case any legacy pipelines depend on it.
466          * Note that this does not work well with the formatSingleDocument option, since only the first output will
467          * use the encoding specified only by the first document.  Fortunately, CSVParser appears to be the only snap
468          * that actually outputs a document with metadata specifying a charset.
469          */
470         String charset = TransformUtils.getCharEncoding(documentUtility, document);
471         JsonEncoding jsonEncoding = JsonEncoding.UTF8;
472         if (StringUtils.isNotBlank(charset)) {
473             jsonEncoding = getJsonEncoding(charset);
474         }
475 
476         // instantiate JsonGenerator with given JsonEncoding
477         try {
478             jsonGenerator = jsonFactory.createJsonGenerator(outputStream, jsonEncoding);
479             if (prettyPrint) {
480                 DefaultPrettyPrinter pp = new DefaultPrettyPrinter();
481                 pp.indentArraysWith(DefaultIndenter.SYSTEM_LINEFEED_INSTANCE);
482                 jsonGenerator.setPrettyPrinter(pp);
483             }
484             else if (jsonLines) {
485                 jsonGenerator.setPrettyPrinter(new MinimalPrettyPrinter(""));
486             }
487         } catch (IOException e) {
488             // TODO test this; how?
489             String error = String.format(ERR_INIT_JSON_FORMATTER, charset, e.getMessage());
490             throw new ExecutionException(e, error)
491                     .withReason(error)
492                     .withResolution(ERR_CUSTOMER_SUPPORT);
493         }
494     }

496     JsonEncoding getJsonEncoding(String charset) {
497         for (JsonEncoding encoding : JsonEncoding.values()) {

499                 return encoding;

501         }

505     private void onEndOfData(OutputStream outputStream) {
506         try {
507             if (docWritingStarted) {
508                 jsonGenerator.writeEndArray();
509                 jsonGenerator.flush();
510                 jsonGenerator.close();
511             } else if (formatSingleDocument) {
512                 jsonGenerator.flush();
513                 jsonGenerator.close();
514                 jsonGenerator = null; // a new instance will need to be created for the next OutputStream
515             } else if (jsonLines) {
516                 if (jsonGenerator != null) {
517                     jsonGenerator.flush();
518                     jsonGenerator.close();
519                 } else if (!ignoreEmptyStream) {
520                     //jsonGenerator is null when there is an empty document in the input
521                     outputStream.write(EMPTY_OBJECT);
522                     outputStream.flush();
523                 }
524             } else if (!ignoreEmptyStream) {
525                 outputStream.write(EMPTY_ARRAY);
526                 outputStream.flush();
527             }
528         } catch (ClosedChannelException e) {
529             // TODO test this; how?
530             // NO OP
531         } catch (IOException e) {
532             // TODO test this; how?
533             throw new ExecutionException(e, JSON_FLUSH_ERROR)
534                     .withReason(JSON_FLUSH_ERROR)
535                     .withResolution(ERR_CUSTOMER_SUPPORT);
536         }
