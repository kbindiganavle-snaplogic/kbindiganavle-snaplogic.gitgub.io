123     public Map<String, List<ColumnDefinition>> process(TableConfig tableConfig,
124                                                        TableDefinition srcTableDef,
125                                                        TableDefinition tgtTableDef,
126                                                        AutoSyncSpec spec,
127                                                        EndpointService endpointService,
128                                                        boolean isPreview) {
129 
130         Map<String, List<ColumnDefinition>> result = new LinkedHashMap<>();
131         try {
132             // Get source columns into 'srcColumns'
133             List<ColumnDefinition> srcColumns = srcTableDef.getColumns();
134             // Retrieve expressions from "src"
135             srcCdm = srcTableDef.getCdm();
136             if (srcCdm == null) {
137                 Map<String, Object> tableDefinition = endpointService.definition(
138                         spec.getSource(), tableConfig, "cdm", spec.getPath(), "source");
139                 srcCdm = OBJECT_MAPPER_NORMAL.convertValue(tableDefinition.get("cdm"),
140                         new TypeReference<>() {

143                     throw new SnapDataException("Cdt preparation requires" +

145                 }

147             }
148

150             boolean targetSchemaExists = false;
151             List<ColumnDefinition> tgtColumns = null;
152             if (tgtTableDef.getColumns() != null) {
153                 targetSchemaExists = true;
154                 tgtColumns = tgtTableDef.getColumns();
155

157             }

159             tgtCdm = tgtTableDef.getCdm();
160             if (tgtCdm == null) {

162                         spec.getTarget(), tableConfig, "cdm", spec.getPath(), "target");

225      * Process the given srcColumns and build output documents.
226      *
227      * @param srcColumns             source column schemas
228      * @param srcCdm                 source expression file contents
229      * @param tgtCdm                 target expression file contents
230      * @param hasFasterLoaderColumns flag to indicate FDL columns exists at 'srcColumns'
231      * @param isPreview
232      * @return output documents with "groupBy" and "srcDef" maps
233      */
234     private List<ColumnDefinition> processSourceColumns(List<ColumnDefinition> srcColumns,
235                                                         DataMapper srcCdm, DataMapper tgtCdm,
236                                                         boolean hasFasterLoaderColumns,
237                                                         TableConfig tableConfig, boolean isPreview)
238             throws ClassCastException {
239         String columnNameConvertExpr = null;
240         String tgtMapColumnName = tgtCdm.getMapColumnName();
241         if (tgtMapColumnName != null && !tgtMapColumnName.isEmpty()) {
242             columnNameConvertExpr = tgtCdm.getMapColumnName();
243         }
244 
245         // Retrieve default data type and type map from map of expressions
246         String srcDefaultDataType = srcCdm.getDefaultDataType();
247         Map<String, String> srcMapDatatypes = srcCdm.getMapDatatypes();
248 
249         // Retrieve default data type and type map from map of expressions
250 
251         Map<String, String> tgtMapDatatypes = tgtCdm.getMapDatatypes();
252         Map<String, DataSize> tgtDatasizeMap = tgtCdm.getDatasize();
253 
254         if (!hasFasterLoaderColumns) {
255             prependAutoSyncColumns(srcColumns);
256         }
257 
258         for (ColumnDefinition colDef : srcColumns) {
259             colDef.sourceType = colDef.dataType;
260             colDef.sourceColumn = colDef.column;
261             colDef.cdt = srcMapDatatypes.getOrDefault(colDef.sourceType, srcDefaultDataType);
262 
263             colDef.isDefaultCDT = !srcMapDatatypes.containsKey(colDef.sourceType);
264 
265             // Add "srcPrecision" to the map.
266             colDef.srcPrecision = colDef.precision;
267         }
268 
269         for (ColumnDefinition colDef : srcColumns) {
270             // map hierarchy here
271             colDef.originalCdt = colDef.cdt;
272             colDef.cdt = hierarchy.getTargetDataTypeFromCdt(colDef.cdt, colDef.precision,
273                     colDef.scale, tgtMapDatatypes, tgtDatasizeMap);
274             getInitialTargetDefinitionValue(colDef, tgtCdm);
275 
276             // Validate and adjust target precision with the tgtDatasizeMap.
277             colDef.precision = getTargetPrecisionValue(colDef, colDef.originalCdt, colDef.cdt,
278                     colDef.dataType, tgtDatasizeMap, tgtMapDatatypes, tableConfig);
279             // Set target table "scale"
280             colDef.scale = getTargetScaleValue(colDef.precision, colDef.scale,
281                     colDef.dataType, tgtDatasizeMap);
282             // Update "table" if necessary:
283             updateTableName(colDef);
284 
285             // Update "column" if necessary:
286             if (columnNameConvertExpr == null || columnNameConvertExpr.isEmpty()) {
287                 colDef.column = nameHandler.getNameAsInDatabase(NameType.COLUMN_NAME,
288                         colDef.column);
289             } else {
290                 updateColumnName(columnNameConvertExpr, colDef);
291             }
292 
293             // Add column clause "columnDefinition"
294             colDef.columnDefinition = buildColumnClause(colDef);
295         }
296 
297         if (!isPreview) {
298             if (tableConfig.getTransformations() != null) {
299                 // Transformations are null in case the ENABLE flag is false
300                 // Process transformations in case Enable is true
301                 Transformation transformation = new TransformationImpl();
302                 transformation.process(srcColumns, tableConfig);
303             }
304         }
305 
306         return srcColumns;
307     }

378     private void getInitialTargetDefinitionValue(ColumnDefinition colDef,
379                                                  DataMapper tgtCdm) {
380         String tgtDefaultDataType = tgtCdm.getDefaultDataType();
381         Map<String, String> tgtMapDatatypes = tgtCdm.getMapDatatypes();
382         String tgtDataType = tgtMapDatatypes.getOrDefault(colDef.cdt, tgtDefaultDataType);
383         if (tgtDataType.contains("(")) {
384             colDef.dataType = tgtDataType.replaceAll("\\(.*\\)", "");
385             int startIndex = tgtDataType.indexOf("(");
386             int commaIndex = tgtDataType.indexOf(",", startIndex);
387             int endIndex = tgtDataType.indexOf(")", startIndex);
388             if (commaIndex >= 0) {
389                 String precString = tgtDataType.substring(startIndex + 1, commaIndex);
390                 String scaleString = tgtDataType.substring(commaIndex + 1, endIndex);
391                 colDef.precision = getValueAsBigInteger(Integer.parseInt(precString.trim()))
392                         .orElse(null);
393                 colDef.scale = getValueAsBigInteger(Integer.parseInt(scaleString.trim()))
394                         .orElse(null);
395             } else {
396                 String precString = tgtDataType.substring(startIndex + 1, endIndex);
397                 colDef.precision = getValueAsBigInteger(Integer.parseInt(precString.trim()))
398                         .orElse(null);
399             }
400         } else {
401             colDef.dataType = tgtDataType;
402         }
403     }

409     /**
410      * Return target precision value after validating
411      * the given source 'precisionValue' (source precision value).
412      * <p>
413      * Note that 'precisionValue' may be BigInteger, integer string, or
414      * even special string value like "max".
415      *
416      * @param srcColDef       source column definition
417      * @param originalCdt     cdt from source datatype mapping
418      * @param finalCdt        cdt that has corresponding target datatype
419      * @param dataTypeValue   target data type
420      * @param tgtDatasizeMap  target data size map with min and max per data type
421      * @param tgtMapDatatypes mapping from cdt to target datatype
422      * @return target scale value
423      */
424     private BigInteger getTargetPrecisionValue(
425             ColumnDefinition srcColDef, String originalCdt, String finalCdt,
426             String dataTypeValue, Map<String, DataSize> tgtDatasizeMap,
427             Map<String, String> tgtMapDatatypes, TableConfig tableConfig) {
428         Object precisionValue = srcColDef.precision;
429         BigInteger sourceScale = srcColDef.scale;
430         String tgtDataType = tgtMapDatatypes.getOrDefault(srcColDef.cdt, srcColDef.dataType);
431 
432         if (tableConfig.getSrcDbType().equalsIgnoreCase(tableConfig.getTgtDbType()) &&
433                 srcColDef.sourceType.equalsIgnoreCase(srcColDef.dataType) &&
434                 srcColDef.dbPrecision != null &&
435                 srcColDef.dbPrecision instanceof String &&
436                 ((String) srcColDef.dbPrecision).isBlank()) {
437             precisionValue = null;
438         }
439 
440         // We may have a String value like "max" from precisionValue
441         precisionValue = CDT_PRECISION_DATA_TYPES.contains(finalCdt) || tgtDataType.contains("(")
442                 ? precisionValue : null;
443         boolean isPrecisionValueNonNumeric =
444                 precisionValue instanceof String &&
445                         !StringUtils.isNumeric((String) precisionValue);
446         BigInteger precisionValueBigInteger =
447                 getValueAsBigInteger(precisionValue).orElse(null);
448 
449         DataSize datasizeEntry = tgtDatasizeMap.getOrDefault(dataTypeValue, null);
450         if (datasizeEntry == null) {
451             return null;
452         }
453         BigInteger maxValue = datasizeEntry.max[0];
454         BigInteger minValue = datasizeEntry.min[0];
455 
456         if (isPrecisionValueNonNumeric &&
457                 StringUtils.equals("max", (String) precisionValue)) {
458             return maxValue;
459         }
460 
461         if (CDT_NUMBER_TYPES.contains(originalCdt) && !CDT_NUMBER_TYPES.contains(finalCdt)
462                 && precisionValueBigInteger != null) {
463             /*
464               P >= S && S > 0 → Actual P = P, Actual S = S
465                    To CDT_STRING → P = P + 2 (dot and sign)
466               P >= S && S = 0 → Actual P = P, Actual S = S
467                    To CDT_STRING → P = P + 1 ( sign )
468               P < S && S >0 → Actual P = S, Actual S = S
469                    To CDT_STRING → P = S + 3 (dot and sign and 0)
470               S < 0 → Actual P = P + abs(S) , Actual S = 0
471                    To CDT_STRING → P = P + 2 (dot and sign)
472               */
473             if (sourceScale.compareTo(BigInteger.ZERO) > 0 &&
474                     precisionValueBigInteger.compareTo(sourceScale) > 0) {
475                 precisionValueBigInteger = precisionValueBigInteger.add(BigInteger.TWO);
476             }
477             if (sourceScale.compareTo(BigInteger.ZERO) == 0 &&
478                     precisionValueBigInteger.compareTo(sourceScale) > 0) {
479                 precisionValueBigInteger = precisionValueBigInteger.add(BigInteger.ONE);
480             }
481             if (sourceScale.compareTo(BigInteger.ZERO) > 0 &&
482                     precisionValueBigInteger.compareTo(sourceScale) < 0) {
483                 precisionValueBigInteger = sourceScale.abs().add(BigInteger.valueOf(3));
484             }
485             if (sourceScale.compareTo(BigInteger.ZERO) < 0) {
486                 precisionValueBigInteger = precisionValueBigInteger.add(sourceScale.abs())
487                         .add(BigInteger.TWO);
488             }
489         }
490 
491         if (precisionValue == null || precisionValue.equals(BigInteger.ZERO)) {
492             if (datasizeEntry.def != null) {
493                 return datasizeEntry.def[0];
494             }
495             return maxValue;
496         }
497 
498         // Validate and return precision value with minValue
499         if (minValue == null) {
500             return null;
501         }
502 
503         if (!isPrecisionValueNonNumeric &&
504                 precisionValueBigInteger != null) {
505 
506             if (precisionValueBigInteger.compareTo(BigInteger.ZERO) < 0) {
507                 return maxValue;
508             }
509 
510             if (precisionValueBigInteger.compareTo(minValue) < 0) {

540                                            String dataTypeValue,
541                                            Map<String, DataSize> tgtDatasizeMap) {
542         if (precisionValue == null) {
543             return null;
544         }
545 
546         DataSize datasizeEntry = tgtDatasizeMap.get(dataTypeValue);
547         if (datasizeEntry == null) {
548             return scaleValue;
549         }
550 
551         if (scaleValue == null) {
552             if (datasizeEntry.def != null) {
553                 return datasizeEntry.def[1];
554             }
555             return null;
556         }
557 
558         // We implicitly assume that an entry from 'tgDatasizeMap' should have all integral values
559         //
560         // For example,
561         //   "float" :            { "min" : [ null,     null ], "max" : [ null,     null ] }
562         // , "number" :           { "min" : [ 1,        0    ], "max" : [ 38,       37   ] }
563         //
564         BigInteger minValue = datasizeEntry.min[1];
565         if (minValue == null) {
566             return null;
567         }
568         if (scaleValue.compareTo(minValue) < 0) {
569             return minValue;
570         }
571 
572         BigInteger maxValue = datasizeEntry.max[1];
573         if (maxValue == null) {
574             return null;
575         }
576         if (scaleValue.compareTo(maxValue) > 0) {
577             return maxValue;
578         }
579 
580         return scaleValue;
