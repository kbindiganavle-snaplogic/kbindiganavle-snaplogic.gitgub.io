225     public void configure(final PropertyValues propertyValues) throws ConfigurationException {
226         // Configure the pipeline
227         pipelineSnodeId = propertyValues.get(PROP_PIPELINE_ID);
228         if (StringUtils.isBlank(pipelineSnodeId)) {
229             // TODO - MK: temp workaround until we can mark required properties as not required
230             //            during value suggest.
231             throw new ConfigurationException(ERR_NO_PIPELINE_ID)
232                     .withReason(ERR_NO_PIPELINE_ID)
233                     .withResolution(PLEASE_PROVIDE_REQUIRED_PIPELINE_NAME);
234         }
235         pipelineFacade = snapiService.createPipelineFacade();
236         Object object = propertyValues.get(PROP_MAX_INSTANCES);
237         maxInstances = object instanceof BigInteger ? ((BigInteger) object).intValue() :
238                 DEFAULT_MAX_INSTANCES;
239         if (maxInstances > MAX_INSTANCES_LIMIT) {
240             LOG.warn("maxInstances is set too high ({}), reducing to {}", maxInstances,
241                     MAX_INSTANCES_LIMIT);
242             maxInstances = MAX_INSTANCES_LIMIT;
243         }
244         isRecurringPipeline = checkRecurringPipeline(propertyValues);
245         snaplexFacade = snapiService.createSnaplexFacade();
246         execMode = ExecMode.getExecMode(propertyValues.<String>get(PROP_EXEC_MODE));
247         // Collect the params
248         List<Map<String, Map<String, String>>> paramList = propertyValues.get(PATH_PARAM);
249         propertyValues.validateTableProperty(paramList, PROP_PARAM_NAME, LBL_PARAM_NAME);
250         Map<String, ExpressionProperty> params = createParamMap(propertyValues, paramList);
251         ForEachViewWriter viewWriter;
252         int batchSize = -1;
253         if (execMode == ExecMode.SYNCHRONOUS) {
254             BigInteger batchSizeValue = propertyValues.get(PROP_BATCH_SIZE);
255             if (batchSizeValue != null) {
256                 batchSize = batchSizeValue.intValue();
257             }
258             viewWriter = new SynchronousViewWriter(outputViews, errorViews,
259                     documentUtility);
260         } else {
261             viewWriter = new FireAndForgetViewWriter(outputViews, documentUtility);
262         }
263         pipelineScheduler = new PipelineScheduler(batchSize, execMode, viewWriter,
264                 propertyValues.getAsExpression(PROP_PLEX_LABEL), params);
265     }

267     @Override
268     public void execute() {
269         Map<String, InputView> inputViewMap = inputViews.getAll();
270 
271         if (inputViewMap.isEmpty()) {
272             process(null, null);
273             return;
274         }
275 
276         int delay = INITIAL_POLL_DELAY;
277 
278         while (true) {
279             Map<String, Document> ready = inputViews.select(delay, TimeUnit.MILLISECONDS);
280 
281             if (ready.isEmpty()) {
282                 delay = Math.min(MAX_POLL_DELAY, delay * 2);
283                 pipelineScheduler.pollForCompletion();
284             } else {
285                 for (Map.Entry<String, Document> entry : ready.entrySet()) {
286                     Document doc = entry.getValue();
287 
288                     if (doc == null) {
289                         return;
290                     }
291                     try {
292                         process(doc, entry.getKey());
293                     } catch (SnapDataException ex) {
294                         errorViews.write(ex, document);
295                     }
296                 }
297                 delay = INITIAL_POLL_DELAY;
298             }
299         }
300     }

302     @Override
303     protected void process(final Document document, final String inputViewName) {
304         try {
305             if (isRecurringPipeline) {
306                 while (!selfInvokable()) {
307                     if (execMode.equals(ExecMode.SYNCHRONOUS)) {
308                         throw new SnapDataException(ERR_MAX_INSTANCES).formatWith(maxInstances)
309                                 .withReason(REASON_MAX_INSTANCES)
310                                 .withResolution(RESOLUTION_ADDRESS_ISSUE);
311                     }
312                     // delay
313                     try {
314                         TimeUnit.MILLISECONDS.sleep(DELAY_IN_MILLISECONDS);
315                     } catch (InterruptedException ex) {
316                         // NO-OP
317                         break;
318                     }
319                 }
320             }
321             this.document = document;
322             pipelineScheduler.run(pipelineFacade, snaplexFacade, document, pipelineSnodeId);
323         } catch (SnapDataException e) {
324             errorViews.write(e, document);
325         }
326     }

328     @Override
329     public void handle(final LifecycleEvent event) {
330         // During validation, pipeline scheduler is null.
331         switch (event) {
332             case SUCCESS:
333             case FAILURE:
334                 if (pipelineScheduler != null) {
335                     try {
336                         pipelineScheduler.cleanup(event);
337                     } catch (SuggestViewAbortException e) {
338                         throw e;
339                     } catch (Throwable e) {
340                         SnapDataException ex = new SnapDataException(e, ERR_CLEANUP)
341                                 .withReason(e.getMessage())
342                                 .withResolution(RESOLUTION_ADDRESS_ISSUE);
343                         errorViews.write(ex, document);
344                     }
345                 }
346                 if (isRecurringPipeline) {
347                     decrementRecurringPipelineCounter();
348                 }
349         }

353     protected boolean checkRecurringPipeline(PropertyValues propertyValues) {
354         Document emptyDoc = documentUtility.newDocument();
355         // get the path of the project which this Snap belongs to
356         String projectPath = expressionUtils.createExpressionProperty(
357                 propertyValues, true, PIPE_PROJECT_PATH).eval(emptyDoc);
358         // get the name of the pipeline which this Snap belongs to
359         String pipeLabel = expressionUtils.createExpressionProperty(
360                 propertyValues, true, PIPE_LABEL).eval(emptyDoc);
361         // get the asset path of the pipeline which this Snap belongs to
362         String pipelinePath = String.format(FORMAT_PATH, projectPath, pipeLabel);
363         // get the snode ID of the pipeline which this Snap belongs to
364         AssetFacade assetFacade= snapiService.createAssetFacade();
365         Map<String, Object> pipelineInfo = assetFacade.fetchAssetInfo(pipelinePath);
366         String pipelineSnodeIdThis = (String) pipelineInfo.get(
367                 PipelineFacade.PipeIdEnum.SNODE_ID.toString());
368         return StringUtils.equals(pipelineSnodeId, pipelineSnodeIdThis);
369     }

388     private synchronized boolean selfInvokable() {
389         Integer counter = RECURRING_PIPELINES.get(pipelineSnodeId);
390         counter = counter == null ? 1 : counter + 1;
391         if (counter < maxInstances) {
392             RECURRING_PIPELINES.put(pipelineSnodeId, counter);
393             return true;
394         }
395         return false;
396     }
397 
398     private synchronized void decrementRecurringPipelineCounter() {
399         Integer counter = RECURRING_PIPELINES.get(pipelineSnodeId);
400         if (counter != null) {
401             RECURRING_PIPELINES.put(pipelineSnodeId, --counter < 0 ? 0 : counter);
402         }
403     }
