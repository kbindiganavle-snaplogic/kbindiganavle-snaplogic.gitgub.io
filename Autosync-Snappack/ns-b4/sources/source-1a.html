724     @Override
725     public void execute() {
726         try {
727             Set<String> pendingViews = new HashSet<>(inputViews.names());
728             Map<String, Document> selectedDocuments =
729                     inputViews.select(TIMEOUT_MILLISECONDS, TimeUnit.MILLISECONDS);
730 
731             // Keep reading till we complete all the input views.
732             int pollTime = TIMEOUT_MILLISECONDS;
733             String inputViewName = StringUtils.EMPTY;
734             Document sqlBindValuesDocument = null;
735 
736             // Polling and looping through views is required for validation mode (even if one only 1
737             // inputview for this snap), where if we dont poll, then the inputSqlString is not
738             // completely received
739             while (!pendingViews.isEmpty()) {
740                 if (!selectedDocuments.isEmpty()) {
741                     for (Map.Entry<String, Document> documentEntry :
742                             selectedDocuments.entrySet()) {
743                         Document doc = documentEntry.getValue();
744                         inputViewName = documentEntry.getKey();
745                         if (doc == null) {
746                             pendingViews.remove(documentEntry.getKey());
747                             continue;
748                         }
749 
750                         String inputDocument = (String) ((Map<?, ?>) doc.get()).get(SNAPSQL);
751 
752                         // empty inputDocument indicates that we might have received
753                         // documents that may be part of preview data from the upstream snap
754                         if (StringUtils.isBlank(inputDocument)) {
755                             pendingViews.remove(inputViewName);
756                             continue;
757                         }
758 
759                         if (StringUtils.isBlank(inputSqlString)) {
760                             inputSqlString = inputDocument;
761                             allBindingValues = extractBindingValues(doc);
762                             sqlBindValuesDocument = doc;
763                         } else {
764                             // More than one input sql received which is unexpected in this snap.
765                             // There could be something wrong with the upstream snap SQL generation
766                             throw new ConfigurationException(ERR_MULTIPLE_INPUT_SQL_RECEIVED)
767                                     .withReason(INTERNAL_ERROR_HAS_OCCURRED)
768                                     .withResolution(CONTACT_SUPPORT);
769                         }
770 
771                         pollTime = TIMEOUT_MILLISECONDS;
772                     }
773                 } else {
774                     // We are not receiving any data. Lets increment poll time.
775                     pollTime += INCREMENTS;
776                     if (pollTime > MAX_POLL_TIME) {
777                         // If the poll time exceeds the max allowed poll time,
778                         // lets fix the poll time as the max allowed poll time.
779                         pollTime = MAX_POLL_TIME;
780                     }
781                 }
782                 selectedDocuments = inputViews.select(pollTime, TimeUnit.MILLISECONDS);
783             }
784             if (StringUtils.isBlank(inputSqlString)) {
785                 throw new ConfigurationException(INPUT_SQL_NOT_RECEIVED)
786                         .withReason(INTERNAL_ERROR_HAS_OCCURRED)
787                         .withResolution(CONTACT_SUPPORT);
788             }
789 
790             if (validationInProgress && !isSchemaGenerated) {
791                 schemaSupportService.defineInputSchema(
792                         schemaProvider, inputSqlString, inputViewName);
793                 isSchemaGenerated = true;
794             }
795 
796             handleSourceColumnValueExpressions(sqlBindValuesDocument);
797             validateAndGatherTargetSourceColMappings();
798             executeTableActionAndSCD2();
799             notifyStatusReporter(statusReporter, SNAP_STATS, statistic);
800 
801             processDocument(null, null);
802         } catch (Exception ex) {
803             statistic.setSnapStatus(StatisticHolder.SnapStatus.FAIL);
804             notifyStatusReporter(statusReporter, SNAP_STATS, statistic);
805             throw ex;
806         }
807     }
