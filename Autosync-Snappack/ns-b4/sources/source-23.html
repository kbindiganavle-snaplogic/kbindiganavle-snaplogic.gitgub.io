79 /**
80  * ELT Window Functions Snap applies window function on input SQL.
81  * Conceptually it implements the following functionality:
82  *
83  * If pass through is disabled:
84  *
85  *    SELECT winfunc1(<sql_expression>)
86  *    OVER (PARTITION BY <expr1>, <expr2>, ....
87  *          ORDER BY <expr3> <ASC | DESC> <NULLS FIRST | NULLS LAST>,
88  *                   <expr4> <ASC | DESC> <NULLS FIRST | NULLS LAST>,
89  *                   ....
90  *          ROWS BETWEEN <lower bound> AND <upper bound>)
91  *    FROM (SELECT ... FROM ...WHERE...) inputSql (exp1, exp2, ...)
92  *
93  *    where <lower bound> could be:
94  *    a. UNBOUNDED PRECEDING
95  *    b. UNBOUNDED FOLLOWING
96  *    c. CURRENT ROW
97  *    d. <value> PRECEDING
98  *    e. <value> FOLLOWING
99  *
100  *    where <upper bound> could be:
101  *    a. UNBOUNDED PRECEDING
102  *    b. UNBOUNDED FOLLOWING
103  *    c. CURRENT ROW
104  *    d. <value> PRECEDING
105  *    e. <value> FOLLOWING
106  *
107  *    If pass through is enabled:
108  *
109  *    SELECT winfunc1(<sql_expression>), *
110  *    OVER (PARTITION BY <expr1>, <expr2>, ....
111  *          ORDER BY <expr3> <ASC | DESC> <NULLS FIRST | NULLS LAST>,
112  *                   <expr4> <ASC | DESC> <NULLS FIRST | NULLS LAST>,
113  *                   ....
114  *          ROWS BETWEEN <lower bound> AND <upper bound>)
115  *    FROM (SELECT ... FROM ...WHERE...) inputSql (exp1, exp2,...)
116  *
117  * @author Pradeep Sathyanarayan
118  */

852     @Override
853     protected void processDocument(final Document document, final String inputViewName) {
854 
855         String inputSqlString = extractQueryFromInputDocument(document);
856         if (StringUtils.isBlank(inputSqlString)){
857             return;
858         }
859 
860         List<String> snapTypes = new ArrayList<>();
861         snapTypes.add(ELT_WINDOW_FUNCTIONS_TITLE);
862         List<String> previousSnapTypes = extractSnapTypeFromInputDocument(document);
863         String randomString = getDerivedTableName();
864         String sqlWindow;
865         Optional<PushdownContext> optionalPushdownContext = PushdownContext.get(document);
866         PushdownContext ctx;
867 
868         boolean isPushdownEnabled = CollectionUtils.isEmpty(rewriterFunctions);
869 
870         if (canPushdown(document, WINDOW_FUNCTIONS_PUSHDOWN_SNAP_LIST)
871                 && !linearRegressionRewrites
872                 && optionalPushdownContext.isPresent()
873                 && isPushdownEnabled
874                 && !hasNestedFunction(document, snapWindowFunctionExpressionList.stream()
875                 .map(e -> {
876                     List<String> fields = new ArrayList<>();
877                     fields.addAll(e.getArgs());
878                     fields.addAll(e.getPartitionByFields());
879                     fields.addAll(e.getOrderByFields().stream()
880                             .map(SortElement::getField)
881                             .collect(Collectors.toList()));
882                     return fields;
883                 })
884                 .flatMap(Collection::stream), inputSqlString)) {
885             ctx = optionalPushdownContext.get();
886             snapTypes.addAll(previousSnapTypes);
887             pushdownWindowFunctions(ctx, document, inputSqlString, true);
888             sqlWindow = pushdownQueryBuilder.buildSql(ctx);
889         } else {
890             String tableAlias = "Window_" + randomString;
891             sqlWindow = buildWindowFunctionsQuery(document, inputSqlString, tableAlias);
892             if (linearRegressionRewrites) {
893                 //disable pushdown for the next snap
894                 ctx = null;
895             } else {
896                 ctx = PushdownContext.create(document, tableAlias);
897                 pushdownWindowFunctions(ctx, document, inputSqlString, false);
898             }
899         }
900 
901 
902         Map<String, Object> output = getOutputViewMap(document, sqlWindow,
903                 snapTypes, ctx);
904 
905         if (handlePreview(documentUtility.newDocument(output), document,
906                 account, validationInProgress, previewRowCount, executionMode,
907                 executeSqlInPreview)) {
908             return;
909         }
910         outputViews.write(documentUtility.newDocument(output), document);
911     }

919     private void pushdownWindowFunctions(PushdownContext ctx,
920                                          Document document,
921                                          String inputSqlString,
922                                          boolean replaceAliases) {
923         if (!snapWindowFunctionExpressionList.isEmpty()) {
924             SelectPart selectPart = ctx.getSelectPart();
925 
926             List<Field> windowsFunctionElementsWIthOriginalFields =
927                     snapWindowFunctionExpressionList.stream()
928                             .map(windowFunctionElement ->
929                                     new Field(windowFunctionElement.build(),
930                                             windowFunctionElement.getAlias(), true))
931                             .collect(Collectors.toList());
932 
933             if (replaceAliases) {
934                 snapWindowFunctionExpressionList = snapWindowFunctionExpressionList.stream()
935                         .map(windowFunctionElement -> replaceFields(selectPart.getExpressions(),
936                                 windowFunctionElement))
937                         .collect(Collectors.toList());
938             }
939 
940             List<Field> windowFunctionElements = snapWindowFunctionExpressionList.stream()
941                     .map(windowFunctionElement -> new Field(windowFunctionElement.build(),
942                             windowFunctionElement.getAlias(), true))
943                     .collect(Collectors.toList());
944 
945             checkColumnsExists(document, windowsFunctionElementsWIthOriginalFields,
946                     inputSqlString);
947             if (passThrough) {
948                 List<Field> expressions = new ArrayList<>(selectPart.getExpressions());
949                 expressions.addAll(0, windowFunctionElements);
950                 selectPart.setExpressions(expressions);
951             } else {
952                 selectPart.setExpressions(windowFunctionElements);
953             }
954         } else {
955             if (!passThrough) {
956                 throw new ConfigurationException(ERR_MSG_INVALID_SNAP_CONFIGURATION)
957                         .withReason(ERR_REASON_INVALID_SNAP_CONFIGURATION_WF)
958                         .withResolution(ERR_RES_INVALID_SNAP_CONFIGURATION_WF);
959             }
960         }
961     }

964      * Replace aliases with field expression
965      * exp "select count(b) c from (select a b from tbl1)"
966      * after pushing rebuilding the query we replace count(b) -> count(a) so the
967      * final query looks like "select count(a) c from tbl1"
968      * @param expressions expressions of the sub-query
969      * @param function function
970      * @return windowFunctionElement
971      */
972     private WindowsFunction replaceFields(List<Field> expressions,
973                                           WindowsFunction function) {
974         function.setArgs(replaceValues(function.getArgs(), expressions));
975         function.setPartitionByFields(replaceValues(function.getPartitionByFields(), expressions));
976         function.setOrderByFields(replaceSortValues(function.getOrderByFields(), expressions));
977         return function;
978     }

980     private List<String> replaceValues(List<String> args, List<Field> expressions) {
981         List<String> newArgs = new ArrayList<>();
982         for (String arg : args) {
983             boolean match = false;
984             for (Field expression : expressions) {
985                 if (expression.getAlias() == null) {
986                     continue;
987                 }
988                 if (expression.getAlias().equalsIgnoreCase(arg)) {
989                     newArgs.add(expression.getExpr());
990                     match = true;
991                     break;
992                 }
993             }
994             if (!match) {
995                 newArgs.add(arg);
996             }
997         }
998         return newArgs;
999     }

1001     private List<SortElement> replaceSortValues(List<SortElement> args, List<Field> expressions) {
1002         List<SortElement> newArgs = new ArrayList<>();
1003         for (SortElement arg : args) {
1004             boolean match = false;
1005             for (Field expression : expressions) {
1006                 if (expression.getAlias() == null) {
1007                     continue;
1008                 }
1009                 if (expression.getAlias().equalsIgnoreCase(arg.field)) {
1010                     newArgs.add(new SortElement(
1011                             expression.getExpr(),
1012                             arg.getSortOrder(),
1013                             arg.getSortNullsOrder(),
1014                             arg.sortWithoutNullsOrder));
1015                     match = true;
1016                     break;
1017                 }
1018             }
1019             if (!match) {
1020                 newArgs.add(arg);
1021             }
1022         }
1023         return newArgs;
1024     }

1026     private String buildWindowFunctionsQuery(Document document, String inputSqlString,
1027                                              String randomString) {
1028         String sqlWindow;
1029         if (!snapWindowFunctionExpressionList.isEmpty()) {
1030             if (linearRegressionRewrites) {
1031                 Set<String> inputSqlStringColumns =
1032                         setUpInputSqlStringColumns(document, inputSqlString);
1033                 sqlWindow = rewriter.process(inputSqlString, inputSqlStringColumns,
1034                         buildExpressionsString(),
1035                         randomString, passThrough);
1036             } else if (passThrough) {
1037                 sqlWindow = String.format(FUNCTIONS_AND_PASS_THROUGH_SQL,
1038                         buildExpressionsString(), inputSqlString, randomString);
1039             } else {
1040                 sqlWindow = String.format(FUNCTIONS_ONLY_SQL, buildExpressionsString(),
1041                         inputSqlString, randomString);
1042             }
1043         } else if (passThrough) {
1044             // if Pass through is enabled and there are no window functions
1045             // arguments in the snap, subquery pushdown is enabled then the
1046             // window functions snap can simply forward the input SQL without any processing.
1047             if (isSubQueryPushdownEnabled()) {
1048                 sqlWindow = inputSqlString;
1049             } else {
1050                 sqlWindow = String.format(PASS_THROUGH_ONLY_SQL_FUNCTIONS, inputSqlString,
1051                         randomString);
1052             }
1053         } else {
1054             throw new ConfigurationException(ERR_MSG_INVALID_SNAP_CONFIGURATION)
1055                      .withReason(ERR_REASON_INVALID_SNAP_CONFIGURATION_WF)
1056                      .withResolution(ERR_RES_INVALID_SNAP_CONFIGURATION_WF);
1057         }
1058         return sqlWindow;
1059     }

1328                 propertyValues.get(VALUE_WINDOW_FUNCTIONS_PATH);
1329         if (userSpecifiedFunctionTable == null) {
1330             return windowFunctionList;
1331         }
1332         for (Map<String, String> userSpecifiedFunctionRow : userSpecifiedFunctionTable) {
1333             String function =
1334                     jsonPathUtil.nullableRead(VALUE_WINDOW_FUNCTION_PATH, userSpecifiedFunctionRow);
1335             String valueBasedWindowFunctionName = getValueBasedWindowFunctionName(function);
1336 
1337             ExpressionProperty field =
1338                     propertyValues.getExpressionPropertyFor(userSpecifiedFunctionRow, FIELD_PROP);
1339 
1340             String originalFunctionName = getOriginalFunctionName(valueBasedWindowFunctionName);
1341             String readOrder = getReadOrder(valueBasedWindowFunctionName);
1342             boolean ignoreNulls = false;
1343             String nullsClause = getNullsClause(valueBasedWindowFunctionName);
1344             if (isDeltalakeDb()) {
1345                 ignoreNulls = StringUtils.isNotBlank(nullsClause);
1346             }
1347 
1348             ExpressionProperty nthValueFieldExpr = propertyValues.getExpressionPropertyFor(
1349                     userSpecifiedFunctionRow, NTH_VALUE_FIELD_PROP);
1350 
1351             ExpressionProperty resultFieldExpr = propertyValues.getExpressionPropertyFor(
1352                     userSpecifiedFunctionRow, RESULT_FIELD_PROP);
1353 
1354             String overClause = EMPTY_STRING;
1355             boolean reverseOrderByClause = false;
1356             if (StringUtils.isNotBlank(readOrder) &&
1357                     readOrder.equalsIgnoreCase(NTH_VALUE_FROM_LAST_OPTION) && isRedshiftDb()) {
1358 
1359                 // Redshift does not support FROM LAST option in the NTH_VALUE
1360                 // function while Snowflake does. Hence we are doing a rewrite to
1361                 // support the FROM LAST option in the NTH_VALUE function for Redshift.
1362 
1363                 // Here we use the NTH_VALUE function with FROM FIRST option to perform
1364                 // the rewrite and the ORDER BY clause for this function needs to be
1365                 // This means if the Sort order is ASC, we need to make it DESC and vice
1366                 // versa. Similarly the NULLS Order needs to be reversed as well. If it
1367                 // NULLS FIRST, it needs to be made NULLS LAST and vice versa.
1368 
1369                 // Care needs to be taken that the reversal needs to be done only for
1370                 // the NTH_VALUE functions using the FROM LAST option. The ordering
1371                 // should not be reversed for other flavors of function whether it be
1372                 // the value based functions or any other functions used in the window
1373                 // snap. This is because if we do, it will cause wrong results and can
1374                 // cause data integrity issues even though the pipeline execution will
1375                 // go through successfully. It needs to be noted that the ORDER BY
1376                 // and PARTITION BY clauses are shared by all the functions in a given
1377                 // Windows snap.
1378 
1379                 // In order to achieve the rewrite, we reverse the ORDER BY
1380                 // and then generate the OVER Clause so that the reversed ORDER BY only
1381                 // takes affect for the NTH_VALUE function using FROM LAST option.
1382                 reverseOrderByClause = true;
1383                 readOrder = EMPTY_STRING;
1384             }
1385             String rewrittenFunctionExpression;
1386             List<String> functionArgs = new ArrayList<>();
1387             if (originalFunctionName.equalsIgnoreCase(ORIGINAL_NTH_VALUE)) {
1388                 if (isBigQueryDb()) {
1389                     rewrittenFunctionExpression = String.format(" %s( %s, %s %s) %s",
1390                             originalFunctionName, "%s", "%s", nullsClause, "%s");
1391                     functionArgs.add(field.eval(null));
1392                     functionArgs.add(nthValueFieldExpr.eval(null));
1393                 } else {
1394                     rewrittenFunctionExpression = String.format(" %s( %s, %s ) %s %s %s",
1395                             originalFunctionName, "%s", "%s", readOrder, nullsClause, "%s");
1396                     functionArgs.add(field.eval(null));
1397                     functionArgs.add(nthValueFieldExpr.eval(null));
1398                 }
1399             } else {
1400                 if (isDeltalakeDb()) {
1401                     String ignoreNullsString = ignoreNulls ? ", True" : "";
1402                     rewrittenFunctionExpression = String.format(" %s( %s%s ) %s",
1403                             originalFunctionName, "%s", ignoreNullsString, "%s");
1404 
1405                     functionArgs.add(field.eval(null));
1406                 } else if (isBigQueryDb()) {
1407                     rewrittenFunctionExpression = String.format(" %s( %s %s) %s",
1408                             originalFunctionName, "%s", nullsClause, "%s");
1409                     functionArgs.add(field.eval(null));
1410                 } else {
1411                     rewrittenFunctionExpression = String.format(" %s( %s ) %s %s",
1412                             originalFunctionName, "%s", nullsClause, "%s");
1413                     functionArgs.add(field.eval(null));
1414                 }
1415             }
1416 
1417             List<SortElement> functionsOrderByFields = orderByFields;
1418             if (reverseOrderByClause) {
1419                 functionsOrderByFields = orderByFields.stream()
1420                         .map(sortElement -> new SortElement(
1421                                 sortElement.getField(),
1422                                 reverseSortOrder(sortElement.getSortOrder()),
1423                                 reverseNullsOrder(sortElement.getSortNullsOrder()),
1424                                 sortElement.isSortWithoutNullsOrder()))
1425                         .collect(Collectors.toList());
1426             }
1427 
1428             WindowsFunction windowsFunction = new WindowsFunction(functionArgs,
1429                     valueBasedWindowFunctionName,
1430                     rewrittenFunctionExpression,
1431                     resultFieldExpr.eval(null),
1432                     partitionByFields,
1433                     functionsOrderByFields);
1434 
1435             windowFunctionList.add(windowsFunction);
1436         }
1437 
1438         return windowFunctionList;
1439     }

1711     /**
1712      * Find rewriting target list type and return the type and
1713      * a compressed list from the given "functions".
1714      *
1715      * 1. If the given "partitionByFieldsList" is blank,
1716      *    return TYPE_ONE_NO_PARTITION_BY_WITH_ONE_KEY_PAIR.
1717      * 2. remove duplicates of the given "functions" list by <function type, argument1, argument 2>
1718      *    triples as key. For an argument pair, two pairs are reduced to one if the function type
1719      *    is commutative.
1720      * 3. Set up the "compressedList" after removing duplicates.
1721      * 4. Return the type.
1722      *
1723      * @param   dbType:                     IN
1724      * @param   partitionByFieldsList:      IN
1725      * @param   functions:                  IN
1726      * @param   map:                        OUT
1727      * @return  RewriteTargetListType
1728      */
1729     private RewriteTargetListType findRewriteTargetListTypeAndCompress(
1730             String dbType,
1731             String partitionByFieldsList,
1732             List<RewriterFunction> functions,
1733             Map<String, RewriterFunction> map) {
1734 
1735         RewriteTargetListType targetListType = null;
1736         // This set is for counting number of unique pairs of arguments to the given "functions".
1737         Set<String> argumentPairSet = new HashSet<>();
1738 
1739         // Remove duplicate functions by argument pairs using map.
1740         int setIdx = 0;
1741         for (int i = 0; i < functions.size(); i++) {
1742             RewriterFunction current = functions.get(i);
1743             current.setOriginalArrayIdx(i);
1744             List<String> keys =
1745                     EltRewriterUtils.getKeysForRewriterTargetFunction(
1746                             current.getFunctionName(),
1747                             current.getField1Str(), current.getField2Str(),
1748                             eltWindowFunctionsList);
1749             String keyToMap = keys.get(0);
1750             current.setKeyToMap(keyToMap);
1751             RewriterFunction mapPrevious = map.get(keyToMap);
1752             argumentPairSet.add(keys.get(1));
1753             if (mapPrevious == null) {
1754                 map.put(keyToMap, current);
1755                 current.setCompressedArrayIdx(setIdx++);
1756                 current.setCompressedArrayEntry(true);
1757             } else {
1758                 // Two cases for current's <field1Str, field2Str>
1759                 // if current is commutative: <f1, f2>, <f2, f1>
1760                 current.setCompressedArrayIdx(mapPrevious.getCompressedArrayIdx());
1761             }
1762         }
1763 
1764         this.argumentPairMap = EltRewriterUtils.buildArgumentPairMap(functions,
1765                 eltWindowFunctionsList);
1766         // Determine and return target list type
1767         if (StringUtils.isBlank(partitionByFieldsList)) {
1768             return RewriteTargetListType.
1769                     TYPE_ONE_NO_PARTITION_BY_WITH_ONE_OR_MORE_UNIQUE_FUNCTION_ARGUMENT_PAIRS;
1770         } else if (argumentPairSet.size() > 1) {
1771             targetListType = RewriteTargetListType.
1772                     TYPE_THREE_PARTITION_BY_WITH_MORE_THAN_ONE_FUNCTION_OR_ARGUMENT_PAIRS;
1773         } else {
1774             targetListType = RewriteTargetListType.
1775                     TYPE_TWO_PARTITION_BY_WITH_ONE_UNIQUE_FUNCTION_AND_ARGUMENT_PAIR;
1776         }
1777 
1778         return targetListType;
1779     }

1925     // This function builds the OVER clause for window functions
1926     // based on the window function name. Each window function
1927     // has different OVER clause support on different databases.
1928     // Some have only PARTITION BY support, some have PARTITION
1929     // BY and ORDER by support and finally some have PARTITION
1930     // BY, ORDER BY and Window Framing Clause support. We need
1931     // to keep in mind, all of these clauses are non-mandatory
1932     // in window functions. But we should not end up passing
1933     // unnecessary clauses to functions which don't honor it and
1934     // it can cause syntax errors. Each of the functions may
1935     // have different support on different databases. Care needs
1936     // to be taken to attach the right clauses to right functions
1937     // on right databases.
1938     private String buildOverClause(String windowFunctionName, String partitionByClause,
1939                                    String orderByClause) {
1940         if (StringUtils.isBlank(windowFunctionName)) {
1941             throw invalidFunctionException(windowFunctionName);
1942         }
1943         final String windowFunctionNameUpperCase = windowFunctionName.toUpperCase();
1944 
1945         if (eltWindowFunctionsList.getPartitionByOnlyFunctions()
1946                 .contains(windowFunctionNameUpperCase)) {
1947             return buildOverClauseWithPartitionBy(partitionByClause);
1948         } else if (eltWindowFunctionsList.getPartitionByAndOrderByOnlyFunctions()
1949                 .contains(windowFunctionNameUpperCase)) {
1950             return buildOverClauseWithPartitionByOrderBy(partitionByClause, orderByClause);
1951         } else if (eltWindowFunctionsList.getPartitionByAndOrderByWindowFrameFunctions()
1952                 .contains(windowFunctionNameUpperCase)) {
1953             return buildOverClauseWithPartitionByOrderByWindowFrame(partitionByClause,
1954                     orderByClause);
1955         } else {
1956             throw invalidFunctionException(windowFunctionName);
1957         }
1958     }
