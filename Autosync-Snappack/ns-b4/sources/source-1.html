502         try {

505             }
506             sqlStatement = sqlStatement.trim();

508             if (pipelineParams == null || pipelineParams.isEmpty()) {
509                 transformedSql = sqlStatement;

537                 long offsetLong = Long.parseLong(offsetValue);

544             long limitInputLong = Long.parseLong(limitInput.strip());
545             long previewRowCountLong = Long.parseLong(previewRowCount);
546             previewRowCount = String.valueOf(Math.min(limitInputLong, previewRowCountLong));

659         try {
660             int numberOfAffectedRecords = 0;
661             dbConnection = jdbcOperations.acquireConnection(account);
662 
663             PreparedStatement statement = dbConnection.prepareStatement(sql);
664             statementMap.put(sql, statement);
665 
666             statement.execute();
667 
668             int updateCount = statement.getUpdateCount();
669 
670             if (updateCount != -1) {
671                 numberOfAffectedRecords = updateCount;
672             } else {
673 //        TODO: DATA-4455 [ELT] Hide credentials before logging sql query

680             }

682                     .formatWith(getMaskedErrorMessages(sql))
683                     .withReason(getMaskedErrorMessages(getSQLErrorMessage(e)))
684                     .withResolution(CHECK_QUERY)

714     private void cancelQuery() {
715         // Currently we only support Abort processing
716         // for snowflake, redshift and azure synapse, bigquery databases. As other databases
717         // support are added, then this logic needs to
718         // be enhanced.
719         if (StringUtils.isNotBlank(getQueryToBeStopped())) {
720 
721             LOGGER.debug("Trying to kill the ELT operation");
722 
723             switch (getName()) {
724                 case SNOWFLAKE:
725                     LOGGER.debug(DEBUG_TEMPLATE_TRYING_TO_KILL, SNOWFLAKE);
726                     cancelQuery(account, this::findSnowflakeCurrentQueryId,
727                             SNOWFLAKE_FORMAT_CANCEL_QUERY);
728                     break;
729                 case REDSHIFT:
730                     LOGGER.debug(DEBUG_TEMPLATE_TRYING_TO_KILL, REDSHIFT);
731                     cancelQuery(account, this::findRedshiftCurrentQueryId,
732                             REDSHIFT_FORMAT_CANCEL_QUERY);
733                     break;
734                 case AZURE_SYNAPSE:
735                     LOGGER.debug(DEBUG_TEMPLATE_TRYING_TO_KILL, AZURE_SYNAPSE);
736                     cancelQuery(account, this::findAzureSynapseCurrentQuerySessionId,
737                             AZURE_SYNAPSE_FORMAT_KILL_QUERY);
738                     break;
739                 case DATABRICKS_LAKEHOUSE_PLATFORM:
740                     LOGGER.debug(DEBUG_TEMPLATE_TRYING_TO_KILL, DATABRICKS_LAKEHOUSE_PLATFORM);
741                     cancelDeltaQuery();
742                     break;
743                 case BIGQUERY:
744                     LOGGER.debug(DEBUG_TEMPLATE_TRYING_TO_KILL, BIGQUERY);
745                     cancelQuery(account, this::findBigQueryCurrentQueryId,
746                             BIGQUERY_FORMAT_CANCEL_QUERY);
747                     break;
748                 default:
749                     LOGGER.debug("Cannot determine database");
750             }
751         }
752     }

771         try (Statement stmt = connection.createStatement()) {
772             if (!stmt.execute(cancelQuery)) {

779             }
780         } catch (SQLException sqlException) {
781             throwCancelException(sqlException);
782         } finally {
783             jdbcOperations.cleanup();
784         }
785     }

967     public boolean handlePreview(Document document, Document original, DatabaseAccount account,
968                                  boolean validationRunning, String previewRowCnt,
969                                  String executionMode,
970                                  boolean executeSqlInPreview) {
971 
972         if (!validationRunning) /* If the user has not pressed the validate button in the UI */ {
973             return false;
974         }
975 
976         /* If the executionMode is set to Execute Only or Disabled in the snap,
977          * then the snap will just not produce any preview output, but the snap
978          * will still produce the generated SQL output and the output will not
979          * be shown in the preview.
980          */
981         if (!executionMode.equalsIgnoreCase(ExecutionMode.ENABLED.toString())) {
982             return false;
983         }
984 
985         Map<String, Object> docMap = (Map) document.get();
986         if (!docMap.containsKey(SNAPSQL)) {
987             return true;
988         }
989 
990         String inputSqlString;
991         inputSqlString = (String) docMap.get(SNAPSQL);
992         previewSqlString = inputSqlString;
993 
994         List<String> previousSnapTypes = extractSnapTypeFromInputDocument(document);
995         if (previousSnapTypes.contains(ELT_LIMIT_TITLE)) {
996             previewSqlString = getStarQueryWith(previewSqlString, VALIDATION_QUERY_NAME_PREFIX);
997         }
998 
999         if (previewSqlString.toLowerCase().contains(SELECT_TOP) && isRedshiftDb()) {
1000             previewSqlString = String.format(LIMIT_SQL_SAMPLE_REDSHIFT, docMap.get(SNAPSQL),
1001                     previewRowCnt);
1002         } else if (isAzureSynapseDb()) {
1003             String originalSql = (String) docMap.get(SNAPSQL);
1004             if (rewriteCteAsSubquery &&
1005                     orderByClauseExistsForTheMainQueryBlock(originalSql)) {
1006                 previewSqlString = buildPreviewSqlForWithClauseSqlAzureSynapse(
1007                         originalSql, previewRowCnt, getDerivedTableName());
1008                 inputSqlString = eliminateOrderByClause(inputSqlString);
1009             } else {
1010                 previewSqlString = String.format(LIMIT_WITH_OFFSET_ZERO_EQUIVALENT_SQL,
1011                         previewRowCnt, originalSql, getDerivedTableName());
1012             }
1013         } else {
1014             previewSqlString = updateLimitClauseForPreview(previewSqlString, previewRowCnt);
1015         }
1016 
1017         List<Object> bindValues = (List) docMap.get(SNAPSQL_BIND_VALUES);
1018 
1019         List<String> snapTypes = extractSnapTypeFromInputDocument(document);
1020         Map<String, Object> output = new LinkedHashMap<>();
1021         output.put(SNAPSQL, inputSqlString);
1022         output.put(SNAPSQL_BIND_VALUES, bindValues);
1023         output.put(ELT_SNAP_TYPE, snapTypes);
1024         if (docMap.containsKey(STATUS)) {
1025             output.put(STATUS, docMap.get(STATUS));
1026         }
1027 
1028         if (docMap.containsKey(TARGET_TABLE_WRITER_SQL)) {
1029             output.put(TARGET_TABLE_WRITER_SQL, docMap.get(TARGET_TABLE_WRITER_SQL));
1030         }
1031         if (docMap.containsKey(OUTPUT_SCHEMA)) {
1032             output.put(OUTPUT_SCHEMA, docMap.get(OUTPUT_SCHEMA));
1033         }
1034 
1035         if (docMap.containsKey(TARGET_TABLE_WRITER_SQL)) {
1036             output.put(TARGET_TABLE_WRITER_SQL, docMap.get(TARGET_TABLE_WRITER_SQL));
1037         }
1038         if (docMap.containsKey(ELT_SORT_SNAP_ORDER_BY_CLAUSE)) {
1039             output.put(ELT_SORT_SNAP_ORDER_BY_CLAUSE, docMap.get(ELT_SORT_SNAP_ORDER_BY_CLAUSE));
1040         }
1041         if (docMap.containsKey(ELT_SORT_SNAP_INPUT_SQL)) {
1042             output.put(ELT_SORT_SNAP_INPUT_SQL, docMap.get(ELT_SORT_SNAP_INPUT_SQL));
1043             if (isSubQueryPushdownEnabled()) {
1044                 output.put(SNAPSQL, docMap.get(ELT_SORT_SNAP_INPUT_SQL));
1045             }
1046         }
1047         PushdownContext.propagate(docMap, output);
1048 
1049         writeToOutputView(output, original);
1050 
1051         /* If the user has not set the account for the snap, then the snap will just
1052          * produce the generated SQL in the preview
1053          */
1054         if (account instanceof EltDefaultAccount) {
1055             return true;
1056         }
1057 
1058         /* If the user has not checked the flag Execute Snap SQL in Preview , then
1059          * the snap will just produce the generated SQL in the preview.
1060          */
1061         if (!executeSqlInPreview) {
1062             return true;
1063         }
1064 
1065         if (jdbcOperations instanceof SnowflakeOperations) {
1066             SnowflakeRecordMapper recordMapper = ((SnowflakeOperations) jdbcOperations)
1067                     .getRecordMapper();
1068             recordMapper.setEltMode(true);
1069         } else if (jdbcOperations instanceof DatabricksLakehouseOperations) {
1070             DatabricksLakehouseRecordMapper recordMapper =
1071                     ((DatabricksLakehouseOperations) jdbcOperations).getRecordMapper();
1072             recordMapper.setEltMode(true);
1073         }
1074 
1075         /* If we have reached here, that means the the Snap will be executed in
1076          * in preview mode. The snap will run the generated SQL on the database
1077          * endpoint and output the preview data.
1078          * The preview output will contain both the generated SQL and the preview
1079          * data.
1080          */
1081         executePreviewMode(document, account, previewSqlString, bindValues);
1082         return true;
1083     }

1458             case JdbcOperations.SNOWFLAKE:
1459             case JdbcOperations.AZURE_SYNAPSE:
1460                 /* This is a performance optimization. If a table
1461                  * was created during the LOAD/INSERT-SELECT/MERGE INTO
1462                  * operation, there is no point to do rollback on
1463                  * the table that is going to be dropped immediately.
1464                  * Instead we simply drop the table and return.
1465                  * This way we save on expensive rollback operation.
1466                  */
1467                 if (isCreateTargetTable) {
1468                     dropTable(targetTablePath, error);
1469                     return;
1470                 }
1471                 if (isCreatedStagedTable) {

1475                     dropFileFormatObject(this.fileFormatNameForExternalStage, error);

1477                 if (isCreatedExternalStage) {
1478                     dropExternalStageObject(this.externalStageName, error);
1479                 }

1482             case JdbcOperations.DATABRICKS_LAKEHOUSE_PLATFORM:
1483             case JdbcOperations.BIGQUERY:
1484                 if (isCreateTargetTable) {
1485                     dropTable(targetTablePath, error);
1486                     return;

1567             if (hasAlterTableBeenExecuted || isCreateTargetTable) {
1568                 // Drop the original table which might be modified and rename the cloned
1569                 // table to original table to retain the table schema and data; can
1570                 // happen before or after the COPY/SCD2 operation
1571                 sqlDropTable = buildDropTableSql(targetTablePath);
1572                 executeSqlStmt(sqlDropTable, bindValues, account);
1573                 // Below check is necessary as there can be case when there is no cloned
1574                 // table (When "Drop and Create Table" is chosen and that table does not
