276     public void execute() {
277         Set<String> pendingViews = Sets.newHashSet(inputViews.names());
278         // Keep reading till we complete all the input views.
279         Map<String, Document> selectedDocuments =
280                 inputViews.select(pendingViews, TIMEOUT_MILLISECONDS, TimeUnit.MILLISECONDS);
281         // Keep reading till we complete all the input views.
282         int pollTime = TIMEOUT_MILLISECONDS;
283         List<Object> leftDocBindValues = new ArrayList<>();
284         List<Object> rightDocBindValues = new ArrayList<>();
285 
286         Set<String> leftInputSqlRenamedColumns = new LinkedHashSet<>();
287         leftInputSqlRenamedColumns.add("*");
288 
289         Set<String> rightInputSqlRenamedColumns = new LinkedHashSet<>();
290         rightInputSqlRenamedColumns.add("*");
291 
292         Document leftInputDoc = null;
293         Document rightInputDoc = null;
294 
295         while (!pendingViews.isEmpty()) {
296             if (!selectedDocuments.isEmpty()) {
297                 for (Map.Entry<String, Document> documentEntry : selectedDocuments.entrySet()) {
298                     String docInputViewName = documentEntry.getKey();
299                     if (docInputViewName.equals(leftSideInputViewName)) {
300                         Document doc = documentEntry.getValue();
301                         if (doc == null) {
302                             // null value is used to mark a view that has completed
303                             pendingViews.remove(documentEntry.getKey());
304                             continue;
305                         } else {
306                             String inputSqlString = extractQueryFromInputDocument(doc);
307                             if (inputSqlString != null) {
308                                 if (validationInProgress) {
309                                     schemaSupportService.defineInputSchema(schemaProvider,
310                                             inputSqlString, docInputViewName);
311                                 }
312                                 leftInputDoc = doc;
313                                 leftInputSqlString = inputSqlString;
314                                 leftDocBindValues = extractBindingValues(leftInputDoc);
315                             }
316                         }
317                     } else if (docInputViewName.equals(rightSideInputViewName)) {
318                         Document doc = documentEntry.getValue();
319                         if (doc == null) {
320                             // null value is used to mark a view that has completed
321                             pendingViews.remove(documentEntry.getKey());
322                             continue;
323                         } else {
324                             String inputSqlString = extractQueryFromInputDocument(doc);
325                             if (inputSqlString != null) {
326                                 if (validationInProgress) {
327                                     schemaSupportService.defineInputSchema(schemaProvider,
328                                             inputSqlString, docInputViewName);
329                                 }
330                                 rightInputDoc = doc;
331                                 rightInputSqlString = inputSqlString;
332                                 rightDocBindValues = extractBindingValues(rightInputDoc);
333                             }
334                         }
335                     }
336                 }
337                 pollTime = TIMEOUT_MILLISECONDS;
338             } else {
339                 // We are not receiving any data. Lets increment poll time.
340                 pollTime += INCREMENTS;
341                 if (pollTime > MAX_POLL_TIME) {
342                     // If the poll time exceeds the max allowed poll time,
343                     // lets fix the poll time as the max allowed poll time.
344                     pollTime = MAX_POLL_TIME;
345                 }
346             }
347             selectedDocuments = inputViews.select(pendingViews, pollTime, TimeUnit.MILLISECONDS);
348         }
349 
350         String dtName1 = "Join1_" + getDerivedTableName();
351         String dtName2 = "Join2_" + getDerivedTableName();
352 
353         if (StringUtils.isNotBlank(leftTableAliasName)) {
354             dtName1 = leftTableAliasName;
355         }
356 
357         if (StringUtils.isNotBlank(rightTableAliasName)) {
358             dtName2 = rightTableAliasName;
359         }
360 
361         String checkTableColumnQuery = buildFetchColumnsSqlString(leftInputSqlString);

363

385

387         String sqlJoin = EMPTY_STRING;
388         switch (joinType) {
389             case JOIN_TYPE_INNER:
390             case JOIN_TYPE_LEFT_OUTER:
391             case JOIN_TYPE_RIGHT_OUTER:

394                 sqlJoin = String.format(JOIN_SQL, resultantColumnNames,
395                         leftInputSqlString, dtName1, joinType,

410                 } else {
411                     // use native natural or cross joins
412                     // (NATURAL INNER, NATURAL LEFT/RIGHT/FULL OUTER, CROSS)
413                     sqlJoin = String.format(NATURAL_OR_CROSS_JOIN_SQL, resultantColumnNames,
414                             leftInputSqlString, dtName1, joinType,
415                             rightInputSqlString, dtName2);
416                     ctx.getFromPart().setTable(String.format("( %s ) %s %s JOIN ( %s ) %s",
417                             leftInputSqlString, dtName1, joinType,
418                             rightInputSqlString, dtName2));
419                 }
420                 break;
421             case JOIN_TYPE_LEFT_SEMI:
422             case JOIN_TYPE_LEFT_ANTI:
423                 if (leftSemiOrleftAntiJoinRewriteEnabled) {
424                     // use rewritten left semi or left anto joins
425                     sqlJoin = leftSemiOrleftAntiJoinRewrite(leftInputSqlColumns,
426                             rightInputSqlColumns, dtName1, dtName2, ctx);
427                 } else {
428                     // use native left semi and left anti joins
429                     sqlJoin = String.format(JOIN_SQL, resultantColumnNames,
430                             leftInputSqlString, dtName1, joinType,
431                             rightInputSqlString, dtName2, joinCondition);
432                     ctx.getFromPart().setTable(String.format("( %s ) %s %s JOIN ( %s ) %s ON %s",
433                             leftInputSqlString, dtName1, joinType,
434                             rightInputSqlString, dtName2, joinCondition));
435                 }
436                 break;
437         }
438 
439         List<Object> bindingValues = new ArrayList<>();
440         bindingValues.addAll(leftDocBindValues);
441         bindingValues.addAll(rightDocBindValues);
442 
443         sqlJoin = addUniqueQueryID(sqlJoin);
444 
445         Map<String, Object> output = getOutputViewMap(sqlJoin, bindingValues,
446                 Collections.singletonList(ELT_JOIN_TITLE), ctx);
447         if (handlePreview(documentUtility.newDocument(output), null, account,
448                 validationInProgress, previewRowCount, executionMode, executeSqlInPreview)) {
449             return;
450         }
451         outputViews.write(documentUtility.newDocument(output));
452     }

454     private String naturalJoinRewrite(Set<String> leftInputSqlColumns,
455                                       Set<String> rightInputSqlColumns,
456                                       String dtName1, String dtName2,
457                                       PushdownContext ctx) {
458         // join type specified is NATURAL type and database type is Synapse
459         // Note : Azure Synapse DB does not support NATURAL JOIN type(s).
460         // So these NATURAL JOIN types (NATURAL INNER, NATURAL LEFT/RIGHT/FULL OUTER) can be
461         // implemented by rewriting SQL query using regular INNER and OUTER JOIN type(s)
462         // which is anyway supported by Azure Synapse DB.
463 
464         // Sample of SQL queries simulating above, and based on which generalized cases have been
465         // implemented below. Lets say we have two tables as following :
466         // CREATE TABLE orders (
467         //  ord_id int,
468         //  cust_id int,
469         //  amount int);
470 
471         // CREATE TABLE customers (
472         //  cust_id int,
473         //  cust_name varchar(100));
474 
475         // Natural Inner Join (implement natural join by rewriting SQL INNER JOIN Query)
476         // SELECT L.ord_id, L.cust_id, L.amount, R.cust_name
477         //    FROM orders L INNER JOIN customers R
478         //        ON L.cust_id = R.cust_id;
479 
480         // Natural Left Outer Join (
481         // implement natural left outer join by rewriting SQL LEFT OUTER JOIN Query)
482         // SELECT L.ord_id, L.cust_id, L.amount, R.cust_name
483         //    FROM orders L LEFT OUTER JOIN customers R
484         //        ON L.cust_id = R.cust_id;
485 
486         // Natural Right Outer Join (
487         // implement natural right outer join by rewriting SQL RIGHT OUTER JOIN Query)
488         // SELECT L.ord_id, R.cust_id, L.amount, R.cust_name
489         //    FROM orders L RIGHT OUTER JOIN customers R
490         //        ON L.cust_id = R.cust_id;
491 
492         // Natural Full Outer Join (
493         // implement natural full outer join by rewriting SQL FULL OUTER JOIN Query)
494         // SELECT L.ord_id,
495         //        CASE
496         //            WHEN L.cust_id is NOT NULL THEN L.cust_id
497         //            ELSE R.cust_id
498         //        END AS cust_id,
499         //        L.amount,
500         //        R.cust_name
501         //    FROM orders L FULL OUTER JOIN customers R
502         //        ON L.cust_id = R.cust_id;
503 
504         // NATURAL join implicitly joins 2 tables on columns with same names (equi join

655     private String leftSemiOrleftAntiJoinRewrite(Set<String> leftInputSqlColumns,
656                                                  Set<String> rightInputSqlColumns,
657                                                  String dtName1, String dtName2,
658                                                  PushdownContext ctx) {
659         // join type specified is LEFT SEMI or LEFT ANTI type and database type is
660         // Snowflake/Redshift/Synapse
661         // Note : Snowflake/Redshift/Azure Synapse DB do not support LEFT SEMI/ANTI JOIN type(s).
662         // So these LEFT SEMI/ANTI can be implemented by rewriting SQL query using regular
663         // JOIN types
664         // supported by any DB. Just ensure to project the columns from left side table only in the
665         // resultant columns
666 
667         // Sample of SQL queries simulating above, and based on which generalized cases have been
668         // implemented below. Lets say we have two tables as following :
669         // CREATE TABLE orders (
670         //  ord_id int,
671         //  cust_id int,
672         //  amount int);
673 
674         // CREATE TABLE customers (
675         //  cust_id int,
676         //  cust_name varchar(100));
677 
678         // LEFT SEMI Join (implement by rewriting SQL INNER JOIN Query and projecting left table
679         // columns only)
680         // Native query:
681         // SELECT * FROM orders L LEFT SEMI JOIN customers ON L.cust_id < R.cust_id;
682         //
683         // Rewritten query:
684         // SELECT * FROM orders L
685         //  INNER JOIN customers R
686         //       ON L.cust_id < R.cust_id);
687 
688         // LEFT ANTI Join (implement by rewriting SQL OUTER LEFT JOIN Query and filtered
689         // on rows with
690         // right table key as NULL. And project left table columns only)
691         // Native query:
692         // SELECT * FROM orders L LEFT ANTI JOIN customers ON L.cust_id < R.cust_id;
693         //
694         // Rewritten query:
695         // SELECT * FROM orders L
696         // LEFT OUTER JOIN (SELECT 'NON_NULL' AS NON_NULL_COLUMN, * FROM customers) R
697         //       ON L.cust_id < R.cust_id)
698         // WHERE R.NON_NULL_COLUMN IS NULL;
699 
700         String sqlJoin = EMPTY_STRING;
701         switch (joinType) {
702             case JOIN_TYPE_LEFT_SEMI:
703                 sqlJoin = String.format(REWRITE_LEFT_SEMI_JOIN_SQL,
704                         dtName1, leftInputSqlString, dtName1, rightInputSqlString, dtName2,
705                         joinCondition);
706                 ctx.getFromPart().setTable(String.format("( %s ) %s INNER JOIN ( %s ) %s ON %s",
707                         leftInputSqlString, dtName1, rightInputSqlString, dtName2,
708                         joinCondition));
709                 ctx.getSelectPart().setExpressions(
710                         Collections.singletonList(new Field(String.format("%s.*", dtName1))));
711                 break;
712             case JOIN_TYPE_LEFT_ANTI:
713                 sqlJoin = String.format(REWRITE_LEFT_ANTI_JOIN_SQL,
714                         dtName1, leftInputSqlString, dtName1, rightInputSqlString, dtName2,
715                         dtName2, joinCondition, dtName2);
716                 ctx.getSelectPart().setExpressions(
717                         Collections.singletonList(new Field(String.format("%s.*", dtName1))));
718                 ctx.getFromPart().setTable(String.format("( %s ) %s LEFT OUTER JOIN " +
719                         "(SELECT 'NON_NULL' AS NON_NULL_COLUMN, * FROM ( %s ) %s) %s ON %s",
720                         leftInputSqlString, dtName1, rightInputSqlString, dtName2,
721                         dtName2, joinCondition));
722                 ctx.setWherePart(new WherePart());
723                 ctx.getWherePart()
724                         .addCondition(String.format("%s.NON_NULL_COLUMN IS NULL", dtName2));
725                 break;
726         }
727 
728         return sqlJoin;
729     }

783     private String buildNaturalJoinRewriteSql(
784             Set<String> selectColumnNames, String leftInputSqlString,
785             String leftAlias, String joinType, String rightInputSqlString,
786             String rightAlias, StringBuffer naturalJoinCondition, PushdownContext ctx) {
787         String sqlJoin;
788 
789         // append natural join condition to sqlJoin (only when identically named columns are found
790         // in left and right tables)
791         // Note : If a NATURAL join is specified but no identically named pairs of columns exist
792         // in the tables to be joined, the query defaults to a cross-join.
793         if (naturalJoinCondition.length() > 0) {
794             sqlJoin = String.format(REWRITE_NATURAL_JOIN_SQL,
795                             StringUtils.join(selectColumnNames, ","), leftInputSqlString,
796                             leftAlias, joinType, rightInputSqlString, rightAlias);
797             sqlJoin += " ON " + naturalJoinCondition;
798             ctx.getFromPart().setTable(String.format("( %s ) %s %s JOIN ( %s ) %s ON %s",
799                     leftInputSqlString, leftAlias, joinType,
800                     rightInputSqlString, rightAlias, naturalJoinCondition));
801         } else {
802             sqlJoin = String.format(REWRITE_NATURAL_JOIN_SQL,
803                             StringUtils.join(selectColumnNames, ","), leftInputSqlString,
804                             leftAlias, "CROSS ", rightInputSqlString, rightAlias);
805             ctx.getFromPart().setTable(String.format("( %s ) %s %s JOIN ( %s ) %s",
806                     leftInputSqlString, leftAlias, "CROSS ",
807                     rightInputSqlString, rightAlias));
808         }
809 
810         return sqlJoin;
811     }
