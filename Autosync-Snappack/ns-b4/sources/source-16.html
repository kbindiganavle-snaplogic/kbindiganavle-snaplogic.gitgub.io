516             }
517             oldSourceTableName = dtName;
518 
519             setupMergeIntoAllOperationsValuesMap(document, inputSqlString, dtName);
520             doAutomaticSourceAliasPrefix(document, inputSqlString, dtName);
521 
522             buildQueries();
523 
524             if (validationInProgress) {
525                 if (!isSchemaGenerated) {
526                     schemaSupportService.defineInputSchema(
527                             schemaProvider, inputSqlString, inputViewName);
528                     isSchemaGenerated = true;
529                 }
530 
531                 if (isCreateTargetTable) {
532                     String createTableQuery = buildCreateTableForPreviewQuery();
533                     executedQueries.add(createTableQuery);
534                     mergeIntoExecuteSqlStmt(
535                             Map.entry(CREATE_TABLE_FOR_PREVIEW_SQL_KEY, createTableQuery));
536                 }
537 
538                 output.put(TARGET_TABLE_WRITER_SQL, queriesToExecute);
539 
540                 if (handlePreview(documentUtility.newDocument(output), document, account,
541                         validationInProgress, previewRowCount, executionMode,
542                         executeSqlInPreview)) {
543                     statistic.setSnapStatus(StatisticHolder.SnapStatus.DONE);
544                     handleSuccess();
545                     return;
546                 }
547             }
548 
549             statusReporter.beginUnknown();
550 
551             executeQueries();
552 
553             statistic.setSnapStatus(StatisticHolder.SnapStatus.DONE);
554             LOG.info("Statistic is: {}", statistic);
555 
556             handleSuccess();
557             notifyStatusReporter(statusReporter, SNAP_STATS, statistic);
558 
559             output.put(STATISTIC_KEY, toMap(statistic));
560             outputViews.write(documentUtility.newDocument(output), document);
561         } catch (Exception ex) {
562             statistic.setSnapStatus(StatisticHolder.SnapStatus.FAIL);
563             notifyStatusReporter(statusReporter, SNAP_STATS, statistic);
564             throw ex;
565         }
566     }

908     private Map<String, String> getInsertValuesMap(final PropertyValues propertyValues) {
909         List<Map<String, Object>> insertValuesTable = propertyValues.get(INSERT_VALUES_PATH);
910         Map<String, String> insertValuesMap = Maps.newLinkedHashMap();
911         if (insertValuesTable != null) {
912             for (Map<String, Object> insertValuesRow : insertValuesTable) {
913                 ExpressionProperty columnNameExpr = propertyValues
914                         .getExpressionPropertyFor(insertValuesRow, INSERT_COLUMN_PROP);
915                 String columnName = columnNameExpr.eval(null);
916                 ExpressionProperty columnValueExpr = propertyValues
917                         .getExpressionPropertyFor(insertValuesRow, INSERT_VALUE_PROP);
918                 String columnValue = columnValueExpr.eval(null);
919                 if (StringUtils.isNotBlank(columnName) && StringUtils.isNotBlank(columnValue)) {
920                     columnValue = doTargetTableNameManagement(columnValue);
921                     columnValue = transformSql(propertyValues, columnValue);
922                     if (isRedshiftDb()) {
923                         // Redshift does not support MERGE INTO natively. Therefore
924                         // MERGE INTO operation is performed using equivalent
925                         // UPDATE, INSERT and DELETE SQL statements. While doing
926                         // INSERT operations, the user is allowed to use SQL
927                         // expressions in the Insert Expression List. However
928                         // if the SQL expression happens to be a literal constant
929                         // or a complex SQL expression involving various arithmetic,
930                         // string operations etc., the SQL expression will be
931                         // associated with a default column name by Redshift. If there
932                         // are multiple SQL expressions, all those will be assigned
933                         // the same column name due to which column name collisions
934                         // occur and non-matching rows table will have duplicate
935                         // column names that can cause the create table operations to
936                         // fail. Hence in the case of Redshift, we blindly assign
937                         // new column names to avoid the collisions and store them
938                         // in a different map to be later used in insert SQL operation.
939                         String newColumnValue = getDerivedTableName();
940                         columnValue = String.format(RENAME_SQL_EXPRESSION,
941                                       columnValue, newColumnValue);
942                         sqlExpressionsRenamedMap.put(columnName, newColumnValue);
943                     }
944                     insertValuesMap.put(columnName, columnValue);
945                 }
946             }
947         }
948         return insertValuesMap;
949     }

1019                    String.format(TABLE_NAME_DOT_COLUMN_NAME, inputSqlAlias, srcColumnsArray[i]));

1022         }

1028                     .withResolution(RES_MERGE_INTO_ALL_TARGET_COLUMNS_ABSENT);

1033                     .withResolution(RES_MERGE_INTO_ALL_SOURCE_COLUMNS_ABSENT);

1121     /**
1122      * Validates merge clauses for correctness
1123      * ie.
1124      * 1. There can be a max of 2 'when matched' and 1 'when not matched' clauses
1125      * 2. 'When matched' can only be used in conjunction with Update and Delete
1126      * 3. 'When not matched' can only be used in conjunction with Insert
1127      * 4. There can only be 1 INSERT, 1 UPDATE and/or 1 DELETE in the entire merge statement
1128      */
1129     private void validateMergeActionClauses() {
1130         int whenMatchedCount = 0;
1131         int whenNotMatchedCount = 0;
1132         int whenNotMatchedBySourceCount = 0;
1133         String firstWhenMatchedCondition = null;
1134         // Error if merge clauses are not configured
1135         if (mergeActionClauseList.isEmpty()) {
1136             throw new ConfigurationException(ERR_TABLE_CONFIG)
1137                     .withReason(REASON_MERGE_ACTION_CLAUSES_NOT_FOUND)
1138                     .withResolution(REASON_MERGE_ACTION_CLAUSES_NOT_FOUND);
1139         }
1140 
1141         if (mergeActionClauseList.size() > maxNumOfWhenClauses) {
1142             throw new ConfigurationException(ERR_TABLE_CONFIG)
1143                     .withReason(REASON_TOO_MANY_WHEN_CLAUSES)
1144                     .withResolution(String.format(RES_TOO_MANY_WHEN_CLAUSES,
1145                             maxNumOfWhenClauses));
1146         }
1147 
1148         for (MergeClauseObject mergeClause : mergeActionClauseList) {
1149             switch (mergeClause.whenClause) {
1150                 case WHEN_MATCHED:
1151                     whenMatchedCount++;
1152                     if (mergeClause.action.equalsIgnoreCase(INSERT_ACTION)) {
1153                         throw new ConfigurationException(ERR_TABLE_CONFIG)
1154                                 .withReason(REASON_WRONG_WHEN_MATCHED_CLAUSE)
1155                                 .withResolution(RES_WRONG_WHEN_MATCHED_CLAUSE);
1156                     }
1157                     // If more than 1 WHEN MATCHED clauses are present, the first one should
1158                     // have an additional condition.
1159                     if (whenMatchedCount == 1) {
1160                         firstWhenMatchedCondition = mergeClause.condition;
1161                     } else if (whenMatchedCount == 2) {
1162                         if (StringUtils.isBlank(firstWhenMatchedCondition)) {
1163                             throw new ConfigurationException(ERR_TABLE_CONFIG)
1164                                     .withReason(REASON_MULTIPLE_WHEN_MATCHED_NO_CONDITION)
1165                                     .withResolution(RES_MULTIPLE_WHEN_MATCHED_NO_CONDITION);
1166                         }
1167                     }
1168                     break;
1169                 case WHEN_NOT_MATCHED:
1170                     whenNotMatchedCount++;
1171                     if (!mergeClause.action.equalsIgnoreCase(INSERT_ACTION)) {
1172                         throw new ConfigurationException(ERR_TABLE_CONFIG)
1173                                 .withReason(REASON_WRONG_WHEN_NOT_MATCHED_CLAUSE)
1174                                 .withResolution(RES_WRONG_WHEN_NOT_MATCHED_CLAUSE);
1175                     }
1176                     if (isAzureSynapseDb()) {
1177                         handleAzureSynapseDbCase();
1178                     }
1179                     break;
1180                 case WHEN_NOT_MATCHED_BY_SOURCE:
1181                     whenNotMatchedBySourceCount++;
1182                     if (!isExtendedMergeInto()) {
1183                         throw new ConfigurationException(ERR_TABLE_CONFIG)
1184                                 .withReason(REASON_WRONG_WHEN_NOT_MATCHED_BY_SOURCE_CLAUSE)
1185                                 .withResolution(
1186                                         RES_WHEN_NOT_MATCHED_BY_SOURCE_CLAUSE_NOT_SUPPORTED);
1187                     }
1188                     if (mergeClause.action.equalsIgnoreCase(INSERT_ACTION)) {
1189                         throw new ConfigurationException(ERR_TABLE_CONFIG)
1190                                 .withReason(REASON_WRONG_WHEN_NOT_MATCHED_BY_SOURCE_CLAUSE)
1191                                 .withResolution(RES_WRONG_WHEN_NOT_MATCHED_BY_SOURCE_CLAUSE);
1192                     }
1193                     break;
1194             }
1195 
1196             switch (mergeClause.action) {
1197                 case INSERT_ACTION:
1198                     insertCount++;
1199                     break;
1200                 case UPDATE_ACTION:
1201                     updateCount++;
1202                     break;
1203                 case DELETE_ACTION:
1204                     deleteCount++;
1205                     break;
1206             }
1207         }
1208 
1209         // Check if valid number of when clauses are used
1210         if (whenMatchedCount > maxNumOfWhenMatchedClauses ||
1211                 whenNotMatchedCount > maxNumOfWhenNotMatchedClauses ||
1212                 whenNotMatchedBySourceCount > maxNumOfWhenNotMatchedBySourceClauses) {
1213             if (isExtendedMergeInto()) {
1214                 throw new ConfigurationException(ERR_TABLE_CONFIG)
1215                         .withReason(REASON_WRONG_USAGE_OF_WHEN_CLAUSES)
1216                         .withResolution(String.format(RES_WRONG_USAGE_OF_EXTENDED_WHEN_CLAUSES,
1217                                 maxNumOfWhenMatchedClauses,
1218                                 maxNumOfWhenNotMatchedClauses,
1219                                 maxNumOfWhenNotMatchedBySourceClauses));
1220             } else {
1221                 throw new ConfigurationException(ERR_TABLE_CONFIG)
1222                         .withReason(REASON_WRONG_USAGE_OF_WHEN_CLAUSES)
1223                         .withResolution(String.format(RES_WRONG_USAGE_OF_WHEN_CLAUSES,
1224                                 maxNumOfWhenMatchedClauses,
1225                                 maxNumOfWhenNotMatchedClauses));
1226             }
1227         }
1228 
1229         totalStmtCount = insertCount + updateCount + deleteCount;
1230         // Check if valid number of INSERT/UPDATE/DELETE clauses are used
1231         if (insertCount > maxNumOfInsertStmts || updateCount > maxNumOfUpdateStmts ||
1232                 deleteCount > maxNumOfDeleteStmts) {
1233             if (isExtendedMergeInto()) {
1234                 throw new ConfigurationException(ERR_TABLE_CONFIG)
1235                         .withReason(REASON_WRONG_USAGE_OF_MERGE_ACTIONS)
1236                         .withResolution(RES_WRONG_USAGE_OF_EXTENDED_MERGE_ACTIONS);
1237             } else {
1238                 throw new ConfigurationException(ERR_TABLE_CONFIG)
1239                         .withReason(REASON_WRONG_USAGE_OF_MERGE_ACTIONS)
1240                         .withResolution(RES_WRONG_USAGE_OF_MERGE_ACTIONS);
1241             }
1242         }
1243     }

1352     private String getOrderedMergeActionClausesForDLP(List<String> whenClauseExprList) {
1353         /* DLP expects the WHEN MATCHED and WHEN NOT MATCHED [BY TARGET] and WHEN NOT MATCHED BY
1354         SOURCE clauses in a certain order as stated in the Databricks Merge Into documentation
1355         ie.
1356         WHEN MATCHED [ AND matched_condition ] THEN matched_action |
1357         WHEN NOT MATCHED [BY TARGET] [ AND not_matched_condition ] THEN not_matched_action |
1358         WHEN NOT MATCHED BY SOURCE [ AND not_matched_by_source_condition ]
1359                                                     THEN not_matched_by_source_action } [...]
1360          */
1361         StringBuilder whenActionClausesStr = new StringBuilder();
1362         List<String> whenMatched = new ArrayList<>();
1363         List<String> whenNotMatched = new ArrayList<>();
1364         List<String> whenNotMatchedBySourceUpdate = new ArrayList<>();
1365         List<String> whenNotMatchedBySourceDelete = new ArrayList<>();
1366 
1367         for (String whenClauseExpr : whenClauseExprList) {
1368             if (StringUtils.startsWithIgnoreCase(whenClauseExpr, WHEN_MATCHED)) {
1369                 whenMatched.add(whenClauseExpr);
1370             } else if (StringUtils.startsWithIgnoreCase(whenClauseExpr,
1371                     WHEN_NOT_MATCHED_BY_SOURCE)) {
1372                 if (StringUtils.endsWith(whenClauseExpr, "THEN " + DELETE_ACTION))
1373                 {
1374                     whenNotMatchedBySourceDelete.add(whenClauseExpr);
1375                 } else {
1376                     whenNotMatchedBySourceUpdate.add(whenClauseExpr);
1377                 }
1378             } else {
1379                 // WHEN NOT MATCHED [optional BY TARGET] clauses go here
1380                 whenNotMatched.add(whenClauseExpr);
1381             }
1382         }
1383 
1384 
1385         if (!whenMatched.isEmpty()) {
1386             whenActionClausesStr.append(String.join(StringUtils.SPACE, whenMatched));
1387         }
1388         if (!whenNotMatched.isEmpty()) {
1389             whenActionClausesStr.append(StringUtils.SPACE);
1390             whenActionClausesStr.append(String.join(StringUtils.SPACE, whenNotMatched));
1391         }
1392         if (!whenNotMatchedBySourceDelete.isEmpty()) {
1393             whenActionClausesStr.append(StringUtils.SPACE);
1394             whenActionClausesStr.append(String.join(StringUtils.SPACE,
1395                     whenNotMatchedBySourceDelete));
1396         }
1397         if (!whenNotMatchedBySourceUpdate.isEmpty()) {
1398             // if the option to enable update all columns by source operation is selected, there
1399             // is no point adding the WHEN NOT MATCHED BY SOURCE UPDATE action clauses, because
1400             // there are no source rows to update the target. By including the 'when not matched by
1401             // source' clause, the efficiency of the query is reduced.
1402 
1403             // Another scenario is where the only clauses configured are the when not matched by
1404             // source update clauses, in which case, we append them to the merge into sql.
1405             if (!isMergeIntoUpdateAllBySource() || !extendedMergeIntoFlag
1406                     || whenActionClausesStr.length() == 0) {
1407                 // append space only there is already some content in StringBuilder
1408                 if (whenActionClausesStr.length() != 0) {
1409                     whenActionClausesStr.append(StringUtils.SPACE);
1410                 }
1411                 whenActionClausesStr.append(String.join(StringUtils.SPACE,
1412                         whenNotMatchedBySourceUpdate));
1413             } else {
1414                 OPTIMIZING_QUERY.report();
1415             }
1416         }
1417         return whenActionClausesStr.toString();
1418     }
