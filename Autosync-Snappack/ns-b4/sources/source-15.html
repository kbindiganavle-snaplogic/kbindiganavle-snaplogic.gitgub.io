444     private Map<String, Object> getMathOutput(String inputSqlString, Document document,
445                                               List<String> snapTypes) {
446         List<String> previousSnapTypes = extractSnapTypeFromInputDocument(document);
447         Map<String, Object> output;
448         String randomString = "Math_".concat(getDerivedTableName());
449         if (canPushdownFunctions(document, MATH_PUSHDOWN_SNAP_LIST, specifiedFunctions,
450                 inputSqlString)) {
451             snapTypes.addAll(previousSnapTypes);
452             output = buildMathPushDownOutput(document, snapTypes, randomString, inputSqlString);
453         } else {
454             output = buildMathOutput(inputSqlString, snapTypes, document, randomString);
455         }
456         return output;
457     }

459     private Map<String, Object> buildMathPushDownOutput(Document document,
460                                                         List<String> snapTypes,
461                                                         String randomString,
462                                                         String inputSqlString) {
463         String resultSql = StringUtils.EMPTY;
464         PushdownContext ctx = PushdownContext.getOrCreate(document, randomString);
465         if (!specifiedFunctions.isEmpty()) {
466             List<Field> allFunctionFieldsWithOriginalNames = buildAllFields(specifiedFunctions);
467             checkColumnsExists(document, allFunctionFieldsWithOriginalNames, inputSqlString);
468 
469             replaceArgumentsIfRenamed(ctx, specifiedFunctions);
470             List<Field> allFunctionFields = buildAllFields(specifiedFunctions);
471             SelectPart selectPart = ctx.getSelectPart();
472             if (passThroughMath) {
473                 List<Field> expressions = new ArrayList<>(selectPart.getExpressions());
474                 expressions.addAll(0, allFunctionFields);
475                 selectPart.setExpressions(expressions);
476             } else {
477                 selectPart.setExpressions(new ArrayList<>(allFunctionFields));
478             }
479             resultSql = pushdownQueryBuilder.buildSql(ctx);
480         } else if (passThroughMath) {
481             resultSql = pushdownQueryBuilder.buildSql(ctx);
482         } else {
483             throwInvalidConfig();
484         }
485         return getOutputViewMap(document, resultSql, snapTypes, ctx);
486     }

517                 // the Math functions snap can simply forward the input SQL without

522                         PASS_THROUGH_ONLY_SQL_FUNCTIONS, inputSqlString, randomString);
