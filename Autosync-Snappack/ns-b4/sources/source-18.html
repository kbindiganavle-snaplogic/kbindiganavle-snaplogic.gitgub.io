282     protected void processDocument(final Document document, final String inputViewName) {
283         String inputSqlString = extractQueryFromInputDocument(document);
284         if (StringUtils.isBlank(inputSqlString)) {
285             return;
286         }
287         List<Object> allBindingValues = extractBindingValues(document);
288         if (validationInProgress) {
289             schemaSupportService.defineInputSchema(schemaProvider, inputSqlString, inputViewName);
290         }
291         createTempSourceTableFromInputSql(inputSqlString, allBindingValues);
292         String dtName = "Pivot_" + getDerivedTableName();
293         String sqlPivot = EMPTY_STRING;
294         Set<String> columns = new LinkedHashSet<>();
295         if (isDynamicPivotValues || isRedshiftDb() || isAzureSynapseDb()) {
296             String checkTableColumnQuery = buildFetchColumnsSqlString(inputSqlString);
297             columns = findTableColumns(document, checkTableColumnQuery);
298             if (!(columns.remove(resolveColumn(pivotColumn)) &&
299                     columns.remove(resolveColumn(pivotValueColumn)))) {
300                 throw new SnapDataException(ERR_PIVOT_MISSING_TABLE_COLUMN)
301                         .withReason(ERR_PIVOT_MISSING_TABLE_COLUMN)
302                         .withResolution(RESOLUTION_PIVOT_MISSING_TABLE_COLUMN);
303             }
304         }
305         boolean useAliases = false;
306         List<String> valuesRead = new ArrayList<>();
307         if (isDynamicPivotValues) {
308             valuesRead = readPivotValues(dtName, allBindingValues);
309             pivotValues.addAll(valuesRead);
310             useAliases = valuesRead.contains(NULL_COLUMN) ||
311                     valuesRead.contains(PIVOT_VALUE_QUOTE_FORMAT_EMPTY);
312         }
313 
314         switch (getName()) {
315             // Native Pivot provided by following CDWs
316             case JdbcOperations.SNOWFLAKE:
317                 sqlPivot = buildSnowflakePivotSql(dtName, columns, useAliases, valuesRead);
318                 break;
319             case JdbcOperations.DATABRICKS_LAKEHOUSE_PLATFORM:
320                 if (isDynamicPivotValues && COUNT.equalsIgnoreCase(pivotAggregateFunction)) {
321                     sqlPivot = buildRewrittenPivotSql(dtName, columns);
322                 } else {
323                     sqlPivot = buildBqDlpPivotSql(dtName, useAliases);
324                 }
325                 break;
326             case JdbcOperations.BIGQUERY:
327                 sqlPivot = buildBqDlpPivotSql(dtName, useAliases);
328                 break;
329             case JdbcOperations.REDSHIFT:
330             case JdbcOperations.AZURE_SYNAPSE:
331                 sqlPivot = buildRewrittenPivotSql(dtName, columns);
332                 break;
333             default:
334                 EltUtils.throwInvalidDbTypeException();
335         }
336         sqlPivot = addUniqueQueryID(sqlPivot);
337         if (tempSourceTableCreated) {
338             newSqlsToExecuteOnCleanUp.add(buildDropTableSql(tempSourceTableName));
339         }
340         Map<String, Object> output = getOutputViewMap(document, sqlPivot,
341                 Collections.singletonList(ELT_PIVOT_TITLE));
342         if (handlePreview(documentUtility.newDocument(output), document, account,
343                 validationInProgress, previewRowCount, executionMode, executeSqlInPreview)) {
344             return;
345         }
346         outputViews.write(documentUtility.newDocument(output), document);
347     }

349     private String buildRewrittenPivotSql(String dtName, Set<String> columns) {
350         // Pivot not supported natively by following CDWs, and hence Query rewrites are used
351         String sqlPivot;
352         //form the PIVOT query for Redshift database
353         StringBuilder columnConstruction = new StringBuilder(2048);
354         StringBuilder groupByConstruction = new StringBuilder(50);
355         for (String column : columns) {
356             columnConstruction.append(column).append(COMMA);
357             groupByConstruction.append(column).append(COMMA);
358         }
359         for (int i = 0; i < pivotValues.size(); i++) {
360             final String colName = pivotValues.get(i);
361             String colNameAlias = getColumnAlias(colName);
362             String equalityOperator = NULL_COLUMN.equals(colName) ? "IS" : "=";
363             columnConstruction.append(String.format(PIVOT_COLUMN_CONSTRUCTION,
364                     pivotAggregateFunction, pivotValueColumn, equalityOperator, colName,
365                     pivotColumn, colNameAlias)).append(COMMA);
366         }
367         //Remove the last comma from columnConstruction
368         String finalColumnConstruction =
369                 columnConstruction.substring(0, columnConstruction.length() - 1);
370         if (StringUtils.isEmpty(groupByConstruction)) {
371             sqlPivot = String.format(PIVOT_REDSHIFT_AZURE_SYNAPSE_SQL,
372                     finalColumnConstruction, tempSourceTableName, dtName);
373         } else {
374             //Remove the last comma from groupByConstruction
375             String finalGroupByConstruction = groupByConstruction.substring(0,
376                     groupByConstruction.length() - 1);
377             sqlPivot = String.format(PIVOT_REDSHIFT_AZURE_SYNAPSE_GROUP_BY_SQL,
378                     finalColumnConstruction, tempSourceTableName, dtName,
379                     finalGroupByConstruction);
380         }
381         return sqlPivot;
382     }

385         String colNameAlias;

399     private String buildBqDlpPivotSql(String dtName, boolean useAliases) {
400         String sqlPivot;
401         final String valuesString = useAliases ?
402                 pivotValues.stream().map(val -> {
403                     if (NULL_COLUMN.equals(val)) {
404                         return String.format(ALIAS_TEMPLATE, val, NULL_COLUMN_ALIAS);
405                     }
406                     if (PIVOT_VALUE_QUOTE_FORMAT_EMPTY.equals(val)) {
407                         return String.format(ALIAS_TEMPLATE, val, EMPTY_COLUMN_ALIAS);
408                     }
409                     return val;
410                 }).collect(Collectors.joining(", "))
411                 : StringUtils.join(pivotValues, COMMA);
412 
413         sqlPivot = String.format(PIVOT_SQL, tempSourceTableName, dtName,
414                 pivotAggregateFunction, pivotColumn, pivotValueColumn, valuesString);
415         return sqlPivot;
416     }

418     private String buildSnowflakePivotSql(String dtName, Set<String> columns, boolean useAliases,
419                              List<String> valuesRead) {
420         String sqlPivot = String.format(PIVOT_SQL, tempSourceTableName, dtName,
421                 pivotAggregateFunction, pivotColumn, pivotValueColumn,
422                 StringUtils.join(pivotValues, COMMA));
423         if (useAliases) {
424             List<String> pivotAliases = new ArrayList<>();
425             pivotAliases.addAll(columns);
426             pivotAliases.addAll(valuesRead);
427             Collections.replaceAll(pivotAliases, NULL_COLUMN, NULL_COLUMN_ALIAS);
428             Collections.replaceAll(pivotAliases, PIVOT_VALUE_QUOTE_FORMAT_EMPTY,
429                     EMPTY_COLUMN_ALIAS);
430             pivotAliases =
431                     pivotAliases.stream().map(value -> StringUtils.strip(value, "'"))
432                             .collect(Collectors.toList());
433             sqlPivot = String.format("%s AS %s (%s)", sqlPivot, getDerivedTableName(),
434                     StringUtils.join(pivotAliases, COMMA));
435         }
436         return sqlPivot;
437     }

439     protected List<String> readPivotValues(String randString, List<Object> allBindingValues) {
440         String fromClause = tempSourceTableCreated ? tempSourceTableName :
441                 tempSourceTableName.concat(" ").concat(randString);
442         final String pivotValueColumnNormalized = resolveColumn(this.pivotValueColumn);
443         List<String> valuesRead = new ArrayList<>();
444 
445         String sqlQuery;
446         if (isAzureSynapseDb() && StringUtils.isNotBlank(dynamicValuesLimit)) {
447             sqlQuery = String.format("SELECT DISTINCT TOP %s %s FROM %s ",
448                     dynamicValuesLimit, pivotValueColumnNormalized, fromClause);
449         } else {
450             sqlQuery = String.format("SELECT DISTINCT %s FROM %s ", pivotValueColumnNormalized,
451                     fromClause);
452         }
453 
454         if (StringUtils.isNotBlank(filterClauseList)) {
455             sqlQuery = String.format("%s WHERE %s ", sqlQuery, filterClauseList);
456         }
457 
458         if (StringUtils.isNotBlank(sortOrder) && !NONE.equals(sortOrder)) {
459             String orderValue = String.format("%s %s", pivotValueColumnNormalized, sortOrder);
460             sqlQuery = String.format("%s ORDER BY %s ", sqlQuery, orderValue);
461         }
462 
463         if (!isAzureSynapseDb() && StringUtils.isNotBlank(dynamicValuesLimit)) {
464             sqlQuery = String.format("%s LIMIT %s ", sqlQuery, dynamicValuesLimit);
465         }
466 
467         try (DatabaseCursor cursor = executeSqlStmt(sqlQuery, allBindingValues, account)) {
468             do {
469                 Map<String, Object> dbData = cursor.fetchOne();
470                 String value = String.valueOf(dbData.get(pivotValueColumnNormalized));
471                 valuesRead.add(NULL_COLUMN.equals(value) ? NULL_COLUMN :
472                         String.format(PIVOT_VALUE_QUOTE_FORMAT, value));
473             } while (cursor.hasNext());
474 
475             if (valuesRead.isEmpty()) {
476                 throw new SnapDataException(ERR_PIVOT_VALUES)
477                         .withReason(ERROR_PIVOT_VALUES_RETRIEVAL)
478                         .withResolution(RESOLUTION_PIVOT_VALUES);
479             }
480 
481         } catch (Exception ex) {
482             throw new SnapDataException(ex, ERR_PIVOT_VALUES)
483                     .withReason(ERROR_PIVOT_VALUES_RETRIEVAL)
484                     .withResolution(RESOLUTION_PIVOT_VALUES);
485         }
486         return valuesRead;
487     }

491 
492         if (validationInProgress) {

496         if (isAzureSynapseDb()) {

499         try {

501             if (!getName().equals(JdbcOperations.DATABRICKS_LAKEHOUSE_PLATFORM)) {
502                 schemaName = extractSchemaNameFromSql(inputSqlString);
503                 if (StringUtils.isBlank(schemaName)) {
