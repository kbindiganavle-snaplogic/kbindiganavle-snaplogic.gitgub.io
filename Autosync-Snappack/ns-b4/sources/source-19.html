170     protected void configureSnap(final PropertyValues propertyValues)
171             throws ConfigurationException {
172         super.configureSnap(propertyValues);
173         EltUtils.checkSnapPlacement(inputViews, ELT_ROUTER_TITLE);
174 
175         // Build an output view label to output view name map.
176         Set<String> outputViewNames = outputViews.names();
177         for (String outputViewName : outputViewNames) {
178             String outputViewLabelPath = new JsonPathBuilder(outputViewName)
179                     .appendCurrentElement(OUTPUT_VIEW_LABEL)
180                     .build();
181             String outputViewLabel = propertyValues.get(PropertyCategory.OUTPUT,
182                     outputViewLabelPath);
183             outputLabelToKeyMapping.put(outputViewLabel, outputViewName);
184         }
185         // Build a property map list
186         List<Map<String, String>> routeMapList =
187                 propertyValues.get(EXPRESSION_TO_OUTPUT_VIEW_MAP_PATH);
188         // if not empty, build a runtime map, "routingMap".
189         if (CollectionUtils.isNotEmpty(routeMapList)) {
190             routingMap = Maps.newLinkedHashMap();
191             for (Map<String, String> route : routeMapList) {
192                 String key = propertyValues
193                         .getExpressionPropertyFor(route, ROUTER_EXPRESSION_PROP)
194                         .eval(null);
195                 if (StringUtils.isBlank(key)) {
196                     throw new ConfigurationException(ERR_INVALID_EXPRESSION_DETECTED)
197                             .withReason(REASON_EMPTY_EXPRESSION_DETECTED)
198                             .withResolution(RESOLUTION_EMPTY_EXPRESSION_DETECTED);
199                 }
200                 String value = propertyValues
201                         .getExpressionPropertyFor(route, OUTPUT_VIEW_NAME_PROP)
202                         .eval(null);
203                 if (StringUtils.isBlank(value)) {
204                     continue;
205                 }
206                 key = transformSql(propertyValues, key);
207                 // Validate the values() set here.
208                 if (routingMap.containsValue(value)) {
209                     throw new ConfigurationException(ERR_INVALID_OUTPUT_VIEW_NAME_DETECTED)
210                             .withReason(String.format(
211                                     REASON_DUPLICATE_OUTPUT_VIEW_NAME_DETECTED, value))
212                             .withResolution(RESOLUTION_DUPLICATE_OUTPUT_VIEW_NAME_DETECTED);
213                 }
214                 routingMap.put(key, value);
215             }
216         }
217         // if the routingMap is empty, set true to 'copyToAllWithNoFilter'.
218         if (MapUtils.isEmpty(routingMap)) {
219             copyToAllWithNoFilter = true;
220         }
221 
222         evaluateExpressionBeforeForwarding =
223                 Boolean.TRUE.equals(propertyValues.get(EVALUATE_EXPRESSION_BEFORE_FORWARDING_PROP));
224     }

237      * Process the given "document" per description of DATA-3517:
238      * https://mysnaplogic.atlassian.net/browse/DATA-3517. Note that
239      * 'appendExpressionToOutputViewSql' is the variable for
240      * "Append the Filter Condition to the input SQL" in the description
241      * and 'evaluateExpressionBeforeForwarding' is the one for
242      * "Evaluate the Filter Condition before forwarding".
243      *
244      * Please note that Case no 1 to 5 as the case numbers are mentioned at
245      * the comments of this method.
246      *
247      * @param document      Document received on input.
248      * @param inputViewName Name of the input view that received the data.
249      */
250     @Override
251     protected void processDocument(final Document document, final String inputViewName) {
252         String inputSqlString = extractQueryFromInputDocument(document);
253 
254         if (StringUtils.isBlank(inputSqlString)) {
255             return;
256         }
257         if (validationInProgress) {
258             schemaSupportService.defineInputSchema(schemaProvider, inputSqlString, inputViewName);
259         }
260 
261         // If evaluateExpressionBeforeForwarding is true and copyToAllWithNoFilter not true,
262         // find "viewToWriteToForRouter" to be used for handlePreview() and this function.
263         if (evaluateExpressionBeforeForwarding && !copyToAllWithNoFilter) {
264             viewsToWriteToForRouter = findViewsToWriteTo(inputSqlString);
265 
266             // Here we should have at least one expression with true as result.
267             // If not, we throw an exception.
268             if (viewsToWriteToForRouter.isEmpty()) {
269                 throw new ConfigurationException(ERR_INVALID_EXPRESSIONS_DETECTED)
270                         .withReason(REASON_ALL_THE_EXPRESSIONS_EVALUATED_TO_FALSE)
271                         .withResolution(RESOLUTION_ALL_THE_EXPRESSIONS_EVALUATED_TO_FALSE);
272             }
273         }
274 
275         // If-then block is executed when copyToAllWithNoFilter
276         // is false. We append filter condition to the input sql string of the output views.
277         // This is for Case no 1 and 2 that are described at the
278         // Description section of https://mysnaplogic.atlassian.net/browse/DATA-3517.
279         if (!copyToAllWithNoFilter) {
280             Map<String, Object> output = null;
281 
282             int i = 0;
283             for (Map.Entry<String, String> route : routingMap.entrySet()) {
284                 i++;
285                 String expression = route.getKey();
286                 outputViewNameToWriteTo = outputLabelToKeyMapping.get(route.getValue());
287                 if (viewsToWriteToForRouter != null) {
288                     // 'evaluateExpressionBeforeForwarding'=true case
289                     if (!viewsToWriteToForRouter
290                             .contains(outputViewNameToWriteTo)) {
291                         outputViewNameToWriteTo = null;
292                         // Skip if the output view is not found from "viewsToWriteToForRouter".
293                         continue;
294                     }
295                 }
296                 output = buildOutputForFilterCondition(
297                         document, inputSqlString, expression.trim());
298                 if (handlePreview(documentUtility.newDocument(output), document, account,
299                         validationInProgress, previewRowCount, executionMode,
300                         executeSqlInPreview)) {
301                     outputViewNameToWriteTo = null;
302                     if (i < routingMap.size()) {
303                         continue;
304                     } else {
305                         return;
306                     }
307                 }
308 
309                 // Write to the output view of the current "outputViewNameToWriteTo".
310                 writeToGivenOutputView(outputViewNameToWriteTo,
311                         documentUtility.newDocument(output), document);
312             }
313         } else { // copyToAllWithNoFilter case
314             // We pass "document" as the first parameter as
315             // we have no filter and thus we pass the input document through.
316             if (handlePreview(document, document, account,
317                     validationInProgress, previewRowCount, executionMode, executeSqlInPreview)) {
318                 return;
319             }
320 
321             // We have true for copyToAllWithNoFilter, thus copy to all the output views.
322             outputViews.write(document);
323         }
324     }

326     private List<String> findViewsToWriteTo(String inputSqlString) {
327         List<String> viewListToWriteTo = new ArrayList<>();
328         for (Map.Entry<String, String> route : routingMap.entrySet()) {
329             String expression = route.getKey();
330             if (evaluateConditionalSqlExpression(inputSqlString, expression)) {
331                 viewListToWriteTo.add(outputLabelToKeyMapping.get(route.getValue()));
332             }
333         }
334         return viewListToWriteTo;
335     }

338      * Build and return a Filter Snap output for the
339      * given "filterCondition".
340      */
341     private Map<String, Object> buildOutputForFilterCondition(
342             final Document document, String inputSqlString, String filterCondition) {
343         List<String> snapTypes = new ArrayList<>();
344         snapTypes.add(ELT_FILTER_TITLE);
345         String dtName = "Filter_" + getDerivedTableName();
346         String sqlFilter = String.format(FILTER_SQL, inputSqlString, dtName, filterCondition);
347 
348         List<String> previousSnapTypes = extractSnapTypeFromInputDocument(document);
349 
350         PushdownContext ctx;
351         if (canPushdown(document, FILTER_PUSHDOWN_SNAP_LIST)) {
352             snapTypes.addAll(previousSnapTypes);
353             ctx = PushdownContext.getOrCreate(document, dtName);
354             checkFilterSQL(document, sqlFilter);
355             pushdownFilterCondition(ctx, filterCondition);
356 
357             sqlFilter = pushdownQueryBuilder.buildSql(ctx);
358         } else {
359             ctx = PushdownContext.create(document, dtName);
360             pushdownFilterCondition(ctx, filterCondition);
361         }
362 
363         return getOutputViewMap(document, sqlFilter, snapTypes, ctx);
364     }

384      * Execute the given "inputSqlString" with the given boolean "expression".
385      * I.e., 'SELECT 1 FROM inputSqlString dtName WHERE expression LIMIT 1'.
386      * Then returns true if the the SELECT SQL returns a result.
387      *
388      * Note that, if Azure Synapse, the SQL for the example will be
389      * 'SELECT TOP 1 1 FROM inputSqlString dtName WHERE expression'.
390      */
391     @VisibleForTesting
392     public boolean evaluateConditionalSqlExpression(String inputSqlString, String expression) {
393         dbConnection = jdbcOperations.acquireConnection(account);
394         ResultSet rs = null;
395 
396         selectWithConditionalSqlExpression =
397                 buildConditionalSqlQueryWithExpression(inputSqlString, expression);
398 
399         try (PreparedStatement stmt = dbConnection.prepareStatement(
400                 selectWithConditionalSqlExpression)) {
401             rs = stmt.executeQuery();
402             if (rs.next()) {
403                 return true;
404             } else {
405                 return false;
406             }
407         } catch (SQLException e) {
408             throw new ConfigurationException(ERR_MSG_INVALID_SQL_QUERY)
409                     .formatWith(selectWithConditionalSqlExpression)
410                     .withReason(e.getMessage())
411                     .withResolution(RESOLUTION_INVALID_ROUTER_FILTER_LIMIT_ONE_SQL);
412         } finally {
413             if (rs != null) {
414                 try {
415                     rs.close();
416                 } catch (SQLException e) {}
417             }
418         }
419     }
