60 /**
61  * Snap that transforms incoming data using expressions. The snap takes in a two column table (an
62  * expression column and a target column). The snap will evaluate the expression and map it to
63  * the target path.
64  *
65  * Performs transformation on a SELECT list expressions from the input SQL.
66  * Conceptually it works like this:
67  *
68  * SELECT <sql_expr1> AS <trans_sql_expr1>, <sql_expr2> AS <trans_sql_exp2>, ...
69  * FROM inputSQL
70  * WHERE ...
71  *
72  * @author Pradeep Sathyanarayan
73  */

165     @Override
166     protected void processDocument(Document document, final String inputViewName) {
167         String inputSqlString = extractQueryFromInputDocument(document);
168         if (StringUtils.isBlank(inputSqlString)) {
169             return;
170         }
171         String dtName = "Transform_" + getDerivedTableName();
172 
173         if (validationInProgress) {
174             schemaSupportService.defineInputSchema(schemaProvider, inputSqlString, inputViewName);
175         }
176 
177         List<String> snapTypes = new ArrayList<>();
178         snapTypes.add(ELT_TRANSFORM_TITLE);
179 
180         String sqlTransform;
181 
182         PushdownContext pushdownContext;
183 
184         if (canPushdown(document, TRANSFORM_PUSHDOWN_SNAP_LIST)
185                 && !hasNestedFunction(document, transformations.stream()
186                 .map(Field::getExpr), inputSqlString)
187         ) {
188             pushdownContext = PushdownContext.getOrCreate(document, dtName);
189 
190             sqlTransform = doTransformRewrite(document, inputSqlString,
191                     snapTypes, pushdownContext);
192         } else {
193             String sqlExprList = buildSqlExprList(document, inputSqlString);
194             sqlTransform = addUniqueQueryID(
195                     String.format(TRANSFORM_SQL, sqlExprList, inputSqlString, dtName));
196             pushdownContext = PushdownContext.create(document, dtName);
197             populatePushdownContext(document, inputSqlString, pushdownContext);
198         }
199 
200         List<Map<String, Object>> outputSchema = null;
201         if (validationInProgress) {
202             outputSchema = schemaSupportService.getSchemaColumns(sqlTransform);
203         }
204 
205         Map<String, Object> output =
206                 getOutputViewMap(document, sqlTransform, snapTypes, pushdownContext, outputSchema);
207         if (handlePreview(documentUtility.newDocument(output), document, account,
208                 validationInProgress, previewRowCount, executionMode, executeSqlInPreview)) {
209             return;
210         }
211         outputViews.write(documentUtility.newDocument(output), document);
212     }

227     private void populatePushdownContext(final Document document,
228                                          final String inputSqlString,
229                                          final PushdownContext pushdownContext) {
230         SelectPart selectPart = pushdownContext.getSelectPart();
231         checkColumnsExists(document, transformations, inputSqlString);
232 
233         List<Field> prevExpressions = selectPart.getExpressions();
234 
235         // only transformations
236         if (isEmpty(columnsToDelete)) {
237             List<Field> mergedTransformations =
238                     mergeTransformations(transformations, prevExpressions);
239             selectPart.setExpressions(mergedTransformations);
240         } else {
241             List<Field> rval = mergeColumns(document, inputSqlString,
242                     prevExpressions, transformations, columnsToDelete);
243             selectPart.setExpressions(rval);
244         }

248                                       final String inputSqlString,
249                                       final List<String> snapTypes,
250                                       final PushdownContext pushdownContext) {
251         List<String> previousSnapTypes = extractSnapTypeFromInputDocument(document);
252         snapTypes.addAll(previousSnapTypes);
253 
254         if (passThroughTransform && isEmpty(transformations) && isEmpty(columnsToDelete)) {

262     private List<Field> mergeColumns(Document document,
263                                      String inputSqlString,
264                                      List<Field> prevTransformations,
265                                      List<Field> currTransformations,
266                                      Set<String> columnsToDelete) {
267 
268         Set<String> allAvailableColumns = getAllAvailableColumns(document, inputSqlString);
269 
270         List<Field> prev = new ArrayList<>(prevTransformations);
271         List<Field> curr = new ArrayList<>(currTransformations);
272 
273         for (Field f: prevTransformations) {
274             String col = f.getExpr();
275             String alias = f.getAlias();
276 
277             if (StringUtils.isBlank(alias)) {
278                 if (columnsToDelete.contains(col)) {
279                     prev.remove(f);
280                     allAvailableColumns.remove(resolveColumn(col));
281                 }
282             } else {
283                 if (columnsToDelete.contains(alias)) {
284                     prev.remove(f);
285                     allAvailableColumns.remove(resolveColumn(alias));
286                 }
287             }
288         }
289 
290         allAvailableColumns.removeAll(columnsToDelete);
291 
292         prev.stream()
293                 .filter(f -> isNotBlank(f.getAlias()))
294                 .forEach(f -> allAvailableColumns.remove(f.getAlias()));
295 
296         List<Field> mergedTransformations;
297 
298         if (isNotEmpty(currTransformations)) {
299             mergedTransformations = mergeTransformations(curr, prev);
300         } else {
301             mergedTransformations = prev;
302         }
303 
304         if (passThroughTransform && mergedTransformations.contains(ASTERISK_FIELD)) {
305             mergedTransformations.remove(ASTERISK_FIELD);
306             mergedTransformations.addAll(allAvailableColumns.stream()
307                     .map(Field::new)
308                     .collect(Collectors.toList()));
309         }
310 
311         return mergedTransformations;
312     }

314     private List<Field> mergeTransformations(List<Field> currentTransformations,
315                                              List<Field> prevExpressions) {
316         List<Field> rval = new ArrayList<>(currentTransformations);
317         List<Field> prev = new ArrayList<>(prevExpressions);
318         List<Field> mergedAliases = new ArrayList<>();
319 
320         for (Field currField: currentTransformations) {
321             String currentFieldExpr = currField.getExpr();
322             String currentFieldAlias = currField.getAlias();
323 
324             for (Field prevField: prevExpressions) {
325                 String prevFieldExpr = prevField.getExpr();
326                 String prevFieldAlias = prevField.getAlias();
327 
328                 if (equals(currentFieldExpr, prevFieldAlias)) {
329                     mergedAliases.add(new Field(prevFieldExpr, currentFieldAlias));
330                     prev.remove(prevField);
331                     rval.remove(currField);
332                 }
333             }
334         }
335 
336         rval.addAll(mergedAliases);
337 
338         if (passThroughTransform) {
339             rval.addAll(0, prev);
340         }
341 
342         return rval;
343     }

376     private String buildSqlExprList(Document document, String inputSqlString) {
377         String transformationSqlExprList;
378 
379         if (passThroughTransform && transformations.isEmpty() && columnsToDelete.isEmpty()) {
380             transformationSqlExprList = ASTERISK;
381         } else {
382             transformationSqlExprList = transformations.stream()
383                     .map(e -> String.format(TRANSFORMATION_PATTERN, e.getExpr(), e.getAlias()))
384                     .collect(Collectors.joining(TRANSFORMATION_DELIMITER));
385 
386             if (passThroughTransform) {
387                 if (columnsToDelete.isEmpty()) {
388                     transformationSqlExprList = isNotBlank(transformationSqlExprList) ?
389                             addAsterisk(transformationSqlExprList) : ASTERISK;
390                 } else {
391                     Set<String> columnSetFromSrcSql = findTableColumns(document,
392                             buildFetchColumnsSqlString(inputSqlString));
393                     columnSetFromSrcSql.removeAll(columnsToDelete);
394                     columnSetFromSrcSql.add(transformationSqlExprList);
395                     transformationSqlExprList = columnSetFromSrcSql.stream()
396                             .filter(StringUtils::isNotBlank)
397                             .collect(Collectors.joining(TRANSFORMATION_DELIMITER));
398                 }
399             }
400         }
401         return transformationSqlExprList;
402     }
