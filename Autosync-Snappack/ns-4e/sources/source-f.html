358      * Helper function that iterates over the table list.
359      *
360      * @param catalog      must match the catalog name as it
361      *                     is stored in the database; "" retrieves those without a catalog;
362      *                     <code>null</code> means that the catalog name should not be used to
363      *                     narrow
364      * @param schema       the schema to filter on
365      * @param includeGraph True if the set of dependent tables should be included in the
366      *                     returned map.
367      *
368      * @return A map of table names.
369      */
370     private Map<String, Set<String>> getTableData(Connection conn, String catalog, String schema,
371             String tableNamePattern, boolean includeGraph, String[] tableTypes) {
372         if (StringUtils.isEmpty(schema)) {
373             schema = null;
374         }
375         Map<String, Set<String>> tableData = new HashMap<>();
376         long startTime = System.currentTimeMillis();
377         try {
378             DatabaseConfig databaseConfig = account.getDatabaseConfig();
379             SQLDialect dialect = databaseConfig.getSQLDialect();
380             java.sql.DatabaseMetaData dbMetaData = conn.getMetaData();
381             if (quotationHandler instanceof GenericQuotationHandler) {
382                 ((GenericQuotationHandler) quotationHandler).setQuotationMarker(
383                         dbMetaData.getIdentifierQuoteString());
384             }
385             String productName = dbMetaData.getDatabaseProductName();
386             if (StringUtils.containsAnyIgnoreCase(productName,
387                     JdbcOperations.AWS_ATHENA_ALL_NAMES)) {
388                 // Athena is not working with the default value of {"TABLE", "VIEW"} for the
389                 //  tableTypes parameter.
390                 // It doesn't work if we just use {"TABLE"} value.
391                 // But it does work with null value
392                 // So setting this tableTypes to null for Athena
393                 tableTypes = null;
394             }
395             if (StringUtils.isNotBlank(tableNamePattern)) {
396                 String[] parts = quotationHandler.splitFullyQualifiedName(tableNamePattern, 0);
397                 if (parts.length == 2) {
398                     tableNamePattern = parts[1];
399                 }
400             }
401             schema = quotationHandler.unescape(schema);
402             tableNamePattern = formatTableNamePattern(
403                     quotationHandler.unescape(tableNamePattern),
404                     dbMetaData);
405 
406             String finalCatalog = catalog;
407             String finalSchema = schema;
408             if (dialect == SQLDialect.MYSQL || MYSQL_PRODUCT_NAME.equals(productName)) {
409                 finalCatalog = schema;
410                 finalSchema = null;
411             } else if (SPARK_PRODUCT_NAME.equals(productName)) {
412                 finalCatalog = "SPARK";
413                 finalSchema = catalog;
414             }
415 
416             try (ResultSet allTables = dbMetaData.getTables(
417                     finalCatalog,
418                     finalSchema,
419                     tableNamePattern,
420                     tableTypes)) {
421                 if (!VERTICA_PRODUCT_NAME.equals(productName)) {
422                     allTables.setFetchSize(FETCH_SIZE_OF_SUGGEST);
423                 }
424                 while (allTables.next()) {
425                     String catName = allTables.getString(TABLE_CATALOG_COL);
426                     String schemaName = allTables.getString(TABLE_SCHEMA_COL);
427                     String tableName = allTables.getString(TABLE_NAME_COL);
428                     String fqTableName = formatTableName(dialect, catName, schemaName, tableName,
429                             quotationHandler);
430                     if (SPARK_PRODUCT_NAME.equals(productName)) {
431                         // Spark does not support schema
432                         fqTableName = tableName;
433                     }
434 
435                     if (includeGraph) {
436                         addTableReference(tableData, fqTableName, null);
437                         // XXX We get an error on the Redshift server when calling
438                         // getImportedKeys.
439                         if (account instanceof RedshiftDatabaseAccount) {
440                             continue;
441                         }
442                         try (ResultSet keys = dbMetaData.getImportedKeys(
443                                 catName, schemaName, tableName)) {
444                             while (keys.next()) {
445                                 String fqOtherTableName = formatTableName(
446                                         dialect,
447                                         keys.getString(PKTABLE_CAT),
448                                         keys.getString(PKTABLE_SCHEM),
449                                         keys.getString(PKTABLE_NAME),
450                                         quotationHandler);
451                                 addTableReference(tableData, fqTableName, fqOtherTableName);
452                             }
453                         }
454                     } else {
455                         tableData.put(fqTableName, null);
456                     }
457                     if (!includeGraph && System.currentTimeMillis() - startTime >
458                             SUGGEST_TIMEOUT_IN_MS) {
459                         tableData.put(TIMEOUT_ERROR_KEY, null);
460                         break;
461                     }
462                 }
463             }
464         } catch (SQLException e) {
465             throw new ExecutionException(e, CREATE_DATABASE_METADATA_FAILED)
466                     .withReason(ERR_ACCESSING_DB_METADATA_ITERATOR)
467                     .withResolutionAsDefect();
468         }
469 
470         return tableData;
471     }

491      * @param dialect       The SQL dialect that is used.
492      * @param catName       The category name for the table.
493      * @param schemaName    The schema name for the table.
494      * @param procedureName The procedure name
495      *
496      * @return A fully-qualified name for the given table.
497      */
498     protected static String formatProcedureName(SQLDialect dialect, String catName,
499             String schemaName, String procedureName, QuotationHandler handler) {
500         if (dialect == SQLDialect.MYSQL) {
501             schemaName = catName;
502         }
503         procedureName = handler.escape(procedureName);
504         schemaName = handler.escape(schemaName);
505         catName = handler.escape(catName);
506         switch (dialect) {
507             case MYSQL:
508                 if (schemaName != null) {
509                     return String.format(TABLE_NAME_FORMAT, catName, procedureName);
510                 }
511                 return procedureName;
512             case ORACLE:
513             case ORACLE10G:
514             case ORACLE11G:
515             case ORACLE12C:
516                 StringBuilder builder = new StringBuilder();
517                 if (schemaName != null) {
518                     builder.append(schemaName).append(DOT_STRING);
519                 }
520                 if (catName != null) {
521                     // The Catalog argument is taken to mean a package name. See the description of
522                     // getProcedureColumns method in the web page:
523                     // http://download.oracle.com/otn_hosted_doc/jdeveloper/905/jdbc-javadoc/oracle
524                     // /jdbc/OracleDatabaseMetaData.html
525                     builder.append(catName).append(DOT_STRING);
526                 }
527                 return builder.append(procedureName).toString();
528             case SQLSERVER:
529             case SQLSERVER2012:
530             case SQLSERVER2008:
531             case SQLSERVER2014:
532                 // trim the double-quote at the beginning and the end of the name
533                 procedureName = CharMatcher.is('"').trimFrom(procedureName);
534                 // trim characters at and after ';' at the end of the name
535                 int i = procedureName.lastIndexOf(';');
536                 if (i > 0) {
537                     return procedureName.substring(0, i);
538                 }
539             default:
540                 return procedureName;
541         }
542     }

569      * Format a table name pattern by appending a wild card character
570      * and converting to lowercase or uppercase based on the case sensitivity setting
571      *
572      * @param tableNamePattern  a table name pattern
573      * @param dbMetaData        a database metadata to retrieve the database configurations
574      * @return                  a table name pattern
575      * @throws SQLException     if a database access error occurs
576      */
577     static String formatTableNamePattern(String tableNamePattern,
578             java.sql.DatabaseMetaData dbMetaData) throws SQLException {
579         if (StringUtils.isBlank(tableNamePattern)) {
580             tableNamePattern = WILD_CARD_CHAR;
581         } else {
582             tableNamePattern += WILD_CARD_CHAR;
583         }
584
