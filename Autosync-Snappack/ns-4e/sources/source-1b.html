290     public boolean evictConnection(final Connection conn, final DatabaseConfig config) {
291         if (conn == null || config == null || !config.isEvictConnectionOnClose()
292             || !(conn instanceof HikariProxyConnection)) {
293             return false;
294         }
295         try {
296             Map<DatabaseConfig, DataSource> snapshot = DATA_SOURCE_CACHE.asMap();
297             DataSource datasource = snapshot.get(config);
298             if (datasource instanceof HikariDataSource) {
299                 if (LOGGER.isDebugEnabled() && !isJUnitTest() && INTENSE_DEBUG) {
300                     dumpPoolStats("PRE-eviction", (HikariDataSource) datasource);
301                 }
302 
303                 // evict should close the connection immediately. otherwise, it will be
304                 // "marked" for eviction and evicted the next time someone tries to acquire
305                 // it from the pool.
306                 ((HikariDataSource)datasource).evictConnection(conn);
307 
308                 if (LOGGER.isDebugEnabled() && !isJUnitTest() && INTENSE_DEBUG) {
309                     dumpPoolStats("POST-eviction", (HikariDataSource) datasource);
310                 }
311 
312                 //TODO: how to determine if eviction really succeeded ? want this:
313                 //return ((HikariProxyConnection)conn).getPoolEntry().isMarkedEvicted();
314                 return true;
315             }
316         } catch (Exception e) {
317             LOGGER.info("(+) Unable to evict connection = @{}",
318                 Integer.toHexString(conn.hashCode()), e);
319         }
320         return false;
321     }

329     public Connection getConnection(final DatabaseConfig config) {
330         try {
331             String password = config.getPassword();
332             password = password == null ? EMPTY : password;
333             LOGGER.debug(String.format(JDBC_URL_FOR_CONNECTION,
334                     config.getJdbcUrl().replace(password, PASSWORD_MASK)));
335             // execute all the Hadoop connections as privileged user action
336             DataSource dataSource = DATA_SOURCE_CACHE.get(config);
337             if (config.getKerberosConfig() != null && (config.useKerberos() || config.useSSL()) &&
338                     config.getKerberosConfig().useHadoopClasses()) {
339                 return runPrivilegedConnection(dataSource, config);
340             } else {
341                 // Only run if we have real HikariDataSource and flag is set
342                 if (!isJUnitTest() && INTENSE_DEBUG) {
343                     LOGGER.debug("Hikari pool connections active count before new " +
344                             "connection is: {}, idle count is: {} and total count is: {}",
345                         getActiveCount((HikariDataSource) dataSource),
346                         getIdleCount((HikariDataSource) dataSource),
347                         getTotalCount((HikariDataSource) dataSource));
348                 }
349                 Connection conn = dataSource.getConnection();
350                 // Only run if we have real HikariDataSource and flag is set
351                 if (!isJUnitTest() && INTENSE_DEBUG) {
352                     LOGGER.debug("Hikari pool connections active count after new " +
353                             "connection is: {}, idle count is: {} and total count is: {}",
354                         getActiveCount((HikariDataSource) dataSource),
355                         getIdleCount((HikariDataSource) dataSource),
356                         getTotalCount((HikariDataSource) dataSource));
357                 }
358                 if (conn.isClosed()) {
359                     conn = retryGetConnection(config);
360                     if (conn == null) {
361                         throw new ExecutionException(ERR_CONNECT)
362                                 .withReason(REASON_CONNECT)
363                                 .withResolution(RESOLUTION_ADDRESS_ISSUE);
364                     }
365                 } else if (conn != null) {
366                     final java.sql.DatabaseMetaData md = conn.getMetaData();
367                     // may be null during unit testing
368                     if (md != null) {
369                         LOGGER.debug(JDBC_URL_SERVER_AND_DRIVER_VERSION, config.getJdbcUrl(),
370                                 md.getDatabaseProductName(), md.getDatabaseProductVersion(),
371                                 md.getDriverName(), md.getDriverVersion());
372                     }
373                 }
374                 return conn;
375             }
376         } catch (java.util.concurrent.ExecutionException e) {
377             // stop immediately if the failure was due to an authentication problem.
378             detectAndHandleLoginFailure(config, e);
379             for (Throwable cause : Throwables.getCausalChain(e)) {
380                 if (cause instanceof GSSException || cause instanceof LoginException) {
381                     safeConnectionPoolClose(config, null);
382                     throw new SnapDataException(cause, RETRIEVE_DATABASE_CONNECTION_FAILED)
383                             .withReason(cause.getLocalizedMessage())
384                             .withResolution(String.format(RESOLUTION_ADDRESS_ISSUE));
385                 } else if (cause instanceof TimeoutException) {
386                     LOGGER.debug("Timeout exception while loading into cache", e);
387                 }
388             }
389 
390             // else retry
391             Connection connection = retryGetConnection(config);
392             if (connection != null) {
393                 return connection;
394             }
395             Throwable root = Throwables.getRootCause(e);
396             throw new SnapDataException(root, CREATE_DATABASE_CONNECTION_FAILED)
397                     .withReason(root.getMessage())
398                     .withResolution(String.format(RESOLUTION_ADDRESS_ISSUE));
399         } catch (IllegalStateException | IllegalArgumentException |
400                 SQLException e) {
401             detectAndHandleLoginFailure(config, e);
402             Connection connection = retryGetConnection(config);
403             if (connection != null) {
404                 return connection;
405             }
406             // Hikari throws IllegalStateException or IllegalArgumentException when the pool
407             // configuration is not right. These exceptions should not result in driver
408             // de-registration.
409             throw new SnapDataException(e, RETRIEVE_DATABASE_CONNECTION_FAILED)
410                     .withReason(e.getMessage())
411                     .withResolution(String.format(RESOLUTION_ADDRESS_ISSUE));
412         } catch (Throwable e) {
413             detectAndHandleLoginFailure(config, e);
414             Connection connection = retryGetConnection(config);
415             if (connection != null) {
416                 return connection;
417             }
418             if (e instanceof UncheckedExecutionException
419                     && e.getCause() instanceof HikariPool.PoolInitializationException
420                     && e.getCause().getCause() != null) {
421                 // Remove HikariCP's fail-fast exception since it's not helpful for troubleshooting
422                 e = e.getCause().getCause();
423             }
424             if (e instanceof UncheckedExecutionException
425                     && e.getCause() instanceof UndeclaredThrowableException
426                     && e.getCause().getCause() != null) {
427                 // can be thrown by privileged block in getHadoopDataSource. The most
428                 // common exceptions should already be stripped but we may see unexpected ones.
429                 e = e.getCause().getCause();
430             }
431             if (e instanceof  UncheckedExecutionException
432                     && e.getCause() instanceof  UncheckedExecutionException
433                     && e.getCause().getCause() instanceof  AbstractMethodError){
434                 throw new SnapDataException(e, FAILED_TO_AUTO_DETECT)
435                         .withReason(e.getMessage())
436                         .withResolution(RESOLUTION_FAILED_TO_AUTO_DETECT);
437             }
438 
439             List<Throwable> causalChain = Throwables.getCausalChain(e);
440             for (Throwable throwable : causalChain) {
441                 // If there is any SQLException in the causal chain, it means the driver is valid.
442                 // Therefore we should not de-register the driver. Lets throw snap data exception.
443                 if (throwable instanceof SQLException) {
444                     // Check for Kerberos authentication issue - we want to deregister the driver
445                     // so we can try reconnecting with updated configuration files later. Otherwise
446                     // we would have to restart the jcc to flush the entry.
447                     if (KerberosUtilities.isKerberosAuthenticationIssue((SQLException) throwable)) {
448                         driverManager.deregisterDriver(config.getDriver());
449                     }
450                     throw new SnapDataException(e, RETRIEVE_DATABASE_CONNECTION_FAILED)
451                             .withReason(e.getMessage())
452                             .withResolution(String.format(RESOLUTION_ADDRESS_ISSUE));
453                 }
454             }
455             // as a precaution we will unload the driver to avoid troubles for other drivers
456             // loaded later in time.
457             driverManager.deregisterDriver(config.getDriver());
458             throw new ExecutionException(e, e.getMessage())
459                     .withResolution(DRIVER_LOADING_ERR_RESOLUTION);
460         }
461     }

518 
519         try {
520             // set temporary hadoop.home.dir for duration of getting data source.
521             if (KerberosUtilities.getHadoopHomeDir() != null) {
522                 System.setProperty(HADOOP_HOME_DIR_PROPERTY, KerberosUtilities.getHadoopHomeDir());
523             }
524             System.setProperty(USE_SUBJECT_CREDS_ONLY_PROPERTY, "false");
525 
526             // attempt to authenticate client to KDC. We do this in a separate
527             // try-catch block so our error message can be more precise that the
528             // problem occurred when authenticating the user.
529             LoginContext lc = null;
530             try {
531                 lc = KerberosUtilities.getKerberosLoginContext(krb5Config, false);
532                 LOGGER.debug("Login Context Hadoop Privileged Action:" + lc);
533                 lc.login();
534             } catch (LoginException e) {
535                 throw new ConfigurationException(ERROR_UNABLE_TO_AUTHENTICATE_TO_KDC_MESSAGE)
536                         .withReason(e.getLocalizedMessage())
537                         .withResolution(RESOLUTION_ADDRESS_ISSUE);
538             }
539 
540             // attempt to connect to server. This could fail since the server does
541             // not support Kerberos, is unknown to the KDC, etc. Again we try to be more
542             // precise about the problem.
543             try {
544                 UserGroupInformation ugi =
545                         UserGroupInformation.getUGIFromSubject(lc.getSubject());
546                 ugi.setAuthenticationMethod(config.getHadoopConfig().useSSL()
547                         ? UserGroupInformation.AuthenticationMethod.KERBEROS_SSL
548                         : UserGroupInformation.AuthenticationMethod.KERBEROS);
549                 ugi.reloginFromKeytab();
550                 ugi.checkTGTAndReloginFromKeytab();
551                 return ugi.doAs(action);
552 
553             } catch (UndeclaredThrowableException e) {
554                 // unwrap the most likely exceptions. This can make subsequent handling
555                 // a little cleaner since it hides the fact we used a privileged block.
556                 Throwable cause = e.getCause();
557                 if (cause instanceof SQLException) {
558                     throw (SQLException) cause;
559                 } else if (cause instanceof SaslException || cause instanceof GSSException ||
560                         cause instanceof LoginException) {
561                     throw new ConfigurationException(cause, ERROR_UNABLE_TO_CONNECT_TO_SERVER)
562                             .withReason(cause.getLocalizedMessage())
563                             .withResolution(RESOLUTION_ADDRESS_ISSUE);
564                 }
565                 throw e;
566             } catch (InterruptedException e) {
567                 throw e;
568             }
569         } catch (InterruptedException e) {
570             throw new SnapException(ERROR_INTERRUPTED_GETTING_CONNECTION_MESSAGE)
571                     .withReason(ERROR_INTERRUPTED_GETTING_CONNECTION_REASON)
572                     .withResolution(ERROR_INTERRUPTED_GETTING_CONNECTION_RESOLUTION);
573         } finally {
574             // restore hadoop.home.dir system property. Besides avoiding messing up other
575             // applications (excepting race conditions) this helps us detect that the
576             // environment variables have changed.
577             if (hadoopHomeDir == null) {
578                 System.clearProperty(HADOOP_HOME_DIR_PROPERTY);
579             } else {
580                 System.setProperty(HADOOP_HOME_DIR_PROPERTY, hadoopHomeDir);
581             }
582             if (useSubjectCredsOnly == null) {
583                 System.clearProperty(USE_SUBJECT_CREDS_ONLY_PROPERTY);
584             } else {
585                 System.setProperty(USE_SUBJECT_CREDS_ONLY_PROPERTY, useSubjectCredsOnly);
586             }
587         }

671          * @param config Configuration for the target connection pool
672          * @return HikariDataSource Connection pool

674          */
675         public static HikariDataSource loadHikariDatasource(DatabaseConfig config)
676                 throws Exception {

678             int checkoutTimeout = config.getConnectionTimeout() + 10000; // 10 seconds buffer time
679             if (checkoutTimeout < WAIT_TIMEOUT_IN_MILLI_SEC) {
680                 checkoutTimeout = WAIT_TIMEOUT_IN_MILLI_SEC;
681             }
682             try {
683                 return executor.submit(
684                         () -> getHikariDataSource(config)
685                 ).get(checkoutTimeout, TimeUnit.MILLISECONDS);
686             } catch (java.util.concurrent.ExecutionException e) {
687                 // Unwrap the concurrent exception to get its original cause. The change is to
688                 // maintain the backward compatibility with the change of SWAT-2349
689                 Throwable cause = e.getCause();
690                 if (cause instanceof Exception) {
691                     throw (Exception) cause;
692                 } else {
693                     throw new UncheckedExecutionException(cause);
694                 }
695             } finally {
696                 executor.shutdown();
697                 try {
698                     if (!executor.awaitTermination(EXECUTOR_SHUTDOWN_WAIT_TIMEOUT_IN_MIN,
699                             TimeUnit.MINUTES)) {
700                         executor.shutdownNow();
701                     }
702                 } catch (InterruptedException e) {
703                     // Ignore interruption
704                 }
705             }
706         }

709          * Get a HikariDataSource. Nearly all databases will use this method.
710          *
711          * @param config
712          *
713          * @return
714          */
715         private static HikariDataSource getHikariDataSource(DatabaseConfig config) throws
716                 ClassNotFoundException {
717             ClassLoader originalClassLoader = Thread.currentThread().getContextClassLoader();
718 
719             try {
720                 if (config.getClassLoader() != null) {
721                     Thread.currentThread().setContextClassLoader(config.getClassLoader());
722                     Class.forName(config.getDriver(), true, config.getClassLoader());
723 
724                 }
725                 final HikariConfig dataSourceConfig = new HikariConfig();
726                 DriverManager.setLoginTimeout(LOGIN_TIMEOUT_IN_SEC);
727                 String testQuery = JdbcDataSourceRegistryImpl.setupConnectionTestQuery(config);
728                 dataSourceConfig.setConnectionTestQuery(testQuery);
729                 String jdbcUrl = config.getJdbcUrl();
730                 dataSourceConfig.setDataSourceProperties(config.getUrlProperties());
731                 dataSourceConfig.setJdbcUrl(jdbcUrl);
732                 dataSourceConfig.setUsername(config.getUsername());
733                 String password = config.getPassword();
734                 password = password == null ? EMPTY : password;
735                 dataSourceConfig.setPassword(password);
736                 dataSourceConfig.setAutoCommit(config.getAutoCommit());
737                 int maxLifetime = config.getMaxLifetime();
738                 dataSourceConfig.setMaxLifetime(maxLifetime);
739                 int idleTimeout = config.getIdleTimeout();
740                 if (idleTimeout >= maxLifetime) {
741                     // When idleTimeout is greater than or equal to maxLifetime,
742                     // lets cut it down to half of max life time.
743                     // If idleTimeout == maxLifeTime then HikariConfig turns it to 0 (infinite)
744                     idleTimeout = maxLifetime / 2;
745                 }
746                 dataSourceConfig.setIdleTimeout(idleTimeout);
747                 dataSourceConfig.setMinimumIdle(0); // Min idle in the pool should always be 0.
748                 dataSourceConfig.setMaximumPoolSize(config.getMaximumPoolSize());
749                 dataSourceConfig.setConnectionTimeout(config.getConnectionTimeout());
750 
751                 // Make sure we have a URL that doesn't contain characters that mess up JMXBeans
752                 int index = jdbcUrl.indexOf(SEMI_COLON);
753                 jdbcUrl = index > 0 ? jdbcUrl.substring(0, index) : jdbcUrl;
754                 jdbcUrl = StringUtils.replace(jdbcUrl, password, PASSWORD_MASK);
755                 jdbcUrl = StringUtils.replace(jdbcUrl, COLON, UNDERSCORE);
756                 jdbcUrl = StringUtils.replace(jdbcUrl, EQUALS, UNDERSCORE)  ;
757 
758                 String poolName = String.format(POOL_NAME_FMT, config.getSQLDialect(),
759                     jdbcUrl, UUID.randomUUID().toString());
760                 dataSourceConfig.setPoolName(poolName);
761                 dataSourceConfig.setLeakDetectionThreshold(LEAK_DETECTION_INTERVAL);
762                 dataSourceConfig.setRegisterMbeans(true);
763                 dataSourceConfig.setDriverClassName(config.getDriver());
764 
765                 StringBuilder sb = new StringBuilder(config.getJdbcUrl());
766                 for (Map.Entry<Object, Object> entry : config.getUrlProperties().entrySet()) {
767                     sb.append(SEMI_COLON + entry.getKey() + "=" + entry.getValue());
768                 }
769                 String url = sb.toString();
770                 LOGGER.debug(String.format(JDBC_URL_FOR_DATASOURCE, url.replace(password,
771                         PASSWORD_MASK)));
772                 return new HikariDataSource(dataSourceConfig);
773 
774             } finally {
775                 Thread.currentThread().setContextClassLoader(originalClassLoader);
776             }
777         }

910      * Detect root cause of the Exception. Close the corresponding connection pool if the cause
911      * is login failure
912      *
913      * @param config Configuration of the connection
914      * @param e      Exception captured from the connection
915      */
916     private void detectAndHandleLoginFailure(final DatabaseConfig config, final Throwable e) {
917         boolean authenticationFailed = false;
918         SQLException sqlException = null;
919         for (Throwable cause : Throwables.getCausalChain(e)) {
920             if (cause instanceof SQLException) {
921                 sqlException = (SQLException) cause;
922                 String sqlState = sqlException.getSQLState();
923                 int errorCode = sqlException.getErrorCode();
924                 if (sqlState != null) {
925                     switch (sqlState) {
926                         case FAIL_AUTH_SQL_STATE_CODE_VERTICA_HIVE:
927                             if (errorCode == 500164 || errorCode == 3781) {
928                                 // 500164 for Hive and 3781 for Vertica
929                                 authenticationFailed = true;
930                             }
931                             break;
932                         case FAIL_AUTH_SQL_STATE_CODE_POSTGRES:
933                         case FAIL_AUTH_SQL_STATE_CODE_MYSQL_REDSHIFT_SAP_HANA_TERADATA:
934                             // Postgres, MySQL, Redshift, SAP Hana and Teradata
935                             authenticationFailed = true;
936                             break;
937                         case FAIL_AUTH_SQL_STATE_CODE_SQL_SERVER:
938                             if (errorCode == 18456) {
939                                 // SQL Server
940                                 authenticationFailed = true;
941                             }
942                             break;
943                         case FAIL_AUTH_SQL_STATE_CODE_SNOWFLAKE:
944                             if (errorCode == 200002) {
945                                 // Snowflake
946                                 authenticationFailed = true;
947                             }
948                             break;
949                         case FAIL_AUTH_SQL_STATE_CODE_ORACLE:
950                             if (errorCode == 1017) {
951                                 // Oracle
952                                 authenticationFailed = true;
953                             }
954                     }
955                 } else if (ERROR_INVALID_CREDENTIALS_CASSANDRA.equals(
956                         sqlException.getMessage())) {
957                     // Cassandra
958                     authenticationFailed = true;
959                 }
960                 if (authenticationFailed) {
961                     safeConnectionPoolClose(config, null);
962                     // If the issue is caused by login failure, then it should be treated as
963                     // ExecutionException
964                     throw new ExecutionException(
965                             sqlException, RETRIEVE_DATABASE_CONNECTION_FAILED)
966                             .withReason(sqlException.getLocalizedMessage())
967                             .withResolution(RESOLUTION_LOGIN_FAILURE);
968                 }
969             }
970         }
971     }

974      * Close connection pool within a time bounded call.
975      * If operation is not performed in that time, TimeoutException would be thrown.
976      * @param config Configuration for the target connection pool
977      * @param connectionPool
978      */
979     private static void safeConnectionPoolClose(DatabaseConfig config,
980             HikariDataSource connectionPool) {
981         shutdownAbandonedConnectionCleanupThread(config.getClassLoader());
982         ExecutorService executor = Executors.newSingleThreadExecutor();
983         try {
984             // If connectionPool object is found, close it directly here.
985             if (Objects.nonNull(connectionPool)) {
986                 executor.submit(
987                         () -> connectionPool.close()
988                 ).get(WAIT_TIMEOUT_IN_MILLI_SEC, TimeUnit.MILLISECONDS);
989             } else {
990                 executor.submit(
991                         () -> closeConnectionPoolFor(config)
992                 ).get(WAIT_TIMEOUT_IN_MILLI_SEC, TimeUnit.MILLISECONDS);
993             }
994         } catch (Exception e) {
995             // Not throwing any error as we do not want to hamper current execution for not being
996             // able to close other connections.
997             LOGGER.error("Unable to shutdown the Connection pool", e);
998             FAILED_TO_SHUTDOWN_POOL.report();
999         } finally {
1000             executor.shutdown();
1001             try {
1002                 if (!executor.awaitTermination(EXECUTOR_SHUTDOWN_WAIT_TIMEOUT_IN_MIN,
1003                         TimeUnit.MINUTES)) {
1004                     executor.shutdownNow();
1005                 }
1006             } catch (InterruptedException e) {
1007                 // Ignore interruption
1008             }
1009         }

1017     private static void closeConnectionPoolFor(DatabaseConfig config) {

1019         // When credential expired, connections in the pool will no longer be valid at all. It is
1020         // safe to close it in this case.
1021         Map<DatabaseConfig, DataSource> snapshot = DATA_SOURCE_CACHE.asMap();
1022         DataSource ds = snapshot.get(config);
1023         if (ds != null) {

1026                 connectionPool.close();
1027                 LOGGER.debug(String.format(CONNECTION_POOL_CLOSED, connectionPool.getPoolName()));

1029                 try {
