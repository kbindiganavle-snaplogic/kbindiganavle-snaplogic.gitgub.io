178      * Check an exception to see if it might've been caused by bad data.  We want to send
179      * bad-data errors to the error view, while other types of errors should stop the
180      * pipeline.
181      *
182      * Unfortunately, the different databases do things quite a bit differently, so this
183      * function has to do some implementation-specific checks and cannot just rely on
184      * JDBC or SQLSTATE values.
185      *
186      * @param throwable The Throwable to check.
187      *
188      * @return True if the given Throwable is a SQLException that was caused by bad data.
189      */
190     public static boolean isDataException(Throwable throwable) {
191         if (!(throwable instanceof SQLException)) {
192             return false;
193         }
194 
195         SQLException sqlException = (SQLException) throwable;
196         String sqlState = sqlException.getSQLState();
197         int errorCode = sqlException.getErrorCode();
198         LOG.debug(CHECK_EXCEPTION_LOG, sqlState, errorCode);
199 
200         if (SQLSTATE_SQL_SERVER.equals(sqlState)) {
201             switch (errorCode) {
202                 case SQL_SERVER_DIVIDE_BY_ZERO:
203                 case SQL_SERVER_CONVERSION_ERROR:
204                     return true;
205             }
206         }
207 
208         return (SQLSTATE_TRANSACTION_CONSTRAINT.equals(sqlState) ||
209                 SQLSTATE_SQL_SERVER_OVERFLOW.equals(sqlState) ||
210                 StringUtils.startsWith(sqlState, SQLSTATE_DATA_ACCESS_PREFIX) ||
211                 StringUtils.startsWith(sqlState, SQLSTATE_CONSTRAINT_PREFIX) ||
212                 StringUtils.startsWith(sqlState, SQLSTATE_EXECUTE_PREFIX) ||
213                 // Oracle incorrectly classifies errors where an operand is expected to be
214                 // a number as a syntax error instead of a semantic error.
215                 StringUtils.startsWith(sqlException.getMessage(), ORACLE_NOT_A_NUMBER) ||
216                 sqlException instanceof SQLIntegrityConstraintViolationException ||
217                 sqlException instanceof SQLDataException);
218     }

272                             return true;

282      * original data is null, then the data results will be returned.
283      *
284      * @param cursor          database cursor
285      * @param original        original document from the input stream
286      * @param outputViews     output views
287      * @param documentUtility document utility
288      * @param viewName        The name of the view to write the results to.  If null, the results
289      *                        will be written to all of the output views.
290      * @param passThrough     pass through original document
291      * @param ignoreEmptyResult  do not pass through original document if no result
292      * @param query           original query
293      * @param targetDbName    Name of the underlying target db (such as Oracle, Sql Server, etc)
294      */
295     public void processSelect(final DatabaseCursor cursor, final Document original,
296             final OutputViews outputViews, final DocumentUtility documentUtility,
297             @Nullable final String viewName, final Boolean passThrough,
298             final Boolean ignoreEmptyResult, String query, String targetDbName)
299             throws SQLException {
300         try {
301             if (!cursor.hasNext()) {
302                 Map<String, Object> successRecord = new LinkedHashMap<>();
303                 successRecord.put("message", SUCCESS);
304                 Document successDoc = documentUtility.newDocument(successRecord);
305                 boolean writeSuccessDocForNoQueryOutput = StringUtils.isNotBlank(query) &&
306                         isProcedureCallOrDeclareSelect(query, targetDbName);
307 
308                 // The cursor had no results so write original data
309                 if (original != null) {
310                     if (passThrough == null) {
311                         // try to be backward-compatible
312                         if (StringUtils.isNotBlank(viewName)) {
313                             outputViews.getDocumentViewFor(viewName).write(original);
314                         } else {
315                             if (writeSuccessDocForNoQueryOutput) {
316                                 outputViews.write(successDoc, original);
317                             } else {
318                                 outputViews.write(original);
319                             }
320                         }
321                     } else if (passThrough && !Boolean.TRUE.equals(ignoreEmptyResult)) {
322                         Document document = documentUtility.newDocument(
323                                 documentUtility.newDocument(new LinkedHashMap<>(0)), original);
324                         if (StringUtils.isNotBlank(viewName)) {
325                             outputViews.getDocumentViewFor(viewName).write(document);
326                         } else {
327                             if (writeSuccessDocForNoQueryOutput) {
328                                 outputViews.write(successDoc, original);
329                             } else {
330                                 outputViews.write(document);
331                             }
332                         }
333                     } else {
334                         if (writeSuccessDocForNoQueryOutput) {
335                             outputViews.write(successDoc);
336                         }
337                         // ack document so platform knows that it was processed.
338                         original.acknowledge();
339                     }
340                 } else {
341                     if (writeSuccessDocForNoQueryOutput) {
342                         outputViews.write(successDoc);
343                     }
344                 }
345             } else {
346                 // The cursor had results so merge result data with original data
347                 Document inputDocument = null;
348                 if (passThrough == null || passThrough) {
349                     inputDocument = original;
350                 }
351                 do {
352                     Map<String, Object> dbData = cursor.fetchOne();
353                     Document dbDoc = documentUtility.newDocumentFor(original, dbData);
354                     if (viewName != null) {
355                         outputViews.getDocumentViewFor(viewName).write(
356                                 documentUtility.newDocument(dbDoc, inputDocument));
357                     } else {
358                         outputViews.write(dbDoc, inputDocument);
359                     }
360                 } while (cursor.hasNext());
361             }
362         } catch (DataAccessException e) {
363             Throwable cause = e.getCause();
364             if (DatabaseUtils.isDataException(cause)) {
365                 throw (SQLException) cause;
366             }
367             throw e;
368         }
369     }

424             throw new ExecutionException(error)
425                     .withReason(ERR_CANNOT_FIND_TABLE_REASON, tableName)
426                     .withResolution(RESOLUTION_TABLE_NAME);
427         }
428         for (String viewName : provider.getRegisteredViewNames()) {
429             if (viewNames.length > 0 && !ArrayUtils.contains(viewNames, viewName)) {
430                 continue;
431             }
432             SchemaBuilder schemaBuilder = provider.getSchemaBuilder(viewName);
433             for (Map.Entry<String, ColumnMetaData> entry : tableMetadata.entrySet()) {
434                 ColumnMetaData columnMetaData = entry.getValue();
435                 String colName = entry.getKey();
436                 if (CollectionUtils.isNotEmpty(selectedColumns) &&
437                         !selectedColumns.contains(colName)) {
438                     continue;
439                 }
440                 DefaultTypeMappingHandler.SnapTypeInfo snapInfoType = typeMappingHandler
441                         .resolve(columnMetaData.getDataType(), columnMetaData.getColumnTypeName());
442

445                             columnMetaData.getDataType(), columnMetaData.getColumnTypeName());
446                 }
447

449                 Schema childSchema = provider.createSchema(snapType,

461             }
462             schemaBuilder.build();

466     public Map<String, Object> getOnConditionValueMap(final Document document,
467             final Matcher matcher) {
468         // Merge on should have a column name -> column value mapping
469         Map<String, Object> onConditionValueMap = Maps.newLinkedHashMap();
470         String expression = null;
471         try {
472             while (matcher.find()) {
473                 expression = matcher.group();
474                 PathTokenizer tokenizer = new PathTokenizer(expression);
475                 PathToken lastToken = tokenizer.removeLastPathToken();
476                 String dataKey = lastToken.getFragment();
477                 Object val;
478                 try {
479                     val = snapExpressionService.evaluateExpression(expression, document);
480                 } catch (IllegalArgumentException e) {
481                     // Value was missing in the document so default to null
482                     val = null;
483                 }
484                 // Cannot update columns used in merge conditions
485                 onConditionValueMap.put(dataKey, val);
486             }
487         } catch (SnapDataException e) {
488             throw new ExecutionException(e, EXPRESSION_EVAL_FAILED, expression);
489         } finally {
490             matcher.reset();
491         }
492         return onConditionValueMap;
493     }

732     /**
733      * Handles results from an update batch. The DB decides if the record gets
734      * create/updated/deleted as part of the batch operation. Records can still be modified even if
735      * one record of the batch fails, hence we only write the failing records to the error view.
736      *
737      * In an auto-commit off mode, this is even trickier. We can't keep all records of all
738      * batches in memory (unless we write all to disk) to decide which view they go to.
739      * For now: We write the failing ones to the error view, all other records will be written to
740      * the output view, even though if we execute a rollback in that case.
741      * @param outputViews          output views
742      * @param errorViews           error views
743      * @param rowsModified         rows modified
744      * @param documentUtility      document utility
745      * @param inputDocuments       input documents
746      * @param batchUpdateException The exception that was thrown during the update.  The
747      *                             message in this exception will be included in the doc
748      *                             written to the error view.
749      * @param runtimeDBName        The database name that gets from JDBC driver at runtime for
750      *                             JDBC snap pack.
751      */
752     public boolean handleUpdateResult(OutputViews outputViews, ErrorViews errorViews,
753             int[] rowsModified, final DocumentUtility documentUtility, List<?> inputDocuments,
754             BatchUpdateException batchUpdateException,
755             final String runtimeDBName, Boolean passThrough) {
756         boolean hasErrors = false;
757         if (SPARKSQL.equals(runtimeDBName) && rowsModified != null && rowsModified.length == 1) {
758             // If the target db is Spark SQL and a batch is executed successfully, rowsModified is
759             // an array of length 1 with its value as number of rows modified. If a batch execution
760             // fails, an exception is thrown and all records in the batch are not inserted.
761             if (rowsModified[0] == inputDocuments.size()) {
762                 for (Object doc : inputDocuments) {
763                     Map<String, Object> data = new LinkedHashMap<>();
764                     data.put(STATUS, 1);
765                     data.put(MESSAGE, SUCCESS);
766                     outputViews.write(documentUtility.newDocument(data), doc);
767                 }
768                 inputDocuments.clear();
769                 return hasErrors;
770             }
771         }
772         if (rowsModified != null && outputViews != null && inputDocuments != null) {
773             int rowsSize = rowsModified.length;
774             boolean isSingleQueryBatch = false;
775             /*
776              * With different versions of Snowflake JDBC drivers the batch operation is executed
777              * in two different ways:
778              *
779              * 1. With driver in version 1.14.0, there is only one query with a binding array
780              * executed, the returned array contains only one element that represents the total
781              * update count of the batch.
782              *
783              * 2. With driver in version 1.24.1, there are queries as many as the number of
784              * the batch size executed. The number of elements in the returned array equals to the
785              * batch size, with each element as 0.
786              */
787             if (rowsModified.length == 1 && rowsModified[0] > 0 &&
788                     StringUtils.equalsIgnoreCase(runtimeDBName, JdbcOperations.SNOWFLAKE) ||
789                     StringUtils.equalsIgnoreCase(runtimeDBName,
790                             JdbcOperations.DATABRICKS_LAKEHOUSE_PLATFORM)) {
791                 isSingleQueryBatch = true;
792                 rowsSize = rowsModified[0];
793             }
794             int docsSize = inputDocuments.size();
795             int min = isSingleQueryBatch ? docsSize : Math.min(rowsSize, docsSize);
796             for (int i = 0; i < min; i++) {
797                 Document doc = null;
798                 Document lineageDoc = (Document) inputDocuments.get(i);
799                 if (passThrough == null || passThrough) {
800                     doc = lineageDoc;
801                 }
802                 int result = isSingleQueryBatch ? rowsModified[0] : rowsModified[i];
803                 Map<String, Object> data = Maps.newLinkedHashMap();
804                 data.put(STATUS, result);
805                 if (result >= 0) {
806                     if (StringUtils.equalsIgnoreCase(runtimeDBName, JdbcOperations.POSTGRES) &&
807                             rowsSize != docsSize) {
808                         hasErrors = true;
809                         String reason = DOCS_NOT_EQUAL_TO_BATCH_SIZE;
810                         if (null != batchUpdateException) {
811                             reason =
812                                     getSqlErrorMessage(batchUpdateException.getNextException());
813                         }
814                         writeDocumentToErrorView(batchUpdateException, errorViews, doc,
815                                 documentUtility, result, reason);
816                     } else {
817                         data.put(MESSAGE, SUCCESS);
818                         outputViews.write(documentUtility
819                                 .newDocumentFor(lineageDoc, data), doc);
820                     }
821                 } else {
822                     switch (result) {
823                         case Statement.SUCCESS_NO_INFO:
824                             if (batchUpdateException == null) {
825                                 data.put(MESSAGE, SUCCESS_NO_INFO);
826                                 outputViews.write(documentUtility.
827                                         newDocumentFor(lineageDoc, data), doc);
828                             } else {
829                                 hasErrors = true;
830                                 writeDocumentToErrorView(batchUpdateException, errorViews, doc,
831                                         documentUtility, result,
832                                         getSqlErrorMessage(batchUpdateException));
833                             }
834                             break;
835                         case Statement.EXECUTE_FAILED:
836                             hasErrors = true;
837                             doc = (Document) inputDocuments.get(i);
838                             writeDocumentToErrorView(batchUpdateException, errorViews, doc,
839                                     documentUtility, result, batchUpdateException != null ?
840                                             getSqlErrorMessage(batchUpdateException) :
841                                             EXECUTE_FAILED);
842                             break;
843                         default:
844                             outputViews.write(documentUtility.newDocumentFor(doc, data), doc);
845                     }
846                 }
847             }
848 
849             // if rowsModified array is shorter than inputDocuments length, send the rest
850             // of the documents to the error view
851             if (!isSingleQueryBatch && rowsSize < docsSize) {
852                 hasErrors = true;
853                 writeErrorView(errorViews, inputDocuments.subList(rowsSize, docsSize),
854                         batchUpdateException);
855             }
856             // Clear the input documents because the batch has executed
857             inputDocuments.clear();
858         } else if (batchUpdateException != null) {
859             // in SQL Server, if stored procedure was executed successfully, do not throw exception
860             String message = batchUpdateException.getMessage();
861             if (RESULT_SET_GENERATED.equalsIgnoreCase(message)) {
862                 Map<String, Object> outputMap = new LinkedHashMap<>();
863                 outputMap.put(MESSAGE, message);
864                 Document outputDocument = documentUtility.newDocument(outputMap);
865                 if (CollectionUtils.isNotEmpty(inputDocuments)) {
866                     for (Object object : inputDocuments) {
867                         if (passThrough == null || passThrough) {
868                             outputViews.write(outputDocument, object);
869                         } else {
870                             outputViews.write(outputDocument);
871                         }
872                     }
873                 } else {
874                     outputViews.write(outputDocument);
875                 }
876                 return false;
877             }
878             // If the driver does not return any update-counts, make sure an error is written.
879             hasErrors = true;
880             writeErrorView(errorViews, inputDocuments, batchUpdateException);
881         }
882         return hasErrors;
883     }

885     protected void writeDocumentToErrorView(BatchUpdateException batchUpdateException,
886             ErrorViews errorViews, Document original, DocumentUtility documentUtility,
887             int status, String reason) {
888         Map<String, Object> error = Maps.newLinkedHashMap();
889         error.put(SimpleSqlSnap.ERROR, BATCH_OPERATION_FAILED);
890         error.put(REASON, reason);
891         error.put(RESOLUTION, getResolutionForInvalidDataError(reason));
892         error.put(STATUS, status);
893         SnapDataException snapDataException = new SnapDataException(documentUtility.newDocumentFor
894                 (original, error), batchUpdateException, BATCH_OPERATION_FAILED)
895                 .withReason(reason)
896                 .withResolution(getResolutionForInvalidDataError(reason));
897         error.put(STACKTRACE, batchUpdateException != null ?
898                 Throwables.getStackTraceAsString(
899                 batchUpdateException.getCause() != null ? batchUpdateException.getCause() :
900                         batchUpdateException) :
901                 Throwables.getStackTraceAsString(snapDataException));
902         errorViews.write(snapDataException, original);
903     }

906      * Writes all input documents to error view and clears inputDocuments.
907      * If no document, write an error to error view.
908      *
909      * @param errorViews     - error view
910      * @param inputDocuments - list of documents
911      * @param throwable      - throwable object
912      */
913     public static void writeErrorView(ErrorViews errorViews, List<?> inputDocuments,
914             Throwable throwable) {
915         SnapDataException snapDataException;
916         if (throwable instanceof BatchUpdateException) {
917             String message = SQL_OPERATION_FAILED;
918             if (CollectionUtils.isNotEmpty(inputDocuments) && inputDocuments.size() > 1) {
919                 message = BATCH_OPERATION_FAILED;
920             }
921             BatchUpdateException batchUpdateException = (BatchUpdateException) throwable;
922             SQLException sqlException = batchUpdateException.getNextException();
923             String reason;
924             if (sqlException != null) {
925                 reason = getSqlErrorMessage(sqlException);
926             } else {
927                 reason = getSqlErrorMessage(batchUpdateException);
928             }
929             snapDataException = new SnapDataException(throwable, message)
930                     .withReason(reason)
931                     .withResolution(getResolutionForInvalidDataError(reason));
932         } else if (throwable instanceof ExecutionException) {
933             ExecutionException e = (ExecutionException) throwable;
934             String message = SQL_OPERATION_FAILED;
935             if (CollectionUtils.isNotEmpty(inputDocuments) && inputDocuments.size() > 1) {
936                 message = BATCH_OPERATION_FAILED;
937             }
938             String reason = StringUtils.isBlank(e.getMessage()) ? e.getReason() :
939                     e.getMessage().equals(e.getReason()) ? e.getMessage() :
940                     String.format(FORMAT_APPEND_MESSAGES, e.getMessage(), e.getReason());
941             snapDataException = new SnapDataException(throwable, message)
942                     .withReason(reason)
943                     .withResolution(e.getResolution());
944         } else {
945             String message = SQL_OPERATION_FAILED;
946             if (CollectionUtils.isNotEmpty(inputDocuments) && inputDocuments.size() > 1) {
947                 message = BATCH_OPERATION_FAILED;
948             }
949             String reason = throwable == null ? message : throwable.getMessage();
950             snapDataException = new SnapDataException(throwable, message)
951                     .withReason(reason)
952                     .withResolution(getResolutionForInvalidDataError(reason));
953         }
954         if (CollectionUtils.isNotEmpty(inputDocuments)) {
955             if (errorViews != null) {
956                 for (Object object : inputDocuments) {
957                     errorViews.write(snapDataException, object);
958                 }
959             }
960             // Clear the input documents because the batch has executed
961             inputDocuments.clear();
962         } else {
963             errorViews.write(snapDataException);
964         }
965     }

1066     /**
1067      * Compute the order of tables in the given graph.  The ordering is from
1068      * least dependent to most-dependent.
1069      *
1070      * @param tableGraph The graph of table names to their dependent tables.
1071      *
1072      * @return An ordered list of table names.
1073      */
1074     public List<String> computeTableGraphOrder(Map<String, Set<String>> tableGraph) {
1075         List<String> retval = new ArrayList<>();
1076         tableGraph = DeepUtils.copy(tableGraph);
1077 
1078         while (!tableGraph.isEmpty()) {
1079             List<String> subset = new ArrayList<>();
1080             String minNodeName = null;
1081             int minNodeSize = Integer.MAX_VALUE;
1082             // Find the set of tables that have no dependencies or, if there are none,
1083             // the table with the fewest.
1084             for (Map.Entry<String, Set<String>> entry : tableGraph.entrySet()) {
1085                 Set<String> dependentTables = entry.getValue();
1086                 if (dependentTables.isEmpty()) {
1087                     subset.add(entry.getKey());
1088                 }
1089                 if (dependentTables.size() < minNodeSize) {
1090                     minNodeSize = dependentTables.size();
1091                     minNodeName = entry.getKey();
1092                 }
1093             }
1094             if (subset.isEmpty()) {
1095                 subset.add(minNodeName);
1096             }
1097 
1098             for (String name : subset) {
1099                 tableGraph.remove(name);
1100                 for (Map.Entry<String, Set<String>> entry : tableGraph.entrySet()) {
1101                     entry.getValue().remove(name);
1102                 }
1103             }
1104             Collections.sort(subset);
1105             retval.addAll(subset);
1106         }
1107 
1108         return retval;
1109     }

1111     private void traverseDependents(Map<String, Set<String>> accumulator, Map<String,
1112             Set<String>> graph, String currentDependent) {
1113         if (!accumulator.get(currentDependent).isEmpty()) {
1114             // We've already traversed this table.
1115             return;
1116         }
1117 
1118         Set<String> nextDependents = new HashSet<>();
1119 
1120         for (Map.Entry<String, Set<String>> entry : graph.entrySet()) {
1121             if (entry.getValue().contains(currentDependent)) {
1122                 nextDependents.add(entry.getKey());
1123             }
1124         }

1314      * Converts binary values into Hex representation.
1315      *
1316      * @param value that should be converted
1317      * @return converted string
1318      */
1319     public static String getBinaryValueInHex(final byte[] value) {
1320         // Snowflake supports three forms of encoding for
1321         // binary data types.
1322         // 1. UTF-8
1323         // 2. BASE64
1324         // 3. HEX
1325         // By default it is always HEX encoding.
1326         // To begin with we don't know what kind of encoding
1327         // was used to store the binary value in the database.
1328         // We will start with the assumption that it may be
1329         // UTF-8 encoding. If it turns out to be UTF-8 encoding
1330         // we just encode the input byte array into HEX and return.
1331         // If it is not UTF-8 encoding, we attempt to see if it
1332         // is BASE64 encoding using the BASE64 decoder. If it
1333         // succeeds, we just encode the decoded value in HEX
1334         // and return it back. If the BASE64 decoding fails it
1335         // raises an exception, in which case we simply assume
1336         // it is HEX encoding and return HEX encoded value.
1337         final String converted = new String(value, StandardCharsets.UTF_8);
1338         final byte[] outputBytes = converted.getBytes(StandardCharsets.UTF_8);
1339         if (Arrays.equals(value, outputBytes)) {
1340             return convertToHex(value); // means it is UTF-8 encoded
1341         }
1342         Base64.Decoder decoder = Base64.getDecoder();
1343         try {
1344             byte[] decoded = decoder.decode(value);
1345             return convertToHex(decoded); //means it is Base64 encoded.
1346         } catch (Exception ex) {
1347             return convertToHex(value);   // means it is Hex encoded
1348         }
1349     }

1372         // to decode it to make sure it is Base64 encoded. If the value is
