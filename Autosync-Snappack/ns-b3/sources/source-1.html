72     private static <T> T fetchObjectFromInputStream(InputStream is, TypeReference typeRef, String mappingExMessage) {
73         try (InputStream inputStream = is) {
74             if (inputStream.available() > 0) {
75                 return (T) OBJECT_MAPPER.readValue(inputStream, typeRef);
76             }
77             throw new ExecutionException(NO_AVAILABLE_BYTES)
78                     .withReason(NO_AVAILABLE_BYTES)
79                     .withResolutionAsDefect();
80         } catch (JsonParseException e) {
81             throw new ExecutionException(e, JSON_PARSING_ERROR)
82                     .withReason(e.getMessage())
83                     .withResolutionAsDefect();
84         } catch (JsonMappingException e) {
85             throw new ExecutionException(e, mappingExMessage)
86                     .withReason(e.getMessage())
87                     .withResolutionAsDefect();
88         } catch (IOException e) {
89             throw new ExecutionException(e, IO_ERROR)
90                     .withReason(e.getMessage())
91                     .withResolutionAsDefect();
92         }
93     }

162     public static BigInteger parseBigIntExpression(ExpressionProperty expr, Document document) {
163         Object value = expr.eval(document);
164         if (value instanceof BigInteger) {
165             return (BigInteger) value;
166         } else if (value == null) {
167             return null;
168         }
169         String s = String.valueOf(value);
170         try {
171             return new BigInteger(s);
172         } catch (NumberFormatException ex) {
173             throw new SnapDataException(String.format(NOT_AN_INTEGER, s))
174                     .withReason(EXPECTING_AN_INTEGER)
175                     .withResolution(RESOLUTION_ADDRESS_ISSUE);
176         }
177     }
