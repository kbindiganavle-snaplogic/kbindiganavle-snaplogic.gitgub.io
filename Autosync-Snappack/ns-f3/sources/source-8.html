278         if (StringUtils.isBlank(destinationName)) {
279             throw new ConfigurationException(JMS_PRODUCER_DESTINATION_ERROR_MSG)
280                     .withReason(JMS_DESTINATION_ERROR_REASON)
281                     .withResolution(JMS_DESTINATION_ERROR_RESOLUTION);
282         }
283         MessageProducer messageProducer = producerCache.getIfPresent(destinationName);
284         if (messageProducer == null) {
285             Destination destination = JMSUtils.createDestination(session, destinationType,
286                     destinationName);
287             JMSUtils.checkDestinationType(destinationType, destination, destinationName);
288             try {
289                 messageProducer = session.createProducer(destination);
290                 producerCache.put(destinationName, messageProducer);
291             } catch (JMSException e) {

293                         .withReason(e.getMessage())

301     @Override
302     protected void process(Document header, ReadableByteChannel readChannel) {
303         String evalDestinationName = destinationNameExp.eval(header);
304         AtomicBoolean firstRun = new AtomicBoolean(true);
305         AtomicBoolean createMessage = new AtomicBoolean();
306         StringBuffer inputMessage = new StringBuffer();
307         final byte[] inputByte;
308         boolean exceptionOccurred = false;
309 
310         try (InputStream inputStream = Channels.newInputStream(readChannel)) {
311             if (msgType.equals(MessageType.BYTES) || msgType.equals(MessageType.STREAM)) {
312                 inputByte = IOUtils.toByteArray(inputStream);
313             } else {
314                 inputMessage.setLength(0);
315                 inputMessage.append(IOUtils.toString(inputStream, Charsets.UTF_8));
316                 inputByte = new byte[0];
317             }
318             Failsafe
319                     .with(retryPolicy)
320                     .get(context -> {
321                         // handle retry
322                         Execution execution = (Execution) context;
323                         if (execution.getAttemptCount() > 0) {
324                             // its  retry
325                             // close prev resources
326                             releaseResourcesAndCleanupCache();
327                             // reconnect
328                             session = account.connect();
329                         }
330 
331                         // initialise resources for first run
332                         if (firstRun.get() && session == null) {
333                             session = account.connect();
334                             firstRun.set(false);
335                         }
336 
337                         final MessageProducer producer = createProducer(evalDestinationName);
338                         createMessage.set(false);
339                         Message message = null;
340                         try {
341                             switch (msgType) {
342                                 case TEXT:
343                                     message = session.createTextMessage(inputMessage.toString());
344                                     break;
345                                 case MAP:
346                                     // TODO: Support MapMessage via a different snap that has DOCUMENT as an
347                                     // input view
348                                     SnapObjectMapper mapper = new SnapObjectMapper();
349                                     Map<String, Object> jsonMap = mapper.readValue(inputMessage.toString(), Map.class);
350                                     if (MapUtils.isNotEmpty(jsonMap)) {
351                                         Iterator<String> iter = jsonMap.keySet().iterator();
352                                         MapMessage mapmessage = session.createMapMessage();
353                                         while (iter.hasNext()) {
354                                             String key = iter.next();
355                                             Object value = jsonMap.get(key);
356                                             if (value == null) {
357                                                 mapmessage.setString(key, StringUtils.EMPTY);
358                                             } else if (value instanceof String) {
359                                                 mapmessage.setString(key, (String) value);
360                                             } else if (value instanceof BigInteger) {
361                                                 mapmessage.setLong(key, ((BigInteger) value).longValue());
362                                             } else if (value instanceof BigDecimal) {
363                                                 mapmessage.setDouble(key, ((BigDecimal) value).doubleValue());
364                                             } else if (value instanceof Boolean) {
365                                                 mapmessage.setBoolean(key, ((Boolean) value));
366                                             } else {
367                                                 mapmessage.setObject(key, jsonMap.get(key));
368                                             }
369                                         }
370                                         message = mapmessage;
371                                     }
372                                     break;
373                                 case STREAM:
374                                     message = session.createStreamMessage();
375                                     ((StreamMessage) message).writeBytes(inputByte);
376                                     break;
377                                 case BYTES:
378                                     message = session.createBytesMessage();
379                                     ((BytesMessage) message).writeBytes(inputByte);
380                                     break;
381                                 default:
382                                     // Will never reach here
383                                     throw new ExecutionException(INVALID_MESSAGE_TYPE, msgType.name());
384                             }
385                         } catch (JMSException e2) {
386                             createMessage.set(true);
387                             throw e2;
388                         }
389 
390                         if (header != null) {
391                             MessageHandler.populateMessageHeader(header, message);
392                         }
393 
394                         if (messageHeaderProperties != null) {
395                             MessageHandler.populateMessageProperties(messageHeaderProperties, message, header);
396                         }
397 
398                         Long msgExp = MessageHandler.getExpiration(header, messageHeaderProperties);
399                         if (msgExp != null) {
400                             producer.send(message, DeliveryMode.NON_PERSISTENT, 0, msgExp);
401                         } else {
402                             producer.send(message);
403                         }
404 
405                         Map<String, String> data = new LinkedHashMap<>(2);
406                         data.put(CORRELATION_ID, message.getJMSCorrelationID());
407                         data.put(MESSAGE_ID, message.getJMSMessageID());
408 
409                         if (isTransacted) {
410                             data.put(TRANSACTION_ID_KEY, getTransactionID(message));
411                         }
412 
413                         outputViews.write(documentUtility.newDocumentFor(header, data), header);
414 
415                         if (isTransacted) {
416                             session.commit();
417                         }
418 
419                         messageCounter.inc();
420                         header.acknowledge();
421 
422                         return null;
423                     });
424         } catch (Exception e1) {
425             exceptionOccurred = true;
426             Throwable cause = e1 instanceof FailsafeException ? e1.getCause() : e1;
427 
428             if (cause instanceof SnapDataException) {
429                 errorViews.write((SnapDataException) cause);
430             } else if (cause instanceof JMSException) {
431                 JMSException jmsException = (JMSException) cause;
432 
433                 if (createMessage.get()) {
434                     throw new SnapDataException(jmsException, ERROR_WHILE_CREATING_MESSAGE)
435                             .withReason(String.format(JMS_PROVIDER_MESSAGE_CREATION_FAILURE,
436                                             jmsException.getMessage()))
437                             .withResolution(JMS_RES_VERIFY_INPUT_DATA);
438                 } else {
439                     if (isTransacted) {
440                         try {
441                             session.rollback();
442                         } catch (JMSException jme) {
443                             throw new SnapDataException(jme, ERROR_ROLL_BACK_SESSION)
444                                     .withReason(jme.getMessage())
445                                     .withResolution(ERROR_ROLL_BACK_SESSION_RESOLUTION);
446                         }
447                     }
448                     SnapDataException snapDataException =
449                             new SnapDataException(jmsException, ERROR_WHILE_SENDING_MESSAGE)
450                                     .withReason(String.format(JMS_PROVIDER_MESSAGE_SENDING_FAILURE,
451                                             jmsException.getMessage()))
452                                     .withResolution(JMS_RES_VERIFY_INPUT_DATA);
453                     errorViews.write(snapDataException, header);
454                 }
455             } else if (cause instanceof IOException) {
456                 IOException ioException = (IOException) cause;
457 
458                 SnapDataException snapDataException =
459                         new SnapDataException(ioException,
460                                 ERROR_WHILE_READING_STREAM_INTO_MESSAGE_BODY)
461                                 .withReason(String.format(READING_FROM_INPUT_VIEW_FAILED,
462                                         ioException.getMessage()))
463                                 .withResolution(JMS_RES_VERIFY_INPUT_DATA);
464                 errorViews.write(snapDataException, header);
465             } else if (cause instanceof SnapException) {
466                 errorViews.write(MessageHandler.getSnapDataException((SnapException) cause, header,
467                         documentUtility), header);
468             } else {
469                 throw new SnapDataException(cause, ERROR_WHILE_SENDING_MESSAGE)
470                         .withResolution(JMS_RES_VERIFY_INPUT_DATA);
471             }
472         } finally {
473             if (exceptionOccurred) {
474                 // there might be resource leak; so invoke cleanup() to be on safe side
475                 releaseResourcesAndCleanupCache();
476             }
477         }
478     }

492      * Try to get the transaction ID from the Message. Should be called only when the producer is
493      * using a transacted session. The method should return values only for Active MQ.
494      *
495      */
496     private String getTransactionID(Message message) {
497         try {
498             Class activeMQMessageClass = message.getClass().getSuperclass();
499             if (activeMQMessageClass != null && ACTIVEMQ_MESSAGE_CLASS_NAME.equals(activeMQMessageClass
500                     .getSimpleName())) {
501                 // Invoke ActiveMQMessage.getTransactionId().getTransactionKey() to get the
502                 // transaction Id for ActiveMQ. Using reflection here since the activemq jar is
503                 // provided by users.
504                 Object transactionID = activeMQMessageClass.getSuperclass()
505                         .getDeclaredMethod(GET_TRANSACTION_ID_METHOD_NAME)
506                         .invoke(message);
507                 return (String) transactionID.getClass().getDeclaredMethod(GET_TRANSACTION_KEY_METHOD_NAME)
508                         .invoke(transactionID);
509             }
510         } catch (IllegalAccessException | InvocationTargetException | NoSuchMethodException e) {
511             LOGGER.debug("Could not return the transaction Id from the message object. Reason:{}",
512                     e.getMessage());
513         }
514         return null;
515     }
