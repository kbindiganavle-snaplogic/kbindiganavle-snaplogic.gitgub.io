36         this.defaultTypeMappingHandler = defaultTypeMappingHandler;
37     }

39     @Override
40     public SnapTypeInfo resolve(final int sqlType, final String typeName) {
41         if (sqlType == Types.OTHER)
42             switch (typeName.toUpperCase()) {
43                 case TIMESTAMPTYPE:
44                     return DefaultTypeMappingHandler.DATE_TIME_TYPE_INFO;
45                 case UUIDTYPE:
46                 default :
47                     LOG.info("unrecognized data type: {}", typeName);
48                     return DefaultTypeMappingHandler.STRING_TYPE_INFO;
49             }
50         return defaultTypeMappingHandler.resolve(sqlType);
51     }
52
