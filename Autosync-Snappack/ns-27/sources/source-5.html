273      * @return integer value evaluated from pipeline parameter or input document,
274      *         or default value if expression is null or evaluated value is neither BigInteger
275      *         nor numeric string
276      */
277     public static Integer evalIntProperty(@Nullable ExpressionProperty expression,
278             @Nullable Document document, int defaultValue, int minValue) {
279         if (expression != null) {
280             Object value = expression.eval(document);
281             // if the evaluated value is BigInteger or numeric string then we convert into
282             // integer otherwise we will take default value
283             int returnValue;
284             if (value instanceof BigInteger) {
285                 returnValue = ((BigInteger) value).intValue();
286             } else if (value instanceof String && StringUtils.isNumeric((String) value)) {
287                 returnValue = Integer.parseInt((String) value);
288             } else {
289                 returnValue = defaultValue;
290             }
291             return Math.max(returnValue, minValue);
292         }
293         return defaultValue;

297      * Evaluates expression retry property to an integer.
298      * It may use pipeline parameters to evaluate the expression.
299      *
300      * @param expression   - expression property
301      * @param defaultValue - default value
302      * @param minValue     - minimum value
303      * @param maxValue     - maximum value
304      * @param label        - retry property label
305      *
306      * @return evaluated integer value, or default value if the value is null
307      * @throws ConfigurationException if it fails to convert to integer
308      */
309     public static Integer evalRetryProperty(@Nullable ExpressionProperty expression,
310             int defaultValue, int minValue, int maxValue, String label) {
311         String reason = null;
312         if (expression != null) {
313             try {
314                 Object value = expression.eval(null);
315                 // The evaluated value can be null if its Snap property is left blank, an empty
316                 // string if its pipeline parameter value is an empty string. Return default
317                 // in these cases.
318                 Long returnValue = null;
319                 switch (value == null ? NULL : value.getClass().getSimpleName()) {
320                     case BIG_INTEGER:
321                         returnValue = ((BigInteger) value).longValue();
322                         break;
323                     case BIG_DECIMAL:
324                         // eval(null) may return BigDecimal object if retry expression property
325                         // is like '1+2'
326                         returnValue = ((BigDecimal) value).longValue();
327                         break;
328                     case INTEGER:
329                         returnValue = Long.valueOf((Integer) value);
330                         break;
331                     case LONG:
332                         returnValue = (Long) value;
333                         break;
334                     case STRING:
335                         String valueStr = (String) value;
336                         if (StringUtils.isNumeric(valueStr)) {
337                             returnValue = Long.parseLong(valueStr);
338                         } else if (StringUtils.isBlank(valueStr)) {
339                             returnValue = (long) defaultValue;
340                         }
341                         break;
342                     case NULL:
343                         returnValue = (long) defaultValue;
344                         break;
345                     default:
346                         reason = format(REASON_INVALID_TYPE, value.getClass().getName());
347                 }
348                 if (returnValue != null) {
349                     if (returnValue < minValue) {
350                         reason = format(REASON_SMALLER_THAN_MIN_VALUE, returnValue,
351                                 minValue);
352                     } else if (returnValue > maxValue) {
353                         reason = format(REASON_LARGER_THAN_MAX_VALUE, returnValue,
354                                 maxValue);
355                     } else {
356                         return returnValue.intValue();
357                     }
358                 }
359             } catch (Exception e) {
360                 throw new ConfigurationException(e, ERR_GET_VALID_VALUE)
361                         .formatWith(label)
362                         .withReason(e.getMessage())
363                         .withResolution(RESOLUTION_ADDRESS_ISSUE);
364             }
365         }
366         if (StringUtils.isBlank(reason)) {
367             return defaultValue;
368         } else {
369             throw new ConfigurationException(ERR_INVALID_VALUE_FOR)
370                     .formatWith(label)
371                     .withReason(reason)
372                     .withResolution(RESOLUTION_ADDRESS_ISSUE);
373         }
374     }

543      * Converts a standard POSIX Shell globbing pattern into a regular expression
544      * pattern. The result can be used with the standard {@link java.util.regex} API to
545      * recognize strings which match the glob pattern.
546      *
547      * This code is copied from: https://stackoverflow.com/questions/1247772/
548      * is-there-an-equivalent-of-java-util-regex-for-glob-type-patterns
549      *
550      * @param glob - glob pattern
551      *
552      * @return regex pattern to recognize the given glob pattern.
553      */
554     public static final String getRegexFrom(String glob) {
555         StringBuilder sb = new StringBuilder(glob.length());
556         int inGroup = 0;
557         int inClass = 0;
558         int firstIndexInClass = -1;
559         char[] arr = glob.toCharArray();
560         for (int i = 0; i < arr.length; i++) {
561             char ch = arr[i];
562             switch (ch) {
563                 case '\\':
564                     if (++i >= arr.length) {
565                         sb.append('\\');
566                     } else {
567                         char next = arr[i];
568                         switch (next) {
569                             case ',':
570                                 // escape not needed
571                                 break;
572                             case 'Q':
573                             case 'E':
574                                 // extra escape needed
575                                 sb.append('\\');
576                             default:
577                                 sb.append('\\');
578                         }
579                         sb.append(next);
580                     }
581                     break;
582                 case '*':
583                     if (inClass == 0) {
584                         sb.append(".*");
585                     } else {
586                         sb.append('*');
587                     }
588                     break;
589                 case '?':
590                     sb.append(inClass == 0 ? '.' : '?');
591                     break;
592                 case '[':
593                     inClass++;
594                     firstIndexInClass = i + 1;
595                     sb.append('[');
596                     break;
597                 case ']':
598                     inClass--;
599                     sb.append(']');
600                     break;
601                 case '.':
602                 case '(':
603                 case ')':
604                 case '+':
605                 case '|':
606                 case '^':
607                 case '$':
608                 case '@':
609                 case '%':
610                     if (inClass == 0 || (firstIndexInClass == i && ch == '^')) {
611                         sb.append('\\');
612                     }
613                     sb.append(ch);
614                     break;
615                 case '!':
616                     sb.append(firstIndexInClass == i ? '^' : '!');
617                     break;
618                 case '{':
619                     inGroup++;
620                     sb.append('(');
621                     break;
622                 case '}':
623                     inGroup--;
624                     sb.append(')');
625                     break;
626                 case ',':
627                     sb.append(inGroup > 0 ? '|' : ',');
628                     break;
629                 default:
630                     sb.append(ch);
631             }
632         }
633         return sb.toString();
634     }

780         int allSemicolons = StringUtils.countMatches(query, QUERY_SEPARATOR);
781         Matcher quotes = semicolonInQuotes.matcher(query);

1023      * @param statusMessage              - StatusMessage object

1025      * @param totalDocumentCount         - total input-document count, set 0 if not available
1026      * @param file                       - File object to be used to get the usable disk space
1027      * @param bytesPerDocument           - average input document size, set 0 if not available
1028      * @return true if it has been paused for a while and resumed
1029      */
1030     public static boolean checkMemoryAndDisk(long minMemory, long minFreeDiskSpace,
1031             long outOfResourceTimeoutMillis, StatusReporter statusReporter,
1032             StatusMessage statusMessage, Logger logger, int totalDocumentCount, File file,
1033             long bytesPerDocument) {
1034         int intervalSeconds = 1;
1035         long startTime = System.currentTimeMillis();
1036         boolean paused = false;
1037         file = file == null ? new File(ROOT) : file;
1038         long availableMemory = getAvailableMemory();
1039         long freeDiskSpace = file.getUsableSpace();
1040         String availableMemoryStr = byteCountDisplay(availableMemory);
1041         String freeDiskSpaceStr = byteCountDisplay(freeDiskSpace);
1042         while (availableMemory < minMemory || freeDiskSpace < minFreeDiskSpace) {
1043             if (!paused) {
1044                 paused = true;
1045                 logger.warn("Paused processing input documents, waiting for more resource, " +
1046                                 "available memory: {}, free disk space: {}", availableMemoryStr,
1047                         freeDiskSpaceStr);
1048             }
1049             if (bytesPerDocument > 0) {
1050                 statusReporter.notify(statusMessage, STATUS_PAUSED, availableMemoryStr,
1051                         freeDiskSpaceStr, byteCountDisplay(bytesPerDocument));
1052             } else {
1053                 statusReporter.notify(statusMessage, STATUS_PAUSED, availableMemoryStr,
1054                         freeDiskSpaceStr);
1055             }
1056             sleepNoInterrupt(intervalSeconds, TimeUnit.SECONDS, logger);
1057             if (System.currentTimeMillis() - startTime > outOfResourceTimeoutMillis) {
1058                 SnapDataException ex;
1059                 if (availableMemory < minMemory) {
1060                     if (totalDocumentCount > 0) {
1061                         ex = new SnapDataException(ERR_MEMORY_BELOW_THRESHOLD)
1062                                 .formatWith(availableMemoryStr, byteCountDisplay(minMemory),
1063                                         totalDocumentCount, byteCountDisplay(RUNTIME.maxMemory()));
1064                     } else {
1065                         ex = new SnapDataException(ERR_MEMORY_BELOW_THRESHOLD_0)
1066                                 .formatWith(availableMemoryStr, byteCountDisplay(minMemory),
1067                                         byteCountDisplay(RUNTIME.maxMemory()));
1068                     }
1069                     throw ex.withReason(REASON_MEMORY_PRESSURE_TOO_HIGH)
1070                             .withResolution(RESOLUTION_MEMORY_BELOW_THRESHOLD);
1071                 } else {
1072                     if (totalDocumentCount > 0) {
1073                         ex = new SnapDataException(ERR_FREE_DISK_SPACE_BELOW_THRESHOLD)
1074                                 .formatWith(freeDiskSpaceStr, byteCountDisplay(minFreeDiskSpace),
1075                                         totalDocumentCount);
1076                     } else {
1077                         ex = new SnapDataException(ERR_FREE_DISK_SPACE_BELOW_THRESHOLD_0)
1078                                 .formatWith(freeDiskSpaceStr, byteCountDisplay(minFreeDiskSpace));
1079                     }
1080                     throw ex.withReason(REASON_FREE_DISK_SPACE_LOWER)
1081                             .withResolution(RESOLUTION_FREE_DISK_SPACE_BELOW_THRESHOLD);
1082                 }
1083             }
1084             intervalSeconds = min(intervalSeconds * 2, MAX_INTERVAL_SECONDS);
1085             availableMemory = getAvailableMemory();
1086             freeDiskSpace = file.getUsableSpace();
1087             availableMemoryStr = byteCountDisplay(availableMemory);
1088             freeDiskSpaceStr = byteCountDisplay(freeDiskSpace);
1089         }

1091             logger.warn("Resumed to process input documents, available memory: {}," +
1092                     "free disk space: {}", availableMemoryStr, freeDiskSpaceStr);
1093         }
1094         if (bytesPerDocument > 0) {
1095             statusReporter.notify(statusMessage, STATUS_AVAILABLE_MEMORY, availableMemoryStr,
1096                     freeDiskSpaceStr, byteCountDisplay(bytesPerDocument));
1097         } else {
1098             statusReporter.notify(statusMessage, STATUS_AVAILABLE_MEMORY, availableMemoryStr,
1099                     freeDiskSpaceStr);
1100         }
1101         return paused;
1102     }
