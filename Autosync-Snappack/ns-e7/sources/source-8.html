125      * Process the worksheet data along with the grid data to preserve data types
126      * and write to output view
127      * @param document
128      */
129     private void processDataWithGridInfo(Document document) {
130         try {
131             spreadsheetId = getSpreadsheetId(googleDrive, spreadsheetNameOrId);
132             List<GridData> values;
133             try {
134                 @SuppressWarnings("unchecked")
135                 RetryPolicy retryPolicy = new RetryPolicy()
136                         .withMaxRetries(numOfRetries)
137                         .withDelay(retryInterval, TimeUnit.SECONDS)
138                         .abortOn(SuggestViewAbortException.class, SnapDataException.class);
139                 values = Failsafe.with(retryPolicy)
140                         .onFailedAttempt(failure -> LOG.warn("Failed to read worksheet: {} in " +
141                                         "spreadsheet: {}, {}", worksheetName, spreadsheetNameOrId,
142                                 failure.getMessage()))
143                         .get(context -> {
144                             logRetryAttempt(context, LOG);
145                             return readWorksheet(spreadsheetId, worksheetName);
146                         });
147             } catch (SnapDataException sde){
148                 throw sde;
149             } catch (Throwable t) {
150                 Throwable t2 = t instanceof FailsafeException ? t.getCause() : t;
151                 throw getErrorReadWorksheet(t2, worksheetName, t2.getMessage());
152             }
153             if (CollectionUtils.isEmpty(values)) {
154                 outputViews.write(documentUtility.newDocumentFor(document, new JSONArray()));
155                 return;
156             }
157             List<RowData> allRows = values.get(0).getRowData();
158             int dataRowIndex = 0;
159             // initialize header
160             int maxColumns = 0;
161             List<String> header;
162             // Get the max row size
163             for (RowData row : allRows){
164                 if (!MapUtils.isEmpty(row)) {
165                     maxColumns = Math.max(maxColumns, row.getValues().size());
166                 }
167             }
168             if (headerExists) {
169                 // first row is header so the data starts at index 1
170                 dataRowIndex = 1;
171                 RowData headerRow = allRows.get(0);
172                 /* HeaderRow size could be < maxColumns in case the
173                  * last column header is null
174                  */
175                 if (MapUtils.isEmpty(headerRow) || headerRow.getValues().size() < maxColumns) {
176                     throw new SnapDataException(ERR_HEADER_COLUMN_EMPTY_MSG)
177                             .withReason(ERR_HEADER_COLUMN_EMPTY_REASON)
178                             .withResolution(ERR_HEADER_COLUMN_EMPTY_RESOLUTION);
179                 }
180                 List<CellData> headerColumns = headerRow.getValues();
181                 header = new ArrayList<>(headerColumns.size());
182                 for (CellData headerColumn : headerColumns) {
183                     String headerName = headerColumn.getFormattedValue();
184                     if (StringUtils.isBlank(headerName)) {
185                         throw new SnapDataException(ERR_HEADER_COLUMN_EMPTY_MSG)
186                                 .withReason(ERR_HEADER_COLUMN_EMPTY_REASON)
187                                 .withResolution(ERR_HEADER_COLUMN_EMPTY_RESOLUTION);
188                     }
189                     if (header.contains(headerName)) {
190                         throw new SnapDataException(ERR_HEADER_COLUMN_REPEATED_MSG)
191                                 .withReason(String.format(ERR_HEADER_COLUMN_REPEATED_REASON,
192                                         headerName))
193                                 .withResolution(ERR_HEADER_COLUMN_REPEATED_RESOLUTION);
194                     }
195                     header.add(headerName);
196                 }
197                 if (CollectionUtils.isEmpty(header)) {
198                     throw new SnapDataException(ERR_WORKSHEET_HEADER_NOT_FOUND)
199                             .formatWith(worksheetName)
200                             .withReason(REASON_EMPTY_WORKSHEET)
201                             .withResolution(RESOLUTION_ADDRESS_ISSUE);
202                 }
203             } else {
204                 header = new ArrayList<>(maxColumns);
205                 for (int i = 1; i <= maxColumns; i++) {
206                     header.add(String.format(FIELD, i));
207                 }
208             }
209             // parse worksheet data
210             int dataRowCount = allRows.size();
211             while (dataRowIndex < dataRowCount) {
212                 Map<Object, Object> record = getRecord(allRows.get(dataRowIndex++).getValues(),
213                         header);
214                 if (displayEmptyRows || !isEmptyRow(record)) {
215                     outputViews.write(documentUtility.newDocumentFor(document, record));
216                 }
217             }
218         } catch (SnapDataException e) {
219             throw e;
220         } catch (SuggestViewAbortException e) {
221             throw e;
222         } catch (Exception e) {
223             errorViews.write(getErrorReadWorksheet(e, worksheetName, e.getMessage()), document);
224         }
225     }

229      * and write to output view. All cell values are returned as formatted
230      * in the spreadsheet.
231      * @param document
232      */
233     private void processDataNoGridInfo(Document document) {
234         try {
235             spreadsheetId = getSpreadsheetId(googleDrive, spreadsheetNameOrId);
236             List<ValueRange> valueRangesList =
237                     readWorksheetNoGridInfo(spreadsheetId, worksheetName);
238             if (CollectionUtils.isEmpty(valueRangesList)) {
239                 outputViews.write(documentUtility.newDocumentFor(document, new JSONArray()));
240                 return;
241             }
242             List<String> header = null;
243             int maxColumns = 0;
244             // Iterate through rows in all value-ranges to find the row with max columns
245             // to set/validate the header
246             for (ValueRange valueRange : valueRangesList) {
247                 List<List<Object>> valuesInRange = valueRange.getValues();
248                 if (CollectionUtils.isNotEmpty(valuesInRange)) {
249                     for (List<Object> values : valuesInRange) {
250                         if (CollectionUtils.isNotEmpty(values)) {
251                             maxColumns = Math.max(maxColumns, values.size());
252                         }
253                     }
254                 }
255             }
256 
257             if (headerExists) {
258                 // first row is header
259                 List<Object> headerRow = valueRangesList.get(0).getValues().get(0);
260                 /* HeaderRow size could be < maxColumns in case the
261                  * last column header is null. Throw error in such case.
262                  */
263                 if (CollectionUtils.isEmpty(headerRow) || headerRow.size() < maxColumns) {
264                     throw new SnapDataException(ERR_HEADER_COLUMN_EMPTY_MSG)
265                             .withReason(ERR_HEADER_COLUMN_EMPTY_REASON)
266                             .withResolution(ERR_HEADER_COLUMN_EMPTY_RESOLUTION);
267                 }
268                 header = new ArrayList<>(headerRow.size());
269                 for (Object headerColumn : headerRow) {
270                     String headerName = headerColumn.toString();
271                     if (StringUtils.isBlank(headerName)) {
272                         throw new SnapDataException(ERR_HEADER_COLUMN_EMPTY_MSG)
273                                 .withReason(ERR_HEADER_COLUMN_EMPTY_REASON)
274                                 .withResolution(ERR_HEADER_COLUMN_EMPTY_RESOLUTION);
275                     }
276                     if (header.contains(headerName)) {
277                         throw new SnapDataException(ERR_HEADER_COLUMN_REPEATED_MSG)
278                                 .withReason(String.format(ERR_HEADER_COLUMN_REPEATED_REASON,
279                                         headerName))
280                                 .withResolution(ERR_HEADER_COLUMN_REPEATED_RESOLUTION);
281                     }
282                     header.add(headerName);
283                 }
284                 if (CollectionUtils.isEmpty(header)) {
285                     throw new SnapDataException(ERR_WORKSHEET_HEADER_NOT_FOUND)
286                             .formatWith(worksheetName)
287                             .withReason(REASON_EMPTY_WORKSHEET)
288                             .withResolution(RESOLUTION_ADDRESS_ISSUE);
289                 }
290             } else {
291                 header = new ArrayList<>(maxColumns);
292                 for (int i = 1; i <= maxColumns; i++) {
293                     header.add(String.format(FIELD, i));
294                 }
295             }
296 
297             // parse worksheet data
298             for (int listItr = 0; listItr < valueRangesList.size(); listItr++) {
299                 List<List<Object>> valuesInRange = valueRangesList.get(listItr).getValues();
300                 // valuesInRange is null if an entire batch is empty. Skip all empty rows
301                 if (valuesInRange == null) {
302                     continue;
303                 }
304                 ListIterator<List<Object>> rowIterator = valuesInRange.listIterator();
305                 if (headerExists && listItr == 0) {
306                     rowIterator.next();
307                 }
308                 while (rowIterator.hasNext()) {
309                     List<Object> row = rowIterator.next();
310                     Map<Object, Object> record = new LinkedHashMap<>();
311                     Iterator<Object> cellIterator = row.iterator();
312                     for (Object key : header) {
313                         Object value = cellIterator.hasNext() ?
314                                 cellIterator.next() : StringUtils.EMPTY;
315                         if (!(value instanceof String)) {
316                             value = String.valueOf(value);
317                         }
318                         record.put(key, value);
319                     }
320                     if (displayEmptyRows || !isEmptyRow(record)) {
321                         outputViews.write(documentUtility.newDocumentFor(document, record));
322                     }
323                 }
324             }
325         } catch (SnapDataException e) {
326             throw e;
327         } catch (SuggestViewAbortException e) {
328             throw e;
329         } catch (Exception e) {
330             errorViews.write(getErrorReadWorksheet(e, worksheetName, e.getMessage()), document);
331         }
332     }

352      * Read worksheet data in batches specified by ranges, with no additional
353      * cell metadata. All data returned are as formatted in the spreadsheet.
354      * @param spreadsheetId
355      * @param worksheetName
356      * @return list of values in ranges of size = READ_BATCH_SIZE
357      */
358     protected List<ValueRange> readWorksheetNoGridInfo(String spreadsheetId, String worksheetName) {
359         GridProperties gp = getSheetGridProperties(spreadsheetId, worksheetName);
360         if (gp == null) {
361             throw new SnapDataException(ERR_READ_WORKSHEET)
362                     .formatWith(worksheetName, spreadsheetNameOrId)
363                     .withReason(ERROR_FETCHING_WORKSHEET_ROW_COL_INFO)
364                     .withResolution(RESOLUTION_VERIFY_WORKSHEET_EXISTS);
365         }
366         int rowCount = gp.getRowCount();
367         List<String> ranges = Lists.newArrayList();
368         int startRow = 1;
369         // Set up ranges to be read
370         while (startRow <= rowCount) {
371             String range = String.format(WORKSHEET_BATCH_GET_RANGE_FORMAT, worksheetName,
372                     FIRST_COLUMN, startRow, LAST_COLUMN, startRow + READ_BATCH_SIZE - 1);
373             ranges.add(range);
374             startRow = startRow + READ_BATCH_SIZE;
375         }
376         try {
377             @SuppressWarnings("unchecked")
378             RetryPolicy retryPolicy = new RetryPolicy()
379                     .withMaxRetries(numOfRetries)
380                     .withDelay(retryInterval, TimeUnit.SECONDS)
381                     .abortOn(SuggestViewAbortException.class, SnapDataException.class);
382             return Failsafe.with(retryPolicy)
383                     .onFailedAttempt(failure -> LOG.warn("Failed to read worksheet: {} in " +
384                             "spreadsheet: {}, {}", worksheetName, spreadsheetNameOrId,
385                             failure.getMessage()))
386                     .get(context -> {
387                        logRetryAttempt(context, LOG);
388                        return readWorksheetNoGridInfo(spreadsheetId, ranges);
389                     });
390         } catch (Throwable t) {
391             Throwable t2 = t instanceof FailsafeException ? t.getCause() : t;
392             throw getErrorReadWorksheet(t2, worksheetName, t2.getMessage());
393         }
394     }

396     protected List<ValueRange> readWorksheetNoGridInfo(String spreadsheetId, List<String> ranges)
397             throws IOException {
398         try {
399             Sheets.Spreadsheets.Values.BatchGet batchGetRequest = sheets
400                     .spreadsheets()
401                     .values()
402                     .batchGet(spreadsheetId);
403             batchGetRequest.setRanges(ranges);
404             batchGetRequest.setValueRenderOption(VALUE_RENDER_OPTION);
405             return batchGetRequest.execute().getValueRanges();
406         } catch (GoogleJsonResponseException e) {
407             String errMessage = e.getMessage();
408             if (StringUtils.startsWithIgnoreCase(errMessage, TOO_MANY_REQUESTS)) {
409                 // if number of requests exceeded the hourly limit, wait longer and retry
410                 sleepNoInterrupt(SLEEP_SECONDS_ERR_429, TimeUnit.SECONDS, LOG);
411             } else if (StringUtils.containsIgnoreCase(errMessage, INVALID_RANGE)) {
412                 // The error "Invalid range" occurs if the worksheet is a chart or an image.
413                 String sheetType = getSheetType(spreadsheetId, worksheetName);
414                 if (sheetType != null && !GRID.equalsIgnoreCase(sheetType)) {
415                     // Worksheet Reader can read 'GRID' type only
416                     throw getErrorReadWorksheet(e, worksheetName, String.format(
417                             REASON_UNSUPPORTED_WORKSHEET_TYPE, sheetType));
418                 }
419             }
420             throw e;
421         }
422     }
