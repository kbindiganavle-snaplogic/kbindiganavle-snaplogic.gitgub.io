416      * Gets the spreadsheet ID from its name by using Google Drive API.
417      *
418      * @param drive - Google Drive
419      * @param name - spreadsheet name
420      *
421      * @return spreadsheet ID (also called 'key')
422      */
423     protected String getId(Drive drive, String name, boolean nullIfNotExists) {
424         FileList fileList;
425         try {
426             String query = String.format(FORMAT_NAME, escapeSheetName(name));
427             fileList = Failsafe.with(getRetryPolicy(numOfRetries, retryInterval))
428                     .onFailedAttempt(e -> LOG.warn("Failed to get spreadsheet file list, {}",
429                             e.getMessage()))
430                     .get(context -> {
431                         logRetryAttempt(context, LOG);
432                         if (keyIncludeSharedDrive) {
433                             return drive.files().list().setFields(FILES_ID_NAME)
434                                     .setQ(query)
435                                     .setSupportsAllDrives(true)
436                                     .setIncludeItemsFromAllDrives(true)
437                                     .execute();
438                         } else {
439                             return drive.files().list().setFields(FILES_ID_NAME)
440                                     .setQ(query)
441                                     .execute();
442                         }
443                     });
444         } catch (Throwable t) {
445             Throwable t2 =  t instanceof FailsafeException ? t.getCause() : t;
446             throw new SnapDataException(t2, ERR_SPREADSHEET_NOT_FOUND)
447                     .formatWith(name)
448                     .withReason(REASON_SPREADSHEET_NOT_FOUND)
449                     .withResolution(RESOLUTION_ADDRESS_ISSUE);
450         }
451         List<String> idList = new ArrayList<>();
452         if (fileList != null) {
453             for (File file : fileList.getFiles()) {
454                 if (StringUtils.equals(file.getName(), name)) {
455                     String id = file.getId();
456                     if (StringUtils.isNotBlank(id)) {
457                         idList.add(id);
458                     } else {
459                         throw new SnapDataException(ERR_UNEXPECTED_SEARCH_RESULT)
460                                 .formatWith(name)
461                                 .withReason(REASON_EMPTY_ID)
462                                 .withResolutionAsDefect();
463                     }
464                 }
465             }
466         }
467         switch (idList.size()) {
468             case 0:
469                 if (nullIfNotExists) {
470                     return null;
471                 }
472                 throw new SnapDataException(ERR_SPREADSHEET_NOT_FOUND)
473                         .formatWith(name)
474                         .withReason(REASON_SPREADSHEET_NOT_FOUND)
475                         .withResolution(RESOLUTION_SPREADSHEET_NOT_FOUND);
476             case 1:
477                 return idList.get(0);
478             default:
479                 throw new SnapDataException(ERR_UNEXPECTED_SEARCH_RESULT)
480                         .formatWith(name)
481                         .withReason(REASON_TOO_MANY)
482                         .withResolution(RESOLUTION_DUPLICATE_SPREADSHEET);
483         }
484     }

487     /**
488      * Checking the given KEY_SPREADSHEET property is ID or Name And
489      * Gets the spreadsheet ID
490      *
491      * @param drive - Google Drive
492      * @param spreadsheetNameOrId - spreadsheet name/id
493      *
494      * @return spreadsheet ID (also called 'key')
495      */
496     protected String getSpreadsheetId(Drive drive, String spreadsheetNameOrId, boolean nullIfNotExists){
497         FileList fileList = getSpreadSheetList(drive);
498         Set<String> setOfSpreadsheetNames = new TreeSet<>();
499         Set<String> setOfSpreadsheetIds = new TreeSet<>();
500         Set<String> spreadsheetNameAndId = new TreeSet<>();
501         for (File file : fileList.getFiles()) {
502             setOfSpreadsheetNames.add(file.getName());
503             setOfSpreadsheetIds.add(file.getId());
504             spreadsheetNameAndId.add(file.getName() + " : " + file.getId());
505         }
506         if (setOfSpreadsheetNames.contains(spreadsheetNameOrId)) {
507             spreadsheetId = getId(drive, spreadsheetNameOrId, nullIfNotExists);
508         } else if (setOfSpreadsheetIds.contains(spreadsheetNameOrId)) {
509             spreadsheetId = spreadsheetNameOrId;
510         } else if (spreadsheetNameAndId.contains(spreadsheetNameOrId)) {
511             // Splitting the given spreadsheetName by separating spreadsheetId
512             String[] spreadsheets = spreadsheetNameOrId.split("\\s:\\s");
513             spreadsheetId = spreadsheets[spreadsheets.length-1];
514         } else {
515             if (nullIfNotExists) {
516                 // For Worksheet Writer
517                 spreadsheetId = null;
518             } else {
519                 //For Worksheet Reader
520                 spreadsheetId = getId(drive, spreadsheetNameOrId);
521             }
522         }
523         return spreadsheetId;
524     }

737     protected Map<Object, Object> getRecord(List<CellData> dataColumns, List<String> header) {
738         CellData dataColumn;
739         CellFormat cellFormat;
740         ExtendedValue cellExtendedValue;
741         int columnIndex = 0;
742         boolean writeValueAsDate = false;
743         Map<Object, Object> record = new LinkedHashMap<>();
744 
745         for (Object key : header) {
746             // If a row is empty, dataColumns List will be null
747             if (dataColumns != null) {
748                 /* This check is to avoid failing in case the last column
749                  * has no data - SWAT-1727.
750                  * In such a case header.size() > dataColumns.size()
751                  * */
752                 if (columnIndex < dataColumns.size()) {
753                     dataColumn = dataColumns.get(columnIndex++);
754                 } else {
755                     dataColumn = null;
756                 }
757                 if (dataColumn != null) {
758                     cellFormat = dataColumn.getEffectiveFormat();
759                     cellExtendedValue = dataColumn.getEffectiveValue();
760                 } else {
761                     cellFormat = null;
762                     cellExtendedValue = null;
763                 }
764                 if (cellFormat != null) {
765                     NumberFormat numberFormat = cellFormat.getNumberFormat();
766                     if (numberFormat != null) {
767                         String type = numberFormat.getType();
768                         if (TYPE_DATE.equalsIgnoreCase(type)) {
769                             writeValueAsDate = true;
770                         }
771                     }
772                 }
773             } else {
774                 cellExtendedValue = null;
775             }
776 
777             record.put(key, parseExtendedValue(cellExtendedValue,
778                     writeValueAsDate));
779             writeValueAsDate = false;
780         }
781         return record;
782     }

784     /**
785      * Parses the given {@link ExtendedValue} to get the actual value
786      *
787      * @param extendedValue
788      * @param writeValueAsDate
789      *
790      * @return actual value with actual type
791      */
792     private Object parseExtendedValue(ExtendedValue extendedValue, boolean writeValueAsDate) {
793         if (extendedValue == null) {
794             return null;
795         }
796 
797         Object finalValue;
798         // attempt boolean value first
799         finalValue = extendedValue.getBoolValue();
800         if (finalValue == null) {
801             // attempt number value next
802             finalValue = extendedValue.getNumberValue();
803             if (finalValue == null) {
804                 // attempt string value next
805                 finalValue = extendedValue.getStringValue();
806                 if (finalValue == null) {
807                     // attempt formula value next
808                     finalValue = extendedValue.getFormulaValue();
809                     if (finalValue == null) {
810                         // attempt error value finally
811                         finalValue = extendedValue.getErrorValue().getType();
812                     }
813                 }
814             } else {
815                 // Date value is represented as number by Google Spreadsheet API
816                 // So if a numeric value is of type date then its converted into
817                 //   Joda DateTime object
818                 if (writeValueAsDate) {
819                     finalValue = new LocalDate(new DateTime(DateUtil.getJavaDate(
820                             (Double) finalValue)));
821                 }
822             }
823         }
824         return finalValue;
825     }

828             throws IOException {
829         try {
830             Sheets.Spreadsheets.Get getRequest = sheets
831                     .spreadsheets()
832                     .get(spreadsheetId)
833                     .setIncludeGridData(true)
834                     .setRanges(Arrays.asList(String.format(DEFAULT_RANGE, worksheetName)));
835             return getRequest.execute().getSheets().get(0).getData();
836         } catch (GoogleJsonResponseException e) {
837             // The error "503 Service Unavailable ..." or "429 Too Many Requests ..." occurs
838             // intermittently. Sleep n seconds, log and retry.
839             String errMessage = e.getMessage();
840             if (StringUtils.startsWithIgnoreCase(errMessage, TOO_MANY_REQUESTS)) {
841                 // if number of requests exceeded the hourly limit, wait longer and retry
842                 sleepNoInterrupt(SLEEP_SECONDS_ERR_429, TimeUnit.SECONDS, LOG);
843             } else if (StringUtils.containsIgnoreCase(errMessage, INVALID_RANGE)) {
844                 // The error "Invalid range" occurs if the worksheet is a chart or an image.
845                 String sheetType = getSheetType(spreadsheetId, this.worksheetName);
846                 if (sheetType != null && !GRID.equalsIgnoreCase(sheetType)) {
847                     // Worksheet Reader can read 'GRID' type only
848                     throw getErrorReadWorksheet(e, worksheetName, String.format(
849                             REASON_UNSUPPORTED_WORKSHEET_TYPE, sheetType));
850                 }
851             } else if (errMessage.contains("Unable to parse range")) {
852                 throw new SnapDataException(ERR_READ_WORKSHEET)
853                         .formatWith(worksheetName, spreadsheetNameOrId)
854                         .withReason(ERROR_FETCHING_WORKSHEET_ROW_COL_INFO)
855                         .withResolution(RESOLUTION_VERIFY_WORKSHEET_EXISTS);
856             }
857             throw e;
858         }
