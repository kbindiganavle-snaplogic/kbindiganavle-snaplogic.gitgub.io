420         props.put(ConsumerConfig.GROUP_ID_CONFIG, group);

423         props.put(ConsumerConfig.AUTO_OFFSET_RESET_CONFIG, autoOffsetReset.toLowerCase());

460             throw new SnapDataException(MULTIPLE_INPUT_DOCUMENTS_ERROR)
461                     .withReason(MULTIPLE_INPUT_DOCUMENTS_ERROR_REASON)
462                     .withResolution(MULTIPLE_INPUT_DOCUMENTS_ERROR_RESOLUTION);
463         }
464         processCalledOnce = true;
465 
466         initPropsFromDoc(inputDoc);
467         if (inputDoc != null) {
468             inputDoc.acknowledge();
469         }
470         Document passThroughDoc = isPassThrough ? inputDoc : null;
471 
472         initializeConsumer();
473         try {
474             prepareForPolling();
475 
476             int msgCount = 0;
477             long maxRate = 0;
478             int batchIndex = 0;
479 
480             boolean mustCommitOffsetsForAssignedPartitions = SEEK_END_LEGACY.equals(seekType) &&
481                     AUTO_OFFSET_RESET_LATEST.equals(autoOffsetReset);
482 
483             final boolean stopWhenEmpty = (maxMsgCount == 0) || (maxMsgCount > 0 && !waitForCount);
484             boolean lastPollMetStopCriteria = false;
485             while (!done) {
486                 Set<TopicPartition> assignmentBeforePoll = consumer.assignment();
487                 LOG.debug("{}: polling from {}", clientId, assignmentBeforePoll);
488                 ConsumerRecords<byte[], byte[]> allRecs = consumer.poll(maxWaitingTime);
489 
490                 if (mustCommitOffsetsForAssignedPartitions) {
491                     Set<TopicPartition> assignment = consumer.assignment();
492                     if (!assignment.isEmpty()) {
493                         // this only needs to happen once
494                         mustCommitOffsetsForAssignedPartitions = false;
495                         commitOffsetsForAssignedPartitionsNotCommittedTo(assignment, allRecs, inputDoc);
496                     }
497                 }
498 
499                 final int batchSize = allRecs.count();
500                 if (batchSize == 0) {
501                     LOG.trace("{}: poll returned no records", clientId);
502                     // When poll returns nothing, we usually just want to continue polling.
503                     // But "Message Count == 0" means we want to stop when there's no more data
504                     // currently available. This is tricky because sometimes poll returns nothing
505                     // even though there's more data available. To avoid premature stopping, we
506                     // check that ALL of the following conditions occur on TWO POLLS IN A ROW:
507                     // 1. The partition assignment was non-empty before the poll.
508                     // 2. The partition assignment was non-empty after the poll.
509                     // 3. The "Message Count" setting's value (maxMsgCount) is 0,
510                     //    or is > 0 and "Wait for count" setting is false.
511                     if (!assignmentBeforePoll.isEmpty() && stopWhenEmpty && !consumer.assignment().isEmpty()) {
512                         if (!lastPollMetStopCriteria) {
513                             LOG.debug("{}: stop criteria occurred once; waiting " +
514                                     "for this to happen twice in a row before stopping", clientId);
515                             lastPollMetStopCriteria = true;
516                         } else {
517                             LOG.debug("{}: stop criteria occurred again; stopping", clientId);
518                             done = true;
519                         }
520                     } else if (lastPollMetStopCriteria) {
521                         LOG.debug("{}: clearing lastPollMetStopCriteria flag", clientId);
522                         lastPollMetStopCriteria = false;
523                     }
524                     continue;
525                 } else if (lastPollMetStopCriteria) {
526                     // poll returned nothing last time, but just returned something
527                     LOG.debug("{}: poll returned something, clearing lastPollMetStopCriteria flag", clientId);
528                     lastPollMetStopCriteria = false;
529                 }
530 
531                 int recIndex = 0;
532                 long startTime = System.nanoTime();
533                 Set<TopicPartition> partitions = allRecs.partitions();
534                 LOG.debug("{}: read {} records from {} partition(s); batchIndex={} _______________",
535                         clientId, batchSize, partitions.size(), batchIndex);
536 
537                 boolean waitForEachRecord = !isAutoCommit && ACK_MODE_RECORD.equals(ackMode);
538                 boolean oneOutputPerRecord = OUTPUT_MODE_RECORD.equals(outputMode);
539                 boolean abortBatch = false;
540 
541                 // batchSize is the number of records returned from the last poll of Kafka,
542                 // but we might need to output a smaller value for batch_size in the output
543                 // depending on maxMsgCount or maxSuggestValue since those limit the actual
544                 // number of output records.
545                 int adjustedBatchSize = batchSize;
546                 if (maxMsgCount > 0) {
547                     adjustedBatchSize = Math.min(adjustedBatchSize, maxMsgCount - msgCount);
548                 }
549                 if (isValidation()) {
550                     adjustedBatchSize = Math.min(adjustedBatchSize, maxSuggestValue - msgCount);
551                 }
552                 if (adjustedBatchSize != batchSize) {
553                     LOG.debug("{}: adjustedBatchSize={}", clientId, adjustedBatchSize);
554                 }
555 
556                 List<Map<String, Object>> batch = oneOutputPerRecord ? null : new ArrayList<>(adjustedBatchSize);
557 
558                 for (Iterator<TopicPartition> iParts = partitions.iterator(); iParts.hasNext() && !abortBatch && !done;) {
559                     TopicPartition tp = iParts.next();
560                     List<ConsumerRecord<byte[], byte[]>> recs = allRecs.records(tp);
561                     int recsCount = recs.size();
562 
563                     Map<TopicPartition, OffsetAndMetadata> recCommitMap = null;
564                     try {
565                         LOG.debug("{}: processing {} records from {}, offsets {}-{}", clientId,
566                                 recsCount, tp, recs.get(0).offset(), recs.get(recsCount - 1).offset());
567                         for (Iterator<ConsumerRecord<byte[], byte[]>> iRecs = recs.iterator(); iRecs.hasNext() && !abortBatch && !done; ) {
568                             ConsumerRecord<byte[], byte[]> rec = iRecs.next();
569                             if (tracker == null) {
570                                 // create and register a tracker only when needed
571                                 tracker = trackerFactory.create();
572                                 trackerIndex = trackerRegistry.put(clientId, tracker);
573                                 LOG.debug("{}: registered new tracker with trackerIndex={}", clientId, trackerIndex);
574                             }
575                             Map<String, Object> outputMap = createOutputMap(rec, adjustedBatchSize, batchIndex,
576                                     recIndex++, oneOutputPerRecord, inputDoc);
577                             if (outputMap != null) {
578                                 if (isAutoCommit) {
579                                     tracker.autoTrack(tp, rec.offset());
580                                 } else {
581                                     tracker.track(tp, rec.offset());
582                                 }
583                                 if (oneOutputPerRecord) {
584                                     Document outputDoc = documentUtility.newDocument(outputMap);
585                                     outputViews.write(outputDoc, passThroughDoc);
586                                 } else {
587                                     batch.add(outputMap);
588                                 }
589                             } else { // outputMap == null
590                                 // there's no output for this record, so autoTrack it
591                                 tracker.autoTrack(tp, rec.offset());
592                             }
593                             msgCount++;
594                             if (maxMsgCount > 0 && msgCount >= maxMsgCount) {
595                                 LOG.debug("{}: stopping because msgCount reached maxMsgCount", clientId);
596                                 done = true;
597                             } else if (isValidation() && msgCount >= maxSuggestValue) {
598                                 LOG.debug("{}: stopping because msgCount reached maxSuggestValue", clientId);
599                                 done = true;
600                             }
601                             if (waitForEachRecord) {
602                                 try {
603                                     LOG.trace("{}: waiting for pending record to be acknowledged", clientId);
604                                     recCommitMap = tracker.awaitCommitMap(ackTimeoutSecs, SECONDS);
605                                 } catch (OffsetTrackerTimeoutException e) {
606                                     try {
607                                         // Write the details of the unacked record to the error view.
608                                         // This will throw an exception if there is no error view,
609                                         // which will return from process() and cause the snap to stop.
610                                         reportRecordTimeout(e, outputMap, inputDoc);
611                                         // If we get here, there is an error view and we're continuing.
612                                         if (ACK_TIMEOUT_POLICY_CONTINUE.equals(ackTimeoutPolicy)) {
613                                             LOG.debug("{}: continuing with next record in batch", clientId);
614                                             recCommitMap = tracker.toCommitMapIfAllAcked();
615                                         } else {
616                                             LOG.debug("{}: reprocessing (aborting batch)", clientId);
617                                             abortBatch = true;
618                                             if (partition == null) {
619                                                 enforceRebalance();
620                                             } else {
621                                                 LOG.debug("{}: seeking to {}:{}", clientId, tp, rec.offset());
622                                                 consumer.seek(tp, rec.offset());
623                                             }
624                                         }
625                                     } finally {
626                                         // set tracker to null so a new one is used for subsequent records
627                                         tracker = null;
628                                     }
629                                 }
630                                 LOG.trace("{}: recCommitMap={}", clientId, recCommitMap);
631                             }
632                         }
633                     } finally {
634                         if (recCommitMap != null) {
635                             commitOffsets(recCommitMap, inputDoc);
636                         }
637                     }
638                 }
639                 if (batch != null) {
640                     Map<String, Object> batchMap = createBatchMap(batch, batchIndex);
641                     Document outputDoc = documentUtility.newDocument(batchMap);
642                     outputViews.write(outputDoc, passThroughDoc);
643                 }
644                 if (!waitForEachRecord) { // == wait for each batch
645                     Map<TopicPartition, OffsetAndMetadata> batchCommitMap = null;
646                     try {
647                         if (isAutoCommit) {
648                             batchCommitMap = tracker.toCommitMap(true);
649                         } else {
650                             LOG.trace("{}: waiting for pending records to be acknowledged", clientId);
651                             batchCommitMap = tracker.awaitCommitMap(ackTimeoutSecs, SECONDS);
652                         }
653                     } catch (OffsetTrackerTimeoutException e) {
654                         try {
655                             batchCommitMap = e.commitMap();
656                             // Write the details of the unacked records to the error view.
657                             // This will throw an exception if there is no error view,
658                             // which will return from process() and cause the snap to stop.
659                             reportBatchTimeout(e, allRecs, adjustedBatchSize, batchIndex, inputDoc);
660                             // If we get here, there is an error view and we're continuing.
661                             if (ACK_TIMEOUT_POLICY_CONTINUE.equals(ackTimeoutPolicy)) {
662                                 LOG.debug("{}: continuing with next batch", clientId);
663                                 batchCommitMap = tracker.toCommitMapIfAllAcked();
664                             } else {
665                                 LOG.debug("{}: reprocessing from first unacked record", clientId);
666                                 if (partition == null) {
667                                     enforceRebalance();
668                                 } else {
669                                     if (batchCommitMap.isEmpty()) {
670                                         LOG.warn("{}: batchCommitMap is empty; can't seek", clientId);
671                                     } else {
672                                         batchCommitMap.forEach((tp, offsetAndMetadata) -> {
673                                             long offset = offsetAndMetadata.offset();
674                                             LOG.debug("{}: seeking to {}:{}", clientId, tp, offset);
675                                             consumer.seek(tp, offset);
676                                         });
677                                     }
678                                 }
679                             }
680                         } finally {
681                             // set tracker to null so a new one is used for subsequent records
682                             tracker = null;
683                         }
684                     } finally {
685                         if (batchCommitMap != null) {
686                             commitOffsets(batchCommitMap, inputDoc);
687                         }
688                     }
689                 }
690 
691                 batchIndex++;
692                 long elapsed = System.nanoTime() - startTime;
693                 long ratePerSec = (adjustedBatchSize * 1_000_000_000L) / elapsed;
694                 if (ratePerSec > maxRate) {
695                     maxRate = ratePerSec;
696                 }
697                 LOG.debug("{}: msgCount: {}, rate: {} recs/s, maxRate: {} recs/s",
698                         clientId, msgCount, ratePerSec, maxRate);
699             }
700         } catch (InterruptedException | InterruptException | SuggestViewAbortException e) {
701             // The InterruptException constructor calls Thread.currentThread().interrupt(),
702             // so let's clear the interrupted flag by calling Thread.interrupted(). If this
703             // isn't done, the consumer.close() will sometimes throw InterruptException again
704             // and not close the consumer cleanly.
705             LOG.info("{}: aborted; interrupted={}", clientId, Thread.interrupted(), e);
706         } catch (SnapException e) {
707             throw e;
708         } catch (AssertionError e) { // errors thrown by EasyMock, etc
709             throw e;
710         } catch (Throwable t) {
711             throw new SnapDataException(inputDoc, t, FAILED_TO_CONSUME)
712                     .formatWith(clientId)
713                     .withReason(t.getMessage())
714                     .withResolution(RESOLUTION_ADDRESS_ISSUE);
715         } finally {
716             LOG.debug("{}: in finally block", clientId);
717             try {
718                 // Commit offsets to reflect records that have been processed/acked.
719                 // This is important when there's an interrupt or a failure such as an ack timeout
720                 // (tracker.awaitCommitMap() throws an OffsetTrackerTimeoutException).
721                 // When the loop terminates normally, offsets will have already been
722                 // committed so the commitMap will be empty here.
723                 if (tracker != null) {
724                     commitOffsets(tracker.toCommitMap(true), inputDoc);
725                 }
726             } catch (AssertionError e) { // errors thrown by EasyMock, etc
727                 throw e;
728             } catch (Throwable t) {
729                 LOG.error("{}: error on attempt to do final commit of offsets from tracker", clientId, t);
730                 errorsOnlyLogged.add(t);
731             }
732             try {
733                 LOG.debug("{}: closing", clientId);
734                 consumer.close();
735             } catch (AssertionError e) { // errors thrown by EasyMock, etc
736                 throw e;
737             } catch (Throwable t) {
738                 LOG.error("{}: error on attempt to close Consumer", clientId, t);
739                 errorsOnlyLogged.add(t);
740             }
741         }
742     }

744     /*
745      * See SNAP-7221.
746      * This is called once when this consumer instance has at least one partition assigned to it,
747      * and only if seekType is SEEK_END_LEGACY and autoOffsetReset is AUTO_OFFSET_RESET_LATEST.
748      * The idea is to commit the correct offset for all assigned partitions that don't already
749      * have an offset committed. Basically, this is the case when the Group ID value is new so
750      * no offsets are committed for it yet. AUTO_OFFSET_RESET_LATEST means that the consumer
751      * should skip all existing records and consume only new records. So for each assigned partition
752      * we want to commit the offset of the last record, plus 1. This is trickier than it seems.
753      * The consumer is assigned partitions by calling poll(), but poll() can also return records
754      * for some of the assigned partitions. See the comments below for more.
755      */
756     private void commitOffsetsForAssignedPartitionsNotCommittedTo(
757             Set<TopicPartition> assignment,
758             ConsumerRecords<byte[], byte[]> allRecs, // the poll() result
759             Document inputDoc) {
760         try {
761             // pollPartitions: the partitions of records in the poll() result
762             Set<TopicPartition> pollPartitions = allRecs.partitions();
763             // committed: Map of each assigned partition to its committed offset, which can be null
764             Map<TopicPartition, OffsetAndMetadata> committed = consumer.committed(assignment);
765             // commitMap: the offsets to commit
766             Map<TopicPartition, OffsetAndMetadata> commitMap = new LinkedHashMap<>(assignment.size());
767             for (TopicPartition tp : assignment) {
768                 if (committed.get(tp) == null) {
769                     // no offset is committed for this partition yet
770                     long offset;
771                     if (pollPartitions.contains(tp)) {
772                         // return the offset of the first record for this partition in the result set;
773                         // position() would return an offset beyond the records that were just read
774                         // but haven't been processed yet
775                         offset = allRecs.records(tp).get(0).offset();
776                     } else {
777                         // allRecs doesn't contain any records for tp, so we can just use position()
778                         offset = consumer.position(tp);
779                     }
780                     commitMap.put(tp, new OffsetAndMetadata(offset));
781                 }
782             }
783             LOG.debug("{}: commitOffsetsForAssignedPartitions:\n" +
784                             "  assignment={}\n" +
785                             "  committed={}\n" +
786                             "  pollPartitions={}\n" +
787                             "  commitMap={}",
788                     clientId, assignment, committed, pollPartitions, commitMap);
789             if (!commitMap.isEmpty()) {
790                 commitOffsets(commitMap, inputDoc);
791             }
792         } catch (AssertionError e) { // errors thrown by EasyMock, etc
793             throw e;
794         } catch (Throwable t) {
795             LOG.error("{}: error on attempt to commit offsets for partitions assigned but not yet committed to", clientId, t);
796             errorsOnlyLogged.add(t);
797         }
798     }

806     private void commitOffsets(Map<TopicPartition, OffsetAndMetadata> commitMap, Document inputDoc) {
807         checkNotNull(commitMap, "commitMap is null");
808         if (!commitMap.isEmpty()) {
809             List<Map<String, ?>> commitMapJson = TopicPartitions.toJsonList(commitMap);
810             try {
811                 LOG.debug("{}: commitSync({})", clientId, commitMapJson);
812                 consumer.commitSync(commitMap);
813             } catch (CommitFailedException e) {

815                 errorDocMap.put("offsets", commitMapJson);

819                         .withReason(e.getMessage())
820                         .withResolution(RESOLUTION_ADDRESS_ISSUE);
821                 LOG.debug("writing to error view: {}", ex.toString());
822                 errorViews.write(ex);
823             }
824         }
825     }

828         if (partition == null) {
829             LOG.debug("{}: subscribing", clientId);
830             consumer.subscribe(Collections.singleton(topic), rebalanceListener);
831         } else {
832             TopicPartition tp = TopicPartitions.of(topic, partition);
833             LOG.debug("{}: assigning {}", clientId, tp);
834             Set<TopicPartition> partitions = Collections.singleton(tp);
835             consumer.assign(partitions);
836             performSeek(partitions);
837         }

840     private final ConsumerRebalanceListener rebalanceListener = new ConsumerRebalanceListener() {
841         @Override
842         public void onPartitionsRevoked(Collection<TopicPartition> partitions) {
843             LOG.info("{}: onPartitionsRevoked: partitions={}", clientId, partitions);
844         }
845         @Override
846         public void onPartitionsLost(Collection<TopicPartition> partitions) {
847             LOG.info("{}: onPartitionsLost: partitions={}", clientId, partitions);
848         }
849         @Override
850         public void onPartitionsAssigned(Collection<TopicPartition> partitions) {
851             LOG.info("{}: onPartitionsAssigned: partitions={}", clientId, partitions);
852             try {

854             } catch (Throwable t) {
855                 // Kafka just logs anything thrown by a ConsumerRebalanceListener
856                 LOG.error("error in onPartitionsAssigned", t);
857                 errorsOnlyLogged.add(t);
858             }
859         }

862     private void performSeek(Collection<TopicPartition> requestedPartitions) {
863         // only seek the first time a partition is assigned during a single execution
864         Set<TopicPartition> partitions = new LinkedHashSet<>(requestedPartitions);
865         if (partitions.removeAll(seekPerformed)) {
866             LOG.debug("{}: performSeek: removed seekPerformed={} from requestedPartitions={}",
867                     clientId, seekPerformed, requestedPartitions);
868         }
869         if (!partitions.isEmpty()) {
870             switch (seekType) {
871                 case SEEK_END_LEGACY:
872                     // Do nothing. Consumption will begin at the committed offset, if one exists,
873                     // or at the earliest or latest offset as specified by autoOffsetReset.
874                     LOG.debug("{}: not seeking for {}; seekType={}", clientId, partitions, seekType);
875                     break;
876                 case SEEK_BEGINNING:
877                     LOG.debug("{}: seeking to beginning for {}", clientId, partitions);
878                     consumer.seekToBeginning(partitions);
879                     break;
880                 case SEEK_END_NEW:
881                     LOG.debug("{}: seeking to end for {}", clientId, partitions);
882                     consumer.seekToEnd(partitions);
883                     break;
884                 case SEEK_OFFSET:
885                     LOG.debug("{}: seeking to {} for {}", clientId, seekOffset, partitions);
886                     for (TopicPartition tp : partitions) {
887                         consumer.seek(tp, seekOffset);
888                     }
889                     break;
890                 default:
891                     throw new ConfigurationException("Unhandled seekType: " + seekType);
892             }
893             seekPerformed.addAll(partitions);
894         }
895     }

897     private Map<String, Object> createOutputMap(ConsumerRecord<byte[], byte[]> record,
898                                                 int batchSize,
899                                                 int batchIndex,
900                                                 int recIndex,
901                                                 boolean incBatchInfo,
902                                                 Document inputDoc) {
903         final Map<String, Object> map = new LinkedHashMap<>();
904 
905         // metadata
906         Map<String, Object> metadata = createMetadataMap(record, batchSize, batchIndex, recIndex, incBatchInfo);
907         Paths.METADATA.writeStatic(map, metadata);
908 
909         // record key
910         try {
911             Object key = keyDeser.deserialize(topic, record.key());
912             Paths.KEY.writeStatic(map, key);
913         } catch (Exception e) {
914             Map<String, Object> errorDocMap = new LinkedHashMap<>();
915             Paths.RECORD_WITHOUT_HEADERS.writeStatic(errorDocMap, map);
916             Document errorDoc = documentUtility.newDocumentFor(inputDoc, errorDocMap);
917             SnapDataException ex = new SnapDataException(errorDoc, e, UNABLE_TO_DESERIALIZE)
918                     .formatWith(RECORD_KEY, keySerdeChoice.label())
919                     .withReason(e.getMessage())
920                     .withResolution(String.format(UNABLE_TO_DESERIALIZE_RES, KEY_DESERIALIZER_LABEL));
921             errorViews.write(ex);
922             return null;
923         }
924 
925         // record value
926         try {
927             Object value = valDeser.deserialize(topic, record.value());
928             Paths.VALUE.writeStatic(map, value);
929         } catch (Exception e) {
930             Map<String, Object> errorDocMap = new LinkedHashMap<>();
931             Paths.RECORD_WITHOUT_HEADERS.writeStatic(errorDocMap, map);
932             Document errorDoc = documentUtility.newDocumentFor(inputDoc, errorDocMap);
933             SnapDataException ex = new SnapDataException(errorDoc, e, UNABLE_TO_DESERIALIZE)
934                     .formatWith(RECORD_VALUE, valSerdeChoice.label())
935                     .withReason(e.getMessage())
936                     .withResolution(String.format(UNABLE_TO_DESERIALIZE_RES, VAL_DESERIALIZER_LABEL));
937             errorViews.write(ex);
938             return null;
939         }
940 
941         // record headers
942         try {
943             Map<String, Object> headers = createHeaderMap(record);
944             if (!headers.isEmpty()) {
945                 Paths.HEADERS.writeStatic(map, headers);
946             }
947         } catch (Exception e) {
948             Map<String, Object> errorDocMap = new LinkedHashMap<>();
949             Paths.RECORD_WITHOUT_HEADERS.writeStatic(errorDocMap, map);
950             Document errorDoc = documentUtility.newDocumentFor(inputDoc, errorDocMap);
951             SnapDataException ex = new SnapDataException(errorDoc, e, UNABLE_TO_DESERIALIZE_HEADERS)
952                     .withReason(e.getMessage())
953                     .withResolution(UNABLE_TO_DESERIALIZE_HEADERS_RES);
954             errorViews.write(ex);
955             return null;
956         }
957 
958         return map;
959     }

961     private Map<String, Object> createMetadataMap(ConsumerRecord<byte[], byte[]> record,
962                                                   int batchSize,
963                                                   int batchIndex,
964                                                   int recIndex,
965                                                   boolean incBatchInfo) {
966         Map<String, Object> metadata = new LinkedHashMap<>();
967         Paths.TOPIC.writeStatic(metadata, record.topic());
968         Paths.PARTITION.writeStatic(metadata, BigInteger.valueOf(record.partition()));
969         Paths.OFFSET.writeStatic(metadata, BigInteger.valueOf(record.offset()));
970         Paths.CONSUMER_GROUP.writeStatic(metadata,  group);
971         Paths.CLIENT_ID.writeStatic(metadata, clientId);
972         Paths.TRACKER_INDEX.writeStatic(metadata, trackerIndex);

981             Paths.TIMESTAMP_TYPE.writeStatic(metadata, String.valueOf(record.timestampType()));
982         }

986     private Map<String, Object> createHeaderMap(ConsumerRecord<?, ?> record) {
987         Map<String, Object> headerMap = new LinkedHashMap<>();
988         for (Header header : record.headers()) {

992             try {

995                     // There's already a value for this header in the map.
996                     // The first time this happens, we replace it with a list containing
997                     // the existing element and the new element.
998                     // When this happens again, we'll already have a list, so we just add
999                     // the new element to this list.
1000                     List<Object> valueList;
1001                     Object oldValue = headerMap.get(key);
1002                     if (oldValue instanceof List) {
1003                         valueList = (List<Object>) oldValue;
1004                     } else {
1005                         valueList = new ArrayList<>();

1007                         valueList.add(oldValue);

1009                     valueList.add(value);
1010                 } else {
1011                     headerMap.put(key, value);
1012                 }
1013             } catch (SerializationException e) {
1014                 SerdeChoice serdeChoice = headerRow != null ? headerRow.serdeChoice : defaultHeaderSerdeChoice;
1015                 String message = String.format(UNABLE_TO_DESERIALIZE, header, serdeChoice.label());
1016                 throw new SerializationException(message, e);
1017             }
1018         }
