248     protected MongoClient validateAuthenticationAndReturnClient(String username, String password,
249                                                                 String databaseName) throws UnknownHostException {
250 
251         //The account database is not required for LDAP/SSL authentication.
252         //If the account type is default, enforce that the account database is present.
253         if (!StringUtils.equals(encryptType, SSL_CERTS_CHECK_LABEL) && StringUtils.isBlank
254                 (accountDatabaseName) && StringUtils.equals(authType, DEFAULT_AUTH_LABEL)) {
255             throw new ExecutionException(ERR_ACCOUNT_INVALID)
256                     .withReason(String.format(REASON_NO_AUTH_DB,
257                             DATABASE_NAME_LABEL,
258                             AUTH_TYPE_LABEL,
259                             authType
260                     ))
261                     .withResolution(RESOLUTION_NO_AUTH_DB);
262         }
263 
264         // The keystore is required for X.509 authentication.
265         if (StringUtils.equals(authType, X509_AUTH_LABEL)) {
266             if (!StringUtils.equals(encryptType, SSL_CERTS_CHECK_LABEL)) {
267                 throw new ExecutionException(ERR_X509_AUTH_REQUIRES_SSL_CERTS_CHECK_MSG)
268                     .withReason(ERR_X509_AUTH_REQUIRES_SSL_CERTS_CHECK_REASON)
269                     .withResolution(ERR_X509_AUTH_REQUIRES_SSL_CERTS_CHECK_RESOLUTION);
270             }
271 
272             // does the keystore field contain anything?
273             if (StringUtils.isBlank(keystorePath)) {
274                 throw new ExecutionException(ERR_KEYSTORE_REQUIRED_MSG)
275                     .withReason(ERR_KEYSTORE_REQUIRED_REASON)
276                     .withResolution(ERR_KEYSTORE_REQUIRED_RESOLUTION);
277             }
278         }
279 
280         /*
281           This is a workaround to let the authentication process fails fast. When the
282           authentication fails, the MongoDB driver won't return anything until reaching the
283           timeout, and then it will throw a timeout exception. To let it fail fast I set a
284           short 10-second timeout to the first MongoClient for fail-fast authentication, if it
285           succeeds then create and return a new MongoClient with the default timeout setting.
286          */
287         for (int retry = 0; retry < MAX_RETRIES; retry++) {
288             int retryNum = retry + 1;
289             LOGGER.debug("Trying Attempt To Connect To Mongo DB:: " + retryNum);
290             if (client != null) {
291                 client.close();
292             }
293             client = null;
294             try {
295                 if (StringUtils.isBlank(username)) {
296                     client = createMongoClient(null, null, null);
297                 } else if (StringUtils.equals(authType, X509_AUTH_LABEL)) {
298                     client = createMongoClient(username, null, databaseName);
299                 } else {
300                     if (StringUtils.isBlank(password)) {
301                         throw new ExecutionException(ERR_MISSING_PASSWORD)
302                                 .withReason(ERR_MISSING_PASSWORD_REASON)
303                                 .withResolution(ERR_MISSING_PASSWORD_RESOLUTION);
304                     }
305                     client = createMongoClient(username, password, databaseName);
306                 }
307                 client.getDatabase(DATABASE_ADMIN).runCommand(new Document(COMMAND_PING, true));
308                 break;
309             } catch (MongoTimeoutException e) {
310                 if (retryNum == MAX_RETRIES) {
311                     LOGGER.debug("No of attempts tried to connect to MongoDB :: " + retryNum);
312                     throw new ExecutionException(e, ERR_VALIDATION_FAILED)
313                             .withReason(ERR_VALIDATION_FAILED_REASON)
314                             .withResolution(ERR_VALIDATION_FAILED_RESOLUTION);
315                 }
316             }
317             DatabaseUtils.delay(retryNum * IN_MILLISECONDS);
318         }
319 
320         return client;
321     }

375         if (StringUtils.equals(encryptType, TLS_ENCRYPT_LABEL)) {
376             // simulate the --ssl and --sslAllowInvalidCertificates option

390         });
391         builder.applyToClusterSettings(bldr -> {

407      * Create a SSLContext that does not validate certificate chains.
408      * The code is introduced from:
409      * http://stackoverflow.com/questions/2893819/telling-java-to-accept-self-signed-ssl-certificate
410      * <p>
411      * As it is warned by the author, the connection will not be safe from this socket, and this
412      * is not recommended. We only use this for the case that SSL authorization is chosen and
413      * allow invalid certificates is enabled
414      *
415      * @return The unsafe SSLSocketFactory
416      */
417     private SSLContext getUnsafeSSLSocketContext() {
418         // Create a trust manager that does not validate certificate chains
419         TrustManager[] trustAllCerts = new TrustManager[]{new X509TrustManager() {
420             public java.security.cert.X509Certificate[] getAcceptedIssuers() {
421                 return new X509Certificate[0];
422             }
423 
424             public void checkClientTrusted(java.security.cert.X509Certificate[] certs,
425                                            String authType) {
426             }
427 
428             public void checkServerTrusted(java.security.cert.X509Certificate[] certs,
429                                            String authType) {
430             }
431         }};
432         // Install the all-trusting trust manager
433         try {
434             SSLContext sc = SSLContext.getInstance(TLS_ENCRYPT);
435             sc.init(null, trustAllCerts, new java.security.SecureRandom());
436             return sc;
437         } catch (GeneralSecurityException e) {
438             throw new ExecutionException(ERR_INIT_SSL_CONNECTION).withReason(e.getLocalizedMessage()).withResolutionAsDefect();

442     protected SSLContext getSSlContext() {
443         try {
444             KeyStore trustStore = getKeyStore(trustStorePath, trustStorePass);
445             TrustManagerFactory trustFactory =
446                     TrustManagerFactory.getInstance(TrustManagerFactory.getDefaultAlgorithm());
447             trustFactory.init(trustStore);
448 
449             SSLContext sc = SSLContext.getInstance(SSL_STRING_LITERAL);
450             if (StringUtils.isNotBlank(keystorePath)) {
451                 KeyStore keyStore = getKeyStore(keystorePath, keystorePass);
452                 KeyManagerFactory keyManagerFactory =
453                         KeyManagerFactory.getInstance(KeyManagerFactory.getDefaultAlgorithm());
454                 keyManagerFactory.init(keyStore, keystorePass.toCharArray());
455                 sc.init(keyManagerFactory.getKeyManagers(), trustFactory.getTrustManagers(), null);
456             } else {
457                 sc.init(null, trustFactory.getTrustManagers(), null);
458             }
459             return sc;
460         } catch (NoSuchAlgorithmException | KeyStoreException | CertificateException |
461                 UnrecoverableKeyException | KeyManagementException e) {
462             throw new ConfigurationException(e, KEY_STORE_LOAD_ERROR)
463                     .withReason(e.getMessage())
464                     .withResolution(KEY_STORE_ERROR_RES);
465         } catch (IOException e) {
466             throw new ConfigurationException(e, IO_ERROR)
467                     .withReason(e.getMessage())
468                     .withResolution(IO_ERROR_RES);
469         }
470     }

472     private KeyStore getKeyStore(String url, String passPhrase) throws NoSuchAlgorithmException,
473             KeyStoreException, CertificateException, IOException {
474         URLConnection urlConnection = urlEncoder.validateAndEncodeURI(url)
475                 .toURL().openConnection();
476         urlConnection.setUseCaches(true);
477         urlConnection.connect();
478         try (InputStream inputStream = urlConnection.getInputStream()) {
479             byte[] keyStoreBits = IOUtils.toByteArray(inputStream);
480             ByteArrayInputStream bais = new ByteArrayInputStream(keyStoreBits);
481             try {
482                 return load(bais, JKS, passPhrase);
483             } catch (NoSuchAlgorithmException | CertificateException | KeyStoreException |
484                     IOException e) {
485                 try {
486                     bais.reset();
487                     return load(bais, JCEKS, passPhrase);
488                 } catch (NoSuchAlgorithmException | CertificateException | KeyStoreException |
489                         IOException ex) {
490                     bais.reset();
491                     return load(bais, PKCS12, passPhrase);
492                 }
493             } finally {
494                 IOUtils.closeQuietly(bais);
495             }
496         }
497     }
