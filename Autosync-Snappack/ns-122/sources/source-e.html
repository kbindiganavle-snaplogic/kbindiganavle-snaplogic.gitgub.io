34 /**
35  * Constructs an {@link MongoTemplate}.
36  *
37  * @author jinloes
38  */
39 @Singleton
40 public class MongoTemplateFactory {
41     /**
42      * Returns a mongo template for the given {@link MongoClient} and database name.
43      *
44      * @param mongo        mongo client
45      * @param databaseName database name
46      *
47      * @return mongo template
48      */
49     public MongoTemplate getTemplate(final MongoClient mongo, final String databaseName) {
50         MongoDatabaseFactory factory = new SimpleMongoClientDatabaseFactory(mongo, databaseName);
51         DbRefResolver resolver = new DefaultDbRefResolver(factory);
52         MappingMongoConverter converter = new MappingMongoConverter(resolver,
53                 new MongoMappingContext());
54         // Passing null disables writing _class metadata to mongo
55         converter.setTypeMapper(new DefaultMongoTypeMapper(null));
56         converter.setCustomConversions(new MongoCustomConversions(
57                 Lists.<Converter>newArrayList(new Converter<DateTime, Date>() {
58                     @Override
59                     public Date convert(DateTime source) {
60                         return source.toDate();
61                     }
62                 },new Converter<LocalTime, String>() {
63                     @Override
64                     public String convert(LocalTime source) {
65                         return source.toString();
66                     }
67                 })
68         ));
69         converter.setMapKeyDotReplacement(".");
70         converter.afterPropertiesSet();
71         return new MongoTemplate(factory, converter);
72     }
73 }
