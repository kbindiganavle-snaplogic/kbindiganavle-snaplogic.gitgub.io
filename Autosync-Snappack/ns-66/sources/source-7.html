367     public void removeEmptyAttributes(Node node) {
368         NamedNodeMap namedNodeMap = node.getAttributes();
369         for (int j = 0; j < namedNodeMap.getLength(); ) {
370             Node attributeNode = namedNodeMap.item(j);
371             if (attributeNode.getNodeValue() == null
372                     || attributeNode.getNodeValue().trim().equals(NO_DATA)
373                     || attributeNode.getNodeValue().trim().startsWith(REFERENCE_PREFIX)) {
374                 namedNodeMap.removeNamedItemNS(attributeNode.getNamespaceURI(),
375                         attributeNode.getLocalName());
376             } else {
377                 if (attributeNode.getNodeValue().equals(EMPTY_DATA)) {
378                     attributeNode.setNodeValue(StringUtils.EMPTY);
379                 }
380                 j++;
381             }
382         }
383     }
