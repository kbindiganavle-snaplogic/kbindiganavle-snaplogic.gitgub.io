36     @Override
37     public void register(BindingRegisterContext<String> ctx) throws SQLException {
38         ctx.statement().registerOutParameter(ctx.index(), Types.NCLOB);
39     }
40 
41     @Override
42     protected Clob newClob(Configuration configuration, String string) throws SQLException {
43         Connection conn = configuration.connectionProvider().acquire();
44         Clob retNclob;
45         try {
46             Clob nclob = OracleReflectionUtils.createTemporaryNclob(conn);
47             nclob.setString(1L, string);
48             retNclob = nclob;
49         } finally {
50             if (conn != null) {
51                 configuration.connectionProvider().release(conn);
52             }
53         }
54 
55         return retNclob;
56     }
