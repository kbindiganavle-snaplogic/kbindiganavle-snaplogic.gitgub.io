242     private URLResolver createUrlResolver(final Map<String, Object> documentData,
243             final APIUtils.Operation operation) {
244         return document -> {
245             try {
246                 URL abfsURL = new URI(baseURL).toURL();
247                 UriBuilder uriBuilder = UriBuilder.fromPath(operation.getPath());
248                 Matcher matcher = PATHFIELD_PARAM.matcher(operation.getPath());
249                 while (matcher.find()) {
250                     String pathField = matcher.group().substring(1, matcher.group().length() - 1);
251                     if (documentData.get(pathField) == null) {
252                         throw new SnapDataException(Messages.INPUT_ERROR)
253                                 .withReason(String.format(Messages.REQUIRED_INPUT_INVALID,
254                                         pathField))
255                                 .withResolution(Messages.PLEASE_FIX_INPUT_MAPPING_AND_RETRY);
256                     }
257                 }
258                 for (Map.Entry<String, String> param : operation.getParams().entrySet()) {
259                     uriBuilder = uriBuilder.queryParam(param.getKey(), param.getValue());
260                 }
261 
262                 for (Map.Entry<String, String> param : params.entrySet()) {
263                     uriBuilder = uriBuilder.queryParam(param.getKey(), param.getValue());
264                 }
265 
266                 return new URL(abfsURL, uriBuilder.buildFromMap(documentData).toString())
267                         .toString();
268             } catch (MalformedURLException | URISyntaxException e) {
269                 throw new ExecutionException(e, Messages.ERR_INVALID_URL_MSG)
270                         .withReason(e.getMessage())
271                         .withResolution(Messages.ERR_INVALID_URL_RESOLUTION);
272             }
273         };
274     }

301     boolean retryImmediately(HttpResponse response) {
302         int sc = response.getStatusLine().getStatusCode();
303         checkNotNull(response, "response is null");
304         if (sc == HttpStatus.SC_UNAUTHORIZED) {
305             account.connect();
306             Header[] headers = restRequestExecutor.getHeaders();
307             for (int i = 0; i < headers.length; i++) {
308                 if ((headers[i].getName().contains(AUTHORIZATION_HEADER))) {
309                     headers[i] = getBasicAuthHeader(account);
310                     break;
311                 }
312             }
313             return true;
314         } else {
315             return false;

341                 Header[] requestHeaders, String url) {
342             if (operation.equals(APIUtils.Operation.READ_FILE) && tempFile != null) {
343                 HttpEntity entity = httpResponse.getEntity();
344                 try (InputStream is = entity.getContent();
345                      OutputStream outputStream = new FileOutputStream(tempFile)) {
346                     IOUtils.copy(is, outputStream);
347                 } catch (IOException e) {

349                             .withReason(e.getMessage())

351                     errorViews.write(ex, document);

353                 return null;
354             } else if (operation.equals(APIUtils.Operation.SET_FILE_PROPERTIES)) {
355                 return null;
356             } else if (operation.equals(APIUtils.Operation.READ_FILE_FOR_APPEND)) {
357                 Map<String, Object> statusLineMap = getStatusLineMap(httpResponse);
358                 Map<String, Object> headers = getHeaderMap(httpResponse.getAllHeaders());
359                 writeToOutput(null, document, statusLineMap, headers);
360                 return null;
361             } else if ((operation.equals(APIUtils.Operation.CREATE_PATH)) ||
362                     (operation.equals(APIUtils.Operation.CREATE_DIR))) {
363                 return super.process(httpResponse, document, requestHeaders, url);
364             } else {

370         protected void writeToOutput(Object content, Document document,
371                     Map<String, Object> statusLine, Map<String, Object> headersResult) {
372             if (content instanceof Map) {
373                 if (operation.equals(APIUtils.Operation.LIST_FILES)) {
374                     Map<String, Object> responseData = (Map) content;
375                     listingFilesDirectories = (List) responseData.get(TS_RESPONSE_PATHS);
376                 }
377             } else if ((operation.equals(APIUtils.Operation.READ_FILE)) ||
378                     (operation.equals(APIUtils.Operation.SET_FILE_PROPERTIES))) {
379                 return;
380             } else if ((operation.equals(APIUtils.Operation.CREATE_DIR))) {
381                 if (setVarBasedOnStatusCode(statusLine)) {
382                     createDir = true;
383                 }
384             } else if ((operation.equals(APIUtils.Operation.CREATE_PATH))) {
385                 if (setVarBasedOnStatusCode(statusLine)) {
386                     createPath = true;
387                 }
388             } else if (operation.equals(APIUtils.Operation.READ_FILE_FOR_APPEND)) {
389                 if (setVarBasedOnStatusCode(statusLine)) {
390                     readFileForAppend = true;
391                     if (headersResult.containsKey(CONTENT_LENGTH)) {
392                         readFileContentLength = Long.parseLong(headersResult
393                                 .get(CONTENT_LENGTH).toString());
394                     }
395                 }
396             } else if (operation.equals(APIUtils.Operation.APPEND_FILE)) {
397                 if (setVarBasedOnStatusCode(statusLine)) {
398                     appendedToFile = true;
399                 }
400             } else if (operation.equals(APIUtils.Operation.FLUSH_DATA)) {
401                 if (setVarBasedOnStatusCode(statusLine)) {
402                     writtenToFile = true;
403                 }
404             } else if (operation.equals(Operation.DELETE_PATH)) {
405                 if (setVarBasedOnStatusCode(statusLine)) {
406                     delete = true;

408             } else {
409                 super.writeToOutput(content, document, statusLine, headersResult);
410             }
411         }

413         private boolean setVarBasedOnStatusCode(final Map<String, Object> statusLine) {

421             return false;
