337     public void createSshTunnel(String sshUser, String sshPassword, String sshHost,
338             String remoteDbHost, int localPort, int remoteDbPort,
339             String keyStoreURL) {
340         try {
341             switch (sshAuthType) {
342                 case AUTH_MODE_PASSWORD:
343                     session = getSession(sshUser, sshHost, null);
344                     session.setPassword(sshPassword);
345                     break;
346                 case AUTH_MODE_KEYFILE:
347                     config = new Properties();
348                     KeyStore keyStore = keyStoreHandler.createKeyStore(keyStoreURL,
349                         keyStorePassword.toCharArray());
350                     PrivateKey key = keyStoreHandler.fetchPrivateKeyForAlias(keyStore,
351                         keyAlias, privateKeyPassPhrase);
352                     String privateKeyPEM = getPrivateKeyPem(key);
353                     session = getSession(sshUser, sshHost, privateKeyPEM);
354                     break;
355             }
356         } catch (NoSuchAlgorithmException | KeyStoreException | UnrecoverableEntryException
357                 | CertificateException e) {
358             throw new ConfigurationException(e, KEY_STORE_LOAD_ERROR)
359                     .withResolution(RES_KEYSTORE_ERROR);
360         } catch (IOException e) {
361             throw new ConfigurationException(e, IO_ERROR)
362                     .formatWith(keyStoreURL)
363                     .withResolution(RES_IO_ERROR, keyStoreURL);
364         } catch (JSchException e) {
365             throw new ConfigurationException(e, INVALID_SESSION)
366                     .formatWith(keyStoreURL)
367                     .withResolution(RES_VERIFY_SSH_PROPERTIES);
368         }
369         setupConnectChannelToRemoteDb(config, session, localPort, remoteDbHost, remoteDbPort,
370                 keyStoreURL);
371     }

382     public void setupConnectChannelToRemoteDb(Properties config, Session session, int localPort,
383             String remoteDbHost, int remoteDbPort, String keyStoreURL) {
384         try {
385             config = new Properties();
386             config.put(STRICT_HOST_KEY_CHECKING, NO);
387             session.setConfig(config);
388             session.connect();
389             session.setPortForwardingL(localPort, remoteDbHost, remoteDbPort);
390         } catch (JSchException e) {
391             if (e.getMessage().equalsIgnoreCase(SESSION_IS_ALREADY_CONNECTED)) {
392                 session.disconnect();
393                 setupConnectChannelToRemoteDb(config, session, localPort, remoteDbHost,
394                         remoteDbPort, keyStoreURL);
395             } else if (e.getMessage().equalsIgnoreCase(AUTH_FAIL) || e.getMessage()
396                     .equalsIgnoreCase(AUTHENTICATION_FAILED)) {
397                 throw new ConfigurationException(Messages.AUTH_FAIL).withReason(INVALID_PASSWORD)
398                         .withResolution(RESOLUTION_AUTH_FAIL);
399             } else if (e.getMessage().contains(PORT_FORWARDING) && e.getMessage().contains
400                     (CANNOT_BE_BOUND)) {
401                 if (retryCount < MAX_RETRIES) {
402                     retryCount++;
403                     session.disconnect();
404                     // In case the connection fails due to given port not being available then
405                     //  we will try again (for MAX_RETRIES number of times) with other available
406                     //  port
407                     int randomLocalPort = getFreeLocalPort();
408                     createSshTunnel(sshUserName, sshPassword, sshHostName,
409                             databaseConfig.getHostName(), randomLocalPort, dbPort.intValue(),
410                             keyStoreURL);
411                 } else {
412                     // Exceeded MAX_RETRIES number of times with random free ports, so
413                     //  throwing exception
414                     throw new ConfigurationException(e.getMessage()).withReason
415                             (UNABLE_TO_ESTABLISH_SSH_CONNECTION)
416                             .withResolution(RESOLUTION_NETWORK_SETTINGS);
417                 }
418             }
419         }
420     }

428     protected BigInteger normalizeFetchSize(BigInteger fetchSize) {
429         // For MySQL, any positive fetch size is converted to Integer.MIN_VALUE
430         // That is the only value which results in the result set being streamed
431         // when the result set is large. See ResultSet section of
432         // http://dev.mysql.com/doc/refman/5.0/es/connector-j-reference-implementation-notes.html
433         if (fetchSize != null && fetchSize.intValue() > 0) {
434             return BigInteger.valueOf(Integer.MIN_VALUE);
435         } else {
436             return fetchSize;
437         }
