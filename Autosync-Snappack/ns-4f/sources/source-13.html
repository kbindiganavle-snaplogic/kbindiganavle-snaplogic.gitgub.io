403     @Override
404     public Connection connect() {
405         /*
406          * Remove all the connections that are not active in the list.
407          * This could keep the list from bloating the memory usage.
408          * Refer to [SWAT-3657] for more details.
409          */
410         connections.removeIf(conn -> {
411             try {
412                 return conn.isClosed();
413             } catch (SQLException e) {
414                 return false;
415             }
416         });
417         /*
418          * The thread-safety of the Connection implementation is fully dependent on the JDBC
419          * drivers. so not reusing the connection here.
420          */
421         Connection connection = dataSourceRegistry.getConnection(databaseConfig);
422         connections.add(connection);
423         return connection;
424     }
