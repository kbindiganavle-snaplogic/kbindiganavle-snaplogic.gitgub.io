408     /**
409      * Validates the account config given by the user
410      *
411      */
412     public void validateAccountSetup() {
413         AmazonS3Client s3Client;
414 
415         if ((StringUtils.isBlank(awsAccessId) || NULL.equalsIgnoreCase(awsAccessId)) &&
416                 (StringUtils.isBlank(awsSecretKey) || NULL.equalsIgnoreCase(awsSecretKey))) {
417             String ec2CredentialForS3 = System.getProperty(IAM_CREDENTIAL_FOR_S3);
418             if (Boolean.parseBoolean(ec2CredentialForS3)) {
419                 // let AWS SDK get S3 credential from EC2 metadata
420                 s3Client = new AmazonS3Client(new InstanceProfileCredentialsProvider(),
421                         UrlUtils.createClientConfiguration());
422             } else {
423                 return;
424             }
425         } else {
426             AWSCredentials awsCredentials;
427             if (StringUtils.isBlank(awsSecurityToken) || NULL.equalsIgnoreCase(awsSecurityToken)) {
428                 awsCredentials = new BasicAWSCredentials(awsAccessId, awsSecretKey);
429             } else {
430                 awsCredentials = new BasicSessionCredentials(awsAccessId,
431                         awsSecretKey, awsSecurityToken);
432             }
433             s3Client = new AmazonS3Client(awsCredentials, UrlUtils.createClientConfiguration());
434         }
435 
436         try {
437             s3Bucket = JfsUtils.updateS3RegionInfo(s3Bucket, s3Client);
438             if (!StringUtils.isBlank(s3Bucket)) {
439                 if (!s3Client.doesBucketExist(s3Bucket)) {
440                     throw new ExecutionException(ERROR_BUCKET_NOT_EXISTS)
441                             .withReason(ERROR_BUCKET_EXPECTED)
442                             .withResolution(ENSURE_BUCKET_EXISTS);
443                 }
444             }
445         } catch (AmazonClientException ae) {
446             throw new ExecutionException(ae, ERROR_S3_ACCESS)
447                     .withReason(ae.getLocalizedMessage())
448                     .withResolution(ERROR_S3_CREDS);
449         } finally {
450             s3Client.shutdown();
451         }
452     }

553     @Override
554     protected String createDatabaseUrl(final String hostname, final BigInteger port,
555             final String databaseName) {
556         if (StringUtils.isBlank(jdbcUrl)) {
557             String url = String.format(JDBC_URL_FMT, JDBC_PREFIX, hostname, port, databaseName);
558             // Setting OpenSourceSubProtocolOverride property to true prevents Amazon Redshift
559             // driver from overriding the postgres driver, it must be added to the url string so
560             // so that connection pool can pick the right driver during initializing.
561             String overrideRedshiftDriverProp = databaseConfig.getUrlProperties().getProperty
562                     (OPEN_SOURCE_SUB_PROTOCOL_OVERRIDE);
563             if (overrideRedshiftDriverProp != null) {
564                 url = String.format(OVERRIDE_FMT, url, overrideRedshiftDriverProp);
565             } else {
566                 //Enable Postgresql driver when using postgresql class.
567                 if (POSTGRESQL_DRIVER_CLASS_NAME.equals(driverClass)) {
568                     url = String.format(OVERRIDE_FMT, url, true);
569                 } else {
570                     // Enable Amazon redshift driver by default.
571                     url = String.format(OVERRIDE_FMT, url, false);
572                 }
573             }
574             return url;
575         } else {
576             return jdbcUrl;
577         }
578     }
