240     @Override
241     public Connection connect() {
242         // Re-assign access token if the account was recently reloaded (i.e. via
243         // OAuthRetryableSnowflakeSnap).
244         Connection connection = null;
245         try {
246             if (isTokenExpired()) {
247                 connection = refreshAccessTokenRetry();
248             } else {
249                 connection = acquireConnection();
250             }
251         } catch (SnapException e) {
252             if (e.getReason().contains(INVALID_REFRESH_TOKEN_MSG)) {
253                 throw e;
254             }
255             LOG.warn("OAuth error on attempting to acquire JDBC connection, retrying...");
256             if (isOAuthRetryAble(e)) {
257                 connection = refreshAccessTokenRetry();
258             }
259         }
260         if (connection == null) {
261             LOG.error("Unable to acquire JDBC connection.");
262             throw new SnapDataException(ACQUIRE_CONNECTION_ERROR)
263                     .withReason(CONNECTION_NULL_MSG)
264                     .withResolution(CONNECTION_NULL_RESOLUTION);
265         } else {
266             connections.add(connection);
267         }
268         return connection;
269     }

271     private Connection refreshAccessTokenRetry() {
272         for (int i = 0; i < 2; i++) {
273             if (refreshAccessToken()) {
274                 return acquireConnection();
275             } else {
276                 LOG.warn("Unable to refresh access token in attempt : " + i);
277             }
278         }
279         throw new SnapDataException(REFRESH_TOKEN_EXPIRED_MSG)
280                 .withReason(REFRESH_TOKEN_EXPIRED_REASON)
281                 .withResolution(REFRESH_TOKEN_EXPIRED_RESOLUTION);
282     }

310     private void loadDriverHelper(final JdbcDriverManager jdbcDriverManager, final String
311             driverClass, final List<Map<String, Map<String, Object>>> jdbcDriversMapList) {
312         List<String> jdbcDriversList = new ArrayList<>();
313         databaseConfig.setDriver(driverClass);
314         if (jdbcDriversMapList != null && !jdbcDriversMapList.isEmpty()) {
315             for (Map<String, Map<String, Object>> jdbcDriverMap : jdbcDriversMapList) {
316                 Object jdbcDriver = jdbcDriverMap.get(JDBC_DRIVER_PROP).get(
317                         SdkConstants.SNAP_PROPERTY_VALUE);
318                 if (jdbcDriver == null) {
319                     throw new ConfigurationException(JDBC_DRIVER_JAR_ERROR_MSG)
320                             .withReason(JDBC_DRIVER_JAR_ERROR_REASON)
321                             .withResolution(JDBC_DRIVER_JAR_ERROR_RESOLUTION);
322                 }
323                 jdbcDriversList.add(jdbcDriver.toString());
324             }
325         }
326 
327         ClassLoader originalClassLoader = Thread.currentThread().getContextClassLoader();
328         // we need this list for jars loaded in a privileged context, e.g., the hive drivers
329         // when using Kerberos.
330         try {
331             databaseConfig.setClassLoader(getSldbClassLoader(jdbcDriversList));
332             if (databaseConfig.getClassLoader() != null) {
333                 Thread.currentThread().setContextClassLoader(databaseConfig.getClassLoader());
334             }
335             // load driver
336             jdbcDriverManager.registerDriver(driverClass, jdbcDriversList);
337         } catch (MalformedURLException e) {
338             throw new ConfigurationException(JDBC_DRIVER_JAR_ERROR_MSG)
339                     .withReason(e.getLocalizedMessage())
340                     .withResolution(JDBC_DRIVER_JAR_ERROR_RESOLUTION);
341         } finally {
342             Thread.currentThread().setContextClassLoader(originalClassLoader);
343         }
344     }

346     private ClassLoader getSldbClassLoader(List<String> jdbcDriversPath)
347             throws MalformedURLException {
348         // our driver might be specified by the snap UI and located in SLDB so we need to
349         // create an appropriate classloader. Copied from JdbcDriverRegistryManager
350         ClassLoader parentClassLoader = Thread.currentThread().getContextClassLoader();
351         ClassLoader loader = parentClassLoader;
352         if (!CollectionUtils.isEmpty(jdbcDriversPath)) {
353             List<URL> urlList = Lists.newArrayListWithExpectedSize(jdbcDriversPath.size());
354             for (String jdbcDriverPath : jdbcDriversPath) {
355                 if (StringUtils.isNotBlank(jdbcDriverPath)) {
356                     URI resolvedPath = urlEncoder.validateAndEncodeURI(jdbcDriverPath);
357                     urlList.add(resolvedPath.toURL());
358                 }
359             }
360             loader = new SldbJarClassLoader(urlList.toArray(new URL[urlList.size()]),
361                     parentClassLoader);
362         }
363         return loader;
364     }

416     /**
417      * Refresh access token.
418      * Logic around this method should be synchronized to avoid multiple refresh requests among
419      * multiple Snap instances.
420      *
421      * @return true if the access token is unexpired, false otherwise
422      */
423     public synchronized boolean refreshAccessToken() {
424         // Try to reload without refreshing the token first.
425         try {
426             if (!snapControl.reloadSnapAccount() && !snapControl.reloadSnapAccount(true)) {
427                 LOG.warn("Failed to refresh access token.");
428                 return false;
429             }
430         } catch (SnapException ex) {
431             if (ex.getCause() instanceof IOException
432                  && StringUtils.contains(ex.getCause().getMessage(), INVALID_REFRESH_TOKEN_MSG)) {
433                 throw new SnapDataException(ex.getCause(), REFRESH_TOKEN_EXPIRED_MSG)
434                         .withReason(REFRESH_TOKEN_EXPIRED_REASON)
435                         .withResolution(REFRESH_TOKEN_EXPIRED_RESOLUTION);
436             }
437             throw ex;
438         }
439         return !isTokenExpired();
440     }

442     private boolean isTokenExpired() {
443         long newExpire = getAccessTokenExpiration(propertyValues);
444         long secondsLeft = newExpire - System.currentTimeMillis() / 1000;
445         if (secondsLeft <= 0) {
446             LOG.info("Access token is expired");
447             return true;
448         }
449         LOG.info("Access token expires in {} seconds", secondsLeft);
450         return false;
451     }

467     private boolean isErrorCodeOAuth(SQLException e) {
468         // https://docs.snowflake.com/en/user-guide/oauth-intro.html#oauth-error-codes
469         // https://github.com/snowflakedb/gosnowflake/issues/160
470         // 390112 should be caught and handled within the driver using the refresh token.
471         if (e.getErrorCode() >= 390302 && e.getErrorCode() < 390319 ||
472                 e.getErrorCode() == 390111 || e.getErrorCode() == 390114) {
473             LOG.error("OAuth retry attempt due to code: " + e.getErrorCode());
474             return true;
475         }
476         return false;
