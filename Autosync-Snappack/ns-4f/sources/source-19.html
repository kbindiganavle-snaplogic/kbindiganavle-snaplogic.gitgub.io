255     boolean validateS3Write(AmazonS3 amazonS3, String s3Bucket, String s3Folder) {
256         String filename = getSecureRandomString(16);
257         String objectKey = concat(s3Folder, filename);
258         amazonS3.putObject(s3Bucket, objectKey, VALIDATING_S3_WRITE);
259         if (amazonS3.doesObjectExist(s3Bucket, objectKey)) {
260             try {
261                 amazonS3.deleteObject(s3Bucket, objectKey);
262             } catch (SdkClientException e) {
263                 throw new SnapDataException(e, ERR_DELETE_S3_OBJECT)
264                         .formatWith(s3Bucket, objectKey)
265                         .withReason(REASON_DELETE_S3_OBJECT)
266                         .withResolution(RESOLUTION_DELETE_S3_OBJECT);
267             }
268             if (amazonS3.doesObjectExist(s3Bucket, objectKey)) {
269                 throw new SnapDataException(ERR_DELETE_S3_OBJECT)
270                         .formatWith(s3Bucket, objectKey)
271                         .withReason(REASON_DELETE_S3_OBJECT)
272                         .withResolution(RESOLUTION_DELETE_S3_OBJECT);
273             } else {
274                 return true;
275             }
276         }
277         throw new SnapDataException(ERR_SILENT_S3_WRITE)
278                 .formatWith(s3Bucket, objectKey)
279                 .withReason(REASON_SILENT_S3_WRITE)
280                 .withResolution(RESOLUTION_TRY_AGAIN);
281     }

585         try {
586             CloudStorageAccount account = CloudStorageAccount.parse(
587                     String.format(AZURE_CONNECTION_STRING, accountName, accountKey));
588             CloudBlobClient client = account.createCloudBlobClient();
589             CloudBlobContainer blobContainer = client.getContainerReference(container);
590             if (!blobContainer.exists()) {
591                 throw new ExecutionException(ERR_ASB_CONTAINER_DOES_NOT_EXISTS_MSG)
592                         .withReason(ERR_ASB_CONTAINER_DOES_NOT_EXISTS_REASON, container)
593                         .withResolution(ERR_ASB_CONTAINER_DOES_NOT_EXISTS_RESOLUTION);
594             }
595             return blobContainer;
596         } catch (URISyntaxException | InvalidKeyException | StorageException e1) {
597             throw new ExecutionException(e1, ERR_ASB_CREDENTIALS_INVALID_MSG)
598                     .withReason(e1.getMessage())
599                     .withResolution(ERR_ASB_CREDENTIALS_INVALID_RESOLUTION);
600         }
601     }

1576     public static SQLDialect getSQLServerDialect(Connection conn, SQLDialect dialect,
1577                                                  DatabaseConfig databaseConfig, Logger logger) {
1578         int majorVersion = MAX_VERSION_SUPPORTED + 1;
1579         try {
1580             DatabaseMetaData dbMetaData = conn.getMetaData();
1581             majorVersion = dbMetaData.getDatabaseMajorVersion();
1582         } catch (SQLException e) {
1583             logger.info("failed to get database major version", e);
