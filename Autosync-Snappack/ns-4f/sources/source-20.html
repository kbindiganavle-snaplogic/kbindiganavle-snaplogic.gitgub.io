143          sshAuthType = propertyValues.get(getDatabasePropertyValuePath(
144                     CONNECTION_PROPERTIES_PROP, AUTHENTICATION_MODE));
145         if (Objects.equals(sshAuthType, AUTH_MODE_KEYFILE)) {

149         dbPort = propertyValues.get(getDatabasePropertyValuePath(
150                 CONNECTION_PROPERTIES_PROP, PORT_PROP));
151         sshUserName = propertyValues.get(getDatabasePropertyValuePath(
152                 CONNECTION_PROPERTIES_PROP, SSH_USERNAME_PROP));
153         sshPassword = propertyValues.get(getDatabasePropertyValuePath(
154                 CONNECTION_PROPERTIES_PROP, SSH_PASSWORD_PROP));
155         privateKey = propertyValues.get(getDatabasePropertyValuePath(
156                 CONNECTION_PROPERTIES_PROP, PRIVATE_KEY));
157         sshHostName = propertyValues.get(getDatabasePropertyValuePath(
158                 CONNECTION_PROPERTIES_PROP, SSH_HOSTNAME_PROP));
159         keyFileUrl = propertyValues.get(getDatabasePropertyValuePath
160                 (CONNECTION_PROPERTIES_PROP, KEY_FILE_URL));
161         keyPassphrase = propertyValues.get(getDatabasePropertyValuePath
162                 (CONNECTION_PROPERTIES_PROP, PRIVATE_KEY_PASSPHRASE));
163         if (propertyValues.get(getDatabasePropertyValuePath
164                 (CONNECTION_PROPERTIES_PROP, SSH_PORT)) instanceof BigInteger) {
165             sshPort = propertyValues.get(getDatabasePropertyValuePath
166                     (CONNECTION_PROPERTIES_PROP, SSH_PORT));
167         } else {
168             sshPort = BigInteger.valueOf(Long.parseLong(propertyValues.get
169                     (getDatabasePropertyValuePath(CONNECTION_PROPERTIES_PROP, SSH_PORT))));
170         }
171 
172         switch (sshAuthType) {
173             case AUTH_MODE_PASSWORD:

175                         StringUtils.isBlank(sshHostName)) {
176                     throw new ConfigurationException(PROVIDE_ALL_DETAILS)
177                             .withReason(REASON_PASSWORD_AUTH)

182                 if (StringUtils.isBlank(keyFileUrl) ||
183                         StringUtils.isBlank(sshHostName) ||

348     public void createSshTunnel(String remoteDbHost, int localPort, URLEncoder urlEncoder) {
349         try {
350             switch (sshAuthType) {
351                 case AUTH_MODE_PASSWORD:
352                     session = getSession(sshUserName, sshHostName, null);
353                     session.setPassword(sshPassword);
354                     break;
355                 case AUTH_MODE_KEYFILE_FILE:
356                     config = new Properties();
357                     String privateKeyPEM = null;
358                     if (StringUtils.isNotBlank(keyFileUrl)) {
359                         URLConnection urlConnection = urlEncoder.validateAndEncodeURI(keyFileUrl)
360                                 .toURL()
361                                 .openConnection();
362                         urlConnection.setUseCaches(true);
363                         urlConnection.connect();
364 
365                         try (InputStream urlInputStream = urlConnection.getInputStream();
366                              ByteArrayInputStream inputStream = new ByteArrayInputStream(
367                                      IOUtils.toByteArray(urlInputStream))) {
368                             privateKeyPEM =
369                                     IOUtils.toString(inputStream, StandardCharsets.UTF_8);
370                         }
371                     }
372                     session = getSession(sshUserName, sshHostName, privateKeyPEM);
373                     break;
374                 case AUTH_MODE_KEYFILE_STRING:
375                     config = new Properties();
376                     session = getSession(sshUserName, sshHostName,
377                             PrivateKeyUtils.convertPrivateKeyToMultiline(privateKey));
378                     break;
379             }
380         } catch (IOException e) {
381             throw new ConfigurationException(e, IO_ERROR)
382                     .formatWith(keyFileUrl)
383                     .withResolution(RES_IO_ERROR, keyFileUrl);
384         } catch (JSchException e) {
385             throw new ConfigurationException(e, INVALID_SESSION)
386                     .withResolution(RES_VERIFY_SSH_PROPERTIES);
387         }
388         setupConnectChannelToRemoteDb(config, session, localPort, remoteDbHost, dbPort.intValue(),
389                 urlEncoder);
390     }

392     public Session getSession(String sshUser, String sshHost, String privateKeyPem)
393             throws JSchException {
394         JSch jSch = new JSch();
395 
396         if (StringUtils.isNotBlank(privateKeyPem)) {
397             jSch.addIdentity(StringUtils.EMPTY, privateKeyPem.getBytes(), null,
398                     keyPassphrase == null ? null : keyPassphrase.getBytes());
399         }
400 
401         Session session = createJschSession(jSch, sshUser, sshHost);
402         sessionList.add(session);
403         return session;
404     }

421     public void setupConnectChannelToRemoteDb(Properties config, Session session, int localPort,
422                                               String remoteDbHost, int remoteDbPort,
423                                               URLEncoder urlEncoder) {
424         try {
425             config = new Properties();
426             config.put(STRICT_HOST_KEY_CHECKING, "no");
427             session.setConfig(config);
428             session.connect();
429             session.setPortForwardingL(localPort, remoteDbHost, remoteDbPort);
430         } catch (JSchException e) {
431             if (e.getMessage().equalsIgnoreCase(SESSION_IS_ALREADY_CONNECTED)) {
432                 session.disconnect();
433                 setupConnectChannelToRemoteDb(config, session, localPort, remoteDbHost,
434                         remoteDbPort, urlEncoder);
435             } else if (e.getMessage().equalsIgnoreCase(AUTH_FAIL) || e.getMessage()
436                     .equalsIgnoreCase(AUTHENTICATION_FAILED) || e.getMessage()
437                     .equalsIgnoreCase(USERAUTH_FAIL)) {
438                 throw new ConfigurationException(Messages.AUTH_FAIL).withReason(INVALID_PASSWORD)
439                         .withResolution(RESOLUTION_AUTH_FAIL);
440             } else if (e.getMessage().contains(PORT_FORWARDING) && e.getMessage().contains
441                     (CANNOT_BE_BOUND)) {
442                 if (retryCount < MAX_RETRIES) {
443                     retryCount++;
444                     session.disconnect();
445                     // In case the connection fails due to given port not being available then
446                     //  we will try again (for MAX_RETRIES number of times) with other available
447                     //  port
448                     int randomLocalPort = FreePortFinder.findFreeLocalPort();
449                     createSshTunnel(remoteDbHost, randomLocalPort, urlEncoder);
450                 } else {
451                     // Exceeded MAX_RETRIES number of times with random free ports, so
452                     //  throwing exception
453                     throw new ConfigurationException(e.getMessage())
454                             .withReason(UNABLE_TO_ESTABLISH_SSH_CONNECTION)
455                             .withResolution(RESOLUTION_NETWORK_SETTINGS);
456                 }
457             }
458         }
459     }
