323         externalStorage =

347     private void retrieveAzureSynapseStorageInfo(final PropertyValues propertyValues) {
348         externalStorage =
349                 propertyValues.get(getDatabasePropPath(CONNECTION_PROPERTIES_PROP,
350                         EXTERNAL_LOCATION_AZURE_SYNAPSE_PROP));
351         storageEndpoint =
352                 getFormattedAzureSynapseStorageEndpoint(propertyValues, AZURE_SYNAPSE_ENDPOINT_PROP,
353                         externalStorage);
354         accountName = propertyValues.get(getDatabasePropPath(CONNECTION_PROPERTIES_PROP,
355                 AZURE_SYNAPSE_SOURCE_ACCOUNT_PROP));
356         container = propertyValues.get(getDatabasePropPath(
357                 CONNECTION_PROPERTIES_PROP, AZURE_SYNAPSE_SOURCE_CONTAINER_PROP));
358         folder = propertyValues.get(getDatabasePropPath(CONNECTION_PROPERTIES_PROP,
359                 AZURE_SYNAPSE_SOURCE_FOLDER_PROP));
360         authType = propertyValues.get(getDatabasePropPath(CONNECTION_PROPERTIES_PROP,
361                 AZURE_AUTH_TYPE_PROP));
362 
363         switch (authType) {
364             case AAD_USER_AUTH:
365                 clientId =
366                         propertyValues.get(getDatabasePropPath(CONNECTION_PROPERTIES_PROP,
367                                 AZURE_AAD_CLIENT_ID_PROP));
368                 clientSecret = propertyValues
369                         .get(getDatabasePropPath(CONNECTION_PROPERTIES_PROP,
370                                 AZURE_AAD_CLIENT_SECRET_PROP));
371                 tenantId =
372                         propertyValues.get(getDatabasePropPath(CONNECTION_PROPERTIES_PROP,
373                                 AZURE_AAD_TENANT_ID_PROP));
374                 break;
375             case AZURE_SYNAPSE_SERVICE_PRINCIPAL_AUTH:
376                 setIdentity(propertyValues);
377                 setSecret(propertyValues);
378                 break;
379             case AZURE_SYNAPSE_SAS_AUTH:
380                 setIdentity(propertyValues);
381                 identity = StringUtils.isBlank(identity) ?
382                         AZURE_SYNAPSE_SAS_AUTH : identity;
383                 setSecret(propertyValues);
384                 storageKey =
385                         propertyValues.get(getDatabasePropPath(CONNECTION_PROPERTIES_PROP,
386                                 AZURE_SYNAPSE_STORAGE_KEY_PROP));
387                 break;
388             case AZURE_SYNAPSE_MANAGED_IDENTITY_AUTH:
389                 setIdentity(propertyValues);
390                 identity = StringUtils.isBlank(identity) ?
391                         AZURE_SYNAPSE_MANAGED_IDENTITY_AUTH : identity;
392                 storageKey =
393                         propertyValues.get(getDatabasePropPath(CONNECTION_PROPERTIES_PROP,
394                                 AZURE_SYNAPSE_STORAGE_KEY_PROP));
395                 secret = storageKey;
396                 break;
397             case AZURE_SYNAPSE_ACCESS_KEY_AUTH:
398                 setIdentity(propertyValues);
399                 setSecret(propertyValues);
400                 storageKey = secret;
401                 break;
402         }
403     }

405     private void retrieveAzureSynapseErrorFileStorageInfo(final PropertyValues propertyValues) {
406         isErrorFileNewConfigEnabled = propertyValues.get(getDatabasePropPath(
407                 CONNECTION_PROPERTIES_PROP, AZURE_ERRORFILE_NEW_CONFIG_PROP));
408         errFileAzureFolder = propertyValues.get(getDatabasePropPath(CONNECTION_PROPERTIES_PROP,
409                 ERRFILE_SOURCE_LOCATION_AZURE_FOLDER_PROP));
410         if (!isErrorFileNewConfigEnabled) {
411             errFileAzureIdentity = identity;
412             errFileAzureSecret = secret;
413             // Reuse account settings
414             errFileExternalStorage = externalStorage;
415             errFileAzureSynapseStorageEndpoint = storageEndpoint;
416             errFileAzureAccountName = accountName;
417             errFileAzureContainer = container;
418             errFileAzureAuthType = authType;
419             validateAzureSynapseAccountSettingsForLoad(errFileAzureSynapseStorageEndpoint,
420                     errFileAzureAccountName, errFileAzureContainer, errFileAzureFolder,
421                     isErrorFileEnabledForAzureSynapse);
422             return;
423         }
424         errFileExternalStorage = propertyValues.get(getDatabasePropPath(CONNECTION_PROPERTIES_PROP,
425                 ERRFILE_EXTERNAL_LOCATION_AZURE_SYNAPSE_PROP));
426         errFileAzureSynapseStorageEndpoint = getFormattedAzureSynapseStorageEndpoint(propertyValues,
427                 ERRFILE_AZURE_SYNAPSE_ENDPOINT_PROP,
428                 errFileExternalStorage);
429         errFileAzureAccountName = propertyValues
430                 .get(getDatabasePropPath(CONNECTION_PROPERTIES_PROP,
431                         ERRFILE_SOURCE_LOCATION_AZURE_ACCOUNT_PROP));
432         errFileAzureContainer = propertyValues
433                 .get(getDatabasePropPath(CONNECTION_PROPERTIES_PROP,
434                         ERRFILE_SOURCE_LOCATION_AZURE_CONTAINER_PROP));
435         errFileAzureAuthType = propertyValues
436                 .get(getDatabasePropPath(CONNECTION_PROPERTIES_PROP,
437                         ERRFILE_AZURE_AUTH_TYPE_PROP));
438 
439         switch (errFileAzureAuthType) {
440             case AZURE_SYNAPSE_SERVICE_PRINCIPAL_AUTH:
441             case AZURE_SYNAPSE_SAS_AUTH:
442             case AZURE_SYNAPSE_ACCESS_KEY_AUTH:
443                 setErrFileAzureSynapseIdentity(propertyValues);
444                 setErrFileAzureSynapseSecret(propertyValues);
445                 break;
446             case AZURE_SYNAPSE_MANAGED_IDENTITY_AUTH:
447                 setErrFileAzureSynapseIdentity(propertyValues);
448                 break;
449             case AAD_USER_AUTH:
450             default:
451                 break;
452         }
453         validateAzureSynapseAccountSettingsForLoad(errFileAzureSynapseStorageEndpoint,
454                 errFileAzureAccountName, errFileAzureContainer, errFileAzureFolder,
455                 isErrorFileEnabledForAzureSynapse);
456     }

484     private void validateExternalStorageAuth() {
485         switch (getAuthType()) {
486             case AAD_USER_AUTH:
487                 validateFields(
488                         getTenantId(), AZURE_TENANT_ID_LABEL,
489                         getClientId(), AZURE_CLIENT_ID_LABEL,
490                         getClientSecret(), AZURE_CLIENT_SECRET_LABEL
491                 );
492                 validateAADStorageClient(getAccountName(), getTenantId(),
493                         getClientId(), getClientSecret(), getStorageEndpoint(),
494                         getExternalStorage());
495                 break;
496             case AZURE_SYNAPSE_SERVICE_PRINCIPAL_AUTH:
497             case AZURE_SYNAPSE_ACCESS_KEY_AUTH:
498                 validateFields(
499                         getIdentity(), AZURE_IDENTITY_LABEL,
500                         getSecret(), AZURE_SECRET_LABEL
501                 );
502                 validateSAKStorageClient(getAccountName(), getStorageEndpoint(),
503                         getSecret(), getExternalStorage());
504                 break;
505             case AZURE_SYNAPSE_SAS_AUTH:
506                 validateFields(
507                         getIdentity(), AZURE_IDENTITY_LABEL,
508                         getSecret(), AZURE_SECRET_LABEL
509                 );
510                 validateStorageSASClient(getAccountName(), getStorageEndpoint(),
511                         getSecret(), getExternalStorage());
512                 break;
513             case AZURE_SYNAPSE_MANAGED_IDENTITY_AUTH:
514                 validateFields(getIdentity(), AZURE_IDENTITY_LABEL);
515                 validateSAKStorageClient(getAccountName(), getStorageEndpoint(),
516                         getSecret(), getExternalStorage());
517                 break;
518         }
519     }
