171     private static Set<String> findKeysOfObject(Map<String, Object> mapObject, String rootKey)
172             throws SnapDataException {
173         logger.trace("In findKeysOfObject start");
174         Set<String> fieldsSet = new HashSet<>();
175         for (Map.Entry<String, Object> m : mapObject.entrySet()) {
176             String key = m.getKey();
177             Object value = m.getValue();
178             Map<String, Object> jsout = null;
179             List jsArr = null;
180             try {
181                 if (value instanceof Map) {
182                     fieldsSet.add(key);
183                     jsout = (Map<String, Object>) value;
184                     if (rootKey != null) {
185                         CollectionUtils.addAll(fieldsSet, findKeysOfObject(jsout, rootKey + "." + key).iterator());
186                     } else {
187                         CollectionUtils.addAll(fieldsSet, findKeysOfObject(jsout, key).iterator());
188                     }
189                 } else if (value instanceof List) {
190                     jsArr = (List) value;
191                     if (rootKey != null) {
192                         findKeysOfArray(jsArr, rootKey + "." + key, fieldsSet);
193                     } else {
194                         findKeysOfArray(jsArr, key, fieldsSet);
195                     }
196                 } else {
197                     fieldsSet.add((rootKey != null) ? (rootKey + "." + key) : (key));
198                 }
199             } catch (Exception e) {
200                 Throwable t = Throwables.getRootCause(e);
201                 throw new SnapDataException(t, ERR_WHILE_PREPARING_JSON_OBJECT)
202                         .withReason(e.getMessage())
203                         .withResolution(ERROR_RESOLUTION);
204             }
205             logger.trace("In findKeysOfObject end");
206         }
207 
208         return fieldsSet;
209     }

346             ExpressionProperty>> queryParams, final String objectType,
347             final boolean isByCrosswalk) {
348         logger.trace("In createURLResolver");
349         if (urlResolver == null) {
350             urlResolver = document -> {
351                 try {
352                     logger.trace("In createURLResolver--resolveUrl");
353                     boolean isFilterAdded = false;
354                     URIBuilder uriBuilder;
355 
356                     if (isBulkExport || isGenericExecute) {
357                         uriBuilder = new URIBuilder(exportOrServiceURL);
358                     } else if (isByCrosswalk) {
359                         uriBuilder = new URIBuilder(tenantURL + "/" + restObject
360                                 + "/_byCrosswalks");
361                     } else if (isAllEntities) {
362                         uriBuilder = new URIBuilder(tenantURL + "/" + restObject
363                                 + "/_scan");
364                     } else {
365                         uriBuilder = new URIBuilder(tenantURL + "/" + restObject);
366                     }
367 
368                     if (queryParams != null) {
369                         for (Pair<String, ExpressionProperty> paramPair : queryParams) {
370                             // left is the param key, right is the value
371                             String paramKey = paramPair.getLeft();
372                             if (paramKey == null) {
373                                 continue;
374                             }
375                             ExpressionProperty paramValueProp = paramPair.getRight();
376                             Object objValue = paramValueProp.eval(document);
377                             String paramValue = String.valueOf(objValue);
378                             if (StringUtils.equalsIgnoreCase(paramKey, FILTER) &&
379                                     !StringUtils.isBlank(objectType)) {
380                                 String objTypeParamValue = "equals(type,'" + objectType + "')";
381                                 paramValue = paramValue + " and " + objTypeParamValue;
382                                 isFilterAdded = true;
383                             }
384                             if (paramKey.equals(MAX_PARAMETER) && isValidation) {
385                                 if (Integer.parseInt(paramValue) > Integer.parseInt
386                                         (MAX_PARAMETER_VALUE)) {
387                                     paramValue = MAX_PARAMETER_VALUE;
388                                 }
389                             }
390                             uriBuilder.addParameter(paramKey, paramValue);
391                         }
392                     } else if (isValidation) {
393                         uriBuilder.addParameter(MAX_PARAMETER, MAX_PARAMETER_VALUE);
394                     }
395 
396                     // adding objectType parameter into uri string for get
397                     // snap --iteration2
398                     if (!StringUtils.isBlank(objectType) && !isFilterAdded) {
399                         logger.trace("In createURLResolver, " +
400                                 "adding objectType parameter into uri string");
401                         String objTypeParamValue = "equals(type,'" + objectType + "')";
402                         uriBuilder.addParameter(FILTER, objTypeParamValue);
403                     }
404 
405 
406                     logger.trace("createURLResolver method :: uriBuilder" + uriBuilder);
407                     return uriBuilder.build().toString();
408                 } catch (URISyntaxException e) {
409                     Throwable t = Throwables.getRootCause(e);
410                     throw new SnapDataException(t, String.format(CANNOT_MAKE_REQUEST, tenantURL))
411                             .withReason(e.getMessage())
412                             .withResolution(PLEASE_VERIFY_TENANT_URL_ON_ACCOUNT);
413                 }
414             };
415         }
416         return urlResolver;
417     }

495                 getUserScopeFromReltio();

522             if (CollectionUtils.isNotEmpty(relationTypesData)) {

526             if (CollectionUtils.isNotEmpty(interactionTypesData)) {

540             return objectNames.toArray(new String[0]);
541         } catch (Exception e) {

544             throw new SnapDataException(t, OBJECT_NOT_FOUND_MSG)

547         }

827     protected void executeRequest(String methodName, Document document, URLResolver urlResolver,
828             boolean isGroup, boolean isGetSnap, RequestDataHandler payLoad) {
829 
830         boolean isSuccess = false;
831         RestResponseProcessor getScanResponseProcess = scanResponseProcessor(methodName, document,
832                 isGroup, isGetSnap, payLoad);
833         Map<String, Object> responseMap;
834         Object result;
835         RestRequestExecutorExtended restRequestExecutorExtended = new RestRequestExecutorExtended(
836                 restHttpClient, errorViews, this);
837         try {
838             result = restRequestExecutorExtended
839                     .withMethod(methodName)
840                     .withResponseProcessor(getScanResponseProcess)
841                     .withHeaders(headers)
842                     .withMaxRetries(maxRetries)
843                     .execute(document, urlResolver, payLoad);
844         }  catch (SnapDataException e) {
845             errorViews.write(e, document);
846             return;
847         }
848         logger.trace("For thread " + Thread.currentThread().getName() + ", Response :"
849                 + result);
850 
851         if (result != null) {
852             responseMap = (Map<String, Object>) result;
853             int statusCode = (int) responseMap.get(STATUS_CODE);
854             if (statusCode >= VALID_STATUS_CODE_START && statusCode <= VALID_STATUS_CODE_END) {
855                 isSuccess = true;
856             }
857 
858             if (!isSuccess) {
859                 //pass the payload to errorView and printing error for each document in batch
860                 String responseString = (String) responseMap.get(REASON_PHRASE);
861                 if (statusCode == UNMERGE_STATUS_CODE && responseString.contains(ERROR_CODE_UNMERGE)) {
862                     SnapDataException ex = new SnapDataException(ERROR_UNMERGE)
863                             .withReason((String) responseMap.get(REASON_PHRASE))
864                             .withResolution(ERROR_EXPONENTIAL_RESOLUTION);
865                     errorViews.write(ex, document);
866                 } else {
867                     SnapDataException ex = new SnapDataException(ERROR_EXECUTING_REQUEST)
868                             .withReason((String) responseMap.get(REASON_PHRASE))
869                             .withResolution(ERROR_EXPONENTIAL_RESOLUTION);
870                     errorViews.write(ex, document);
871                 }
872             }
873 
874             if (isSuccess && (responseArray.size() != 0)) {
875                 Map<String, Object> response = new LinkedHashMap<>();
876                 List<Object> arrayResponses = new ArrayList<>();
877                 arrayResponses.addAll(responseArray);
878                 response.put(RESPONSE, arrayResponses);
879                 response.put(HEADERS, headersResultArray);
880                 response.put(STATUS_LINE, statusLinesArray);
881                 writeRecordsToOutput(response, document);
882                 responseArray.clear();
883             }
884         } else {
885             logger.debug("In ReltioSnapConnectorCommon:: Nothing found in result");
886         }
887     }
