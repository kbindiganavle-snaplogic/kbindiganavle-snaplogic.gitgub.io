218     public HttpSnapResponse request(HttpSnapRequest request) throws HttpCoreDataException,
219             HttpCoreConnectionException {
220         HttpClientContext clientContext = initClientContext();
221         request.withResponseType(responseType);
222         if ((authorization != null) && (authorization.getCredentials() != null)) {
223             clientContext.setAttribute(ACCOUNT_PASSWORD,
224                     authorization.getCredentials().getPassword());
225         }
226         if ((authorization != null) && (authorization.getAuthHeader() != null)) {
227             List<Pair<String, String>> headers = authorization.getAuthHeader();
228             Set<String> accountHeaders = new HashSet<>();
229             for (Pair<String, String> header : headers) {
230                 accountHeaders.add(header.getKey());
231             }
232             clientContext.setAttribute(ACCOUNT_HEADERS, accountHeaders);
233         }
234         try {
235             HttpSnapResponse response =
236                     Failsafe.with(retryPolicies)
237                             .get(() -> request.executeOn(client, authorization, clientContext,
238                                     customResponseHandler));
239             if (treatErrorResponseAsException && !isSuccessful(response)) {
240                 throw new HttpCoreConnectionException(String.format("Got response %d - %s from " +
241                         "the endpoint", response.getHttpStatusCode(), response.getHttpStatusText()),
242                         response);
243             }
244             return response;
245         } catch (FailsafeException fse) {
246             if (fse.getCause() != null && fse.getCause() instanceof HttpCoreConnectionException) {
247                 throw (HttpCoreConnectionException) fse.getCause();
248             }
249             throw new HttpCoreDataException("Exception encountered when requesting the endpoint",
250                     fse.getCause());
251         }

263     private HttpClientContext initClientContext() throws HttpCoreDataException {
264         try {
265             HttpClientContext context = HttpClientContext.create();
266             SSLConnectionSocketFactory sslConnectionSocketFactory;
267             if (trustAllCerts) {
268                 // client for trust all certificates
269                 TrustStrategy trustAllStrategy = (cert, authType) -> true;
270                 SSLContext sslContext = SSLContexts.custom()
271                         .loadTrustMaterial(trustAllStrategy)
272                         .build();
273                 sslConnectionSocketFactory =
274                         new SSLConnectionSocketFactory(sslContext, NoopHostnameVerifier.INSTANCE);
275             } else {
276                 SSLContext sslContext = SSLContext.getInstance(SSLConnectionSocketFactory.TLS);
277                 KeyManager[] keyManagers = authorization == null ? null :
278                         authorization.getKeyManagers();
279                 TrustManager[] trustManagers = authorization == null ? null :
280                         authorization.getTrustManagers();
281                 ;
282                 sslContext.init(keyManagers, trustManagers, new SecureRandom());
283                 sslConnectionSocketFactory =
284                         new SSLConnectionSocketFactory(sslContext, new DefaultHostnameVerifier());
285             }
286             // This is a undocumented feature from the Apache HttpClient to configure ssl socket
287             // factory for each route.
288             // ref: https://stackoverflow.com/a/36540658
289             // If this is not working in the future, we will have to create a new client for each
290             // KeyManager/TrustManager settings
291             Registry<ConnectionSocketFactory> socketFactoryRegistry =
292                     RegistryBuilder.<ConnectionSocketFactory>create()
293                             .register("https", sslConnectionSocketFactory)
294                             .register("http", PlainConnectionSocketFactory.getSocketFactory())
295                             .build();
296             context.setAttribute(SOCKET_FACTORY_REGISTRY, socketFactoryRegistry);
297             RequestConfig requestConfig = RequestConfig.copy(DEFAULT_REQUEST_CONFIG)
298                     .setRedirectsEnabled(followRedirects)
299                     .setRelativeRedirectsAllowed(allowRelativeRedirects)
300                     .setCircularRedirectsAllowed(allowCircularRedirects)
301                     .setMaxRedirects(maxRedirects)
302                     .setConnectTimeout(connectionTimeoutSec * 1000)
303                     .setSocketTimeout(readTimeoutSec * 1000)
304                     .setCookieSpec(cookieSpec)
305                     .build();
306             context.setRequestConfig(requestConfig);
307             context.setCookieStore(cookieStore);
308             if (recordHistory) {
309                 context.setAttribute(HISTORY_RECORD_CTX_KEY, historyList);
310             }
311             context.setAttribute(FOLLOW_ORIGINAL_METHOD_CTX_KEY, followOriginalMethod);
312             return context;
313         } catch (NoSuchAlgorithmException | KeyStoreException | KeyManagementException e) {
314             throw new HttpCoreDataException("Error initializing http clients", e);
315         }

551         private List<Policy<HttpSnapResponse>> buildRetryPolicies() {
552             LinkedList<Policy<HttpSnapResponse>> retryPolicies = new LinkedList<>();
553             // The order failsafe evaluate the policies is reversed:
554             // 429 -> other responses -> fallback for exceptions
555             RetryPolicy<HttpSnapResponse> retry429Policy = new RetryPolicy<HttpSnapResponse>()
556                     .handleResultIf(response -> autoRetry429 && response.getHttpStatusCode() == 429)
557                     .handleIf(exception -> false) // All exception retry will not be handled here
558                     .withDelay((result, failure, context) -> {
559                         if (result != null && useRetryAfterHeader) {
560                             List<Pair<String, String>> headers = result.getHeaders();
561                             for (Pair<String, String> header : headers) {
562                                 if (StringUtils.equalsIgnoreCase(
563                                         HttpHeaders.RETRY_AFTER, header.getKey())) {
564                                     // We wait for 5 mins maximum
565                                     long timeout = MAX_WAIT_TIME_FOR_RETRY_AFTER_SEC;
566                                     if (StringUtils.isNumeric(header.getValue())) {
567                                         timeout = Math.min(timeout,
568                                                 Integer.parseInt(header.getValue()));
569                                     } else {
570                                         // If the Retry-After header is in Date format, we
571                                         // calculate the delay second according to the current time
572                                         Date retryAfter = DateUtils.parseDate(header.getValue());
573                                         timeout = Math.min(timeout,
574                                                 retryAfter.getTime() - new Date().getTime());
575                                     }
576                                     LOG.debug("[*] 429 retry delays: {}s", timeout);
577                                     return Duration.ofSeconds(TimeUnit.SECONDS.toSeconds(timeout));
578                                 }
579                             }
580                         }
581                         LOG.debug("[*] 429 retry delays: {}s", retryIntervalSec);
582                         return Duration.ofSeconds(retryIntervalSec);
583                     })
584                     .onRetry(event -> LOG.warn(
585                             "[*] Retrying for 429 response. Details {} - {} - {}",
586                             event.getLastResult().getHttpStatusCode(),
587                             event.getLastResult().getHttpStatusText(),
588                             event.getLastResult().getEntity()))
589                     .withMaxRetries(retry);
590             retryPolicies.addFirst(retry429Policy);
591             RetryPolicy<HttpSnapResponse> retryPolicy = new RetryPolicy<HttpSnapResponse>()
592                     .handleResultIf(this::shouldRetry)
593                     .handleIf(exception -> {
594                         // Handle all HttpCoreconnectionException except for those caused by
595                         // redirect violation
596                         Throwable rootCause = Throwables.getRootCause(exception);
597                         return exception instanceof HttpCoreConnectionException
598                                 && !(rootCause instanceof RedirectException)
599                                 && !(StringUtils
600                                 .contains(rootCause.getMessage(), "Relative redirect location"));
601                     })
602                     .onRetry(event -> {
603                         if (event.getLastResult() != null) {
604                             HttpSnapResponse response = event.getLastResult();
605                             LOG.info("Got response {} {}. Details: {}. Will retry...",
606                                     response.getHttpStatusCode(), response.getHttpStatusText(),
607                                     response.getEntity());
608                         } else if (event.getLastFailure() != null) {
609                             LOG.info("Encountered error {}. Will retry...",
610                                     event.getLastFailure().getMessage());
611                             LOG.debug("ERROR", event.getLastFailure());
612                         }
613                     })
614                     .withMaxRetries(retry);
615             if (retryIntervalSec > 0) {
616                 retryPolicy.withDelay(Duration.ofSeconds(retryIntervalSec));
617             }
618             retryPolicies.addFirst(retryPolicy);
619             Fallback<HttpSnapResponse> fallback = Fallback.of(
620                     (CheckedConsumer<ExecutionAttemptedEvent<? extends HttpSnapResponse>>) event
621                             -> {
622                         HttpSnapResponse lastResp = event.getLastResult();
623                         if (lastResp != null) {
624                             throw new HttpCoreConnectionException(
625                                     "Error making HTTP call to the endpoint",
626                                     lastResp);
627                         } else if (event.getLastFailure() != null) {
628                             throw new HttpCoreConnectionException(
629                                     "Error making HTTP call to the endpoint",
630                                     event.getLastFailure());
631                         } else {
632                             throw new HttpCoreConnectionException(
633                                     "Error making HTTP call to the endpoint");
634                         }
635                     })
636                     .handle(HttpCoreConnectionException.class)
637                     .handleResultIf(response -> shouldRetry(response)
638                             && treatErrorResponseAsException);
639             retryPolicies.addFirst(fallback);
640             if (additionalRetryPolicies != null) {
641                 retryPolicies.addAll(additionalRetryPolicies);
642             }
643             return retryPolicies;
644         }

646         private boolean shouldRetry(HttpSnapResponse resp) {
647             if (resp.getHttpStatusCode() == 429) {
648                 // The 429 response has been handled by another policy
649                 return false;
650             }
651             switch (retryCondition) {
652                 case CONNECTION_AND_AUTH_ERROR_401_AND_403:
653                     return resp.getHttpStatusCode() == 401 || resp.getHttpStatusCode() == 403;
654                 case CONNECTION_AND_4XX_ERROR:
655                     return resp.getHttpStatusCode() >= 400 && resp.getHttpStatusCode() < 500;
656                 case CONNECTION_AND_5XX_ERROR:
657                     return resp.getHttpStatusCode() >= 500;
658                 case CONNECTION_AND_4XX_AND_5XX_ERROR:
659                     return resp.getHttpStatusCode() >= 400;
660             }
661             return false;
662         }
663     }
