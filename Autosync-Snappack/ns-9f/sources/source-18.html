305     @Override
306     public void process(Document inDoc, String inputViewName) throws ExecutionException {
307         // validate directory URI
308         String directory = dirExp.eval(inDoc);
309 
310         if (StringUtils.isNotBlank(findSftpSessionsChannels)) {
311             SftpDiagnostics.findMaxSftpSessionsChannels(directory, userInfo,
312                     findSftpSessionsChannels, outputViews, errorViews, documentUtility,
313                     statusReporter, SFTP_DIAG);
314             return;
315         }
316 
317         isEmpty = true;
318         if (StringUtils.isBlank(directory)) { // it is sldb current project if blank
319             directory = URL_SEPARATOR;
320         } else if (!directory.endsWith(URL_SEPARATOR)) {
321             directory = directory.concat(URL_SEPARATOR);
322         }
323         // resolve numOfRetries & retryInterval
324         numOfRetries = evalIntProperty(numOfRetriesExpr, inDoc, MIN_RETRIES, MIN_RETRIES,
325                 errorViews);
326         retryInterval = evalIntProperty(retryIntervalExpr, inDoc, MIN_RETRY_INTERVAL,
327                 MIN_RETRY_INTERVAL, errorViews);
328         FileSystem fileSystem = null;
329         try {
330             // create a matcher for glob pattern
331             String filter = filterExp.eval(inDoc);
332             PathMatcher matcher = FileSystems.getDefault().getPathMatcher(StringUtils.join(
333                     GLOB_PATTERN, filter));
334             URI pathUri = urlUtil.insertAuthority(directory, accountType, userInfo,
335                     UrlUtil.PATTERN);
336             String scheme = pathUri.getScheme();
337             if (PROTOCOL_S3.equalsIgnoreCase(scheme)) {
338                 browseS3Directory(directory, matcher, inDoc);
339                 return;
340             }
341             fileFinder.setAccount(account);
342             fileFinder.setRetries(numOfRetries, retryInterval);
343             Map<String, String> env = getSasEnvProperties(advPropMap, inDoc);
344             BinaryUtils.embedAuthenticationConfigIfNeeded(env, account);
345             if (PROTOCOL_WASB.equalsIgnoreCase(scheme) || PROTOCOL_WASBS.equalsIgnoreCase(scheme)) {
346                 env.put(USE_FAST_FILE_TYPE_LABEL, getUseFastFileType(advPropMap, inDoc));
347             }
348             if ((fileSystem = fileFinder.getFileSystemFor(pathUri, directory, env)) == null) {
349                 if (!ignoreEmptyResult) {
350                     outputViews.write(documentUtility.newDocument(Maps.newLinkedHashMap()), inDoc);
351                 }
352                 return;
353             }
354             Path path;
355             /*
356              * Obtaining path to the user-specified directory/file from the FileSystem.
357              * For file protocol it has to be pathUri.getPath().
358              * For all other protocols it has to be pathUri.toString().
359              */
360             if (URL_PROTOCOL_FILE.equalsIgnoreCase(scheme)) {
361                 path = FileFinder.getLocalPath(fileSystem, pathUri);
362             } else if (URL_PROTOCOL_HDFS.equalsIgnoreCase(scheme)) {
363                 path = ((HdfsFileSystem)fileSystem).getPath(pathUri.getScheme(),
364                         pathUri.getPort(), pathUri.getPath());
365             } else {
366                 path = fileSystem.getPath(pathUri.toString());
367             }
368             displayAttributes(path, directory, matcher, inDoc);
369             if (isEmpty) {
370                 if (Files.isDirectory(path)) {
371                     if (!ignoreEmptyResult) {
372                         outputViews.write(
373                                 documentUtility.newDocument(Maps.newLinkedHashMap()),
374                                 inDoc);
375                     }
376                 } else {
377                     throw new SnapDataException(ERR_DIR_NOT_FOUND_NO_DETAIL).formatWith(directory)
378                             .withReason(REASON_DIR_NOT_FOUND)
379                             .withResolution(RESOLUTION_DIRECTORY);
380                 }
381             }
382         } catch (IOException e) {
383             SnapDataException snapEx = new SnapDataException(e, DIR_BROWSER_IO_ERROR_MSG)
384                     .withReason(e.getMessage())
385                     .withResolution(DIR_BROWSER_IO_ERROR_RESOLUTION);
386             errorViews.write(snapEx, inDoc);
387         } catch (IllegalArgumentException e) {
388             SnapDataException snapEx = new SnapDataException(e, DIR_BROWSER_URI_ERROR_MSG)
389                     .withReason(DIR_BROWSER_URI_ERROR_REASON)
390                     .withResolution(DIR_BROWSER_URI_ERROR_RESOLUTION);
391             errorViews.write(snapEx, inDoc);
392         } catch (SuggestViewAbortException e4) {
393             throw e4;
394         } catch (SnapDataException e) {
395             errorViews.write(e, inDoc);
396         } catch (SnapException e) {
397             SnapDataException ex = new SnapDataException(e, e.getMessage())
398                     .withReason(e.getReason())
399                     .withResolution(e.getResolution());
400             errorViews.write(ex, inDoc);
401         } catch (Throwable e) {
402             SnapDataException snapEx = new SnapDataException(e, DIR_BROWSER_OTHER_ERROR_MSG)
403                     .withResolution(DIR_BROWSER_OTHER_ERROR_RESOLUTION);
404             errorViews.write(snapEx, inDoc);
405         } finally {
406             fileFinder.closeFileSystem(fileSystem, directory);
407         }
408     }

444     /**
445      * Get all the attributes for an object and return it as a map
446      *
447      * @param directory The full path to an object in the file system
448      * @param entry an entry inside the path
449      * @return Map of all attributes found
450      *
451      * @throws IOException Thrown if there is a problem getting the attributes
452      */
453     Map<String, String> gatherAttributesFor(Path directory, Path entry) throws IOException {
454         Map<String, String> map = new LinkedHashMap<>();
455         String fullPath = JfsUtils.getFileNoAuth(entry.toUri());
456         String directoryNoAuth = JfsUtils.getFileNoAuth(directory.toString());
457         if (URL_PROTOCOL_FILE.equalsIgnoreCase(directory.toUri().getScheme())) {
458             // in Windows Groundplex, fullPath may have back-slash
459             fullPath = fullPath.replaceAll(WINDOWS_URL_SEPARATOR, URL_SEPARATOR);
460         }
461         String filename = FilenameUtils.getName(JfsUtils.endsWithoutUrlSeparator(fullPath));
462         map.put(DIR_BROWSER_MAP_FILE_NAME, filename);
463         fullPath = validateFilePath(fullPath);
464         map.put(DIR_BROWSER_MAP_PATH, fullPath);
465         map.put(DIR_BROWSER_MAP_FILE_TYPE, Files.isDirectory(entry) ?
466                 DIR_BROWSER_DIRECTORY : DIR_BROWSER_FILE);
467         try {
468             BasicFileAttributes basicAttrs = Files.readAttributes(entry,
469                     BasicFileAttributes.class);
470             FileTime creationTime = basicAttrs.creationTime();
471             map.put(DIR_BROWSER_MAP_FILE_CREATE_DATE, fileTimeToString(creationTime));
472         } catch (UnsupportedOperationException x) {
473             map.put(DIR_BROWSER_MAP_FILE_CREATE_DATE, DIR_BROWSER_VALUE_UNKNOWN);
474             LOGGER.debug(DIR_BROWSER_UNSUPPORTED_ATTRIBUTE, DIR_BROWSER_MAP_FILE_CREATE_DATE,
475                     directoryNoAuth);
476         }
477         try {
478             FileTime lastModifiedTime = Files.getLastModifiedTime(entry);
479             map.put(DIR_BROWSER_MAP_FILE_UPDATE_DATE, fileTimeToString(lastModifiedTime));
480         } catch (UnsupportedOperationException x) {
481             map.put(DIR_BROWSER_MAP_FILE_UPDATE_DATE, DIR_BROWSER_VALUE_UNKNOWN);
482             LOGGER.debug(DIR_BROWSER_UNSUPPORTED_ATTRIBUTE, DIR_BROWSER_MAP_FILE_UPDATE_DATE,
483                     directoryNoAuth);
484         }
485         try {
486             map.put(DIR_BROWSER_MAP_FILE_SIZE, String.valueOf(Files.size(entry)));
487         } catch (UnsupportedOperationException x) {
488             map.put(DIR_BROWSER_MAP_FILE_SIZE, DIR_BROWSER_VALUE_UNKNOWN);

490                     directoryNoAuth);
491         }
492         PosixFileAttributes posixAttrs;
493         try {
494             PosixFileAttributeView fileAttributeView = Files.getFileAttributeView(entry,
495                     PosixFileAttributeView.class);
496             if (fileAttributeView == null) {
497                 // Windows is not POSIX-compliant
498                 map.put(DIR_BROWSER_MAP_FILE_OWNER, DIR_BROWSER_VALUE_UNKNOWN);
499                 map.put(DIR_BROWSER_MAP_FILE_PERMISSIONS, DIR_BROWSER_VALUE_UNKNOWN);
500             } else {
501                 posixAttrs = fileAttributeView.readAttributes();
502                 try {
503                     map.put(DIR_BROWSER_MAP_FILE_OWNER, posixAttrs.owner().getName());
504                 } catch (UnsupportedOperationException x) {
505                     map.put(DIR_BROWSER_MAP_FILE_OWNER, DIR_BROWSER_VALUE_UNKNOWN);
506                     LOGGER.debug(DIR_BROWSER_UNSUPPORTED_ATTRIBUTE,
507                             DIR_BROWSER_MAP_FILE_OWNER, directoryNoAuth);
508                 }
509                 try {
510                     map.put(DIR_BROWSER_MAP_FILE_PERMISSIONS,
511                             PosixFilePermissions.toString(posixAttrs.permissions()));
512                 } catch (UnsupportedOperationException x) {
513                     map.put(DIR_BROWSER_MAP_FILE_PERMISSIONS, DIR_BROWSER_VALUE_UNKNOWN);
514                     LOGGER.debug(DIR_BROWSER_UNSUPPORTED_ATTRIBUTE,
515                             DIR_BROWSER_MAP_FILE_PERMISSIONS, directoryNoAuth);
516                 }
517             }
518         } catch (UnsupportedOperationException x) {
519             map.put(DIR_BROWSER_MAP_FILE_OWNER, DIR_BROWSER_VALUE_UNKNOWN);
520             map.put(DIR_BROWSER_MAP_FILE_PERMISSIONS, DIR_BROWSER_VALUE_UNKNOWN);
521             LOGGER.debug(DIR_BROWSER_UNSUPPORTED_ATTRIBUTE2, DIR_BROWSER_MAP_FILE_OWNER,
522                     DIR_BROWSER_MAP_FILE_PERMISSIONS, directoryNoAuth);
523         }
524         return map;

572     private File getAttributes(Path path, final PathMatcher matcher, Document document)
573             throws IOException {
574         DirectoryStream.Filter<Path> filter = new DirectoryStream.Filter<Path>() {
575             public boolean accept(Path file) throws IOException {
576                 return matcher.matches(file.getFileName());
577             }
578         };
579         // if numOfRetries > 0, stage attributes in a temp file
580         File tempFile = null;
581         if (numOfRetries > 0) {
582             tempFile = File.createTempFile(PREFIX_TEMP_FILE, null, tempDir);
583             if (objectMapper == null) {
584                 objectMapper = new ObjectMapper()
585                         .registerModule(new JsonFactoryModule())
586                         .registerModule(new JodaModule())
587                         .registerModule(new JavaTimeModule());
588             }
589         }
590         isEmpty = true;
591         try (OutputStream outputStream = tempFile != null ?
592                 new GzipCompressorOutputStream(new FileOutputStream(tempFile)) : null;
593                 DirectoryStream<Path> stream = Files.newDirectoryStream(path, filter)) {
594             for (Path entry : stream) {
595                 Map<String, String> attribute = gatherAttributesFor(path, entry);
596                 if (numOfRetries > 0) {
597                     String jsonLine = objectMapper.writeValueAsString(attribute).concat(NEW_LINE);
598                     outputStream.write(jsonLine.getBytes());
599                 } else {
600                     outputViews.write(documentUtility.newDocument(attribute), document);
601                 }
602                 isEmpty = false;
603             }
604         } catch (Exception e){
605             if (tempFile != null) {
606                 tempFile.delete();
607             }
608             throw e;
609         }
610         return tempFile;
611     }

614             throws Throwable {
615         // get s3 bucket and prefix from directory path
616         String bucket;
617         String prefix;

619         switch (strings.length) {

621             case 1:

631             default:

642             amazonS3 = s3Builder.build();

644             final AmazonS3 amazonS3Final = amazonS3;

676     private List<Map<String, Object>> browseS3(String directory, String bucket, String prefix,
677             AmazonS3 s3client, PathMatcher matcher, Document document) {
678         List<Map<String, Object>> results = new ArrayList<>();
679         // init DateFormat
680         DateFormat dateFormat = new SimpleDateFormat(DATE_FORMAT);
681         // request to get list of all S3 objects within a bucket with specified prefix
682         ListObjectsRequest req = new ListObjectsRequest()
683                 .withBucketName(bucket)
684                 .withPrefix(prefix);
685         ObjectListing objectListing = null;
686         PrefixMetadata metadata = null;
687         boolean isTruncated = false;
688         Map<Object, PrefixMetadata> resultMap = new LinkedHashMap<Object, PrefixMetadata>();
689         do {
690             objectListing = s3client.listObjects(req);
691             for (S3ObjectSummary objectSummary : objectListing.getObjectSummaries()) {
692                 String entryKey = objectSummary.getKey();
693                 if (StringUtils.isNotBlank(prefix) && entryKey.startsWith(prefix)) {
694                     entryKey = entryKey.substring(prefix.length());
695                     if (StringUtils.isBlank(entryKey)) {
696                         continue;
697                     }
698 
699                 }
700                 if (entryKey.contains(URL_SEPARATOR)) {
701                     metadata = new PrefixMetadata();
702                     metadata.setLastModifiedTime(objectSummary.getLastModified());
703                     Owner owner = objectSummary.getOwner();
704                     if (owner != null) {
705                         metadata.setOwner(owner.getDisplayName());
706                     } else {
707                         metadata.setOwner(DIR_BROWSER_VALUE_UNKNOWN);
708                     }
709                     int firstUrlSeparatorIndex = entryKey.indexOf('/');
710                     String dirName = entryKey.substring(0, firstUrlSeparatorIndex);
711                     if (StringUtils.isNotBlank(dirName) &&
712                             matcher.matches(new DefaultPath(dirName))) {
713                         String dirPath;
714                         if (directory.endsWith(URL_SEPARATOR)) {
715                             dirPath = directory + dirName;
716                         } else {
717                             dirPath = directory + URL_SEPARATOR + dirName;
718                         }
719                         metadata.setPath(dirPath);
720                         updateResultMap(resultMap, dirName, metadata);
721                     }
722                 } else {
723                     Map<String, Object> map = new LinkedHashMap<>();
724                     if (isTruncated) {
725                         entryKey = decode(entryKey);
726                     }
727                     if (matcher.matches(new DefaultPath(entryKey))) {
728                         map.put(DIR_BROWSER_MAP_FILE_NAME, entryKey);
729                         map.put(DIR_BROWSER_MAP_FILE_TYPE, DIR_BROWSER_FILE);
730                         map.put(DIR_BROWSER_MAP_FILE_SIZE,
731                                 String.valueOf(objectSummary.getSize()));
732                         String filePath;
733                         if (directory.endsWith(URL_SEPARATOR)) {
734                             filePath = directory + entryKey;
735                         } else {
736                             filePath = directory + URL_SEPARATOR + entryKey;
737                         }
738                         map.put(DIR_BROWSER_MAP_PATH, filePath);
739                         Owner owner = objectSummary.getOwner();
740                         map.put(DIR_BROWSER_MAP_FILE_OWNER,
741                                 owner != null ? owner.getDisplayName() : DIR_BROWSER_VALUE_UNKNOWN);
742                         map.put(DIR_BROWSER_MAP_FILE_UPDATE_DATE,
743                                 dateFormat.format(objectSummary.getLastModified()));
744                         map.put(DIR_BROWSER_MAP_FILE_CREATE_DATE, DIR_BROWSER_VALUE_UNKNOWN);
745                         map.put(DIR_BROWSER_MAP_FILE_PERMISSIONS, DIR_BROWSER_VALUE_UNKNOWN);
746                         results.add(map);
747                     }
748                 }
749             }
750             req.setMarker(objectListing.getNextMarker());
751         } while (objectListing.isTruncated());
752         for (Object dirName : resultMap.keySet()) {
753             Map<String, Object> map = new LinkedHashMap<>();
754             PrefixMetadata dirMetadata = resultMap.get(dirName);
755             map.put(DIR_BROWSER_MAP_FILE_NAME, dirName + URL_SEPARATOR);
756             map.put(DIR_BROWSER_MAP_FILE_TYPE, DIR_BROWSER_DIRECTORY);
757             map.put(DIR_BROWSER_MAP_FILE_SIZE, 0);
758             map.put(DIR_BROWSER_MAP_PATH, dirMetadata.getPath() + URL_SEPARATOR);
759             map.put(DIR_BROWSER_MAP_FILE_OWNER, dirMetadata.getOwner());
760             map.put(DIR_BROWSER_MAP_FILE_UPDATE_DATE,
761                     dateFormat.format(dirMetadata.getLastModifiedTime()));
762             map.put(DIR_BROWSER_MAP_FILE_CREATE_DATE, DIR_BROWSER_VALUE_UNKNOWN);
763             map.put(DIR_BROWSER_MAP_FILE_PERMISSIONS, DIR_BROWSER_VALUE_UNKNOWN);
764             results.add(map);
765         }
766         return results;
767     }

770             String prefix, AmazonS3 s3client, PathMatcher matcher, Document document) {
771         List<Map<String, Object>> results = new ArrayList<>();
772         Set<String> names = new HashSet<>();
773         // default size of each batch is 1000
774         ListObjectsRequest req = new ListObjectsRequest()
775                 .withBucketName(bucket)
776                 .withPrefix(prefix)
777                 .withDelimiter(URL_SEPARATOR);
778         ObjectListing objectListing = s3client.listObjects(req);
779         boolean isTruncated = false;
780         int numBatches = 0;
781         // init DateFormat
782         DateFormat dateFormat = new SimpleDateFormat(DATE_FORMAT);
783         do {
784             // get a list of regular files
785             for (S3ObjectSummary objectSummary : objectListing.getObjectSummaries()) {
786                 String key = objectSummary.getKey();
787                 // TODO - the following if statement can be removed if AWS S3 SDK is debugged
788                 // of a defect (If filename contains colon characters, they are encoded to
789                 // %3A in the next batch.)
790                 if (isTruncated) {
791                     key = decode(key);
792                 }

794                 if (!names.contains(name) && StringUtils.isNotBlank(name) &&
795                         matcher.matches(new DefaultPath(name))) {

797                             document));

799                 }

801             // get a list of subdirectories
802             for (String key : objectListing.getCommonPrefixes()) {
803                 key = decode(key);
804                 String name = key.substring(prefix.length());
805                 if (StringUtils.isNotBlank(name) && matcher.matches(new DefaultPath(name))) {
806                     results.add(displayS3Attributes(directory, name, null, dateFormat, document));
807                 }
808             }
809             // TODO - psung : remove "&& ++numBatches < 300" when AWS S3 library is updated in the
810             // future. Due to a defect in the current version of AWS S3 library, it may run
811             // in an infinite loop. Due to this defect, Director Browser cannot support
812             // reliably a directory with more than 1000 files. "names" HasSet may not be
813             // needed in the new version of AWS S3 SDK
814             if (isTruncated = (objectListing.isTruncated() &&
815                     ++numBatches < BATCH_THRESHOLD_LIMIT)) {
816                 objectListing = s3client.listNextBatchOfObjects(objectListing);
817             }
818         } while (isTruncated);
819         return results;
