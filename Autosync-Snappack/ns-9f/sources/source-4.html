343     protected void process(final Document header, final InputStream is) {
344         this.header = header;
345         env = getSasEnvProperties(advPropMap, header);
346         BinaryUtils.embedAuthenticationConfigIfNeeded(env, account);
347         userInfo = account.connect();
348         AccountType accountType = binaryUtils.getAccountType(account);
349         Object object = filePathProperty.eval(header);
350         filenameNoAuth = String.valueOf(object);
351         if (object == null || StringUtils.isBlank(filenameNoAuth) || filenameNoAuth.equals(DOT)) {
352             consume(is, header);
353             SnapDataException ex = new SnapDataException(ERR_EMPTY_FILENAME)
354                     .withReason(ERR_BLANK_FILENAME)
355                     .withResolution(RESOLUTION_EMPTY_FILENAME);
356             errorViews.write(ex, header);
357             return;
358         }
359         LOG.info(WRITING_TO_FILE, filenameNoAuth);
360         try {
361             uri = urlUtil.insertAuthority(filenameNoAuth, accountType, userInfo, UrlUtil.PATTERN,
362                     encodingInstruction);
363         } catch (SnapDataException e) {
364             errorViews.write(e, header);
365             return;
366         }
367         // Append file action is supported by SFTP, FTP and FTPS protocols only
368         String scheme = uri.getScheme();
369         if (fileAction == FileAction.APPEND && !(scheme.equals(SFTP) || scheme.equals(FTP) ||
370                 scheme.equals(FTPS) || scheme.equals(ADL) || scheme.equals(URL_PROTOCOL_FILE))) {
371             consume(is, header);
372             SnapDataException ex = new SnapDataException(ERR_APPEND_UNSUPPORTED)
373                     .formatWith(uri.getScheme())
374                     .withReason(APPEND_SUPPORT_PROTOCOLS)
375                     .withResolution(RESOLUTION_APPEND_UNSUPPORTED);
376             errorViews.write(ex, header);
377             return;
378         }
379         writeData(header, is);
380         switch (writeResult) {
381             case FAILED:
382             case NOT_WRITTEN:
383                 return;
384             case IGNORED:
385                 break;
386             default:
387                 writeHeaderFile(header);
388         }
389         // set permissions for the actual file
390         setPermissions(uri, filenameNoAuth);
391         // send the write result to the output view
392         Map<String, String> data = new LinkedHashMap<>(3);
393         data.put(FILE_NAME_PROP, filenameNoAuth);
394         data.put(RESULT, writeResult.toString().toLowerCase());
395         if (userProvidedFilePermissions != null) {
396             data.put(PERMISSIONS, PosixFilePermissions.toString(userProvidedFilePermissions));
397         }
398         if (urlConnection instanceof S3UrlConnection) {
399             // report the version ID if an S3 file is written into version-enabled bucket
400             String versionId = urlConnection.getRequestProperty(VERSION_ID_PROP);
401             if (StringUtils.isNotBlank(versionId)) {
402                 data.put(VERSION_ID_PROP, versionId);
403             }
404         }
405         outputViews.write(documentUtility.newDocumentFor(header, data), header);
406     }

439     @VisibleForTesting
440     protected void writeHeaderFile(Document header) {
441         if (header != null && writeHeader) {
442             Object object = header.get();
443             if (object != null && object instanceof Map && !((Map) object).isEmpty()) {
444                 Map headerMap = (Map) object;
445                 int headerSize = headerMap.size();
446                 // Don't write a header if it just contains content-type/content-encoding
447                 if (!((headerSize == 1 && headerMap.containsKey(
448                         HttpHeaders.CONTENT_TYPE.toLowerCase())) || (headerSize == 2 &&
449                         headerMap.containsKey(HttpHeaders.CONTENT_TYPE.toLowerCase()) &&
450                         headerMap.containsKey(HttpHeaders.CONTENT_ENCODING.toLowerCase())))) {
451                     try {
452                         String path = uri.getPath();
453                         URI headerUri = new URI(uri.toString()
454                                 .replace(path, path.concat(EXTENSION_HEADER)));
455                         initHeaderOutputStream(headerUri);
456                         mapper.writeValue(outputStreamHeader, object);
457                         urlUtils.checkResponse(urlConnectionHeader);
458                         // set permissions on header file
459                         setPermissions(headerUri, StringUtils.EMPTY);
460                     } catch (Throwable e) {
461                         String message = e.getMessage();
462                         String detail = StringUtils.isNotBlank(message) ? message :
463                                 e.getCause() != null ? e.getCause().getMessage() : e.toString();
464                         SnapDataException ex = new SnapDataException(e, ERR_CANNOT_WRITE_HEADER)
465                                 .formatWith(filenameNoAuth)
466                                 .withReason(String.format(REASON_CANT_WRITE, detail))
467                                 .withResolution(RESOLUTION_CANT_ACCESS);
468                         errorViews.write(ex, header);
469                     } finally {
470                         try {
471                             cleanup(outputStreamHeader, urlConnectionHeader, header);
472                         } finally {
473                             outputStreamHeader = null;
474                             urlConnectionHeader = null;
475                         }
476                     }
477                 }
478             }
479         }
480     }

505      * <pre>
506      * 3 cases for <code>writeBufferSize</code> (Flush Interval)
507      *  - -1: flush once at the end.
508      *  - 0: flush every time the snap read the input (limit to the size of {@link #readBuffer}).
509      *  - Otherwise, wrap the OutputStream with BufferedOutputStream to handle the flushing.
510      * </pre>
511      */
512     protected void copy(InputStream inStream, OutputStream outStream, byte[] readBuffer,
513                         int bytesRead, int writeBufferSize) throws IOException {
514         if (outStream == null) {
515             return;
516         }
517 
518         outStream = wrapWithBufferedOutputStream(outStream, writeBufferSize);
519 
520         while (bytesRead != -1) {
521             if (bytesRead > 0) {
522                 outStream.write(readBuffer, 0, bytesRead);
523                 updateMetrics(bytesRead);
524                 if (writeBufferSize == 0) {
525                     outStream.flush();
526                 }
527             }
528             bytesRead = inStream.read(readBuffer);
529         }
530         outStream.flush();
531     }

550     protected void writeData(final Document header, InputStream inputStream) {
551         try {
552             if (createDir) {
553                 createDirIfNotPresent(uri);
554             } else if (uri.getScheme().equalsIgnoreCase(ADL)) {
555                 FileSystem fileSystem = getFileSystem();
556                 try {
557                     Path path = fileFinder.createPath(fileSystem, uri, false);
558                     Path parent = path.getParent();
559                     checkParentPathExists(parent);
560                 } finally {
561                     fileFinder.closeFileSystem(fileSystem, filenameNoAuth);
562                 }
563             }
564 
565             int bytesRead = inputStream.read(readBuffer);
566             if (bytesRead == -1 && !writeEmptyFile) {
567                 writeResult = WriteResult.NOT_WRITTEN;
568                 return;
569             }
570             initOutputStream(header);
571 
572             // configure chunk size (MB) for Azure Blob Storage
573             int wasbChunkSize = 0;
574             if (account instanceof BinaryAzureAccount) {
575                 wasbChunkSize = ((BinaryAzureAccount) account).getRequestSizeInMB();
576                 if (urlConnection instanceof WasbUrlConnection) {
577                     ((WasbUrlConnection) urlConnection).setRequestSize(wasbChunkSize);
578                 }
579             }
580 
581             int writeBufferSize = getWriteBufferSize(flushInterval, wasbChunkSize * 1024 * 1024);
582             copy(inputStream, outputStream, readBuffer, bytesRead, writeBufferSize);
583             LOG.info("Total number of bytes written successfully to the output stream: {}",
584                     byteCounter.getStats());
585             urlUtils.checkResponse(urlConnection);
586         } catch (SuggestViewAbortException e) {
587             throw e;
588         } catch (Exception e) {
589             if (retries-- > 0) {
590                 retryWriting(inputStream, header, e.getMessage());
591             } else if (e instanceof SnapDataException){
592                 writeResult = WriteResult.FAILED;
593                 errorViews.write((SnapDataException) e, header);
594             } else {
595                 throwExceptionCantWrite(e, header);
596             }
597             return;
598         } finally {
599             consume(inputStream, header);
600         }
601         // validate the file write
602         if (validate) {
603             switch (writeResult) {
604                 case CREATED:
605                 case OVERWRITTEN:
606                 case APPENDED:
607                     BinaryUtils.sleep(DELAY_IN_SECONDS);
608                     FileSystem fileSystem = getFileSystem();
609                     String uriNoAuth = URLUtils.getFileNoAuth(uri.toString());
610                     Path path = null;
611                     try {
612                         LOG.info("Trying to fetch the metadata for the file just written {}",
613                                 uriNoAuth);
614                         try {
615                             ContextualCallable<Path> callable = context -> getPath(fileSystem);
616                             CheckedConsumer<Throwable> failedAttemptListener = e -> LOG.warn(
617                                     "binary_retry - Failed to fetch the metadata after writing {}" +
618                                             " due to {}", uriNoAuth, e.getMessage());
619                             path = failsafeExecute(callable, failedAttemptListener,
620                                     retries, retryInterval);
621                             LOG.info("Successfully fetched metadata for the file {}", uriNoAuth);
622                         } catch (FailsafeException e1) {
623                             Throwable cause = e1.getCause();
624                             throw new SnapDataException(cause,
625                                     ERR_READING_METADATA_AFTER_WRITE_MSG)
626                                     .formatWith(uriNoAuth)
627                                     .withReason(cause.getMessage())
628                                     .withResolution(ERR_READING_METADATA_AFTER_WRITE_RESOLUTION);
629                         }
630                         handleFileNotExists(path, inputStream, header);
631                     } finally {
632                         fileFinder.closeFileSystem(fileSystem, filenameNoAuth);
633                         if (path instanceof S3Path) {
634                             ((S3Path) path).shutdown();
635                         }
636                     }
637                     break;
638             }
639         }
640     }

661     protected void cleanup(OutputStream outputStream, URLConnection urlConnection, Document header)
662             throws ExecutionException {
663         if (outputStream != null) {
664             try {
665                 // for some ftp sites (e.g. box), we have to wait a few seconds before
666                 // closing output stream, otherwise only a partial data may be written
667                 String scheme = uri.getScheme();
668                 if (FTP.equals(scheme)) {
669                     BinaryUtils.sleep(3);
670                     LOG.info("Total bytes should be written before closing the outputstream: " +
671                             byteCounter.getStats());
672                 }
673                 closeOutputStream(outputStream, header);
674                 switch (scheme) {
675                     case FTP:
676                         // It was also observed that we have to wait a short while after the output
677                         // stream is closed and before closing the FTP connection if the FTP server
678                         // software is Serv-U. [SWAT-1223]
679                         LOG.warn("binary_delay - Delay before closing FTP connection: " +
680                                 "{} seconds", ftpDelayBeforeClosing);
681                         BinaryUtils.sleep(ftpDelayBeforeClosing);
682                         break;
683                     case FTPS:
684                         // It was also observed that we have to wait a short while after the output
685                         // stream is closed and before closing the FTPS connection if
686                         // the FTPS server software is FileZilla. [SWAT-1936]
687                         LOG.warn("binary_delay - Delay before closing FTPS connection: " +
688                                 "{} seconds", ftpsDelayBeforeClosing);
689                         BinaryUtils.sleep(ftpsDelayBeforeClosing);
690                         break;
691                 }
692 
693             } catch (SuggestViewAbortException e) {
694                 throw e;
695             } catch (Exception e) {
696                 if (SFTP.equals(uri.getScheme())) {
697                     // For some sftp server, the output stream fails to close for unknown reason
698                     // even if the file has been written successfully.
699                     LOG.debug(LOG_CLOSE_OUTPUT_STREAM, e.getMessage());
700                 } else {
701                     throwExceptionCantWrite(e, header);
702                 }
703             } finally {
704                 urlUtils.disconnect(urlConnection);
705             }
706         } else {
707             urlUtils.disconnect(urlConnection);
708         }
709     }

789     /**
790      * Initializes the output stream for the file. Output stream is kept open until all binary
791      * inputs are read.
792      *
793      * @param header
794      */
795     protected void initOutputStream(final Document header) throws Exception {
796         if (outputStream == null) {
797             if (fileAction == FileAction.OVERWRITE) {
798                 // if file action is overwrite, do not check if the file exists
799                 writeResult = WriteResult.OVERWRITTEN;
800             } else {
801                 // check if the file exists
802                 if (exists()) {
803                     switch (fileAction) {
804                         case APPEND:
805                             List<NameValuePair> qparams = Lists.newArrayList();
806                             qparams.add(new BasicNameValuePair(FILE_ACTION_PROP,
807                                     FileAction.APPEND.toString().toLowerCase()));
808                             String urlParams = URLEncodedUtils.format(qparams, UTF_8);
809                             String formattedString = String.format(FORMAT_QUERY_URL,
810                                     uri.toString(), urlParams);
811                             Pattern p = Pattern.compile("\\bfileAction=append?\\b");
812                             Matcher m = p.matcher(formattedString);
813                             if (m.find()) {
814                                 formattedString = formattedString.split("fileAction=append?")[0]
815                                         .concat("fileAction=append");
816                             }
817                             LOG.info("Appended URI: {}" , URLUtils.getFileNoAuth(uri));
818                             uri = new URI(formattedString);
819                             writeResult = WriteResult.APPENDED;
820                             break;
821                         case IGNORE:
822                             writeResult = WriteResult.IGNORED;
823                             return;
824                         case ERROR:
825                             writeResult = WriteResult.IGNORED;
826                             throw new SnapDataException(ERR_FILE_EXISTS)
827                                     .formatWith(filenameNoAuth)
828                                     .withReason(REASON_FILE_EXISTS)
829                                     .withResolution(RESOLUTION_FILE_EXISTS);
830                         default:
831                             throw new SnapDataException(ERR_UNSUPPORTED_FILE_ACTION)
832                                     .formatWith(fileAction.toString())
833                                     .withReason(REASON_UNSUPPORTED_FILE_ACTION)
834                                     .withResolution(RESOLUTION_UNSUPPORTED_FILE_ACTION);
835                     }
836                 } else {
837                     writeResult = WriteResult.CREATED;
838                 }
839             }
840             // get output stream
841             openURLConnection();
842             if (urlConnection != null) {
843                 urlConnection.setRequestProperty(TEMP_DIR, tempDir.toURI().getPath());
844                 if (header != null) {
845                     Map<String, Object> map = documentUtility.getAsMap(header, errorViews);
846                     for (Map.Entry<String, Object> entry : map.entrySet()) {
847                         Object value = entry.getValue();
848                         String stringValue;
849 
850                         if (value == null) {
851                             continue;
852                         }
853                         if (value instanceof DateTime) {
854                             DateTime dateTime = (DateTime) value;
855                             stringValue = DateUtil.formatDate(dateTime.toDate());
856                         } else if (value instanceof Date) {
857                             stringValue = DateUtil.formatDate((Date) value);
858                         } else {
859                             stringValue = value.toString();
860                         }
861                         urlConnection.setRequestProperty(entry.getKey(), stringValue);
862                     }
863                 }
864                 // set user-defined object metadata
865                 setRequestProperties(metadataExpr, PREFIX_USER_DEFINED_METADATA);
866                 // set object tags
867                 setRequestProperties(tagsExpr, PREFIX_OBJECT_TAG);
868                 // set retry properties
869                 if (maxRetries > 0) {
870                     urlConnection.setRequestProperty(RETRIES_PROP,  String.valueOf(maxRetries));
871                     urlConnection.setRequestProperty(RETRY_INTERVAL_PROP,
872                             String.valueOf(retryInterval));
873                 }
874 
875                 if (account instanceof UrlAccount) {
876                     urlConnection = ((UrlAccount) account).withHeaders(urlConnection);
877                 }
878 
879                 // set the request size before connecting to azure API
880                 if (urlConnection instanceof WasbUrlConnection &&
881                     account instanceof BinaryAzureAccount){
882                     ((WasbUrlConnection)urlConnection).setRequestSize(
883                         ((BinaryAzureAccount)account).getRequestSizeInMB()
884                     );
885                 }
886 
887                 String scheme = uri.getScheme();
888                 switch (scheme) {
889                     case JfsUtils.PROTOCOL_S3:
890                         // set the buffer size to be used as per user's choice
891                         setS3OutputStreamBufferSize();
892                         // set the number of S3 multipart upload threads
893                         if (maxUploadThreads != null && maxUploadThreads.intValue() > 0) {
894                             urlConnection.setRequestProperty(MAX_UPLOAD_THREADS,
895                                     maxUploadThreads.toString());
896                         }
897                         break;
898                 }
899                 urlUtils.prepareForPut(urlConnection);
900                 setRequestPropertyForAuthorization(scheme, userInfo, urlConnection);
901                 setFilesystemProperties(urlConnection, account, maxRetries, retryInterval, env);
902                 urlConnection.connect();
903                 outputStream = urlConnection.getOutputStream();
904             }
905         }
906     }

1098     protected void retryWriting(InputStream inputStream, Document header, String reason) {
1099         IOUtils.closeQuietly(inputStream);
1100         IOUtils.closeQuietly(outputStream);
1101         JfsUtils.sleepNoInterrupt(retryInterval, TimeUnit.SECONDS, LOG);
1102         LOG.warn("{} - File Writer failed, {}", RETRY_SIGNATURE, reason);
1103         LOG.warn("{} - Retry attempt: {}", RETRY_SIGNATURE, maxRetries - retries);
1104         inputStream = getInputStreamFrom(tempFile, header);
1105         resetMetrics();
1106         if (uri.getScheme().equalsIgnoreCase(ADL)) {
1107             // it gets reopened in initoutputstream() method
1108             outputStream = null;
1109         }
1110         if (inputStream != null) {
1111             writeData(header, inputStream);
1112         }
1113         retries = maxRetries;
1114     }
