228      * Compute the Content-Disposition header field value from a given filename.
229      *
230      * @param filename The filename to derive the attachment filename from.
231      *
232      * @return The header field value or NULL if one could not be derived.
233      */
234     String getDispositionFormat(String filename, URI uri) {
235         // Find the base name from the filename, we'll use that as the name in the disposition.
236         Path path;
237         try {
238             path = Paths.get(filename);
239         } catch (InvalidPathException e) {
240             // in Windows Groundplex, this exception is thrown if filename = "/c:/opt/usr/..."
241             String osName = System.getProperty(OS_NAME).toLowerCase();
242             if (osName.contains(WIN) && filename.startsWith(URL_SEPARATOR) &&
243                    URL_PROTOCOL_FILE.equals(uri.getScheme())) {
244                 // absolute files on Windows do not start with "/"
245                 filename = filename.substring(1, filename.length());
246             }
247             try {
248                 path = Paths.get(filename);
249             } catch (InvalidPathException ex) {
250                 return null;
251             }
252         }
253         String basename = path.getFileName().toString();
254         if (!basename.isEmpty()) {
255             // Make sure the filename in the disposition is simple and sane.
256             return String.format(DISPOSITION_FORMAT, basename.replaceAll(
257                     DISPOSITION_FILENAME_SCRUBBER, "_"));
258         }
259         return null;
260     }

262     /**
263      * Create a binary header document from a filename and URLConnection.
264      *
265      * @param sensitiveUri The URI that was used to open the file, which may contain
266      *                     sensitive user information.
267      * @param urlConnection The connection to the file.
268      *
269      * @return A map containing HTTP header fields that can be used in a binary document header.
270      */
271     Map<String, Object> createHeader(URI sensitiveUri, URLConnection urlConnection) {
272         Map<String, Object> headerMap = new LinkedHashMap<>();
273         String filename = sensitiveUri.getPath();
274         // Guess the Content-Type, but allow the URLConnection to overwrite it.
275         String contentType = guessContentTypeFromName(filename);
276         headerMap.put(HttpHeaders.CONTENT_TYPE.toLowerCase(), contentType);
277         for (Map.Entry<String, List<String>> entry :
278                 urlConnection.getHeaderFields().entrySet()) {
279             String key = entry.getKey();
280             if (key == null) {
281                 continue;
282             }
283             key = key.toLowerCase();
284 
285             if (key.startsWith("x-sl") || key.startsWith("sl-x") ||
286                     key.startsWith("access-control")) {
287                 continue;
288             }
289             switch (entry.getValue().size()) {
290                 case 0:
291                     break;
292                 case 1:
293                     headerMap.put(key, entry.getValue().get(0));
294                     break;
295                 default:
296                     headerMap.put(key, entry.getValue());
297                     break;
298             }
299         }
300         // Always set the Content-Location and Content-Disposition.
301         headerMap.put(HttpHeaders.CONTENT_LOCATION.toLowerCase(),
302                 JfsUtils.getFileNoAuth(sensitiveUri));
303         long lastModified = urlConnection.getLastModified();
304         if (lastModified > 0) {
305             headerMap.put(HttpHeaders.LAST_MODIFIED.toLowerCase(), new DateTime(lastModified,
306                     DateTimeZone.UTC));
307         }
308         String dispositionValue = getDispositionFormat(filename, sensitiveUri);
309         if (dispositionValue != null) {
310             headerMap.put(HttpHeaders.CONTENT_DISPOSITION.toLowerCase(), dispositionValue);
311         }
312         if (PROTOCOL_S3.equalsIgnoreCase(sensitiveUri.getScheme())) {
313             // get S3 object tags if any
314             for (Map.Entry<String, List<String>> entry :
315                     urlConnection.getRequestProperties().entrySet()) {
316                 String key = entry.getKey();
317                 if (StringUtils.startsWith(key, PREFIX_OBJECT_TAG)) {
318                     List<String> values = entry.getValue();
319                     if (CollectionUtils.isNotEmpty(values)) {
320                         key = key.substring(PREFIX_OBJECT_TAG.length());
321                         if (headerMap.containsKey(key)) {
322                             // Prefix the object tag key witn PREFIX_TAG if headerMap already
323                             // contains the same key.
324                             key = PREFIX_TAG.concat(key);
325                         }
326                         headerMap.put(key, values.get(0));
327                     }
328                 }
329             }
330         }
331         return headerMap;
332     }

343         int unit = si ? 1000 : 1024;

453                                 .onComplete(smbRetryListener())

516             if (t instanceof SuggestViewAbortException) {

639     void copyFileWithAzcopy(String sourceStr, String targetStr, String fileOperation,
640                                    String azcopyPathInput, String sasUri,

642                                    LinkedHashMap<String, String> output,

644         try {

647             sasUri = SAS_TOKEN_SEPARATOR + splits[1];
648             String source = sourceStr;
649             String target = targetStr;
650             if (sourceStr.startsWith("https")) {
651                 source = sourceStr + sasUri;
652             } else if (sourceStr.startsWith("file:///")) {
653                 source = sourceStr.substring(7);
654             }
655             if (targetStr.startsWith("https")) {
656                 target = targetStr + sasUri;

659                     source, target);

661                     source);
662             if (fileOperation.equals(COPY)) {
663                 invokeCommand(copyCommand, output, tempDirFile);
664                 output.put(STATUS,STATUS_COPIED);
665             } else if (fileOperation.equals(MOVE)) {
666                 invokeCommand(copyCommand, output, tempDirFile);
667                 if (sourceStr.startsWith("https")) {
668                     invokeCommand(rmCommand, output, tempDirFile);
669                 } else {
670                     File f = new File (source);
671                     if (f.isFile()) {
672                         f.delete();
673                     } else {
674                         deleteFolder(f);
675                     }
676                 }
677                 output.put(STATUS,STATUS_MOVED);
678             } else {
679                 copyCommand = String.format(AZCOPY_CP_COMMAND_FORMAT, azcopyPathInput,
680                         sourceStr + RENAME_WILDCARD + advPropMap.get(SAS_URI_NAME),
681                         targetStr + advPropMap.get(SAS_URI_NAME));
682                 invokeCommand(copyCommand, output, tempDirFile);
683                 invokeCommand(rmCommand, output, tempDirFile);
684                 output.put(STATUS,STATUS_RENAMED);
685             }
686         } catch (Throwable e) {
687             Throwable ex = e instanceof FailsafeException ? e.getCause() : e;
688             if (ex instanceof ConfigurationException) {
689                 throw (ConfigurationException) ex;
690             } else {
691                 throw new IOException(ex.getMessage(), ex);
692             }

717     void lookForAzcopy(String azcopyPathInput) {
718         // a caution sysadmin will either drop the PATH environment or
719         // set it to /. In these cases we should return
720         // immediately.
721         if (!System.getenv().containsKey(PATH_ENV_VARIABLE)) {
722             return;
723         }
724         // see above.
725         final String path = System.getenv(PATH_ENV_VARIABLE).trim();
726         if (path.isEmpty()) {
727             return;
728         }

731             if (file.exists() && file.canRead() && file.canExecute()) {

741     void invokeCommand(String command, LinkedHashMap<String, String> output,
742                                File tempDirFile)
743             throws IOException, InterruptedException {
744         try {
745             String[] envp = new String[2];
746             envp[0] = String.format(AZCOPY_SET_LOG_LOCATION_FORMAT,tempDirFile.getAbsolutePath());
747             envp[1] = String.format(AZCOPY_SET_PLAN_LOCATION_FORMAT,tempDirFile.getAbsolutePath());
748             Runtime rt = Runtime.getRuntime();
749             Process pr = rt.exec(command,envp);
750             LOGGER.debug(EXIT_VALUE + pr.waitFor());
751             BufferedReader input = new BufferedReader(new InputStreamReader(pr.getInputStream()));
752             BufferedReader inputErr =
753                     new BufferedReader(new InputStreamReader(pr.getErrorStream()));
754             readFromReader(inputErr, ERROR_STREAM, output);
755             readFromReader(input, INPUT_STREAM, output);
756         } catch (Exception e) {
757             throw e;
758         }
759     }

761     void readFromReader(BufferedReader reader, String streamType,
762                                LinkedHashMap<String, String> output) throws IOException {
763         String line;
764         StringBuilder sb = new StringBuilder();
765         boolean isFirstLine = true;
766         while ((line = reader.readLine()) != null) {
767             LOGGER.debug(streamType + line);
768             if (isFirstLine) {
769                 isFirstLine = false;
770             } else {
771                 sb.append(System.lineSeparator());
772             }
773             sb.append(line);
774             if (line.contains(AZCOPY_AUTH_ERR)) {
775                 throw new ConfigurationException(ERR_AZCOPY_AUTHENTICATION_MSG)
776                         .withReason(ERR_AZCOPY_AUTHENTICATION_REASON)
777                         .withResolution(ERR_AZCOPY_AUTHENTICATION_RESOLUTION);
778             }
779         }
780         String message = sb.toString();
781         if (!StringUtils.isEmpty(message)) {
782             if (streamType.equals(ERROR_STREAM)) {
783                 throw new SnapDataException(AZCOPY_ERR)
784                         .withReason(message)
785                         .withResolution(AZCOPY_ERR_RESOLUTION);
786             } else {
787                 output.put(OUTPUT,message);
788             }
789         }
790     }
