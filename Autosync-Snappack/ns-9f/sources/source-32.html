233     @Override
234     protected void process(Document document, String inputViewName) {
235         Object object = filePath.eval(document);
236         if (object instanceof String && ((String) object).length() < MAX_FILE_PROPERTY_SIZE) {
237             // a long delay has been observed when the evaluated object is several-mega-byte string
238             filename = (String) object;
239         } else {
240             SnapDataException ex = new SnapDataException(ERR_FILE_PROPERTY)
241                     .withReason(REASON_FILE_PROPERTY)
242                     .withResolution(RESOLUTION_ADDRESS_ISSUE);
243             errorViews.write(ex, document);
244             return;
245         }
246         if (StringUtils.isBlank(filename)) {
247             SnapDataException ex = new SnapDataException(ERR_BLANK_FILENAME)
248                     .withReason(ERR_BLANK_FILENAME)
249                     .withResolution(ERR_BLANK_FILENAME_RESOLUTION);
250             errorViews.write(ex, document);
251             return;
252         }
253         LOGGER.debug(READING_ZIP_FILE_FROM, filename);
254         URI pathUri = urlUtil.insertAuthority(filename, accountType, userInfo, UrlUtil.PATTERN,
255                 encodingInstruction);
256         // resolve numOfRetries and retryInterval
257         numOfRetries = evalIntProperty(numOfRetriesExpr, document, MIN_RETRIES, MIN_RETRIES,
258                 errorViews);
259         retryInterval = evalIntProperty(retryIntervalExpr, document, MIN_RETRY_INTERVAL,
260                 MIN_RETRY_INTERVAL, errorViews);
261         URLConnection urlConnection = null;
262         try {
263             if (numOfRetries > 0 && tmpDir != null) {
264                 // download zip file with retries
265                 final long limit = computeRetryLimit();
266                 ContextualCallable<URLConnection> callable =
267                         context -> downloadZipFile(filename, pathUri, limit, document);
268                 CheckedConsumer<Throwable> failedAttemptListener = e -> LOGGER.warn(
269                         "{} - Failed to read file: {}, {}",
270                         RETRY_SIGNATURE, filename, e.getMessage());
271                 urlConnection = failsafeExecute(callable, failedAttemptListener,
272                         numOfRetries, retryInterval);
273             } else {
274                 urlConnection = getURLConneciton(filename, pathUri, document);
275             }
276             // unzip
277             try (PushbackInputStream inputStream = new PushbackInputStream(tmpFile != null ?
278                     new FileInputStream(tmpFile) : urlConnection.getInputStream(),
279                     MAX_STREAM_HEADER_SIZE)) {
280                 try {
281                     byte[] streamHeader = new byte[MAX_STREAM_HEADER_SIZE];
282                     int inLength;
283                     /* Peek into the first few bytes in the stream to check the header. */
284                     inLength = inputStream.read(streamHeader);
285                     if (inLength == -1) {
286                         inLength = 0;
287                     }
288                     inputStream.unread(streamHeader, 0, inLength);
289                     if (inLength >= MAX_STREAM_HEADER_SIZE) {
290                         if (Arrays.equals(streamHeader, ZIP_HEADER)) {
291                             //ZIP header bytes
292                             LOGGER.debug(DETECTED_PKZIP_FILE, filename);
293                             processZip(document, inputStream, pathUri, urlConnection);
294                         } else if (Arrays.equals(streamHeader, SEVENZ_HEADER)) {
295                             //7-zip header bytes
296                             LOGGER.debug(DETECTED_7Z_FILE, filename);
297                             processSevenzip(document, inputStream, pathUri, urlConnection);
298                         } else {
299                             throw new SnapDataException(String.format(
300                                     ERR_FORMAT_NOT_SUPPORTED, filename))
301                                     .withReason(ERR_FORMAT_NOT_SUPPORTED_REASON)
302                                     .withResolution(ERR_FORMAT_NOT_SUPPORTED_RES);
303                         }
304                     }
305                 } catch (SuggestViewAbortException | SnapDataException e) {
306                     throw e;
307                 } catch (Exception e) {
308                     handleException(e, ERROR_UNZIP_ZIP_FILE, filename, ERROR_UNZIP_ZIP_FILE,
309                             ERROR_UNZIP_ZIP_FILE_RES, documentUtility, errorViews, document);
310                 }
311             }
312         } catch (Throwable e) {
313             Throwable ex = e instanceof FailsafeException ? e.getCause() : e;
314             if (ex instanceof SuggestViewAbortException) {
315                 throw (SuggestViewAbortException) ex;
316             } else if (ex instanceof SnapDataException) {
317                 errorViews.write((SnapDataException) ex, document);
318             } else if (ex instanceof SnapException) {
319                 errorViews.write((SnapDataException) new SnapDataException(ex, ex.getMessage())
320                         .withReason(((SnapException) ex).getReason())
321                         .withResolution(((SnapException) ex).getResolution()), document);
322             } else {
323                 handleException(e, ERR_CANT_READ, filename, e.getMessage(),
324                         RESOLUTION_CANT_ACCESS, documentUtility, errorViews, document);
325             }
326         } finally {
327             statusReporter.reset();
328             if (urlConnection != null) {
329                 urlUtil.disconnect(urlConnection);
330             }
331             if (tmpFile != null) {
332                 tmpFile.delete();
333                 tmpFile = null;
334             }
335         }
336     }

377     private void processZip(final Document inputDocument, InputStream inputStream, URI uri,
378         URLConnection urlConnection) {
379         try (CountingInputStream counter = new CountingInputStream(inputStream);
380                 ZipArchiveInputStream zis = new ZipArchiveInputStream(
381                 new BufferedInputStream(counter))) {
382             ArchiveEntry entry;
383             while ((entry = zis.getNextEntry()) != null) {
384                 statusReporter.worked(counter.getByteCount());
385                 final ArchiveEntry zipEntry = entry;
386                 if (entry.isDirectory()) {
387                     LOGGER.debug(SKIPPING_DIRECTORY_ENTRY, zipEntry.getName());
388                     continue;
389                 }
390                 LOGGER.debug(READING_ZIP_ENTRY, zipEntry.getName());
391                 final BigInteger size = BigInteger.valueOf(zipEntry.getSize());
392                 statusReporter.begin(size.longValue(), "bytes")
393                         .notify(EXTRACT_FILE, zipEntry.getName());
394                 final Document header = getHeader(inputDocument, zipEntry.getName(), size,
395                         zipEntry.getLastModifiedDate(), uri, urlConnection);
396                 outputViews.write(new BinaryOutput() {
397                     long bytesRead = -1;
398 
399                     @Override
400                     public Document getHeader() {
401                         if (size.compareTo(BigInteger.ZERO) < 0 && bytesRead > -1) {
402                             ((Map<String, Object>) header.get())
403                                     .put(HttpHeaders.CONTENT_LENGTH.toLowerCase(),
404                                             BigInteger.valueOf(bytesRead));
405                         }
406                         return header;
407                     }
408 
409                     @Override
410                     public void write(final WritableByteChannel writeChannel)
411                             throws IOException {
412                         try (OutputStream os = Channels.newOutputStream(writeChannel)) {
413                             long limit;
414 
415                             if (isPreview) {
416                                 limit = Math.min(MAX_SUGGEST_FILE_SIZE, zipEntry.getSize());
417                             } else {
418                                 limit = zipEntry.getSize();
419                             }
420                             bytesRead = StatusReportingUtils.copy(statusReporter, zis, os, limit);
421                         } finally {
422                             statusReporter.done();
423                         }
424                     }
425                 });
426             }
427         } catch (IOException e) {
428             handleException(e, ERROR_UNZIP_ZIP_FILE, filename, ERROR_UNZIP_ZIP_FILE,
429                     ERROR_UNZIP_ZIP_FILE_RES, documentUtility, errorViews, inputDocument);
430         }

433     private void processSevenzip(final Document inputDocument, InputStream inputStream, URI uri,
434         URLConnection urlConnection) {
435         File tempZipFile = null;
436         FileOutputStream fo = null;
437         try {
438             statusReporter.notify(DOWNLOAD_FILE, filename);
439             tempZipFile = File.createTempFile(TEMP_7ZIP_FILE_PREFIX, TEMP_7ZIP_FILE_SUFFIX);
440             fo = new FileOutputStream(tempZipFile);
441             StatusReportingUtils.copy(statusReporter, inputStream, fo, -1);
442         } catch (IOException e) {
443             handleException(e, ERROR_CREATE_LOCAL_ZIP_FILE, filename,
444                     ERROR_CREATE_LOCAL_ZIP_FILE, ERROR_CREATE_LOCAL_ZIP_FILE_RES,
445                     documentUtility, errorViews, inputDocument);
446             deleteTempFile(tempZipFile);
447             return;
448         } finally {
449             IOUtils.closeQuietly(fo);
450             statusReporter.done();
451         }
452         try (final SevenZFile sevenZFile = new SevenZFile(tempZipFile)) {
453             SevenZArchiveEntry entry;
454             while ((entry = sevenZFile.getNextEntry()) != null) {
455                 if (entry.isDirectory()) {
456                     continue;
457                 }
458 
459                 final SevenZArchiveEntry sevenZipEntry = entry;
460                 final BigInteger size = BigInteger.valueOf(sevenZipEntry.getSize());
461                 statusReporter.begin(size.longValue(), "bytes")
462                         .notify(EXTRACT_FILE, sevenZipEntry.getName());
463                 final Document header = getHeader(inputDocument, sevenZipEntry.getName(), size,
464                         sevenZipEntry.getLastModifiedDate(), uri, urlConnection);
465                 outputViews.write(new BinaryOutput() {
466                     long total = 0;
467 
468                     @Override
469                     public Document getHeader() {
470                         // in an earlier version of this code the 'content-length' value would
471                         // be set in this method if it was initially unknown (size < 0). Since
472                         // this method isn't synced with the write() loop the number could change
473                         // during successive calls. To fix that the setter has been moved to the
474                         // tail of the write() method.
475                         return header;
476                     }
477 
478                     @Override
479                     public void write(final WritableByteChannel writeChannel)
480                             throws IOException {
481                         try (OutputStream os = Channels.newOutputStream(writeChannel)) {
482                             // note: we've seen files that would provide 3 GB of output in the
483                             // field.
484                             final int bufferSize = (int) ((size.longValue() <= 0) ? BUFFER_SIZE
485                                 : Math.min(BUFFER_SIZE, size.longValue()));
486                             byte[] buffer = new byte[bufferSize];
487                             int length;
488                             while ((length = sevenZFile.read(buffer)) > 0) {
489                                 total += length;
490                                 if (isPreview) {
491                                     // if preview, the file size cannot exceed 100MB
492                                     if (total > MAX_SUGGEST_FILE_SIZE) {
493                                         length -= total - MAX_SUGGEST_FILE_SIZE;
494                                         os.write(buffer, 0, length);
495                                         statusReporter.incrementalWork(length);
496                                         return;
497                                     }
498                                 }
499                                 os.write(buffer, 0, length);
500                                 statusReporter.incrementalWork(length);
501                             }
502 
503                             // if the file was written in streaming mode the size
504                             // of the entry may not have been known when the header
505                             // was written - size would have been -1. We had been
506                             // setting this value in getHeaders() but it could change
507                             // if we hadn't fully read the entry. To fix that it's
508                             // been moved here - the value will only be set once we
509                             // know the final value.
510                             if (size.longValue() < 0) {
511                                 ((Map<String, Object>) header.get())
512                                         .put(HttpHeaders.CONTENT_LENGTH.toLowerCase(),
513                                                BigInteger.valueOf(total));
514                             }
515                         } finally {
516                             statusReporter.done();
517                         }
518                     }

535     private Document getHeader(final Document inputDocument, String name, BigInteger size, Date
536             lastModifiedDate, URI uri, URLConnection urlConnection) {
537         Map<String, Object> header = new LinkedHashMap<>();
538         if (StringUtils.isNotBlank(filePathURI)) {
539             name = StringUtils.join(filePathURI, FORWARD_SLASH, name);
540         }
541         header.put(HttpHeaders.CONTENT_LOCATION.toLowerCase(), name);
542         String contentType = binaryUtils.guessContentTypeFromName(name);
543         header.put(HttpHeaders.CONTENT_TYPE.toLowerCase(), contentType);
544         if (size.compareTo(BigInteger.ZERO) > -1) {
545             header.put(HttpHeaders.CONTENT_LENGTH.toLowerCase(), size);
546         }
547         header.put(HttpHeaders.LAST_MODIFIED.toLowerCase(), new DateTime(lastModifiedDate,
548                 DateTimeZone.UTC));
549         header.put(ZIP_FILENAME, filename);
550 
551         // Add the rest of the url connection headers. Only add them if they are missing
552         // to be backwards compatible. The above headers were released before
553         // this code.
554         Map<String, Object> connHeader = binaryUtils.createHeader(uri, urlConnection);
555         for (Map.Entry<String, Object> entry : connHeader.entrySet()) {
556             header.putIfAbsent(entry.getKey(), entry.getValue());
557         }
558 
559         return documentUtility.newDocumentFor(inputDocument, header);
560     }
