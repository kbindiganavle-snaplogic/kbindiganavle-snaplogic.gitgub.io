267     @Override
268     protected void doWork() {
269         for (InputView inputView : inputViews) {
270             inputViewName = inputView.getName();
271             counter = 0;
272             for (BinaryInput binaryData : inputViews.binaryInputs(inputView)) {
273                 Document header = binaryData.getHeader();
274                 // evaluate File name property
275                 if (StringUtils.isBlank(filename)) {
276                     filename = filenameExpr.eval(header);
277                 }
278                 if (StringUtils.isBlank(filename) || filename.equals(DOT)) {
279                     errorViews.write((SnapDataException) new ExecutionException(ERR_EMPTY_FILENAME)
280                             .withReason(ERR_BLANK_FILENAME)
281                             .withResolution(RESOLUTION_EMPTY_FILENAME), header);
282                     return;
283                 }
284                 AccountType accountType = binaryUtils.getAccountType(account);
285                 uri = urlUtil.insertAuthority(filename, accountType, userInfo, UrlUtil.PATTERN);
286                 // evaluate retry properties
287                 numOfRetries = evalIntProperty(numOfRetriesExpr, header, MIN_RETRIES, MIN_RETRIES,
288                         errorViews);
289                 retryInterval = evalIntProperty(retryIntervalExpr, header, MIN_RETRY_INTERVAL,
290                         MIN_RETRY_INTERVAL, errorViews);
291                 fileFinder.setRetries(numOfRetries, retryInterval);
292                 // write binary data
293                 writeData(binaryData);
294                 if (writeResult == WriteResult.FAILED) {
295                     return;
296                 }
297             }
298         }
299         if (tmpFile != null) {
300             IOUtils.closeQuietly(zipOutput);
301             zipOutput = null;
302             // copy the zip data ro the target file with retries
303             try {
304                 ContextualRunnable runnable = context -> copyTmpFileToTarget();
305                 CheckedConsumer<Throwable> failedAttemptListener = e -> LOGGER.warn(
306                         "{} - Failed to write zip file: {}, {}",
307                         RETRY_SIGNATURE, filename, e.getMessage());
308                 failsafeExecute(runnable, failedAttemptListener, numOfRetries, retryInterval);
309             } catch (Throwable e) {
310                 Throwable ex = e instanceof FailsafeException ? e.getCause() : e;
311                 if (ex instanceof SuggestViewAbortException) {
312                     throw (SuggestViewAbortException) ex;
313                 } else if (ex instanceof SnapDataException) {
314                     errorViews.write((SnapDataException) ex);
315                 } else if (ex instanceof SnapException) {
316                     errorViews.write((SnapDataException)
317                             new SnapDataException(ex, ex.getMessage())
318                                     .withReason(((SnapException) ex).getReason())
319                                     .withResolution(((SnapException) ex).getResolution()));
320                 } else {
321                     handleException(ex, ERROR_WRITING_ZIP_FILE, filename,
322                             ex.getMessage(), RESOLUTION_ADDRESS_ISSUE, documentUtility,
323                             errorViews, null);
324                     writeResult = WriteResult.FAILED;
325                 }
326             }
327         }
328     }

330     private void writeData(BinaryInput binaryData) {
331         Document header = binaryData.getHeader();
332         env = getSasEnvProperties(advPropMap, header);
333         BinaryUtils.embedAuthenticationConfigIfNeeded(env, account);
334         env.put(HttpHeaders.CONTENT_TYPE, MediaType.ZIP.toString());
335 
336         try (InputStream inputStream = Channels.newInputStream(binaryData.getChannel())) {
337             if (numOfRetries > 0) {
338                 initTmpFileOutputStream();
339             } else {
340                 initOutputStream();
341             }
342             if (outputStream != null) {
343                 prepareZipOutputStream(header, inputStream);
344             }
345         } catch (SnapDataException e) {
346             errorViews.write(e, header);
347             writeResult = WriteResult.FAILED;
348         } catch (IOException e) {
349             handleException(e, ERROR_WRITING_ZIP_FILE, filename, e.getMessage(),
350                     RESOLUTION_ADDRESS_ISSUE, documentUtility, errorViews, header);
351             writeResult = WriteResult.FAILED;
352         }
353     }

355     private void prepareZipOutputStream(final Document header, final InputStream inputStream)
356             throws IOException {
357         counter++;
358         String zipEntryFilename;
359         InputView inputView = inputViews != null ? inputViews.get(inputViewName) : null;
360 
361         if (useInputViewLabel && inputView != null) {
362             zipEntryFilename = getFilenameFromInputViewLabel(inputView.getLabel(), counter);
363         } else if (header != null) {
364             zipEntryFilename = getFilenameFromHeader(header, inputViewName, counter);
365         } else {
366             zipEntryFilename = StringUtils.join(inputViewName, counter);
367         }
368 
369         ZipEntry entry = new ZipEntry(formatEntryName(baseDirectory, zipEntryFilename));
370         long time = getModifiedTime(header);
371         if (time > 0) {
372             entry.setTime(time);
373         }
374         zipOutput.putNextEntry(entry);
375         IOUtils.copyLarge(inputStream, zipOutput);
376     }

429                     filePath = uri.getPath();

435             }
436         }
437         if (filePath != null) {
438             if (filePath.indexOf(DOT) > 0) {
439                 filename = filePath;
440             } else {
441                 String fileExt = getFileExtension(header);
442                 filename = StringUtils.join(filePath, fileExt);
443             }
444         } else {
445             String fileExt = getFileExtension(header);
446             filename = StringUtils.join(inputViewName, counter, fileExt);
447         }
448         return filename;

503     /**
504      * Initializes the output stream for the file. Output stream is kept open until all binary
505      * inputs are read.
506      *
507      * @throws IOException
508      * @throws MalformedURLException
509      */
510     protected void initOutputStream() throws IOException {
511         if (zipOutput == null) {
512             if (fileAction == FileAction.OVERWRITE) {
513                 // if file action is overwrite, do not check if the file exists
514                 writeResult = WriteResult.OVERWRITTEN;
515             } else {
516                 // check if the file exists
517                 FileSystem fileSystem = fileFinder.getFileSystemFor(uri, filename, env);
518                 if (fileSystem == null) {
519                     writeResult = WriteResult.FAILED;
520                     throw new SnapDataException(ERR_FILESYSTEM_NOT_FOUND)
521                             .formatWith(filename)
522                             .withReason(REASON_FILESYSTEM_NOT_FOUND)
523                             .withResolution(RESOLUTION_ADDRESS_ISSUE);
524                 }
525                 try {
526                     Path path = fileFinder.createPath(fileSystem, uri, false);
527                     if (Files.exists(path, LinkOption.NOFOLLOW_LINKS)) {
528                         switch (fileAction) {
529                             case IGNORE:
530                                 writeResult = WriteResult.IGNORED;
531                                 return;
532                             case ERROR:
533                                 writeResult = WriteResult.FAILED;
534                                 throw new SnapDataException(ERR_FILE_EXISTS)
535                                         .formatWith(filename)
536                                         .withReason(REASON_FILE_EXISTS)
537                                         .withResolution(RESOLUTION_FILE_EXISTS);
538                             default:
539                                 writeResult = WriteResult.FAILED;
540                                 throw new SnapDataException(ERR_UNSUPPORTED_FILE_ACTION)
541                                         .formatWith(fileAction.toString())
542                                         .withReason(REASON_UNSUPPORTED_FILE_ACTION)
543                                         .withResolution(RESOLUTION_UNSUPPORTED_FILE_ACTION);
544                         }
545                     } else {
546                         writeResult = WriteResult.CREATED;
547                     }
548                 } finally {
549                     fileFinder.closeFileSystem(fileSystem, filename);
550                 }
551             }
552             // get output stream
553             urlConnection = jfsUtils.openURLConnection(uri);
554             if (urlConnection instanceof S3UrlConnection && cannedAccessControlList != null) {
