256     @Override
257     protected void process(Document document, String inputViewName) {
258         // init Snap properties and account
259         String containerName = containerExpr.eval(document);
260         String blobName = blobExpr.eval(document);
261         Object value = expireInHoursExpr.eval(document);
262         int expireInHours = value instanceof BigInteger ? ((BigInteger) value).intValue() : ZERO;
263         value = expireInMinutesExpr.eval(document);
264         int expireInMinutes = value instanceof BigInteger ? ((BigInteger) value).intValue() : ONE;
265         CloudStorageAccount cloudStorageAccount = getCloudStorageAccount(account);
266         // get CloudBlobContainer object
267         CloudBlobClient cloudBlobClient = cloudStorageAccount.createCloudBlobClient();
268         CloudBlobContainer container = null;
269         try {
270             container = StringUtils.isBlank(containerName) ? null :
271                     cloudBlobClient.getContainerReference(containerName);
272         } catch (URISyntaxException | StorageException e) {
273             errorViews.write((SnapDataException)
274                     new SnapDataException(e, ERR_CONTAINER_NOT_FOUND)
275                     .formatWith(containerName)
276                     .withResolution(RESOLUTION_ADDRESS_ISSUE), document);
277         }
278         // Create Shared Access Policy.
279         SharedAccessBlobPolicy sasBlobPolicy = new SharedAccessBlobPolicy();
280         SharedAccessAccountPolicy sasAccountPolicy = new SharedAccessAccountPolicy();
281         if (expireInHours > 0 || expireInMinutes > 0) {
282             // Create a UTC Gregorian calendar value.
283             GregorianCalendar calendar = new GregorianCalendar(
284                     TimeZone.getTimeZone(UTC));
285             // Specify the current time as the start time for the shared access
286             // signature.
287             calendar.setTime(new Date());
288             sasBlobPolicy.setSharedAccessStartTime(calendar.getTime());
289             sasAccountPolicy.setSharedAccessStartTime(calendar.getTime());
290             // Use the start time plus n hours and m minutes as the end time for the shared
291             // access signature.
292             calendar.add(Calendar.HOUR, expireInHours);
293             calendar.add(Calendar.MINUTE, expireInMinutes);
294             Date expiryTime = calendar.getTime();
295             sasBlobPolicy.setSharedAccessExpiryTime(expiryTime);
296             sasAccountPolicy.setSharedAccessExpiryTime(expiryTime);
297         }
298         // Set permissions
299         Set<SharedAccessAccountPermissions> accounPermissions = new HashSet<>(4);
300         for (SharedAccessBlobPermissions permission : blobPermissions) {
301             accounPermissions.add(SharedAccessAccountPermissions.valueOf(permission.name()));
302         }
303         sasBlobPolicy.setPermissions(EnumSet.copyOf(blobPermissions));
304         sasAccountPolicy.setPermissions(EnumSet.copyOf(accounPermissions));
305 
306         Map<String , Object> data = new LinkedHashMap<>();
307         String sas;
308         String sasUri;
309         if (StringUtils.isBlank(containerName)) {
310             try {
311                 // Generate Account SAS
312                 sasAccountPolicy.setServices(EnumSet.of(SharedAccessAccountService.BLOB));
313                 sasAccountPolicy.setResourceTypes(EnumSet.of(
314                         SharedAccessAccountResourceType.CONTAINER,
315                         SharedAccessAccountResourceType.OBJECT,
316                         SharedAccessAccountResourceType.SERVICE
317                 ));
318                 sas = cloudStorageAccount.generateSharedAccessSignature(sasAccountPolicy);
319                 sasUri = String.format(FORMAT_APPEND_QUERY, blobEndpoint, sas);
320                 data.put(SAS_TYPE, ACCOUNT_SAS);
321                 data.put(SAS_URI, sasUri);
322             } catch (StorageException | InvalidKeyException e) {
323                 errorViews.write((SnapDataException)
324                         new SnapDataException(e, ERR_GENERATE_ACCOUNT_SAS)
325                                 .withResolution(RESOLUTION_ADDRESS_ISSUE), document);
326             }
327         } else if (StringUtils.isBlank(blobName)) {
328             try {
329                 if (storeAccessPolicy) {
330                     // Create the container permissions.
331                     BlobContainerPermissions containerPermissions = new BlobContainerPermissions();
332                     // Turn public access to the container off.
333                     containerPermissions.setPublicAccess(BlobContainerPublicAccessType.CONTAINER);
334                     // up to 5 policies can be stored in a container
335                     containerPermissions.getSharedAccessPolicies().put(SNAP_POLICY, sasBlobPolicy);
336                     container.uploadPermissions(containerPermissions);
337                     sas = container.generateSharedAccessSignature(sasBlobPolicy, SNAP_POLICY);
338                     String containerPath = JfsUtils.concat(blobEndpoint, containerName);
339                     sasUri = String.format(FORMAT_APPEND_QUERY, encode(containerPath), sas);
340                     data.put(SAS_TYPE, STORED_ACCESS_POLICY);
341                     data.put(SAS_URI, sasUri);
342                 } else {
343                     // Generate Service SAS (Shared Access Signature) on a container
344                     sas = container.generateSharedAccessSignature(sasBlobPolicy, null);
345                     String containerPath = JfsUtils.concat(blobEndpoint, containerName);
346                     sasUri = String.format(FORMAT_APPEND_QUERY, encode(containerPath), sas);
347                     data.put(SAS_TYPE, SERVICE_SAS_ON_CONTAINER);
348                     data.put(SAS_URI, sasUri);
349                 }
350             } catch (StorageException | InvalidKeyException e) {
351                 errorViews.write((SnapDataException)
352                         new SnapDataException(e, ERR_GENERATE_SAS_ON_CONTAINER)
353                                 .formatWith(containerName)
354                                 .withResolution(RESOLUTION_ADDRESS_ISSUE), document);
355             }
356         } else {
357             // Generate Service SAS on a blob
358             try {
359                 CloudBlockBlob cloudBlockBlob = container.getBlockBlobReference(blobName);
360                 sas = cloudBlockBlob.generateSharedAccessSignature(sasBlobPolicy, null, null);
361                 String blobPath = JfsUtils.concat(blobEndpoint, containerName, blobName);
362                 sasUri = String.format(FORMAT_APPEND_QUERY, encode(blobPath), sas);
363                 data.put(SAS_TYPE, SERVICE_SAS_ON_BLOB);
364                 data.put(SAS_URI, sasUri);
365             } catch (StorageException | URISyntaxException | InvalidKeyException e) {
366                 errorViews.write((SnapDataException)
367                         new SnapDataException(e, ERR_GENERATE_SAS_ON_BLOB)
368                                 .formatWith(blobName)
369                                 .withResolution(RESOLUTION_ADDRESS_ISSUE), document);
370             }
371         }
372         // When the generated SAS is used straight away, we get an authorization error
373         // intermittently. Sleeping for 2 seconds seems to fix the issue.
374         JfsUtils.sleepNoInterrupt(2, TimeUnit.SECONDS, LOG);
375         outputViews.write(documentUtility.newDocument(data), document);
376     }

380     private void suggestContainers(SuggestionBuilder suggestionBuilder, Account account) {
381         CloudBlobClient cloudBlobClient = getCloudBlobClient(account);
382         List<String> suggestionList = new ArrayList<>();
383         for (CloudBlobContainer container : cloudBlobClient.listContainers()) {
384             suggestionList.add(container.getName());
385         }
386         if (CollectionUtils.isEmpty(suggestionList)) {
387             // there should be at least one empty entry in the suggestions
388             suggestionList.add(StringUtils.EMPTY);
389         }
390         String[] suggestions = suggestionList.toArray(new String[suggestionList.size()]);
391         suggestionBuilder.node(CONTAINER_PROP).suggestions(suggestions);
392     }

394     private void suggestBlobs(SuggestionBuilder suggestionBuilder, Account account,
395             PropertyValues propertyValues) {
396         CloudBlobClient cloudBlobClient = getCloudBlobClient(account);
397         String containerName = propertyValues.get(CONTAINER_PROP);
398         List<String> suggestionList = new ArrayList<>();
399         if (StringUtils.isNotBlank(containerName)) {
400             CloudBlobContainer container;
401             try {
402                 container = cloudBlobClient.getContainerReference(containerName);
403             } catch (URISyntaxException | StorageException e) {
404                 throw new ConfigurationException(e, ERR_CONTAINER_NOT_FOUND)
405                         .formatWith(containerName)
406                         .withResolution(RESOLUTION_ADDRESS_ISSUE);
407             }
408             int index = containerName.length() + 2;
409             String prefix = propertyValues.get(BLOB_PROP);
410             prefix = StringUtils.startsWith(prefix, URL_SEPARATOR.concat(containerName)) ?
411                     prefix.substring(index) : prefix;
412             for (ListBlobItem item : container.listBlobs(prefix)) {
413                 suggestionList.add(item.getUri().getPath().substring(index));
414             }
415         }
416         if (CollectionUtils.isEmpty(suggestionList)) {
417             // there should be at least one empty entry in the suggestions
418             suggestionList.add(StringUtils.EMPTY);
419         }
420         String[] suggestions = suggestionList.toArray(new String[suggestionList.size()]);
421         suggestionBuilder.node(BLOB_PROP).suggestions(suggestions);
422     }

424     private CloudStorageAccount getCloudStorageAccount(Account account) {
425         // get account credential
426         Map<String, String> properties;
427         if (account instanceof  FileSystemPropertiesProvider) {
428             account.connect();
429             properties = ((FileSystemPropertiesProvider) account).getRequestProperties();
430         } else {
431             throw new ConfigurationException(ERR_INVALID_ACCOUNT)
432                     .formatWith(account.getClass().getSimpleName())
433                     .withReason(REASON_INVALID_ACCOUNT)
434                     .withResolution(RESOLUTION_ADDRESS_ISSUE);
435         }
436         String accountName = properties.get(ACCOUNT_NAME);
437         String primaryAccessKey = properties.get(PRIMARY_ACCESS_KEY);
438         String blobEndpoint = properties.get(BLOB_ENDPOINT);
439         // format storage connection string
440         String storageConnection = String.format(FORMAT_STORAGE_CONNECTION_STRING,
441                 PROTOCOL_HTTPS, accountName, primaryAccessKey);
442         if (StringUtils.isNotBlank(blobEndpoint)) {
443             storageConnection = storageConnection.concat(
444                     String.format(FORMAT_APPEND, BLOB_ENDPOINT, blobEndpoint));
445         }
446         // get and return CloudStorageAccount object
447         try {
448             return CloudStorageAccount.parse(storageConnection);
449         } catch (URISyntaxException | InvalidKeyException e) {
450             throw new ConfigurationException(e, ERR_INVALID_ACCOUNT)
451                     .formatWith(account.getClass().getSimpleName())
452                     .withResolution(RESOLUTION_ADDRESS_ISSUE);
453         }
454     }
