114             PGPEncryptedDataList pgpEncryptedDataList, InputStream keyIn, char[] passwd)
115             throws IOException, NoSuchProviderException, PGPException {
116         InputStream inputStream = null;
117 
118         // find the secret key
119         Iterator it = pgpEncryptedDataList.getEncryptedDataObjects();
120         PGPPrivateKey sKey = null;
121         PGPPublicKeyEncryptedData pbe = null;
122         PGPSecretKeyRingCollection pgpSec = new PGPSecretKeyRingCollection(
123                 PGPUtil.getDecoderStream(keyIn), new BcKeyFingerprintCalculator());
124         while (sKey == null && it.hasNext()) {
125             pbe = (PGPPublicKeyEncryptedData) it.next();
126             sKey = findSecretKey(pgpSec, pbe.getKeyID(), passwd);
127         }
128         if (sKey == null) {
129             if (pbe != null) {
130                 throw new IllegalArgumentException(
131                         ERR_SECRET_KEY_NOT_FOUND + ": \"" + Long.toHexString(pbe.getKeyID())
132                                 .toUpperCase() + '"');
133             } else {
134                 throw new IllegalArgumentException(ERR_SECRET_KEY_NOT_FOUND);
135             }
136         }
137         PublicKeyDataDecryptorFactory decryptorFactory = new BcPublicKeyDataDecryptorFactory(sKey);
138         InputStream clear = pbe.getDataStream(decryptorFactory);
139         PGPObjectFactory plainFact = new PGPObjectFactory(clear, new BcKeyFingerprintCalculator());
140         Object message = plainFact.nextObject();
141         if (message instanceof PGPCompressedData) {
142             PGPCompressedData cData = (PGPCompressedData) message;
143             PGPObjectFactory pgpFact = new PGPObjectFactory(cData.getDataStream(),
144                     new BcKeyFingerprintCalculator());
145             message = pgpFact.nextObject();
146         }
147         if (message instanceof PGPLiteralData) {
148             PGPLiteralData ld = (PGPLiteralData) message;
149             inputStream = ld.getInputStream();
150         } else if (message instanceof PGPOnePassSignatureList) {
151             throw new PGPException(ERR_PGP_SIGNED_MESSAGE);
152         } else {
153             throw new PGPException(ERR_PGP_UNKNOWN_MESSAGE_TYPE);
154         }
155         return Pair.of(inputStream != null ? inputStream : IOUtils.toInputStream("", UTF_8), pbe);
156     }

206                 signatureList = (PGPSignatureList) message;

209             }

229                     if (!ops.verify(signature)) {
230                         throw new PGPException(ERR_SIGNATURE_VERIFICATION_FAILED);
231                     }
232                 }
233             }
234         }
235         if (!skipVerification && publicKey == null) {
236             throw new PGPException(ERR_PUBLIC_KEY_INVALID);
237         }
238         return Pair.of(new ByteArrayInputStream(output), pbe);

256     public static void encryptFile(OutputStream targetOut, InputStream sourceIn, InputStream keyIn,
257             boolean withIntegrityCheck, String encryptedFilename, int symmetricKeyAlgorithmTag,
258             final String encryptionKeyId)
259             throws IOException, PGPException {
260         PGPPublicKey publicKey;
261         try {
262             publicKey = readPublicKey(keyIn, encryptionKeyId);
263             LOG.debug(LOG_PUBLIC_KEY_STRENGTH, publicKey.getBitStrength());
264             LOG.debug(LOG_PUBLIC_KEY_ALGORITHM, publicKey.getAlgorithm());
265             LOG.debug(LOG_PUBLIC_KEY_VERSION, publicKey.getVersion());
266         } catch (PGPException e) {
267             throw new ExecutionException(e, ERR_PGP_PUBLIC_KEY_READ).withReason(
268                     ERR_PGP_PUBLIC_KEY_READ_REASON).withResolution(
269                     ERR_PGP_PUBLIC_KEY_READ_RESOLUTION);
270         }
271 
272         if (isKeyExpired(publicKey)) {
273             throw new PGPException(ERR_PGP_PUBLIC_KEY_EXPIRED);
274         }
275         PGPEncryptedDataGenerator encryptedGen = null;
276         PGPCompressedDataGenerator compressedGen = null;
277         PGPLiteralDataGenerator literalGen = null;
278 
279         // Data flow: source -> literal -> compressed -> encrypted -> target
280         try {
281             // encrypted data stream -> target data stream
282             byte[] buffer = new byte[BUFFER_SIZE];
283             encryptedGen = new PGPEncryptedDataGenerator(
284                     new JcePGPDataEncryptorBuilder(symmetricKeyAlgorithmTag)
285                             .setWithIntegrityPacket(withIntegrityCheck)
286                             .setSecureRandom(new SecureRandom())
287                             .setProvider(BOUNCY_CASTLE_PROVIDER));
288             encryptedGen.addMethod(new JcePublicKeyKeyEncryptionMethodGenerator(publicKey)
289                     .setProvider(BOUNCY_CASTLE_PROVIDER));
290             OutputStream encryptedOut = encryptedGen.open(targetOut, buffer);
291 
292             // compressed data stream -> encrypted data stream
293             compressedGen = new PGPCompressedDataGenerator(PGPCompressedDataGenerator.ZIP);
294             OutputStream compressedOut = compressedGen.open(encryptedOut);
295 
296             // literal data stream -> compressed data stream
297             byte[] buffer2 = new byte[BUFFER_SIZE];
298             literalGen = new PGPLiteralDataGenerator();
299             encryptedFilename = StringUtils.isBlank(encryptedFilename)
300                     ? DEFAULT_ENCRYPTED_FILENAME : encryptedFilename;
301             OutputStream literalOut = literalGen.open(compressedOut,
302                     PGPLiteralDataGenerator.BINARY, encryptedFilename, new Date(), buffer2);
303 
304             // source stream -> literal data stream
305             IOUtils.copy(sourceIn, literalOut);
306         } finally {
307             if (literalGen != null) {
308                 literalGen.close();
309             }
310             if (compressedGen != null) {
311                 compressedGen.close();
312             }
313             if (encryptedGen != null) {
314                 encryptedGen.close();
315             }
316         }
317     }

330     static void encryptAndSign(
331             InputStream plainInput, OutputStream encryptedOutput, String stringPrivateKey,
332             String password, PGPPublicKey publicKey, final String encryptionKeyId)
333             throws IOException, PGPException {
334 
335         PGPSecretKey secretKey;
336         try {
337             secretKey = readPrivateKey(IOUtils.toInputStream(stringPrivateKey, UTF_8), password,
338                     encryptionKeyId);
339         } catch (Exception e) {
340             throw new PGPException(ERR_PGP_PRIVATE_KEY_NOT_FOUND_IN_KEY_RING, e);
341         }
342 
343         if (isKeyExpired(publicKey)) {
344             throw new ExecutionException(ERR_PGP_PUBLIC_KEY_EXPIRED);
345         }
346         // setup data encryptor & generator
347         BcPGPDataEncryptorBuilder encryptor = new
348                 BcPGPDataEncryptorBuilder(PGPEncryptedData.TRIPLE_DES);
349         encryptor.setWithIntegrityPacket(true);
350         encryptor.setSecureRandom(new SecureRandom());
351 
352         // add public key recipients
353         PGPEncryptedDataGenerator encGen = new PGPEncryptedDataGenerator(encryptor);
354         encGen.addMethod(new BcPublicKeyKeyEncryptionMethodGenerator(publicKey));
355 
356         OutputStream encryptedOut = encGen.open(encryptedOutput, new byte[BUFFER_SIZE]);
357 
358         // setup compressed data generator
359         PGPCompressedDataGenerator compGen = new PGPCompressedDataGenerator(PGPCompressedData.ZIP);
360         OutputStream compressedOut = compGen.open(encryptedOut, new byte[BUFFER_SIZE]);
361 
362         PBESecretKeyDecryptor pbeSecretKeyDecryptor = new JcePBESecretKeyDecryptorBuilder()
363                 .setProvider(BOUNCY_CASTLE_PROVIDER).build(password.toCharArray());
364         PGPPrivateKey privateKey = secretKey.extractPrivateKey(pbeSecretKeyDecryptor);
365 
366         PGPSignatureGenerator signatureGenerator = new PGPSignatureGenerator(
367                 new JcaPGPContentSignerBuilder(
368                         secretKey.getPublicKey().getAlgorithm(),
369                         HashAlgorithmTags.SHA1).setProvider(BOUNCY_CASTLE_PROVIDER));
370         signatureGenerator.init(PGPSignature.BINARY_DOCUMENT, privateKey);
371 
372         Iterator<String> it = secretKey.getPublicKey().getUserIDs();
373         if (it.hasNext()) {
374             PGPSignatureSubpacketGenerator spGen = new PGPSignatureSubpacketGenerator();
375             spGen.addSignerUserID(false, it.next());
376             signatureGenerator.setHashedSubpackets(spGen.generate());
377         }
378 
379         signatureGenerator.generateOnePassVersion(false).encode(compressedOut);
380 
381         // Initialize literal data generator
382         PGPLiteralDataGenerator literalGen = new PGPLiteralDataGenerator();
383         OutputStream literalOut = literalGen.open(
384                 compressedOut,
385                 PGPLiteralData.BINARY,
386                 "",
387                 new Date(),
388                 new byte[PGPIntegrationUtil.BUFFER_SIZE]);
389 
390         // read the "in" stream, compress, encrypt and write to the "out" stream
391         // this must be done if clear data is bigger than the buffer size
392         // but there are other ways to optimize...
393         byte[] buf = new byte[PGPIntegrationUtil.BUFFER_SIZE];
394         int len;
395         while ((len = plainInput.read(buf)) > 0) {
396             literalOut.write(buf, 0, len);
397             signatureGenerator.update(buf, 0, len);
398         }
399 
400         literalGen.close();
401 
402         // generate the signature, compress, encrypt and write to the "out" stream
403         signatureGenerator.generate().encode(compressedOut);
404         compGen.close();
405         encGen.close();
406     }

408     static void sign(InputStream inputStream, OutputStream outputStream, PGPSecretKey signingKey,
409             String password, int signatureAlgorithm, boolean detachedMode, boolean armorMode)
410             throws PGPException, IOException {
411         PGPCompressedDataGenerator compressedDataGenerator = null;
412         BCPGOutputStream binaryOutputStream = null;
413         PGPLiteralDataGenerator literalGen = null;
414         OutputStream literalOutputStream = null;
415         if (armorMode) {
416             outputStream = new ArmoredOutputStream(outputStream);
417         }
418         // set up base signing pieces
419         PBESecretKeyDecryptor keyDecryptor = new JcePBESecretKeyDecryptorBuilder()
420                 .setProvider(BOUNCY_CASTLE_PROVIDER).build(password.toCharArray());
421         PGPPrivateKey privateKey = signingKey.extractPrivateKey(keyDecryptor);
422         PGPPublicKey publicKey = signingKey.getPublicKey();
423 
424         PGPSignatureGenerator sigGenerator = new PGPSignatureGenerator(
425                 new JcaPGPContentSignerBuilder(publicKey.getAlgorithm(), signatureAlgorithm)
426                         .setProvider(BOUNCY_CASTLE_PROVIDER));
427 
428         sigGenerator.init(PGPSignature.BINARY_DOCUMENT, privateKey);
429 
430         // set up file compression
431         if (!detachedMode) {
432             Iterator<String> it = publicKey.getUserIDs();
433             if (it.hasNext()) {
434                 PGPSignatureSubpacketGenerator spGen = new PGPSignatureSubpacketGenerator();
435                 spGen.addSignerUserID(false, it.next());
436                 sigGenerator.setHashedSubpackets(spGen.generate());
437             }
438             compressedDataGenerator = new PGPCompressedDataGenerator(
439                     PGPCompressedData.ZLIB);
440             binaryOutputStream = new BCPGOutputStream(
441                     compressedDataGenerator.open(outputStream));
442             sigGenerator.generateOnePassVersion(false).encode(binaryOutputStream);
443             literalGen = new PGPLiteralDataGenerator();
444             literalOutputStream = literalGen.open(binaryOutputStream, PGPLiteralData.BINARY,
445                     "", new Date(), new byte[PGPIntegrationUtil.BUFFER_SIZE]);
446         } else {
447             binaryOutputStream = new BCPGOutputStream(outputStream);
448         }
449 
450         // actually do the signature
451         byte[] buf = new byte[PGPIntegrationUtil.BUFFER_SIZE];
452         int len;
453         while ((len = inputStream.read(buf)) > 0) {
454             if (literalOutputStream != null) {
455                 literalOutputStream.write(buf, 0, len);
456             }
457             sigGenerator.update(buf, 0, len);
458         }
459         if (literalGen != null) {
460             literalGen.close();
461         }
462         sigGenerator.generate().encode(binaryOutputStream);
463         if (compressedDataGenerator != null) {
464             compressedDataGenerator.close();
465         }
466         if (armorMode) {
467             // we only want to close the Armor Output Stream, it doesn't close the internal stream
468             outputStream.close();
469         }
470     }

472     static PGPSecretKey getSigningKey(String privateKey, String phrase)
473             throws IOException, PGPException {
474         PGPSecretKeyRingCollection pgpSec = new PGPSecretKeyRingCollection(
475                 PGPUtil.getDecoderStream(IOUtils.toInputStream(privateKey, UTF_8)),
476                 new BcKeyFingerprintCalculator());
477         Security.addProvider(BOUNCY_CASTLE_PROVIDER);
478         Iterator<PGPSecretKeyRing> keyRingIter = pgpSec.getKeyRings();
479         Map<Date, PGPSecretKey> signingKeys = new HashMap<>();
480         PGPPublicKey masterPublicKey = null;
481         while (keyRingIter.hasNext()) {
482             PGPSecretKeyRing keyRing = keyRingIter.next();
483             Iterator<PGPSecretKey> keyIter = keyRing.getSecretKeys();
484             while (keyIter.hasNext()) {
485                 PGPSecretKey key = keyIter.next();
486                 if (key.isMasterKey()) {
487                     masterPublicKey = key.getPublicKey();
488                 }
489                 if (!isKeyExpired(key.getPublicKey(), masterPublicKey) && isSigningKey(key)) {
490                     signingKeys.put(key.getPublicKey().getCreationTime(), key);
491                 }
492             }
493         }
494         Optional<Map.Entry<Date, PGPSecretKey>> signingKey = signingKeys.entrySet().stream()
495                 .max(Map.Entry.comparingByKey());
496         if (signingKey.isEmpty()) {
497             throw new PGPException(ERR_PGP_NO_VALID_SIGNING_KEY);
498         }
499         PBESecretKeyDecryptor decryptor = new JcePBESecretKeyDecryptorBuilder()

620         if (key.isMasterKey()) {
621             return true;
622         }
623         if (!key.isSigningKey()) {
624             return false;

627                 .getSignaturesOfType(PGPSignature.SUBKEY_BINDING);
