225     public void process(Document document, String inputViewName) throws ExecutionException {
226         // resolve numOfRetries and retryInterval
227         numOfRetries = evalIntProperty(numOfRetriesExpr, document, MIN_RETRIES, MIN_RETRIES,
228                 errorViews);
229         retryInterval = evalIntProperty(retryIntervalExpr, document, MIN_RETRY_INTERVAL,
230                 MIN_RETRY_INTERVAL, errorViews);
231         fileFinder.setRetries(numOfRetries, retryInterval);
232         // init exitOnFirstMatches
233         exitOnFirstMatches = evalExitOnFirstMatches(document);
234         // get polling interval in seconds and polling timeout in milliseconds
235         int interval, timeout;
236         try {
237             interval = evalIntProperty(intervalExp, document, POLLING_INTERVAL_DEFAULT_SEC,
238                     MIN_POLLING_INTERVAL_SEC, errorViews);
239             timeout = evalIntProperty(timeoutExp, document, POLLING_TIMEOUT_DEFAULT_MIN,
240                     -1, errorViews);
241         } catch (SnapDataException e) {
242             errorViews.write(e, document);
243             return;
244         }
245         long timeoutInMs = 0;
246         if (timeout > 0) {
247             switch (timeUnit) {
248                 case HOURS:
249                     timeoutInMs = timeout * 3600000;
250                     break;
251                 case MINUTES:
252                     timeoutInMs = timeout * 60000;
253                     break;
254                 case SECONDS:
255                     timeoutInMs = timeout * 1000;
256                     break;
257                 default:
258                     throw new ConfigurationException(ERR_UNSUPPORTED_TIME_UNIT)
259                             .formatWith(timeUnit.toString())
260                             .withResolutionAsDefect();
261             }
262             if (interval > timeoutInMs / 1000) {
263                 throw new ConfigurationException(ERR_INVALID_VALUE)
264                         .formatWith(interval, POLLING_INTERVAL_LABEL)
265                         .withReason(REASON_POLLING_INTERVAL)
266                         .withResolution(RESOLUTION_ADDRESS_ISSUE);
267             }
268         }
269         // start polling
270         long startTimeInMs = System.currentTimeMillis();
271         pollingLoop(document, interval, timeout, timeoutInMs, startTimeInMs);
272     }

274     void pollingLoop(Document document, int interval, int timeout, long timeoutInMs,
275             long startTimeInMs) {
276         try {
277             while (looping) {
278                 pollInBackground(document, timeoutInMs);
279                 if (timeout == 0) {
280                     return; // if timeout is 0, it executes only once
281                 }
282                 try {
283                     synchronized (this) {
284                         if (looping) {
285                             this.wait(TimeUnit.SECONDS.toMillis(interval));
286                         }
287                     }
288                 } catch (InterruptedException e) {
289                     return;
290                 }
291                 // if timeout is negative, it runs forever
292                 if (timeout < 0 || (System.currentTimeMillis() - startTimeInMs) < timeoutInMs) {
293                     continue;
294                 }
295                 return;
296             }
297         } finally {
298             LOGGER.info(EXITED_POLLING_LOOP);
299         }
300     }

306     void pollInBackground(Document document, long timeoutInMs) {
307         Future<Void> future = pollingExecutor.submit(() -> poll(document), null);
308         long effectiveTimeoutMs = getEffectiveTimeout(timeoutInMs);
309         try {
310             future.get(effectiveTimeoutMs, TimeUnit.MILLISECONDS);

315             LOGGER.warn(ERR_POLLING_EXCEPTION, e);

319     void poll(Document document) {
320         // validate directory URI
321         String directory = dirExp.eval(document);
322         // If the Directory property is blank, it is regarded as sldb current project.
323         // Make sure the directory ends with "/".
324         if (!StringUtils.endsWith(directory, URL_SEPARATOR)) {
325             directory = StringUtils.join(directory, URL_SEPARATOR);
326         }
327         try {
328             String filter = filterExp.eval(document);
329             URI pathUri = urlUtil.insertAuthority(directory, accountType, userInfo,
330                     UrlUtil.PATTERN);
331             Map<String, String> env = getSasEnvProperties(advPropMap, document);
332             BinaryUtils.embedAuthenticationConfigIfNeeded(env, account);
333             FileSystem fileSystem = fileFinder.getFileSystemFor(pathUri, directory, env);
334             if (fileSystem == null) {
335                 return;
336             }
337             final String dirFinal = directory;
338             ContextualCallable<List<String>> callable =
339                     context -> poll(fileSystem, pathUri, dirFinal, filter);
340             CheckedConsumer<Throwable> failedAttemptListener = e -> LOGGER.warn(
341                     "{} - Failed to poll directory: {}, {}",
342                     RETRY_SIGNATURE, dirFinal, e.getMessage());
343             List<String> files = failsafeExecute(callable, failedAttemptListener,
344                     numOfRetries, retryInterval);
345             for (String filename : files) {
346                 Map<String, String> map = new LinkedHashMap<>();
347                 map.put(PATH, filename);
348                 outputViews.write(documentUtility.newDocument(map), document);
349             }
350             if (exitOnFirstMatches && CollectionUtils.isNotEmpty(files)) {
351                 looping = false;
352             }
353         } catch (Throwable e) {
354             Throwable ex = e instanceof FailsafeException ? e.getCause() : e;
355             if (ex instanceof SuggestViewAbortException) {
356                 throw (SuggestViewAbortException) ex;
357             } else if (ex instanceof SnapDataException) {
358                 errorViews.write((SnapDataException) ex, document);
359             } else if (ex instanceof SnapException) {
360                 errorViews.write((SnapDataException) new SnapDataException(ex, ex.getMessage())
361                         .withReason(((SnapException) ex).getReason())
362                         .withResolution(((SnapException) ex).getResolution()), document);
363             } else {
364                 errorViews.write((SnapDataException) new SnapDataException(ex, ERR_ACCESS_DIR)
365                         .formatWith(directory)
366                         .withReason(ex.getMessage())
367                         .withResolution(RESOLUTION_ADDRESS_ISSUE), document);
368             }
369         }
370     }

372     protected List<String> poll(FileSystem fileSystem, URI uri, String directory, String filter)
373             throws IOException {
374         Path path = null;
375         try {
376             if (URL_PROTOCOL_FILE.equalsIgnoreCase(uri.getScheme())) {
377                 path = FileFinder.getLocalPath(fileSystem, uri);
378             } else {
379                 path = fileSystem.getPath(uri.toString());
380             }
381             if (!Files.isDirectory(path)) {
382                 throw new SnapDataException(DIR_BROWSER_NOT_FILE_MSG)
383                         .withReason(DIR_BROWSER_NOT_FILE_REASON)
384                         .withResolution(DIR_BROWSER_NOT_FILE_RESOLUTION);
385             }
386             // create a matcher for glob pattern
387             final PathMatcher matcher = FileSystems.getDefault().getPathMatcher(StringUtils.join(
388                     GLOB_PATTERN, filter));
389             DirectoryStream.Filter<Path> pathFilter = new DirectoryStream.Filter<Path>() {
390                 public boolean accept(Path file) throws IOException {
391                     String pathStr = file instanceof SftpPath ? ((SftpPath) file).getPath() :
392                             file.toUri().getPath();
393                     if (file instanceof S3Path) {
394                         return acceptS3Path(uri.getPath(), pathStr, matcher);
395                     }
396                     String filename = FilenameUtils.getName(pathStr);
397                     return matcher.matches(Paths.get(filename)) && Files.isRegularFile(file);
398                 }
399             };
400             Map<String, FileTime> newFileState = new HashMap<>();
401             // list regular files in the directory
402             List<String> files = new ArrayList<>();
403             try (DirectoryStream<Path> stream = Files.newDirectoryStream(path, pathFilter)) {
404                 for (Path entry : stream) {
405                     String fullPath = JfsUtils.getFileNoAuth(entry.toUri());
406                     if (onChange) {
407                         FileTime modTime = Files.getLastModifiedTime(entry);
408                         FileTime prevTime = fileState.get(fullPath);
409 
410                         newFileState.put(fullPath, modTime);
411                         if (modTime.equals(prevTime)) {
412                             continue;
413                         }
414                     }
415                     if (URL_PROTOCOL_FILE.equalsIgnoreCase(uri.getScheme()) &&
416                             System.getProperty(OS_NAME).toLowerCase().contains(WIN)) {
417                         fullPath = StringUtils.replaceChars(fullPath, '\\', '/');
418                         if (fullPath.matches(REGEX_WINDOWS_PATH)) {
419                             fullPath = URL_SEPARATOR.concat(fullPath);
420                         }
421                     }
422                     files.add(fullPath);
423                 }
424             }
425             fileState = newFileState;
426             return files;
427         } finally {
428             fileFinder.closeFileSystem(fileSystem, directory);
429             if (path instanceof S3Path) {
430                 ((S3Path) path).shutdown();
431             }
432         }
433     }

435     @VisibleForTesting
436     boolean acceptS3Path(String dirPath, String pathStr, PathMatcher matcher) {
437         // get the filename under the directory path
438         String filename = pathStr.substring(dirPath.length());
439         // if directory doesn't end with a slash, remove the slash from the filename
440         filename = dirPath.endsWith(URL_SEPARATOR) ? filename : filename.substring(1);
441         boolean isMatch = matcher.matches(Paths.get(filename));
442         // if filename contains URL_SEPARATOR, the file must be in a subdirectory
443         return !StringUtils.contains(filename, URL_SEPARATOR) && isMatch;
444     }

484     private boolean evalExitOnFirstMatches(Document document) {
485         Object value = advPropMap.get(EXIT_ON_FIRST_MATCHES_PROP);
486         if (value instanceof ExpressionProperty) {
487             value = ((ExpressionProperty) value).eval(document);
488             if (value instanceof Boolean) {
489                 return (Boolean) value;
490             } else if (value != null) {
491                 return StringUtils.equalsIgnoreCase(value.toString(), Boolean.TRUE.toString());
492             }
493         }
494         return false;
