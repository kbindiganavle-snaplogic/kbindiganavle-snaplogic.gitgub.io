175      * @param initializationVector initialization vector string
176      * @param password             string used to generate secret key
177      * @return IV parameter specification
178      */
179     private IvParameterSpec createIVParameterSpec(String initializationVector, String password) {
180         IvParameterSpec ivParameterSpec = null;
181         if (StringUtils.isNotBlank(initializationVector)) {
182             try {
183                 byte[] ivParameterBytes = DatatypeConverter.parseHexBinary(initializationVector);
184 
185                 ivParameterSpec = new IvParameterSpec(ivParameterBytes);
186             } catch (NumberFormatException e) {
187                 throw new ConfigurationException(e, INVALID_ALGORITHM_PARAMETER)
188                         .withReason(INVALID_ALGORITHM_PARAMETER)
189                         .withResolution(CHECK_IV);
190             } catch (IllegalArgumentException e) {
191                 throw new ConfigurationException(e, INVALID_ALGORITHM_PARAMETER)
192                         .withReason(INVALID_ALGORITHM_PARAMETER)
193                         .withResolution(CHECK_IV);
194             }
195         } else if (StringUtils.isNotBlank(password)) {
196             byte[] key = password.getBytes();
197             MessageDigest sha;
198             try {
199                 sha = MessageDigest.getInstance(SHA1_ALGORITHM);
200             } catch (NoSuchAlgorithmException e) {
201                 throw new ConfigurationException(e, IV_GENERATION_FAILED)
202                         .withReason(IV_GENERATION_FAILED)
203                         .withResolutionAsDefect();
204             }
205             byte[] shaBytes = sha.digest(key);
206             if (CryptoAlgorithm.BLOWFISH.getName().equals(cryptoAlgorithm)) {
207                 ivParameterSpec = new IvParameterSpec(shaBytes, 0, BLOWFISH_IV_SIZE_IN_BYTES);
208             } else {
209                 ivParameterSpec = new IvParameterSpec(shaBytes, 0, IV_SIZE_IN_BYTES);
210             }
211         }
212         return ivParameterSpec;
213     }

216      * Generates secret key from password
217      *
218      * @param password string used to generate secret key
219      * @return secret key
220      * @throws NoSuchAlgorithmException
221      * @throws InvalidKeySpecException
222      */
223     private Key generateSecretKeyFromPassword(String password) throws NoSuchAlgorithmException,
224             InvalidKeySpecException {
225         byte[] passwordBytes = password.getBytes();
226         MessageDigest md = MessageDigest.getInstance(MD5_ALGORITHM);
227         byte[] mdBytes = md.digest(passwordBytes);
228         return new SecretKeySpec(mdBytes, 0, KEY_SIZE_IN_BYTES, cryptoAlgorithm);
229     }

232      * Obtains secret key from snap properties or account
233      *
234      * @param propertyValues property values map
235      * @return secret key
236      */
237     private Key getSecretKeyFromProperties(PropertyValues propertyValues) {
238         String secretKeyString;
239         accountType = binaryUtils.getAccountType(account);
240         if (AccountType.NONE.equals(accountType)) {
241             secretKeyString = propertyValues.get(SECRET_KEY_PROP);
242         } else {
243             secretKeyString = account.connect();
244         }
245         if (StringUtils.isBlank(secretKeyString)) {
246             throw new ConfigurationException(KEY_NULL_OR_EMPTY)
247                     .withReason(KEY_NULL_OR_EMPTY)
248                     .withResolution(PROVIDE_SECRET_KEY);
249         }
250         if (!Base64.isBase64(secretKeyString)) {
251             throw new ConfigurationException(INVALID_SECRET_KEY)
252                     .withReason(INVALID_SECRET_KEY)
253                     .withResolution(CHECK_SECRET_KEY);
254         }
255         byte[] secretKeyBytes = Base64.decodeBase64(secretKeyString);
256         return new SecretKeySpec(secretKeyBytes, cryptoAlgorithm);
257     }

267     protected void process(final Document header, final ReadableByteChannel readChannel) {
268         try {
269             if (ivParameterSpec == null) {
270                 cipher.init(cryptoMode, secretKey);
271             } else {
272                 cipher.init(cryptoMode, secretKey, ivParameterSpec);
273             }
274             outputViews.write(new BinaryOutput() {
275                 @Override
276                 public Document getHeader() {
277                     return header;
278                 }
279 
280                 @Override
281                 public void write(WritableByteChannel writeChannel) throws IOException {
282                     try (InputStream inputStream = Channels.newInputStream(readChannel);
283                          OutputStream outputStream = Channels.newOutputStream(writeChannel)) {
284                         CipherOutputStream cipherOutputStream = new CipherOutputStream(outputStream,
285                                 cipher);
286                         IOUtils.copyLarge(inputStream, cipherOutputStream);
287                         cipherOutputStream.close();
288                     } catch (IOException e) {
289                         SnapDataException sde = new SnapDataException(e, ERROR_PROCESSING_DATA)
290                                 .withReason(ERROR_PROCESSING_DATA)
291                                 .withResolutionAsDefect();
292                         errorViews.write(sde, header);
293                     }
294                 }
295             });
296         } catch (InvalidKeyException e) {
297             errorViews.write((SnapDataException) new SnapDataException(e, e.toString())
298                     .withReason(INVALID_SECRET_KEY)
299                     .withResolution(CHECK_SECRET_KEY), header);
300         } catch (InvalidAlgorithmParameterException e) {
301             errorViews.write((SnapDataException) new SnapDataException(e, e.toString())
302                             .withReason(INVALID_ALGORITHM_PARAMETER)
303                             .withResolution(CHECK_IV), header);
304         } catch (SuggestViewAbortException e) {
305             throw e;
306         } catch (Exception e) {
307             errorViews.write((SnapDataException)
308                     new SnapDataException(e, ERR_PROCESS_ENCRYPTION)
309                     .withReason(e.getMessage())
310                     .withResolution(RESOLUTION_ADDRESS_ISSUE), header);
311         }
312     }
