107      * Produces a list of files matching the given wildcard or regex. Its scope is package-default
108      * because this method is called from another class in the same package.
109      *
110      * @param propertyValues - Snap property values
111      * @param document       - input document to evaluate expression properties
112      * @param accountType    - account type
113      * @param userInfo       - user credential, e.g. "yourUsername:yourPassword@"
114      * @param errorViews     - error views
115      *
116      * @return list of directory Path objects
117      */
118     List<Path> populate(final PropertyValues propertyValues, Document document,
119             AccountType accountType, String userInfo, ErrorViews errorViews,
120             Map<String, String> env) throws Throwable {
121         // selectedFiles - list of map<K,V> data with following keys:
122         //      "fileFolder" - file or folder path
123         //      "wildcard" - wildcard
124         //      "includeSubfolders" - traverse all subfolders or not
125         List<Path> dirPaths = new ArrayList<>();
126         totalSelectedFileList.clear();
127         List<Map<String, Object>> selectedFiles = propertyValues.get(SELECTED_FILES_PROP);
128         if (selectedFiles == null) {
129             return dirPaths;
130         }
131         String fileNoAuth;
132         String wildcard;
133         FileSystem fileSystem;
134         for (Map<String, Object> selectedFile : selectedFiles) {
135             selectedFileList.clear();
136             // File/Folder property may include a pipeline parameter
137             fileNoAuth = propertyValues.getExpressionPropertyFor(selectedFile, FILE_FOLDER_PROP)
138                     .eval(document);
139             if (StringUtils.isBlank(fileNoAuth)) {
140                 continue;
141             }
142             wildcard = propertyValues.getExpressionPropertyFor(selectedFile, WILDCARD_PROP)
143                     .eval(document);
144             // if includeSubfolders value is either missing or explicitly null, use false
145             final boolean includeSubfolders = firstNonNull(
146                     includeSubfoldersPath.read(selectedFile, false), false);
147             final URI pathUri = urlUtil.insertAuthority(fileNoAuth, accountType, userInfo,
148                     UrlUtil.PATTERN);
149             checkForSameScheme(pathUri);
150             final String fileNoAuthFinal = fileNoAuth;
151             // get file system object
152             fileSystem = getFileSystemFor(pathUri, fileNoAuthFinal, env);
153             if (fileSystem != null) {
154                 // If the selected file is a folder and its wildcard/Regex field is empty, we
155                 // will interpret this as users want to select all files in the folder.
156                 if (StringUtils.isBlank(wildcard)) {
157                     wildcard = DEFAULT_WILD_CARD_PATTERN;
158                 }
159                 try {
160                     // Get a Pattern object from wildcard/regex
161                     Pattern pattern = getRegexPattern(wildcard);
162                     // get path object
163                     final Path path = createPath(fileSystem, pathUri, includeSubfolders);
164                     dirPaths.add(path);
165                     ContextualRunnable runnable = context ->
166                             populateFailsafe(path, pathUri, fileNoAuthFinal, pattern,
167                                     includeSubfolders, document, errorViews);
168                     CheckedConsumer<Throwable> failedAttemptListener = e -> LOG.warn(
169                             "{} - Failed to populate the selected file list for: {}, {}",
170                             RETRY_SIGNATURE, fileNoAuthFinal, e.getMessage());
171                     failsafeExecute(runnable, failedAttemptListener, numOfRetries, retryInterval);
172                 } catch (Throwable e) {
173                     Throwable ex = e instanceof FailsafeException ? e.getCause() : e;
174                     if (ex instanceof SnapDataException) {
175                         errorViews.write((SnapDataException) ex, document);
176                     } else if (ex instanceof SnapException) {
177                         errorViews.write(
178                                 (SnapDataException) new SnapDataException(ex, ex.getMessage())
179                                 .withReason(((SnapException) ex).getReason())
180                                 .withResolution(((SnapException) ex).getResolution()), document);
181                     } else {
182                         throw ex;
183                     }
184                 } finally {
185                     closeFileSystem(fileSystem, fileNoAuth);
186                 }
187             }
188             totalSelectedFileList.addAll(selectedFileList);
189         }
190         return dirPaths;
191     }

193     private void populateFailsafe(Path path, URI pathUri, String fileNoAuth, Pattern pattern,
194             boolean includeSubfolders, Document document, ErrorViews errorViews) {
195         selectedFileList.clear();
196         if (Files.isDirectory(path)) {
197             // If it is a directory, go get the list of files in the directory.
198             findFiles(path, pattern, includeSubfolders, fileNoAuth);
199         } else if (Files.isRegularFile(path)) {
200             // If it is a regular file, add it to the selected file list.
201             selectedFileList.add(pathUri);
202         } else if (Files.exists(path, LinkOption.NOFOLLOW_LINKS)) {
203             // a file which is not a regular file nor a directory is not supported
204             throw new SnapDataException(ERR_UNSUPPORTED_FILE_TYPE)
205                     .formatWith(fileNoAuth)
206                     .withReason(ERR_UNSUPPORTED_FILE_TYPE_REASON)
207                     .withResolution(ERR_UNSUPPORTED_FILE_TYPE_RESOLUTION);
208         } else if (URL_PROTOCOL_S3.equalsIgnoreCase(pathUri.getScheme())) {
209             // If an S3 directory has been created by directly writing an S3 file,
210             // that S3 directory object actually does not exists. We may need to go
211             // ahead to try to list the given directory
212             int count = selectedFileList.size();
213             findFiles(path, pattern, includeSubfolders, fileNoAuth);
214             if (count == selectedFileList.size()) {
215                 throw new SnapDataException(ERR_FILE_NOT_FOUND)
216                         .formatWith(fileNoAuth)
217                         .withReason(REASON_S3_OBJECT_NOT_FOUND)
218                         .withResolution(ERR_FILE_NOT_FOUND_RESOLUTION);
219             }
220         } else {
221             // the file does not exist or the server is accessible
222             throw new SnapDataException(ERR_FILE_NOT_FOUND)
223                     .formatWith(fileNoAuth)
224                     .withReason(ERR_FILE_NOT_FOUND_REASON)
225                     .withResolution(ERR_FILE_NOT_FOUND_RESOLUTION);
226         }
227     }

245      * @param pathUri    - URI object for the path
246      * @param fileNoAuth - file path string without user info
247      * @param env        - a map of provider specific properties to configure the file system
248      *                     may be empty
249      *
250      * @return FileSystem object
251      */
252     FileSystem getFileSystemFor(final URI pathUri, String fileNoAuth, Map<String, ?> env) {
253         String scheme = pathUri.getScheme().toLowerCase();
254         if (PROTOCOL_FILE.equals(scheme)) {
255             // local filesystem is a default file system
256             return FileSystems.getDefault();
257         }
258         final Map<String, ?> envFinal = env == null ? new HashMap<>() : env;
259         // check other sources of environment map
260         Map<String, String> env2 = null;
261         if (PROTOCOL_GS.equals(scheme)) {
262             env2 = urlUtil.getEnv(account);
263         } else if (account instanceof FileSystemPropertiesProvider) {
264             env2 = ((FileSystemPropertiesProvider) account).getRequestProperties();
265         }
266         if (MapUtils.isNotEmpty(env2)) {
267             envFinal.putAll((Map) env2);
268         }
269         try {
270             ContextualCallable<FileSystem> callable =
271                     context -> JfsUtils.getFileSystem(pathUri, envFinal);
272             CheckedConsumer<Throwable> failedAttemptListener = e -> LOG.warn(
273                     "{} - Failed to get FileSystem for: {}, {}",
274                     RETRY_SIGNATURE, fileNoAuth, e.getMessage());
275             return failsafeExecute(callable, failedAttemptListener, numOfRetries, retryInterval);
276         } catch (Throwable e) {
277             // if FailSafe wraps an exception, unwrap it
278             Throwable ex = e instanceof FailsafeException ? e.getCause() : e;
279             if (ex instanceof SuggestViewAbortException) {
280                 throw (SuggestViewAbortException) ex;
281             } else {
282                 throw new SnapDataException(ex, ERR_CREATE_FILESYSTEM)
283                         .formatWith(fileNoAuth)
284                         .withReason(ex.getMessage())
285                         .withResolution(ERR_CREATE_FILESYSTEM_RESOLUTION);
286             }
287         }
288     }

303      * Finds all files which matches wildcard or regex in a given directory.
304      *
305      * @param path             Path object of the directory
306      * @param pattern          regular expression pattern
307      * @param includeSubfolder a flag to search subdirectories recursively if true
308      * @param fileNoAuth       same string as in the path argument without user info
309      */
310     @VisibleForTesting
311     protected void findFiles(Path path, final Pattern pattern, final boolean includeSubfolder,
312             final String fileNoAuth) {
313         // prepare a filter object
314         DirectoryStream.Filter<Path> filter = new DirectoryStream.Filter<Path>() {
315             public boolean accept(Path file) throws IOException {
316                 String pathStr = file instanceof SftpPath ? ((SftpPath) file).getPath() :
317                         file.toUri().getPath();
318                 String fileName = FilenameUtils.getName(pathStr);
319                 if (fileName != null && pattern.matcher(fileName).matches() &&
320                         Files.isRegularFile(file)) {
321                     return true;
322                 } else if (includeSubfolder && Files.isDirectory(file) &&
323                         !StringUtils.startsWithIgnoreCase(fileNoAuth, URL_PROTOCOL_S3)) {
324                     // in S3 API, there is no need to traverse subdirectories
325                     findFiles(file, pattern, includeSubfolder, fileNoAuth);
326                 }
327                 return false;
328             }
329         };
330         // Search for regular files in the directory. If found, add them to the selected file
331         // list.
332         try (DirectoryStream<Path> stream = Files.newDirectoryStream(path, filter)) {
333             for (Path entry : stream) {
334                 URI uri = entry.toUri();
335                 String pathStr = uri.toString();
336                 // In case of local file UNC syntax in Windows Groundplex, the platform may return
337                 // "file://computer_name/shared_dir/..." instead of
338                 // "file:////computer_name/shared_dir/...".
339                 // The two-slash path will cause a failure in getting its input stream.
340                 if (StringUtils.startsWith(pathStr, FILE_PREFIX) &&
341                         !StringUtils.startsWith(pathStr, FILE_PREFIX_SLASH)) {
342                     pathStr = FILE_PREFIX.concat(pathStr.substring(5));
343                     uri = URI.create(pathStr);
344                 }
345                 selectedFileList.add(uri);
346             }
347         } catch (IOException e) {
348             // Directory could not be found for a given path object.
349             String error = String.format(ERR_DIR_NOT_FOUND, fileNoAuth, e.getMessage());
350             throw new SnapDataException(e, error)
351                     .withReason(error)
352                     .withResolution(RESOLUTION_DIR_NOT_FOUND);
353         }
354     }

378     /**
379      * Converts a wildcard pattern to a regex pattern.
380      *
381      * @param wildcard - wildcard pattern
382      *
383      * @return - regex pattern that is equivalent to the wildcard pattern
384      */
385     @VisibleForTesting
386     protected String wildcardToRegex(String wildcard) {
387         if (wildcard == null) {
388             return null;
389         }
390         StringBuilder buffer = new StringBuilder(STRING_CARET);
391         char[] chars = wildcard.toCharArray();
392         for (char character : chars) {
393             if (character == '*') {
394                 buffer.append(DOT_STAR);
395             } else if (character == '?') {
396                 buffer.append('.');
397             } else if (REGEX_PATTERN_WILDCARD.indexOf(character) != -1) {
398                 // prefix all meta-characters with backslash
399                 buffer.append('\\').append(character);
400             } else {
401                 buffer.append(character);
402             }
403         }
404         buffer.append('$');
405         return buffer.toString();
406     }

408     protected Path createPath(FileSystem fileSystem, URI uri, boolean includeSubfolders) {
409         String fileFolder = uri.toString();
410         LOG.debug(URI_LOG, URLUtils.getFileNoAuth(fileFolder));
411         switch (uri.getScheme()) {
412             case URL_PROTOCOL_ADL:
413                 fileFolder = JfsUtils.decode(fileFolder);
414                 break;
415             case URL_PROTOCOL_FILE:
416                 return getLocalPath(fileSystem, uri);
417             case URL_PROTOCOL_HDFS:
418                 return ((HdfsFileSystem)fileSystem).getPath(uri.getScheme(), uri.getPort(),
419                         uri.getPath());
420             case URL_PROTOCOL_S3:
421             case URL_PROTOCOL_S3N:
422                 if (includeSubfolders) {
423                     fileFolder = String.format(FORMAT_INCLUDESUBFOLDERS, fileFolder);
424                 }
425         }
426         return fileSystem.getPath(fileFolder);
427     }
428     private void checkForSameScheme(URI uri) {
429         String scheme = uri.getScheme();
430         if (firstUri) {
431             lastScheme = scheme;
432             firstUri = false;
433         } else {
434             if (!scheme.equals(lastScheme)) {
435                 // All files or folders in a Read Snap instance must have the same scheme. This
436                 // rule is found violated here.
437                 String error = String.format(ERR_DIFFERENT_SCHEMES, lastScheme, scheme);
438                 throw new ExecutionException(error)
439                         .withReason(error)
440                         .withResolution(RESOLUTION_DIFFERENT_SCHEMES);
441             }
442         }
443     }
