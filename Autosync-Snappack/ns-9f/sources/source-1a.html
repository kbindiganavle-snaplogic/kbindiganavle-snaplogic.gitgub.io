227     @Override
228     protected void process(final Document document, final String inputViewName) {
229         numOfRetries = evalIntProperty(numOfRetriesExpr, document, MIN_RETRIES, MIN_RETRIES,
230                 errorViews);
231         retryInterval = evalIntProperty(retryIntervalExpr, document, MIN_RETRY_INTERVAL,
232                 MIN_RETRY_INTERVAL, errorViews);
233         String source = sourceExpr.eval(document);
234         String target = targetExpr.eval(document);
235         if (StringUtils.isBlank(source)) {
236             writeError(null, document, REASON_REQUIRED_PROPERTY, ERR_BLANK_SOURCE);
237             return;
238         }
239         if (StringUtils.isBlank(target)) {
240             writeError(null, document, REASON_REQUIRED_PROPERTY, ERR_BLANK_TARGET);
241             return;
242         }
243         LinkedHashMap<String, String> output = new LinkedHashMap<>();
244         output.put(SOURCE_LABEL, source);
245         output.put(TARGET_LABEL, target);
246         if (source.equals(target)) {
247             output.put(STATUS, STATUS_SAME_SOURCE_TARGET);
248             outputViews.write(documentUtility.newDocument(output), document);
249             return;
250         }
251         String fileOperation = fileOperationExpr.eval(document);
252         boolean isCopy = false;
253         if (StringUtils.equalsIgnoreCase(fileOperation, MOVE) ||
254                 StringUtils.equalsIgnoreCase(fileOperation, RENAME)) {
255             isCopy = false;
256         } else if (StringUtils.equalsIgnoreCase(fileOperation, COPY)) {
257             isCopy = true;
258         } else {
259             errorViews.write((SnapDataException)
260                     new SnapDataException(ERR_INVALID_FILE_OPERATION).formatWith(fileOperation)
261                             .withReason(REASON_MOVE_OR_COPY)
262                             .withResolution(RESOLUTION_ADDRESS_ISSUE), document);
263         }
264         // insert user name and password into file name
265         URI srcUri;
266         URI targetUri;
267         try {
268             if (account instanceof BinaryAbfsAccount) {
269                 srcUri = urlUtil.insertAbfsAuthority(source, userInfo, PATTERN,
270                         (BinaryAbfsAccount) account);
271                 targetUri = urlUtil.insertAbfsAuthority(target, userInfo, PATTERN,
272                         (BinaryAbfsAccount) account);
273             } else {
274                 srcUri = urlUtil.insertAuthority(source, accountType, userInfo, PATTERN);
275                 targetUri = urlUtil.insertAuthority(target, accountType, userInfo, PATTERN);
276             }
277         } catch (SnapException e) {
278             errorViews.write((SnapDataException) new SnapDataException(e.getMessage())
279                     .withReason(e.getReason())
280                     .withResolution(e.getResolution()), document);
281             // Stop Snap execution if an exception occurs while processing src or target URI
282             throw new ExecutionException(e.getMessage()).withReason(e.getReason());
283         }
284         if (!validatePath(srcUri, document) || !validatePath(targetUri, document)) {
285             return;
286         }
287         // get file system object
288         FileSystem fileSystem = null;
289         String error = isCopy ? ERR_COPY : ERR_MOVE;
290         Path srcPath = null;
291         Path targetPath = null;
292         try {
293             fileFinder.setAccount(account);
294             fileFinder.setRetries(numOfRetries, retryInterval);
295             Map<String, String> env = getSasEnvProperties(advPropMap, document);
296             BinaryUtils.embedAuthenticationConfigIfNeeded(env, account);
297             getSimpleFileOperation(advPropMap, document, env);
298             azcopyPathInput = advPropMap.containsKey(AZCOPY_ABS_PATH) ?
299                     advPropMap.get(AZCOPY_ABS_PATH).eval(document) : null;
300             if (null == azcopyPathInput || azcopyPathInput.isEmpty()) {
301                 fileSystem = fileFinder.getFileSystemFor(srcUri, source, env);
302                 // instantiate Path objects for source and target
303                 srcPath = fileFinder.createPath(fileSystem, srcUri, false);
304                 if (srcPath instanceof S3Path) {
305                     setKmsProperties((S3Path) srcPath);
306                 }
307                 targetPath = fileFinder.createPath(fileSystem, targetUri, false);
308                 // init option to replace an existing file if it exists
309                 moveOrCopyFailSafe(srcPath, targetPath, source, target, isCopy, output);
310             } else {
311                 if (source.startsWith(PROTOCOL_WASBS) || target.startsWith(PROTOCOL_WASBS)) {
312                     throw new ConfigurationException(ERR_AZCOPY_UNSUPPORTED_OPR_MSG)
313                             .withReason(ERR_REASON_WASBS_UNSUPPORTED_WITH_AZCOPY)
314                             .withResolution(RESOLUTION_WASBS_UNSUPPORTED_WITH_AZCOPY);
315                 }
316                 binaryUtils.copyFileWithAzcopy(source, target, fileOperation, azcopyPathInput,
317                         env.get(SAS_URI), advPropMap, output, tempDirFile);
318             }
319             outputViews.write(documentUtility.newDocument(output), document);
320         } catch (NoSuchFileException e) {
321             try {
322                 if (binaryUtils.existsWithRetry(srcPath, source, numOfRetries, retryInterval,
323                         LOGGER)) {
324                     Path targetParent = targetPath.getParent();
325                     String targetPathStr = getFileNoAuth(targetParent.toUri());
326                     if (!binaryUtils.existsWithRetry(targetParent, targetPathStr, numOfRetries,
327                             retryInterval, LOGGER)) {
328                         try {
329                             JfsUtils.createDirectory(targetParent);
330                             moveOrCopyFailSafe(srcPath, targetPath, source, target, isCopy, output);
331                             outputViews.write(documentUtility.newDocument(output), document);
332                         } catch (IOException ex) {
333                             String reason = e.getMessage();
334                             reason = StringUtils.isBlank(reason) ?
335                                     e.getClass().getSimpleName() : reason;
336                             writeError(e, document, reason, error, source, target);
337                         }
338                     } else {
339                         writeError(e, document, REASON_UNKNOWN, error, source, target);
340                     }
341                 } else {
342                     writeError(e, document, String.format(ERR_FILE_NOT_FOUND, source), error,
343                             source, target);
344                 }
345             } catch (IOException ex) {
346                 writeError(ex, document, ex.getMessage(), error, source, target);
347                 return;
348             }
349         } catch (FileAlreadyExistsException e) {
350             writeError(e, document, REASON_TARGET_ALREADY_EXISTS, error, source, target);
351         } catch (DirectoryNotEmptyException e) {
352             writeError(e, document, String.format(REASON_DIRECTORY_NOT_EMPTY, e.getFile()), error,
353                     source, target);
354         } catch (ExecutionException e) {
355             String reason = StringUtils.isBlank(e.getReason()) ? e.getMessage() : e.getReason();
356             writeError(e, document, reason, error, source, target);
357         } catch (SnapDataException e) {
358             errorViews.write(e, document);
359         } catch (SuggestViewAbortException | ConfigurationException e) {
360             throw e;
361         } catch (Exception e) {
362             String reason = e.getMessage();
363             reason = StringUtils.isBlank(reason) ? e.getClass().getSimpleName() : reason;
364             writeError(e, document, reason, error, source, target);
365         } finally {
366             // should not attempt to close local file system
367             if (!PROTOCOL_FILE.equalsIgnoreCase(srcUri.getScheme())) {
368                 IOUtils.closeQuietly(fileSystem);
369             }
370             if (srcPath instanceof S3Path) {
371                 ((S3Path) srcPath).shutdown();
372             }
373             if (targetPath instanceof S3Path) {
374                 ((S3Path) targetPath).shutdown();
375             }
376         }
377     }

390     private void moveOrCopyFailSafe(Path source, Path target, String sourceStr, String targetStr,
391                                     boolean isCopy, LinkedHashMap<String, String> output)
392             throws IOException {
393         try {
394             CheckedConsumer<Throwable> failedAttemptListener = e -> LOGGER.warn(
395                     "{} - Failed to perform file operation from {} to {}, {}",
396                     RETRY_SIGNATURE, sourceStr, targetStr, e.getMessage());
397             ContextualRunnable runnable = context -> moveOrCopy(source, target, isCopy, output);
398             failsafeExecute(runnable, failedAttemptListener, numOfRetries, retryInterval);
399         } catch (Throwable e) {
400             Throwable ex = e instanceof FailsafeException ? e.getCause() : e;
401             if (ex instanceof SuggestViewAbortException) {
402                 throw (SuggestViewAbortException) ex;
403             } else if (ex instanceof IOException) {
404                 throw (IOException) ex;
405             } else {
406                 throw new IOException(ex.getMessage(), ex);
407             }
408         }
409     }

411     private void moveOrCopy(Path source, Path target, boolean isCopy,
412                             LinkedHashMap<String, String> output) throws IOException {
413         if (errorIfExists) {
414             if (isCopy) {
415                 Files.copy(source, target);
416             } else {
417                 if (StringUtils.startsWith(target.toString(), PROTOCOL_SFTP) &&
418                         Files.exists(target, LinkOption.NOFOLLOW_LINKS)) {
419                     throw new FileAlreadyExistsException(
420                             JfsUtils.getFileNoAuth(target.toString()), null, ERR_TARGET_EXISTS);
421                 }
422                 Files.move(source, target);
423             }
424         } else {
425             if (isCopy) {
426                 Files.copy(source, target, StandardCopyOption.REPLACE_EXISTING);
427             } else {
428                 Files.move(source, target, StandardCopyOption.REPLACE_EXISTING);
429             }
430         }
431         output.put(STATUS, isCopy ? STATUS_COPIED : STATUS_MOVED);
432     }
