23 /**
24  * AbstractJoinFactory is the factory that returns the appropriate {@link JoinFactory} based on
25  * the sort order.

27  * @author ksubramanian

30 public class AbstractJoinFactory {
31     /**
32      * Returns the join factory implementation based on the sort order.
33      *
34      * @param sortOrder
35      *
36      * @return joinFactory
37      */
38     public static JoinFactory getJoinFactoryFor(SortOrder sortOrder) {
39         switch (sortOrder) {
40             case ASCENDING:
41                 return new AscendingSortedJoinFactory();
42             case DESCENDING:
43                 return new DescendingSortedJoinFactory();
44             case UNSORTED:
45                 return new UnsortedJoinFactory();
46             default:
47                 // This should never happen
48                 throw new ExecutionException(SORT_ORDER_IS_NOT_VALID)
49                         .formatWith(sortOrder.name().toString())
50                         .withResolutionAsDefect();
51         }
52     }
