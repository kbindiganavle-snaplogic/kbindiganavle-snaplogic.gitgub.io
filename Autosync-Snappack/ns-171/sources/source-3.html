68     @Override
69     public int compareTo(final ComparableList o) {
70         Iterator<Comparable> thisIterator = this.iterator();
71         Iterator<Comparable> otherIterator = o.iterator();
72         boolean hasNextThis = thisIterator.hasNext();
73         boolean hasNextOther = otherIterator.hasNext();
74         while (hasNextThis && hasNextOther) {
75             Comparable thisComparable = thisIterator.next();
76             Comparable otherComparable = otherIterator.next();
77             if (thisComparable == null && otherComparable == null) {
78                 hasNextThis = thisIterator.hasNext();
79                 hasNextOther = otherIterator.hasNext();
80                 continue;
81             }
82             if (thisComparable == null) {
83                 return -1;
84             }
85             if (otherComparable == null) {
86                 return 1;
87             }
88             try {
89                 int comparison = thisComparable.compareTo(otherComparable);
90                 if (comparison != 0) {
91                     return comparison;
92                 }
93             } catch (ClassCastException e) {
94                 String msg = String.format(CLASSCAST_REASON,
95                         thisComparable.getClass().getSimpleName(),
96                         otherComparable.getClass().getSimpleName());
97                 throw new SnapDataException(e, msg)
98                         .withReason(msg)
99                         .withResolution(CLASSCAST_RES);
100             } catch (Exception e) {
101                 throw new ExecutionException(e, ERR_COMPARE_TYPE)
102                         .withReason(REASON_INTERNAL_ERROR)
103                         .withResolution(RESOLUTION_REPORT_ISSUE);
104             }
105             hasNextThis = thisIterator.hasNext();
106             hasNextOther = otherIterator.hasNext();
107         }
108         if (hasNextThis) {
109             return 1;
110         } else if (hasNextOther) {
111             return -1;
112         }
113         return 0;
114     }
