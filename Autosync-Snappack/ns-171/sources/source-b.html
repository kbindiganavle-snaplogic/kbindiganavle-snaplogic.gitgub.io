282      * then an exception is thrown.
283      *

285      *

288     public static JoinType fromString(String joinType) {
289         JoinType type = STRING_TO_ENUM.get(StringUtils.lowerCase(joinType));
290         if (type == null) {
291             throw new ExecutionException(JOIN_TYPE_NOT_SUPPORTED, joinType);
292         }
293         return type;
