33         Object val = prop.eval(doc);
34         if (val instanceof Number) {
35             return ((Number) val).intValue();

64     public static Long evalTimestamp(Document doc, ExpressionProperty prop, Long defaultVal) {
65         Object val = prop.eval(doc);
66         if (val instanceof Number) {
67             return ((Number) val).longValue();
68         } else if (val instanceof ReadableInstant) { // handle joda objects
69             return ((ReadableInstant) val).getMillis();
70         } else if (val instanceof TemporalAccessor) { // handle java.time objects
71             return Instant.from((TemporalAccessor) val).toEpochMilli();
72         } else if (val instanceof String) {
73             String s = ((String) val).trim();
74             if (s.isEmpty()) {
75                 return defaultVal;
76             } else {
77                 return Long.valueOf(s);
78             }
79         } else {
80             return defaultVal;

84     public static <T> T evalExpression(String expression,
85                                        ExpressionUtils expressionUtils,
86                                        PropertyValues propertyValues,
87                                        String defaultPropOnException) {
88         ExpressionProperty expProp = expressionUtils.createExpressionProperty(
89                 propertyValues, true, expression);
90         try {
91             return expProp.eval(null);
92         } catch (ExpressionException e) {
93             // When unit tests are executed, the standard global variables like pipe.startDate,
94             // snap.label, etc, aren't defined, so any expression that references them will result
95             // in this exception. This fallback logic will look for a property named
96             // by defaultPropOnException, and if defined, will use that property's value
97             // as the return value. This property will only be defined by individual unit tests.
98             T defaultOnException = propertyValues.get(defaultPropOnException);
99             if (defaultOnException != null) {
100                 return defaultOnException;
101             } else {
102                 throw e;
103             }
104         }
105     }

107     public static String evalClientID(ExpressionUtils expressionUtils, PropertyValues props) {
108         String result = evalExpression(DEFAULT_CLIENT_ID_EXPR, expressionUtils, props, DEFAULT_CLIENT_ID_PROP);
