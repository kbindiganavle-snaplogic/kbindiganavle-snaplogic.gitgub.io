399     /**
400      * If the given schema element contains children that refer back to XML Schema definition
401      * element then XSOM parser is unable to handle this use case.
402      *
403      * Example:
404      *
405      * <pre>
406      * {@code
407      * < s:schema xmlns:s="http://www.w3.org/2001/XMLSchema">
408      *     < s:element name="name" type="string"/>
409      *     < s:element ref="s:schema"/>
410      * < /s:schema>
411      * }
412      * </pre>
413      *
414      * XSOM parser will throw error while parsing the above schema element.  This seems to be
415      * limitation with the parser.
416      *
417      * To suppress this error, we will internally change such child elements to complex type
418      * with name "childReferringToXmlSchemaRootElement" having no sub-elements as follows:
419      *
420      * Workaround:
421      *
422      * <pre>
423      * {@code
424      * < s:schema xmlns:s="http://www.w3.org/2001/XMLSchema">
425      *     < s:element name="name" type="string"/>
426      *     < s:element name="childReferringToXmlSchemaRootElement">
427      *         < s:complexType/>
428      *     < /s:element>
429      * < /s:schema>
430      * }
431      * </pre>
432      *
433      * We are changing the child element name temporarily so as to set the right namespace while
434      * generating SOAP envelope.
435      *
436      * NOTE: By chance if the given WSDL already contains an element with name
437      * "childReferringToXmlSchemaRootElement" then that will break this code but such probability
438      * is almost zero.
439      *
440      * And finally, while generating SOAP envelope, we will change the element name back to
441      * "schema" with namespace "http://www.w3.org/2001/XMLSchema". But this part is done in
442      * japis/jsoap project.
443      *
444      * @param element
445      * @param document
446      */
447     private void handleChildSchemaElementsIfAny(Element element, Document document) {
448         Node refAttribute = null;
449         NamedNodeMap allAttributes = element.getAttributes();
450         if (allAttributes != null) {
451             refAttribute = allAttributes.getNamedItem(REF_ATTRIBUTE);
452         }
453         if (refAttribute != null && elementRefersToXsdRootSchema(refAttribute)) {
454             allAttributes.removeNamedItem(REF_ATTRIBUTE);
455             element.setAttribute(NAME_ATTRIBUTE,
456                     TEMP_NAME_FOR_CHILD_ELEMENT_OF_TYPE_XSD_SCHEMA_ROOT);
457             element.appendChild(document.createElementNS(
458                     element.getNamespaceURI(), COMPLEX_TYPE_ELEMENT));
459         } else {
460             NodeList childNodes = element.getChildNodes();
461             int noOfChildNodes = childNodes.getLength();
462             for (int i = 0; i < noOfChildNodes; i++) {
463                 Node childNode = childNodes.item(i);
464                 if (childNode instanceof Element) {
465                     handleChildSchemaElementsIfAny((Element) childNode, document);
466                 }
467             }
468         }
469     }

471     /**
472      * Checks if the given attribute has name "ref" and value "somePrefix:schema".
473      *
474      * NOTE: schema here refers to XSD root schema element.  So namespace referred to has to be
475      * "http://www.w3.org/2001/XMLSchema";
476      *
477      * @param refAttribute
478      *
479      * @return
480      */
481     private boolean elementRefersToXsdRootSchema(Node refAttribute) {
482         String attrValue = refAttribute.getNodeValue();
483         int colonIndex = attrValue.indexOf(COLON);
484         String prefix = null;
485         String value = attrValue;
486 
487         if (colonIndex > 0) {
488             prefix = attrValue.substring(0, colonIndex);
489             value = attrValue.substring(colonIndex + 1);
490         }
491 
492         if (prefix != null &&
493                 HTTP_WWW_W3_ORG_2001_XMLSCHEMA.equals(refAttribute.lookupNamespaceURI(prefix)) &&
494                 value.equals(SCHEMA)) {
495             return true;
496         }
497         return false;
498     }

516         if (nodeList.getLength() == 0) {
517             // Here we will assume there could be WSDL import element in the given WSDL.
518             // So we will attempt reading schema elements from the location provided in that
519             //  import element
520             wsdlDocument = domUtil.getResourceAsDocument(
521                     url, httpContextProvider, timeout);
522             nodeList = getNodeListForXPath(XPATH_FOR_IMPORT_ELEMENT, wsdlDocument,
523                     wsdlNamespaceContext);
524             if (nodeList.getLength() == 0) {
525                 // No import elements found
526                 return null;
527             } else {
528                 NodeList schemaElements = null;
529                 for (int i = 0; i < nodeList.getLength(); i++) {
530                     // Found import element
531                     // Now get the Types info from WSDL pointed to by "location" attribute
532                     NamedNodeMap map = nodeList.item(i).getAttributes();
533                     Node location = map.getNamedItem(ATTRIBUTE_LOCATION);
534                     if (location == null) {
535                         // No "location" attribute found
536                         continue;
537                     } else {
538                         // Found "location" attribute that refers to another WSDL
539                         // So get Types from that WSDL
540                         String newWsdl = location.getNodeValue();
541                         URL newWsdlUrl;
542                         try {
543                             newWsdlUrl = new URL(
544                                     new URI(wsdlDocument.getDocumentURI()).toURL(), newWsdl);

559                                     domUtil.getResourceAsDocument(newWsdlUrl, httpContextProvider,

804     private void populateElemAndAttrFormDefaults(Node node, Map<String,
805             FormChoice> tnsToElemFormDefaultMap, Map<String, FormChoice> tnsToAttrFormDefaultMap) {
806         // The given node may not be schema node when this method is being called while resolving
807         // external entities, so we need to find the root schema node.
808         node = getSchemaNode(node);
809         node.normalize();
810         NamedNodeMap namedNodeMap = node.getAttributes();
811         Node targetNamespaceNode = namedNodeMap.getNamedItem(TARGET_NAMESPACE);
812         String targetNamespace;
813         if (targetNamespaceNode != null) {
814             targetNamespace = targetNamespaceNode.getNodeValue();
815         } else {
816             targetNamespace = DEFAULT_TARGET_NS + counter.getAndIncrement();
817             ((Element) node).setAttribute(TARGET_NAMESPACE, targetNamespace);
818         }
819         Node elementFormDefaultNode = namedNodeMap.getNamedItem(ELEMENT_FORM_DEFAULT);
820         FormChoice elemFormChoice = elementFormDefaultNode == null ?
821                 FormChoice.UNQUALIFIED : FormChoice.valueOf(elementFormDefaultNode
822                 .getNodeValue().toUpperCase());
823         Node attributeFormDefaultNode = namedNodeMap.getNamedItem(ATTR_FORM_DEFAULT);
824         FormChoice attrFormChoice = attributeFormDefaultNode == null ?
825                 FormChoice.UNQUALIFIED : FormChoice.valueOf(attributeFormDefaultNode
826                 .getNodeValue().toUpperCase());
827         tnsToElemFormDefaultMap.put(targetNamespace, elemFormChoice);
828         tnsToAttrFormDefaultMap.put(targetNamespace, attrFormChoice);
829     }
