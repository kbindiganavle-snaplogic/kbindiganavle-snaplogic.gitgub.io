13 import org.joda.time.DateTime;
14 import org.joda.time.DateTimeZone;
15 
16 import java.sql.Timestamp;

18 /**
19  * DateTime Converter which produces DateTime with DateTimeZone.UTC
20  *
21  * @since Dec, 2017
22  */
23 public class DateTimeUtcConverter extends DateTimeConverter {
24 
25     private boolean useEpoch;
26 
27     public DateTimeUtcConverter() {
28         this(false);
29     }
30 
31     public DateTimeUtcConverter(boolean useEpoch) {
32         this.useEpoch = useEpoch;
33     }
34 
35     @Override
36     public DateTime from(final Timestamp timestamp) {
37         if (timestamp == null) {
38             return null;
39         }
40 
41         if (useEpoch) {
42             return new DateTime(timestamp.getTime(), DateTimeZone.UTC);
43         } else {
44             // One use case of this conversion is MySQL TIMESTAMP data type.
45             // When the java.sql.Timestamp object is converted to Joda DateTime, UTC times zone
46             // should be set. MySQL TIMESTAMP stores date/time data in UTC zone.
47             // Without DateTimeZone.UTC argument below, the system time zone will be picked up.
48             return new DateTime(timestamp.toString().replace(' ', 'T'), DateTimeZone.UTC);
49         }
50     }
51 }
