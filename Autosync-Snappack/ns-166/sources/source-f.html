276     private Document decryptOneField(final Document document, final WalkResult walkResult) {
277         Object value = walkResult.getValue();
278         if (value instanceof String) {
279             String stringValue = ((String) value);
280             if (StringUtils.startsWith(stringValue, ENC_PREFIX) &&
281                     StringUtils.endsWith(stringValue, ENC_SUFFIX)) {
282                 String bareString = stringValue.substring(4, stringValue.length() - 4);
283                 try {
284                     Map<String, Object> wrapper = snapObjectMapper.readValue(bareString, Map.class);
285                     return decryptUnencapsulatedData(document, walkResult, wrapper);
286                 } catch (IOException e) {
287                     throw new SnapDataException(e, ERR_READ_ENCRYPTED)
288                             .formatWith(walkResult.getActualPath().toString())
289                             .withReason(e.getMessage())
290                             .withResolutionAsDefect();
291                 }
292             } else {
293                 return decryptUnencapsulatedDataRaw(document, walkResult, String.valueOf(value));
294             }
295         } else if (value instanceof Map) {
296             Map<String, Object> map = ((Map) value);
297             if (map.containsKey(CIPHERTEXT)) {
298                 return decryptUnencapsulatedData(document, walkResult, map);
299             }
300         }
301 
302         return document;
303     }

317         return decryptUnencapsulatedData(document, walkResult, key, transformation, ivBytes,

354     private CryptoKeyHolder getKeyHolder(final Document document, final String transformation) {
355         CryptoKeyHolder keyHolder;
356         if (account instanceof CryptoEmptyAccount) {
357             String key = keyExpr.eval(document);
358             if (org.apache.commons.lang.StringUtils.isBlank(key)) {
359                 throw new SnapDataException(ERR_EMPTY_KEY)
360                         .withReason(REASON_EMPTY_KEY)
361                         .withResolution(RESOLUTION_EMPTY_KEY);
362             }
363             if (StringUtils.contains(key, BEGIN_RSA_PRIVATE_KEY)) {
364                 keyHolder = new CryptoKeyHolderForPrivateKey(key);
365             } else {
366                 keyHolder = new CryptoKeyHolderForSymmetricKey(key,
367                         CoercionUtils.getAlgorithmFromTransform(transformation));
368             }
369         } else {
370             keyHolder = account.connect();
371         }
372         return keyHolder;
373     }

377     private Document decryptUnencapsulatedData(final Document document, final WalkResult walkResult,
378                                                final Key key, final String transformation,
379                                                final byte[] ivBytes, final byte[] cipherText,
380                                                final String plaintextTypeString) {
381         try {
382             Cipher cipher = Cipher.getInstance(transformation, CRYPTO_PROVIDER);
383             if (ivBytes != null) {
384                 cipher.init(Cipher.DECRYPT_MODE, key, new IvParameterSpec(ivBytes));
385             } else {
386                 cipher.init(Cipher.DECRYPT_MODE, key);
387             }
388             byte[] plaintext = cipher.doFinal(cipherText);
389             Constants.PlaintextType plaintextType;
390             try {
391                 plaintextType = Constants.PlaintextType.valueOf(plaintextTypeString);
392             } catch (IllegalArgumentException e) {
393                 throw new SnapDataException(e, ERR_INVALID_PLAIN_TEXT)
394                         .formatWith(plaintextTypeString)
395                         .withReason(e.getMessage())
396                         .withResolution(String.format(RESOLUTION_INVALID_PLAIN_TEXT, Joiner
397                                 .on(JOINER_FORMAT).join(Constants.PlaintextType.values())));
398             }
399             Object out;
400             switch (plaintextType) {
401                 case BYTE_ARRAY:
402                 default:
403                     out = plaintext;
404                     break;
405                 case STRING:
406                     out = new String(plaintext, StandardCharsets.UTF_8);
407                     break;
408                 case JSON:
409                 case DOCUMENT:
410                     try {
411                         out = readObjectMapper.readValue(plaintext, Object.class);
412                     } catch (IOException e) {
413                         throw new SnapDataException(e, ERR_PARSE_JSON)
414                                 .formatWith(new String(plaintext, StandardCharsets.UTF_8))
415                                 .withResolution(RESOLUTION_ADDRESS_ISSUE);
416                     }
417                     break;
418             }
419             if (plaintextType == PlaintextType.DOCUMENT) {
420                 return documentUtility.newDocumentFor(document, out);
421             } else {
422                 walkResult.setValue(out);
423                 return document;
424             }
425         } catch (GeneralSecurityException e) {
426             throw new SnapDataException(e, ERR_DECRYPT_FIELD)
427                     .formatWith(walkResult.getActualPath())
428                     .withResolution(RESOLUTION_ADDRESS_ISSUE);
429         }
430     }
