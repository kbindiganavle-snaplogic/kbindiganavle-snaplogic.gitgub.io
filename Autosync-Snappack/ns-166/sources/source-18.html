79  * writes binary data to the output view. The skip rows property allows skipping the top rows

145         List<Map<String, Map<String, Object>>> fieldProperties = propertyValues
146                 .get(FWF_TABLE_PROP_FIELD_CONFIG);
147         if (fieldProperties != null) {
148             fieldInfoMap = new LinkedHashMap<>(fieldProperties.size());
149             String fieldName, fmtType, actType, formatString = STRING_NO_PADDING;
150             FormatType formatType;

155             for (Map<String, Map<String, Object>> map : fieldProperties) {

172                     formatString = STRING_NO_PADDING;
173                 } else {

176                             formatString = String.format(STRING_PADDING_FORMAT, fieldWidth);

181                         case NONE:

183                     }
184                 }
185                 fieldInfo = new FieldInfo();
186                 fieldInfo.setFieldName(fieldName);
187                 fieldInfo.setFieldWidth(fieldWidth);
188                 fieldInfo.setFormatString(formatString);
189                 fieldInfo.setActionType(actionType);
190                 fieldInfoMap.put(fieldName, fieldInfo);
191             }
192         }
193         isNullSafeAccess = Boolean.TRUE.equals(propertyValues.get(FWF_PROP_NULL_SAFE_ACCESS));

269      * Returns the header data that contains the field names formatted according to user specified
270      * widths
271      *
272      * @return headerData
273      */
274     private String getHeaderData() {
275         StringBuilder headerData = new StringBuilder();
276         String headerVal;
277         FieldInfo fieldInfo;
278         for (String fieldName : fieldInfoMap.keySet()) {
279             fieldInfo = fieldInfoMap.get(fieldName);
280             // Apply the format based on the selection for each field
281             if (fieldName.length() > fieldInfo.getFieldWidth()) {
282                 headerVal = fieldName.substring(0, fieldInfo.getFieldWidth());
283             } else if (fieldName.length() < fieldInfo.getFieldWidth()) {
284                 headerVal = String.format(fieldInfo.getFormatString(), fieldName);
285             } else {
286                 headerVal = fieldName;
287             }
288             headerData.append(headerVal);
289         }
290         return headerData.toString();
291     }

304             @Override
305             public void write(final WritableByteChannel writeChannel) throws IOException {
306                 OutputStream outputStream = Channels.newOutputStream(writeChannel);
307                 if (addHeader) {
308                     try {
309                         outputStream.write(getHeaderData().getBytes(StandardCharsets.UTF_8));
310                         outputStream.write(lineSeparator.getBytes(StandardCharsets.UTF_8));
311                     } catch (Exception ex) {
312                         log.error(Messages.ERROR_IN_PROCESSING, ex);
313                         throw new ExecutionException(ex, Messages.ERROR_IN_PROCESSING)
314                                 .withReason(ex.getMessage())
315                                 .withResolution(Messages.MESSAGE_CHECK_INPUT_DATA);
316                     }
317                 }
318                 for (Document document : inputViews.documents(inputViews.get())) {
319                     if (document != null) {
320                         // stop processing top n rows for the value in the property "SkipLines"
321                         if (skippedRows < rowsToSkip) {
322                             skippedRows++;
323                             continue;
324                         }
325                         process(document, outputStream);
326                     } else {
327                         break;
328                     }
329                 }
330                 onEndOfData(outputStream);
331             }
332         });
333     }

339     protected void process(Document inDoc, OutputStream outputStream) {
340         Map<String, Object> map = (Map<String, Object>) inDoc.get();
341         FieldInfo fieldInfo = null;
342         String fieldVal, formattedVal = null;
343         StringBuilder rowData = new StringBuilder();
344         String fieldName;
345         try {
346             for (Map.Entry<String, FieldInfo> entrySet : fieldInfoMap.entrySet()) {
347                 fieldName = entrySet.getKey();
348                 try {
349                     Object value = jsonPathUtil.read(fieldName, map);
350                     fieldVal = value == null ? StringUtils.EMPTY : ObjectType.toString(value);
351                     fieldInfo = entrySet.getValue();
352                     // Apply the format based on the selection for each field
353                     if (fieldVal.length() > fieldInfo.getFieldWidth()) {
354                         switch (fieldInfo.getActionType()) {
355                             case TRUNCATE_AT_END:
356                                 formattedVal = fieldVal.substring(0, fieldInfo.getFieldWidth());
357                                 break;
358                             case WRITE_TO_ERROR_VIEW:
359                                 Map<String, Object> error = Maps.newLinkedHashMap();
360                                 error.put(ERROR, Messages.RECORD_REDIRECTION_MESSAGE);
361                                 SnapDataException snapDataException = new SnapDataException(
362                                         documentUtility.newDocument(error),
363                                         Messages.RECORD_REDIRECTION_MESSAGE)
364                                         .withReason(Messages.RECORD_REDIRECTION_MESSAGE)
365                                         .withResolution(RECORD_REDIRECTION_MESSAGE_RESOLUTION,
366                                                 fieldName);
367                                 errorViews.write(snapDataException, inDoc);
368                                 continue;
369                             case NONE:
370                                 formattedVal = fieldVal;
371                                 break;
372                         }
373                     } else {
374                         formattedVal = String.format(fieldInfo.getFormatString(), fieldVal);
375                     }
376                     rowData.append(formattedVal);
377                 } catch (InvalidPathException e) {
378                     if (!isNullSafeAccess) {
379                         Map<String, Object> error = Maps.newLinkedHashMap();
380                         String errorMessage = String.format(ERROR_INVALID_FIELD_NAME_REASON,
381                                 fieldName);
382                         error.put(ERROR, errorMessage);
383                         SnapDataException ex = new SnapDataException(documentUtility
384                                 .newDocument(error), Messages.ERROR_IN_PROCESSING)
385                                 .withReason(errorMessage)
386                                 .withResolution(Messages.ERROR_INVALID_FIELD_NAME_RESOLUTION);
387                         errorViews.write(ex, inDoc);
388                     } else {
389                         if (fieldInfo != null) {
390                             formattedVal = String.format(fieldInfo.getFormatString(), StringUtils.EMPTY);
391                             rowData.append(formattedVal);
392                         }
393                     }
394                 }
395             }
396             outputStream.write(rowData.toString().getBytes(StandardCharsets.UTF_8));
397             outputStream.write(lineSeparator.getBytes(StandardCharsets.UTF_8));
398         } catch (IOException ex) {
399             log.error(Messages.ERROR_IN_PROCESSING, ex);
400             throw new ExecutionException(ex, Messages.ERROR_IN_PROCESSING)
401                     .withReason(ex.getMessage())
402                     .withResolution(Messages.MESSAGE_CHECK_INPUT_DATA);
403         }
404     }
