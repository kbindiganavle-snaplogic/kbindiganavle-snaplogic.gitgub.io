66 /**
67  * ExcelParser Snap reads binary data in XLS and XLSX formats from specified sheet
68  * and outputs document data
69  * <p>
70  * <p>
71  * The snap uses the following properties:
72  * <ul>
73  * <li>Sheet index or name - Sheet index (0 to n) or sheet name</li>
74  * <li>Star row - Number of row to start reading from</li>
75  * <li>End row - Number of row to stop reading at</li>
76  * <li>Contains headers - Specify whether or not the excel data contains a header line</li>
77  * <li>Column names - Specify a header to map the columns of the excel. If Contains headers
78  * property is false and column names are not set, column names are taken from excel data (A,
79  * B..etc.). This property is not required.
80  * </li>
81  * </ul>

231     public void configure(final PropertyValues propertyValues) throws ConfigurationException {
232         this.propertyValues = propertyValues;
233         sheetNameExprProperty = propertyValues.getAsExpression(SHEET_NAME_PROP);
234         sheetIndexExprProperty = propertyValues.getAsExpression(SHEET_INDEX_PROP);
235         startRowExprProperty = propertyValues.getAsExpression(START_ROW_PROP);
236         endRowExprProperty = propertyValues.getAsExpression(END_ROW_PROP);
237         headerRowExprProperty = propertyValues.getAsExpression(HEADER_ROW_PROP);
238         lastBlankHeadersExprProperty = propertyValues.getAsExpression(LAST_BLANK_HEADERS_PROP);
239         String sheetName = sheetNameExprProperty.eval(null);
240         String sheetIndex = sheetIndexExprProperty.eval(null);
241         if (StringUtils.isNotEmpty(sheetName)) {
242             sheetIndexOrName = sheetName;
243             isSheetFiteredByIndex = false;
244         } else if (StringUtils.isNotEmpty(sheetIndex)) {
245             sheetIndexOrName = sheetIndex;
246             isSheetFiteredByIndex = StringUtils.isNumeric(sheetIndexOrName);
247         } else {
248             throw new ConfigurationException(ERR_EXCEL_PARSE_SHEET_NOT_PROVIDED)
249                     .withReason(ERR_EXCEL_PARSE_SHEET_NOT_PROVIDED)
250                     .withResolution(ERR_EXCEL_PARSE_SHEET_NOT_PROVIDED_RES);
251         }
252         String startRowValue = startRowExprProperty.eval(null);
253         String endRowValue = endRowExprProperty.eval(null);
254 
255         startRow = getNumericPropertyValue(startRowValue, START_ROW_LABEL);
256         endRow = getNumericPropertyValue(endRowValue, END_ROW_LABEL);
257         //For support old version
258         isContainsHeader = Boolean.TRUE.equals(propertyValues.get(CONTAINS_HEADERS_PROP));
259         nullMissingColumns = Boolean.TRUE.equals(propertyValues.get(NULL_EMPTY_COLUMNS_PROP));
260         // startRow cannot be one when isContainsHeader is true
261         if (isContainsHeader && ONE.equals(startRow)) {
262             throw new ConfigurationException(ERR_EXCEL_START_ROW_ONE_WITH_HEADER_MSG)
263                     .withReason(ERR_EXCEL_START_ROW_ONE_WITH_HEADER_REASON)
264                     .withResolution(ERR_EXCEL_START_ROW_ONE_WITH_HEADER_RESOLUTION);
265         }
266 
267         if (startRow != null && endRow != null && startRow.compareTo(endRow) == 1) {
268             throw new ConfigurationException(ERR_EXCEL_PARSE_START_ROW_GREATER_THAN_END_ROW)
269                     .withReason(ERR_EXCEL_PARSE_START_ROW_GREATER_THAN_END_ROW)
270                     .withResolution(String.format(ERR_EXCEL_PARSE_START_END_ROW_INVALID_RES,
271                             START_ROW_LABEL));
272         }
273 
274         if (sheetIndexOrName == null || sheetIndexOrName.isEmpty()) {
275             throw new ConfigurationException(ERR_EXCEL_PARSE_SHEET_NOT_PROVIDED)
276                     .withReason(ERR_EXCEL_PARSE_SHEET_NOT_PROVIDED)
277                     .withResolution(ERR_EXCEL_PARSE_SHEET_NOT_PROVIDED_RES);
278         }
279         ExpressionProperty columnNamesExpressionProp = propertyValues.getAsExpression
280                 (COLUMN_NAMES_PROP);
281         if (columnNamesExpressionProp != null) {
282             List<Map<String, String>> columnNamesMapData = columnNamesExpressionProp.eval(null);
283             if (CollectionUtils.isNotEmpty(columnNamesMapData)) {
284                 columnPathExpressions = new ArrayList<>(columnNamesMapData.size());
285                 columnNames = new ArrayList<>();
286                 columnNamesMapData.forEach(tableRow -> {
287                     ExpressionProperty row = propertyValues.getExpressionPropertyFor(tableRow,
288                             COLUMN_PROP);
289                     columnPathExpressions.add(row);
290                 });
291                 columnPathExpressions.forEach(columnExp -> {
292                     String colName = columnExp.eval(null);
293                     columnNames.add(colName);
294                 });
295             }
296         }
297         columnNames = trimAll(columnNames);
298         isCellFormulasEvaluated = Boolean.FALSE.equals(propertyValues.get(EVALUATE_FORMULAS_PROP));
299         roundDateToSeconds = Boolean.TRUE.equals(propertyValues.get(ROUND_DATE_PROP));
300         setCustomLocale(propertyValues);
301         cellFormatting = propertyValues.get(CELL_FORMATTING_PROP);
302         if (cellFormatting == null) {
303             cellFormatting = FORMATTED;
304         }
305         // init header row
306         Object headerRowProp = headerRowExprProperty.eval(null);
307         Number headerRowPropNum = (Number) headerRowProp;
308         if (headerRowPropNum != null) {
309             headerRow = headerRowPropNum.intValue();
310             if (!isContainsHeader || headerRow < DEFAULT_HEADER_ROW) {
311                 headerRow = DEFAULT_HEADER_ROW;
312             }
313         }
314         // startRow should be set to Header row + 1 if it is blank and 'Contains header' property
315         // is checked
316         if (isContainsHeader && (startRow == null || startRow <= headerRow)) {
317             startRow = (long) headerRow + 1;
318         }
319         Object lastBlankHeadersProp = lastBlankHeadersExprProperty.eval(null);
320         Number lastBlankHeaderNum = (Number) lastBlankHeadersProp;
321         if (lastBlankHeaderNum != null) {
322             lastBlankHeaders = lastBlankHeaderNum.intValue();
323         }
324     }

326     private void setCustomLocale(PropertyValues propertyValues) {
327         String customLocaleInput = propertyValues.get(CUSTOM_LOCALE_PROP);
328         if (StringUtils.isNotBlank(customLocaleInput) &&
329                 !customLocale.getDisplayName().equals(customLocaleInput) &&
330                 !customLocale.toLanguageTag().equals(customLocaleInput)) {
331             for (Locale locale : LocaleUtils.availableLocaleList()) {
332                 if (locale.getDisplayName().equals(customLocaleInput) ||
333                         locale.toLanguageTag().equals(customLocaleInput)) {
334                     customLocale = locale;
335                     break;
336                 }
337             }
338             if (customLocale.equals(Locale.getDefault())) {
339                 throw new ConfigurationException(ERR_CUSTOM_LOCALE_INVALID)
340                         .withReason(REASON_CUSTOM_LOCALE_INVALID)
341                         .withResolution(RESOLUTION_CUSTOM_LOCALE_INVALID);
342             }
343         }

346     @Override
347     protected void doWork() {
348         if (inputViews.isEmpty()) {
349             LOGGER.debug(EXCEL_VIEW_NOT_CONNECTED);
350             return;
351         }
352         for (final BinaryInput binaryInput : inputViews.binaryInputs(inputViews.get())) {
353             BufferedInputStream inputStream = null;
354             ProcessSheet processSheet = null;
355             try {
356                 inputStream = new BufferedInputStream(
357                         Channels.newInputStream(binaryInput.getChannel()));
358                 processSheet = ProcessSheetFactory.create(binaryInput.getHeader(), inputStream,
359                         isCellFormulasEvaluated);
360                 processSheet.setSnapObjects(errorViews, documentUtility);
361                 if (processSheet instanceof XSSFProcessSheet) {
362                     tmpDir = getTmpDir(expressionUtils, propertyValues, errorViews);
363                     if (tmpDir == null) {
364                         return;
365                     }
366                     ((XSSFProcessSheet) processSheet).init(tmpDir);
367                 }
368                 processSheet.validateSheet(sheetIndexOrName, isSheetFiteredByIndex);
369                 ProcessSheetCompleteHandler handler = new ProcessSheetCompleteHandler() {
370                     @Override
371                     public void onComplete(Map<String, String> data) {
372                         // when data is null it will simply increment the dataCounter
373                         // otherwise it will do both - increment dataCounter and write the data
374                         //    to the output view
375                         incrementDataCounter();
376                         if (data != null) {
377                             writeData(binaryInput.getHeader(), data);
378                         }
379                     }
380                 };
381                 if (processSheet instanceof HSSFProcessSheet) {
382                     ((HSSFProcessSheet) processSheet).setExcelParser(this);
383                     if (isContainsHeader) {
384                         dataCounter = headerRow;
385                     }
386                 }
387                 snapEndRowProcessed = false;
388                 processSheet.parse(handler, isContainsHeader, headerRow, columnNames,
389                         nullMissingColumns, lastBlankHeaders, roundDateToSeconds,
390                         cellFormatting, customLocale);
391             } catch (IOException e) {
392                 SnapDataException ex = new SnapDataException(e,
393                         ERR_EXCEL_PARSE_READING_THE_INCOMING_BINARY_DATA)
394                         .withReason(ERR_EXCEL_PARSE_READING_THE_INCOMING_BINARY_DATA)
395                         .withResolution(ERR_EXCEL_PARSE_READING_INPUT_DATA);
396                 errorViews.write(ex, createErrorDocument(binaryInput));
397             } catch (ExecutionException e) {
398                 SnapDataException ex = new SnapDataException(e, e.getMessage())
399                         .withReason(e.getReason())
400                         .withResolution(e.getResolution());
401                 errorViews.write(ex, createErrorDocument(binaryInput));
402             } catch (SuggestViewAbortException e) {
403                 throw e;
404             } catch (Exception e) {
405                 SnapDataException ex = new SnapDataException(e, ERR_PARSE)
406                         .withResolution(RESOLUTION_ADDRESS_ISSUE);
407                 errorViews.write(ex, createErrorDocument(binaryInput));
408             } finally {
409                 // Need to close stream here because must handle case
410                 // when ProcessSheet constructor throws exception
411                 consumeClose(inputStream, LOGGER);
412                 if (processSheet != null) {
413                     processSheet.cleanup();
414                 }
415             }
416         }
417     }

461     private Long getNumericPropertyValue(String propValue, String property) {
462         Long value = null;
463         String message;
464         if (!StringUtils.isBlank(propValue)) {
465             if (NumberUtils.isNumber(propValue)) {
466                 value = Long.valueOf(propValue);
467                 if (value < DEFAULT_MIN_START_END_ROW_VALUE) {
468                     message = String.format(ERR_EXCEL_PARSE_START_END_ROW_INVALID, property, value,
469                             DEFAULT_MIN_START_END_ROW_VALUE);
470                     throw new ConfigurationException(message)
471                             .withReason(message)
472                             .withResolution(String.format(ERR_EXCEL_PARSE_START_END_ROW_INVALID_RES,
473                                     property));
474                 }
475             } else {
476                 message = String.format(ERR_EXCEL_PARSE_START_END_ROW_SHOULD_BE_NUMERIC, property);
477                 throw new ConfigurationException(message)
478                         .withReason(message)
479                         .withResolution(String.format(ERR_EXCEL_PARSE_START_END_ROW_INVALID_RES,
480                                 property));
481             }
482         }
483 
484         return value;
485     }

500     private Document createErrorDocument(BinaryInput binaryInput) {
501         Map<String, Object> errorMap = new LinkedHashMap<>();
502         Document headerDocument = binaryInput.getHeader();
503         Object headerContents = null;
504         if (headerDocument != null) {
