59  * Joins data from two input views. Inner, Left, and Right equi-joins are supported. If the two
60  * input streams are ordered, then a merge join optimization can be applied to the document
61  * streams. The order of the output is determined by the left side input view. Joins can also
62  * only be performed on documents that have a map as it's root data structure.
63  *
64  * <pre>
65  * {@code
66  * Left Input View Data: inp1
67  * [
68  *  {
69  *      "name": "Bob"
70  *  },
71  *  {
72  *      "name": "Sally"
73  *  },
74  *  {
75  *      "name": "Sam"
76  *  }
77  * ]
78  *
79  * Right Input View Data: inp2
80  * [
81  *  {
82  *      "name": "Bob"
83  *      "position": 2
84  *  },
85  *  {
86  *      "name": "Pete"
87  *      "position": 3
88  *  },
89  *  {
90  *      "name": "Sally"
91  *      "position": 1
92  *  }
93  * ]
94  *
95  * Inner join with left path "name" and right path "name" result:
96  *
97  * [
98  *  {
99  *      "name": "Bob",
100  *      "inp2.name": "Bob",
101  *      "position": 2
102  *  },
103  *  {
104  *      "name": "Sally",
105  *      "inp2.name": "Sally",
106  *      "position": 1
107  *  }
108  * ]
109  *
110  * Left join with left path "name" and right path "name" result:
111  *
112  * [
113  *  {
114  *      "name": "Bob",
115  *      "inp2.name": "Bob",
116  *      "position": 2
117  *  },
118  *  {
119  *      "name": "Sally",
120  *      "inp2.name": "Sally",
121  *      "position": 1
122  *  },
123  *  {
124  *      "name": "Sam"
125  *  }
126  * ]
127  *
128  * Right join with left path "name" and right path "name" result:
129  *
130  * [
131  *  {
132  *      "name": "Bob",
133  *      "inp2.name": "Bob",

135  *  },

291     @Override
292     public void  execute() throws ExecutionException {
293         // do nothing if input view is not connected
294         if (inputViews.size() < 2) {
295             return;
296         }
297         int pollTime = INITIAL_POLL_TIME;
298         Set<String> viewNames = Sets.newHashSet(leftSideInputViewName, rightSideInputViewName);
299         Map<String, Document> dataMap = inputViews.select(viewNames, pollTime, TIME_UNIT);
300         while (dataMap != null && !viewNames.isEmpty()) {
301             if (!dataMap.isEmpty()) {
302                 if (dataMap.containsKey(leftSideInputViewName)) {
303                     if (processData(dataMap.get(leftSideInputViewName), JoinSide.LEFT)) {
304                         viewNames.remove(leftSideInputViewName);
305                     }
306                 }
307                 if (dataMap.containsKey(rightSideInputViewName)) {
308                     if (processData(dataMap.get(rightSideInputViewName), JoinSide.RIGHT)) {
309                         viewNames.remove(rightSideInputViewName);
310                     }
311                 }
312                 // We are receiving data actively, lets keep the poll time fixed at the initial
313                 // poll rate.
314                 pollTime = INITIAL_POLL_TIME;
315             } else {
316                 // We are not receiving any data. Lets increment poll time.
317                 pollTime += INCREMENTS;
318                 if (pollTime > MAX_POLL_TIME) {
319                     // If the poll time exceeds the max allowed poll time,
320                     // lets fix the poll time as the max allowed poll time.
321                     pollTime = MAX_POLL_TIME;
322                 }
323             }
324             dataMap = inputViews.select(viewNames, pollTime, TIME_UNIT);
325         }
326     }

346     private boolean processData(final Document document, JoinSide joinSide) {
347         boolean isEnd = false;
348         try {
349             if (document != null) {
350                 executeJoinOn(joinSide, document);
351             } else {
352                 isEnd = true;
353                 executeJoinOn(joinSide, null);
354             }
355         } catch (ClassCastException ex) {
356             RuntimeException runtimeException = new RuntimeException(ERR_TYPE_NOT_COMPARABLE, ex);
357             handleException(document, joinSide, runtimeException);
358         }
359         return isEnd;
360     }

417     public void generateSample(InputViews inputViews, OutputViews outputViews,
418                                ErrorViews errorViews) throws ExecutionException {
419         // Get a list of all document iterators
420         // do nothing if input view is not connected
421         if (inputViews.size() < 2) {
422             return;
423         }
424         Set<String> viewNames = Sets.newHashSet(leftSideInputViewName, rightSideInputViewName);
425         Map<String, Document> dataMap = inputViews.select(viewNames);
426         while (dataMap != null) {
427             if (!dataMap.isEmpty()) {
428                 if (dataMap.containsKey(leftSideInputViewName) &&
429                         dataMap.containsKey(rightSideInputViewName)) {
430                     if (dataMap.get(leftSideInputViewName) != null &&
431                             dataMap.get(rightSideInputViewName) != null) {
432                         // Keep writing to output until one of the two document iterators is empty
433                         outputViews.write(fakeJoin(dataMap.get(leftSideInputViewName),
434                                 dataMap.get(rightSideInputViewName)));
435                     } else {
436                         break;
437                     }
438                 } else {
439                     break;
440                 }
441             } else {
442                 break;
443             }
444             dataMap = inputViews.select(viewNames);
445         }

448     /**
449      * Join is a schema altering snap that is capable of not having any output during prepare.
450      * That is an unacceptable outcome for SampleDataGenerator since it needs the potential
451      * altered output schema in case there are documents further down the stream.
452      *
453      * Here we simulate a join by joining documents regardless of whether or not they meet the join
454      * condition.  The output data types are the same during fake join and normal join so this junk
455      * data satifies SampleDataGenerator's ability to store documents from which one could determine
456      * output schema at a later time (parallelize).
457      *
458      * @param left left incoming document
459      * @param right right incoming document
460      * @return the fake join of the two incoming documents
461      */
462     private Document fakeJoin(Document left, Document right) {
463         Map<String, Object> leftObject = left.get(Map.class);
464         Map<String, Object> rightObject = right.get(Map.class);
465         Map<String, Object> joinObject = new LinkedHashMap(leftObject.size() + rightObject.size());
466         for (Map.Entry<String, Object> leftColumn : leftObject.entrySet()) {
467             joinObject.put(String.format(LEFT_PREFIX_PATTERN, leftColumn.getKey()),
468                     leftColumn.getValue());
469         }
470         for (Map.Entry<String, Object> rightColumn : rightObject.entrySet()) {
471             joinObject.put(String.format(RIGHT_PREFIX_PATTERN, rightColumn.getKey()),
472                     rightColumn.getValue());
473         }
474         return documentUtility.newDocument(joinObject);
475     }
