81  * Excel Multi Sheet Formatter Snap reads the documents from multiple inputs, converts the documents
82  * in XLSX format into specified sheets and writes out binary data.
83  *
84  * <p>
85  * The snap uses the following property:
86  * <ul>
87  * <li>Sheet name - Sheet name</li>
88  * <li>Write header - This option specifies whether or not the Snap should write the header
89  * to the output.</li>
90  * </ul>
91  *
92  * @author kgade
93  * @author smudassir

143     public void configure(final PropertyValues propertyValues) throws ConfigurationException {
144         // Note that an input view "name" is a view ID instead of its label. The view label is user-
145         // defined, and the view name is system-generated in ascending order.
146         for (InputView inputView : inputViews.getAll().values()) {
147             // Initially, values of viewNameToSheetMap are view labels, which are used as default
148             // sheet names.
149             viewNameToSheetMap.put(inputView.getName(), inputView.getLabel());
150         }
151         int viewSize = viewNameToSheetMap.size();
152         // Prepare another map of view labels to view names,
153         // and also make sure all view labels are unique
154         Map<String, String> viewLabelToNameMap = new HashMap<>(viewSize);
155         Set<String> viewLabels = new HashSet<>(viewSize);
156         for (Map.Entry<String, String> entry : viewNameToSheetMap.entrySet()) {
157             String viewLabel = entry.getValue();
158             viewLabelToNameMap.put(viewLabel, entry.getKey());
159             if (viewLabels.contains(viewLabel)) {
160                 throw new ConfigurationException(ERR_DUPLICATE_INPUT_VIEW_LABEL)
161                         .formatWith(viewLabel)
162                         .withReason(REASON_DUPLICATE_NAMES)
163                         .withResolution(RESOLUTION_ADDRESS_ISSUE);
164             } else {
165                 viewLabels.add(viewLabel);
166             }
167         }
168         // process 'Sheet mapping' table property
169         List<Map<String, String>> routeList = propertyValues.get(SHEET_MAPPING_PROP);
170         if (CollectionUtils.isNotEmpty(routeList)) {
171             for (Map<String, String> route : routeList) {
172                 String viewLabel = JsonPath.read(route, viewPath);
173                 String sheetName = propertyValues.getExpressionPropertyFor(route, SHEET_NAME_PROP)
174                         .eval(null);
175                 if (StringUtils.isNotBlank(sheetName)
176                         && viewLabelToNameMap.containsKey(viewLabel)) {
177                     viewNameToSheetMap.put(viewLabelToNameMap.get(viewLabel), sheetName);
178                 }
179             }
180         }
181 
182         isWriteHeader = propertyValues.getBoolean(WRITE_HEADER_PROP, true);
183         ignoreEmptyStream = propertyValues.getBoolean(IGNORE_EMPTY_STREAM_PROP, false);
184         if (propertyValues.inImmediateMode()) {
185             throw new ConfigurationException(INCOMPATIBLE_USAGE_OF_THIS_SNAP)
186                     .withReason(THE_EXCEL_FORMATTER_SNAP_CANNOT_BE_USED_IN_AN_ALWAYS_ON_TASK)
187                     .withResolution(REMOVE_THE_SNAP_FROM_THE_PIPELINE);
188         }
189         convertString = propertyValues.getBoolean(CONVERT_NUMERIC_STRING_PROP, false);
190         translateDateTimes = propertyValues.getBoolean(TRANSLATE_DATEANDTIME_TYPE_PROP, false);
191         dateCellStyle.setDataFormat(createHelper.createDataFormat()

193         convertFormulas = propertyValues.getBoolean(CONVERT_FORMULAS, false);

206     @Override
207     protected void doWork() {
208         if (inputViews.isEmpty()) {
209             LOGGER.debug(EXCEL_VIEW_NOT_CONNECTED);
210             return;
211         }
212         // Check if there are any non-empty input documents.
213         boolean hasNext = false;
214         int viewSize = inputViews.size();
215         final Map<String, Document> firstDocs = new LinkedHashMap<>();
216         Map<String, Document> documents;
217         do {
218             // We should use select() because the Snap has multiple input views.
219             documents = inputViews.select(TIMEOUT_MILLISECONDS, TimeUnit.MILLISECONDS);
220             if (!documents.isEmpty()) {
221                 for (Map.Entry<String, Document> entry : documents.entrySet()) {
222                     Document document = entry.getValue();
223                     // Copy to firstDocs, which will be used as the first set of input documents
224                     // in BinaryOutput() below.
225                     firstDocs.put(entry.getKey(), document);
226                     if (document != null) {
227                         hasNext = true;
228                     }
229                 }
230                 if (hasNext) {
231                     // Found non-empty input documents
232                     break;
233                 }
234             }
235         } while (documents.size() != viewSize);
236 
237         if (ignoreEmptyStream && !hasNext && documents.size() == viewSize) {
238             // Found the end of empty input document streams for all input views
239             LOGGER.debug("No input document");
240             return;
241         }
242 
243         outputViews.write(new BinaryOutput() {
244             @Override
245             public Document getHeader() {
246                 Map<String, Object> headerMap = new LinkedHashMap<>(1);
247                 headerMap.put(SdkConstants.CONTENT_TYPE, MediaType.OOXML_SHEET.toString());
248                 return documentUtility.newDocument(headerMap);
249             }

251             @SuppressWarnings("unchecked")
252             @Override
253             public void write(final WritableByteChannel writeChannel) throws IOException {
254                 // init workbook sheets in the correct order
255                 int viewSize = inputViews.size();
256                 Map<String, SXSSFSheet>  viewSheetMap = new HashMap<>(viewSize);
257                 Map<String, Integer> rowNumbers = new HashMap<>(viewSize);
258                 // input view names in the value set of viewNameToSheetMap are in the correct order
259                 for (Map.Entry<String, String> entry : viewNameToSheetMap.entrySet()) {
260                     String viewName = entry.getKey();
261                     String sheetName = entry.getValue();
262                     // SXSSF (Streaming Usermodel API) is an API-compatible streaming extension
263                     // of XSSF to be used when very large spreadsheets have to be produced,
264                     // and heap space is limited.
265                     // keep 100 (SXSSFWorkbook.DEFAULT_WINDOW_SIZE) rows in memory,
266                     // exceeding rows will be flushed to disk
267                     SXSSFSheet sheet = StringUtils.isBlank(sheetName) ? workbook.createSheet()
268                             : workbook.createSheet(sheetName);
269                     viewSheetMap.put(viewName, sheet);
270                     rowNumbers.put(viewName, 0);
271                 }
272                 // format documents to sheets
273                 boolean hasNext;
274                 Map<String, Document> documents = firstDocs;
275                 do {
276                     hasNext = false;
277                     for (Map.Entry<String, Document> entry : documents.entrySet()) {
278                         String viewName = entry.getKey();
279                         Document document = entry.getValue();
280                         if (document != null) {
281                             hasNext = true;
282                             Object data = document.get();
283                             if (data instanceof Map) {
284                                 int rowNum = rowNumbers.get(viewName);
285                                 SXSSFSheet sheet = viewSheetMap.get(viewName);
286                                 try {
287                                     // format one row
288                                     rowNum = process(sheet, rowNum, (Map<String, Object>) data);
289                                 } catch (IllegalArgumentException e) {
290                                     SnapDataException ex =
291                                             new SnapDataException(ERR_WRITING_DATA_TO_EXCEL_SHEET)
292                                                     .formatWith(sheet.getSheetName())
293                                                     .withReason(e.getMessage())
294                                                     .withResolution(PLEASE_PROVIDE_VALID_INPUT);
295                                     errorViews.write(ex, document);
296                                 } finally {
297                                     document.acknowledge();
298                                     rowNumbers.put(viewName, ++rowNum);
299                                 }
300                             } else {
301                                 String className = data.getClass().getSimpleName();
302                                 SnapDataException ex = new SnapDataException(
303                                         ERR_FLATTENED_MAP_REASON).formatWith(className)
304                                         .withReason(ERR_FLATTENED_MAP_REASON)
305                                         .withResolution(ERR_FLATTENED_MAP_RESOLUTION);
306                                 errorViews.write(ex, document);
307                             }
308                         }
309                     }
310                     // The do-while condition has to be determined before inputViews::select().
311                     // Otherwise, the do-while loop will exit prematurely and the "input document
312                     // not consumed" error will occur.
313                     // If all document values are null, all input document streams have ended.
314                     hasNext = hasNext || documents.size() != viewSize;
315                     // Pull more input documents for the next iteration of the do-while loop.
316                     documents = inputViews.select(TIMEOUT_MILLISECONDS, TimeUnit.MILLISECONDS);
317                 } while (hasNext);
318                 // handle ignoreEmptyStream
319                 if (ignoreEmptyStream) {
320                     // check if there is no document in all input views
321                     boolean areAllEmpty = true;
322                     for (Map.Entry<String, Integer> entry : rowNumbers.entrySet()) {
323                         if (entry.getValue() != 0) {
324                             areAllEmpty = false;
325                             break;
326                         }
327                     }
328                     if (areAllEmpty) {
329                         // if there is no document in all input views, an empty output binary
330                         // object with zero-byte data is written to the output view
331                         workbook.dispose();
332                         return;
333                     }
334                     // remove empty sheets
335                     for (Map.Entry<String, Integer> entry : rowNumbers.entrySet()) {
336                         if (entry.getValue() == 0) {
337                             String viewName = entry.getKey();
338                             SXSSFSheet sheet = viewSheetMap.get(viewName);
339                             int index = workbook.getSheetIndex(sheet);
340                             workbook.removeSheetAt(index);
341                         }
342                     }
343                 }
344                 // write workbook to output stream
345                 try (OutputStream outputStream = Channels.newOutputStream(writeChannel)) {
346                     workbook.write(outputStream);
347                 } finally {
348                     workbook.dispose();
349                 }
350             }

352             private int process(Sheet sheet, int rowNum,  Map<String, Object> data) {
353                 Row row = sheet.createRow(rowNum);
354                 //Column names should be the first row
355                 if (rowNum == 0 && isWriteHeader) {
356                     createCell(row, data.keySet());
357                     row = sheet.createRow(++rowNum);
358                 }
359                 createCell(row, data.values());
360                 return rowNum;
361             }

363             @SuppressWarnings("rawtypes")
364             private void createCell(Row row, Collection data) {
365                 int cellCounter = 0;
366                 for (Object value : data) {
367                     if (value != null) {
368                         Cell cell = row.createCell(cellCounter);
369                         if (value instanceof LocalDate && translateDateTimes) {
370                             cell.setCellValue(((LocalDate) value).toDate());
371                             cell.setCellStyle(dateCellStyle);
372                         } else if (value instanceof DateTime && translateDateTimes) {
373                             cell.setCellValue(((DateTime) value)
374                                     .toGregorianCalendar());
375                             cell.setCellStyle(dateCellStyle);
376                         } else if (value instanceof Number) {
377                             cell.setCellValue(((Number) value).doubleValue());
378                         } else if (value instanceof String) {
379                             String stringValue = (String) value;
380                             if (convertFormulas && stringValue.startsWith("=")) {
381                                 cell.setCellFormula(stringValue.substring(1));
382                             } else {
383                                 if (convertString && NumberUtils.isParsable(stringValue)) {
384                                     cell.setCellValue(Double.parseDouble(stringValue));
385                                 } else {
386                                     cell.setCellValue(stringValue);
387                                 }
388                             }
389                         } else {
390                             cell.setCellValue(value.toString());
391                         }
392                     } else {
393                         // No value to set for this cell
394                         row.createCell(cellCounter);
395                     }
396                     cellCounter++;
397                 }
398             }
399         });
