283     @Override
284     @SuppressWarnings("unchecked")
285     protected void process(final Document document, final String inputViewName) {
286         checkMemoryCount++;
287         totalDocumentCount++;
288         if (checkMemoryCount > CHECK_MEMORY_INTERVAL) {
289             checkMemoryCount = 0;
290             checkMemory(minMemory, outOfResourceTimeoutMillis, statusReporter, STATUS_MESSAGE,
291                     LOG, totalDocumentCount);
292         }
293         Object data = document.get();
294         if (data instanceof Map) {
295 
296             LOG.trace("process: data = {}", data);
297 
298             // we just got new data, so if there's an outstanding flushTask, cancel it
299             if (flushTask != null) {
300                 if (flushTask.cancel()) {
301                     LOG.trace("process:  cancelled flushTask");
302                 }
303                 flushTask = null;
304             }
305 
306             boolean startFlushTask = false;
307 
308             synchronized (group) {
309                 group.add((Map<String, Object>) data);
310                 if (flushTimer != null) {
311                     LOG.trace("process:  added data to group; size = {}", group.size());
312                 }
313                 if (group.size() == groupSize) {
314                     // we have a full group, so write it; no flush task needed at the moment
315                     write(group, false);
316                 } else if (flushTimer != null) {
317                     // we have a partial group, so start a timer task to flush it
318                     startFlushTask = true;
319                 }
320             }
321 
322             if (startFlushTask) {
323                 // schedule a flushTask to write the partial group; if process() is called again
324                 // with a new input document before that, this task will be cancelled (see above)
325                 flushTask = new TimerTask() {
326                     @Override
327                     public void run() {
328                         LOG.debug("running flushTask");
329                         write(group, false);
330                     }
331                 };
332                 LOG.trace("process:  scheduling flushTask with {} ms delay", flushTimeoutMs);
333                 flushTimer.schedule(flushTask, flushTimeoutMs);
334             }
335 
336             document.acknowledge();
337         } else {
338             errorViews.write((SnapDataException) new SnapDataException(ERR_INVALID_INPUT_DATA)
339                     .withReason(REASON_NOT_MAP_DATA)
340                     .withResolution(RESOLUTION_ADDRESS_ISSUE), document);
341         }
342     }

344     private void write(final List<Map<String, Object>> group, boolean last) {
345         synchronized (group) {
346             if (!group.isEmpty()) {
347                 LOG.debug("write: writing group {} of size {}", ++groupIndex, group.size());
348                 try {
349                     Map<String, Object> map = new LinkedHashMap<>(1);
350                     targetField.write(map, new ArrayList<>(group));
351                     outputViews.write(documentUtility.newDocument(map));
352                     group.clear();
353                 } catch (InvalidPathException e) {
354                     errorViews.write((SnapDataException) new SnapDataException(e, ERR_INVALID_PATH)
355                             .formatWith(targetField.toString())
356                             .withReason(e.getMessage())
357                             .withResolution(RESOLUTION_ADDRESS_ISSUE));
358                 }
359                 if (!last && dynamicMode) {
360                     adjustGroupSize();
361                 }
362             }
363         }
364         if (flushTimer != null) {
365             flushTimer.purge(); // most tasks will be cancelled, so this is important
366         }
367     }

402     private void adjustGroupSize() {
403         checkState(dynamicMode);
404 
405         // The key metric used here: the fraction of max memory that's available.
406         // The value ranges from 0 to 1.
407         double availMem = memoryUtil.availableMemoryFraction();
408 
409         // Calculate the moving average (MA) of availMem using memSamples, an EvictingQueue which
410         // stores the most recent NUM_MEM_SAMPLES values of availMem. Current value is added below.
411         double availMemMA = availMemSamples.isEmpty() ? availMem : meanOf(availMemSamples);
412 
413         // Calculate how much the current value deviates from the moving average.
414         double availMemDelta = availMem - availMemMA;
415 
416         // Now that we've already used memSamples to calculate the MA, add the new sample to it.
417         availMemSamples.add(availMem);
418 
419         // Check whether we should trigger a garbage collection.
420         if (memoryUtil.requestGC(availMem)) {
421             gcCount++;
422         }
423 
424         // Calculate the sigmoid function for the partial value for the groupSizeFraction.
425         // This value also ranges from 0 to 1.
426         double sigmoidResult = sigmoidFunc.applyAsDouble(availMem);
427 
428         // Calculate the groupSizeFraction by adding two components:
429         // sigmoidResult, based on the current available memory, and
430         // availMemDelta, how much the current value deviates from the moving average.
431         // Including the availMemDelta helps to dampen spikes (rapid increases/decreases).
432         double groupSizeFraction = sigmoidResult + availMemDelta;
433 
434         // Multiply groupSize from settings by groupSizeFraction to get effective groupSize.
435         int newGroupSize = (int) round(requestedGroupSize * groupSizeFraction);
436 
437         // Constrain group size to the range [minGroupSize..requestedGroupSize].
438         newGroupSize = max(minGroupSize, min(requestedGroupSize, newGroupSize));
439 
440         // Track stats and report when groupSize is changing.
441         minAvailMem = min(minAvailMem, availMem);
442         maxAvailMem = max(maxAvailMem, availMem);
443         sumGroupSize += newGroupSize;
444         if (newGroupSize != groupSize) {
445             minAdjustedGroupSize = min(minAdjustedGroupSize, newGroupSize);
446             maxAdjustedGroupSize = max(maxAdjustedGroupSize, newGroupSize);
447             int avgGroupSize = (int) (sumGroupSize / (groupIndex + 1));
448             statusReporter.notify(GROUP_SIZE_STATUS,
449                     availMem * 100, minAvailMem * 100, maxAvailMem * 100,
450                     newGroupSize, minAdjustedGroupSize, maxAdjustedGroupSize, avgGroupSize,
451                     sigmoidFunc.describe(),
452                     gcCount);
453         }
454 
455         groupSize = newGroupSize;
456     }
