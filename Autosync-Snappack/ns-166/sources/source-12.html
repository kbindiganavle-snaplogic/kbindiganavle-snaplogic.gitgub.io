225     private String getTransformation(final Key key, Object requested) {
226         if (requested == null || AUTO.equals(requested) || StringUtils.EMPTY.equals(requested)) {
227             String retval = DEFAULT_TRANSFORMS.get(key.getAlgorithm());
228             if (retval == null) {
229                 throw new SnapDataException(ERR_TRANSFORMATION_TYPE)
230                         .withReason(REASON_TRANSFORMATION_TYPE)
231                         .withResolution(RESOLUTION_TRANSFORMATION_TYPE);
232             }
233             return retval;
234         }
235         return requested.toString();
236     }

238     private Document encryptOneField(final Document document, final WalkResult walkResult) {
239         try {
240             Object value = walkResult.getValue();
241             PlaintextType plaintextType;
242             byte[] plaintext;
243             if (value instanceof byte[]) {
244                 plaintext = (byte[]) value;
245                 plaintextType = Constants.PlaintextType.BYTE_ARRAY;
246             } else if (value instanceof String) {
247                 plaintext = ((String) value).getBytes(StandardCharsets.UTF_8);
248                 plaintextType = Constants.PlaintextType.STRING;
249             } else {
250                 String flatValue = writeObjectMapper.writeValueAsString(walkResult.getValue());
251                 plaintext = flatValue.getBytes(StandardCharsets.UTF_8);
252                 plaintextType = walkResult.getActualPath().isRoot() ? PlaintextType.DOCUMENT :
253                         PlaintextType.JSON;
254             }
255             Object transformationObject = transformType.eval(document);
256             CryptoKeyHolder keyHolder;
257             if (account instanceof CryptoEmptyAccount) {
258                 String key = advancedOptions.cryptoKey.eval(document).orElse(StringUtils.EMPTY);
259                 if (StringUtils.isBlank(key)) {
260                     throw new SnapDataException(ERR_EMPTY_KEY)
261                             .withReason(REASON_EMPTY_KEY)
262                             .withResolution(RESOLUTION_EMPTY_KEY);
263                 }
264                 if (StringUtils.contains(key, BEGIN_CERTIFICATE)) {
265                     keyHolder = new CryptoKeyHolderForX509CertificatePublicKey(key);
266                 } else {
267                     keyHolder = new CryptoKeyHolderForSymmetricKey(key,
268                             CoercionUtils.getAlgorithmFromTransform(transformationObject));
269                 }
270             } else {
271                 if (account instanceof KeyStoreAccount) {
272                     ((KeyStoreAccount) account).setEncrypt();
273                 }
274                 keyHolder = account.connect();
275             }
276             Key key = keyHolder.getKey();
277             String transformation = getTransformation(key, transformType.eval(document));
278             Cipher cipher = Cipher.getInstance(transformation, CRYPTO_PROVIDER);
279             Optional<Object> optionalIV = advancedOptions.iv.eval(document);
280             if (!optionalIV.isPresent()) {
281                 cipher.init(Cipher.ENCRYPT_MODE, key, secureRandom);
282             } else {
283                 Object iv = optionalIV.get();
284                 IvParameterSpec ivParameterSpec = new IvParameterSpec(
285                         CoercionUtils.toByteArray(iv, advancedOptions.iv.getExpression()));
286                 cipher.init(Cipher.ENCRYPT_MODE, key, ivParameterSpec, secureRandom);
287             }
288             byte[] ivBytes = cipher.getIV();
289             Map<String, Object> wrapper = new LinkedHashMap<>();
290             wrapper.put(TRANSFORMATION, transformation);
291             if (ivBytes != null && ivBytes.length > 0) {
292                 wrapper.put(IV_PROP, Base64.encodeToString(ivBytes, true));
293             }
294             if (StringUtils.equalsIgnoreCase(transformation, AES_CTR_NOPADDING)) {
295                 cipher.init(Cipher.ENCRYPT_MODE, key, new IvParameterSpec(ivBytes), secureRandom);
296             }
297             wrapper.put(TYPE, plaintextType.toString());
298             // get 'aad' value from input data
299             Map<String, Object> data = document.get(Map.class);
300             Object aadValue = data.get(AAD);
301             if (StringUtils.equalsIgnoreCase(transformation, AES_GCM_NOPADDING) &&
302                     aadValue != null) {
303                 if (keyHolder instanceof CryptoKeyHolderForPassPhrase) {
304                     key = ((CryptoKeyHolderForPassPhrase) keyHolder).getSHA256Key();
305                 }
306                 Cipher aeadCipher = Cipher.getInstance(transformation);
307                 aeadCipher.init(Cipher.ENCRYPT_MODE, key,
308                         new GCMParameterSpec(AUTH_TAG_SIZE, ivBytes));
309                 byte[] aad = aadValue.toString()
310                         .getBytes(StandardCharsets.UTF_8);
311                 aeadCipher.updateAAD(aad);
312                 byte[] ciphertext = aeadCipher.doFinal(plaintext);
313                 byte[] aead = Arrays.copyOfRange(ciphertext, ciphertext.length - AEAD_SIZE,
314                         ciphertext.length);
315                 ciphertext = Arrays.copyOfRange(ciphertext, 0, ciphertext.length - AEAD_SIZE);
316                 wrapper.put(CIPHERTEXT, Base64.encodeToString(ciphertext, true));
317                 wrapper.put(AAD, Base64.encodeToString(aad, true));
318                 wrapper.put(AEAD_TAG, Base64.encodeToString(aead, true));
319                 wrapper.put(KEY_PARAMS, keyHolder instanceof CryptoKeyHolderForPassPhrase ?
320                         ((CryptoKeyHolderForPassPhrase) keyHolder).getSHA256Parameters() :
321                         keyHolder.getParameters());
322             } else {
323                 byte[] ciphertext = cipher.doFinal(plaintext);
324                 wrapper.put(CIPHERTEXT, Base64.encodeToString(ciphertext, true));
325                 wrapper.put(KEY_PARAMS, keyHolder.getParameters());
326             }
327             Object newRoot;
328             if (advancedOptions.encapsulate) {
329                 String flatWrapper = snapObjectMapper.writeValueAsString(wrapper);
330                 newRoot = walkResult.setValue(String.format(ENC_FORMAT, flatWrapper));
331             } else {
332                 newRoot = walkResult.setValue(wrapper);
333             }
334             if (walkResult.getActualPath().isRoot()) {
335                 Map<String, Object> rootWrapper = new LinkedHashMap<>();
336                 rootWrapper.put("value", newRoot);
337 
338                 return documentUtility.newDocumentFor(document, rootWrapper);
339             } else {
340                 return document;
341             }
342         } catch (IOException e) {
343             throw new SnapDataException(e, ERR_FIELD_TO_JSON)
344                     .withReason(e.getMessage())
345                     .withResolution(RESOLUTION_ADDRESS_ISSUE);
346         } catch (GeneralSecurityException e) {
347             throw new SnapDataException(e, ERR_GENERAL_SECURITY)
348                     .withReason(e.getMessage())
349                     .withResolution(RESOLUTION_ADDRESS_ISSUE);
350         }
351     }
