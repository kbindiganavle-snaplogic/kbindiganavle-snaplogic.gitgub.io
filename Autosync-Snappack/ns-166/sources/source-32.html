303      * The values specified in this snap's table are used to fetch the corresponding entries from
304      * the Xml input. These values are then combined with the specified Velocity template to create
305      * a new document that follows the template
306      */
307     @Override
308     protected void process(final Document document, final String inputViewName) {
309         String templateOut = editorProperty.eval(document);
310         if (!useDefaultValue) {
311             try {
312                 templateOut = xmlUtils.stripNodeAndAttributeNodes(templateOut,
313                         documentBuilderFactory);
314             } catch (ExecutionException e) {
315                 SnapDataException ex = new SnapDataException(e, e.getMessage())
316                         .withResolution(e.getReason())
317                         .withResolution(e.getResolution());
318                 errorViews.write(ex, document);
319                 return;
320             }
321         }
322         if (shouldValidate) {
323             try {
324                 xmlUtils.validate(schemaValidator, IOUtils.toInputStream(templateOut, UTF_8));
325             } catch (Exception e) {
326                 errorViews.write(Utils.createDataExceptionForInvalidXML(e), document);
327                 return;
328             }
329         }
330         Map<String, String> xmlOutput = new LinkedHashMap<>();
331         // Attempt to remove extra spaces in the xml string, leave it as it was if error happens
332         // since the xml string here is supposed to be a valid one.
333         templateOut = getMinifiedXml(templateOut);
334         xmlOutput.put(XML_FIELDNAME, templateOut);
335         outputViews.write(documentUtility.newDocument(xmlOutput), document);
336     }

369     private Types setValidatorAndGetTypesFrom(String schemaFile, String xPathForSchemaElement,
370             List<Map<String, String>> namespaceContextTable) {
371         if (StringUtils.isBlank(schemaFile)) {
372             return Types.EMPTY_TYPES;
373         }
374         URL schemaFileURL;
375         try {
376             schemaFileURL = urlEncoder.validateAndEncodeURI(schemaFile).toURL();
377         } catch (MalformedURLException e) {
378             throw new ConfigurationException(e, COULD_NOT_ENCODE_URL)
379                     .formatWith(schemaFile)
380                     .withResolution(PLEASE_CHECK_THAT_URL_IS_WELL_FORMED);
381         }
382         int timeout = 60 * 1000;
383         org.w3c.dom.Document schemaDocument = domUtil.getResourceAsDocument(schemaFileURL, null,
384                 timeout);
385         if (StringUtils.isBlank(xPathForSchemaElement)) {
386             if (shouldValidate) {
387                 schemaValidator = xmlUtils.getSchemaValidator(schemaDocument);
388                 if (schemaValidator == null) {
389                     throw new ConfigurationException(NO_SCHEMA_VALIDATION_FILE_DEFINED)
390                             .withReason(NO_SCHEMA_VALIDATION_FILE_DEFINED)
391                             .withResolution(NO_SCHEMA_VALIDATION_FILE_DEFINED_RESOLUTION);
392                 }
393             }
394             // Assuming that the given file adheres to XSD format
395             return typesFactory.generateTypes(schemaDocument.getFirstChild());
396         }
397         XPath xPath = createXPathWithNSContext(namespaceContextTable);
398         try {
399             NodeList nodeList = (NodeList) xPath.evaluate(xPathForSchemaElement, schemaDocument,
400                     XPathConstants.NODESET);
401             if (shouldValidate) {
402                 schemaValidator = xmlUtils.getSchemaValidator(nodeList);
403                 if (schemaValidator == null) {
404                     throw new ConfigurationException(NO_SCHEMA_VALIDATION_FILE_DEFINED)
405                             .withReason(NO_SCHEMA_VALIDATION_FILE_DEFINED)
406                             .withResolution(NO_SCHEMA_VALIDATION_FILE_DEFINED_RESOLUTION);
407                 }
408             }
409             return typesFactory.generateTypes(nodeList);
410         } catch (XPathExpressionException e) {
411             throw new ConfigurationException(e, COULD_NOT_EVALUATE_XPATH)
412                     .withResolution(XPATH_EVAL_RESOLUTION);
413         }
414     }

416     private XPath createXPathWithNSContext(List<Map<String, String>> namespaceContextTable) {
417         XPath xPathForSchemaElement = XPATH_FACTORY.newXPath();
418         if (CollectionUtils.isNotEmpty(namespaceContextTable)) {
419             final Map<String, String> prefixToNamespaceMap = new HashMap<>();
420             final Map<String, List<String>> namespaceToPrefixMap = new HashMap<>();
421             for (Map<String, String> namespaceCtx : namespaceContextTable) {
422                 String namespace = JsonPath.read(namespaceCtx, NAMESPACE_PATH);
423                 String prefix = JsonPath.read(namespaceCtx, PREFIX_PATH);
424                 prefixToNamespaceMap.put(prefix, namespace);
425                 List<String> listPrefix = namespaceToPrefixMap.get(namespace);
426                 if (listPrefix == null) {
427                     listPrefix = new ArrayList<>();
428                     namespaceToPrefixMap.put(namespace, listPrefix);
429                 }
430                 listPrefix.add(prefix);
431             }
432             xPathForSchemaElement.setNamespaceContext(new NamespaceContext() {
433                 @Override
434                 public String getNamespaceURI(String prefix) {
435                     return prefixToNamespaceMap.get(prefix);
436                 }
437 
438                 @Override
439                 public String getPrefix(String namespaceURI) {
440                     List<String> listPrefix = namespaceToPrefixMap.get(namespaceURI);
441                     return ((listPrefix != null) ? listPrefix.get(0) : null);
442                 }
443 
444                 @Override
445                 public Iterator getPrefixes(String namespaceURI) {
446                     return namespaceToPrefixMap.get(namespaceURI).iterator();
447                 }
448             });
449         }
450         return xPathForSchemaElement;
451     }

471      * Attempt to minify the given XML String. Fall back to original if error happens
472      * @param xml   The XML String
473      * @return  The minified xml string or the original xml if exception happens
474      */
475     private String getMinifiedXml(String xml) {
476         try {
477             DocumentBuilder documentBuilder = documentBuilderFactory.newDocumentBuilder();
478             org.w3c.dom.Document w3cDoc = documentBuilder.parse(new ByteArrayInputStream(
479                     xml.getBytes(UTF_8)));
480             removeEmptyNodes(w3cDoc);
481             StreamResult result = new StreamResult(new StringWriter());
482             Transformer transformer = xmlUtils.getTransformFactory().newTransformer();
483             transformer.setOutputProperty(OutputKeys.ENCODING, UTF_8.name());
484             transformer.transform(new DOMSource(w3cDoc), result);
485             return result.getWriter().toString();
486         } catch (Exception e) {
487             LOGGER.warn("Failed to minify the XML output, due to error: {}", e);
488             return xml;
489         }
490     }

492     private boolean removeEmptyNodes(Node node) {
493         if (node.getNodeType() == Node.TEXT_NODE) {
494             String nodeValue = node.getNodeValue();
495             if (node == null || nodeValue.trim().isEmpty()) {
496                 // Remove unexpected spaces in the xml string
497                 // e.g <a><b>    </b></a> or <a>  <b>abc</b>  </a>
498                 node.getParentNode().removeChild(node);
499                 return true;
500             }
501         } else {
502             NodeList children = node.getChildNodes();
503             for (int i = 0; i < children.getLength();) {
504                 if (!removeEmptyNodes(children.item(i))) {
505                     i++;
506                 }
507             }
