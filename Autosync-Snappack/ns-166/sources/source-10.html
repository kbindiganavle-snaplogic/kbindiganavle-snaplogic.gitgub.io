69 /**
70  * Compares 2 sorted streams (both sorted the same way) of documents (Original and New) and returns
71  * 4 output views:
72  * <ul>
73  * <li>Deletions - Contains documents that exist in the Original view but not the New view.</li>
74  * <li>Insertions - Contains documents that exist in the New view but not the Original view.</li>
75  * <li>Modified - Contains documents that exist in both views, but are different in some property.
76  * </li>
77  * <li>Unmodified - Contains documents that are the same in both input views.</li>
78  * </ul>
79  *
80  * <p>
81  * The snap uses the following properties:
82  * <ul>
83  * <li>Sort Paths - List of paths to sort on. Paths should be obtained from the upstream schema.
84  * </li>
85  * <li>Sort Order - Ordering of the input streams. Allowed values are ascending or descending.
86  * Default value is ascending.</li>
87  * <li>Output view mapping - Mapping of output view names to types (Deletions, Insertions, Modified,
88  * Unmodified)</li>
89  * </ul>
90  *
91  * @author itanasiychuk, yrudyy
92  */

235         }

243         if (originalInputView == null || newInputView == null) {
244             throw new ConfigurationException(ERR_DIFF_VIEWS_NOT_CONNECTED)

247         }
248

250                 inputViews.getDocumentsFrom(originalInputView), ORIGINAL_VIEW_NAME);
251         PeekingDocumentIterator newDocumentIterator = new PeekingDocumentIterator(
252                 inputViews.getDocumentsFrom(newInputView), NEW_VIEW_NAME);
253 
254         diffDocumentIterator = new DiffDocumentIterator(dataComparator, originalDocumentIterator,
255                 newDocumentIterator);
256 
257         // This object is used for sorting validation and processing documents
258         documentHolder = new LinkedList<Map<String, DiffDocument>>() {
259             private static final long serialVersionUID = 1L;

265

269             }
270         };
271 
272         Map<String, String> outputLabelToKeyMapping = Maps.newHashMap();
273         for (String outputViewName : outputViews.names()) {
274             String outputViewLabelPath = new JsonPathBuilder(outputViewName).appendCurrentElement(
275                     LABEL)
276                     .build();
277             String outputViewLabel = propertyValues.get(PropertyCategory.OUTPUT,
278                     outputViewLabelPath);
279             outputLabelToKeyMapping.put(outputViewLabel, outputViewName);
280         }

282         List<Map<Object, Object>> list = propertyValues.get(OUTPUT_VIEW_MAP_PROP);
283 
284         if (CollectionUtils.isEmpty(list)) {
285             throw new ConfigurationException(ERR_DIFF_MAPPING_PROP_EMPTY).withReason(
286                     ERR_DIFF_MAPPING_PROP_EMPTY)
287                     .withResolution(DIFF_PLEASE_ADD_MAPPING);
288         }
289

291             throw new ConfigurationException(ERR_DIFF_ICORRECT_MAPPINGS_NUM).withReason(

304             }

307                 throw new ConfigurationException(ERR_DIFF_DUPLICATE_MAPPING_VIEWS).withReason(
308                         ERR_DIFF_DUPLICATE_MAPPING_VIEWS)

316     @Override
317     public void execute() throws ExecutionException {
318         while (diffDocumentIterator.hasNext()) {
319             Map<String, DiffDocument> current = diffDocumentIterator.peek();
320             documentHolder.add(current);
321 
322             if (documentHolder.size() == 2) {
323                 Map<String, DiffDocument> previous = documentHolder.getFirst();
324                 validateSorting(previous, current);
325                 writeData(previous);
326             }
327         }
328 
329         if (documentHolder.size() == 1) {
330             writeData(documentHolder.getFirst());
331         }
332     }

339     /**
340      * Validate sorting using two subsequent documents from the same view
341      *
342      * @param previous previous document
343      * @param current current document
344      */
345     private void validateSorting(Map<String, DiffDocument> previous,
346             Map<String, DiffDocument> current) {
347         if (previous == null && current == null) {
348             return;
349         }
350         Set<String> keySet = previous != null ? previous.keySet() : current.keySet();
351         for (String key : keySet) {
352             DiffDocument previousDocument = previous.get(key);
353             DiffDocument currentDocument = current.get(key);
354 
355             if (previousDocument == null || currentDocument == null ||
356                     previousDocument.getDocument() == null ||
357                     currentDocument.getDocument() == null) {
358                 // Only one document this time, let's try another view's pair
359                 continue;
360             }
361 
362             int result = dataComparator.compare(previousDocument.getDocument().get(),
363                     currentDocument.getDocument().get());
364             if (result > 0) {
365                 String errMessage;
366                 SnapDataException sde;
367                 switch (sortType) {
368                     case ASCENDING:
369                         errMessage = String.format(ERR_DIFF_STREAM_NOT_IN_ASCENDING_ORDER, key);
370                         sde = new SnapDataException(currentDocument.getDocument(), errMessage);
371                         sde.withReason(errMessage)
372                                 .withResolution(DIFF_STREAM_NOT_IN_RIGHT_ORDER);
373                         writeToErrorView(sde);
374                         break;
375                     case DESCENDING:
376                         errMessage = String.format(ERR_DIFF_STREAM_NOT_IN_DESCENDING_ORDER, key);
377                         sde = new SnapDataException(currentDocument.getDocument(), errMessage);
378                         sde.withReason(errMessage)
379                                 .withResolution(DIFF_STREAM_NOT_IN_RIGHT_ORDER);
380                         writeToErrorView(sde);
381                         break;
382                     case UNSORTED:
383                         // Do nothing, we don't need it for now
384                 }
385             }
386         }
387     }
