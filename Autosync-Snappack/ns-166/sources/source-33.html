329     @Override
330     protected void doWork() {
331         // do nothing if input view is not connected
332         if (inputViews.isEmpty()) {
333             return;
334         }
335         for (BinaryInput binaryInput : inputViews.binaryInputs(inputViews.get())) {
336             Document header = binaryInput.getHeader();
337             evaluateSchemaFileExpression(header);
338             documentCounter.inc();
339             try (BufferedInputStream resettableInputStream = new BufferedInputStream(
340                     new BOMInputStream(Channels.newInputStream(binaryInput.getChannel())));
341                  PushbackInputStream pushbackInputStream = new PushbackInputStream(
342                          resettableInputStream)) {
343                 if (shouldValidate) {
344                     if (schemaValidator != null) {
345                         // Stream has to be opened again as the schema validator will close the
346                         // stream when the xml data is validated.
347                         try {
348                             resettableInputStream.mark(Integer.MAX_VALUE);
349                             InputStream uncloseableInputStream = new CloseShieldInputStream
350                                     (resettableInputStream);
351                             xmlUtils.validate(schemaValidator, uncloseableInputStream);
352                             resettableInputStream.reset();
353                         } catch (SAXException e) {
354                             LOGGER.warn(IGNORING_XML_DATA_AS_IT_DOES_NOT_CONFORM_TO_THE_SCHEMA);
355                             // Writes the validation error to the error view.
356                             errorViews.write(Utils.createDataExceptionForInvalidXML(e), header);
357                             continue;
358                         }
359                     } else {
360                         throw new ConfigurationException(NO_SCHEMA_VALIDATION_FILE_DEFINED)
361                                 .withReason(NO_SCHEMA_VALIDATION_FILE_DEFINED)
362                                 .withResolution(NO_SCHEMA_VALIDATION_FILE_DEFINED_RESOLUTION);
363                     }
364                 }
365                 if (!StringUtils.isBlank(splitterExpression)) {
366                     Source xmlSource = new StreamSource(resettableInputStream);
367                     xmlSplitter = new XmlSplitter.DefaultXmlSplitter(xmlSource,
368                             splitterExpression, namespaceContext);
369                     try {
370                         boolean outputWritten = false;
371                         Result result = xmlSplitter.next();
372                         while (result != null) {
373                             if (result instanceof StreamResult) {
374                                 StreamResult streamResult = (StreamResult) result;
375                                 try {
376                                     Object data = xmlUtils.convertToJson(xmlInputFactory,
377                                             jsonOutputFactory, new StreamSource(new StringReader(
378                                                     streamResult.getWriter().toString())));
379                                     if (data != null) {
380                                         if (matchDataTypes || shouldValidate) {
381                                             matchDataTypesWithOutputBasedOnXSD(null, data, DOLLAR,
382                                                     null,
383                                                     header);
384                                         }
385                                         writeToOutputViews(header, data);
386                                         outputWritten = true;
387                                     }
388                                 } catch (XMLStreamException e) {
389                                     SnapDataException snapDataException =
390                                             new SnapDataException(e, COULD_NOT_CONVERT_XML)
391                                                     .withReason(e.getMessage())
392                                                     .withResolution(PLEASE_VERIFY_THE_XML);
393                                     errorViews.write(snapDataException, header);
394                                 }
395                             } else {
396                                 LOGGER.warn(UNEXPECTED_TRANSFORMATION_RESULT, result.getClass());
397                             }
398                             result = xmlSplitter.next();
399                         }
400                         if (!outputWritten) {
401                             header.acknowledge();
402                         }
403                     } catch (XMLStreamException e) {
404                         SnapDataException ex = new SnapDataException(e, ERROR_WHILE_PARSING_XML)
405                                 .withReason(e.getMessage())
406                                 .withResolution(PLEASE_VERIFY_THE_XML);
407                         errorViews.write(ex);
408                     } finally {
409                         try {
410                             xmlSplitter.close();
411                         } catch (XMLStreamException e) {
412                             LOGGER.warn(ERROR_WHILE_CLOSING_SPLITTER, e);
413                         }
414                     }
415                 } else {
416                     if (Utils.isNotEmpty(pushbackInputStream)) {
417                         try {
418                             Object data = xmlUtils.convertToJson(xmlInputFactory,
419                                     jsonOutputFactory, new StreamSource(pushbackInputStream));
420                             if (data != null) {
421                                 if (matchDataTypes || shouldValidate) {
422                                     matchDataTypesWithOutputBasedOnXSD(null, data, DOLLAR, null,
423                                             header);
424                                 }
425                                 writeToOutputViews(header, data);
426                             }
427                         } catch (XMLStreamException | RuntimeException e) {
428                             SnapDataException snapDataException = new SnapDataException(e,
429                                     FAILED_TO_CONVERT_XML_TO_JSON)
430                                     .withResolution(PLEASE_CHECK_IF_THE_XML_IS_WELL_FORMED);
431                             errorViews.write(snapDataException, header);
432                         }
433                     } else {
434                         // Encountered empty document.
435                         LOGGER.debug(SKIPPING_EMPTY_DOCUMENT);
436                         if (header != null) {
437                             header.acknowledge();
438                         }
439                     }
440                 }
441             } catch (JSONException jsonException) {
442                 SnapDataException snapDataException = new SnapDataException(jsonException,
443                         UNABLE_TO_CONVERT_TO_JSON)
444                         .withResolution(CHECK_IF_XML_IS_WELL_FORMED);
445                 errorViews.write(snapDataException, header);
446             } catch (IOException e) {
447                 SnapDataException snapDataException = new SnapDataException(e,
448                         EXCEPTION_WHILE_READING_THE_INCOMING_BINARY_DATA)
449                         .withReason(ERROR_WHILE_READING_INPUT_DATA);
450                 errorViews.write(snapDataException, header);
451             }
452         }
453     }

509     @SuppressWarnings("unchecked")
510     private void setSchema(Map<String, Object> element, String jsonPath) {
511         String key = null;
512         String type = null;
513         Object value = null;
514         for (Map.Entry<String, Object> entry : element.entrySet()) {
515             switch (entry.getKey()) {
516                 case XS_ELEMENT:
517                 case XSD_ELEMENT:
518                     value = entry.getValue();
519                     checkTypeSetSchema(jsonPath, value);
520                     return;
521                 case XS_ATTRIBUTE:
522                 case XSD_ATTRIBUTE:
523                     Object attribute = entry.getValue();
524                     checkTypeSetSchema(jsonPath, attribute);
525                     break;
526                 case AT_NAME:
527                     key = (String) entry.getValue();
528                     jsonPath = String.format(FORMAT_DOT, jsonPath, key);
529                     break;
530                 case MAX_OCCURS:
531                     if (key != null) {
532                         String val = (String) entry.getValue();
533                         if (!StringUtils.equalsIgnoreCase(val, MAX_OCCURS_DEFAULT_VALUE)) {
534                             arrayFoundBasedOnSchema.add(jsonPath);
535                         }
536                     }
537                     break;
538                 case AT_TYPE:
539                 case AT_BASE:
540                     type = (String) entry.getValue();
541                     break;
542                 default:
543                     Object object = entry.getValue();
544                     if (object instanceof Map || object instanceof List) {
545                         value = object;
546                     }
547             }

653     private String getDataType(String jsonPath) {
654         if (StringUtils.isNotBlank(splitterExpression) && firstMatch &&
655                 StringUtils.isNotBlank(jsonPath)) {
656             // data-type schema map needs to be updated to support splitter
657             String path = jsonPath.substring(1); // jsonPath without '$' prefix
658             String prefix = null;
659             int index = 0;
660             for (Map.Entry<String, String> entry : schema.entrySet()) {
661                 if (entry.getKey().endsWith(path)) {
662                     firstMatch = false;
663                     prefix = entry.getKey();
664                     index = prefix.indexOf(path);
665                     prefix = prefix.substring(0, index);
666                     break;
667                 }
668             }
669             if (!firstMatch && prefix != null) {
670                 // get new data-type schema map
671                 Map<String, String> newSchema = new HashMap<>(schema.size());
672                 for (Map.Entry<String, String> entry : schema.entrySet()) {
