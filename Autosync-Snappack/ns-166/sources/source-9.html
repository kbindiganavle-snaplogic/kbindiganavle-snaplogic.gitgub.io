276     @Override
277     protected void doWork() {
278         // do nothing if input view is not connected
279         if (inputViews.isEmpty()) {
280             return;
281         }
282         List<String> headers = null;
283         try {
284             InputView inputView = null;
285             if (inputViews.size() == 2) {
286                 // if there is a second input view open, get CSV headers and data types
287                 for (InputView view : inputViews) {
288                     if (Utils.INPUT0.equals(view.getName())) {
289                         inputView = view;
290                     } else {
291                         headers = getHeadersFrom(view);
292                         if (containsHeaders && skipLinesValue == 0) {
293                             skipLinesValue++;
294                         }
295                     }
296                 }
297             } else {
298                 inputView = inputViews.get();
299             }
300             for (BinaryInput binaryInput : inputViews.binaryInputs(inputView)) {
301                 if (binaryInput == null) {
302                     continue;
303                 }
304                 contentLocation = getContentLocation(binaryInput.getHeader());
305                 linesRead = 0;
306                 org.apache.commons.csv.CSVParser parser = null;
307                 InputStream inputStream = null;
308                 try {
309                     inputStream = getInputStream(binaryInput);
310                     parser = createParser(inputStream);
311                     Iterator<CSVRecord> dataIterator = parser.iterator();
312                     // process Skip lines property
313                     for (int i = skipLinesValue; i > 0 && dataIterator.hasNext(); i--) {
314                         dataIterator.next();
315                     }
316                     // CSV header may have to be initialized for each input binary stream
317                     if (inputViews.size() == 1) {
318                         headers = createHeader(binaryInput, dataIterator);
319                     }
320                     if (CollectionUtils.isEmpty(headers)) {
321                         continue;
322                     }
323                     while (true) {
324                         try {
325                             if (dataIterator.hasNext()) {
326                                 final CSVRecord record = dataIterator.next();
327                                 linesRead = parser.getRecordNumber();
328                                 processRecord(binaryInput, record, headers);
329                             } else {
330                                 break;
331                             }
332                         } catch (SuggestViewAbortException | ExecutionException e) {
333                             // return it directly this is expected
334                             throw e;
335                         } catch (Exception e) {
336                             // If the pipeline execution is stopped while parsing, hasNext() above
337                             // throws an CloseByInterruptException or ClosedChannelException
338                             // wrapped around RuntimeException. In this case, the CSV parser
339                             // should break out of the while loop.
340                             // Otherwise, it will run in an infinite loop.
341                             Throwable cause = e.getCause();
342                             if (cause instanceof ClosedChannelException) {
343                                 return;
344                             }
345                             long failedLineNumber = parser.getCurrentLineNumber();
346                             SnapDataException ex = new SnapDataException(e, ERROR_PARSING_AT_RECORD)
347                                     .formatWith(linesRead, contentLocation)
348                                     .withReason(e.getMessage())
349                                     .withResolution(String.format(ERROR_PARSING_RESOLUTION,
350                                             failedLineNumber));
351                             errorViews.write(ex, createErrorDocument(binaryInput, headers, null));
352 
353                             // This logic will relieve ourselves of the need to have the custom
354                             // version of the commons-csv code, as it will parse through the end
355                             // of the current "line" in order to clear out the invalid tokens until
356                             // the next csv record is available to us.
357                             if (cause instanceof IOException) {
358                                 do {
359                                     try {
360                                         // This will clear out the "current" value within the
361                                         // CSVIterator so that we can queue up the next Record.
362                                         dataIterator.next();
363                                         // This will queue up the next "record" without nullifying
364                                         // it, so that we can re-"read" this value with `hasNext`
365                                         // in the while(true) loop encapsulating the parent try.
366                                         dataIterator.hasNext();
367                                     } catch (UncheckedIOException | IllegalStateException ignored) {
368                                         // We ignore IllegalStateException and UncheckedIOException
369                                         // as they indicate further potential invalid tokens within
370                                         // the CSV row. We want to get through all of those invalid
371                                         // tokens until we reach the end of the line, then we can
372                                         // move onto the next row (or record) to continue parsing
373                                         // the CSV.
374                                     } catch (NoSuchElementException endOfFileException) {
375                                         // If it's the end of the file, just get out of the loop
376                                         // NoSuchElementException occurs only in a closed reader
377                                         // or when it's the end of the file. In this case, we know
378                                         // that we've already had SOME invalid token for the current
379                                         // record, and now we know we've either closed the file or
380                                         // reached the end of the file already. This break will
381                                         // force us into the try -> else statement.
382                                         break;
383                                     }
384                                 } while (parser.getCurrentLineNumber() == failedLineNumber);
385                             }
386                         }
387                     }
388                 } finally {
389                     consumeClose(inputStream, LOG);
390                     if (parser != null) {
391                         parser.close();
392                     }
393                     if (!ignoreEmptyData && linesRead == 0) {
394                         outputViews.write(documentUtility.newDocumentFor(binaryInput.getHeader(),
395                                 new LinkedHashMap<>()));
396                     }
397                 }
398             }
399         } catch (SnapDataException e) {
400             errorViews.write(e);
401         } catch (SuggestViewAbortException | ExecutionException e) {
402             throw e;
403         } catch (Exception e) {
404             throw new ExecutionException(e, ERR_PARSE)
405                     .withReason(String.format(REASON_CSV_PARSE, e.getMessage()))
406                     .withResolution(RESOLUTION_CSV_PARSE);
407         }
408     }

415     private List<String> getHeadersFrom(InputView inputView) {
416         Iterator<BinaryInput> binaryDataIterator = inputViews.getBinaryInputsFrom(inputView);
417         if (binaryDataIterator.hasNext()) {
418             org.apache.commons.csv.CSVParser parser = null;
419             BinaryInput binaryInput = binaryDataIterator.next();
420             try (InputStream inputStream = getInputStream(binaryInput)) {
421                 parser = createParser(inputStream);
422                 Iterator<CSVRecord> dataIterator = parser.iterator();
423                 // get CSV headers
424                 CSVRecord record = dataIterator.hasNext() ? dataIterator.next() : null;
425                 if (record == null) {
426                     throw new SnapDataException(ERR_HEADERS_INPUTVIEW)
427                             .withReason(REASON_NO_METADATA)
428                             .withResolution(RESOLUTION_ADDRESS_ISSUE);
429                 }
430                 List<String> headers = createHeadersFromInputStream(record);
431                 // get data types
432                 CSVRecord csvRecord = dataIterator.hasNext() ? dataIterator.next() : null;
433                 if (csvRecord != null) {
434                     Iterator<String> iterator = csvRecord.iterator();
435                     for (String header : headers) {
436                         String dataType = iterator.hasNext() ? iterator.next() : null;
437                         if (StringUtils.isNotBlank(dataType) && !STRING.equalsIgnoreCase
438                                 (dataType)) {
439                             columnDataTypes.put(header, dataType);
440                         }
441                     }
442                 }
443                 return headers;
444             } catch (IOException e) {
445                 throw new SnapDataException(e, ERR_HEADERS_INPUTVIEW)
446                         .withReason(e.getMessage())
447                         .withResolution(RESOLUTION_ADDRESS_ISSUE);
448             } finally {
449                 if (parser != null) {
450                     try {
451                         parser.close();
452                     } catch (IOException e) {
453                         LOG.debug("Cannot close apache CSVParser", e);
454                     }
455                 }
456             }
457         }
458         throw new SnapDataException(ERR_HEADERS_INPUTVIEW)
459                 .withReason(REASON_NO_METADATA)
460                 .withResolution(RESOLUTION_ADDRESS_ISSUE);
461     }

477     private void processRecord(final BinaryInput binaryInput, CSVRecord record,
478                                List<String> headers) {
479         int recordSize = record.size();
480         final Map<String, Object> outputDocumentData = new LinkedHashMap<>(headers.size());
481         int headerIndex = 0;
482         for (String header : headers) {
483             if (headerIndex < recordSize) {
484                 Object value = convertValue(header, record.get(headerIndex++));
485                 outputDocumentData.put(header, value);
486             } else {
487                 if (errorPolicy.equalsIgnoreCase(BOTH) || errorPolicy
488                         .equalsIgnoreCase(FAIL_TO_LARGE)) {
489                     SnapDataException ex = new SnapDataException(ERROR_PARSING_AT_RECORD)
490                             .formatWith(linesRead, contentLocation)
491                             .withReason(THE_CSV_RECORD_HAS_MORE_COLUMNS_THAN_DATA)
492                             .withResolution(String.format(ERROR_PARSING_RESOLUTION,
493                                     linesRead));
494                     errorViews.write(ex, createErrorDocument(binaryInput, headers, record));
495                     return;
496                 }
497             }
498         }
499         int keySize = headers.size();
500         if (keySize < recordSize) {
501             handleError(binaryInput, record, headers);
502             // If there is more data than defined fields, we'll write a record to the error
503             // view.  But, we'll also write a document to the output view since the data is
504             // still good enough.
505             if (errorPolicy.equalsIgnoreCase(FAIL_TO_LARGE)) {
506                 return;
507             }
508         }
509         if (charsetName == null) {
510             outputViews.write(documentUtility.newDocumentFor(binaryInput.getHeader(),
511                     outputDocumentData));
512         } else {
513             // set character set name to document metadata,
514             // which may be used by formatters in the pipeline
515             Document document = docUtil.newDocument(outputDocumentData);

567     private org.apache.commons.csv.CSVParser createParser(InputStream inputStream) throws
568             IOException {
569         // check for ByteOrderMark at the beginning of input stream
570         BufferedInputStream bufferedIn = new BufferedInputStream(inputStream);
571         InputStreamReader inputStreamReader;
572         switch (charsetProp) {
573             case AUTO_BOM_DETECT:
574                 bufferedIn.mark(0);
575                 BOMInputStream bomIn = new BOMInputStream(bufferedIn, ByteOrderMark.UTF_8,
576                         ByteOrderMark.UTF_16LE, ByteOrderMark.UTF_16BE,
577                         ByteOrderMark.UTF_32LE, ByteOrderMark.UTF_32BE);
578                 if (bomIn.hasBOM()) {
579                     ByteOrderMark bom = bomIn.getBOM();
580                     // get character set name
581                     charsetName = bom.getCharsetName();
582                     // remove ByteOrderMark
583                     inputStreamReader = new InputStreamReader(bomIn, charsetName);
584                 } else {
585                     // check if BOM itself is written in UTF-16LE
586                     inputStreamReader = new InputStreamReader(bufferedIn, UTF_16LE);
587                     char[] chars = new char[BOM_UTF_16LE_LENGTH];
588                     if (inputStreamReader.read(chars, 0, BOM_UTF_16LE_LENGTH) ==
589                             BOM_UTF_16LE_LENGTH) {
590                         String string = new String(chars);
591                         if (string.contains(UTF_16LE) && string.endsWith(SQUARE_BRACKET_RIGHT)) {
592                             // found a special UTF-16LE BOM where BOM itself is in UTF-16LE
593                         } else {
594                             // no BOM
595                             inputStreamReader = resetInputStream(bufferedIn);
596                         }
597                     } else {
598                         // no BOM
599                         inputStreamReader = resetInputStream(bufferedIn);
600                     }
601                 }
602                 break;
603             case ISO_LATIN_1:
604                 inputStreamReader = new InputStreamReader(bufferedIn, ISO_8859_1);
605                 break;
606             default:
607                 charsetName = charsetProp;
608                 inputStreamReader = new InputStreamReader(bufferedIn, charsetName);
609         }
610         return CSVFormat.Builder.create().setDelimiter(delimiter)
611                 .setQuote(quoteCharacter)
612                 .setEscape(StringUtils.isEmpty(escapeCharacterStr) ? null :
613                         escapeCharacterStr.charAt(0))
614                 .setIgnoreEmptyLines(true)
615                 .setIgnoreSurroundingSpaces(!preserveSurroundingSpaces).build()
616                 .parse(inputStreamReader);
617     }

633         if (containsHeaders) {
634             // input provides the header information
635             if (dataIterator.hasNext()) {
636                 final CSVRecord record = dataIterator.next();
637                 return createHeadersFromInputStream(record);
638             }

645                     // read the first record to determine length
646                     final CSVRecord record = dataIterator.next();

649                     // process the first record
650                     processRecord(binaryInput, record, headers);
651                     return headers;

654                 return Collections.emptyList();
655             } else {
656                 return columnNames;

661     private List<String> createHeadersFromInputStream(CSVRecord headerRecord) {
662         // headers are provided from the input stream and have been read as a record.
663         Iterator<String> header = headerRecord.iterator();
664         List<String> headers = Lists.newArrayListWithExpectedSize(headerRecord.size());
665         while (header.hasNext()) {
666             headers.add(header.next());
667         }
668         // write to error view if the header validation fails
669         if (validateHeaders && CollectionUtils.isNotEmpty(columnNames) &&
670                 !columnNames.equals(headers)) {
671             Map<String, Object> errorMap = Maps.newHashMapWithExpectedSize(3);
672             errorMap.put(ERROR, ERR_VALIDATE_HEADERS);
673             errorMap.put(HEADERS_FROM_INPUT, headers);
674             errorMap.put(COLUMN_NAMES_LABEL, columnNames);
675             SnapDataException ex = new SnapDataException(documentUtility.newDocument(errorMap),
676                     ERR_VALIDATE_HEADERS)
677                     .withReason(ERR_VALIDATE_HEADERS)
678                     .withResolution(RESOLUTION_ADDRESS_ISSUE);
679             errorViews.write(ex);
680         }
681         return headers;
682     }
