235     protected void doWork() {
236         try {
237             DataIterator dataIterator = null;
238             Map<String, String> columnData;
239             for (BinaryInput binaryInput : inputViews.binaryInputs(inputViews.get())) {
240                 if (binaryInput == null) {
241                     break;
242                 }
243                 try {
244                     InputStream inputStream = Channels.newInputStream(binaryInput.getChannel());
245                     dataIterator = createDataIterator(inputStream);
246                     while (dataIterator.hasNext()) {
247                         // Get the next object to be written, if all the field values are parsed
248                         // then write the document to output view otherwise write to error view.
249                         columnData = (Map<String, String>) dataIterator.next();
250                         if (columnData.size() != columnInfoList.size()) {
251                             // TODO - kamal: Reason resolution missing
252                             SnapDataException snapDataException = new SnapDataException(documentUtility
253                                     .newDocumentFor(binaryInput.getHeader(), columnData),
254                                     ERROR_COLUMN_NAMES_REASON);
255                             errorViews.write(snapDataException);
256                             continue;
257                         }
258                         outputViews.write(documentUtility.newDocumentFor(binaryInput.getHeader(),
259                                 columnData));
260                     }
261                 } finally {
262                     IOUtils.closeQuietly(dataIterator);
263                 }
264             }
265         } catch (IOException ex) {
266             throw new ExecutionException(ex, ERROR_READ)
267                     .withReason(String.format(ERROR_READ_REASON, ex.getMessage()))
268                     .withResolution(ERROR_CUSTOMER_SUPPORT);
269         }

336                     while (skippedLines++ < linesToSkip) {

351         private boolean isIgnoreLine(String line) {
352             if (ignoreConditions != null) {
353                 Document document = documentUtility.newDocument();
354                 for (Map<String, Object> ignoreCondition : ignoreConditions) {
355                     for (Entry<String, Object> entry : ignoreCondition.entrySet()) {
356                         ExpressionProperty exp = (ExpressionProperty) entry.getValue();
357                         String val = exp.eval(document);
358                         switch (entry.getKey()) {
359                             case FUNCTION_CONTAINS:
360                                 if (StringUtils.contains(line, val)) {
361                                     return true;
362                                 }
363                                 break;
364                             case FUNCTION_STARTS_WITH:
365                                 if (StringUtils.startsWith(line, val)) {
366                                     return true;
367                                 }
368                                 break;
369                             case FUNCTION_ENDS_WITH:
370                                 if (StringUtils.endsWith(line, val)) {
371                                     return true;
372                                 }
373                                 break;
374                             case FUNCTION_REGEX:
375                                 Pattern pattern;
376                                 try {
377                                     pattern = Pattern.compile(val);
378                                 } catch (PatternSyntaxException e) {
379                                     throw new ExecutionException(ERROR_INVALID_REGEX)
380                                             .withReason(ERROR_INVALID_REGEX_REASON)
381                                             .withResolution(ERROR_INVALID_REGEX_RESOLUTION);
382                                 }
383                                 if (pattern.matcher(line).matches()) {
384                                     return true;
385                                 }
386                                 break;
387                         }
388                     }
389                 }
390             }
391             return false;
392         }

394         private Map<String, String> extractColumns(String line) {
395             if (columnInfoList == null || columnInfoList.isEmpty()) {
396                 throw new ConfigurationException(ERR_COLUMN_INFORMATION_MISSING)
397                     .withReason(REASON_COLUMN_INFORMATION_MISSING)
398                     .withResolution(RESOLUTION_COLUMN_INFORMATION_MISSING);
399             }
400             Map<String, String> columnData = new LinkedHashMap<>(columnInfoList.size());
401             String colVal;
402             try {
403                 // extract column data from the input line
404                 for (ColumnInfo colInfo : columnInfoList) {
405                     if (line.length() <= colInfo.getEndPos()) {
406                         colVal = line.substring(colInfo.getStartPos() - 1, line.length());
407                     } else {
408                         colVal = line.substring(colInfo.getStartPos() - 1, colInfo.getEndPos());
409                     }
410                     if (colInfo.isTrimData()) {
411                         colVal = colVal.trim();
412                     }
413                     columnData.put(colInfo.getColumnName(), colVal);
414                     if (line.length() <= colInfo.getEndPos()) {
415                         break;
416                     }
417                 }
418             } catch (Exception ex) {
419                 log.error(ERROR_COLUMN_NAMES, ex);
420             }
421             return columnData;
422         }

464         @Override
465         public String readLine() throws IOException {
466             if (separatorChar == null) {
467                 return super.readLine();
468             } else {
469                 // Read characters from the input stream until the separator
470                 StringBuilder sb = new StringBuilder();
471                 int ch;
472                 while ((ch = super.read()) != separatorChar) {
473                     if (ch == EOF) {
474                         break;
475                     }
476                     sb.append((char) ch);
477                 }
478                 if (sb.length() > 0) {
479                     return sb.toString();
480                 } else {
481                     return null;
482                 }
483             }
484         }
