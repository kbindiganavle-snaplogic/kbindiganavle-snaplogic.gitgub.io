190     @Override
191     protected void process(Document document, String inputViewName) {
192         Map<String, Object> documentMap = super.getDocumentContentAsMap(document);
193         if (documentMap == null) {
194             // check that it is a map, otherwise will write it to the error view.
195             return;
196         }
197         Document currentDocument = document;
198         //Import requests are require aggregation of the documents if a batch value is specified
199         if (isImportRequest(objectName) && batchSize.intValue() > DEFAULT_BATCH_SIZE) {
200             currentQueueSize++;
201             String batchPath = batchNodePathProp.eval(document);
202             //Batchpath is the Leaf Node where all the batching of data into a single SOAPMessage
203             if (!StringUtils.isBlank(batchPath) && batchPath.startsWith(ROOT_PATH)) {
204                 Collection sourceLeaf;
205                 //save the first document as the default SOAP message
206                 if (batchMapData == null) {
207                     batchMapData = documentMap;
208                     Object data = traverseMapLeaf(batchMapData, batchPath);
209                     //The node should always be a collection
210                     if (data instanceof Collection) {
211                         desLeafNode = (Collection) data;
212                     } else {
213                         throw new SnapException(INVALID_BATCH_NODE_PATH)
214                                 .withReason(EXPECTED_COLLECTION_BATCH_NODE_PATH)
215                                 .withResolution(GENERIC_RESOLUTION);
216                     }
217 
218                 } else {
219                     //Aggregate all all the batch size documents to saved SOAPMessage
220                     Object data = traverseMapLeaf(documentMap, batchPath);
221                     if (data instanceof Collection) {
222                         sourceLeaf = (Collection) data;
223                         desLeafNode.addAll(sourceLeaf);
224                     } else {
225                         throw new SnapException(INVALID_BATCH_NODE_PATH)
226                                 .withReason(EXPECTED_COLLECTION_BATCH_NODE_PATH)
227                                 .withResolution(GENERIC_RESOLUTION);
228                     }
229                 }
230             } else {
231                 throw new SnapException(INVALID_BATCH_NODE_PATH)
232                         .withReason(PROVIDE_VALID_BATCH_PATH_NODE)
233                         .withResolution(GENERIC_RESOLUTION);
234             }
235 
236             //Generate a new document if the total aggregated data matches the batch size else keep
237             // consuming.
238             if (currentQueueSize < batchSize.intValue()) {
239                 return;
240             }
241             //Generate a new document from the aggregated data reset the counter
242             currentDocument = documentUtility.newDocument(batchMapData);
243 
244         }
245         processWorkdayRequest(currentDocument, isSync);
246

261      * Generates a SOAP Envelope from the given document and get's the status of the submitted
262      * job if its a synchronized request.
263      *
264      * @param document
265      * @param isSynchronized
266      */
267     private void processWorkdayRequest(final Document document, boolean isSynchronized) {
268         String originalEnvelope = null;
269         final String strippedEnvelope;
270         try {
271             LOGGER.debug("Before SOAP template evaluation");
272             originalEnvelope = editorProperty.eval(document);
273             LOGGER.debug("Before stripping empty SOAP envelope nodes");
274             strippedEnvelope = xmlUtils.stripNodeAndAttributeNodes(originalEnvelope,
275                     documentBuilderFactory);
276         } catch (SuggestViewAbortException e) {
277             throw e;
278         } catch (SnapDataException e) {
279             errorViews.write(e, document);
280             return;
281         } catch (ExecutionException e) {
282             SnapDataException ex = new SnapDataException(e, e.getMessage())
283                     .withReason(e.getReason())
284                     .withResolution(e.getResolution());
285             errorViews.write(ex, document);
286             return;
287         } catch (Exception e) {
288             SnapDataException ex = new SnapDataException(e, ERR_PREPARE_ENVELOPE_DATA)
289                     .withReason(e.getMessage())
290                     .withResolution(RESOLUTION_ADDRESS_ISSUE);
291             errorViews.write(ex, document);
292             return;
293         }
294 
295         maxRetries = evalIntProperty(maxRetryProperty, document, MIN_RETRIES, MIN_RETRIES,
296             errorViews);
297         retryInterval = evalIntProperty(retryIntervalProperty, document, MIN_RETRY_INTERVAL,
298             MIN_RETRY_INTERVAL, errorViews);
299 
300         try {
301             LOGGER.debug("Before invoking SOAP service");
302 
303             SOAPMessage response = Failsafe.with(getRetryPolicy(maxRetries, retryInterval))
304                 .onFailedAttempt(e ->
305                     LOGGER.warn("processWorkdayRequest_retry - failed {}", e.getMessage()))
306                 .onRetry(retryLogListener("processWorkdayRequest_retry"))
307                 .get(context -> {
308                     LOGGER.debug("SOAP Request: {}", strippedEnvelope);
309                     return invocationService.call(clientBuilder, strippedEnvelope);
310                 });
311 
312             if (response != null) {
313                 LOGGER.debug("Before serializing SOAP response");
314                 Object writeResponseData = invocationService.serialize(response);
315                 Object data = writeResponseData;
316                 // Synchronous mode - get the status of job using the WID form the previous request
317                 if (isImportRequest(objectName) && isSynchronized && writeResponseData instanceof
318                         Map) {
319                     response = makeAPIRequest(parseImportStatusResponse((Map<String,
320                                     Object>) writeResponseData), IMPORT_STATUS_RESOURCE_TEMPLATE,
321                             IMPORT_INTEGRATIONS_SERVICE_NAME, IMPORT_STATUS_OBJECT_NAME);
322                     data = invocationService.serialize(response);
323                     // check and wait for Response to complete status
324                     long startTime = 0;
325                     String responseStatusInfo = PROCESS_STRING;
326                     while (responseStatusInfo.toLowerCase().contains(PROCESS_STRING) && startTime <=
327                             TimeUnit.MINUTES.toMillis(MAX_WAIT_TIME)) {
328                         Thread.sleep(TimeUnit.SECONDS.toMillis(MIN_SLEEP_TIME));
329                         startTime += TimeUnit.SECONDS.toMillis(MIN_SLEEP_TIME);
330                         responseStatusInfo = fetchAndUpdateRequestStatus((Map<String, Object>)
331                                 writeResponseData, (Map<String, Object>) data);
332                     }
333 
334                 }
335                 if (data != null) {
336                     outputViews.write(documentUtility.newDocumentFor(document, data), document);
337                     return;
338                 }
339             }
340             passThroughInputOnly(document);
341 
342         } catch (FailsafeException fse) {
343             // Failsafe will only wrap checked exceptions (which includes interrupted)
344             Throwable t = fse.getCause();
345             if (t instanceof SOAPException || t instanceof XMLStreamException) {
346                 handleWorkdayWebServiceException((Exception) t, originalEnvelope,
347                     strippedEnvelope, errorViews, documentUtility, document, xmlUtils);
348                 return;
349             } else if (t instanceof InterruptedException) {
350                 LOGGER.error("InterruptedException: {}", t.getMessage());
351             } else {
352                 throw new ExecutionException(fse, RETRY_FAULT)
353                     .withResolution(ADDRESS_RETRY_FAULT_MESSAGE_AND_RETRY);
354             }
355 
356         } catch (XMLStreamException | WebServiceException | SOAPException | SoapFault e) {
357             handleWorkdayWebServiceException(e,
358                     originalEnvelope, strippedEnvelope, errorViews, documentUtility, document,
359                     xmlUtils);
360         } catch (InterruptedException e) {
361             LOGGER.error("Error during sleep {}", e.getMessage());
362         } finally {
363             LOGGER.debug("End Workday request and response process");
364             resetBatchProcess();
365         }
366     }

369     Map<String, Object> data) throws SOAPException, XMLStreamException, InterruptedException {
370         String requestStatus = "";
371         SOAPMessage importProcessesResponse = makeAPIRequest(parseImportProcessesResponse
372                         (writeResponseData), IMPORT_PROCESSES_RESOURCE_TEMPLATE,
373                 IMPORT_INTEGRATIONS_SERVICE_NAME, IMPORT_PROCESSES_OBJECT_NAME);
374         Map<String, Object> responsMap = (Map<String, Object>) invocationService
375                 .serialize(importProcessesResponse);
376 
377         List importProcessStatusObject = (List) traverseMapLeaf(responsMap,
378                 TYPE_STATUS_REFERENCE_JSONPATH);
379         for (Object status : importProcessStatusObject) {
380             if (status instanceof Map) {
381                 String value = (String) ((Map) status).get(WD_TYPE);
382                 if (BACKGROUND_PROCESS_INSTANCE_STATUS_ID.equalsIgnoreCase(value)) {
383                     requestStatus = (String) ((Map) status).get(LEAF_NODE_VALUE_KEY);
384                     data.put(REQUEST_STATUS_KEY, requestStatus);
385                 }
386             }
387         }
388         return requestStatus;
