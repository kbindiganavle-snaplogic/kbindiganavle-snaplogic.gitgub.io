423                         if (pages != null && pageNumber == 0) {
424                             int numPages = Integer.parseInt(pages);

427                                 response = invokeWorkday(document, originalEnvelope, i + 1,
428                                         defaultAsOfEffectiveDate);

436                                 }
437                             }

439                     }
440                 } else {

443             }
444         } catch (TransformerException | ParserConfigurationException | SAXException e) {

461                     .withEnvelope(originalEnvelope, xmlUtils)
462                     .withClientBuilder(clientBuilder)
463                     .write(documentUtility, errorViews);
464         }
465     }

467     private String createEnvelope(Document document) {
468         // Create empty document if no input view so that in .eval Velocity Context still

470         if (document == null) {

473         }
474         // check that the document is a map
475         Map<String, Object> documentMap = super.getDocumentContentAsMap(document);
476         if (documentMap == null) {
477             return null;
478         }
479 
480         for (Map<String, Map<String, String>> mapOutputs : outputs) {

484             String outputValue = mapOutputValue.get(Constants.SNAP_PROPERTY_VALUE);

519         }
520         for (int i = 0; i < list.getLength(); ) {
521             if (!xmlUtils.removeEmptyElementsRecursively(list.item(i))) {
522                 i++;
523             }
524         }
525         StreamResult result = new StreamResult(new StringWriter());
526         Transformer transformer = transformerFactory.newTransformer();
527         transformer.transform(new DOMSource(w3cDoc), result);
528 
529         strippedEnvelope = result.getWriter().toString();
530         InputStream is = new ByteArrayInputStream(strippedEnvelope.getBytes());
531         requestMessage = messageFactory.createMessage(null, is);
532         IOUtils.closeQuietly(is);
533 
534         SOAPFactory factory = SOAPFactory.newInstance();
535         SOAPPart soapMessage = requestMessage.getSOAPPart();
536         if (soapMessage.getEnvelope() == null) {
537             new SoapException(MISSING_SOAP_ENVELOPE, PLEASE_CONTACT_SUPPORT)
538                     .write(documentUtility, errorViews);
539             return null;
540         }
541         SOAPEnvelope soapEnvelope = soapMessage.getEnvelope();
542         if (soapEnvelope.getHeader() == null) {
543             soapEnvelope.addHeader();
544         }
545         SOAPBody soapBody = soapEnvelope.getBody();
546         SOAPBodyElement operationRequestElement = getSoapBodyOperationRequestElement(soapBody);
547         if (hasPaging) {
548             addResponsePagingToSoapBody(factory, operationRequestElement, pageNum,
549                     defaultAsOfEffectiveDate);
550         }
551         LOGGER.debug("before executeRead");
552         LOGGER.debug(CALLING_SOAP_ENDPOINT, dispatch.getEndpointReference().toString());
553         return executeRead(requestMessage, originalEnvelope, strippedEnvelope, document);
554     }

557      * Executes the read request
558      *
559      * @param requestMessage   the soap message being executed
560      * @param originalEnvelope the original envelope
561      * @param strippedEnvelope the stripped envelope
562      * @param document         the input document
563      * @return the soap response message
564      */

566     protected SOAPMessage executeRead(SOAPMessage requestMessage, String originalEnvelope,
567             String strippedEnvelope, Document document) {
568         try {
569             return Failsafe.with(getRetryPolicy(maxRetries, retryInterval))
570                 .onFailedAttempt(e ->
571                     LOGGER.warn("executeRead_retry - failed {}", e.getMessage()))
572                 .get(context -> {
573                     logRetryAttempt(context, LOGGER);
574                     logRequest(requestMessage);
575                     return dispatch.invoke(requestMessage);
576                 });
577         } catch (FailsafeException fse) {
578             Throwable e = fse.getCause();
579             if (e instanceof SoapFault) {
580                 throw new ExecutionException(e, SOAP_FAULT)
581                     .withResolution(ADDRESS_SOAP_FAULT_MESSAGE_AND_RETRY);
582             } else if (e instanceof WebServiceException) {
583                 handleWorkdayWebServiceException(
584                     (Exception) e, originalEnvelope, strippedEnvelope, errorViews, documentUtility,
585                     document, xmlUtils);
586                 return null;
587             } else {
588                 throw new ExecutionException(e, RETRY_FAULT)
589                     .withResolution(ADDRESS_RETRY_FAULT_MESSAGE_AND_RETRY);
590             }
591         } catch (SoapFault sfe) {
592             throw new ExecutionException(sfe, SOAP_FAULT)
593                     .withResolution(ADDRESS_SOAP_FAULT_MESSAGE_AND_RETRY);
594         } catch (WebServiceException wse) {
595             handleWorkdayWebServiceException(
596                     wse, originalEnvelope, strippedEnvelope, errorViews, documentUtility, document,
597                     xmlUtils);
598             return null;
599         }

1030     /**
1031      * Computes the correct path based on the json schema that was created for the view so that
1032      * we can populate the template correctly.
1033      *
1034      * @param validTemplateOutputs as set of path separated by .
1035      * @return the final set of json paths
1036      */
1037     private List<String> processSchema(final Set<String> validTemplateOutputs) {
1038         JsonSchema collectedSchema = ((SoapEditorSuggestionsProviderImpl) editorSuggestionProvider)
1039                 .getCollectedSchema();
1040         ObjectSchema rootSchema = schemaFactory.createSchema(SnapType.COMPOSITE, null);
1041         jsonSchemaTransformer.transform(collectedSchema, rootSchema);
1042         List<String> finalOutputPaths = Lists.newArrayListWithExpectedSize(
1043                 validTemplateOutputs.size());
1044         for (String path : validTemplateOutputs) {
1045             try {
1046                 List<PathToken> tokens = PathParser.parse(path, EMPTY_EXPR_HANDLER);
1047                 StringBuilder actualPath = new StringBuilder();
1048                 Schema schema = rootSchema;
1049                 int curIndex = 0;
1050                 for (; curIndex < tokens.size() - 1; curIndex++) {
1051                     PathToken token = tokens.get(curIndex);
1052                     String tokenValue = token.toString();
1053                     if (tokenValue.equals(JsonPathBuilder.ROOT_ELEMENT)) {
1054                         continue;
1055                     }
1056                     schema = ((ObjectSchema) schema).getChild(token.toString());
1057                     if (schema.getType() == SnapType.COMPOSITE) {
1058                         actualPath = actualPath.append(tokenValue).append(JsonPathBuilder
1059                                 .CURRENT_ELEMENT);
1060                     } else if (schema.getType() == SnapType.TABLE) {
1061                         actualPath = actualPath.append(tokenValue).append(JsonPathBuilder
1062                                 .ROOT_ARRAY);
1063                     }
1064                 }
1065                 PathToken lastToken = tokens.get(curIndex);
1066                 actualPath = actualPath.append(lastToken.toString());
1067                 finalOutputPaths.add(actualPath.toString());
1068             } catch (InvalidPathException e) {
1069                 throw new ConfigurationException(e, INVALID_PATH)
1070                         .withReason(String.format(INVALID_PATH_REASON, path))
1071                         .withResolution(INVALID_PATH_RESOLUTION);
1072             }
1073         }
1074         return finalOutputPaths;
1075     }
