175     protected Configuration getConfigurationFromEnv(Map<String, ?> env) {
176         final Configuration conf = new Configuration();
177         if (env != null) {
178             for (Map.Entry<String, ? extends Object> entry : env.entrySet()) {
179                 if (entry.getValue() instanceof Boolean) {
180                     conf.setBoolean(entry.getKey(), (Boolean) entry.getValue());
181                 } else if (entry.getValue() instanceof Double) {
182                     conf.setDouble(entry.getKey(), (Double) entry.getValue());
183                 } else if (entry.getValue() instanceof Float) {
184                     conf.setFloat(entry.getKey(), (Float) entry.getValue());
185                 } else if (entry.getValue() instanceof Integer) {
186                     conf.setInt(entry.getKey(), (Integer) entry.getValue());
187                 } else if (entry.getValue() instanceof Long) {
188                     conf.setLong(entry.getKey(), (Long) entry.getValue());
189                 } else if (entry.getValue() instanceof Pattern) {
190                     conf.setPattern(entry.getKey(), (Pattern) entry.getValue());
191                 } else if (entry.getValue() instanceof InetSocketAddress) {
192                     conf.setSocketAddr(entry.getKey(), (InetSocketAddress) entry.getValue());
193                 } else if (entry.getValue() instanceof String[]) {
194                     conf.setStrings(entry.getKey(), (String[]) entry.getValue());
195                 } else {
196                     conf.set(entry.getKey(), String.valueOf(entry.getValue()));
197                 }
198             }

216     public void close() throws IOException {
217         if (isOpen && (fs != null)) {
218             lock.lock();
219             try {
220                 if (isOpen && (fs != null)) {
221                     try {
222                         // the lock should be enough but the synchronized set javadoc
223                         // also recommends to use a synchronization block. There's no
224                         // harm in doing both.
225                         synchronized (DELETE_ON_CLOSE_CACHE) {
226                             for (org.apache.hadoop.fs.Path path : DELETE_ON_CLOSE_CACHE) {
227                                 try {
228                                     if (fs.exists(path)) {
229                                         fs.delete(path, fs.isDirectory(path));
230                                     }
231                                 } catch (IOException e) {
232                                     LOG.debug("exception when deleting file on close: {}",
233                                             e.getMessage());
234                                 }
235                             }
236                         }
237                         fs.close();
238                     } finally {
239                         isOpen = false;
240                     }
241                 }
242             } catch (MalformedURLException e) {
243                 LOG.debug("unexpected MalformedURLException: {}", e.getMessage());
244                 throw new IOException("unexpected MalformedURLException", e);
245             } finally {
246                 lock.unlock();

308      * @see java.nio.file.FileSystem#getPathMatcher(java.lang.String)
309      */
310     @Override
311     public PathMatcher getPathMatcher(String syntaxAndPattern) {
312         final int idx = syntaxAndPattern.indexOf(':');
313         if (idx < 0) {
314             throw new UnsupportedOperationException();
315         }
316 
317         final String syntax = syntaxAndPattern.substring(0, idx);
318         final String pattern = syntaxAndPattern.substring(idx + 1);
319 
320         Pattern regex = null;
321         switch(syntax) {
322             case "glob":
323                 // convert glob to standard regex
324                 final StringBuilder sb = new StringBuilder("\\A");
325                 for (int i = 0; i < pattern.length(); i++) {
326                     char ch = pattern.charAt(i);
327                     switch(ch) {
328                         case '.':
329                             sb.append("\\x2e");
330                             break;
331                         case '*':
332                             if ((i + 1 < pattern.length()) && (pattern.charAt(i + 1) == '*')) {
333                                 sb.append(".*");
334                                 i++;
335                             } else {
336                                 sb.append("[^/]*");
337                             }
338                             break;
339                         case '?':
340                             sb.append("[^/]?");
341                             break;
342                         case '\\':
343                             sb.append("\\\\");
344                             break;
345                         default:
346                             sb.append(ch);
347                     }
348                 }
349                 sb.append("\\z");
350                 regex = Pattern.compile(sb.toString());
351                 break;
352             case "regex":
353                 regex = Pattern.compile(pattern);
354                 break;
355             default:
356                 throw new UnsupportedOperationException();
357         }
358 
359         return new HadoopPathMatcher<AbstractHadoopFileSystem<FS>>(regex);
360     }

389      * Create an output stream.

392      *      java.nio.file.StandardOpenOption[])
393      */
394     FSDataOutputStream newOutputStream(Path path, OpenOption... options) throws IOException {
395         checkPath(path);
396 
397         final org.apache.hadoop.fs.Path hadoopPath = new org.apache.hadoop.fs.Path(
398                 ((AbstractUrlPath<?>) path).getPath());
399 
400         boolean append = false;
401         boolean deleteOnClose = false;
402         boolean truncate = false;
403         for (OpenOption option : options) {
404             if (option instanceof StandardOpenOption) {
405                 switch ((StandardOpenOption) option) {
406                     case APPEND:
407                         append = true;
408                         break;
409                     case CREATE_NEW:
410                         if (fs.exists(hadoopPath)) {
411                             throw new FileAlreadyExistsException(hadoopPath.toString());
412                         }
413                         break;
414                     case DELETE_ON_CLOSE:
415                         deleteOnClose = true;
416                         break;
417                     case TRUNCATE_EXISTING:
418                         truncate = true;
419                         break;
420                 }
421             }
422         }
423 
424         // truncate file if requested
425         if (truncate) {
426             fs.truncate(hadoopPath, 0);
427         }
428 
429         // either create a new file or open an existing one for append.
430         final FSDataOutputStream os = (append) ? fs.append(hadoopPath)
431                 : fs.create(hadoopPath, true);
432 
433         if (deleteOnClose) {
434             DELETE_ON_CLOSE_CACHE.add(hadoopPath);
435 
436             // fallback
437             fs.deleteOnExit(hadoopPath);
438         }
439 
440         return os;
441     }

521         for (AccessMode mode : modes) {

524                     sb.setCharAt(0, 'r');

533         }

536             fs.access(hadoopPath, FsAction.getFsAction(sb.toString()));

652         final org.apache.hadoop.fs.Path hadoopPath = new org.apache.hadoop.fs.Path(
653                 ((AbstractUrlPath<?>) path).getPath());
654 
655         final String[] components = attribute.split(":");
656         final String key = (components.length == 1) ? components[0] : components[1];
657         switch (key) {
658             case HadoopPosixFileAttributes.LAST_MODIFIED_TIME:
659                 fs.setTimes(hadoopPath, ((FileTime) value).toMillis(), -1);
660                 break;
661             case HadoopPosixFileAttributes.LAST_ACCESS_TIME:
662                 fs.setTimes(hadoopPath, -1, ((FileTime) value).toMillis());
663                 break;
664             case HadoopPosixFileAttributes.OWNER:
665                 if (value instanceof String) {
666                     fs.setOwner(hadoopPath, (String) value, null);
667                 } else if (value instanceof UserPrincipal) {
668                     fs.setOwner(hadoopPath, ((UserPrincipal) value).getName(), null);
669                 }
670                 break;
671             case HadoopPosixFileAttributes.GROUP:
672                 if (value instanceof String) {
673                     fs.setOwner(hadoopPath, null, (String) value);
674                 } else if (value instanceof GroupPrincipal) {
675                     fs.setOwner(hadoopPath, null, ((GroupPrincipal) value).getName());
676                 }
677                 break;
678             case HadoopPosixFileAttributes.PERMISSIONS:
679                 if (value instanceof Set) {
680                     @SuppressWarnings("unchecked")
681                     final Set<PosixFilePermission> perms = (Set<PosixFilePermission>) value;
682                     final FsPermission permissions =
683                             HadoopPermissionUtils.convertToFsPermission(perms);
684                     fs.setPermission(hadoopPath, permissions);
685                 }
686         }
