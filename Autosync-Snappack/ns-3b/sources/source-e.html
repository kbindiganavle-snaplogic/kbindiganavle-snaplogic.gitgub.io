67      * contains a Session object, counter and last-access-time. The counter value means the number
68      * of active SFTP channels in the same session.
69      */

73         public SessionMap(final Lock lock) {

77         public Session getSession() {
78             Iterator<Entry<Session, Pair<Integer, Long>>> iterator = this.entrySet().iterator();
79             while (iterator.hasNext()) {
80                 Entry<Session, Pair<Integer, Long>> entry = iterator.next();
81                 Session session = entry.getKey();
82                 Pair<Integer, Long> sessionAttributes = entry.getValue();
83                 Integer counter = sessionAttributes.getKey();
84                 Long lastAccessTime = sessionAttributes.getValue();
85                 long now = System.currentTimeMillis();
86                 if (now - lastAccessTime > sftpSessionTimeoutMillis && counter < 1) {
87                     disconnectQuietly(session);
88                     LOGGER.warn("Idle SFTP session removed for {}@{}, channels={}",
89                             session.getUserName(), session.getHost(), counter);
90                     iterator.remove();
91                     continue;
92                 }
93                 if (counter < maxSftpChannelsPerSession) {
94                     entry.setValue(Pair.of(++counter, now));
95                     LOGGER.warn("SFTP session borrowed, authorities: {}, sessions: {} {}",
96                             SESSIONS.size(), this.size(), this.values());
97                     return session;
98                 }
99             }
100             return null;
101         }

104             return lock;
105         }

107         public void putSession(Session session) {
108             put(session, Pair.of(1, System.currentTimeMillis()));
109             LOGGER.warn("SFTP session created, authorities: {}, sessions: {} {}",

188      * Gets a connected Session object. A lock is applied to the same authority while creating a
189      * new Session object.
190      *
191      * @param authority - username:password@host:port
192      *
193      * @return connected Session object
194      * @throws IOException if it fails to get or create a new Session object
195      */
196     public Session getSession(SftpAuthority authority) throws IOException {
197         Lock lock;
198         boolean isLockAcquired;
199         Session session;
200         SessionMap sessionMap;
201         // lock out all other threads
202         synchronized (SESSIONS) {
203             cleanupSessionCache();
204             sessionMap = SESSIONS.get(authority);
205             if (sessionMap != null) {
206                 session = sessionMap.getSession();
207                 while (session != null) {
208                     if (session.isConnected()) {
209                         LOGGER.debug("Got session from cache, session.isConnected() returned true");
210                         return session;
211                     }
212                     disconnectQuietly(session);
213                     sessionMap.remove(session);
214                     LOGGER.debug("Session is not connected and removed from cache");
215                     session = sessionMap.getSession();
216                 }
217                 lock = sessionMap.getLock();
218             } else {
219                 lock = new ReentrantLock();
220                 sessionMap = new SessionMap(lock);
221                 SESSIONS.put(authority, sessionMap);
222                 LOGGER.warn("New SFTP authority added, {}@{}:{}", authority.getUsername(),
223                         authority.getHost(), authority.getPort());
224             }
225             isLockAcquired = lock.tryLock();
226         }
227         // lock out only other threads with the same authority
228         if (!isLockAcquired) {
229             lock.lock();
230         }
231         try {
232             session = findConnectedSession(sessionMap);
233             if (session != null) {
234                 return session;
235             }
236             session = connectSession(authority);
237             sessionMap.putSession(session);
238             return session;
239         } finally {
240             lock.unlock();
241         }
242     }

254      * Returns a previously-borrowed Session object.
255      * The Session object which all borrowers have returned is disconnected and removed if
256      * removeSession is true. Otherwise, it is kept in the cache for a limited time in case
257      * the session needs to be used again.
258      *
259      * @param sftpAuthority - username:password@host:port
260      * @param session       - Session object which a borrower has finished to use
261      * @param removeSession - Disconnect and remove session if there is no channel in the session
262      */
263     public void returnSession(SftpAuthority sftpAuthority, Session session, boolean removeSession) {
264         synchronized (SESSIONS) {
265             cleanupSessionCache();
266             SessionMap sessionMap = SESSIONS.get(sftpAuthority);
267             if (sessionMap != null) {
268                 Pair<Integer, Long> sessionAttributes = sessionMap.get(session);
269                 Integer counter = sessionAttributes == null ? null : sessionAttributes.getKey();
270                 if (counter != null) {
271                     if (counter > 1) {
272                         sessionMap.put(session, Pair.of(--counter, System.currentTimeMillis()));
273                     } else if (removeSession || !cacheSftpSessions) {
274                         disconnectQuietly(session);
275                         sessionMap.remove(session);
276                         LOGGER.warn("Session object is removed from the cache: {}, {}@{}",
277                                 session, sftpAuthority.getUsername(), sftpAuthority.getHost());
278                     } else {
279                         sessionMap.put(session, Pair.of(0, System.currentTimeMillis()));
280                         LOGGER.warn("Session object is kept in the cache even if no channel is " +
281                                 "using it, session: {}, {}@{}", session,
282                                 sftpAuthority.getUsername(), sftpAuthority.getHost());
283                     }
284                 }
285             }
286         }

326     @VisibleForTesting
327     protected Session connectSession(SftpAuthority authority) throws IOException {
328         Session session = null;
329         String username = authority.getUsername();
330         String password = authority.getPassword();
331         String userPass = StringUtils.EMPTY;
332         String host = authority.getHost();
333         JSch jsch = new JSch();
334         try {
335             // If client is using SSH authentication, password should be a private key, and
336             // the first line should be "-----BEGIN RSA PRIVATE KEY-----" and the last line,
337             // "-----END RSA PRIVATE KEY-----", without which jsch.addIdentity() throws
338             // JSchException with an error message "invalid private key".
339             boolean basicAuth = true;
340             if (password.contains(PRIVATE) && password.contains(KEY) && (password.contains("\n")
341                     || password.contains("\r"))) {
342                 String passphrase = StringUtils.EMPTY;
343                 String combination;
344                 int index;
345                 int userPassIndex;
346                 if ((index = password.lastIndexOf(PASSPHRASE_SEP)) != -1) {
347                     combination = password.substring(index + 11);
348                     if ((userPassIndex = combination.lastIndexOf(PASSWORD_SEP)) != -1) {
349                         userPass = combination.substring(userPassIndex + 10);
350                         passphrase = combination.substring(0, userPassIndex);
351                     } else {
352                         passphrase = combination;
353                     }
354                     password = password.substring(0, index);
355                 }
356                 noAuthUrl = String.format(FORMAT_NO_AUTH_URL, host);
357                 LOGGER.debug(LOG_HAS_PRIVATE_KEY, noAuthUrl, index, !passphrase.isEmpty());
358                 KeyPair kp = KeyPair.load(jsch, password.getBytes(StandardCharsets.UTF_8), null);
359                 if (kp.isEncrypted()) {
360                     if (!kp.decrypt(passphrase)) {
361                         throw new IOException(INVALID_PASSPHRASE_FOR_ENCRYPTED_KEY);
362                     }
363                 }
364                 jsch.addIdentity(SSH_IDENTITY, password.getBytes(StandardCharsets.UTF_8), null,
365                         passphrase.getBytes(StandardCharsets.UTF_8));
366                 basicAuth = false;
367             }
368             // it is nested retries, kept at minimum
369             int retryCount = MIN_RETRY;
370             while (session == null || !session.isConnected()) {
371                 session = jsch.getSession(username, host, authority.getPort());
372                 session.setConfig(KEY_STRICT_HOST_KEY_CHECKING, VALUE_NO);
373                 appendOverrideAlgorithms(session);
374                 if (basicAuth) {
375                     session.setPassword(password);
376                     session.setConfig(KEY_PREFERRED_AUTHENTICATIONS, USER_AUTH_PASSWORD);
377                 } else {
378                     if (StringUtils.isNotEmpty(userPass)) {
379                         session.setPassword(userPass);
380                         session.setConfig(KEY_PREFERRED_AUTHENTICATIONS, USER_AUTH_TWO_FACTOR_SSH);
381                     } else {
382                         session.setConfig(KEY_PREFERRED_AUTHENTICATIONS, USER_AUTH_PUBLIC_KEY);
383                     }
384                 }
385                 try {
386                     session = connect(session);
387                 } catch (JSchException e) {
388                     if (retryCount-- == 0 || StringUtils.startsWithIgnoreCase(e.getMessage(),
389                             AUTH_FAIL)) {
390                         throw new IOException(ERR_GET_SESSION, e);
391                     }
392                     LOGGER.debug(LOG_SESSION_RETRY, MIN_RETRY - retryCount, e.getMessage());
393                     SftpUrlConnection.delay(RETRY_DELAY_IN_MILLISECONDS);
394                 }
395             }
396         } catch (JSchException e) {
397             LOGGER.warn("Failed to get SFTP session for {}, error: {}", noAuthUrl, e.getMessage());
398             throw new IOException(String.format(ERR_CONNECTION, noAuthUrl), e);
399         }
400         return session;
401     }

404      * It has been reported that stale sessions in the cache exist, which caused resource leak.
405      * This method finds, disconnects, and removes any stale session which hasn't been used to open
406      * an additional channel or to return a channel to the cache. If a session has not be accessed
407      * for more than sftpSessionTimeoutMillis and its counter is zero, it is considered stale.
408      */
409     public void cleanupSessionCache() {
410         if ((System.currentTimeMillis() - lastCleanupTime) >
411                 sftpSessionTimeoutMillis) {
412             removeIdleSessions(sftpSessionTimeoutMillis);
413         }
414     }

416     /**
417      * Removes all sessions from cache which have been idling more that expiry time in ms.
418      *
419      * @param expiry_ms - expiry time in milli-seconds
420      */
421     protected void removeIdleSessions(long expiry_ms) {
422         synchronized (SESSIONS) {
423             long now = System.currentTimeMillis();
424             for (SessionMap sessionMap : SESSIONS.values()) {
425                 Iterator<Map.Entry<Session, Pair<Integer, Long>>> iterator =
426                         sessionMap.entrySet().iterator();
427                 while (iterator.hasNext()) {
428                     Map.Entry<Session, Pair<Integer, Long>> entry = iterator.next();
429                     Pair<Integer, Long> sessionAttributes = entry.getValue();
430                     long msInCache = now - sessionAttributes.getValue();
431                     Integer counter = sessionAttributes.getKey();
432                     if (msInCache > expiry_ms && counter < 1) {
433                         Session session = entry.getKey();
434                         LOGGER.warn("Idle session removed for username: {}, host: {}," +
435                                 " channels: {}, Minutes in cache: {}", session.getUserName(),
436                                 session.getHost(), counter, msInCache / 1000 / 60);
437                         disconnectQuietly(session);
438                         iterator.remove();
439                     }
440                 }
441             }
442             lastCleanupTime = now;
443         }
444     }
