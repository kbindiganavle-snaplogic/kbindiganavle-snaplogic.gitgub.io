140     public boolean checkDate(Object value) {
141         boolean checkTZFormats = false;
142         boolean checkNoTZFormats = false;
143 
144         checkTZFormats = checkFormats(value.toString(), true,
145                 WITH_TZ_FORMATS);

152     public boolean checkFormats(String dateString, boolean hasTZ, String[] formats) {
153         boolean valid = false;
154 
155         for (String pattern : formats) {
156             DateTime dt = parse(dateString, pattern, hasTZ);
157             if (dt != null) {
158                 valid = true;
159                 break;
160             }
161         }
162 
163         return valid;
164     }

166     public DateTime parse(String dateString, String pattern, boolean hasTZ) {
167         DateTimeFormatter jodaDateFormatter = null;
168 
169         if (Date.Parse.canUseJoda(pattern)) {
170             jodaDateFormatter = DateTimeFormat.forPattern(pattern);
171         }
172 
173         DateTime retval = null;
174 
175         if (jodaDateFormatter != null) {
176             retval = parseWithJoda(dateString, jodaDateFormatter);
177         }
178 
179         if (retval != null && hasTZ) {
180             retval = retval.withZone(DateTimeZone.UTC);
181         } else if (retval != null) {
182             retval = retval.withZoneRetainFields(DateTimeZone.UTC);
183         }
184 
185         return retval;
186     }
187 
188     private DateTime parseWithJoda(String dateString, DateTimeFormatter jodaDateFormatter) {
189         try {
190             return jodaDateFormatter.parseDateTime(dateString);
191         } catch (IllegalArgumentException e) {
192             return null;
193         }
194     }
