23 /**

26 public class DetectionResult {

28 
29     /**
30      * Map of {SourceDrift : SourceDriftInfo}

33      * For example, if we detect two ADD_COLUMN drifts, single SourceDriftInfo object

35      */
36     private Map<SourceDrift, SourceDriftInfo> driftInfoMap;

41     }

47     }

59     }

62         this.driftInfoMap = driftInfoMap;
63     }
64 
65     public boolean noDriftDetected() {
66         return isEmpty();
67     }
68     
69     public void addDrift(SourceDrift drift, DriftUnitType unitType,
70                          Map<String, Object> schemaInfo) {
71         SourceDriftInfo info = driftInfoMap.getOrDefault(
72                 drift, new SourceDriftInfo(drift, unitType));
73         info.addColumnSchemaInfo(schemaInfo);
74         driftInfoMap.put(drift, info);
75     }

77     public void addDrift(SourceDrift drift, DriftUnitType unitType,
78                          List<Map<String, Object>> schemaInfoList) {
79         SourceDriftInfo info =
80                 driftInfoMap.getOrDefault(drift, new SourceDriftInfo(drift, unitType));
81         info.addColumnSchemaInfoSet(new LinkedHashSet<>(schemaInfoList));
82         driftInfoMap.put(drift, info);
83     }
