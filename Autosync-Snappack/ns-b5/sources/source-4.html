393         String regexpr = quoteRegex ? Pattern.quote(regex) : regex;
394         Pattern pattern = Pattern.compile(regexpr);
395         Matcher matcher = pattern.matcher(sb);
396         int start = 0;
397         while (matcher.find(start)) {
398             sb.replace(matcher.start(), matcher.end(), replacement);
399             start = matcher.start() + replacement.length();
400         }
