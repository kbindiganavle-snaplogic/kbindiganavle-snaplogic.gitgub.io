239     @Override
240     @SuppressWarnings("unchecked")
241     protected void process(Document document, String inputViewName) {
242         if (batch != null && batch.size() == batchSize) {
243             executeAndClearBatch(document);
244         }
245         if (batch != null && batch.size() < batchSize) {
246             // Evaluating the header fields needs to be done with one of the original
247             // input documents.  So, we save the last one received and use that.
248             lastBatchDocument = document;
249             Object result = httpEntity.eval(document);
250             batch.add(result);
251             if (batch.size() == batchSize){
252                 executeAndClearBatch(document);
253 
254             }
255         } else {
256             getRequestExecutor(document).execute(document, urlResolver, requestHandler);
257         }
258     }

260     private void executeAndClearBatch(Document document){
261         Document batchDocument = documentUtility.newDocumentFor(document, batch);
262         getRequestExecutor(lastBatchDocument).execute(lastBatchDocument, batchDocument,
263             urlResolver, requestHandler);
264         batch.clear();
265     }
