359     protected void processDocument(final Document document, final String inputViewName) {
360         checkLoadingThread();
361         // Check if this is the schema document and save it for later.
362         if (inputViewName != null && inputViewName.equals(schemaViewName)) {
363             if (document != null) {
364                 schema = document;
365                 columnTranslation = databaseUtils.buildColumnTranslation(schema);
366             }
367             return;
368         }
369         if (document == null || !successful) {
370             return;
371         }
372         conn = jdbcOperations.acquireConnection(account);
373 
374         if (firstRecord) {
375             firstRecord = false;
376             if (autoCreateTable) {
377                 createTableIfAbsent(document);
378             }
379             tableMetaData = jdbcOperations.getTableMetaData(conn,
380                     account.getDatabaseConfig().getDatabaseName(), tableName);
381             if (!getInputSchemaFields()) {
382                 successful = false;
383                 return;

418             }
419         } catch (IOException e) {

421                     .withReason(e.getMessage())
422                     .withResolution(PLEASE_ADDRESS_REPORTED_ISSUE);
423         }
424 
425         // Handle batch
426         batchSize++;
427         counterForInterval++;
428         if (counterForInterval == maxRecordCountForInterval) {
429             counterForInterval = 0;
430             try {
431                 Thread.sleep(100);
432             } catch (InterruptedException ipe) {
433             }
434         }

440     @VisibleForTesting
441     protected String processRecord(Map<String, Object> record, List<String> columns) {
442         Object value;
443         StringBuilder builder = new StringBuilder();
444         for (int i = 0; i < columns.size(); ++i) {
445             String mappedKey;
446             String actualKey = columns.get(i);
447             mappedKey = getMappedKeyIgnoreCase(keySet, actualKey);
448             if (mappedKey == null) {
449                 if (getMappedKeyIgnoreCase(virtualColumns, actualKey) == null) {
450                     // it is not a virtual column and its value is required
451                     SnapDataException ex = new SnapDataException(ERR_INVALID_DATA_MSG)
452                             .withReason(String.format(ERR_INVALID_DATA_REASON_FMT,
453                                     actualKey))
454                             .withResolution(String.format(ERR_INVALID_DATA_RESOLUTION_FMT,
455                                     actualKey));
456                     errorViews.write(ex);
457                     return null;
458                 } else if (replaceOnDuplicates) {
459                     // Add default value of virtual column if 'On duplicate' is REPLACE.
460                     // MySQL throws SQLException if input data does not contain virtual
461                     // column value.
462                     builder.append(DEFAULT_VALUE);
463                 } else {
464                     // skip virtual column
465                 }
466             } else {
467                 value = record.get(mappedKey);
468                 builder.append(convertValue(value));
469             }
470             if (i < (columns.size() - 1)) {
471                 builder.append(FIELD_DELIMITER);
472             }
473         }
474         return builder.toString();
475     }

486     private void createAndRunLoadingThread() {
487         loadingThread = new Thread(new Runnable() {
488             @Override
489             public void run() {
490                 while (true) {
491                     String fileName = null;
492                     try {
493                         fileName = fileQueue.take();
494                         if (fileName.equals(COMPLETED_MESSAGE)) {
495                             break;
496                         }
497                         st.clearParameters();
498                         if (!(st instanceof MockStatement)) {
499                             // Use reflection to avoid classloading issue
500                             Method method = st.getClass().getMethod("setLocalInfileInputStream",
501                                     InputStream.class);
502                             method.invoke(st, new FileInputStream(new File(fileName)));
503                         }
504                         st.execute();
505                         updateCount += st.getUpdateCount();
506                     } catch (FileNotFoundException | NoSuchMethodException |
507                             IllegalAccessException | InvocationTargetException e) {
508                         throw new ExecutionException(e, ERR_LOADING_DATA)
509                                 .withReason(e.getMessage())
510                                 .withResolutionAsDefect();
511                     } catch (SQLException e) {
512                         SnapDataException snapDataException = new SnapDataException(e,
513                                 ERR_LOADING_DATA)
514                                 .withReason(e.getMessage())
515                                 .withResolution(PLEASE_ADDRESS_REPORTED_ISSUE);
516                         errorViews.write(snapDataException);
517                         return;
518                     } catch (InterruptedException e) {
519                         // no op
520                     } finally {
521                         if (fileName != null) {
522                             FileUtils.deleteQuietly(new File(fileName));
523                         }
524                     }
525                 }
526             }
527         });
528         loadingThread.setUncaughtExceptionHandler(new Thread.UncaughtExceptionHandler() {
529             @Override
530             public void uncaughtException(final Thread t, final Throwable e) {
531                 loadingException = e;
532                 loadingExceptionThrown = true;
533             }
534         });
535         loadingThread.start();
536     }

538     protected String convertValue(final Object value) {
539         if (value == null) {
540             // An unquoted NULL string is used to represent a NULL value when FIELD ENCLOSED BY
541             // is specified.
542             return NULL_STR;
543         } else {
544             String valueStr = StringUtils.replace(value.toString(), FIELD_ENCLOSED_BY_STR,
545                     FIELD_ENCLOSED_BY_STR + FIELD_ENCLOSED_BY_STR);
546             // if value ends with odd number of backslash, it has to be escaped
547             if (valueStr.endsWith(BACKSLASH)) {
548                 // count the number of backslashes at the end
549                 int count = 0;
550                 for (int i = valueStr.length() - 1; i >= 0; i--) {
551                     if (valueStr.charAt(i) == BACKSLASH_CHAR) {
552                         count++;
553                     } else {
554                         break;
555                     }
556                 }
557                 valueStr = count % 2 == 1 ? valueStr.concat(BACKSLASH) : valueStr;
558             }
559             return String.format(QUOTED_VALUE_FMT, FIELD_ENCLOSED_BY_STR, valueStr,
560                     FIELD_ENCLOSED_BY_STR);
561         }
562     }
