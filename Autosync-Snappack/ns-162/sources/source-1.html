801     /**
802      * gets Evaluated velocity template from the provided Tableau XML schema
803      * The Tableau XML schema is a global schema - combines all operation payloads into one xsd.
804      * Extraneous input document elements that are not defined in the global Tableau schema will
805      * be dropped.
806      *
807      * @param schema                 vendor XML Schema
808      * @param inputValues            inputs for Velocity
809      * @param xmlTemplateEvaluator   template evaluator
810      * @param xmlUtils               Xml Utils
811      * @param documentBuilderFactory document builder factory
812      * @return
813      */
814     public static String evaluateTableauSchema(String schema, Map<String, Object> inputValues,
815             XMLTemplateEvaluatorImpl xmlTemplateEvaluator, XmlUtils xmlUtils,
816             DocumentBuilderFactory documentBuilderFactory,
817             DOMUtil domUtil, TypesFactoryImpl typesFactoryImpl)
818             throws SnapDataException {
819 
820         String xmlTemplate = getVelocityTemplate(schema, domUtil, typesFactoryImpl);
821         String templateOut = evaluateTemplate(inputValues, xmlTemplateEvaluator, xmlUtils,
822                 documentBuilderFactory, xmlTemplate);
823         return templateOut;
824     }
825 
826     private static String evaluateTemplate(Map<String, Object> inputValues,
827             XMLTemplateEvaluatorImpl xmlTemplateEvaluator, XmlUtils xmlUtils,
828             DocumentBuilderFactory documentBuilderFactory, String xmlTemplate) {
829         try {
830             String xmlResult = xmlTemplateEvaluator.evaluate(xmlTemplate, inputValues);
831             return xmlUtils.stripNodeAndAttributeNodes(xmlResult, documentBuilderFactory);
832         } catch (SnapException e) {
833             throw new SnapDataException(Messages.INVALID_MAPPING)
834                     .withReason(e.getMessage())
835                     .withResolution(Messages.PLEASE_FIX_MAPPING_AND_RETRY);
836         }
837     }
838 
839     /*
840     Released Tableau schemas are provided, extensible for future Tableau release schemas
841      */
842     private static String getVelocityTemplate(String schema, DOMUtil domUtil,
843             TypesFactoryImpl typesFactoryImpl) throws SnapDataException {
844         String velocityResource = SCHEMA_MAP.get(schema);
845         String xmlTemplate;
846         if (StringUtils.isNotBlank(velocityResource)) {
847             try {
848                 URL url = Thread.currentThread().getContextClassLoader()
849                         .getResource(velocityResource);
850                 xmlTemplate = TEMPLATE_CACHE.getIfPresent(url);
851                 if (xmlTemplate == null) {
852                     InputStream is = url.openStream();
853                     xmlTemplate = IOUtils.toString(is);
854                     TEMPLATE_CACHE.put(url, xmlTemplate);
855                 }
856             } catch (IOException e) {
