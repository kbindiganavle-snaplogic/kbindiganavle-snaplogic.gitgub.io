236         @Override
237         protected void writeToOutput(Object content, Document document, Map<String,
238                 Object> statusLine, Map<String, Object> headersResult) {
239             Document outputDocument;
240             if (!isFirstRowProcessed) {
241                 isFirstRowProcessed = true;
242             }
243             int count = 1;
244             if (content instanceof Map) {
245                 Map<String, Object> responseData = (Map<String, Object>) content;
246                 Map<String, Object> tsResponse = (Map<String, Object>) responseData.get(
247                         TS_RESPONSE);
248                 for (Map.Entry<String, Object> object : tsResponse.entrySet()) {
249                     if (object.getValue() instanceof Map && !object.getKey().startsWith(AT_CHAR)
250                             && !object.getKey().equalsIgnoreCase(PAGINATION)) {
251                         Map<String, Object> tsResponseChild =
252                                 (Map<String, Object>) object.getValue();
253                         Map<String, Object> outputMap =
254                                 new LinkedHashMap<>(tsResponseChild.size() + 1);
255                         for (Map.Entry<String, Object> entry : tsResponseChild.entrySet()) {
256                             if (StringUtils.isBlank(entry.getKey())) {
257                                 continue;
258                             }
259                             if (entry.getValue() instanceof List) {
260                                 List<Object> listObject = (List<Object>) entry.getValue();
261                                 if (operation.isPaged()) {
262                                     for (Object listChildObject : listObject) {
263                                         if (listChildObject instanceof Map) {
264                                             Map<String, Object> listChildMap =
265                                                     (Map<String, Object>) listChildObject;
266                                             for (Map.Entry<String, Object> listChildMapEntry :
267                                                     listChildMap.entrySet()) {
268                                                 if (!writeObject(outputMap,
269                                                         listChildMapEntry.getKey(),
270                                                         listChildMapEntry.getValue())) {
271                                                     continue;
272                                                 }
273                                             }
274                                         } else if (listChildObject instanceof String ||
275                                                 listChildObject instanceof  List) {
276                                             outputMap.put(entry.getKey().replaceAll(AT_CHAR,
277                                                     StringUtils.EMPTY), object);
278                                         } else if (listChildObject != null) {
279                                             SnapDataException sde = new SnapDataException(
280                                                     ERROR_WRITING_OUTPUT)
281                                                     .withReason(String.format(
282                                                             UNEXPECTED_TYPE_S_FOR_FIELD_S,
283                                                             listChildObject.getClass().toString(),
284                                                             entry.getKey()))
285                                                     .withResolutionAsDefect();
286                                             errorViews.write(sde);
287                                         }
288                                         outputDocument = documentUtility.newDocumentFor(document,
289                                                 outputMap);
290                                         outputViews.write(outputDocument);
291                                         // we need to create a new instance of the
292                                         //  {@link LinkedHashMap} after writing to the output view
293                                         outputMap = new LinkedHashMap<>();
294                                         count++;
295                                         if (isPreview && count > maxSuggestValue.intValue()) {
296                                             return;
297                                         }
298                                     }
299                                 } else {
300                                     outputMap.put(entry.getKey(), listObject);
301                                 }
302                             } else {
303                                 if (!writeObject(outputMap, entry.getKey(), entry.getValue())) {
304                                     continue;
305                                 }
306                             }
307                         }
308                         if (!operation.isPaged()) {
309                             outputDocument = documentUtility.newDocumentFor(document, outputMap);
310                             outputViews.write(outputDocument);
311                             count++;
312                             if (isPreview && count > 50) {
313                                 return;
314                             }
315                         }
316                     }
317                 }
318             }  else {
319                 super.writeToOutput(content, document, statusLine, headersResult);
320             }
321             if (operation.isPaged()) {
322                 Object totalAvailableObject = TOTAL_AVAIL_JSONPATH.readStatic(content, null);
323                 Object pageNumberObject = PAGE_NUMBER_JSONPATH.readStatic(content, null);
324                 Object pageSizeObject = PAGE_SIZE_JSONPATH.readStatic(content, null);
325                 BigInteger totalAvailable = new BigInteger((String) totalAvailableObject);
326                 BigInteger pageNumber = new BigInteger((String) pageNumberObject);
327                 BigInteger pageSize = new BigInteger((String) pageSizeObject);
328                 BigInteger remaining = totalAvailable.subtract(pageNumber.multiply(pageSize));
329                 if (remaining.compareTo(BigInteger.ZERO) == 1) {
330                     nextUrlObject = String.format(PAGE_NUMBER_PARAMETER, pageNumber.add(
331                             BigInteger.ONE));
332                 } else {
333                     nextUrlObject = StringUtils.EMPTY;
334                 }
335             }
336         }

338         private boolean writeObject(Map<String, Object> outputMap, String key, Object object) {
339             if (object instanceof Map) {
340                 Map<String, Object> map = (Map<String, Object>) object;
341                 Map<String, Object> entryOutputMap = new LinkedHashMap<>(map.size() + 1);
342                 for (Map.Entry<String, Object> mapEntry : map.entrySet()) {
343                     if (!writeObject(entryOutputMap, mapEntry.getKey(), mapEntry.getValue())) {
344                         continue;
345                     }
346                     outputMap.put(key.replaceAll(AT_CHAR, StringUtils.EMPTY), entryOutputMap);
347                 }
348             } else if (object instanceof String || object instanceof  List) {
349                 outputMap.put(key.replaceAll(AT_CHAR, StringUtils.EMPTY), object);
350             } else if (object != null) {
351                 SnapDataException sde = new SnapDataException(ERROR_WRITING_OUTPUT)
352                         .withReason(String.format(UNEXPECTED_TYPE_S_FOR_FIELD_S,
353                                 object.getClass().toString(), key))
354                         .withResolutionAsDefect();
355                 errorViews.write(sde);
356                 return false;
357             }
358             return true;
359         }

418     private void defineInputSchemaMap(ObjectSchema schema, SchemaProvider provider,
419             Map<String, Object> inputValues) {
420         for (Map.Entry<String, Object> entrySet : inputValues.entrySet()) {
421             if (entrySet.getValue() instanceof Map) {
422                 Map<String, Object> childMap = (Map<String, Object>) entrySet.getValue();
423                 ObjectSchema childSchema = provider.createSchema(SnapType.COMPOSITE,
424                         entrySet.getKey());
425                 schema.addChild(childSchema);
426                 defineInputSchemaMap(childSchema, provider, childMap);
427             } else if (entrySet.getValue() instanceof List) {
428                 List<Object> list = (List<Object>) entrySet.getValue();
429                 ObjectSchema tableSchema = provider.createSchema(SnapType.TABLE,
430                         entrySet.getKey());
431                 schema.addChild(tableSchema);
432                 ObjectSchema objectSchema = provider.createSchema(SnapType.COMPOSITE, null,
433                         entrySet.getKey());
434                 tableSchema.addChild(objectSchema, SnapType.TABLE);
435                 for (Object obj : list) {
436                     if (obj instanceof Map) {
437                         Map<String, Object> objectMap = (Map<String, Object>) obj;
438                         ObjectSchema childObjectSchema =
439                                 provider.createSchema(SnapType.COMPOSITE, null, entrySet.getKey());
440                         objectSchema.addChild(childObjectSchema);
441                         defineInputSchemaMap(childObjectSchema, provider, objectMap);
442                     }
443                 }
444             } else {
445                 ObjectSchema childSchema =
446                         provider.createSchema(SnapType.STRING, entrySet.getKey().replace(AT_CHAR,
447                                 StringUtils.EMPTY));
448                 schema.addChild(childSchema);
449             }
450         }

542     protected void process(Document document, String inputViewName) {
543         BigInteger timeoutValue = timeoutExpression.eval(document);
544         timeout = (timeoutValue == null || timeoutValue.intValue() == 0) ?
545                 DEFAULT_SOCKET_TIMEOUT_IN_SEC : timeoutValue.intValue();
546         BigInteger value = maxRetriesExpression.eval(document);
547         maxRetries = value == null ? DEFAULT_MAX_RETRIES : value.intValue();
548         chunkSize = chunkSizeExpression.eval(document);
549         chunkSize = chunkSize == null ? DEFAULT_CHUNK_SIZE : chunkSize;
550         String operationString = restOperationExpressionProperty.eval(document);
551 
552         try {
553             operation = getValidatedOperation(operationString);
554             Map<String, Object> documentData = (document != null) ? (Map) document.get() :
555                     new LinkedHashMap<>();
556             List<Header> headers = new ArrayList<>();
557             headers.addAll(operation.getHeaderList());
558             if (!this.equals(APIUtils.Operation.SIGN_IN)) {
559                 String token = tableauRestAccount.connect();
560                 if (StringUtils.isBlank(token)) {
561                     SnapDataException sde = new SnapDataException(AUTHENTICATION_ERROR)
562                             .withReason(NO_TOKEN)
563                             .withResolution(PLEASE_VERIFY_ACCOUNT_AND_RETRY);
564                     errorViews.write(sde, document);
565                     return;
566                 }
567                 headers.add(new BasicHeader(APIUtils.AUTH_HEADER_KEY, token));
568                 documentData.computeIfAbsent(SITE_ID, k -> tableauRestAccount.getSiteId());
569                 documentData.computeIfAbsent(USER_ID, k -> tableauRestAccount.getUserId());
570             }
571 
572             uploadFile = uploadFileExpressionProperty.eval(document);
573             if (chunkSize.equals(BigInteger.ZERO) &&
574                     operation == APIUtils.Operation.APPEND_FILE_UPLOAD) {
575                 throw new SnapDataException(ERR_APPEND_FILE_UPLOAD_CHUNKSIZE)
576                         .withReason(ERR_APPEND_FILE_UPLOAD_CHUNKSIZE_REASON)
577                         .withResolution(ERR_APPEND_FILE_UPLOAD_CHUNKSIZE_RESOLUTION);
578             } else if ((documentData.get(UPLOAD_SESSION_ID) != null) &&
579                     (operation == APIUtils.Operation.PUBLISH_DATASOURCE ||
580                             operation == APIUtils.Operation.PUBLISH_WORKBOOK)) {
581                 finalizingUpload(document, documentData, headers);
582                 return;
583             } else if ((documentData.get(UPLOAD_SESSION_ID) != null) &&
584                     operation == APIUtils.Operation.APPEND_FILE_UPLOAD) {
585                 try {
586                     checkUploadFileParameter();
587                     appendFileOperation(document, documentData, headers);
588                     return;
589                 } catch (IOException e) {
590                     String reason = e.getMessage();
591                     throw new SnapDataException(FILE_UPLOAD_ERROR)
592                             .withReason(StringUtils.isNotBlank(reason) ? reason :
593                                     REASON_INPUTSTREAM)
594                             .withResolution(RESOLUTION_ADDRESS_ISSUE);
595                 }
596 
597             } else if (StringUtils.isNotBlank(operation.getBinaryName()) &&
598                     (operation.getHttpMethod().equals(APIUtils.HttpMethod.POST)
599                             || operation.getHttpMethod().equals(APIUtils.HttpMethod.PUT))
600                     && !chunkSize.equals(BigInteger.ZERO)) {
601                 checkUploadFileParameter();
602                 processChunkBinaryUpload(document, documentData, headers);
603                 return;
604             }
605 
606             if (operation == APIUtils.Operation.PUBLISH_WORKBOOK || operation == APIUtils
607                     .Operation.PUBLISH_DATASOURCE) {
608                 checkUploadFileParameter();
609             }
610             TableauRestResponseProcessor restResponseProcessor =
611                     new TableauRestResponseProcessor(xmlUtils, operation.getHttpMethod().name()
612                             .toString(),
613                             outputViews, errorViews, documentUtility, operation);
614             if (operation == APIUtils.Operation.DOWNLOAD_DATASOURCE ||
615                     operation == APIUtils.Operation.DOWNLOAD_WORKBOOK) {
616                 restResponseProcessor.withRawData(true);
617             }
618 
619             String payload = APIUtils.evaluateTableauSchema(
620                     tableauRestAccount.getSchema(tableauRestAccount.getServerVersion()),
621                     documentData, xmlTemplateEvaluator, xmlUtils, documentBuilderFactory, domUtil,
622                     typesFactoryImpl);
623             if (payload.equalsIgnoreCase(EMPTY_PAYLOAD) &&
624                     operation.getDocumentedPayload() != null) {
625                 payload = payload + operation.getDocumentedPayload();
626             }
627             final HttpEntity entity = getHttpEntity(document, operation, documentData, headers,
628                     payload);
629             // RestRequestExecutor retries behaves as attempts, so adjust +1 so that 0 retry still
630             // tries at least once
631             RestRequestExecutor restRequestExecutor = new RestRequestExecutor(
632                     restHttpClient, errorViews)
633                     .withHeaders(headers.toArray(new Header[0]))
634                     .withMethod(operation.getHttpMethod().name().toString())
635                     .withRetryInterval(5)
636                     .withMaxRetries(maxRetries + 1)
637                     .withSocketTimeout(timeout)
638                     .withResponseProcessor(restResponseProcessor);
639 
640             URL url = getUrl(documentData, operation, tableauRestAccount.getURLPath());
641             PagingURLResolver pagingURLResolver = new PagingURLResolver();
642             String originalURLString = url.toString();
643             pagingURLResolver.setURLString(originalURLString);
644             RequestDataHandler requestDataHandler = new RequestDataHandler<HttpEntity>() {
645                 @Override
646                 public HttpEntity createEntity(Document document) {
647                     return entity;
648                 }
649             };
650 
651             restRequestExecutor.execute(document, pagingURLResolver, requestDataHandler);
652             if (operation.isPaged()) {
653                 String nextUrlObject = restResponseProcessor.getNextUrlObject();
654                 while (StringUtils.isNotBlank(nextUrlObject)) {
655                     url = (originalURLString.contains("?")) ?
656                             new URL(originalURLString + "&" + nextUrlObject)
657                             : new URL(originalURLString + "?" + nextUrlObject);
658                     pagingURLResolver.setURLString(url.toString());
659                     restRequestExecutor.execute(document, pagingURLResolver, requestDataHandler);
660                     nextUrlObject = restResponseProcessor.getNextUrlObject();
661                 }
662             }
663         } catch (SnapDataException sde) {
664             if (tableauRestAccount.getErrorMap() != null) {
665               SnapDataException snapDataException =
666                       new SnapDataException(documentUtility.newDocument(
667                               tableauRestAccount.getErrorMap()),
668                               sde.getMessage())
669                       .withReason(sde.getReason())
670                       .withResolution(sde.getResolution());
671                 errorViews.write(snapDataException, document);
672                 return;
673             }
674             errorViews.write(sde, document);
675             return;
676         } catch (MalformedURLException | URISyntaxException e) {
677             SnapDataException sde = new SnapDataException(CONNECTION_ERROR)
678                     .withReason(INVALID_SERVER_URL)
679                     .withResolution(PLEASE_CORRECT_SERVER_NAME);
680             errorViews.write(sde, document);
681             return;
682         }

845     protected URL getUrl(Map<String, Object> documentData, APIUtils.Operation operation,
846             String serverURL) throws MalformedURLException, URISyntaxException {
847         URL tableauServerURL = new URI(serverURL).toURL();
848         UriBuilder uriBuilder = UriBuilder.fromPath(operation.getPath());
849         Matcher matcher = PATHFIELD_PARAM.matcher(operation.getPath());
850         while (matcher.find()) {
851             String pathField = matcher.group().substring(1, matcher.group().length() - 1);
852             if (documentData.get(pathField) == null) {
853                 throw new SnapDataException(INPUT_ERROR)
854                         .withReason(String.format(REQUIRED_INPUT_INVALID, pathField))
855                         .withResolution(PLEASE_FIX_INPUT_MAPPING_AND_RETRY);
856             }
857         }
858         for (String param : operation.getParams()) {
859             if (documentData.get(param) != null) {
860                 uriBuilder = uriBuilder.queryParam(param, "{" + param + "}");
861             }
862         }
863         return new URL(tableauServerURL, uriBuilder.buildFromMap(documentData).toString());
864     }

914     private HttpEntity getHttpEntity(Document document, APIUtils.Operation operation,
915             Map<String, Object> documentData,
916             List<Header> headers, String payload) {
917         if (!operation.getHttpMethod().equals(APIUtils.HttpMethod.POST) &&
918                 !operation.getHttpMethod().equals(APIUtils.HttpMethod.PUT)) {
919             return null;
920         }
921         ContentBody contentBody = null;
922         if (StringUtils.isNotBlank(operation.getBinaryName())) {
923             String uploadFile = uploadFileExpressionProperty.eval(document);
924             contentBody = getBinaryContent(uploadFile);
925         }
926         if (contentBody == null || StringUtils.isBlank(operation.getBinaryName())) {
927             return new StringEntity(payload, UTF_8);
928         } else {
929             HttpEntity entity;
930             if (contentBody == null) {
931                 String content = (String) documentData.get(CONTENT);
932                 String content_disposition = (String) documentData.get(CONTENT_DISPOSITION);
933                 BasicHeader basicHeader = new BasicHeader(CONTENT_DISPOSITION,
934                         content_disposition);
935                 HeaderElement[] headerElements = basicHeader.getElements();
936                 String fileName = DEFAULT;
937                 if (headerElements.length > 0) {
938                     HeaderElement headerElement = headerElements[0];
939                     if (headerElement.getName().equalsIgnoreCase(ATTACHMENT)) {
940                         NameValuePair nmv = headerElement.getParameterByName(FILENAME);
941                         if (nmv != null) {
942                             fileName = nmv.getValue();
943                         }
944                     }
945                 }
946                 entity = MultipartEntityBuilder
947                         .create()
948                         .addTextBody(REQUEST_PAYLOAD, payload, ContentType.TEXT_XML)
949                         .addBinaryBody(CONTENT, content.getBytes(),
950                                 ContentType.APPLICATION_OCTET_STREAM, fileName)
951                         .build();
952                 headers.add(new BasicHeader(HttpHeaders.CONTENT_TYPE, MULTIPART_MIXED
953                         + entity.getContentType().getValue().split(";")[1]));
954             } else {
955                 entity = MultipartEntityBuilder
956                         .create()
957                         .addTextBody(REQUEST_PAYLOAD, payload, ContentType.TEXT_XML)
958                         .addPart(operation.getBinaryName(), contentBody)
959                         .build();
960                 headers.add(new BasicHeader(HttpHeaders.CONTENT_TYPE, MULTIPART_MIXED
961                         + entity.getContentType().getValue().split(";")[1]));
962             }
963             return entity;
964         }
