376     @Override
377     public String connect() throws SnapDataException {
378         HttpEntityEnclosingRequestBase uriRequest = null;
379         try {
380             // if user provides both access-key ID and secret-key, verify them
381             if (StringUtils.isNotBlank(accessKeyID) && StringUtils.isNotBlank(accessKeySecret)) {
382                 verifyS3Credentials();
383             }
384             URL tableauServerURL = new URI(getURLPath()).toURL();
385             URL url = new URL(tableauServerURL, APIUtils.Operation.SIGN_IN.getPath());
386             uriRequest = new HttpPost(url.toURI());
387             HttpEntity requestEntity = getHttpEntity();
388             uriRequest.setEntity(requestEntity);
389             uriRequest.setHeaders(APIUtils.SIGNIN_HEADERS.toArray(new Header[0]));
390             HttpResponse response = restHttpClient.executeRequest(uriRequest, false);
391             HttpEntity responseEntity = response.getEntity();
392             if (response.getStatusLine().getStatusCode() != 200
393                     && response.getStatusLine().getStatusCode() != 201) {
394                 errorMap = new LinkedHashMap<>();
395                 if (responseEntity != null) {
396                     try {
397                         errorMap = mapXMLContent(responseEntity);
398                     } catch (Exception e) {
399                         // skip attempt of getting more detail from response
400                     }
401                 }
402                 throw new SnapDataException(Messages.AUTHENTICATION_ERROR)
403                         .withReason(String.format(Messages.CODE_REASON_RESPONSE,
404                                 response.getStatusLine().getStatusCode(),
405                                 response.getStatusLine().getReasonPhrase()))
406                         .withResolution(Messages.VERIFY_ACCOUNT_CONFIGURATION_AND_RETRY);
407             }
408             if (responseEntity == null) {
409                 throw new SnapDataException(Messages.AUTHENTICATION_ERROR)
410                         .withReason(Messages.NO_ENTITY_CONTENT_FROM_ENDPOINT)
411                         .withResolution(Messages.VERIFY_ACCOUNT_CONFIGURATION_AND_RETRY);
412             }
413             Map<String, Object> statusMap = mapXMLContent(responseEntity);
414             String token = TOKEN_ID.readStatic(statusMap, null);
415             siteId = SITE_ID.readStatic(statusMap, null);
416             userId = USER_ID.readStatic(statusMap, null);
417             authToken = token;
418             return token;
419         } catch (IOException e) {
420             throw new SnapDataException(Messages.AUTHENTICATION_ERROR)
421                     .withReason(e.getMessage())
422                     .withResolution(Messages.VERIFY_ACCOUNT_CONFIGURATION_AND_RETRY);
423         } catch (URISyntaxException | InvalidPathException e) {
424             throw new SnapDataException(Messages.AUTHENTICATION_ERROR)
425                     .withReason(e.getMessage())
426                     .withResolution(Messages.VERIFY_ACCOUNT_CONFIGURATION_AND_RETRY);
427         } finally {
428             if (uriRequest != null) {
429                 uriRequest.releaseConnection();
430             }
431         }
432     }

442     public Map<String, Object> mapXMLContent(HttpEntity responseEntity) throws SnapDataException {
443         try (InputStream is = responseEntity.getContent()) {
444             if (is == null) {
445                 throw new SnapDataException(Messages.AUTHENTICATION_ERROR)
446                         .withReason(Messages.ACCOUNT_ENTITY_INPUT_STREAM_IS_MISSING)
447                         .withResolution(Messages.VERIFY_ACCOUNT_CONFIGURATION_AND_RETRY);
448             }
449             Header contentTypeHeader = responseEntity.getContentType();
450             if (contentTypeHeader == null || !contentTypeHeader.getValue().contains(XML)) {
451                 throw new SnapDataException(Messages.AUTHENTICATION_ERROR)
452                         .withReason(Messages.ACCOUNT_CONTENT_NOT_XML)
453                         .withResolution(Messages.VERIFY_ACCOUNT_CONFIGURATION_AND_RETRY);
454             }
455             Map<String, Object> xmlConvertedContent = (Map<String, Object>)
456                     xmlUtils.convertToJson(xmlInputFactory, jsonOutputFactory,
457                             new StreamSource(is));
458             return xmlConvertedContent;
459         } catch (IOException | XMLStreamException e) {
460             throw new SnapDataException(Messages.UNEXPECTED_XML_IN_ACCOUNT_CONTENT)
461                     .withReason(e.getMessage())
462                     .withResolution(Messages.VERIFY_ACCOUNT_CONFIGURATION_AND_RETRY);
463         }
464     }

466     public HttpEntity getHttpEntity() throws InvalidPathException {
467         Map<String, Object> inputValues = Maps.newLinkedHashMap();
468         for (Map.Entry<String, JsonPath> mapEntry : AUTH_INPUT_REQD.entrySet()) {
469             if (mapEntry.getKey().equals(SITE) &&
470                     propertyValues.get(SITE) == null) {
471                 mapEntry.getValue().write(inputValues, StringUtils.EMPTY);
472             } else {
473                 mapEntry.getValue().write(inputValues, propertyValues.get(mapEntry.getKey()));
474             }
475         }
476         String payload = APIUtils.evaluateTableauSchema(getSchema(getServerVersion()), inputValues,
477                 xmlTemplateEvaluator, xmlUtils, documentBuilderFactory, domUtil, typesFactoryImpl);
478         return new StringEntity(payload, UTF_8);
479     }

546     private void verifyS3Credentials() {
547         try {
548             AmazonS3Client client = new AmazonS3Client(
549                     new BasicAWSCredentials(accessKeyID, accessKeySecret),
550                     DEFAULT_CLIENT_CONFIGURATION);
551             // the statement below will throw error if the credentials are invalid
