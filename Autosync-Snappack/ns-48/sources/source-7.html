152     private Pair<Schema, Type> findSchema(QName typeQname) {
153         for (Schema schema : types.getSchemas()) {
154             Type type = schema.getType(typeQname);
155             if (type == null) {
156                 Element elem = schema.getElement(typeQname);
157                 if (elem == null) {
158                     continue;
159                 }
160                 type = elem.getType();
161                 if (type == null) {
162                     continue;
163                 }
164             }
165             return Pair.of(schema, type);
166         }
167         String message = String.format(NO_TYPE_FOUND_FOR_TYPE_QNAME, typeQname.toString());
168         throw new ExecutionException(message)

438                 } else {
439                     // TODO: Handle value types as required
440                     LOGGER.debug(UNKNOWN_VALUE_TYPE_ENCOUNTERED, value.getClass());
441                 }
442             }
443         }
444     }

459      * Normalizes the localPart : i.e. replaces any '.' with '_', finds the
460      * index of the first alphabet and returns the substring of localPart,
461      * beginning at that index
462      *
463      * @param localPart
464      * @return a normalized localPart
465      */
466     private String normalizeLocalPart(String localPart) {
467         localPart = localPart.replace(DOT, UNDERSCORE);
468         char[] chars = localPart.toCharArray();
469         int indexOfFirstAlphabet = 0;
470         for (int i = 0; i < chars.length; i++) {
471             if (Character.isUpperCase(chars[i]) || Character.isLowerCase(chars[i])) {
472                 indexOfFirstAlphabet = i;
473                 break;
474             }
475         }
476         return localPart.substring(indexOfFirstAlphabet);
477     }

479     private QName createQNameFrom(AbstractSchemaElementImpl parent,
480             String elementName) {
481         String namespaceURI;
482         String prefix;
483         if (parent instanceof ComplexType) {
484             if (((ComplexTypeImpl) parent).getQName() == null) {
485                 Pair<String, String> namespacePrefixPair = findNamespaceAndPrefixFromChild(
486                         (ComplexTypeImpl) parent);
487                 namespaceURI = namespacePrefixPair.getLeft();
488                 prefix = namespacePrefixPair.getRight();
489             } else {
490                 namespaceURI = ((ComplexTypeImpl) parent).getQName().getNamespaceURI();
491                 prefix = ((ComplexTypeImpl) parent).getQName().getPrefix();
492             }
493         } else if (parent instanceof SimpleType) {
494             namespaceURI = ((SimpleTypeImpl) parent).getQName().getNamespaceURI();
495             prefix = ((SimpleTypeImpl) parent).getQName().getPrefix();
496         } else {
497             LOGGER.debug(UNKNOWN_PARENT_TYPE_ENCOUNTERED, parent.getClass());
498             return null;
499         }
500         if (StringUtils.isEmpty(prefix)) {
501             prefix = namespaceToPrefixMap.get(namespaceURI);
502         }
503         if (StringUtils.isEmpty(prefix)) {
504             if (namespaceToCreatedPrefixMap.containsKey(namespaceURI)) {
505                 prefix = namespaceToCreatedPrefixMap.get(namespaceURI);
506             } else {
507                 prefix = DEFAULT_QNAME_PREFIX + counter++;
508                 namespaceToCreatedPrefixMap.put(namespaceURI, prefix);
509             }
510         }
511         return new QName(namespaceURI, elementName, prefix);
512     }

747     private void populateNamespaceMaps() {
748         Set<String> visitedNamespaces = new HashSet<>();
749         for (Schema schema : types.getSchemas()) {
750             if (!visitedNamespaces.contains(schema.getTargetNamespace())) {
751                 NamespaceMapperImpl namespaceMapper = schema.getAllNamespaces();
752                 Map<String, String> prefixNamespaceMap = namespaceMapper.getNamespaces();
753                 for (Map.Entry<String, String> entry : prefixNamespaceMap.entrySet()) {
754                     namespaceToPrefixMap.put(entry.getValue(), entry.getKey());
755                 }
756                 populateNamespaceMapsFrom(schema, visitedNamespaces);
757                 visitedNamespaces.add(schema.getTargetNamespace());
758             }
759         }
760     }

762     private void populateNamespaceMapsFrom(Schema schema, Set<String> visitedNamespaces) {
763         for (Import imported : schema.getImports()) {
764             if (!visitedNamespaces.contains(imported.getNamespaceURI())) {
765                 if (imported.getSchema() != null) {
766                     NamespaceMapperImpl namespaceMapper = imported.getSchema().getAllNamespaces();
767                     Map<String, String> prefixNamespaceMap = namespaceMapper.getNamespaces();
768                     for (Map.Entry<String, String> entry : prefixNamespaceMap.entrySet()) {
769                         namespaceToPrefixMap.put(entry.getValue(), entry.getKey());
770                     }
771                     visitedNamespaces.add(imported.getNamespaceURI());
772                     populateNamespaceMapsFrom(imported.getSchema(), visitedNamespaces);
773                 }
774             }
775         }
