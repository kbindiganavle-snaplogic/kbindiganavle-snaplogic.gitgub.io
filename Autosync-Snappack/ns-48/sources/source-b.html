21 /**
22  * This interceptor intercepts the inbound NetSuite message and sets the
23  * Content-Type as text/xml since NetSuite returns a response which is a valid
24  * XML but has text/html Content-Type.
25  *
26  * @author slad@snaplogic.com (Samarth Lad)
27  */
28 public class NetSuiteResponseInterceptor extends AbstractPhaseInterceptor<Message> {
29     private static final String INBOUND_MESSAGE_NULL = "INBOUND_MESSAGE property is null";
30     private static final String TEXT_XML = "text/xml";
31 
32     public NetSuiteResponseInterceptor() {
33         super(Phase.POST_STREAM);
34         addBefore(StaxInInterceptor.class.getName());
35     }
36 
37     public void handleMessage(Message message) {
38         Object inBound = message.get(Message.INBOUND_MESSAGE);
39         Preconditions.checkArgument(inBound != null, INBOUND_MESSAGE_NULL);
40 
41         Boolean isInBound = (Boolean) inBound;
42         if (isInBound) {
43             message.put(Message.CONTENT_TYPE, TEXT_XML);
44         }
45     }
46 }
