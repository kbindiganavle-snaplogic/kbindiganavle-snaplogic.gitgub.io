379     private void processHttpCall(Document document,
380                                  HttpSnapExecutor.HttpRequestConfig requestConfig)
381             throws HttpCoreDataException, HttpCoreConnectionException {
382         if (ViewType.DOCUMENT.equals(config.getResponseOutputView().getType())) {
383             Map<String, Object> response = executor.execute(requestConfig);
384             Document output = documentUtility.newDocument(response);
385             // Pagination - get nextPageConf before writing output document to the output view in
386             // case that Snap like Mapper is in downstream which modifies the document
387             HttpSnapExecutor.HttpRequestConfig nextPageConf =
388                     config.nextPageConfig(output, document);
389             while (nextPageConf != null) {
390                 writeResponseView(output, document);
391                 output = documentUtility.newDocument(executor.execute(nextPageConf));
392                 nextPageConf = config.nextPageConfig(output, document);
393             }
394             writeResponseView(output, document);
395         } else {
396             config.getBinaryEntityHandler().setOriginal(document);
397             // For Binary outputView, the entity is handled by the binary handler
398             executor.execute(requestConfig);
399         }
400     }

1353         private List<Pair<String, String>> overridePairList(List<Pair<String, String>> pairList,
1354                                                             Map<String, Object> overrideMap) {
1355             List<Pair<String, String>> newList = new ArrayList<>();
1356             if (pairList != null) {
1357                 for (Pair<String, String> pair : pairList) {
1358                     if (overrideMap.containsKey(pair.getKey())) {
1359                         // Override the parameter if both key and value is present. If the
1360                         // value itself is empty, the key should be removed
1361                         String val = String.valueOf(overrideMap.get(pair.getKey()));
1362                         if (overrideMap.get(pair.getKey()) != null && StringUtils.isNotEmpty(val)) {
1363                             newList.add(Pair.of(pair.getKey(), val));
1364                         }
1365                         // remove it from the overrideMap to avoid duplicate insertion
1366                         overrideMap.remove(pair.getKey());
1367                     } else {
1368                         // Otherwise add the element from the original list to the new list
1369                         newList.add(pair);
1370                     }
1371                 }
1372             }
1373             if (!overrideMap.isEmpty()) {
1374                 // Add the elements that is not in the original list but in the overrideMap
1375                 for (Map.Entry<String, Object> entry : overrideMap.entrySet()) {
1376                     newList.add(Pair.of(entry.getKey(), String.valueOf(entry.getValue())));
1377                 }
1378             }
1379             return newList;
1380         }
