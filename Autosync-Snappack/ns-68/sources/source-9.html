24 /**
25  * Collects the JSON schema information from <code>XSParticle</code>
26  *
27  * @author slad@snaplogic.com (Samarth Lad)
28  */
29 public class Particle implements SchemaCollectable<XSParticle> {
30     @Override
31     public void collect(SchemaCollectionContext context, SchemaCollector collector,
32             XSParticle xsParticle, SchemaVisitor visitor) {
33         XSTerm xsTerm = xsParticle.getTerm();
34         if (xsTerm.isModelGroup()) {
35             visitor.visit(xsTerm.asModelGroup());
36         } else if (xsTerm.isModelGroupDecl()) {
37             visitor.visit(xsTerm.asModelGroupDecl());
38         } else if (xsTerm.isElementDecl()) {
39             context.storeXSTermIfRepeated(xsParticle);
40             visitor.visit(xsTerm.asElementDecl());
41         } else if (xsTerm.isWildcard()) {
42             visitor.visit(xsTerm.asWildcard());
43         } else {
44             throw new ExecutionException(String.format(INVALID_XSD_TERM_ENCOUNTERED,
45                     xsTerm.getClass()));
46         }
47     }
48 }
