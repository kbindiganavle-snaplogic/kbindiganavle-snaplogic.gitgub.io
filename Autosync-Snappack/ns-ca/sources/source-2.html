262     protected DefaultRedshiftFileSchemaDetector(final Builder builder) {
263         super();
264         LOGGER = builder.LOGGER;
265         cloudStorageType = builder.cloudStorageType;
266         s3Client = builder.s3Client;
267         inFilePath = builder.inFilePath;
268         s3Bucket = builder.s3Bucket;
269         inFileKey = builder.inFileKey;
270         awsAccessKey = builder.awsAccessKey;
271         awsSecretKey = builder.awsSecretKey;
272         fileFormatType = builder.fileFormatType;
273         fieldDelimiter = builder.fieldDelimiter;
274         dateFormat = builder.dateFormat;
275         quoteCharacter = builder.quoteCharacter;
276         encoding = builder.encoding;
277         escape = builder.escape;
278         ignoreHeader = builder.ignoreHeader;
279         nullAsStr = builder.nullAsStr;
280         timeFormat = builder.timeFormat;
281         escapeCharacter = builder.escapeCharacter;
282         preserveSurroundingSpaces = builder.preserveSurroundingSpaces;
283         checkUnknownType = builder.checkUnknownType;
284         compression = builder.compression;
285         doNotThrowException = builder.doNotThrowException;
286 
287         // Set default value to "fieldDelimiter" if a value is not given.
288         if (fieldDelimiter == Character.MIN_VALUE) {
289             fieldDelimiter = StringUtils.equalsIgnoreCase(fileFormatType, CSV) ?
290                     REDSHIFT_DEFAULT_CSV_DELIMITER :
291                     REDSHIFT_DEFAULT_NONCSV_DELIMITER;
292         }
293 
294         // Check if the given dataFormat and timeFormat are valid and supported
295         // for schema inference.
296         if (!EltRedshiftTypeInferenceUtils.isThisDateFormatSupported(dateFormat)) {
297             EltRedshiftTypeInferenceUtils.throwUnsupportedDateFormatException(dateFormat);
298         }
299 
300         if (!EltRedshiftTypeInferenceUtils.isThisTimeFormatSupported(timeFormat)) {
301             EltRedshiftTypeInferenceUtils.throwUnsupportedTimeFormatException(timeFormat);
302         }
303 
304         // Build CloudStorage
305         switch(cloudStorageType) {
306             case CLOUD_STORAGE_AWS_S3:
307                 if (StringUtils.equals(CSV, fileFormatType) ||
308                         StringUtils.equals(JSON, fileFormatType)) {
309                     cloudStorage = new AwsS3CloudStorage(
310                             s3Client, inFilePath, s3Bucket, inFileKey);
311                 } else {
312                     cloudStorage = new AwsS3CloudStorage(
313                             inFilePath, s3Bucket, awsAccessKey, awsSecretKey);
314                 }
315                 break;
316             default:
317                 EltUtils.throwInvalidCloudStorageTypeException(cloudStorageType);
318         }
319 
320         // Prepary type map and build DetectorInputFile
321         DetectorInputFileConfig detectorInputFileConfig;
322         typePatternMap =
323                 EltRedshiftTypeInferenceUtils
324                         .buildTypePatternMap(nullAsStr, dateFormat, timeFormat);
325 
326         switch(fileFormatType) {
327             case CSV:
328                 detectorInputFileConfig = new DetectorInputFileConfig(
329                         encoding, ignoreHeader,
330                         fieldDelimiter, quoteCharacter, checkUnknownType,
331                         preserveSurroundingSpaces,
332                         doNotThrowException);
333                 inputFile =
334                         new CSVDetectorInputFileImpl(detectorInputFileConfig, REDSHIFT, LOGGER);
335                 break;
336             case JSON:
337                 detectorInputFileConfig = new DetectorInputFileConfig(
338                         encoding, ignoreHeader,
339                         quoteCharacter,
340                         checkUnknownType, preserveSurroundingSpaces,
341                         doNotThrowException);
342                 inputFile =
343                         new JSONDetectorInputFileImpl(detectorInputFileConfig, REDSHIFT, LOGGER);
344                 break;
345             case PARQUET:
346                 // Build Parquet file column type to Redshift column type map
347                 parquetColumnTypeMap =
348                         EltRedshiftTypeInferenceUtils.buildParquetColumnTypeMap();
349                 detectorInputFileConfig = new DetectorInputFileConfig(
350                         encoding,
351                         quoteCharacter,
352                         REDSHIFT_VARCHAR_COLUMN_TYPE_DEFAULT,
353                         checkUnknownType,
354                         doNotThrowException);
355                 inputFile =
356                         new ParquetDetectorInputFileImpl(detectorInputFileConfig, REDSHIFT, LOGGER);
357                 break;
358             case ORC:
359                 orcColumnTypeMap =
360                         EltRedshiftTypeInferenceUtils.buildOrcColumnTypeMap();
361                 detectorInputFileConfig = new DetectorInputFileConfig(
362                         encoding,
363                         quoteCharacter,
364                         REDSHIFT_VARCHAR_COLUMN_TYPE_DEFAULT,
365                         checkUnknownType,
366                         doNotThrowException);
367                 inputFile =
368                         new ORCDetectorInputFileImpl(detectorInputFileConfig, REDSHIFT, LOGGER);
369                 break;
370             case AVRO:
371                 avroColumnTypeMap =
372                         EltRedshiftTypeInferenceUtils.buildAvroColumnTypeMap();
373                 detectorInputFileConfig = new DetectorInputFileConfig(
374                         encoding,
375                         quoteCharacter,
376                         REDSHIFT_VARCHAR_COLUMN_TYPE_DEFAULT,
377                         checkUnknownType,
378                         doNotThrowException);
379                 inputFile =
380                         new AVRODetectorInputFileImpl(detectorInputFileConfig, REDSHIFT, LOGGER);
381                 break;
382             default:
383                 throwInvalidFileFormatTypeException(fileFormatType);
384                 break;
385         }
