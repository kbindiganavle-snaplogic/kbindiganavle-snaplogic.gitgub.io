89     /**
90      * Merge sourceColumnMap's values() of the given "other" DetectResults into "this".
91      * This also updates isDataTypeListComplete of "this" accordingly.
92      *
93      * @param other Another SchemaDetectionResult
94      * @precondition "this" and "other" has the same keySets for sourceColumnMap's. I.e.,
95      *              the column names are equal.
96      */
97     public void merge(SchemaDetectionResult other) {
98         if (isDataTypeListComplete) {
99             // Nothing to to. Just return.
100             return;
101         }
102 
103         if (other == null || this.equals(other)) {
104             return;
105         }
106 
107         if (!sourceColumnMap.keySet().equals(other.sourceColumnMap.keySet())) {
108             areColumnNamesNotMatchingAmongFiles = true;
109             return;
110         }
111 
112         boolean isComplete = true;
113         for (Map.Entry<String,String> e : sourceColumnMap.entrySet()) {
114             if (isUnknownColumnType(e.getValue())) {
115                 sourceColumnMap.put(e.getKey(), other.sourceColumnMap.get(e.getKey()));
116                 if (isComplete &&
117                         isUnknownColumnType(other.sourceColumnMap.get(e.getKey()))) {
118                     isComplete = false;
119                 }
120             }
121         }
122         if (isComplete) {
123             isDataTypeListComplete = isComplete;
124         }
125     }
126 
127     /**
