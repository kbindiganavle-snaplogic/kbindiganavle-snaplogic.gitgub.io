473     protected void processDocument(final Document document, final String inputViewName) {
474         evalRetryProperties(document);

476                 ((String) schemaNameAsExpression.eval(document));
477         packageName = jdbcOperations.normalizeAndEscape

479         try {
480             storedProcName = jdbcOperations.normalizeAndEscape
481                     ((String) storedProcNameAsExpression.eval(document));
482         } catch (SnapDataException sde) {
483             throw new ExecutionException(sde, sde.getMessage())

490         Set<String> seqs = getProcedureSequences(
491                 schemaName,
492                 packageName,
493                 storedProcName);
494         if ((CollectionUtils.isEmpty(seqs) && StringUtils.isNotBlank(procedureSeq))) {
495             throw new ConfigurationException(PROCEDURE_SEQUENCE_NUMBER_ERR_MSG)
496                     .withReason(NON_OVERLOADED_PROCEDURE_ERR_REASON)
497                     .withResolution(NON_OVERLOADED_PROCEDURE_ERR_RESOLUTION);
498         } else if (!CollectionUtils.isEmpty(seqs) && !seqs.contains(procedureSeq)) {
499             throw new ConfigurationException(PROCEDURE_SEQUENCE_NUMBER_ERR_MSG)
500                     .withReason(PROCEDURE_SEQUENCE_OUT_OF_BOUND_ERR_REASON)
501                     .withResolution(PROCEDURE_SEQUENCE_OUT_OF_BOUND_ERR_RESOLUTION);
502         }

504         List<Map<String, Object>> paramValuesList = propertyValues.get(PARAMETER_TABLE_PATH);
505         expressionProps = Lists.newArrayListWithCapacity(paramValuesList.size());
506         for (Map<String, Object> paramValue : paramValuesList) {
507             final String expression = JsonPathUtil.checkedRead(EXPRESSION_PROPERTY_PATH,
508                     paramValue);
509             final Boolean isExpression = JsonPathUtil.checkedRead(EXPRESSION_PATH, paramValue);
510             ExpressionProperty prop = expressionUtils.createExpressionProperty(propertyValues,
511                     isExpression, expression);
512             expressionProps.add(prop);
513         }
514         parameterTypeMapping = new ArrayList<>();
515         List<Map<String, Map<String, Object>>> customTypesInfo = propertyValues
516                 .get(CUSTOM_TYPE_PARAMS_INFO);
517         if (CollectionUtils.isNotEmpty(customTypesInfo)) {
518             // Pass the procedure name that doesn't contain package or schema name
519             int paramsCount = getParamsCount(
520                     schemaName,
521                     packageName,
522                     storedProcName,
523                     procedureSeq);
524             for (int i = 0; i <= paramsCount; ++i) {
525                 parameterTypeMapping.add(null);
526             }
527             for (Map<String, Map<String, Object>> typeInfo : customTypesInfo) {
528                 Map<String, Object> typeInfoIndex = typeInfo.get(CUSTOM_TYPE_PARAM_INDEX);
529                 int location;
530                 if (typeInfoIndex != null) {
531                     Object index = typeInfoIndex.get(VALUE);
532                     if (index != null) {
533                         location = Integer.parseInt((String) index);
534                         if (location > paramsCount) {
535                             throw new ConfigurationException(INDEX_INVALID_ERR_MSG)
536                                     .withReason(INDEX_INVALID_ERR_REASON)
537                                     .withResolution(INDEX_INVALID_ERR_RESOLUTION);
538                         }
539                     } else {
540                         throw new ConfigurationException(TYPE_INDEX_ERR_MSG)
541                                 .withReason(TYPE_INDEX_ERR_REASON)
542                                 .withResolution(TYPE_INDEX_ERR_RESOLUTION);
543                     }
544                 } else {
545                     throw new ConfigurationException(TYPE_INDEX_ERR_MSG)
546                             .withReason(TYPE_INDEX_ERR_REASON)
547                             .withResolution(TYPE_INDEX_ERR_RESOLUTION);
548                 }
549                 Map<String, Object> typeInfoName = typeInfo.get(CUSTOM_TYPE_PARAM_TYPE);
550                 if (typeInfoName != null) {
551                     String name = jdbcOperations.normalizeAndEscape(
552                             (String) typeInfoName.get(VALUE));
553                     if (StringUtils.isNotBlank(schemaName)) {
554                         String[] nameParts = quotationHandler.splitFullyQualifiedName(name, 0);
555                         if (nameParts != null && nameParts.length == 1) {
556                             // Generate fully qualified type name
557                             name = String.format
558                                     (TYPE_NAME_FMT, quotationHandler.escape(schemaName),
559                                             quotationHandler.escape(name));
560                         }
561                     }
562                     if (name != null) {
563                         if (parameterTypeMapping.get(location) != null) {
564                             throw new ConfigurationException(TYPE_NAME_ERR_MSG)
565                                     .withReason(TYPE_MAPPING_ERR_REASON)
566                                     .withResolution(TYPE_MAPPING_ERR_RESOLUTION);
567                         } else {
568                             parameterTypeMapping.remove(location);
569                             parameterTypeMapping.add(location, name);
570                         }
571                     } else {
572                         throw new ConfigurationException(TYPE_NAME_ERR_MSG)
573                                 .withReason(TYPE_NAME_ERR_REASON)
574                                 .withResolution(TYPE_NAME_ERR_RESOLUTION);
575                     }
576                 } else {
577                     throw new ConfigurationException(TYPE_NAME_ERR_MSG)
578                             .withReason(TYPE_NAME_ERR_REASON)
579                             .withResolution(TYPE_NAME_ERR_RESOLUTION);
580                 }
581             }
582         }
583         List<Object> evaluatedParamValues = Lists.newArrayListWithCapacity(expressionProps.size());
584         for (ExpressionProperty property : expressionProps) {
585             Object value = property.eval(document);
586             evaluatedParamValues.add(value);
587         }
588 
589         if (evaluatedParamValues.isEmpty() && document != null) {
590             procedureMetadata = jdbcOperations.getCallableStatementMetaData(fullyQualifiedName(),
591                     account, parameterTypeMapping, procedureSeq);
592             // The parameter names are case insensitive
593             Map<String, Object> documentData =
594                     new CaseInsensitiveMap(document.get(Map.class));
595             procedureMetadata
596                     .stream()
597                     // All IN/INOUT parameters
598                     .filter(parameterMetaData ->
599                             isInputParamColumnType(parameterMetaData.getColumnType()))
600                     .forEachOrdered(parameterMetaData ->
601                             evaluatedParamValues.add(documentData.get(parameterMetaData.
602                                     getColumnName())));
603         }
604         CallableStatementWrapper stmt;
605         try {
606             // TODO - MK: fix casting in september platform release, also move all sql interfaces
607             // from jsdk to jsql. ImmutableschemaProperty has a reference to JdbcOperations
608             // .DuplicateKeyAction. that needs to be cleaned up.
609             if (JdbcOperations.SQLSERVER.equals(getName()) && retryInterval >= 1) {
610                 stmt = ((JdbcOperationsImpl) jdbcOperations)
611                         .executeCallableStatement(fullyQualifiedName(), evaluatedParamValues,
612                                 account, parameterTypeMapping, procedureSeq, maxRetries,
613                                 retryInterval);
614             } else {
615                 stmt = ((JdbcOperationsImpl) jdbcOperations)
616                         .executeCallableStatement(fullyQualifiedName(), evaluatedParamValues,
617                                 account, parameterTypeMapping, procedureSeq);
618             }
619 
620         } catch (SnapDataException e) {
621             // thrown by CustomTypeConverter
622             errorViews.write(e, document);
623             return;
624         } catch (OracleExecutionException e) {
625             SnapDataException ex = new SnapDataException(e, e.getMessage())
626                     .withReason(e.getCause().getMessage())
627                     .withResolution(PLEASE_ADDRESS_THE_REPORTED_ISSUE);
628             errorViews.write(ex, document);
629             return;
630         }
631         try {
632             if (JdbcOperations.SQLSERVER.equals(getName()) ||
633                     JdbcOperations.AZURE_SYNAPSE.equals(getName())) {
634                 // Get result sets at first for SQL Server because stmt.getObject() which is
635                 // called while getting output values would corrupt result sets
636                 if (retryInterval >= 1) {
637                     RetryPolicy retryPolicy = DatabaseUtils.getRetryPolicy(maxRetries,
638                             retryInterval);
639                     DatabaseConfig databaseConfig = account.getDatabaseConfig();
640                     SQLDialect dialect = databaseConfig.getSQLDialect();
641                     Failsafe.with(retryPolicy)
642                             .onFailedAttempt(e -> LOGGER.warn("failed to get result set data: {}",
643                                     e.getMessage()))
644                             .onRetry((result, failure, context) -> {
645                                 jdbcOperations.logRetryAttempt(context, LOGGER);
646                                 jdbcOperations.handleOnRetry(dialect.getName(), failure);
647                                 jdbcOperations.acquireConnection(account);
648                             })
649                             .run(cntx -> {
650                                 stmt.processResultSet(resultSetOutputView, documentUtility,
651                                         document, passThrough);
652                                 paramOutputView.write(generateOutputDoc(stmt, document));
653                             });
654                 } else {
655                     stmt.processResultSet(resultSetOutputView, documentUtility, document,
656                             passThrough);
657                     paramOutputView.write(generateOutputDoc(stmt, document));
658                 }
659             } else {
660                 paramOutputView.write(generateOutputDoc(stmt, document));
661                 stmt.processResultSet(resultSetOutputView, documentUtility, document,
662                         passThrough);
663             }
664         } catch (SQLException e) {
665             SnapDataException ex = new SnapDataException(e, READ_RESULTS_FAILED)
666                     .withReason(e.getMessage())
667                     .withResolution(PLEASE_ADDRESS_THE_REPORTED_ISSUE);
668             errorViews.write(ex, document);
669         } catch (SnapDataException e) {
670             errorViews.write(e, document);
671         } finally {
672             JDBCUtils.safeClose(stmt);
673         }
674     }

697     protected void validateName() {
698         if (StringUtils.isEmpty(storedProcName)) {
699             throw new ConfigurationException(EMPTY_PROCEDURE_ERR_MSG)
700                     .withReason(EMPTY_PROCEDURE_ERR_REASON)
701                     .withResolution(EMPTY_PROCEDURE_ERR_RESOLUTION);
702         }
703         String rawSchemaName = schemaName;
704         String rawPackageName = packageName;
705         String[] segments = quotationHandler.splitFullyQualifiedName(storedProcName, 0);
706         String[] normalizedSegments = new String[segments.length];
707         for (int i = 0; i < normalizedSegments.length; i++) {
708             normalizedSegments[i] = jdbcOperations.normalizeAndEscape(segments[i]);
709         }
710         if (StringUtils.isNotEmpty(rawSchemaName) || StringUtils.isNotEmpty(packageName)) {
711             if (normalizedSegments.length == 3) {
712                 if (!normalizedSegments[0].equals(rawSchemaName)
713                         || (StringUtils.isNotEmpty(rawPackageName)
714                         && !normalizedSegments[1].equals(rawPackageName))) {
715                     throw new ConfigurationException(PROCEDURE_NAME_MISMATCH_ERR_MSG)
716                             .withReason(PROCEDURE_NAME_MISMATCH_ERR_REASON)
717                             .withResolution(PROCEDURE_NAME_MISMATCH_ERR_RESOLUTION);
718                 }
719                 schemaName = normalizedSegments[0];
720                 packageName = normalizedSegments[1];
721             } else if (normalizedSegments.length == 2) {
722                 if ((StringUtils.isNotEmpty(rawSchemaName) &&
723                         StringUtils.isNotEmpty(rawPackageName))
724                         || (StringUtils.isNotEmpty(rawSchemaName + rawPackageName) &&
725                         !normalizedSegments[0].equals(rawSchemaName) &&
726                         !normalizedSegments[0].equals(rawPackageName))) {
727                     throw new ConfigurationException(PROCEDURE_NAME_MISMATCH_ERR_MSG)
728                             .withReason(PROCEDURE_NAME_MISMATCH_ERR_REASON)
729                             .withResolution(PROCEDURE_NAME_MISMATCH_ERR_RESOLUTION);
730                 }
731                 if (StringUtils.isNotEmpty(rawSchemaName)) {
732                     schemaName = normalizedSegments[0];
733                 }
734                 if (StringUtils.isNotEmpty(rawPackageName)) {
735                     packageName = normalizedSegments[0];
736                 }
737             }
738             storedProcName = normalizedSegments[segments.length - 1];
739         } else {
740             if (segments.length == 3) {
741                 schemaName = normalizedSegments[0];
742                 packageName = normalizedSegments[1];
743                 storedProcName = normalizedSegments[2];
744             } else if (segments.length == 2) {
745                 schemaName = normalizedSegments[0];
746                 packageName = null;
747                 storedProcName = normalizedSegments[1];
748             } else {
749                 schemaName = null;
750                 packageName = null;
751             }
752         }
753     }
