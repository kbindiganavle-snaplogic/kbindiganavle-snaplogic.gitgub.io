204                         errorList.add(new BatchElementExecutionResult(snapDataException,

207                     }
208                 } else {

210                     result.put(SUCCESSFUL_DOCUMENTS, results);
211                 }
212             }

219         /**
220          * creates the result list, representing the structure under error or successful documents
221          *
222          * @param batchElemExecutionResults the result list
223          *
224          * @return the object list
225          */
226         @SuppressWarnings("unchecked")
227         protected List<Map<String, Object>> createResultList(List<BatchElementExecutionResult>
228                 batchElemExecutionResults) {
229             List<Map<String, Object>> results = Lists.newArrayList();
230             for (BatchElementExecutionResult batchElemExecutionResult : batchElemExecutionResults) {
231                 Map<String, Object> result = Maps.newLinkedHashMap();
232                 Object leftValue = batchElemExecutionResult.getResult();
233                 if (leftValue instanceof Document) {
234                     // copy the output document
235                     Object content = ((Document) leftValue).get();
236                     if (content instanceof Map) {
237                         result.putAll((Map) content);
238                     } else if (content != null) {
239                         result.put(DOCUMENT, content);
240                     }
241                 } else if (leftValue instanceof SnapDataException) {
242                     SnapDataException exception = (SnapDataException) leftValue;
243                     Document errorDocument = exception.getErrorDocument();
244                     Object errObj = errorDocument != null ? errorDocument.get() : null;
245                     if (errObj instanceof Map) {
246                         result.putAll((Map) errObj);
247                     } else if (errObj != null) {
248                         // write anything else under error key
249                         result.put(ERROR, errObj);
250                     } else {
251                         // transform the exception into an error map if we don't have a document
252                         result.putAll(createErrorMap(exception));
253                     }
254                 } else {
255                     throw new UnsupportedOperationException(RESULT_TYPE_NOT_SUPPORTED);
256                 }
257                 // attach the original
258                 result.put(ORIGINAL, batchElemExecutionResult.getInput());
259                 results.add(result);
260             }
261             return results;
262         }

569     @Override
570     protected void processDocument(Document document, String inputViewName) {
571         StatementsExecutionResult statementsExecutionResult = new StatementsExecutionResult();
572         int index = 0;
573         for (ExecutionRow executionRow : executions) {
574             ExpressionProperty pathExpression = executionRow.getExpressionProperty();
575             Object value = pathExpression.eval(document);
576             String transformedSql = executionRow.getSqlStatement();
577             try {
578                 Number temp = maxRetryProperty.eval(document);
579                 maxRetries = temp == null ? 0 : temp.intValue();
580                 temp = retryIntervalProperty.eval(document);
581                 retryInterval = temp == null ? 1 : temp.longValue();
582                 List<JsonPath> variablePaths = executionRow.getVariablePaths();
583                 if (value instanceof Collection) {
584                     for (Object entry : ((Collection) value)) {
585                         statementsExecutionResult.merge(processStatement(variablePaths, entry,
586                                 transformedSql, index, executionRow.isPlainQuery()));
587                     }
588                 } else if (value instanceof Map) {
589                     statementsExecutionResult.merge(processStatement(variablePaths, value,
590                             transformedSql, index, executionRow.isPlainQuery()));
591                 } else {
592                     SnapDataException snapDataException = new SnapDataException(
593                             DOCUMENT_NOT_OF_TYPE_MAP)
594                             .withReason(DOCUMENT_NOT_OF_TYPE_MAP)
595                             .withResolution(DOCUMENT_NOT_OF_TYPE_MAP_RESOLUTION);
596                     errorViews.write(snapDataException, document);
597                     return;
598                 }
599                 // flush the batch after each execution
600                 flushMultiBatchStatement(statementsExecutionResult, index, transformedSql);
601             } catch (Exception ex) {
602                 SnapDataException snapDataException = new SnapDataException(documentUtility
603                         .newDocument(createErrorMap(ex)), ex, ROLLBACK_MESSAGE)
604                         .withResolution(ROLLBACK_RESOLUTION);
605                 statementsExecutionResult.add(new BatchElementExecutionResult(snapDataException,
606                         value, transformedSql), index);
607             }
608             index++;
609         }
610         if (!statementsExecutionResult.hasFailed()) {
611             // commit only after all executions succeeded for the current document.
612             // write it out
613             jdbcOperations.commit();
614             // write one output document for the entire execution
615             writeResults(document, statementsExecutionResult, outputViews);

769      * Handles results from an update batch.
770      *
771      * @param replacedSql          the transformed SQL (using ? for variables)
772      * @param index                the index of the statement in the executions table
773      * @param rowsModified         rows modified
774      * @param batchUpdateException The exception that was thrown during the update.  The
775      *                             message in this exception will be included in the doc
776      *                             written to the error view.
777      *
778      * @return a wrapper with documents (success) or snap data exceptions (error)
779      *         paired with their input document if any and their statement that was being executed.
780      */
781     protected StatementsExecutionResult handleUpdateResult(String replacedSql, int index,
782             int[] rowsModified, BatchUpdateException batchUpdateException) {
783         // copied from DataBaseUtils.handleUpdateResult(...), here we add each success/failure to
784         // a result list so that we can process it when we commit/rollback the transaction
785         if (rowsModified != null) {
786             StatementsExecutionResult statementsExecutionResult = new StatementsExecutionResult();
787             // If there was an exception during batch processing, there will be fewer rowsModified
788             // values than inputDocuments.
789             if (hasAtLeastSameResultsAsInputDocs && rowsModified.length > inputDocuments.size()) {
790                 throw new ExecutionException(DOCS_NOT_EQUAL_TO_BATCH_SIZE_MSG)
791                         .withReason(DOCS_NOT_EQUAL_TO_BATCH_SIZE)
792                         .withResolutionAsDefect();
793             }
794             int exceptionIndex = -1;
795             if (batchUpdateException != null) {
796                 // MySQL (at least) throws the exception for the last error it encountered, so
797                 // we need to find the index of the last error.  This will need more tweaking
798                 // as other DBs are tested.
799                 for (int i = rowsModified.length - 1; i >= 0; i--) {
800                     if (rowsModified[i] == Statement.EXECUTE_FAILED) {
801                         exceptionIndex = i;
802                         break;
803                     }
804                 }
805             }
806             for (int i = 0; i < rowsModified.length; i++) {
807                 Object doc = inputDocuments.get(i);
808                 Map<String, Object> data = Maps.newLinkedHashMapWithExpectedSize(1);
809                 int result = rowsModified[i];
810                 data.put(DatabaseUtils.STATUS, result);
811                 if (result >= 0) {
812                     data.put(DatabaseUtils.MESSAGE, DatabaseUtils.SUCCESS);
813                     statementsExecutionResult.add(new BatchElementExecutionResult(
814                             documentUtility.newDocument(data), doc, replacedSql), index);
815                 } else {
816                     switch (result) {
817                         case Statement.SUCCESS_NO_INFO:
818                             data.put(DatabaseUtils.MESSAGE, SUCCESS_NO_INFO);
819                             statementsExecutionResult.add(new BatchElementExecutionResult(
820                                     documentUtility.newDocument(data), doc, replacedSql), index);
821                             break;
822                         case Statement.EXECUTE_FAILED:
823                             String reason = batchUpdateException != null ?
824                                     DatabaseUtils.getSqlErrorMessage(batchUpdateException) :
825                                     EXECUTE_FAILED;
826                             SnapDataException snapDataException = new SnapDataException(
827                                     SQL_OPERATION_FAILED)
828                                     .withReason(reason)
829                                     .withResolution(RESOLUTION_CHECK_PROPERTIES_INPUT_DATA);
830                             statementsExecutionResult.add(new BatchElementExecutionResult(
831                                     snapDataException, doc, replacedSql), index);
832                             break;
833                         default:
834                             statementsExecutionResult.add(new BatchElementExecutionResult(
835                                     documentUtility.newDocument(data), doc, replacedSql), index);
836                     }
837                 }
838             }
839             if (exceptionIndex == -1 && batchUpdateException != null) {
840                 // If the driver doesn't return an update-count array with EXECUTE_FAILED
841                 // values, then we need to make sure the exception is written to the error
842                 // view.
843                 Object doc = null;
844 
845                 if (rowsModified.length < inputDocuments.size()) {
846                     // The rowsModified array contains only successful inserts, so the
847                     // document with the error is the one after the last success.
848                     doc = inputDocuments.get(rowsModified.length);
849                 }
850                 SnapDataException snapDataException = new SnapDataException(
851                         SQL_OPERATION_FAILED)
852                         .withReason(batchUpdateException.getMessage())
853                         .withResolution(RESOLUTION_CHECK_PROPERTIES_INPUT_DATA);
854                 statementsExecutionResult.add(new BatchElementExecutionResult(snapDataException,
855                         doc, replacedSql), index);
856             }
857             // Clear the input documents because the batch has executed
858             inputDocuments.clear();
859             return statementsExecutionResult;
860         } else if (batchUpdateException != null) {
861             StatementsExecutionResult resultList = new StatementsExecutionResult();
862             // If the driver does not return any update-counts, make sure an error is written.
863             SnapDataException snapDataException = new SnapDataException(
864                     SQL_OPERATION_FAILED)
865                     .withReason(batchUpdateException.getMessage())
866                     .withResolution(RESOLUTION_CHECK_PROPERTIES_INPUT_DATA);
867             resultList.add(new BatchElementExecutionResult(snapDataException, null, replacedSql),
868                     index);
869             return resultList;
870         }
871         return null;
872     }
