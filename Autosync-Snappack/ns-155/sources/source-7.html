64 /**
65  * Executes a given SQL statement.
66  * <p/>
67  * <p>If the statement is a select statement, then the results will be written as documents.
68  * The snap will substitute values from an incoming document into the select statement if it finds
69  * a json path substring in the query. If a substitution value is missing in the incoming document,
70  * then the document will be written to an error view. In addition, results from a query will be

351     protected void processQuery(final Document document, final List<Object> values,
352                                 final String replacedSql) {
353         try {
354             LOG.debug("Executing the SELECT query");
355             long startTime = System.currentTimeMillis();
356             jdbcOperations.acquireConnection(account);
357             SQLDialect dialect = account.getDatabaseConfig().getSQLDialect();
358             RetryPolicy retryPolicy = jdbcOperations.getDefaultRetryPolicy(maxRetries,
359                 retryInterval);
360             if (retryPolicy == null) {
361                 try (DatabaseCursor cursor = jdbcOperations.executeQuery(replacedSql, values,
362                     account, maxRetries, retryInterval)) {
363                     databaseUtils.processSelect(cursor, document, outputViews, documentUtility,
364                         passThrough, ignoreEmptyResult, getQueryWithoutComments(replacedSql),
365                             getName());
366                 }
367             } else {
368                 Failsafe.with(retryPolicy)
369                     .onFailedAttempt(e -> LOG.warn("simple_sql_execute_snap_retry - failed to " +
370                         "processQuery: {}", e.getMessage()))
371                     .onRetry(throwable -> {
372                         jdbcOperations.handleOnRetry(dialect.getName(), throwable);
373                     })
374                     .run(cntx -> {
375                         jdbcOperations.acquireConnection(account);
376                         jdbcOperations.logRetryAttempt(cntx, LOG);
377                         try (DatabaseCursor cursor = jdbcOperations.executeQuery(replacedSql,
378                             values, account, maxRetries, retryInterval)) {
379                             databaseUtils.processSelect(cursor, document, outputViews,
380                                 documentUtility, passThrough, ignoreEmptyResult,
381                                 getQueryWithoutComments(replacedSql), getName());
382                         }
383                     });
384             }
385             long endTime = System.currentTimeMillis();
386             long diff = endTime - startTime;
387             String time = String.valueOf(diff);
388             LOG.debug("DONE executing the SELECT query: {} ms", time);
389             // commit() should be executed if database is PostgreSQL and auto-commit in
390             // the connection is false, when the SELECT query is for a stored procedure [SWAT-675]
391             // To support BEGIN TRAN in Sql server execute, committing the connection
392             if (!jdbcOperations.getAutoCommit() &&
393                     AUTOCOMMIT_OFF_DB_SET.contains(jdbcOperations.getName())) {
394                 jdbcOperations.commit();
395             }
396         } catch (SQLException e) {
397             SnapDataException snapDataException = new SnapDataException(e,
398                     UNABLE_TO_EXECUTE_QUERY)
399                     .formatWith(replacedSql)
400                     .withReason(e.getLocalizedMessage())
401                     .withResolution(PLEASE_ADDRESS_THE_REPORTED_ISSUE);
402             errorViews.write(snapDataException, document);
403         } catch (SuggestViewAbortException | ExecutionException e) {
404             throw e;
405         } catch (SnapDataException e) {
406             errorViews.write(e, document);
407         } catch (Exception e) {
408             errorViews.write((SnapDataException) new SnapDataException(e, EXECUTE_QUERY_FAILED)
409                     .withReason(e.getMessage())
410                     .withResolution(RESOLUTION_ADDRESS_ISSUE), document);
411         }
412     }

462     protected String trimToFirstKeyword(String sql) {
463         while (StringUtils.isNotBlank(sql)) {
464             sql = StringUtils.trim(sql).concat(NEW_LINE);
465             if (StringUtils.startsWithAny(sql, COMMENT_MARKER)) {
466                 int index = sql.indexOf(NEW_LINE);
467                 sql = index > 0 ? sql.substring(index + 1) : sql;
468             } else if (StringUtils.startsWith(sql, COMMENT_START)) {
469                 int index = sql.indexOf(COMMENT_END);
470                 if (index > 0) {
471                     sql = sql.substring(index + 2);
472                 } else {
473                     throw new SnapDataException(ERR_INVALID_COMMENT)
474                             .formatWith(sql)
475                             .withReason(REASON_COMMENT_NOT_CLOSED)
476                             .withResolution(RESOLUTION_ADDRESS_ISSUE);
477                 }
478             } else if (StringUtils.startsWith(sql, PARENTHESES)) {
479                 sql = sql.substring(sql.indexOf(PARENTHESES) + 1);
480             } else {
481                 break;
482             }
483         }
484         return StringUtils.trim(sql);

487     /**
488      * Tells if the given WITH query does read or write operation
489      *
490      * @param query
491      * @return true if read operation, false otherwise
492      */
493     protected boolean isWithQueryRead(String query) {
494         char escapeChar = (char) 0;
495         boolean escapeMode = false;
496         int length;
497         char c;
498         StringBuilder queryWithoutEscapedTableAndColumnNames = new StringBuilder();
499 
500         // replace all new lines with single space
501         query = query.replaceAll(System.lineSeparator(), SINGLE_SPACE);
502         // replace multiple spaces with single space
503         query = query.replaceAll(MULTI_SPACE, SINGLE_SPACE);
504         // convert the query to lowercase
505         query = query.toLowerCase();
506 
507         // remove the table and column names from the query that are escaped using allowed
508         //    database escape characters such as ', ", ` and []
509         // this will remove table and column names containing keywords such as insert, update,
510         //    delete, etc
511         length = query.length();
512         for (int i = 0; i < length; i++) {
513             c = query.charAt(i);
514             if (!escapeMode) {
515                 if (ArrayUtils.contains(ESCAPE_CHARS_START, c)) {
516                     escapeMode = true;
517                     escapeChar = c;
518                 } else {
519                     queryWithoutEscapedTableAndColumnNames.append(c);
520                 }
521             } else {
522                 if (ArrayUtils.contains(ESCAPE_CHARS_END, c) &&
523                         isEscapeCharsMatching(escapeChar, c)) {
524                     escapeMode = false;
525                 }
526             }
527         }
528 
529         // check if the remaining part of the query contains write keyword phrases
530         // if so then its write query otherwise read query
531         String remainingQueryPart = queryWithoutEscapedTableAndColumnNames.toString();
532         for (int i = 0; i < QUERY_WRITE_PHRASES.length; i++) {
533             if (remainingQueryPart.contains(QUERY_WRITE_PHRASES[i])) {
534                 return false;
535             }
536         }
537         return true;
538     }
