91 /**
92  * Executes a simple select on a database.
93  * <p/>
94  * <p>This component supports the following syntax:
95  * SELECT * FROM [table] WHERE [where clause] ORDER BY [ordering] LIMIT [limit] OFFSET [offset]
96  * <p/>
97  * <p>The where clause supports document value substitution using Json Path. If the value does not
98  * exist in the incoming document, then the document will be written to an error view.
99  * <p/>
100  * <p>If more powerful functionality is desired, then the Execute component should be
101  * used.
102  * <p/>
103  * <p>If an input document stream is provided, then values will be extracted from the document
104  * and inserted into the select statement values. After a query is executed, the query's results
105  * will be merged into the incoming document overwriting any existing keys. The original document
106  * will be output if there are no results from the query.
107  *
108  * @author jinloes
109  */

426                 (String) propertyValues.get(DATABASE_NAME_PROP));

435         return tableMetaData.keySet().toArray(new String[0]);

502     private void configureAutoCommit() {
503         if (getAutoCommitPropertyState().equals(AutoCommitPropertyState.DO_NOT_CREATE)) {
504             return;
505         }
506         // init Auto commit property
507         String autoCommitValue = propertyValues.get(AUTO_COMMIT_PROP);
508         if (autoCommitValue == null) {
509             // for production pipeline backward compatibility
510             if (!jdbcOperations.TERADATA.equals(getName())) {
511                 autoCommitOnSnapLevel = Boolean.FALSE;
512             }
513         } else {
514             switch (autoCommitValue) {

543     protected void generateSelectBuilder(final Document document) {
544 
545         // evaluate WHERE clause
546         String whereClause = whereClauseExpr.eval(document);
547         List<JsonPath> variablePaths = null;
548         String queryHints = null;
549         if (StringUtils.isNotEmpty(whereClause)) {
550             try {
551                 EscapeParser escapeParser = new EscapeParser(whereClause,
552                         quotationHandler.getQuotationMarker());
553                 transformedSql = escapeParser.getTransformedStatement();
554                 variablePaths = propertyValues.getJsonPathsFor(escapeParser.getVariablePaths());
555             } catch (SQLException e) {
556                 throw new SnapDataException(INVALID_WHERE_CLAUSE).formatWith(whereClause)
557                         .withReason(e.getMessage())
558                         .withResolution(PLEASE_CORRECT_THE_WHERE_CLAUSE_SYNTAX);
559             }
560         }
561 
562         // retrieve database name if provided
563         databaseName = (String) evaluateName(databaseNameExpr, DATABASE_NAME_LABEL, document);
564         // evaluate the schema and table name
565         schemaName = (String) evaluateName(schemaNameExpr, SCHEMA_NAME_LABEL, document);
566         tableName = (String) evaluateName(tableNameExpr, SELECT_TABLE_NAME_LABEL, document);
567         if (tableName == null) {
568             return;
569         }
570         tableName = updateTableName(databaseName, schemaName, tableName);
571 
572         // evaluation on Limit offset and Limit rows properties can return a BigInteger or a String
573         Object offsetObj = limitOffset.eval(document);
574         BigInteger offset = getIntValue(offsetObj);
575         BigInteger rows = getRows(document);
576         BigInteger rowsOrg = getIntValue(limitRowsOrg.eval(document));
577         if (queryHintsExpr != null) {
578             queryHints = queryHintsExpr.eval(document);
579         }
580         if (rowsOrg != null && rows != null) {
581             // This block runs only if the target DB supports LIMIT clause in the SELECT stmt.
582             // if the Limit rows property is not empty and its value is less than the one replaced
583             // by configureForSuggest(), restore it to its original value
584             rows = rowsOrg.min(rows);
585         }
586         validateOffset(offset);
587         // evaluate 'Order by' expression property
588         List<Map<String, Map<String, Object>>> orderObj =
589                 copyOrderByFieldsObj(propertyValues.get(SELECT_ORDER_BY_PROP));
590         evaluateColumnNames(orderObj, SELECT_ORDER_BY_COLUMNS_PROP, document,
591                 ORDER_BY_COLUMNS_PATH, ERR_ORDER_BY);
592         List<Map<String, Map<String, String>>> order = copyOrderByFields(orderObj);
593         removeExpressionProp(order, SELECT_ORDER_BY_COLUMNS_PROP);
594         // evaluate 'Output Fields' expression property
595         List<Map<String, Map<String, Object>>> fieldsObj =
596                 copyOutputFieldsObj(propertyValues.get(OUTPUT_FIELDS_PROP));
597         evaluateColumnNames(fieldsObj, OUTPUT_FIELD_PROP, document, OUTPUT_FIELD_PATH,
598                 ERR_OUTPUT_FIELDS);
599         List<Map<String, Map<String, String>>> fields = copyOutputFields(fieldsObj);
600         selectBuilder = buildSelect(order, fields, offset, rows, transformedSql,
601                 variablePaths, queryHints, document);
602     }

699         generateSelectBuilder(document);
700         // selectBuilder is null if there's a bad configuration.
701         if (selectBuilder == null) {
702             return;
703         }
704 
705         Number temp = maxRetryProperty.eval(document);
706         maxRetries = temp == null ? 0 : temp.intValue();
707         temp = retryIntervalProperty.eval(document);
708         retryInterval = temp == null ? 1 : temp.longValue();
709 
710         String stagingString = propertyValues.get(STAGING_PROP);
711         if (maxRetries == 0) {
712             stagingState = StagingState.NO_STAGING;
713         } else if (MEMORY_STAGING.equals(stagingString)) {
714             stagingState = StagingState.MEMORY_STAGING;
715         } else {
716             stagingState = StagingState.DISK_STAGING;

724             String dbName = StringUtils.isNotBlank(databaseName) ? databaseName :
725                     account.getDatabaseConfig().getDatabaseName();

731                         OUTPUT_FIELD_PROP, document);

733                     Map<String, Object> columns = (Map<String, Object>) schema.get(COLUMNS);
734                     Map<String, Object> result = columns.entrySet()
735                             .stream()
736                             .filter(map -> outputFieldsList.contains(map.getKey()))
737                             .collect(Collectors.toMap(map -> map.getKey(), map -> map.getValue()));

756     protected void select(Document document) {
757         try {
758             jdbcOperations.acquireConnection(account);
759             SQLDialect dialect = account.getDatabaseConfig().getSQLDialect();
760             RetryPolicy defaultRetryPolicy = jdbcOperations.getDefaultRetryPolicy(
761                     maxRetries, retryInterval);
762             if (defaultRetryPolicy == null) {
763                 selectAndSpoolResults(document);
764             } else {
765                 Failsafe.with(defaultRetryPolicy)
766                         .onFailedAttempt(e -> LOG.warn("simple_sql_select_snap_retry - failed to " +
767                                 "selectAndSpoolResults: {}", e.getMessage()))
768                         .onRetry(throwable -> {
769                             jdbcOperations.handleOnRetry(dialect.getName(), throwable);
770                         })
771                         .run(cntx -> {
772                             jdbcOperations.acquireConnection(account);
773                             jdbcOperations.logRetryAttempt(cntx, LOG);
774                             selectAndSpoolResults(document);
775                         });
776             }
777         } catch (SnapDataException e) {
778             errorViews.write(e, document);
779         } catch (SuggestViewAbortException | ExecutionException e) {
780             throw e;
781         } catch (Exception e) {
782             throw new ExecutionException(e, EXECUTE_QUERY_FAILED)
783                     .withReason(e.getMessage())
784                     .withResolution(PLEASE_ADDRESS_ISSUE);
785         }
786     }

789         if (jdbcOperations instanceof JdbcOperationsImpl){
790             ((JdbcOperationsImpl) jdbcOperations)
791                 .setConvertUniqueIdentifierColumnValueToUpperCase

794                     .setDisplayRowversionAsInt(convertRowversionToBigInt);
795         }

797                 maxRetries, retryInterval)) {
798             switch (stagingState) {
