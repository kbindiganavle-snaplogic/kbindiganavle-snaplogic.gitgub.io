265         truncateTable();

276             // We want to create the table even if there are no rows to be loaded.
277             createTableIfAbsent(null);

280             if (!successful || dataWriter == null) {
281                 return;

394     protected void loadAction(String targetTable, Document original) {
395         if (connection == null) {
396             // Connection might have been created already if truncate was done
397             connection = jdbcOperations.acquireConnection(account);
398         }
399 
400         filePattern = fileNamePatternExpression.eval(original);
401         fileFormatType = fileFormatTypeExpression.eval(original);
402         fileFormatType = StringUtils.isBlank(fileFormatType) ? CSV : fileFormatType;
403         if (CollectionUtils.isNotEmpty(copyStatementArgumentExpression)) {
404             // Evaluate file format options
405             copyStatementArguments = new ArrayList<>();
406             for (ExpressionProperty optionsExpression : copyStatementArgumentExpression) {
407                 String option = optionsExpression.eval(original);
408                 if (StringUtils.isBlank(option)) {
409                     continue;
410                 }
411                 if (StringUtils.startsWithIgnoreCase(option, ERROR)) {
412                     throw new SnapDataException(String.format(ERROR_INVALID_OPTION, option))
413                             .withReason(ERROR_INVALID_OPTION_REASON)
414                             .withResolution(PLEASE_ADDRESS_MESSAGE);
415                 }
416                 copyStatementArguments.add(option);
417             }
418         }
419         buildCopyCommand();
420         // Output results
421         LOGGER.debug(LOG_COPY_COMMAND, getCopyWithObfuscatedCredentials());
422         try (DatabaseCursor cursor = jdbcOperations.executeQuery(copyCommand, null,
423                 account, 0, 1)) {
424             Map<String, Object> successRecord = new LinkedHashMap<>();
425             successRecord.put("query", getCopyWithObfuscatedCredentials());
426             if (StringUtils.isNotEmpty(errFileAzureFolder)) {
427                 successRecord.put("errorFile", errFileAzureFolder);
428             }
429             successRecord.put("message", SUCCESS);
430             // Not to miss If any status codes from resultset
431             while (cursor.hasNext()) {
432                 successRecord.putAll(cursor.fetchOne());
433             }
434             Document successDoc = documentUtility.newDocument(successRecord);
435             outputViews.write(successDoc, original);
436         } catch (ExecutionException | SnapDataException e) {
437             throw e;
438         } catch (SQLException e) {
439             SnapDataException exception = new SnapDataException(e, ERROR_PERFORMING_LOAD)
440                     .withReason(e.getLocalizedMessage())
441                     .withResolutionAsDefect();
442             throw exception;
443         } catch (NumberFormatException ne) {
444             throw new SnapDataException(ne, ERROR_PERFORMING_LOAD)
445                     .withReason(ne.getLocalizedMessage())
446                     .withResolution(RESOLUTION_ADDRESS_ISSUE);
447         } catch (Exception e) {
448             throw new ExecutionException(e, ERROR_PERFORMING_LOAD)
449                     .withReason(e.getLocalizedMessage())
450                     .withResolution(RESOLUTION_ADDRESS_ISSUE);
451         } finally {
452             JDBCUtils.safeClose(connection);
453             connection = null;
454         }
455         LOGGER.debug(LOG_COPY_DONE);
456     }

462      * @param document
463      * @return A list of column names.

466     private List<String> getQueryColumns(Document document) {
467         List<String> columns;
468         if (!queryColumns.isEmpty()) {
469             Connection conn = jdbcOperations.acquireConnection(account);
470             tableMetaData = jdbcOperations.getTableMetaData(conn,
471                     account.getDatabaseConfig().getDatabaseName(), tableName);
472             columns = new ArrayList<>();
473             queryColumns.forEach(columnSettings -> {
474                 String column = columnSettings.get(0).eval(document);
475                 String defaultVal = columnSettings.get(1).eval(document);
476                 BigInteger position = columnSettings.get(2).eval(document);
477                 boolean enableQuote = columnSettings.get(3).eval(document);
478                 if (StringUtils.isNotEmpty(column)
479                         && tableMetaData.containsKey(column)) {
480                     StringBuilder builder = new StringBuilder();
481                     builder.append(String.format(ID_QUOTE_FORMAT, column));
482                     if (StringUtils.isNotBlank(defaultVal)) {
483                         builder.append(StringUtils.SPACE);
484                         if (enableQuote) {
485                             builder.append(String.format(DEAFULT_WITH_QUOTE_FORMAT, defaultVal));
486                         } else {
487                             builder.append(String.format(DEAFULT_FORMAT, defaultVal));
488                         }
489                     }
490                     if (position != null) {
491                         builder.append(StringUtils.SPACE)
492                                 .append(position.toString());
493                     }
494                     columns.add(builder.toString());
495                     copyColumns.add(column);
496                 }
497             });
498         } else {
499             columns = null;
500         }
501         return columns;

523                 } catch (SnapDataException sde) {
524                     errorViews.write(sde);
525                 }
526             case FAILURE:

528                     rollback();
529                 }
530                 if (INPUT_VIEW.equals(dataSource)) {
531                     purgeFilesFromExternalStorage();

533                 break;
534             default:

536         }
