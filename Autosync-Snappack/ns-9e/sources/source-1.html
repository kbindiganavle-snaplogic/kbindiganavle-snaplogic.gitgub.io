401      * Suggests a list of table columns. This gets all columns in the table and filters out
402      * columns that the user has been selected.
403      *
404      * @param suggestionBuilder
405      * @param propertyValues
406      */
407     protected void suggestTableColumns(SuggestionBuilder suggestionBuilder,
408                                        PropertyValues propertyValues) {
409         //Configure table related properties to initialize table metadata
410         initOperations();
411         isInSuggest = true;
412         super.configureSnap(propertyValues);
413         Set<String> columnsToSuggest = new HashSet<>(tableMetaData.keySet());
414         if (columnsToSuggest.isEmpty()) {
415             throw new ExecutionException(NO_COLUMNS_AVAILABLE);
416         }
417         //Filter columns that have already been selected
418         List<List<ExpressionProperty>> rows = getTableColumns(propertyValues);
419         rows.forEach(row -> {
420             String inputColumn = row.get(0).eval(null);
421             if (inputColumn != null) {
422                 columnsToSuggest.remove(inputColumn);
423             }
424         });
425         if (columnsToSuggest.isEmpty()) {
426             throw new ExecutionException(ERR_ALL_COLUMNS_SELECTED)
427                     .withReason(ERR_ALL_COLUMNS_SELECTED_REASON)
428                     .withResolution(ERR_ALL_COLUMNS_SELECTED_RESOLUTION);
429         }
430         suggestionBuilder.node(COLUMN_NAMES_PROP)
431                 .over(TABLE_COLUMN_NAMES_PROP)
432                 .suggestions(columnsToSuggest.toArray(new String[columnsToSuggest.size()]));
433     }

578      * Creates absolute paths for each file in the list
579      *
580      * @param filesList
581      * @param accountName
582      * @param endpoint
583      * @param container
584      * @param folder
585      * @param validateFilePaths will be false for error file
586      */
587     void buildExternalFilePaths(List<String> filesList, String accountName, String endpoint,
588                                 String container, String folder, boolean validateFilePaths) {
589         if (CollectionUtils.isNotEmpty(filesList)) {
590             List<String> blobFiles = null;
591             /**
592              * external file validation not required for datasource inputview
593              * so skipping blob lookup
594              */
595             if (validateFilePaths && EXTERNAL_STORAGE.equals(dataSource)) {
596                 blobFiles = fetchBlobFiles();
597             }
598             for (String filePath : filesList) {
599                 String filePathReformatted;
600                 Matcher extFileMatcher = EXTFILE_PATTERN.matcher(filePath);
601                 if (!extFileMatcher.matches()) {
602                     filePathReformatted = reformatUrlFilePaths(filePath);
603                     StringBuilder filePathBuilder = new StringBuilder(2048);
604                     if (StringUtils.isNotBlank(container)) {
605                         filePathBuilder.append(StringUtils.strip(container, SLASH));
606                     }
607                     if (StringUtils.isNotBlank(folder)) {
608                         filePathBuilder.append(SLASH);
609                         filePathBuilder.append(StringUtils.strip(folder, SLASH));
610                     }
611                     if (StringUtils.isNotBlank(filePathReformatted)) {
612                         filePathBuilder.append(SLASH);
613                         filePathBuilder.append(StringUtils.strip(filePathReformatted, SLASH));
614                     }
615                     if (CollectionUtils.isNotEmpty(blobFiles) && !blobFiles.stream().anyMatch(f ->
616                             StringUtils.equalsIgnoreCase(f,
617                                     StringUtils.strip(StringUtils.strip(folder, SLASH) + SLASH +
618                                     StringUtils.strip(filePathReformatted, SLASH), SLASH)))) {
619                         throw new ExecutionException(ERR_NO_FILES_FOUND_TO_COPY_FROM)
620                                 .withReason(String.format(REASON_NO_FILES_FOUND,
621                                         filePathReformatted,
622                                         StringUtils.strip(StringUtils.isEmpty(folder) ?
623                                                 container : folder, SLASH)))
624                                 .withResolution(RES_NO_FILES_FOUND);
625                     }
626                     String fullFilePath = String.format(EXTERNAL_FILE_FMT, accountName,
627                             endpoint, filePathBuilder);
628                     int index = filesList.indexOf(filePathReformatted);
629                     filesList.set(index, fullFilePath);
630                 }
631             }
632         }
633     }
