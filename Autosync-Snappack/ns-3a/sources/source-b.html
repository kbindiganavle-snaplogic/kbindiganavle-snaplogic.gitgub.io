149     /**
150      * Parse access key, secret key, bucket name and file key from url. Throw exception if the
151      * given
152      * url is not in correct format.
153      *
154      * @param url
155      */
156     public S3UrlConnection(URL url) throws IOException {
157         super(url); // url = "s3://accessKeyID:secretKey@s3/bucketName/folder/.../filename"
158         URIBuilder builder = null;
159         try {
160             builder = new URIBuilder(url.toURI());
161         } catch (URISyntaxException e) {
162             throw new MalformedURLException(
163                     String.format(ERR_URL_SYNTAX, getFileNoAuth(url.toString()), e.getMessage()));
164         }
165         // parse access key ID and secret key
166         String userInfo = builder.getUserInfo(); // userInfor = "accessKeyID:secretKey"
167         if (StringUtils.isNotBlank(userInfo)) {
168             int index = userInfo.indexOf(':');
169             if (index > 0 && index < userInfo.length()) {
170                 awsAccessKey = userInfo.substring(0, index);
171                 awsSecretKey = userInfo.substring(index + 1);
172                 String[] tokens = StringUtils.split(awsSecretKey, ":", 2);
173                 if (tokens.length > 1) {
174                     // In the case of Cross account IAM, the secretKey from the userInfo is
175                     // formated by secretKey:sessionToken
176                     awsSecretKey = tokens[0];
177                     sessionToken = tokens[1];
178                 }
179             }
180         }
181         // parse bucket name and file key
182         String path = builder.getPath(); // path = "/bucketName/folder/.../filename"
183         if (path != null && path.startsWith(URL_SEPARATOR)) {
184             int index = path.indexOf(URL_SEPARATOR, 1);
185             if (index > 1 && index < path.length()) {
186                 bucketName = path.substring(1, index);
187                 fileKey = path.substring(index + 1); // fileKey = "folder/.../filename"
188                 index = bucketName.indexOf(REGION_DELIMITER);
189                 s3Region = index > 0 ? bucketName.substring(index + 1) : StringUtils.EMPTY;
190             }
191         }
192     }

198     public static ClientConfiguration createClientConfiguration() {
199         ClientConfiguration config = new ClientConfiguration();
200         /**
201          * The HTTPS protocol handler will use the same nonProxyHosts property as the HTTP Protocol.
202          *
203          * networking properties reference:
204          * https://docs.oracle.com/javase/8/docs/technotes/guides/net/proxies.html
205          */
206         String nonProxyHosts = System.getProperty(HTTP_NON_PROXY);
207         String proxyHost = System.getProperty(HTTPS_PROXY);
208         String proxyPort = System.getProperty(HTTPS_PORT);
209         String username = System.getProperty(HTTPS_PROXY_USER);
210         String password = System.getProperty(HTTPS_PROXY_PASSWORD);
211         if (StringUtils.isBlank(proxyHost) || StringUtils.isBlank(proxyPort)) {
212             proxyHost = System.getProperty(HTTP_PROXY);
213             proxyPort = System.getProperty(HTTP_PORT);
214             username = System.getProperty(HTTP_PROXY_USER);
215             password = System.getProperty(HTTP_PROXY_PASSWORD);
216         }
217         if (StringUtils.isNotBlank(proxyHost) && StringUtils.isNotBlank(proxyPort)) {
218             config.withProxyHost(proxyHost)
219                     .withProxyPort(Integer.parseInt(proxyPort));
220             if (!StringUtils.isBlank(username) && !StringUtils.isBlank(password)) {
221                 config.withProxyUsername(username)
222                         .withProxyPassword(password);
223             }
224         }
225         if (StringUtils.isNotBlank(nonProxyHosts)) {
226             config.withNonProxyHosts(nonProxyHosts);
227         }
228         String prefix = S3Utils.getApnUserAgentHeaderFor(COMPANY_NAME, PROJECT_NAME, VERSION);
229         return config

253     private AWSCredentialsProvider getAwsCredentialsProvider() throws MalformedURLException {
254         if (StringUtils.isBlank(awsAccessKey)) {
255             String ec2CredentialForS3 = System.getProperty(IAM_CREDENTIAL_FOR_S3);
256             if (Boolean.parseBoolean(ec2CredentialForS3)) {
257                 return new InstanceProfileCredentialsProvider();
258             }
259             throw new MalformedURLException(
260                     String.format(ERR_ACCESS_KEY_ID, getFileNoAuth(url.toString())));
261         }
262         if (StringUtils.isBlank(awsSecretKey)) {
263             throw new MalformedURLException(
264                     String.format(ERR_SECRET_KEY, getFileNoAuth(url.toString())));
265         }
266         String securityToken = getRequestProperty(SECURITY_TOKEN);
267         if (StringUtils.isNotEmpty(sessionToken)) {
268             securityToken = sessionToken;
269         }
270         if (StringUtils.isBlank(securityToken)) {
271             awsCredentials = new BasicAWSCredentials(awsAccessKey, awsSecretKey);
272         } else {
273             awsCredentials = new BasicSessionCredentials(awsAccessKey, awsSecretKey, securityToken);
274         }
275         return new StaticCredentialsProvider(awsCredentials);
276     }

307     public Map<String, List<String>> getHeaderFields() {
308         if (objectMetadata != null) {
309             Map<String, List<String>> retval = new LinkedHashMap<>();
310 
311             addHeader(retval, HttpHeaders.CACHE_CONTROL, objectMetadata.getCacheControl());
312             addHeader(retval, CONTENT_DISPOSITION, objectMetadata.getContentDisposition());
313             addHeader(retval, HttpHeaders.CONTENT_ENCODING, objectMetadata.getContentEncoding());
314 
315             String unencrypted = objectMetadata.getUserMetaDataOf(UNENCRYPTED_CONTENT_LENGTH);
316             String cryptoKeymapAlgorithm =
317                     objectMetadata.getUserMetaDataOf(CRYPTO_KEYWRAP_ALGORITHM);
318             // if unencrypted content length is available, use it as content length
319             String contentLength = StringUtils.isNotBlank(unencrypted) ? unencrypted :
320                     // if client-side encrypted, but unencrypted content length is not available,
321                     // set content length negative to disable file size check
322                     String.valueOf(StringUtils.isBlank(cryptoKeymapAlgorithm) ?
323                     objectMetadata.getContentLength() : -1);
324             addHeader(retval, CONTENT_LENGTH, contentLength);
325 
326             addHeader(retval, HttpHeaders.CONTENT_TYPE, objectMetadata.getContentType());
327             addHeader(retval, HttpHeaders.ETAG, objectMetadata.getETag());
328             Date lastModified = objectMetadata.getLastModified();
329             // TODO - psung: AWS Java SDK 1.7.11 or newer has a bug in which getLastModified()
330             // returns null while 1.7.9 or older returns a correct date. The same error appears
331             // in up to 1.8.9.1 (the latest as of writing)
332             if (lastModified != null) {
333                 addHeader(retval, HttpHeaders.LAST_MODIFIED, DateUtil.formatDate(lastModified));
334             }
335             for (Map.Entry<String, String> entry : objectMetadata.getUserMetadata().entrySet()) {
336                 retval.put(entry.getKey(), Arrays.asList(entry.getValue()));
337             }
338 
339             for (Map.Entry<String, Object> entry : objectMetadata.getRawMetadata().entrySet()) {
340                 //This includes headers such as server-side encryption settings
341                 Object value = entry.getValue().toString();
342                 if (value != null) {
343                     addHeader(retval, entry.getKey(), value.toString());
344                 }
345             }
346 
347             return retval;
348         }
349         return Collections.emptyMap();
350     }

352     @Override
353     public InputStream getInputStream() throws IOException {
354         if (StringUtils.isBlank(bucketName) || StringUtils.isBlank(fileKey)) {
355             throw new MalformedURLException(
356                     String.format(ERR_BUCKET_FILE_KEY_NAME, getFileNoAuth(url.toString())));
357         }
358         try {
359             // get S3 object for given bucket name and file key
360             GetObjectRequest getObjectRequest = new GetObjectRequest(bucketName, fileKey);
361             String versionId = getRequestProperty(VERSION_ID_PROP);
362             if (StringUtils.isNotBlank(versionId)) {
363                 getObjectRequest.setVersionId(versionId);
364             }
365             s3Object = s3.getObject(getObjectRequest);
366         } catch (AmazonS3Exception e) {
367             Map<String, String> details = e.getAdditionalDetails();
368             String reason;
369             if (MapUtils.isNotEmpty(details) &&
370                     StringUtils.equalsIgnoreCase(DELETE_MARKER, details.get(RESOURCE_TYPE))) {
371                 reason = REASON_DELETE_MARKER;
372             } else {
373                 reason = String.format(FORMAT_EXTENDED_REQUEST_ID, e.getMessage(),
374                         e.getExtendedRequestId());
375             }
376             throwSnapDataException(e, reason);
377         } catch (AmazonClientException e) {
378             throwSnapDataException(e, e.getMessage());
379         }
380         if (s3Object == null) {
381             throw new IOException(
382                     String.format(ERR_CANNOT_FIND_FILE, bucketName, fileKey, s3Region));
383         }
384 
385         objectMetadata = s3Object.getObjectMetadata();
386 
387         //Logs Amazon AWS Authorization Header to check Signature.
388         //Signature V4 example
389         //AWS4-HMAC-SHA256 Credential=AKIAIEOHKZSBPQUCKPLQ/20190826/us-east-1/s3/aws4_request
390         S3ObjectInputStream s3ObjectInputStream = s3Object.getObjectContent();
391         if (null != s3ObjectInputStream) {
392             HttpRequestBase httpRequestBase = s3ObjectInputStream.getHttpRequest();
393             if (null != httpRequestBase) {
394                 if (httpRequestBase.containsHeader(AUTHORIZATION_HEADER)){
395                     HeaderIterator headerIterator = httpRequestBase.headerIterator();
396                     while (headerIterator.hasNext()) {
397                         Header header = headerIterator.nextHeader();
398                         if (header.getName().equals(AUTHORIZATION_HEADER)){
399                             LOG.debug(String.format(
400                                     "AWS Authorization Header Signature : %s", header.getValue()));
401                         }
402                     }
403                 }
404             }
405             // get object tagging
406             readObjectTags();
407         }
408         // Return S3 file input stream
409         return new BufferedInputStream(s3ObjectInputStream);
410     }

474     /**
475      * Creates a S3 client builder. If client-side authentication is used then this will create
476      * an encryption client builder using the KMS key.
477      * @param credentialsProvider
478      * @param clientConfiguration
479      * @return AmazonS3Builder which is either a AmazonS3ClientBuilder or
480      * AmazonS3EncryptionClientV2Builder instance.
481      */
482     private AmazonS3Builder getS3ClientBuilder(AWSCredentialsProvider credentialsProvider,
483             ClientConfiguration clientConfiguration) {
484 
485         S3EncryptionType encryptType = getEncryptionType();
486         if (S3EncryptionType.CSE_KMS.equals(encryptType)) {
487             // client-side encryption
488             String kmsKey = this.getRequestProperty(KMS_KEY);
489             String kmsRegion = this.getRequestProperty(KMS_REGION);
490             if (StringUtils.isBlank(kmsKey)) {
491                 //If no KMS key is provided, supply a default one. In the case where the encryption
492                 //client is only reading s3 objects it does not need a KMS key. The client is smart
493                 //enough to look at the object metadata and download the KMS key/certificate.
494                 //However, there is no way to create an encryption client in the AWS SDK without
495                 //supplying a KMS key. We supply a default value so the client can be used for
496                 //read only scenarios without forcing the end user to supply a key that won't be
497                 //used.
498                 kmsKey = DEFAULT_CSE_KMS;
499             }
500             if (StringUtils.isBlank(kmsRegion)) {
501                 //If no KMS region is provided, use the default us-east-1
502                 kmsRegion = Regions.US_EAST_1.getName();
503             }
504             KMSEncryptionMaterialsProvider kms = new KMSEncryptionMaterialsProvider(kmsKey);
505 
506             // minimal crypto configuration in V2; default CryptoMode.StrictAuthenticatedEncryption
507             CryptoConfigurationV2 cryptoConfig = new CryptoConfigurationV2()
508                     .withAwsKmsRegion(RegionUtils.getRegion(kmsRegion))
509                     // The following setting allows the client to read V1 encrypted objects
510                     .withCryptoMode(CryptoMode.AuthenticatedEncryption);
511 
512             return AmazonS3EncryptionClientV2
513                     .encryptionBuilder()
514                     .withCredentials(credentialsProvider)
515                     .withClientConfiguration(clientConfiguration)
