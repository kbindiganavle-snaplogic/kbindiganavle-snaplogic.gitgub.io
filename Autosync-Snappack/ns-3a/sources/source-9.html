125      * "s3://{yourAccessKeyId}:{yourSecretKey}@s3/{bucketName}/{folderPath}/{fileName}"

128      *

222      * Connect to S3 service with AWS credential, bucket name and key,
223      * and get S3 object listing. Invalid AWS credential or bucket name will cause exception.
224      */
225     protected ObjectListing getS3ObjectListing() {
226         ListObjectsRequest listObjectRequest = getListObjectsRequest();
227 
228         // if there is no includeSubfolders query, there is no need to list grandchildren
229         if (!QUERY_INCLUDESUBFOLDERS.equalsIgnoreCase(getQuery()) && StringUtils.isNotBlank(key)) {
230             listObjectRequest = listObjectRequest.withDelimiter(URL_SEPARATOR);
231         }
232         try {
233             objectListing = s3.listObjects(listObjectRequest);
234         } catch (AmazonServiceException e) {
235             throw new SnapDataException(e, ERR_S3_SERVICE)
236                     .withReason(e.getMessage())
237                     .withResolution(ERR_S3_SERVICE_RESOLUTION);
238         }
239 
240         // Users often forget to put "/" at the end of a S3 directory path. If it happens,
241         // we should try again listObjects() with modified key.
242         if (objectListing.getObjectSummaries().size() == 0 && !key.endsWith("/")) {
243             key += "/";
244             listObjectRequest = listObjectRequest.withPrefix(key);
245             objectListing = s3.listObjects(listObjectRequest);
246         }
247         return objectListing;
248     }

250     private String parseS3Path(String s3Path, AmazonS3 optionalS3Override) {
251         // get userInfo and S3Client
252         URIBuilder builder = URLUtils.getURIBuilder(s3Path);
253         userInfo = builder.getUserInfo();
254         Map<String, String> map = URLUtils.getUsernamePassword(userInfo);
255         accessKeyId = map.get(URLUtils.USERNAME);
256         secretKey = map.get(URLUtils.PASSWORD);
257         // get s3 bucket and key
258         String path = builder.getPath();
259         if (StringUtils.isBlank(path) || path.length() == 1) {
260             throw new SnapDataException(ERR_S3_BUCKET_NAME)
261                     .withReason(REASON_URI_SYNTAX)
262                     .withResolution(RESOLUTION_URI_SYNTAX);
263         }
264         String[] strings = path.split(URLUtils.URL_SEPARATOR);
265         switch (strings.length) {
266             case 0:
267             case 1:
268                 throw new SnapDataException(ERR_S3_BUCKET_NAME)
269                         .withReason(REASON_URI_SYNTAX)
270                         .withResolution(RESOLUTION_URI_SYNTAX);
271             case 2:
272                 bucketName = strings[1];
273                 key = StringUtils.EMPTY;
274                 break;
275             default:
276                 bucketName = strings[1];
277                 key = path.substring(path.indexOf(bucketName) + bucketName.length() + 1);
278         }
279         bucketPath = String.format(FORMAT_S3_BUCKET_PATH, bucketName);
280         boolean s3ClientOverrideProvided = (null != optionalS3Override);
281         if (s3ClientOverrideProvided) {
282             // non-default region support
283             // do NOT pass the s3Client if we provided an override s3Client because we want to
284             // reuse this client and it's region parameters. We still want to parse the bucketName
285             bucketName = JfsUtils.updateS3RegionInfo(bucketName, null);
286             s3 = optionalS3Override;
287             LOG.warn("s3ClientOverrideProvided=true");
288         } else {
289             s3 = getAmazonS3Client();
290         }
291 
292         // get query, expected to have no query or only one query "includeSubfolders"
293         List<NameValuePair> queryParams = builder.getQueryParams();
294         if (queryParams.isEmpty()) {
295             query = StringUtils.EMPTY;
296         } else {
297             query = queryParams.get(0).getName();
298         }
299         return builder.toString();
300     }

321     @VisibleForTesting
322     protected S3ClientConfig getS3ClientConfig() {
323         if (StringUtils.isNotBlank(accessKeyId) && StringUtils.isNotBlank(secretKey) &&
324                 isIAMCredential) {
325             // accessKeyId/secretKey are not blank isIAMCredential should be set false
326             isIAMCredential = false;
327         }
328         String[] tokens = StringUtils.split(secretKey, ":", 2);
329         if (tokens != null && tokens.length > 1) {
330             // In the case of Cross account IAM, the secretKey from the userInfo is
331             // formatted by secretKey:sessionToken
332             secretKey = tokens[0];
333             securityToken = tokens[1];
334         }
335         // parse region
336         int index = StringUtils.indexOf(bucketName, '@');
337         String region = null;
338         if (index > 1) {
339             region = bucketName.substring(index + 1);
340             bucketName = bucketName.substring(0, index);
341         }
342         // build S3ClientConfig
343         return S3ClientConfig.builder()
344                 .accessKeyId(accessKeyId)
345                 .secretKey(secretKey)
346                 .securityToken(securityToken)
347                 .region(region)
348                 .isIAMRole(isIAMCredential)
349                 .build();
350     }

359     /**
360      * Returns Amazon S3 client
361      *
362      * @param awsAccessKey access key ID
363      * @param awsSecretKey secret key
364      * @return Amazon S3 client
365      */
366     protected AmazonS3ClientBuilder getS3ClientBuilder(String awsAccessKey, String awsSecretKey,
367             String sessionToken) {
368         if (StringUtils.isBlank(awsAccessKey)) {
369             if (isIAMCredential) {
370                 return AmazonS3ClientBuilder.standard()
371                         .withCredentials(new InstanceProfileCredentialsProvider())
372                         .withClientConfiguration(S3UrlConnection.createClientConfiguration())
373                         .withForceGlobalBucketAccessEnabled(true);
374             } else {
375                 throw new SnapDataException(ERR_AWS_ACCESS_KEY_ID)
376                         .withReason(ERR_URI_SYNTAX)
377                         .withResolution(RESOLUTION_URI_SYNTAX);
378             }
379         }
380         AWSCredentials credentials = StringUtils.isBlank(sessionToken) ?
381                 new BasicAWSCredentials(awsAccessKey, awsSecretKey) :
382                 new BasicSessionCredentials(awsAccessKey, awsSecretKey, sessionToken);
383         return AmazonS3ClientBuilder.standard()
384                 .withCredentials(new AWSStaticCredentialsProvider(credentials))
385                 .withClientConfiguration(S3UrlConnection.createClientConfiguration())
386                 .withForceGlobalBucketAccessEnabled(true);
387     }
