275     @Override
276     public void close() throws IOException {
277         if (isClosed) {
278             return;
279         }
280         try {
281             if (!isFailed) {
282                 if (byteBuffer.position() > 0 || !uploadThreadInfoList.isEmpty()) {
283                     // If byteBuffer.position() is true this means there is more data to upload
284                     // Upload it and signal it is the last part.
285 
286                     // If uploadThreadInfoList is not empty and the byteBuffer is empty it means
287                     // all data has been written. Write an empty last part to signal the last part
288                     // to the S3 client.
289 
290                     // The S3 client must know when the last part is received when using
291                     // client side encryption so that it can encrypt the data before sending it to
292                     // S3.
293                     startUploadThread(byteBuffer, true);
294                 }
295             }
296 
297             // wait for multipart upload to complete
298             List<PartETag> partETags = new ArrayList<>();
299             for (UploadThreadInfo uploadThreadInfo : uploadThreadInfoList) {
300                 Future<PartETag> future = uploadThreadInfo.getFuture();
301                 waitForCompletion(future);
302                 bytesUploaded += uploadThreadInfo.getPartSize();
303                 UploadPart uploadPart = uploadThreadInfo.getUploadPart();
304                 PartETag partETag = uploadPart.getPartEtag();
305                 if (partETag == null) {
306                     // Since the upload of a part failed even if AmazonS3Client has retried,
307                     // the multipart upload should be aborted
308                     Exception e = uploadPart.getException();
309                     SnapDataException ex;
310                     if (e == null) {
311                         ex = new SnapDataException(ERR_UPLOAD_PART)
312                                 .withReason(REASON_RETRIES_UPLOAD_PART);
313                     } else {
314                         ex = new SnapDataException(e, ERR_UPLOAD_PART)
315                                 .withReason(e.getMessage());
316                     }
317                     throw ex.formatWith(bucketName, key, bytesUploaded, uploadPart.getPartNumber())
318                             .withResolution(RESOLUTION_ADDRESS_ISSUE);
319                 } else {
320                     partETags.add(partETag);
321                 }
322             }
323             completeMultipartUpload(partETags);
324         } catch (SnapDataException e) {
325             isFailed = true;
326             throw e;
327         } catch (Exception e) {
328             isFailed = true;
329             throw new SnapDataException(e, ERR_UPLOAD_FILE)
330                     .formatWith(bucketName, key, bytesUploaded)
331                     .withResolution(RESOLUTION_ADDRESS_ISSUE);
332         } finally {
333             if (isFailed) {
334                 abortMultipartUpload();
335             }
336             shutdownThreadPool(fixedThreadPool);
337             isClosed = true;
338         }
339     }

376         if (partNumber > MAX_PART_NUMBER) {
377             isFailed = true;
378             throw new SnapDataException(ERR_MAX_UPLOAD_PARTS)

389             waitForThreadPool();

395             // save the upload thread info
396             uploadThreadInfoList.add(new UploadThreadInfo(uploadPart, buffer.position(), future));
397         } catch (SnapDataException e) {
398             throw e;
399         } catch (Exception e) {
400             isFailed = true;
401             throw new SnapDataException(e, ERR_UPLOAD_FILE)
402                     .formatWith(bucketName, key, bytesUploaded)
403                     .withResolution(RESOLUTION_ADDRESS_ISSUE);

433     private void completeMultipartUpload(List<PartETag> partETags) throws IOException {
434         if (partETags.size() == 0) {
435             byte[] md5Digest = DigestUtils.md5(new byte[0]);
436             try (ByteArrayInputStream inputStream = new ByteArrayInputStream(new byte[0])) {
437                 // no data written to upload
438                 // create request to upload an empty part
439                 UploadPartRequest uploadRequest = new UploadPartRequest()
440                         .withBucketName(bucketName).withKey(key)
441                         .withUploadId(uploadId)
442                         .withPartNumber(partNumber)
443                         .withInputStream(inputStream)
444                         .withMD5Digest(Base64.getEncoder().encodeToString(md5Digest))
445                         .withLastPart(true)
446                         .withPartSize(0);
447 
448                 partETags.add(s3.uploadPart(uploadRequest).getPartETag());
449             }
450         }
451         CompleteMultipartUploadRequest compRequest =
452                 new CompleteMultipartUploadRequest(bucketName, key, uploadId, partETags);
453         CompleteMultipartUploadResult result = s3.completeMultipartUpload(compRequest);
454         String versionId = result != null ? result.getVersionId() : null;
455         if (StringUtils.isNotBlank(versionId)) {
456             urlConnection.setRequestProperty(VERSION_ID_PROP, versionId);
457         }
458     }

465      * Shuts down gracefully the give thread pool
466      */
467     protected void shutdownThreadPool(final ExecutorService threadPool) {
468         // first, no more new tasks accepted
469         threadPool.shutdown();
470         try {
471             // wait for some time for the scheduled tasks to complete
472             if (!threadPool.awaitTermination(DEFAULT_THREAD_POOL_COMPLETE_TIME,
473                     DEFAULT_THREAD_POOL_COMPLETE_TIME_UNIT)) {
474                 // still tasks didn't complete, so cancel them
475                 threadPool.shutdownNow();
476                 // wait for some time for cancelling the scheduled tasks
477                 if (!threadPool.awaitTermination(DEFAULT_THREAD_POOL_CANCEL_TIME,
478                         DEFAULT_THREAD_POOL_CANCEL_TIME_UNIT)) {
479                     LOGGER.error("Thread pool did not shutdown gracefully");
480                 }
481             }
482         } catch (InterruptedException e1) {
483             // attempt to cancel the tasks again as the thread got interrupted
484             threadPool.shutdownNow();
485         }
486     }

548             RetryPolicy retryPolicy = new RetryPolicy()

557         private PartETag tryUpload() throws Exception {
558             //read bytes till data
559             byte[] data = new byte[buffer.position()];
560             //flip from 'write' mode to 'read' mode
561             buffer.flip();
562             buffer.get(data,  buffer.position(), buffer.limit());
563             try (ByteArrayInputStream inputStream =
564                          new ByteArrayInputStream(data)) {
565                 // create request to upload a part
566                 UploadPartRequest uploadRequest = new UploadPartRequest()
567                         .withBucketName(bucketName)
568                         .withKey(key)
569                         .withUploadId(uploadId)
570                         .withPartNumber(partNumber)
571                         .withInputStream(inputStream)
572                         .withLastPart(isLastPart)
573                         .withPartSize(partSize)
574                         .withGeneralProgressListener(this.progressListener);
575                 // Amazon S3 Encryption Client in the AWS SDK will encrypt data locally to ensure
576                 // its security. If we calculate md5sum before uploading it will not match with
577                 // server calculated md5 as sdk uploading encrypted data.
578                 if (!S3EncryptionType.CSE_KMS.equals(encryptionType)) {
579                     //128-bit MD5 digest of the part data.
580                     byte[] md5Digest = DigestUtils.md5(data);
581                     //base64-encoded 128-bit MD5 digest of the part data
582                     uploadRequest.withMD5Digest(Base64.getEncoder().encodeToString(md5Digest));
583                 }
584                 // upload part
585                 partETag = s3.uploadPart(uploadRequest).getPartETag();
586             } catch (Exception e) {
587                 LOGGER.debug("S3 Upload part failed for bucket: {}, key: {}, part number: {}",
588                         bucketName, key, partNumber);
589                 exception = e;
590                 parent.isFailed = true;
591             } finally {
592                 buffer.clear();
593                 queue.offer(buffer, BLOCKING_QUEUE_TIMEOUT, TimeUnit.HOURS);
594             }
595             return partETag;
596         }
