513     /**
514      * Validates the Discovery Document acquired from the OpenID Provider.
515      */
516     public void validate(String jwsAlg) {
517         // verify that the required fields are present
518         validateRequiredFields();
519 
520         // validate that the OpenID Provider
521         // supports the JWS algorithm selected by the user
522         if (!idTokenSigningAlgValuesSupported.contains(jwsAlg)) {
523             throw new SnapDataException("The provided JWS Algorithm is not " +
524                     "supported by the OIDC provider. Please select one of " +
525                     "the following: " +
526                     String.join(", ", idTokenSigningAlgValuesSupported));
527         }
528     }
529 
530     /**
531      * Validates that the fields mentioned as required in OpenID Discovery
532      * documentation are present.
533      */
534     private void validateRequiredFields() {
535         // based on the future iteration; more checks could be added
536         String missingRequiredField = null;
537 
538         if (StringUtils.isEmpty(issuer)) {
539             missingRequiredField = "issuer";
540         } else if (StringUtils.isEmpty(jwksUri)) {
541             missingRequiredField = "jwks_uri";
542         } else if (idTokenSigningAlgValuesSupported == null) {
543             missingRequiredField = "id_token_signing_alg_values_supported";
544         } else if (!idTokenSigningAlgValuesSupported.contains(
545                 SupportedJWSAlgorithm.RS256.getAlgorithmName())) {
546            missingRequiredField = "RS256 in id_token_signing_alg_values_supported";
547         }
548 
549         if (missingRequiredField != null) {
550             throw new SnapDataException(String.format("Discovery document does not " +
551                     "contain %s.", missingRequiredField))
552                     .withResolution("Make sure the OpenID Discovery Document URL " +
553                             "is correct.");
554         }
555     }
