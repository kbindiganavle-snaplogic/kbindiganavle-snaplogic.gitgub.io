291         // XXX This IS needed since the platform currently only avoids running policies that
292         // that have ALL @EnsuresContext properties already satisfied.
293         if (StandardContexts.SECURITY.getFromContext(requestContext).isPresent()) {
294             return;
295         }
296 
297         if (RATE_LIMIT_REQUESTS) {
298             // We don't want unauthenticated requests consuming resources, so we rate-limit the
299             // client based on their IP address.
300             RateLimitContext context = new RateLimitContext();
301             context.taskPath = StandardContexts.ASSET_PATH.getFromContext(requestContext).get();
302 
303             RateLimitEntity entity = new RateLimitEntity();
304             entity.instanceId = instanceId;
305             entity.instanceVersion = instanceVersion;
306             entity.clientId = StandardContexts.CLIENT_IP.getFromContext(requestContext)
307                     .orElse(StringUtils.EMPTY);
308 
309             RateLimitLocalService.AcquireResult acquireResult =
310                     rateLimitLocalService.acquire(context, entity, 0);
311 
312             if (acquireResult.recommendation != RateLimitLocalService.AcquireRecommendation.ALLOW) {
313                 Response response = ResponseGenerator.toResponse(Response.Status.TOO_MANY_REQUESTS,
314                         "Too many unauthenticated requests have been made.");
315                 requestContext.abortWith(response);
316                 return;
317             }
318         }
319 
320         // This state will be stored in plexfs and retrieved in the callback
321         PrivateAuthState privateState = new PrivateAuthState();
322         String qs = requestContext.getUriInfo().getRequestUri().getQuery();
323         privateState.uri = String.format("%1$s?%2$s", requestContext.getUriInfo().getPath(),
324                 qs == null ? "" : qs);
325         // Note: The callback method won't have access to the "request" object, so we need to
326         // evaluate any properties in this method and store them in this state object.
327         privateState.redirectUri = redirectUri.eval(null).orElse(StringUtils.EMPTY);
328         privateState.body = IOUtils.toByteArray(requestContext.getEntityStream());
329         privateState.method = requestContext.getMethod();
330         privateState.headers = copyHeaders(requestContext);
331         String clientIdVal = clientId.eval(null);
332         privateState.clientID = clientIdVal;
333         String clientSecVal = clientSecret.eval(null);
334         privateState.clientSecret = clientSecVal;
335         String accessTokenUrlVal = accessTokenUrl.eval(null);
336         privateState.accessTokenURL = accessTokenUrlVal;
337         LOG.debug("writing private auth state with method={} & entity size={}", privateState.method,
338                 privateState.body.length);
339 
340         String stateUUID = UUID.randomUUID().toString();
341         AssetAuthState assetAuthState = new AssetAuthState();
342         assetAuthState.policyInstanceId = instanceId;
343         assetAuthState.assetPath = StandardContexts.ASSET_PATH.getFromContext(requestContext).get();
344         assetAuthState.id = stateUUID;
345         assetAuthState.proxy = StandardContexts.PROXY.getFromContext(requestContext).get();
346 
347         long expiresAt = System.currentTimeMillis() +
348                 TimeUnit.SECONDS.toMillis(stateTtl.eval(null));
349         URI stateUri = URI.create(String.format("plexfs:/_oauth2_states/%s?expires=%s",
350                 stateUUID, expiresAt));
351         HttpURLConnection urlConnection = (HttpURLConnection) stateUri.toURL().openConnection();
352         urlConnection.setRequestMethod(HttpMethod.PUT);
353         urlConnection.setDoOutput(true);
354         try (OutputStream outputStream = urlConnection.getOutputStream()) {
355             MAPPER.writeValue(outputStream, privateState);
356         }
357

359             URIBuilder uriBuilder = new URIBuilder(loginUrl.eval(null))

361                     .addParameter("scope", requiredScopes.stream()

364                     .addParameter("client_id", clientIdVal)
365                     .addParameter("state", MAPPER.writeValueAsString(assetAuthState));
366             if (StringUtils.isNotBlank(privateState.redirectUri)) {
367                 uriBuilder.addParameter("redirect_uri", privateState.redirectUri);

380     @Override
381     public Response callback(final String code, final AssetAuthState state) throws IOException {
382         URI stateUri = URI.create(String.format("plexfs:/_oauth2_states/%s", state.id));
383         HttpURLConnection urlConnection = (HttpURLConnection) stateUri.toURL().openConnection();
384         PrivateAuthState privateAuthState;
385         try (InputStream inputStream = urlConnection.getInputStream()) {
386             privateAuthState = MAPPER.readValue(inputStream, PrivateAuthState.class);
387         } catch (FileNotFoundException e) {
388             throw new ExecutionException(e, "Could not find oauth2 request for state: %s")
389                     .formatWith(state);
390         }
391 
392         Map<String, Object> container = new LinkedHashMap<>();
393         Document doc = documentUtility.newDocument(container);
394 
395         try {
396             String postUrl = privateAuthState.accessTokenURL;
397             URIBuilder postUriBuilder = new URIBuilder(postUrl);
398             List<NameValuePair> params = new ArrayList<>();
399             params.add(new BasicNameValuePair("grant_type", "authorization_code"));
400             params.add(new BasicNameValuePair("client_id", privateAuthState.clientID));
401             params.add(new BasicNameValuePair("client_secret", privateAuthState.clientSecret));
402             params.add(new BasicNameValuePair("code", code));
403             params.add(new BasicNameValuePair("state", MAPPER.writeValueAsString(state)));
404             if (StringUtils.isNotBlank(privateAuthState.redirectUri)) {
405                 params.add(new BasicNameValuePair("redirect_uri", privateAuthState.redirectUri));
406             }
407 
408             HttpPost post = new HttpPost(postUriBuilder.build());
409             HttpEntity entity = new UrlEncodedFormEntity(params, "UTF-8");
410             post.setEntity(entity);
411 
412             post.addHeader(HttpHeaders.ACCEPT, MediaType.APPLICATION_JSON);
413             post.addHeader(HttpHeaders.CONTENT_TYPE, MediaType.APPLICATION_FORM_URLENCODED);
414 
415             String username = null;
416             List<String> roles = null;
417 
418             HttpResponse response = httpUtils.executeRequest(httpClient, post);
419             try (InputStream inputStream = httpUtils.getInputStreamFromResponse(response,
420                     post.getURI())) {
421                 Object result = MAPPER.readValue(inputStream, Object.class);
422 
423                 if (isOIDC) {
424                     // 1. Fetch/Validate the Discovery Document
425                     String discoveryURL =
426                             oidcDiscoveryDocumentURL.eval(null).orElse(null);
427                     if (discoveryURL == null) {
428                         throw new SnapDataException("OIDC Discovery URL is required");
429                     }
430                     Object retval = getDiscoveryDocument(discoveryURL);
431                     OpenIDDiscoveryDocument discoveryDocument;
432                     try {
433                         discoveryDocument =
434                                 MAPPER.readValue(MAPPER.writeValueAsString(retval),
435                                         OpenIDDiscoveryDocument.class);
436                         // validate the document
437                         discoveryDocument.validate(jwsAlgorithm.getAlgorithmName());
438                     } catch (JsonMappingException e) {
439                         throw new SnapDataException("Unable to read discovery document.");
440                     } catch (JsonProcessingException e) {
441                         throw new SnapDataException("Unable to parse discovery document.");
442                     }
443 
444                     // 2. Validate ID Token
445                     Map<String, String> tokenObj = MAPPER.convertValue(result,
446                             new TypeReference<LinkedHashMap<String, String>>() {
447                             });
448 
449                     Issuer issuer = new Issuer(discoveryDocument.getIssuer());
450                     ClientID clientID = new ClientID(privateAuthState.clientID);
451                     JWSAlgorithm jwsAlg = jwsAlgorithm.getJWSAlgorithm();
452 
453                     // Create validator for signed ID tokens
454                     IDTokenValidator validator;
455                     if (SupportedJWSAlgorithm.isRSA(jwsAlg)) {
456                         URL jwkSetURL = new URL(discoveryDocument.getJwksUri());
457                         validator = new IDTokenValidator(issuer, clientID, jwsAlg, jwkSetURL);
458                     } else {
459                         Secret secret = new Secret(privateAuthState.clientSecret);
460                         validator = new IDTokenValidator(issuer, clientID, jwsAlg, secret);
461                     }
462 
463                     // Parse the ID token
464                     IDTokenClaimsSet claims;
465                     try {
466                         JWT idToken = JWTParser.parse(tokenObj.getOrDefault("id_token", ""));
467                         // TODO - We are not doing CSRF validation by default.
468                         Nonce expectedNonce = null; // new Nonce(<CSRF token>); or null
469                         claims = validator.validate(idToken, expectedNonce);
470                         username = claims.getSubject().getValue();
471                     } catch (BadJOSEException e) {
472                         // Invalid signature or claims (iss, aud, exp...)
473                         throw new SnapDataException("Invalid signature or JWT claim");
474                     } catch (JOSEException | ParseException e) {
475                         // Internal processing exception
476                         throw new SnapDataException("Unable to parse ID token");
477                     }
478 
479                     // 3. Try to fetch User Profile from Access Token
480                     // If the userinfo_endpoint exists, we should get the username
481                     // using that instead of via claims
482                     if (StringUtils.isNotEmpty(discoveryDocument.getUserinfoEndpoint())) {
483                         String userInfoEndpoint = discoveryDocument.getUserinfoEndpoint();
484                         String accessToken = tokenObj.get("access_token");
485                         String type = tokenObj.getOrDefault("token_type", "Bearer");
486                         Map<String, Object> userInfo =
487                                 getUserInfo(userInfoEndpoint, accessToken, type);
488                         username = (String) userInfo.getOrDefault("sub", username);
489                     }
490 
491                     // Since scopes supported are recommended and
492                     // not required in a discovery document. Although it is written
493                     // that we can assume openid will always be in scope
494                     List<String> scopesSupported = discoveryDocument.getScopesSupported();
495                     roles = List.of("openid");
496                     if (scopesSupported != null) {
497                         roles = discoveryDocument.getScopesSupported();
498                     }
499                 }
500 
501                 if (ERROR_PATH.existsIn(result)) {
502                     throw new SnapDataException("Access token request failed with error: %s")
503                             .formatWith(ERROR_PATH.<String>readStatic(result))
504                             .withReason(ERROR_DESCRIPTION_PATH.readStatic(result, null))
505                             .withResolution("Check the following URI for more information: %s",
506                                     ERROR_URI_PATH.<String>readStatic(result, null));
507                 }
508                 container.put("token", result);
509             } catch (IOException e) {
510                 throw new SnapDataException(e, "Failed to get access token from: %s")
511                         .formatWith(postUrl)
512                         .withResolution("Make sure the URL is correct");
513             }
514 
515             TaskSession.OauthSession oauthSession = new TaskSession.OauthSession();
516             oauthSession.expiresAt =
517                     System.currentTimeMillis() + TimeUnit.SECONDS.toMillis(ttl.eval(null));
518 
519             if (!isOIDC) {
520                 for (HttpRequestProperties infoURL : infoURLs) {
521                     try {
522                         infoURL.execute(doc, container);
523                     } catch (IgnoredRequestException e) {
524                         // policy admin defines when Ignore Exception is thrown,
525                         // and the response is supposed to be ignored and not reported as an error
526                         continue;
527                     }
528                 }
529                 // since securityProps is set to Nullable, we need to perform this check
530                 // although this should never happen
531                 if (securityProps == null) {
532                     throw new SnapDataException("Unable to Extract User Info");
533                 }
534                 username = securityProps.getUsername(doc);
535                 roles = securityProps.getRoles(doc);
536             }
537 
538             // This should only happen in the case where OIDC is not used
539             // Since we set the User ID Expression and Roles Expression as not required,
540             // users might forget to input that field
541             if (StringUtils.isEmpty(username) || roles == null) {
542                 throw new SnapDataException("Unable to fetch username or roles.");
543             }
544 
545             oauthSession.username = username;
546             oauthSession.roles = roles;
547             oauthSession.sessionId = state.id;
548             taskSessionService.storeSession(oauthSession);
549 
550             String cookiePathId = pathId;
551 
552             if (state.proxy) {
553                 String pid = pathId.startsWith("/") ? pathId.substring(1) : pathId;
554                 String[] pidSplits = pid.split("/");
555                 if (pidSplits.length == 1) {
556                     // policy is assigned to the org
557                     cookiePathId = "/";
558                 } else if (pidSplits.length == 3) {
559                     // length == 2 should never happen, treated as proxy level
560                     cookiePathId = pidSplits[2];
561                 } else if (pidSplits.length == 4) {
562                     // policy is assigned to the endpoint
563                     cookiePathId = String.format("%s/%s", pidSplits[2], pidSplits[3]);
564                 } else {
565                     // this should never happen
566                     throw new SnapDataException("Cannot assign policy to this asset");
567                 }
568             }
569 
570             NewCookie authCookie = taskSessionService.createCookieFromSession(oauthSession,
571                     privateAuthState.uri, cookiePathId);
572 
573             return Response.status(Response.Status.FOUND)
574                     .header(HttpHeaders.LOCATION, privateAuthState.uri)
575                     .cookie(authCookie)
576                     .build();
577         } catch (IOException | URISyntaxException | InvalidPathException e) {
578             throw new SnapDataException(doc, e, "Unable to finish oauth2 login");
579         } catch (SnapDataException e) {
580             throw e.withDocument(doc);
581         }
582     }

669     /**
670      * Extracts the response body from the http response. Only works for application/json
671      */
672     private void extractResponseBody(String url, Map<String, Object> responseMap,
673                                      HttpResponse urlResponse, InputStream inputStream) {
674         Header contentTypeHeader = urlResponse.getFirstHeader(HttpHeaders.CONTENT_TYPE);
675         MediaType contentType;
676 
677         if (urlResponse.getStatusLine().getStatusCode() == Response.Status.OK.getStatusCode()) {
678             contentType = MediaType.APPLICATION_JSON_TYPE;
679         } else {
680             contentType = MediaType.TEXT_PLAIN_TYPE;
681         }
682 
683         if (contentTypeHeader != null) {
684             try {
685                 contentType = MediaType.valueOf(contentTypeHeader.getValue());
686             } catch (IllegalArgumentException e) {
687                 throw new SnapDataException(e, "Invalid Content-Type header from URL: %s")
688                         .formatWith(url)
689                         .withResolution("Ensure the endpoint is responding correctly");
690             }
691         }
692 
693         for (Map.Entry<MediaType, BiFunction<String, InputStream, Object>> entry :
694                 CONTENT_PARSERS.entrySet()) {
695             if (!contentType.isCompatible(MediaType.APPLICATION_JSON_TYPE)) {
696                 continue;
697             }
698             responseMap.put("entity", entry.getValue().apply(url, inputStream));
699         }
700 
701         if (!responseMap.containsKey("entity")) {
702             // No parser was found, so just convert the response into a string
703             throw new SnapDataException("Cannot parse the response. " +
704                     "Make sure the response is in application/json.");
705         }
706     }
