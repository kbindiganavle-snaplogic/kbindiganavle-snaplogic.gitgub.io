92      * Load model into memory.
93      *
94      * @param document Serialized model trained by Trainer Snap
95      * @return Mode type which can be classification or regression.
96      */
97     public DatasetType loadModel(Document document) throws IOException, ClassNotFoundException {
98 
99         boolean isLoadWekaModel = false;
100 
101         Map<String, Object> clsMap = document.get(Map.class);
102         if (!clsMap.containsKey(MODEL_MODEL) || !clsMap.containsKey(MODEL_METADATA)) {
103             throw new SnapDataException(ERR_LOAD_MODEL)
104                     .withReason(ERR_LOAD_MODEL_REASON)
105                     .withResolution(ERR_LOAD_MODEL_RESO);
106         }
107         Object metadata = clsMap.get(MODEL_METADATA);
108         if (metadata instanceof String) {
109             metadata = Serializer.deserialize((String) metadata);
110         }
111         if (metadata instanceof Map) {
112             Map<String, Object> metadataMap = (Map) metadata;
113             if (metadataMap.containsKey(MODEL_SUMMARY_ENGINE) && metadataMap.get(MODEL_SUMMARY_ENGINE).equals(AutoMLEngine.H2O_ENGINE.getDisplayName())) {
114                 engine = AutoMLEngine.H2O_ENGINE;
115                 labelField = (String) metadataMap.get(MODEL_LABEL_FIELD);
116                 fields = (Map) metadataMap.get(MODEL_FIELDS);
117                 modelType = metadataMap.get(MODEL_MODEL_TYPE).equals(
118                         CLASSIFICATION) ? DatasetType.CLASSIFICATION : DatasetType.REGRESSION;
119 
120                 Base64.Decoder decoder = Base64.getDecoder();
121                 InputStream modelStream = new ByteArrayInputStream(
122                         decoder.decode((String) clsMap.get(MODEL_MODEL)));
123 
124                 MojoReaderBackend reader = MojoReaderBackendFactory.createReaderBackend(modelStream,
125                         MojoReaderBackendFactory.CachingStrategy.MEMORY);
126                 MojoModel model = ModelMojoReader.readFrom(reader);
127                 h2oModel = new EasyPredictModelWrapper(model);
128                 EasyPredictModelWrapper.Config config = new EasyPredictModelWrapper.Config();
129                 config.setModel(model);
130                 config.setConvertUnknownCategoricalLevelsToNa(true);
131                 h2oModel = new EasyPredictModelWrapper(config);
132             } else if (metadataMap.containsKey(MODEL_SUMMARY_ENGINE) && metadataMap.get(MODEL_SUMMARY_ENGINE).equals(AutoMLEngine.WEKA_ENGINE.getDisplayName())) {
133 
134                 dataset = (Instances) metadataMap.get(MODEL_METADATA);
135                 isLoadWekaModel = true;
136 
137             } else {
138                 throw new SnapDataException(ERR_LOAD_MODEL)
139                         .withReason(ERR_LOAD_MODEL_REASON)
140                         .withResolution(ERR_LOAD_MODEL_RESO);
141             }
142         } else {
143             // support previous built model and model from trainer
144             dataset = (Instances) metadata;
145             isLoadWekaModel = true;
146         }
147 
148         if (isLoadWekaModel) {
149             String modelStr = (String) clsMap.get(MODEL_MODEL);
150             cls = (Classifier) Serializer.deserialize(modelStr);
151 
152             modelType = dataset.classAttribute().isNumeric() ?
153                     DatasetType.REGRESSION : DatasetType.CLASSIFICATION;
154             engine = AutoMLEngine.WEKA_ENGINE;
155         }
156 
157         return modelType;
158     }

161      * Get prediction for incoming document. And also for the documents in pending queue.
162      *
163      * @param document Null if it is the model. Else, it is the sample to be predicted.
164      * @param top      Maximum number of classes to be outputted.
165      * @param showConf Include confidence level in the output.
166      * @return List of predictions.
167      */
168     public List<Object> predict(Document document, int top, boolean showConf) throws SnapDataException {
169 
170         // Model is not ready. Add document to queue and return.
171         if ((engine.equals(AutoMLEngine.WEKA_ENGINE) && (cls == null || dataset == null)) ||
172                 (engine.equals(AutoMLEngine.H2O_ENGINE) && h2oModel == null)) {
173             if (document != null) {
174                 docQueue.add(document);
175             }
176             return new ArrayList<>();
177         }
178 
179         // Model is ready, process incoming document and all documents in queue.
180 
181         // Add document to queue to be processed.
182         if (document != null) {
183             docQueue.add(document);
184         }
185         List<Object> labList = new ArrayList<>();
186         while (!docQueue.isEmpty()) {
187             Document curDoc = docQueue.remove();
188             outputQueue.add(curDoc);
189             Map<String, Object> sample = curDoc.get(Map.class);
190             int validState = 0;
191 
192             // Weka engine.
193             if (engine.equals(AutoMLEngine.WEKA_ENGINE)) {
194                 double[] values = new double[dataset.numAttributes()];
195 
196                 // For each field in the document.
197                 for (Map.Entry<String, Object> entry : sample.entrySet()) {
198                     String colName = entry.getKey();
199 
200                     // Field is not required by the model.
201                     if (dataset.attribute(colName) == null ||
202                             dataset.attribute(colName).index() == dataset.classIndex()) {
203                         continue;
204                     }
205 
206                     String value = String.valueOf(entry.getValue());
207                     int colIndex = dataset.attribute(colName).index();
208                     boolean numericType = dataset.attribute(colName).isNumeric();
209 
210                     // Model requires numeric value in this field.
211                     if (numericType) {
212                         try {
213                             values[colIndex] = Double.parseDouble(value);
214                         } catch (NumberFormatException e) {
215                             labList.add(new SnapDataException(ERR_REQ_NUMERIC)
216                                     .formatWith(value, colName)
217                                     .withReason(ERR_REQ_NUMERIC_REASON)
218                                     .withResolution(ERR_REQ_NUMERIC_RESO));
219                             validState = 2;
220                             break;
221                         }
222                     }
223 
224                     // Model requires categorical value in this field.
225                     else {
226                         values[colIndex] = dataset.attribute(colIndex).indexOfValue(value);
227                         if (values[colIndex] == -1) {
228                             values[colIndex] = dataset.attribute(colIndex).indexOfValue(
229                                     MLKeys.UNKNOWN_VALUE);
230                         }
231                     }
232                     validState = 1;
233                 }
234 
235                 // Document is not valid. Either cannot parse to numeric or does not contain a
236                 // single field matching the model.
237                 if (validState != 1) {
238                     // No field matches the model. Basically, we cannot give prediction for empty
239                     // input.
240                     if (validState == 0) {
241                         labList.add(new SnapDataException(ERR_EMPTY_INPUT)
242                                 .withReason(ERR_EMPTY_INPUT_REASON)
243                                 .withResolution(ERR_EMPTY_INPUT_RESO));
244                     }
245                     continue;
246                 }
247 
248                 Instance inst = new DenseInstance(1.0, values);
249                 dataset.add(inst);
250 
251                 List<Object> predictions;
252                 try {
253 
254                     Instances is = dataset;
255 
256                     // Apply model.
257                     predictions = this.predict(is, top, showConf);
258                     labList.addAll(predictions);
259                 } catch (Exception e) {
260                     labList.add(new SnapDataException(e, ERR_PREDICT)
261                             .withReason(ERR_PREDICT_REASON)
262                             .withResolution(ERR_PREDICT_RESO));
263                 } finally {
264                     dataset.delete();
265                 }
266             }
267 
268             // H2O engine.
269             else if (engine.equals(AutoMLEngine.H2O_ENGINE)) {
270                 RowData row = new RowData();
271                 // For each fields in the model.
272                 for (Map.Entry<String, String> field : fields.entrySet()) {
273                     String fieldName = field.getKey();
274                     String fieldType = field.getValue();
275 
276                     // Field not found.
277                     if (fieldName.equals(labelField) || !sample.containsKey(fieldName)) {
278                         continue;
279                     }
280 
281                     Object value = sample.get(fieldName);
282 
283                     // Model requires numeric value in this field.
284                     if (fieldType.equals("numeric")) {
285                         double valueNumeric;
286                         try {
287                             valueNumeric = Double.parseDouble(String.valueOf(value));
288                         } catch (NumberFormatException e) {
289                             labList.add(new SnapDataException(ERR_REQ_NUMERIC)
290                                     .formatWith(value, fieldName)
291                                     .withReason(ERR_REQ_NUMERIC_REASON)
292                                     .withResolution(ERR_REQ_NUMERIC_RESO));
293                             validState = 2;
294                             break;
295                         }
296                         row.put(fieldName, valueNumeric);
297                     }
298                     // Model requires categorical value in this field.
299                     else {
300                         row.put(fieldName, String.valueOf(value));
301                     }
302                     validState = 1;
303                 }
304 
305                 // Document is not valid. Either cannot parse to numeric or does not contain a
306                 // single field matching the model.
307                 if (validState != 1) {
308                     // No field matches the model. Basically, we cannot give prediction for empty
309                     // input.
310                     if (validState == 0) {
311                         labList.add(new SnapDataException(ERR_EMPTY_INPUT)
312                                 .withReason(ERR_EMPTY_INPUT_REASON)
313                                 .withResolution(ERR_EMPTY_INPUT_RESO));
314                     }
315                     continue;
316                 }
317 
318                 datasetH2O.add(row);
319 
320                 List<Object> predictions;
321                 try {
322                     predictions = this.predictH2O(datasetH2O, top, showConf);
323                     labList.addAll(predictions);
324                 } catch (PredictException e) {
325                     labList.add(new SnapDataException(e, ERR_PREDICT)
326                             .withReason(ERR_PREDICT_REASON)
327                             .withResolution(ERR_PREDICT_RESO));
328                 } finally {
329                     datasetH2O.clear();
330                 }
331             }
332         }
333         return labList;
334     }

336     /**
337      * Get predictions for instances.
338      *
339      * @param unlabInstances Instances to be predicted.
340      * @param top            Maximum number of classes to be outputted.
341      * @param showConf       Include confidence level in the output.
342      * @return List of predictions.
343      */
344     public List<Object> predict(Instances unlabInstances, int top, boolean showConf) throws Exception {
345         List<Object> labList = new ArrayList<>();
346         Instances labInstances = new Instances(unlabInstances);
347         for (int i = 0; i < unlabInstances.numInstances(); i++) {
348             // Regression
349             if (unlabInstances.classAttribute().isNumeric()) {
350                 double predVal = cls.classifyInstance(unlabInstances.instance(i));
351                 labInstances.instance(i).setClassValue(predVal);
352                 Object predList = BigDecimal.valueOf(predVal);
353                 labList.add(predList);
354             }
355             // Classification
356             else {
357                 List<Object> predList = new ArrayList<>();
358                 double[] predProb = cls.distributionForInstance(unlabInstances.instance(i));
359                 List<List<Object>> topList = this.getTop(predProb, top);
360                 for (List<Object> pred : topList) {
361                     BigInteger topPredIndex = (BigInteger) pred.get(0);
362                     BigDecimal topPredProb = (BigDecimal) pred.get(1);
363                     String predLabStr = unlabInstances.classAttribute().value(topPredIndex.intValue());
364                     if (showConf) {
365                         List<Object> predPair = new ArrayList<>();
366                         predPair.add(predLabStr);
367                         predPair.add(topPredProb);
368                         predList.add(predPair);
369                     } else {
370                         predList.add(predLabStr);
371                     }
372                 }
373                 if (top == 1) {
374                     labList.add(predList.get(0));
375                 } else {
376                     labList.add(predList);
377                 }
378             }
379         }
380         unlabInstances.delete();
381         return labList;
382     }

384     /**
385      * Get predictions for rows.
386      *
387      * @param datasetH2O List of rows to be predicted.
388      * @param top        Maximum number of classes to be outputted.
389      * @param showConf   Include confidence level in the output.
390      * @return List of predictions.
391      */
392     public List<Object> predictH2O(List<RowData> datasetH2O, int top, boolean showConf) throws PredictException {
393         List<Object> labList = new ArrayList<>();
394         for (RowData row : datasetH2O) {
395 
396             // Regression
397             if (modelType.equals(DatasetType.REGRESSION)) {
398                 AbstractPrediction prediction = h2oModel.predict(row);
399                 double predVal = ((RegressionModelPrediction) prediction).value;
400                 labList.add(BigDecimal.valueOf(predVal));
401             }
402 
403             // Classification
404             else {
405                 AbstractPrediction prediction = h2oModel.predict(row);
406                 double[] probs = new double[0];
407 
408                 // Multiclass classificaion.
409                 if (prediction instanceof MultinomialModelPrediction) {
410                     probs = ((MultinomialModelPrediction) prediction).classProbabilities;
411                 }
412                 // Binary classification.
413                 else if (prediction instanceof BinomialModelPrediction) {
414                     probs = ((BinomialModelPrediction) prediction).classProbabilities;
415                 }
416 
417                 // Get predictions as list of <label,confidence>.
418                 String[] classes = h2oModel.getResponseDomainValues();
419                 List<Pair<String, BigDecimal>> pairs = new ArrayList<>();
420                 for (int i = 0; i < classes.length; i++) {
421                     pairs.add(Pair.of(classes[i], BigDecimal.valueOf(probs[i])));
422                 }
423 
424                 // Sort by confidence level.
425                 Collections.sort(pairs, (o1, o2) -> {
426                     int result = 0 - o1.getValue().compareTo(o2.getValue());
427                     return result == 0 ? 0 - o1.getKey().compareTo(o2.getKey()) : result;
428                 });
429 
430                 // Prepare output.
431                 List<Object> predList = new ArrayList<>();
432                 for (int i = 0; i < top && i < pairs.size(); i++) {
433                     String predLabStr = pairs.get(i).getKey();
434                     BigDecimal topPredProb = pairs.get(i).getValue();
435                     if (showConf) {
436                         List<Object> predPair = new ArrayList<>();
437                         predPair.add(predLabStr);
438                         predPair.add(topPredProb);
439                         predList.add(predPair);
440                     } else {
441                         predList.add(predLabStr);
442                     }
443                 }
444                 if (top == 1) {
445                     labList.add(predList.get(0));
446                 } else {
447                     labList.add(predList);
448                 }
449             }
450         }
451         datasetH2O.clear();
452         return labList;
453     }

462     /**
463      * Get top predictions based on confidence level.
464      *
465      * @param arr Array containing confidence levels of classes.
466      * @param top Maximum number of classes to be outputted.
467      * @return
468      */
469     private List<List<Object>> getTop(double[] arr, int top) {
470         if (arr.length < top) {
471             top = arr.length;
472         }
473         Set<Integer> topSet = new HashSet<>();
474         List<List<Object>> topList = new ArrayList<>();
475         // Top is always small number. Double loops are efficient.
476         for (int i = 0; i < top; i++) {
477             int maxIndex = 0;
478             for (int j = 0; j < arr.length; j++) {
479                 maxIndex = arr[j] > arr[maxIndex] ? j : maxIndex;
480             }
481             if (!topSet.contains(maxIndex)) {
482                 List<Object> predList = new ArrayList<>();
483                 predList.add(BigInteger.valueOf(maxIndex));
484                 predList.add(BigDecimal.valueOf(arr[maxIndex]));
485                 topList.add(predList);
486                 arr[maxIndex] = -1;
487             } else {
488                 break;
489             }
490         }
491         return topList;
492     }
