171     protected void processDocument(final Document document, final String inputViewName) {
172         try {
173             // Check if this is the schema document and save it for later.
174             if (inputViewName != null && inputViewName.equals(schemaViewName)) {
175                 if (document != null) {
176                     schema = document;
177                     columnTranslation = databaseUtils.buildColumnTranslation(schema);
178                 }
179                 return;
180             }
181             if (document == null) {
182                 return;
183             }
184             if (firstRecord) {
185                 firstRecord = false;
186                 if (autoCreateTable) {
187                     createTableIfAbsent(document);
188                 }
189                 Connection conn = jdbcOperations.acquireConnection(account);
190                 tableMetaData = jdbcOperations.getTableMetaData(conn,
191                         account.getDatabaseConfig().getDatabaseName(), tableName);
192                 if (!setInputSchemaFields()) {
193                     return;
194                 }
195             }
196             if (document.get() instanceof Map) {
197                 Map<String, Object> record = (Map<String, Object>) document.get();
198                 if (keys == null) {
199                     keys = record.keySet();
200                 }
201                 for (int i = 0; i < inputViewFields.size(); ++i) {
202                     String actualKey = inputViewFields.get(i);
203                     String mappedKey = null;
204                     for (String key : keys) {
205                         if (key.equalsIgnoreCase(actualKey)) {
206                             mappedKey = key;
207                             break;
208                         }
209                     }
210                     Object value;
211                     if (mappedKey == null) {
212                         value = null;
213                     } else {
214                         value = record.get(mappedKey);
215                     }
216                     if (value != null) {
217                         dataWriter.write(stringifyValue(mappedKey, value));
218                     }
219                     if (i < (inputViewFields.size() - 1)) {
220                         dataWriter.write(fieldDelimiter);
221                     }
222                 }
223             }
224             dataWriter.write(rowTerminator);
225             dataWriter.flush();
226         } catch (IOException e) {
227             SnapDataException dataException = new SnapDataException(e, ERR_WRITE_TO_DATA_FILE_MSG)
228                     .withReason(e.getMessage());
229             if (e.getMessage().toLowerCase().contains(NOT_ENOUGH_SPACE.toLowerCase())) {
230                 dataException = dataException.withResolution(ERR_NOT_ENOUGH_SPACE_RESOLUTION);
231             } else {
232                 dataException = dataException.withResolutionAsDefect();
233             }
234             errorViews.write(dataException);
235             isErrorOccurred = true;
236         }
237     }

239     private String stringifyValue(String key, @Nonnull Object obj) {
240         String value;
241         if (obj instanceof DateTime) {
242             // convert datetime to "wall-time" in UTC zone (no-op if zone is UTC)
243             DateTime dateTimeUTC = ((DateTime) obj).withZone(DateTimeZone.UTC);
244 
245             // TODO: research why SQL Server doesn't support ISO 8601 format
246             // Ex: 2020-09-11T10:05:23
247             //  -> SQLState = 22005, Invalid character value for cast specification
248             value = dateTimeUTC.toString(DATE_TIME_FORMAT);
249 
250         } else if (obj instanceof LocalDateTime) {
251             value = ((LocalDateTime) obj).toString(DATE_TIME_FORMAT);
252         } else {
253             value = stringValueForDatatype(key, obj);
254         }
255         return value;
256     }

258     private String stringValueForDatatype(String key, @Nonnull Object obj) {
259         String value = obj.toString();
260 
261         if (tableMetaData != null) {
262             Integer dataType = tableMetaData.getColumnDataType(key);
263             if (dataType != null) {
264                 switch (dataType) {
265                     case Types.BIT:
266                         value = coerceBitFieldValue(value);
267                         break;
268                     case Types.CHAR:
269                     case Types.VARCHAR:
270                     case Types.NCHAR:
271                     case Types.NVARCHAR:
272                         value = updateEmptyStringValue(value);
273                         break;
274                 }
275             }
276         }
277         return value;
278     }

280     private String updateEmptyStringValue(@Nonnull String value) {
281         if ("".equals(value)) {
282             // In order to load an empty string into SQL Server, you need to set it to NUL (\0)
283             // This then appropriately translates to an empty string as the value in SQL Server
284             // See SNAP-11048 for more details on the investigation of this change.
285             return "\0";
286         }
287         return value;
288     }

290     private String coerceBitFieldValue(@Nonnull String value) {
291         if ("true".equalsIgnoreCase(value)) {
292             return "1";

294             return "0";

304     @Override
305     public void execute() {
306         if (inputViews.isEmpty()) {
307             return;
308         }
309 
310         try {
311             dataWriter = new BufferedWriter(new FileWriter(dataFileName));
312             for (String viewName : getInputViewOrder()) {
313                 for (Document document : inputViews.documents(inputViews.get(viewName))) {
314                     process(document, viewName);
315                     if (isErrorOccurred) {
316                         return;
317                     }
318                 }
319             }
320         } catch (IOException e1) {
321             SnapDataException dataException = new SnapDataException(e1, ERR_CREATE_DATA_FILE_MSG)
322                     .withReason(e1.getMessage());
323             if (e1.getMessage().toLowerCase().contains(NOT_ENOUGH_SPACE.toLowerCase())) {
324                 dataException = dataException.withResolution(ERR_NOT_ENOUGH_SPACE_RESOLUTION);
325             } else {
326                 dataException = dataException.withResolutionAsDefect();
327             }
328             errorViews.write(dataException);
329             return;
330         } finally {
331             IOUtils.closeQuietly(dataWriter);
332         }
333 
334         if (!firstRecord && autoCreateTable && schema != null) {
335             // We want to create the table even if there are no rows to be loaded.
336             createTableIfAbsent(null);
337         }
338         prepareScriptFileFromCommands();
339 
340         if (inputViews.get(rowViewName).getObjectCount() > 0) {
341             invokeBcpCommand();
342         }
343     }

347         Map<String, String> map = new HashMap<>(1);
348         String[] logLines = logInfo.split(TAB);
349         String result = logInfo;
350         Pattern resultPattern = Pattern.compile(RESULT_PATTERN);
351         for (String logLine : logLines) {
352             if (resultPattern.matcher(logLine).matches()) {
353                 result = logLine;
354                 // result only contains part of the bcp output info, log the whole bcp output info
355                 // for debugging purpose.
356                 LOGGER.debug(logInfo);
357                 break;
358             }
359         }
360         map.put(RESULT, result);
361         outputViews.write(documentUtility.newDocument(map));
362     }

365     protected String generateScriptCommands(DatabaseAccount account) {
366         DatabaseConfig dbConfig = account.getDatabaseConfig();
367         String rawTableName = StringUtils.remove(tableName, DOUBLE_QUOTE);
368 
369         // Get host name and port number from account configuration
370         String url = dbConfig.getJdbcUrl();
371         url = url.substring(url.indexOf(JDBC_URL_HOST_PREFIX) + JDBC_URL_HOST_PREFIX.length());
372         int colonIndex = url.indexOf(COLON);
373 
374         String host = url;
375         String port = null;
376         if (colonIndex >= 0) {
377             host = url.substring(0, colonIndex);
378             port = url.substring(colonIndex + 1);
379             if (port.indexOf(SEMICOLON) > 0) {
380                 port = port.substring(0, port.indexOf(SEMICOLON));
381             }
382         }
383         String batchSizeParam = "";
384         if (batchSize >= 1) {
385             batchSizeParam = String.format( "-b %s",batchSize);
386         }
387         String windowsCommand = ((BaseDatabaseAccount)account).isWindowsAuth() ? "-T" : "";
388         String finalDataCommand = String.format(BCP_COMMAND_WRITE, bcpPath, rawTableName,
389                 dataFileName, doEscape(host, port), doEscape(dbConfig.getDatabaseName()),
390                 doEscape(dbConfig.getUsername()), doEscape(dbConfig.getPassword()),
391                 FIELD_DELIMITER_IN_SCRIPT, errorFileName, ROW_TERMINATOR_IN_SCRIPT, maxError,
392                 batchSizeParam, windowsCommand);
393         String maskedDataCommand = String.format(BCP_COMMAND_WRITE, bcpPath, rawTableName,
394                 dataFileName, doEscape(host, port), doEscape(dbConfig.getDatabaseName()),
395                 doEscape(dbConfig.getUsername()), MASK_STR, FIELD_DELIMITER_IN_SCRIPT,
396                 errorFileName, ROW_TERMINATOR_IN_SCRIPT, maxError, batchSizeParam,

410     protected boolean setInputSchemaFields() {
411         inputViewFields = new ArrayList<>();
412         ResultSet resultSet = null;
413         try {
414             Connection connection = jdbcOperations.acquireConnection(account);
415             DatabaseMetaData metaData = connection.getMetaData();
416             String[] tableParts = quotationHandler.splitFullyQualifiedName(tableName, 0);
417             String table = tableParts[tableParts.length - 1];
418             resultSet = metaData.getColumns(null, quotationHandler.unescape(schemaName),
419                     quotationHandler.unescape(table), null);
420             while (resultSet.next()) {
421                 inputViewFields.add(resultSet.getString(COLUMN_NAME));
422             }
423             return true;
424         } catch (SQLException e1) {
425             if (!isInputSchemaErrorCaptured) {
426                 isInputSchemaErrorCaptured = true;
427                 SnapDataException dataException = new SnapDataException(e1,
428                         ERR_FETCH_TABLE_SCHEMA_MSG)
429                         .withReason(e1.getMessage())
430                         .withResolution(ERR_FETCH_TABLE_SCHEMA_RESOLUTION);
431                 errorViews.write(dataException);
432             }
433             return false;
434         } finally {
435             JDBCUtils.safeClose(resultSet);
436         }
437     }
