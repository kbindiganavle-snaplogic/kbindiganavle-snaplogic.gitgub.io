211             channel = channelExp.eval(document);
212             if (StringUtils.isBlank(channel)) {
213                 throw new SnapDataException(ERR_BAD_INPUT)
214                         .withReason(ERR_MISSING_CHANNEL)
215                         .withResolution(ERR_MISSING_CHANNEL_RESOLUTION);
216             }
217             int colonIndex = channel.indexOf(COLON);
218             if (colonIndex != -1) {
219                 channel = channel.split(COLON)[1];
220             }
221         }
222         switch (channelOperation) {
223             case OPERATION_RENAME_CHANNEL:
224             case OPERATION_CREATE_CHANNEL:
225                 newChannelName = newChannelNameExp.eval(document);
226                 if (StringUtils.isBlank(newChannelName)) {
227                     throw new SnapDataException(ERR_BAD_INPUT)
228                             .withReason(ERR_MISSING_CHANNEL_NAME)

237                             .withReason(ERR_MISSING_USER)

244     private List<Map> listChannelMembers() throws IOException, SlackApiException {
245         String nextCursor = null;
246         List<Map> maps = new ArrayList<>();
247 
248         do {
249             ConversationsMembersResponse response = makeConversationsMembersRequest(nextCursor);
250             if (response.isOk()) {
251                 nextCursor = response.getResponseMetadata().getNextCursor();
252                 List<String> members = response.getMembers();
253                 for (String member : members) {
254                     Map<String, Object> record = new LinkedHashMap<>();
255                     record.put(OUTPUT_FIELD_MEMBER, member);
256                     maps.add(record);
257                 }
258             } else {
259                 throw new SnapDataException(ERR_LIST_CHANNEL_MEMBERS)
260                         .withReason(response.getError())
261                         .withResolution(ERR_COMMON_RESOLUTION);
262             }
263         } while (StringUtils.isNotBlank(nextCursor));
264 
265         return maps;
266     }

297     private void putChannelInfo(Conversation conversation, Map map) {
298         map.put(OUTPUT_FIELD_ID, conversation.getId());
299         map.put(OUTPUT_FIELD_NAME, conversation.getName());
300         map.put(OUTPUT_FIELD_CREATED, conversation.getCreated());
301         map.put(OUTPUT_FIELD_IS_CHANNEL, conversation.isChannel());
302         map.put(OUTPUT_FIELD_IS_GROUP, conversation.isGroup());
303         map.put(OUTPUT_FIELD_IS_ARCHIVED, conversation.isArchived());
304         map.put(OUTPUT_FIELD_CREATOR, conversation.getCreator());
305         Topic topic = conversation.getTopic();
306         if (topic != null) {
307             map.put(OUTPUT_FIELD_TOPIC, conversation.getTopic().getValue());
308         }
309         Purpose purpose = conversation.getPurpose();
310         if (purpose != null) {
311             map.put(OUTPUT_FIELD_PURPOSE, conversation.getPurpose().getValue());
312         }
