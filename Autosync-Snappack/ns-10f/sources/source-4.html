257     public void process(Document document, String inputViewName) {
258         if (++count == STATUS_REPORT_INTERVAL_IN_DOCS) {
259             monitorResource();
260             count = 0;
261         } else if (count > STATUS_REPORT_INTERVAL_IN_DOCS) {
262             // The Snap has already failed because the available memory or disk space was less than
263             // the threshold. Do not process input documents.
264             return;
265         }
266         try {
267             if (!jsonPathFieldSet.isEmpty()) {
268 
269                 Set<JsonPath> iteratedSet = new LinkedHashSet<>(jsonPathFieldSet);
270                 Map docMap = document.get(Map.class);
271 
272                 for (JsonPath fieldPath : iteratedSet) {
273                     try {
274                         Object value = fieldPath.read(docMap);
275                         if (value == null)
276                             continue;
277                         jsonPathFieldSet.remove(fieldPath);
278                     } catch (InvalidPathException e) {
279                     }
280                 }
281             }
282 
283             masterDataset.add(document);
284             writeToJson(document, masterDocIndex);
285             masterDocIndex++;
286 
287         } catch (SnapDataException e) {
288             errorViews.write(e, document);
289         } catch (SuggestViewAbortException e) {
290             // SuggestViewAbortException is thrown when jcc needs to stop the pipeline and should
291             // not be suppressed by wrapping it with SnapDataException and written into the error
292             // view. It should be thrown back so that the pipeline can stop gracefully.
293             throw e;
294         } catch (Exception e) {
295             SnapDataException ex = new SnapDataException(e, ERR_PROCESS_DOC)
296                     .withResolution(RESOLUTION_ADDRESS_ISSUE);
297             errorViews.write(ex, document);

305                 try {
306                     fileWriter1.close();
307

309                         return;
310                     }
311 
312                     if (!jsonPathFieldSet.isEmpty()) {
313                         throw new SnapDataException(ERR_POLICY_FIELD_MISSING)
314                                 .withReason(ERR_POLICY_FIELD_MISSING_REASON, jsonPathFieldSet.toString())

320                     this.proc = new Processor(config);

322                     MatchResult matchResult = new MatchResult(masterDataset);
323                     this.proc.addMatchListener(matchResult);
324                     // Execute proc.deduplicate() in a separate thread and monitor available
325                     // memory and free disk space. If available resource is less than the minimum
326                     // threshold, stop to deduplicate to avoid out-of-memory crash.
327                     launchAndMonitorDeduplicateTask();
328 
329                     // Get output views.
330                     DocumentOutputView matchOutputView =
331                             outputViews.getDocumentViewFor(outputViewNameList.get(0));

333                             outputViews.getDocumentViewFor(outputViewNameList.get(1)) : null;
334 
335                     DuplicateSet[] duplicateSets = matchResult.getMapping();
336 
337                     // first iteration provides first output
338                     int groupId = 0;
339                     for (int i = 0; i < masterDataset.size(); i++) {
340 
341                         // if no duplicate set at this index, this index is not duplicate with others
342                         // if there is duplicate set at this index and the current index is representative data,
343                         // we will calculate group id and set the id to the group
344                         // if is is not representative index, we will send it out in the second iteration
345                         if (duplicateSets[i] == null || i == duplicateSets[i].getRepresentativeIndex()) {
346                             groupId++;
347                             if (includeGroupID) {
348                                 Map<String, Object> record = (Map<String, Object>) masterDataset.get(i).get();
349                                 String key = GROUP_ID;
350                                 int suffixIndex = 1;
351                                 while (record.containsKey(key)) {
352                                     key = GROUP_ID_PREFIX + suffixIndex;
353                                     suffixIndex++;
354                                 }
355                                 record.put(key, groupId);
356                                 matchOutputView.write(documentUtility.newDocument(record));
357                             } else {
358                                 matchOutputView.write(masterDataset.get(i));
359                             }
360 
361                             if (duplicateSets[i] != null)
362                                 duplicateSets[i].setGroupId(groupId);
363                         }
364                     }
365 
366                     // the second iteration will focus only provide duplicated data
367                     if (duplicateOutputView1 != null) {
368 
369                         for (int i = 0; i < masterDataset.size(); i++) {
370 
371                             if (duplicateSets[i] != null && i != duplicateSets[i].getRepresentativeIndex()) {
372 
373                                 DuplicateSet duplicateSet = duplicateSets[i];
374 
375                                 Map<String, Object> docMap1 = masterDataset.get(i).get(Map.class);
376                                 Optional<Double> confidenceValue = duplicateSet.getConfidenceWithRepresentative(i);
377                                 double confidence;
378 
379                                 if (confidenceValue.isPresent()) {
380 
381                                     confidence = confidenceValue.get();
382                                 } else {
383 
384                                     // if there is no pair recorded, we need to get confidence value from duke processor
385                                     Record firstRecord = createRecord(masterDataset.get(i), 1);
386 
387                                     int representativeIndex = duplicateSet.getRepresentativeIndex();
388                                     Record secondRecord = createRecord(masterDataset.get(representativeIndex), 2);
389 
390                                     confidence = this.proc.compare(firstRecord, secondRecord);
391                                 }
392 
393                                 if (includeConf) {
394                                     String key = CONFIDENCE;
395                                     int suffixIndex = 1;
396                                     while (docMap1.containsKey(key)) {
397                                         key = CONFIDENCE_PREFIX + suffixIndex;
398                                         suffixIndex++;
399                                     }
400                                     docMap1.put(key, BigDecimal.valueOf(confidence));
401                                 }
402                                 if (includeGroupID) {
403                                     String key = GROUP_ID;
404                                     int suffixIndex = 1;
405                                     while (docMap1.containsKey(key)) {
406                                         key = GROUP_ID_PREFIX + suffixIndex;
407                                         suffixIndex++;
408                                     }
409                                     docMap1.put(key, duplicateSet.getGroupId());
410                                 }
411                                 duplicateOutputView1.write(documentUtility.newDocument(docMap1));
412 
413                             }
414                         }
415                     }
416                 } catch (SnapDataException e) {
417                     errorViews.write(e);
418                 } catch (SuggestViewAbortException e) {
419                     throw e;
420                 } catch (Exception e) {
421                     SnapDataException ex = new SnapDataException(e, ERR_COMPLETE)
422                             .withResolution(RESOLUTION_ADDRESS_ISSUE);
423                     errorViews.write(ex);
424                 }
425                 break;
426             case STOP:

429         }

450     private void launchAndMonitorDeduplicateTask() {
451         // launch a separate thread for the deduplicate task
452         ExecutorService executorService = Executors.newSingleThreadExecutor();
453         CompletableFuture deduplicateFuture = CompletableFuture.runAsync(new DeduplicateTask(),
454                 executorService);
455         try {
456             // monitor available memory and disk
457             do {
458                 monitorResource();
459                 sleepNoInterrupt(MONITOR_RESOURCE_INTERVAL_IN_MILLIS, TimeUnit.MILLISECONDS, LOG);
460             } while (!deduplicateFuture.isDone());
461         } finally {
462             if (!deduplicateFuture.isDone()) {
463                 // stop the deduplicate thread if it's still in progress
464                 deduplicateFuture.cancel(true);
465             }
466             // clean up resources
467             executorService.shutdownNow();
468             proc.close();
469         }
470     }

472     private void monitorResource() {
473         long availableMemory = getAvailableMemory();
474         long freeDiskSpace = snapTmpDir.getUsableSpace();
475         String availableMemoryStr = byteCountDisplay(availableMemory);
476         String freeDiskSpaceStr = byteCountDisplay(freeDiskSpace);
477         statusReporter.notify(STATUS_MESSAGE, availableMemoryStr, freeDiskSpaceStr);
478         if (availableMemory < minMemory) {
479             masterDataset.clear();
480             throw new SnapDataException(ERR_STOPPED_TO_DUPLICATE)
481                     .formatWith(availableMemoryStr)
482                     .withReason(String.format(REASON_MEMORY_TOO_LOW, byteCountDisplay(minMemory)))
483                     .withResolution(RESOLUTION_MEMORY_TOO_LOW);
484         } else if (freeDiskSpace < minFreeDiskSpace) {
485             masterDataset.clear();
486             throw new SnapDataException(ERR_STOPPED_TO_DUPLICATE_LOW_DISK)
487                     .formatWith(freeDiskSpaceStr)
488                     .withReason(String.format(REASON_DISK_TOO_LOW,
489                             byteCountDisplay(minFreeDiskSpace)))
490                     .withResolution(RESOLUTION_DISK_TOO_LOW);
491         }
492     }

545         Map<String, Object> data = new LinkedHashMap<>();

560             policyIndex++;
561         }
562         return data;
