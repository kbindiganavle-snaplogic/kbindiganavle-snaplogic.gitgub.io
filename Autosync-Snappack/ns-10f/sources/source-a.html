274     public void handle(final LifecycleEvent event) {
275         switch (event) {
276             case SUCCESS:
277                 try {
278                     fileWriter1.close();
279                     fileWriter2.close();
280 
281                     // Start Duke job.
282                     Configuration config = ConfigLoader.load(confPath);
283                     this.proc = new Processor(config);
284                     this.proc.setThreads(runtimeResource.getRuntimeAvailableProcessors());
285 
286                     MatchResult matchResult;
287                     if (this.matchAll) {
288                         matchResult = new MatchAllResult(masterDocIndex, refDocIndex);
289                     } else {
290                         matchResult = new MatchResult(masterDocIndex, refDocIndex);
291                     }
292 
293                     this.proc.addMatchListener(matchResult);
294                     this.proc.link();
295                     this.proc.close();
296 
297                     // Get output views.
298                     DocumentOutputView matchOutputView =
299                             outputViews.getDocumentViewFor(outputViewNameList.get(0));
300                     Optional<DocumentOutputView> nonMatchOutputView1Opt = outputViewNameList.size() > 1 ?
301                             Optional.of(outputViews.getDocumentViewFor(outputViewNameList.get(1))) : Optional.empty();
302                     Optional<DocumentOutputView> nonMatchOutputView2Opt = outputViewNameList.size() > 2 ?
303                             Optional.of(outputViews.getDocumentViewFor(outputViewNameList.get(2))) : Optional.empty();
304 
305                     if (matchAll) {
306                         boolean[] refOutput = new boolean[refDataset.size()];
307                         for (int i=0; i<masterDataset.size(); i++) {
308                             List<String> matchList = ((MatchAllResult) matchResult).getAllMatches(i);
309                             List<Map<String, Object>> resultList = new ArrayList<>();
310                             if (matchList.size() > 0) {
311                                 for (String match : matchList) {
312                                     String[] matchSplit = match.split("_");
313                                     int refIndex = Integer.parseInt(matchSplit[0]);
314                                     refOutput[refIndex] = true;
315                                     double conf = Double.parseDouble(matchSplit[1]);
316                                     Map<String, Object> output = new LinkedHashMap<>();
317                                     if (includeConf) {
318                                         output.put(DATA, new LinkedHashMap<>(refDataset.get(refIndex).get(Map.class)));
319                                         output.put(CONFIDENCE, BigDecimal.valueOf(conf));
320                                     } else {
321                                         output = (Map<String, Object>) refDataset.get(refIndex).get();
322                                     }
323                                     resultList.add(output);
324                                 }
325                                 Map<String, Object> outputMap = new LinkedHashMap<>();
326                                 outputMap.put(MASTER_RECORD, new LinkedHashMap<>(masterDataset.get(i).get(Map.class)));
327                                 outputMap.put(MATCH_LIST, resultList);
328                                 matchOutputView.write(documentUtility.newDocument(outputMap));
329                             }
330                             // Output non-matched master documents.
331                             else {
332                                 var masterData = masterDataset.get(i);
333                                 nonMatchOutputView1Opt.ifPresent(view -> view.write(masterData));
334                             }
335                         }
336 
337                         // Output non-matched ref documents.
338                         for (int i=0; i < refOutput.length; i++) {
339                             if (!refOutput[i]) {
340                                 var refData = refDataset.get(i);
341                                 nonMatchOutputView2Opt.ifPresent(view -> view.write(refData));
342                             }
343                         }
344                     } else {
345                         boolean[] refMatch = new boolean[refDocIndex];
346                         for (int i = 0; i < masterDataset.size(); i++) {
347                             int refIndex = matchResult.getMapping(i);
348                             double confidence = matchResult.getConfidence(i);
349                             // Output matched documents.
350                             if (refIndex >= 0) {
351                                 Document result = merge(masterDataset.get(i),
352                                         refDataset.get(refIndex), confidence, includeConf);
353                                 matchOutputView.write(result);
354                                 refMatch[refIndex] = true;
355                             }
356                             // Output non-matched master documents.
357                             else {
358                                 var masterData = masterDataset.get(i);
359                                 nonMatchOutputView1Opt.ifPresent(view -> view.write(masterData));
360                             }
361                         }
362 
363                         // Output non-matched ref documents.
364                         if (nonMatchOutputView2Opt.isPresent()) {
365                             for (int i = 0; i < refMatch.length; i++) {
366                                 if (!refMatch[i]) {
367                                     var refData = refDataset.get(i);
368                                     nonMatchOutputView2Opt.ifPresent(view -> view.write(refData));
369                                 }
370                             }
371                         }
372                     }
373 
374                 } catch (Exception e) {
375                     SnapDataException ex = new SnapDataException(e, ERR_COMPLETE)

377                     errorViews.write(ex);

413     private void writeToJson(Document doc, int docIndex, int datasetIndex) {
414         Map<String, Object> docMap = doc.get(Map.class);
415         Map<String, Object> data = new LinkedHashMap<>();
416         data.put(ID, datasetIndex + "_" + docIndex);
417         int policyIndex = 0;
418         for (RecordLinkageProperty policy : policies) {
419             JsonPath path = datasetIndex == 1 ? policy.getMasterField() : policy.getRefField();
420             Object obj;
421             try {
422                 obj = path.read(docMap);
423                 if (obj == null) {
424                     obj = "";
425                 }
426             } catch (InvalidPathException e) {
427                 obj = "";
428             }
429             data.put(getPolicyColumn(policyIndex), obj);
430             policyIndex++;
431         }
432         String dataString;
433         try {
434             dataString = objectMapper.writeValueAsString(data);
435         } catch (JsonProcessingException e) {
436             throw new SnapDataException(e, ERR_PROCESS_DATA).formatWith(docIndex)
437                     .withResolution(RESOLUTION_ADDRESS_ISSUE);
438         }
439         PrintWriter writer = datasetIndex == 1 ? fileWriter1 : fileWriter2;
440         writer.write(dataString);
441         writer.write("\n");
442     }

444     /**
445      * Build Duke configuration file.
446      */
447     private void buildConf() {
448         int policyIndex = 0;
449 
450         String properties = "";
451         String columns1 = "";
452         String columns2 = "";
453         for (RecordLinkageProperty policy : policies) {
454             if (policy.getLow() >= policy.getHigh()) {
455                 throw new ConfigurationException(ERR_LOW_GTE_HIGH_REASON, ERR_OPTION_NOT_AVAI);
456             }
457             String colName = getPolicyColumn(policyIndex);
458             String comparatorClass = policy.getComparator();
459             String cleanerClass = policy.getCleaner();
460             properties += String.format(SNAP_LOCALE, PROPERTY, colName, comparatorClass, policy.getLow(),
461                     policy.getHigh());
462             columns1 += String.format(SNAP_LOCALE, COLUMN, colName, colName, cleanerClass);
463             columns2 += String.format(SNAP_LOCALE, COLUMN, colName, colName, cleanerClass);
464             policyIndex++;
465         }
466 
467         String conf = String.format(SNAP_LOCALE, RECORD_LINKAGE_CONF_TEMPLATE, threshold, properties,
468                 filePath1, columns1, filePath2, columns2);
469 
470         confWriter.write(conf);
471         confWriter.close();
472     }

474     /**
475      * Merge two documents and append confidence if specified.
476      */
477     private Document merge(Document doc1, Document doc2, double confidence,
478                            boolean includeConfidence) {
479         Map<String, Object> docMap1 = doc1.get(Map.class);
480         if (!(docMap1 instanceof LinkedHashMap)) {
481             docMap1 = new LinkedHashMap<>(docMap1);
482         }
483         Map<String, Object> docMap2 = doc2.get(Map.class);
484 
485         for (Map.Entry<String, Object> entry : docMap2.entrySet()) {
486             String key = entry.getKey();
487             Object value = entry.getValue();
488             while (docMap1.containsKey(key)) {
489                 key = INPUT1_PREFIX + key;
490             }
491             docMap1.put(key, value);
492         }
493 
494         if (includeConfidence) {
495             String key = CONFIDENCE;
496             Object value = BigDecimal.valueOf(confidence);
497             int suffixIndex = 1;
498             while (docMap1.containsKey(key)) {
499                 key = CONFIDENCE_PREFIX + suffixIndex;
500                 suffixIndex++;
501             }
502             docMap1.put(key, value);
503         }
504 
505         return documentUtility.newDocument(docMap1);
506     }
