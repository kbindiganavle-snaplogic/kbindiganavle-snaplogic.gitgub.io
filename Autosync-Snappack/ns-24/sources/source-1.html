14     public void validate(String contentType, Set<String> invalidContentTypes) {
15         /* Parse charset parameter from content type. Assume RFC 2046 Content Type formatting:
16          *         type / subtype ["+" suffix]*; [ parameter=value ];
17          *         [       Type (first)       ]; [  params follow  ]; ....
18          */
19         String[] contentTypeSplit = contentType.toLowerCase().split(";");
20         String type = contentTypeSplit[0].trim();
21 
22         // Validate the type
23         if (!allowableContentTypes.contains(type)) {
24             invalidContentTypes.add(contentType);
25             return;
26         }
27 
28         if (contentTypeSplit.length == 1) {
29             return; // no more checks to perform
30         }
31 
32         /* Now validate the charset.
33          * May contain other parameters for message types with user-defined params, just get the
34          * block with the charset. Start looking at index 1 since TYPE is already processed.
35          *
36          *
37          * Check that only one charset key/value pair is defined. Invalidate multiple definitions
38          * and invalidate those that don't strictly follow format in RFC 2046 (i.e. include spaces).
39          */
40         boolean firstAndOnlyCharset = true;
41         String charsetBlock = "";
42         for (int i = 1; i < contentTypeSplit.length; i++) {
43             charsetBlock = contentTypeSplit[i].stripLeading(); // may have a leading space
44             if (charsetBlock.contains("charset=")) {
45                 if (firstAndOnlyCharset) {
46                     firstAndOnlyCharset = false;
47 
48                     if (charsetBlock.contains(" ")) {
49                         invalidContentTypes.add(contentType);
50                         return;
51                     }
52                 } else {
53                     // Identified more than one definition identified
54                     invalidContentTypes.add(contentType);
55                     return;
56                 }
57             }
58         }
59 
60         if (charsetBlock.isEmpty()) {
61             return;
62         }
63 
64         String charsetValue = "";
65         String[] charsetBlockSplit = charsetBlock.split("=");
66         if (charsetBlockSplit.length > 1) {
67             // May include "" around the defined charset value. Remove these to evaluate
68             charsetValue = charsetBlockSplit[1].trim().replaceAll("\"", "");
69         }
70 
71         if (!allowableCharsets.contains(charsetValue)) {
72             invalidContentTypes.add(contentType);
73         }
74     }
