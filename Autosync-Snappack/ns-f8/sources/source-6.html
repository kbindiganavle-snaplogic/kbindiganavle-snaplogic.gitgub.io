189     public void process(Document document, String inputViewName) {
190         try {
191             Map<String, Object> data = document.get(Map.class);
192             if (isFirstDocument) {
193                 isFirstDocument = false;
194                 // init CSV headers and CSV builder
195                 csvHeaders = List.of(data.keySet().toArray(new String[0]));
196                 // TODO maybe use CSV library as in other Snaps
197                 csvBuilder = new StringBuilder(join(COMMA, csvHeaders));
198             }
199             Set<String> values = new LinkedHashSet<>(data.size());
200             for (String header : csvHeaders) {
201                 Object value = data.get(header);
202                 values.add(value == null ? StringUtils.EMPTY : value.toString());
203             }
204             String csvRecord = join(COMMA, values);
205             if (csvBuilder.length() + csvRecord.length() + 2 > MAX_BATCH_SIZE) {
206                 // If the CSV builder is about to exceed the limit, process the bulk import batch.
207                 processBatch();
208                 // init a new CSV builder
209                 csvBuilder = new StringBuilder(join(COMMA, csvHeaders));
210             }
211             // write a CSV record
212             csvBuilder.append(NEW_LINE).append(csvRecord);
213         } catch (SuggestViewAbortException e) {
214             throw e;
215         } catch (SnapDataException e) {
216             errorViews.write(e, document);
217         } catch (Exception e) {
218             errorViews.write((SnapDataException)
219                     new SnapDataException(e, "Failed to prepare CSV data for: %s")
220                             .formatWith(bulkImportConfig.entity)
221                             .withResolution(RESOLUTION_ADDRESS_ISSUE));
222         } finally {
223             document.acknowledge();
224         }
225     }

279     private String createBatch() throws Exception {
280         // init input map data
281         inputMap = new HashMap<>();
282         try {
283             // prepare for multipart entity
284             openApiHelper = new BulkImportHelper(v2CompatibleOpenApiSpec);
285             // init openApiPath & operation to create bulk-import batch job
286             openApiPath = paths.get(0);
287             operation = HttpPost.METHOD_NAME.toLowerCase();
288             switch (bulkImportConfig.entity) {
289                 case LEADS:
290                     writeValueToInput(LIST_ID_PARAMETER, bulkImportConfig.listId);
291                     writeValueToInput(LOOKUP_FIELD_PARAMETER, bulkImportConfig.lookupField);
292                     writeValueToInput(PARTITION_NAME_PARAMETER, bulkImportConfig.partitionName);
293                     break;
294                 case PROGRAM_MEMBERS:
295                     writeValueToInput(PROGRAM_ID_PARAMETER, bulkImportConfig.programId);
296                     writeValueToInput(PROGRAM_MEMBER_STATUS_PARAMETER,
297                             bulkImportConfig.programMemberStatus);
298                     break;
299                 case CUSTOM_OBJECTS:
300                     writeValueToInput(API_NAME_PARAMETER, bulkImportConfig.apiName);
301                     break;
302                 default:
303                     throw new SnapDataException(FMT_ERROR_UNSUPPORTED_MARKETO_ENTITY)
304                             .formatWith(bulkImportConfig.entity)
305                             .withResolutionAsDefect();
306             }
307             compile("$.parameters.query.format").write(inputMap, CSV);
308             compile("$.requestBody.file").write(inputMap, csvBuilder.toString());
309             // create bulk-import batch job
310             outDoc = failsafeOAuthProcess(inputMap, null);
311             if (outDoc != null) {
312                 // check if bulk-import batch job is successfully created
313                 Map<String, Object> outMap = outDoc.get(Map.class);
314                 if (isSuccess(outMap)) {
315                     Object value = readJsonPath(outMap, ENTITY, RESULT_0, "batchId");
316                     if (value != null) {
317                         // batch ID found
318                         return value.toString();
319                     }
320                 }
321                 // throw an exception
322                 Object value = readJsonPath(outMap, ENTITY, "errors");
323                 throw new SnapDataException("Failed to create import batch job for %s")
324                         .formatWith(bulkImportConfig.entity)
325                         .withReason(value == null ? "Import ID not found" : value.toString())
326                         .withResolution(RESOLUTION_ADDRESS_ISSUE);
327             }
328             return null;
329 
330         } finally {
331             ((BulkImportHelper) openApiHelper).close();
332             inputMap.clear();
333         }
334     }

342     private void checkStatus(String batchId) throws Exception {
343         openApiHelper = new OpenApiHelper(v2CompatibleOpenApiSpec);
344         openApiPath = "/bulk/v1/leads/batch/{batchId}.json";
345         operation = HttpGet.METHOD_NAME.toLowerCase();
346         inputMap.clear();
347         compile("$.parameters.path.batchId").write(inputMap, batchId);
348 
349         outDoc = failsafeOAuthProcess(inputMap, null);
350 
351         int statusCalls = 1;
352         statusReporter.begin(MAX_TRIES_TIMEOUT, TIMES);
353         String status = null;
354         int delaySeconds = 0;
355         Map<String, Object> outMap = null;
356         try {
357             do {
358                 // wait progressively
359                 delaySeconds += INCREMENT_SECONDS;
360                 sleepNoInterrupt(delaySeconds, TimeUnit.SECONDS, LOG);
361                 // refresh access token if necessary
362                 updateAuthHeader();
363                 // execute http request
364                 outDoc = failsafeOAuthProcess(inputMap, null);
365                 if (outDoc == null) {
366                     break;
367                 }
368                 outMap = (Map<String, Object>) readJsonPath(outDoc.get(Map.class),
369                         ENTITY, RESULT_0);
370                 Object value = outMap.get(STATUS);
371                 if (value instanceof String) {
372                     status = (String) value;
373                 } else {
374                     throw new SnapDataException(
375                             "Failed to check the bulk-import job status for %s")
376                             .formatWith(bulkImportConfig.entity)
377                             .withReason(outDoc.get(Map.class).get(ENTITY).toString())
378                             .withResolutionAsDefect();
379                 }
380                 LOG.info("Marketo bulk job status: {}", status);
381                 statusReporter.worked(statusCalls);
382                 if (statusCalls++ >= MAX_TRIES_TIMEOUT && !COMPLETE.equalsIgnoreCase(status)) {
383                     throw new SnapDataException(
384                             "Timed out while checking the bulk-import job status for %s")
385                             .formatWith(bulkImportConfig.entity)
386                             .withReason("Bulk-import job could not complete in 30 minutes")
387                             .withReason(RESOLUTION_ADDRESS_ISSUE);
388                 }
389             } while (!COMPLETE.equalsIgnoreCase(status));
390             if (MapUtils.isNotEmpty(outMap)) {
391                 outMap = postProcessResponseMap(outMap, bulkImportConfig.entity);
392                 writeDocuments(outMap);
393             }
394         } finally {
395             statusReporter.done();
396             if ("Processing".equalsIgnoreCase(status)) {
397                 // refresh access token if necessary
398                 updateAuthHeader();
399                 // Cancel the bulk-export job in progress.
400                 openApiPath = paths.get(4);
401                 operation = HttpPost.METHOD_NAME.toLowerCase();
402                 failsafeOAuthProcess(inputMap, null);
403             }
404         }
405     }
