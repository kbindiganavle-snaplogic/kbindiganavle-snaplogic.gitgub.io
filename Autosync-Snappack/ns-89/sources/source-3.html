18 public class DefaultNameHandler extends NameHandler {

24     public DefaultNameHandler(Function<String, String> converter,
25                               String startQuote, String endQuote) {

28         this.endQuote = endQuote;
29     }

31     @Override
32     public boolean isEnquoteRequired(NameType nameType, String name) {

34         return nameType != NameType.TABLE_NAME;
35     }
36 
37     @Override
38     public String enquote(NameType nameType, String name) {
39         if (name.startsWith(String.valueOf(startQuote))
40                 && name.endsWith(String.valueOf(endQuote))) {
41             return converter.apply(name);
42         }
43         return String.format("%s%s%s", startQuote,
44                 converter.apply(name),
45                 endQuote);
46     }
47 
48     @Override
49     public String getNameAsInDatabase(NameType nameType, String name) {
50         return converter.apply(name);
51     }
