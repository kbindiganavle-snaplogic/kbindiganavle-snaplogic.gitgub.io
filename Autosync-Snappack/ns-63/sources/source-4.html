37  * This class is copied from "https://raw.githubusercontent .com/twitter/commons/master/src/java/
38  * com/twitter/common/objectsize/ObjectSizeCalculator.java".
39  * Contains utility methods for calculating the memory usage of objects. It
40  * only works on the HotSpot JVM, and infers the actual memory layout (32 bit
41  * vs. 64 bit word size, compressed object pointers vs. uncompressed) from
42  * best available indicators. It can reliably detect a 32 bit vs. 64 bit JVM.
43  * It can only make an educated guess at whether compressed OOPs are used,
44  * though; specifically, it knows what the JVM's default choice of OOP
45  * compression would be based on HotSpot version and maximum heap sizes, but if
46  * the choice is explicitly overridden with the <tt>-XX:{+|-}UseCompressedOops</tt> command line
47  * switch, it can not detect
48  * this fact and will report incorrect sizes, as it will presume the default JVM
49  * behavior.

188     public long calculateObjectSize(Object obj) {
189         // Breadth-first traversal instead of naive depth-first with recursive
190         // implementation, so we don't blow the stack traversing long linked lists.
191         try {
192             for (; ; ) {
193                 visit(obj);
194                 if (pending.isEmpty()) {
195                     return size;
196                 }
197                 obj = pending.removeFirst();
198             }
199         } finally {
200             alreadyVisited.clear();
201             pending.clear();
202             size = 0;
203         }
204     }

206     private void visit(Object obj) {
207         if (alreadyVisited.contains(obj)) {
208             return;
209         }
210         final Class<?> clazz = obj.getClass();
211         if (clazz == ArrayElementsVisitor.class) {
212             ((ArrayElementsVisitor) obj).visit(this);
213         } else {
214             alreadyVisited.add(obj);
215             if (clazz.isArray()) {
216                 visitArray(obj);
217             } else {
218                 classSizeInfos.getUnchecked(clazz).visit(obj, this);
219             }
220         }
221     }

223     private void visitArray(Object array) {
224         final Class<?> componentType = array.getClass().getComponentType();
225         final int length = Array.getLength(array);
226         if (componentType.isPrimitive()) {
227             increaseByArraySize(length, getPrimitiveFieldSize(componentType));
228         } else {
229             increaseByArraySize(length, referenceSize);
230             // If we didn't use an ArrayElementsVisitor, we would be enqueueing every
231             // element of the array here instead. For large arrays, it would
232             // tremendously enlarge the queue. In essence, we're compressing it into
233             // a small command object instead. This is different than immediately
234             // visiting the elements, as their visiting is scheduled for the end of
235             // the current queue.
236             switch (length) {
237                 case 0: {
238                     break;
239                 }
240                 case 1: {
241                     enqueue(Array.get(array, 0));
242                     break;
243                 }
244                 default: {
245                     enqueue(new ArrayElementsVisitor((Object[]) array));
246                 }
247             }
248         }
249     }

286     private class ClassSizeInfo {
287         // Padded fields + header size
288         private final long objectSize;
289         // Only the fields size - used to calculate the subclasses' memory
290         // footprint.
291         private final long fieldsSize;
292         private final Field[] referenceFields;
293 
294         public ClassSizeInfo(Class<?> clazz) {
295             long fieldsSize = 0;
296             final List<Field> referenceFields = new LinkedList<Field>();
297             try {
298                 for (Field f : clazz.getDeclaredFields()) {
299                     if (Modifier.isStatic(f.getModifiers())) {
300                         continue;
301                     }
302                     final Class<?> type = f.getType();
303                     if (type.isPrimitive()) {
304                         fieldsSize += getPrimitiveFieldSize(type);
305                     } else {
306                         f.setAccessible(true);
307                         referenceFields.add(f);
308                         fieldsSize += referenceSize;
309                     }
310                 }
311                 final Class<?> superClass = clazz.getSuperclass();
312                 if (superClass != null) {
313                     final ClassSizeInfo superClassInfo = classSizeInfos.getUnchecked(superClass);
314                     fieldsSize += roundTo(superClassInfo.fieldsSize, superclassFieldPadding);
315                     referenceFields.addAll(Arrays.asList(superClassInfo.referenceFields));
316                 }
317             } catch (NoClassDefFoundError ex) {
318                 // NO OP
319             }
320             this.fieldsSize = fieldsSize;
321             this.objectSize = roundTo(objectHeaderSize + fieldsSize, objectPadding);
322             this.referenceFields = referenceFields.toArray(
323                     new Field[referenceFields.size()]);
324         }
325 
326         void visit(Object obj, ObjectSizeCalculator calc) {
327             calc.increaseSize(objectSize);
328             enqueueReferencedObjects(obj, calc);
329         }
330 
331         public void enqueueReferencedObjects(Object obj, ObjectSizeCalculator calc) {
332             for (Field f : referenceFields) {
333                 try {
334                     calc.enqueue(f.get(obj));
335                 } catch (IllegalAccessException e) {
336                     final AssertionError ae = new AssertionError(
337                             "Unexpected denial of access to " + f);
338                     ae.initCause(e);
339                     throw ae;
340                 }
341             }
342         }
343     }

345     private static long getPrimitiveFieldSize(Class<?> type) {
346         if (type == boolean.class || type == byte.class) {
347             return 1;
348         }
349         if (type == char.class || type == short.class) {
350             return 2;
351         }
352         if (type == int.class || type == float.class) {
353             return 4;
354         }
355         if (type == long.class || type == double.class) {
356             return 8;
357         }
358         throw new AssertionError("Encountered unexpected primitive type " +
359                 type.getName());
360     }
