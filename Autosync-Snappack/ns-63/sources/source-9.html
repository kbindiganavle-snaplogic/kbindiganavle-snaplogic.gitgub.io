164     public static <O> O validate(SnapType type, Object value, boolean isExpression)
165             throws ClassCastException {
166         if (isExpression){
167             return (O) String.class.cast(value);
168         }
169         switch (type) {
170             case NUMBER:
171                 if (value instanceof BigInteger) {
172                     return (O) value;
173                 } else if (value instanceof BigDecimal) {
174                     return (O) value;
175                 } else if (value instanceof Integer) {
176                     return (O) BigInteger.valueOf(((Integer) value));
177                 } else if (value instanceof Long) {
178                     return (O) BigInteger.valueOf(((Long) value));
179                 } else if (value instanceof Short) {
180                     return (O) BigInteger.valueOf(((Short) value));
181                 } else if (value instanceof Double) {
182                     return (O) BigDecimal.valueOf(((Double) value));
183                 } else if (value instanceof Float) {
184                     return (O) BigDecimal.valueOf(((Float) value));
185                 }
186                 break;
187             case INTEGER:
188                 if (value instanceof BigInteger) {
189                     return (O) BigInteger.class.cast(value);
190                 } else if (value instanceof Integer) {
191                     return (O) Integer.class.cast(value);
192                 } else if (value instanceof Long) {
193                     return (O) Long.class.cast(value);
194                 } else if (value instanceof Short) {
195                     return (O) Short.class.cast(value);
196                 }
197                 break;
198             case DATETIME:
199                 return (O) DateTime.class.cast(value);
200             case LOCALDATETIME:
201                 return (O) LocalDateTime.class.cast(value);
202             case DATE:
203                 return (O) LocalDate.class.cast(value);
204             case TIME:
205                 return (O) LocalTime.class.cast(value);
206             case COMPOSITE:
207                 return (O) Map.class.cast(value);
208             case TABLE:
209                 return (O) List.class.cast(value);
210             case STRING:
211                 return (O) String.class.cast(value);
212             case BOOLEAN:
213                 return (O) Boolean.class.cast(value);
214             case BINARY:
215                 return (O) byte[].class.cast(value);
216             case ANY:
217                 return (O) value;
218             default:
219                 break;
220         }
221         if (value != null) {
222             throw new ClassCastException(String.format(UNSUPPORTED_CLASS, value.getClass()));
223         }
224         return null;
