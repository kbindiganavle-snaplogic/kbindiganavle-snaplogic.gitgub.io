201     protected void processDocument(Document document, String inputViewName) {
202         // Check if this is the schema document and save it for later.
203         if (inputViewName != null && inputViewName.equals(schemaViewName)) {
204             if (document != null) {
205                 schema = document;
206                 columnTranslation = databaseUtils.buildColumnTranslation(schema);
207             }
208             return;
209         }
210         // If input document contains null or empty map, skip to the next document.
211         Map<String, Object> data = document.get(Map.class);
212         if (MapUtils.isEmpty(data)) {
213             return;
214         }
215 
216         databaseUtils.translateColumnNamesInDoc(columnTranslation, document);
217         if (firstRecord) {
218             firstRecord = false;
219             if (autoCreateTable) {
220                 createTableIfAbsent(document);
221             }
222             Connection conn = jdbcOperations.acquireConnection(account);
223             tableMetaData = jdbcOperations.getTableMetaData(conn,
224                     account.getDatabaseConfig().getDatabaseName(), tableName);
225             addKeyColumns();
226             try {
227                 keyFile = String.format(FILE_KEY, keyFileUUID);
228                 tempFile = File.createTempFile(keyFile, ".tmp", tempDir);
229                 fileOutputStream = new FileOutputStream(tempFile);
230             } catch (IOException ex) {
231                 SnapDataException snapException = new SnapDataException(ex, ERROR_CREATING_TEMP_FILE)
232                         .withReason(ex.getLocalizedMessage())
233                         .withResolution(ERROR_FIX_FILE_CREATION);
234                 errorViews.write(snapException);
235                 return;
236             }
237             for (String column : data.keySet()) {
238                 String normalizedColumn = jdbcOperations.normalize(column);
239                 if (tableMetaData.containsKey(normalizedColumn) ||
240                         tableMetaData.containsKey(column)) {
241                     copyColumns.add(column);
242                     quotedCopyColumns.add(String.format(ID_QUOTE_FORMAT, column));
243                 }
244             }
245             if (copyColumns.isEmpty()) {
246                 SnapDataException snapException = new SnapDataException(ERROR_INVALID_INPUT)
247                         .withReason(ERROR_INVALID_INPUT_REASON)
248                         .withResolution(ERROR_INVALID_INPUT_RESOLUTION);
249                 errorViews.write(snapException);
250                 return;
251             }
252         }
253         inputCount++;
254         try {
255             boolean firstColumn = true;
256             for (String column : copyColumns) {
257                 Object value = data.get(column);
258                 if (!firstColumn) {
259                     fileOutputStream.write(DELIMITER);
260                 }
261                 firstColumn = false;
262                 if (value != null) {
263                     if (value instanceof String) {
264                         fileOutputStream.write(getEncodedString((String) value).getBytes());
265                     } else if (value instanceof DateTime || value instanceof LocalDateTime) {
266                         writeDateField(column, value);
267                     } else {
268                         fileOutputStream.write(value.toString().getBytes());
269                     }
270                 }
271             }
272             fileOutputStream.write(NEWLINE);
273         } catch (IOException ioe) {
274             SnapDataException snapException = new SnapDataException(ioe, ERROR_STREAMING_DATA)
275                     .withReason(ioe.getLocalizedMessage())
276                     .withResolution(ERROR_FIX_FILE_CREATION);
277             errorViews.write(snapException);
278         }
279     }

285     private String getEncodedString(final String value) {
286         StringBuilder newValue = new StringBuilder();
287 
288         for (char c : value.toCharArray()) {
289             if (c == DELIMITER || c == NEWLINE || c == CARRIAGE_RETURN || c == ESCAPE_CHARACTER) {
290                 newValue.append(ESCAPE_CHARACTER);
291             }
292             newValue.append(c);
293         }
294         return newValue.toString();
295     }

297     private void writeDateField(String columnName, Object value) throws IOException {
298         String normalizedColumn = jdbcOperations.normalize(columnName);
299         Integer sqlType = tableMetaData.getColumnDataType(normalizedColumn);
300         String dateString;
301         if (value instanceof LocalDateTime) {
302             if (sqlType == Types.DATE) {
303                 dateString = dateFormatter.print((LocalDateTime) value);
304             } else {
305                 dateString = dateTimeFormatter.print((LocalDateTime) value);
306             }
307         } else {
308             if (sqlType == Types.DATE) {
309                 dateString = dateFormatter.print((DateTime) value);
310             } else {
311                 dateString = dateTimeFormatter.print((DateTime) value);
312             }
313         }
314         fileOutputStream.write(dateString.getBytes());
315     }

331     private void finish() throws ExecutionException {
332         try {
333             if (!firstRecord && autoCreateTable && schema != null) {
334                 // We want to create the table even if there are no rows to be loaded.
335                 createTableIfAbsent(null);
336             }
337             if (!successful || fileOutputStream == null) {
338                 return;
339             }
340             log.debug(LOG_WRITING_TO_TEMP_FILE);
341             fileOutputStream.close();
342             log.debug(LOG_WRITING_TO_TEMP_FILE_DONE);
343             preLoadAction();
344             loadAction(tableName);
345             writeErrorView();
346             writeOutputView();
347         } catch (SQLException se) {
348             SnapDataException snapException = new SnapDataException(se, ERROR_PERFORMING_LOAD)
349                     .withReason(se.getLocalizedMessage())
350                     .withResolution(ERROR_FIX_LOAD_ISSUE);
351             errorViews.write(snapException);
352         } catch (IOException ioe) {
353             SnapDataException snapException = new SnapDataException(ioe, ERROR_PERFORMING_LOAD)
354                     .withReason(ioe.getLocalizedMessage())
355                     .withResolutionAsDefect();
356             errorViews.write(snapException);
357         } finally {
358             // DatabaseAccountImpl.disconnect() is a no-op, close JDBC connection here
359             // Calling a disconnect on account only removes the reference to the connection,
360             // so when the connection is being closed in the after it can't properly
361             // close the connection, so at the end of the pipeline execution its closed as
362             // a leaked connection. So as a rule of thumb in DB snap,
363             // we need close the connection in the same context where it was created.
364             // That's why the order of closing the account and connection matters here.
365             try {
366                 if (verticaConnection != null && !verticaConnection.isClosed()) {
367                     verticaConnection.close();
368                 }
369                 if (account != null) {
370                     account.disconnect();
371                 }
372             } catch (SQLException e) {
373                 log.error(Messages.ERROR_CLOSING_DB_CONNECTION, e);
374             }
375             if (tempFile != null) {
376                 tempFile.delete();
377             }
378             if (tempExceptionFile != null) {
379                 tempExceptionFile.delete();
380             }
381         }
382     }

398     protected void loadAction(String targetTable) throws SQLException {
399         String columns = StringUtils.join(quotedCopyColumns, COMMA);
400         String copyCommand = String.format(COPY_COMMAND, targetTable, columns,
401                 tempFile.getAbsolutePath(), DELIMITER, ESCAPE_CHARACTER, maxErrors,
402                 tempExceptionFile.getAbsolutePath());
403         try (Statement stmt = verticaConnection.createStatement()) {
404             log.debug(LOG_RUNNING_COPY, copyCommand);
405             stmt.execute(copyCommand);
406             updatedCount = stmt.getUpdateCount();
407             log.debug(LOG_COPY_DONE);
408         } catch (SQLException ex) {
409             SnapDataException snapException = new SnapDataException(ex, ERROR_PERFORMING_LOAD)
410                     .withReason(ex.getLocalizedMessage())
411                     .withResolution(ERROR_FIX_LOAD_ISSUE);
412             errorViews.write(snapException);
413         }
414     }
