778     @Override
779     @SuppressWarnings("ToArrayCallWithZeroLengthArrayArgument")
780     protected void process(final Document document, final String inputViewName) {
781         if (isError) {
782             return;
783         }
784         String envelope;
785         String strippedEnvelope = null;
786         Object documentData = document != null ? document.get() : null;
787         List<Header> headerList = soapUtils.buildHeaderList(document, documentData,
788                 httpHeaders);
789         attachments = evaluateAttachments(document);
790         if (account != null && account instanceof HttpRequestAuthProvider) {
791             Header authHeader = ((HttpRequestAuthProvider) account).createAuth();
792             if (authHeader != null) {
793                 headerList.add(authHeader);
794                 headerList.add(new BasicHeader(HttpHeaders.ACCEPT, MediaType.WILDCARD));
795             }
796         }
797         if (soapMessageSenderInterceptor != null) {
798             Header[] headers = headerList.toArray(new Header[0]);
799             soapMessageSenderInterceptor.setHeaders(headers);
800         }
801 
802         envelope = evaluateGeneratedEnvelope(document);
803         if (removeIllegalCharacters) {
804             envelope = envelope.replaceAll(ILLEGAL_CHAR_PATTERN, "");
805         }
806         if (!useDefaultValueChecked) {
807             strippedEnvelope = xmlUtils.stripElementsAndAttributesExceptSOAPHeader(envelope,
808                     documentBuilderFactory, preserveInputEmptyElementsChecked);
809             envelope = strippedEnvelope;
810         }
811         LOGGER.debug(DISPATCHING_SOAP_REQUEST, envelope);
812         try {
813             invokeSoapRequest(document, envelope, headerList);
814             writeToDebugView(envelope, headerList);
815         } catch (SOAPFaultException e) {
816             writeToDebugView(envelope, headerList);
817             SOAPFault soapFault = e.getFault();
818             Map<String, Object> errorMap = Maps.newHashMap();
819             errorMap.put(SOAP_FAULT_EXCEPTION, e.getMessage());
820             errorMap.put(SOAP_FAULT_CODE, soapFault.getFaultCode());
821             errorMap.put(SOAP_FAULT_STRING, soapFault.getFaultString());
822             errorMap.put(SOAP_FAULT_ACTOR, soapFault.getFaultActor());
823             Detail newDetail = soapFault.getDetail();
824             includeSoapFaultDetails(newDetail, errorMap);
825             Throwable rootCause = Throwables.getRootCause(e);
826             SnapDataException ex;
827             if (standardizeErrorOutput) {
828                 ex = new SnapDataException(e, EXCEPTION_OCCURRED)
829                         .withReason(formatReason(errorMap))
830                         .withResolution(VERIFY_THE_INPUT_DATA_IS_CORRECT);
831             } else {
832                 String reason = rootCause.getMessage();
833                 errorMap.put(KEY_ERROR, reason);
834                 errorMap.put(KEY_REASON, reason);
835                 ex = new SnapDataException(documentUtility.newDocument(errorMap), e,
836                         EXCEPTION_OCCURRED)
837                         .withReason(e.getMessage())
838                         .withResolution(VERIFY_THE_INPUT_DATA_IS_CORRECT);
839             }
840             errorViews.write(ex, document);
841             return;
842         } catch (ExecutionException ex) {
843             writeToDebugView(envelope, headerList);
844             throw ex;
845         } catch (Exception e) {
846             Throwable rootCause = Throwables.getRootCause(e);
847             String reason = rootCause.getMessage();
848             if (rootCause instanceof HTTPException &&
849                     SC_LENGTH_REQUIRED == ((HTTPException) rootCause).getResponseCode()) {
850                 httpClientPolicy.setAllowChunking(false);
851                 httpConduit.setClient(httpClientPolicy);
852             }
853             if (standardizeErrorOutput) {
854                 Throwable cause = e.getCause();
855                 reason = cause == null ? reason : cause.getMessage();
856                 SnapDataException ex = new SnapDataException(e, EXCEPTION_OCCURRED)
857                         .withReason(reason)
858                         .withResolution(SOAP_EXCEPTION_RESOLUTION);
859                 errorViews.write(ex, document);
860                 return;
861             }
862             Map<String, Object> errorMap = Maps.newHashMap();
863             errorMap.put(KEY_ERROR, reason);
864             errorMap.put(STACKTRACE, Throwables.getStackTraceAsString(e.getCause() != null ?
865                     e.getCause() : e));
866             SnapDataException dataException = new SnapDataException(documentUtility
867                     .newDocument(errorMap), e, EXCEPTION_OCCURRED)
868                     .withReason(reason)
869                     .withResolution(SOAP_EXCEPTION_RESOLUTION);
870             // serialize headers
871             Map<String, Object> soapHeaders = Maps.newHashMap();
872 
873             Dispatch<SOAPMessage> dispatchClient = clientBuilder.getDispatchClient();
874             Map<?, Object> requestContext = dispatchClient.getRequestContext();
875             if (requestContext != null) {
876                 soapUtils.serializeHeader(requestContext, soapHeaders);
877             }
878 
879             if (soapMessageSenderInterceptor != null) {
880                 Message message = soapMessageSenderInterceptor.getMessage();
881                 if (message != null) {
882                     // protocol headers are the ones passed into the http/s request.
883                     soapUtils.serializeHeader((Map<?, Object>)
884                             message.get(Message.PROTOCOL_HEADERS), soapHeaders);
885                 }
886             }
887 
888             for (Header header : headerList) {
889                 String name = header.getName();
890                 if (name != null) {
891                     soapHeaders.put(name, header.getValue());
892                 }
893             }
894 
895             if (soapHeaders.containsKey(KEY_AUTHORIZATION)) {
896                 soapHeaders.put(KEY_AUTHORIZATION, MASKED_AUTH_HEADER);
897             }
898             errorMap.put(REQUEST_HEADERS, soapHeaders);
899             // serialize the envelope
900             if (envelope != null) {
901                 try {
902                     errorMap.put(KEY_ENVELOPE, parseXML2JSON(envelope, xmlUtils));
903                 } catch (Exception ex) {
904                     errorMap.put(KEY_ENVELOPE, envelope);
905                 }
906             }
907             if (strippedEnvelope != null) {
908                 try {
909                     errorMap.put(KEY_STRIPPED_ENVELOPE, parseXML2JSON(strippedEnvelope, xmlUtils));
910                 } catch (Exception ex) {
911                     errorMap.put(KEY_STRIPPED_ENVELOPE, strippedEnvelope);
912                 }
913             }
914             writeToDebugView(envelope, headerList);
915             errorViews.write(dataException, document);
916         }
917     }

960     private String formatReason(Map<String, Object> errorMap) {
961         StringBuilder builder = new StringBuilder();
962         for (Map.Entry<String, Object> entry : errorMap.entrySet()) {
963             String value = String.valueOf(entry.getValue());
964             if (StringUtils.isNotBlank(value)) {
965                 String key = entry.getKey();
966                 builder.append(key);
967                 if (!StringUtils.endsWith(key, COLON_SPACE)) {
968                     builder.append(COLON_SPACE);
969                 }
970                 builder.append(value).append(COMMA_SPACE);
971             }
972         }
973         String reason = builder.toString();
974         return StringUtils.isBlank(reason) ? reason :
975                 reason.substring(0, reason.length() - COMMA_SPACE.length());
976     }

979     /**
980      * Retrieves the contents of the elements recursively and
981      * adds them to the errorMap.
982      *
983      * @param iterator Iterator returned by getChildElements
984      * @param errorMap Place to store the results
985      */
986     public void getFullDetails(Iterator iterator, Map<String, Object> errorMap) {
987         while (iterator.hasNext()) {
988             Node node = (Node) iterator.next();
989 
990             if (node instanceof SOAPElement) {
991                 SOAPElement element = (SOAPElement) node;
992                 Iterator childElements = element.getChildElements();
993                 getFullDetails(childElements, errorMap);
994             } else if (node instanceof Text) {
995                 Text text = (Text) node;
996                 String content = text.getValue();
997                 errorMap.put(text.getParentElement().getElementName().getLocalName(), content);
998             }
999         }
1000     }

1002     /**
1003      * If a soap fault occurs capture the details and put them into the
1004      * error map
1005      *
1006      * @param soapFaultDetail Beginning of the Soap fault
1007      * @param errorMap        Collection of all of the errors found
1008      */
1009     protected void includeSoapFaultDetails(Detail soapFaultDetail, Map<String, Object> errorMap) {
1010         if (soapFaultDetail != null) {
1011             // Keep compatibility with older versions of the snap
1012             errorMap.put(SOAP_FAULT_DETAIL_ENTRY, soapFaultDetail.getTextContent());
