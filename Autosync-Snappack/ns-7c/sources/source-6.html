44 /**
45  * Hamcrest matcher for database records.
46  *
47  * The record comparator is only partially implemented. It compares most datatypes using
48  * string equality. However some datatypes map multiple input values to the same value,
49  * e.g., a boolean may accept 't', 'yes', and '1' but will always produce a 'true' output.
50  * Another good example is the variants of timestamps (with/without a 'T', with/without
51  * separators) but the database will always return the value in a single format.
52  *
53  * This comparator handles booleans and timestamps but does not handle dates or times.
54  *
55  * This comparator does not handle floating point numbers well.
56  *
57  * There are probably at least two ways to define the expected values. The first is
58  * pre-computed object, the second is a file in the classpath.
59  */

137     @Override
138     public boolean matchesSafely(DatabaseServer server) {
139         boolean matches = true;
140 
141         try (Connection conn = server.getConnection();
142                 Statement stmt = conn.createStatement();
143                 ResultSet rs = stmt.executeQuery(query)) {
144             if (!rs.next()) {
145                 LOG.info("no records found");
146                 return false;
147             }
148 
149             // identify comparator.
150             // we include 'NullRecordComparator' in case it wants to gather statistics, etc.
151             final ResultSetMetaData md = rs.getMetaData();
152             RecordComparator recComparator = null;
153             switch (comparator) {
154                 case NULL:
155                     recComparator = new NullRecordComparator(md);
156                     break;
157                 case LENIENT:
158                     recComparator = new LenientRecordComparator(md, differences);
159                     break;
160             }
161 
162             // compare each record vs. map entry.
163             for (Map<String, Object> expected : expected) {
164                 try {
165                     matches = recComparator.matches(rs, expected);
166                 } catch (SQLException e) {
167                     LOG.info("Exception while comparing record: {}", e.getMessage(), e);
168                     throw new AssertionFailedError("Exception while comparing record: " +
169                             e.getMessage());
170                 }
171                 if (rs.next()) {
172                     break;
173                 }
174             }
175            recordNotFound = false;
176 
177         } catch (SQLException e) {
178             throw new AssertionFailedError("Unable to check table: " + e.getMessage());
179         }
180 
181         return matches;
182     }

331         @Override
332         public boolean matches(ResultSet actual, Map<String, Object> expected) throws SQLException {
333             boolean matches = true;
334 
335             for (String key : expected.keySet()) {
336                 keys.put(key.toUpperCase(), key);
337             }
338 
339             // two rules:
340             // - it's okay if a column doesn't have a corresponding entry in the map
341             // - if it does they need to match as strings.
342             for (int colno = 1; colno <= md.getColumnCount(); colno++) {
343                 final String key = md.getColumnName(colno).toUpperCase();
344                 if (expected.containsKey(keys.get(key))) {
345                     final Object expectedValue = expected.get(keys.get(key));
346                     switch (md.getColumnType(colno)) {
347                         // a boolean value is often reported as a bit.
348                         case Types.BIT:
349                         case Types.BOOLEAN:
350                             if (!booleanMatches(actual, colno, expectedValue)) {
351                                 differences.add(new ResultSetDifference(md.getColumnName(colno),
352                                         null, expectedValue));
353                                 matches = false;
354                             }
355                             break;
356 
357                         case Types.TIMESTAMP:
358                             if (!timestampMatches(actual, colno, expectedValue)) {
359                                 differences.add(new ResultSetDifference(md.getColumnName(colno),
360                                         null, expectedValue));
361                                 matches = false;
362                             }
363                             break;
364 
365                         case Types.TIME_WITH_TIMEZONE:
366                         case Types.TIMESTAMP_WITH_TIMEZONE:
367                         default:
368                             final String actualValue = actual.getString(colno);
369                             if (actual.wasNull()) {
370                                 if (expected != null) {
371                                     differences.add(new ResultSetDifference(md.getColumnName(colno),
372                                             null, expectedValue));
373                                     matches = false;
374                                 }
375                             } else {
376                                 // check for string equality
377                                 if (!actualValue.equals(String.valueOf(expectedValue))) {
378                                     differences.add(new ResultSetDifference(md.getColumnName(colno),
379                                             actualValue, expectedValue));
380                                     matches = false;
381                                 }
382                             }
383                     }
384                 }
385             }
386 
387             return matches;
388         }

390         /**
391          * Special handling for BOOLEAN fields since postgresql accepts several
392          * different string values so we can't do a simple string comparison of
393          * 'actual' and 'expected'.
394          *
395          * @param actual
396          * @param colno
397          * @param expected
398          * @return
399          * @throws SQLException
400          */
401         boolean booleanMatches(ResultSet actual, int colno, Object expected) throws SQLException {
402             final boolean expectedNull = (expected == null) ||
403                     (expected instanceof String) && StringUtils.isBlank((String) expected);
404             final boolean actualBool = actual.getBoolean(colno);
405 
406             // was either 'actual' or 'expected' null?
407             if (actual.wasNull()) {
408                 return !expectedNull;
409             } else if (expectedNull) {
410                 return false;
411             }
412 
413             // trivial case
414             if (expected instanceof Boolean) {
415                 return ((Boolean) expected).equals(actualBool);
416             }
417 
418             // PostgreSQL accepts a lot of possible string values for booleans.
419             final String s = String.valueOf(expected).toUpperCase();
420             if (!BOOL_MAP.containsKey(s)) {
421                 return false;
422             }
423 
424             return BOOL_MAP.get(s).equals(actualBool);
425         }

427         /**
428          * Special handling for TIMESTAMP fields since postgresql accepts several
429          * different string values so we can't do a simple string comparison of
430          * 'actual' and 'expected'.
431          *
432          * @param actual
433          * @param colno
434          * @param expected
435          * @return
436          * @throws SQLException
437          */
438         boolean timestampMatches(ResultSet actual, int colno, Object expected) throws SQLException {
439             final boolean expectedNull = (expected == null) ||
440                     (expected instanceof String) && StringUtils.isBlank((String) expected);
441             final Timestamp actualTimestamp = actual.getTimestamp(colno);
442 
443             // was either 'actual' or 'expected' null?
444             if (actual.wasNull()) {
445                 return !expectedNull;
446             } else if (expectedNull) {
447                 return false;
448             }
449 
450             // trivial cases
451             if (expected instanceof Timestamp) {
452                 return ((Timestamp) expected).equals(actualTimestamp);
453             } else if (expected instanceof Date) {
454                 final Timestamp expectedTimestamp = new Timestamp(((Date) expected).getTime());
455                 return (expectedTimestamp.equals(actualTimestamp));
456             } // FIXME - additional temporal formats
457 
458             final String expectedValue = String.valueOf(expected);
459             Date expectedTimestamp = null;
460             for (Map.Entry<Pattern, DateFormat> entry : TIMESTAMP_FORMAT_MAP.entrySet()) {
461                 if (entry.getKey().matcher(expectedValue).matches()) {
462                     try {
463                         final Date date = entry.getValue().parse(expectedValue);
464                         expectedTimestamp = new Timestamp(date.getTime());
465                         return actualTimestamp.equals(expectedTimestamp);
466                     } catch (ParseException e) {
467                         LOG.info("bad format? '{}' for '{}'", entry.getKey().pattern(),
468                                 expectedValue);
469                     }
470                     break;
471                 }
472             }
473 
474             // unrecognized timestamp format.
475             return false;
476         }
