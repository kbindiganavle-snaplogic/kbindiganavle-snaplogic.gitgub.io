137         public DatabaseTestContainerModuleServer build() {

142             DatabaseStrategy strategy = null;

151                     }
152                     break;
153 
154                 case DB2:

159                     }
160                     break;
161 
162                 case MARIADB:

164                         strategy = new MariadbDatabaseTestContainerStrategy();
165                     } else {

167                     }

201 
202                 default:
203                     throw new IllegalArgumentException("The " + db.getName()
204                             + " database is not supported.");
