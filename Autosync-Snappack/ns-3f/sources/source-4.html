117     public HttpSnapAWSV4Signer withRequestDetails(String httpMethodName, URI requestUri,
118                                                   List<Pair<String, String>> headers,
119                                                   HttpSnapEntity payload)
120             throws HttpCoreDataException {
121         this.httpMethodName = httpMethodName;
122         String host = requestUri.getHost();
123         this.headersProvided = new ArrayList<>();
124         this.headersProvided.add(Pair.of("host", host));
125         this.canonicalURI = requestUri.getPath();
126         this.queryParamsStr = requestUri.getQuery() == null ?
127                 StringUtils.EMPTY : requestUri.getQuery();
128         if (StringUtils.isEmpty(serviceName) || StringUtils.isEmpty(regionName)) {
129             int domainIndex = host.indexOf(".amazonaws.com");
130             if (domainIndex == -1) {
131                 throw new HttpCoreDataException("Error parsing " +
132                         "the request URI for AWS")
133                         .withReason("Domain name 'amazonaws.com' is missing from the URI")
134                         .withResolution("Provide the AWS Region and Service Name in the " +
135                                 "AWS SigV4 account if using a CNAME URL.");
136             }
137             String hostPrefix = host.substring(0, domainIndex);
138             this.regionName = hostPrefix.substring(hostPrefix.lastIndexOf('.') + 1);
139             hostPrefix = hostPrefix.substring(0, hostPrefix.lastIndexOf('.'));
140             this.serviceName = hostPrefix.substring(hostPrefix.lastIndexOf('.') == -1 ? 0 :
141                     hostPrefix.lastIndexOf('.') + 1);
142         }
143         this.headersProvided.addAll(headers);
144         if (payload != null) {
145             try {
146                 if (payload instanceof HttpBinaryEntity) {
147                     // Need to stage the entity, in order to generate the body hash.
148                     ((HttpBinaryEntity) payload).setNeedRetry(true);
149                 }
150                 HttpEntity entity = payload.getEntity();
151                 if (entity == null) {
152                     this.payloadStr = StringUtils.EMPTY;
153                 } else if (entity.isRepeatable() && !entity.isStreaming()) {
154                     this.payloadStr = EntityUtils.toString(entity, StandardCharsets.UTF_8);
155                 } else {
156                     // For all other entities (streaming or un-repeatable), we choose to use the
157                     // "Unsigned payload option", which is described in the documentation:
158                     // https://docs.aws.amazon.com/AmazonS3/latest/API/sig-v4-header-based-auth.html
159                     this.payloadHash = UNSIGNED_PAYLOAD;
160                 }
161             } catch (IOException e) {
162                 throw new HttpCoreDataException("Error converting Http entity to String", e);
163             }
164         }
165         return this;
166     }

181      * Do the signature process and return the authorization header.
182      *
183      * @return The authorization headers in String pairs
184      * @throws HttpCoreDataException for any issue parsing the given information
185      */
186     public List<Pair<String, String>> sign() throws HttpCoreDataException {
187         // Get current UTC timestamp
188         xAmzDate = getCurTimeStamp();
189         currentDate = getCurDate();
190         List<Pair<String, String>> headers = new ArrayList<>();
191         headers.add(Pair.of(X_AMZ_DATE, xAmzDate));
192         if (StringUtils.isNotEmpty(securityToken)) {
193             // Refer:
194             // https://docs.aws.amazon.com/general/latest/gr/sigv4-add-signature-to-request.html
195             headers.add(Pair.of(X_AMZ_SECURITY_TOKEN, securityToken));
196         }
197         headers.addAll(headersProvided);
198         try {
199             String canonicalRequest = prepareCanonicalRequest(httpMethodName, serviceName,
200                     canonicalURI, queryParamsStr, headers, payloadStr);
201             String stringToSign = prepareStringToSign(canonicalRequest, xAmzDate,
202                     regionName, serviceName);
203             String signature = calculateSignature(stringToSign, currentDate,
204                     regionName, serviceName);
205             List<Pair<String, String>> signedHeaders = Lists.newArrayList(
206                     Pair.of(X_AMZ_DATE, xAmzDate),
207                     Pair.of(HttpHeaders.AUTHORIZATION, buildAuthorizationString(signature,
208                             currentDate, regionName, serviceName, strSignedHeader)),
209                     Pair.of(X_AMZ_CONTENT_SHA256, payloadHash)
210             );
211             if (StringUtils.isNotEmpty(securityToken)) {
212                 // Refer:
213                 // https://docs.aws.amazon.com/general/latest/gr/sigv4-add-signature-to-request.html
214                 signedHeaders.add(Pair.of(X_AMZ_SECURITY_TOKEN, securityToken));
215             }
216             return signedHeaders;
217         } catch (NoSuchAlgorithmException | UnsupportedEncodingException | InvalidKeyException e) {
218             throw new HttpCoreDataException("Error doing signature for AWS Signature V4", e);
219         }
220     }

226      * Task 1: Create a Canonical Request for Signature Version 4.
227      */
228     @VisibleForTesting
229     String prepareCanonicalRequest(String httpMethodName, String serviceName, String canonicalURI,
230                                    String queryParamsStr,
231                                    List<Pair<String, String>> headers, String payloadStr)
232             throws NoSuchAlgorithmException {
233         StringBuilder canonicalURL = new StringBuilder();
234         /* Step 1.1 Start with the HTTP request method (GET, PUT, POST, etc.), followed by a
235         newline character. */
236         canonicalURL.append(httpMethodName).append("\n");
237         /* Step 1.2 Add the canonical URI parameter, followed by a newline character. */
238         canonicalURI = canonicalURI == null || canonicalURI.trim().isEmpty() ? "/" : canonicalURI;
239         canonicalURI = encodeURI(canonicalURI, true);
240         if (!"s3".equalsIgnoreCase(serviceName)) {
241             // According to
242             // https://docs.aws.amazon.com/general/latest/gr/sigv4-create-canonical-request.html
243             // all services except for S3 requires double encoding.
244             canonicalURI = encodeURI(canonicalURI, true);
245         }
246         canonicalURL.append(canonicalURI).append("\n");
247         /* Step 1.3 Add the canonical query string, followed by a newline character. */
248         canonicalURL.append(canonicalizeQuery(queryParamsStr));
249         canonicalURL.append("\n");
250         /* Step 1.4 Add the canonical headers, followed by a newline character. */
251         StringBuilder signedHeaders = new StringBuilder();
252         if (headers != null && !headers.isEmpty()) {
253             headers.sort((p1, p2) ->
254                     StringUtils.compareIgnoreCase(p1.getKey(), p2.getKey()));
255             for (Pair<String, String> entrySet : headers) {
256                 String key = StringUtils.normalizeSpace(entrySet.getKey().toLowerCase());
257                 String value = StringUtils.normalizeSpace(entrySet.getValue());
258                 signedHeaders.append(key).append(";");
259                 /* Note: Each individual header is followed by a newline character, meaning the
260                     complete list ends with a newline character. */
261                 canonicalURL.append(key).append(":").append(value).append("\n");
262             }
263 
264         }
265         canonicalURL.append("\n");
266         /* Step 1.5 Add the signed headers, followed by a newline character. */
267         strSignedHeader = signedHeaders.substring(0, signedHeaders.length() - 1); // Remove last ";"
268         canonicalURL.append(strSignedHeader).append("\n");
269         /* Step 1.6 Use a hash (digest) function like SHA256 to create a hashed value from the
270         payload in the body of the HTTP or HTTPS. */
271         if (!StringUtils.equals(payloadHash, UNSIGNED_PAYLOAD)) {
272             payloadStr = payloadStr == null ? StringUtils.EMPTY : payloadStr;
273             payloadHash = hash(payloadStr);
274         }
275         canonicalURL.append(payloadHash);
276         LOG.debug("[AWS SigV4][Task 1][CanonicalRequest]{}", canonicalURL.toString());
277         return canonicalURL.toString();
278     }

281     /**
282      * Encode the URI for Signature V4.
283      *
284      * The AWS Signature V4 requires rfc3986 URI encoding. and it has several rules that make
285      * the URLEncoder does not work directly, refer to the definition of UriEncode() here:
286      * https://docs.aws.amazon.com/AmazonS3/latest/API/sig-v4-header-based-auth.html
287      *
288      * Including:
289      * - Spaces (' ') need to be encoded into "%20"
290      * - Asterisks ('*') need to be encoded into '%2A'
291      * - '~' needs to be skipped
292      * - Slashes ('/') need to be skipped if we are encoding a Path string
293      *
294      * Also see the implementation in AWS SDK:
295      * https://github.com/aws/aws-sdk-java/blob/master/aws-java-sdk-core/src/main
296      * /java/com/amazonaws/util/SdkHttpUtils.java#L66
297      *
298      * @param uri The String to be encoded
299      *
300      * @return  The encoded URI
301      */
302     private String encodeURI(String uri, boolean isPath) {
303         String encoded =  URLEncoder.encode(uri, StandardCharsets.UTF_8)
304                 .replaceAll("\\+", "%20")
305                 .replaceAll("\\*", "%2A")
306                 .replaceAll("%7E", "~");
307         if (isPath) {
308                 encoded = encoded.replaceAll("%2F", "/");
309         }
310         return encoded;
311     }

313     /**
314      * Parse the given query string and canonicalize it.
315      *
316      * According to AWS documentation, the query strings need to be normalized in the following way:
317      * 1. Queries need to be sorted by key in ascending order. Duplicate keys are sorted by value
318      * 2. Both key and values need to be URI encoded
319      *
320      * @param rawQuery The original query string
321      * @return the normalized query string
322      */
323     @VisibleForTesting
324     protected String canonicalizeQuery(String rawQuery) {
325         String[] queries = rawQuery.split("&") ;
326         String finalQuery = "";
327         List<Pair<String, String>> queryPairs = new ArrayList<>();
328         String extraKey = null;
329         // Parse the query string into pairs.
330         // Accumulate the queries without '=' into one query key. this seems to be how AWS
331         // treat parameters of this kind according to the note:
332         // https://github.com/lucasweb78/aws-v4-signer-java/blob/master/src/main/java/uk/co/
333         // lucasweb/aws/v4/signer/CanonicalRequest.java#L142
334         // where "foo&bar=tar" will be treated as (foo&bar, tar)
335         for (String query : queries) {
336             String[] nameValue = query.split("=", 2);
337             if (nameValue.length == 1) {
338                 extraKey = extraKey == null ? nameValue[0] : extraKey + "&" + nameValue[0];
339             } else if (nameValue.length == 2) {
340                 String key = nameValue[0];
341                 if (extraKey != null) {
342                     key = extraKey + "&" + key;
343                     extraKey = null;
344                 }
345                 queryPairs.add(Pair.of(key, nameValue[1]));
346             }
347         }
348         if (extraKey != null) {
349             queryPairs.add(Pair.of(extraKey, ""));
350         }
351 
352         // Sort the queries by keys in ascending order. If the keys are equal, then sort by value
353         queryPairs.sort((p1, p2) -> {
354             if (p1.getLeft().equals(p2.getLeft())) {
355                 return p1.getRight().compareTo(p2.getRight());
356             }
357             return p1.getLeft().compareTo(p2.getLeft());
358         });
359         List<String> encodedQueries = new ArrayList<>();
360 
361         // URIEncode the query key and query value
362         for (Pair<String, String> query : queryPairs) {
363             if (StringUtils.isNotEmpty(query.getKey())) {
364                 encodedQueries.add(encodeURI(query.getKey(), false) + "="
365                         + encodeURI(query.getRight(), false));
366             }
367         }
368         return String.join("&", encodedQueries);
369     }

381         /* Step 2.2 Append the request date value, followed by a newline character. */
382         stringToSign += xAmzDate + "\n";
383         /* Step 2.3 Append the credential scope value, followed by a newline character. */
384         stringToSign += currentDate + "/" + regionName + "/" + serviceName + "/" + AWS_4_REQUEST +
385                 "\n";
386         /* Step 2.4 Append the hash of the canonical request that you created in Task 1: Create a
387          Canonical Request for Signature Version 4. */
388         stringToSign += hash(canonicalRequest);
389         LOG.debug("[AWS SigV4][Task 2][StringToSign]{}", stringToSign);
390         return stringToSign;

394      * Task 3: Calculate the AWS Signature Version 4.
395      */
396     @VisibleForTesting
397     String calculateSignature(String stringToSign, String currentDate,
398                               String regionName, String serviceName)
399             throws UnsupportedEncodingException, InvalidKeyException, NoSuchAlgorithmException {
400         /* Step 3.1 Derive your signing key */
401         byte[] signatureKey = getSignatureKey(secretKey, currentDate, regionName,
402                 serviceName);
403         /* Step 3.2 Calculate the signature. */
404         byte[] signature = HmacSHA256(signatureKey, stringToSign);
405         /* Step 3.2.1 Encode signature (byte[]) to Hex */
406         return Hex.encodeHexString(signature).toLowerCase();

428     private String hash(byte[] bytes) throws NoSuchAlgorithmException {
429         MessageDigest messageDigest;
430         messageDigest = MessageDigest.getInstance("SHA-256");
431         messageDigest.update(bytes);
432         byte[] digest = messageDigest.digest();
433         return String.format("%064x", new java.math.BigInteger(1, digest));
434     }

461      * Generate AWS signature key.
462      *
463      * @reference http://docs.aws.amazon.com/general/latest/gr/signature-v4-examples
464      * .html#signature-v4-examples-java
465      */
466     private byte[] getSignatureKey(String key, String date, String regionName,
467                                    String serviceName) throws UnsupportedEncodingException,
468             InvalidKeyException, NoSuchAlgorithmException {
469         byte[] kSecret = ("AWS4" + key).getBytes("UTF8");
470         byte[] kDate = HmacSHA256(kSecret, date);
471         byte[] kRegion = HmacSHA256(kDate, regionName);
472         byte[] kService = HmacSHA256(kRegion, serviceName);
473         return HmacSHA256(kService, AWS_4_REQUEST);
474     }

476     /**
477      * Apply HmacSHA256 on data using given key.
478      *
479      * @reference: http://docs.aws.amazon.com/general/latest/gr/signature-v4-examples
480      * .html#signature-v4-examples-java
481      */
482     private byte[] HmacSHA256(byte[] key, String data)
483             throws NoSuchAlgorithmException, InvalidKeyException {
484         String algorithm = "HmacSHA256";
485         Mac mac = Mac.getInstance(algorithm);
486         mac.init(new SecretKeySpec(key, algorithm));
487         return mac.doFinal(data.getBytes(StandardCharsets.UTF_8));
488     }
