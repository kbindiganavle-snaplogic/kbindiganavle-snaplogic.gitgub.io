96          */

165                                               String compositeSubqueryAlias) {
166         StringBuilder predicateBuilder = new StringBuilder("(");
167         boolean firstItem = true;
168         for (int j = 0; j < this.partitionByFields.size(); j++) {
169             if (firstItem) {
170                 firstItem = false;
171             } else {
172                 predicateBuilder.append(AND_IN_SPACES);
173             }
174             String currentExpr = this.partitionByFields.get(j);
175             if (!currentExpr.equals(this.partitionByColumnList.get(j))) {
176                 StringBuilder updateStringBuilder = new StringBuilder(currentExpr);
177                 String columnName = this.partitionByColumnList.get(j);
178                 replaceAllColumnNames(updateStringBuilder, columnName,

233     class TypeThreeRewriterWithCTEPerWindowFunctionProcessor implements PerWindowFunctionProcessor {
234         /**
235          * Perform per-function processing:
236          * create a CTE clause, a base subquery and a composite subquery for the given "function"
237          * and
238          * create and return the field of the function for the main query SELECT clause.
239          *
240          * @param function
241          * @param inputString
242          * @param randomString
243          * @return main SELECT expression for the given "function"
244          */
245         @Override
246         public String processPerWindowFunction(
247                 RewriterFunction function, RewriterFunction first, String inputString,
248                 String randomString, boolean isFirstFunction, List<String> partitionByFields) {
249             if (function.getOriginalArrayIdx() != function.getCompressedArrayIdx()) {
250                 // Skip adding subquery if an equivalent function to the current function already
251                 // exists.
252                 return String.format(DUPLICATE_ITEM_TYPE_ONE_WITH_SUBQUERY_FMT,
253                         compressedMap.get(function.getKeyToCompressedMap()).getResultExprStr(),
254                         function.getResultExprStr());
255             }
256             if (isFirstFunction) {
257                 function.setUpContextForRewriteWithCte(true, null, partitionByFields);
258             } else {
259                 function.setUpContextForRewriteWithCte(false, getPreviousFunction(function),
260                         partitionByFields);
261             }
262 
263             String rewrittenSqlStr = rewriterDatabase.rewriteFormulae(
264                     function.getFunctionName(),
265                     function.getField1Str(),
266                     function.getField2Str(),
267                     inputString,
268                     randomString,
269                     partitionByFieldsList);
270             function.setCteStr(setupCteStr(function, rewrittenSqlStr, inputString, randomString));
271             if (isFirstFunction) {
272                 function.setBaseSubqueryStr(setUpBaseSubqueryFromCteStr(function));
273             } else {
274                 function.setCompositeSubqueryStr(setUpCompositeSubqueryStr(function,
275                         rewrittenSqlStr, inputString, randomString));
276             }
277 
278             return String.format(TABLE_NAME_DOT_COLUMN_FMT, function.getCompositeSubqueryAlias(),
279                     function.getResultExprStr());
280         }
281     }

334                 compositeSubquerySelectFields,
335                 getFirstSubqueryStr(prevFunction),
336                 currentFunction.getBaseSubqueryStr(),

417     protected static void replaceAll(StringBuilder sb, String regex, String replacement,
418                                      boolean quoteRegex) {
419         String regexpr = quoteRegex ? Pattern.quote(regex) : regex;
420         Pattern pattern = Pattern.compile(regexpr);
421         Matcher matcher = pattern.matcher(sb);
422         int start = 0;
423         while (matcher.find(start)) {
424             sb.replace(matcher.start(), matcher.end(), replacement);
425             start = matcher.start() + replacement.length();
426         }
427     }
