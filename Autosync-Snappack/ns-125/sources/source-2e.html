254     public QName getSearchType(WsdlVersion wsdlVersion, boolean advanced) {
255         if (searchLocalPart == null) {
256             return null;
257         } else {

264                 ns = Namespace.ACCOUNTING_NS;

269             return ns.qname(wsdlVersion, localPart);

275      * for the searchValue field of ItemSearchBasic.type or TransactionSearchBasic.type,

284      */
285     public String getTypeSearchValue() {

321                 // TransactionTypes that don't follow the default pattern (see below)
322                 case PURCHASEREQUISITION:
323                     return "_requisition";
324                 case JOURNALENTRY:
325                 case INTERCOMPANYJOURNALENTRY:
326                 case STATISTICALJOURNALENTRY:

328                 case TRANSFERORDER:

331                 case ITEM:

333                     // for a generic item or transaction search, no type should be specified
334                     return null;
335                 default:
336                     // use the default pattern; e.g. SalesOrder -> _salesOrder
337                     return "_" + name.substring(0, 1).toLowerCase() + name.substring(1);
338             }
339         }
340         else {
341             throw new UnsupportedOperationException("not applicable for " + this);
342         }

375     public boolean supports(OperationType operationType) {
376         switch (operationType) {

380             case SEARCH:

384             default:
