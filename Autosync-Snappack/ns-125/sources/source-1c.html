406     /**
407      * Posts the given SOAP request to NetSuite
408      *
409      * @return a SOAP response
410      */
411     protected SOAPMessage postRequest(String request, final Document inputDocument) {
412         account.withNamespacePrefixContext(namespacePrefixContext)
413                 .withClientBuilder(clientBuilder)
414                 .withTimeout(timeoutInSecs)
415                 .connect();
416         logger.debug("Netsuite Operation: "  + operationQName);
417         String strippedRequest;
418         // The function, stripNodeAndAttributeNodes in xmlUtils is not thread safe
419         synchronized (xmlUtils) {
420             strippedRequest = xmlUtils.stripNodeAndAttributeNodes(request, documentBuilderFactory);
421         }
422         final String strippedRequestFinal = sanitizeStrippedRequest(strippedRequest, request);
423         return doWithRetry(inputDocument, new SnapCallable<SOAPMessage>() {
424             @Override
425             public SOAPMessage call() throws SnapDataException {
426                 try {
427                     SOAPMessage response = invocationService.call(clientBuilder, strippedRequestFinal);
428                     responseJson = invocationService.serialize(response);
429                     if (routRecordsToAppropriateViews) {
430                         parseResponseJson();
431                         return response;
432                     }
433                     boolean writeError = false;
434                     if (hasAsyncStatusResult()) {
435                         // TODO
436                     } else {
437                         /* A response may have more than one element with an isSuccess attribute.
438                           If there
439                          * are no such elements, or there is at least one element whose isSuccess
440                           * attribute is
441                          * not "true", report an error.  For example, an error response for
442                          * 'upsertList' may
443                          * have a 'writeResponseList' element whose 'status' child has
444                          * isSuccess="true" but
445                          * whose 'writeResponse' child has a 'status' child with isSuccess="false".
446                          */
447                         // TODO: why are doing all this XML stuff? we could just look at
448                         // responseJson instead
449                         ByteArrayOutputStream responseContainerStream = new ByteArrayOutputStream();
450                         response.writeTo(responseContainerStream);
451                         DocumentBuilder documentBuilder = documentBuilderFactory
452                                 .newDocumentBuilder();
453                         org.w3c.dom.Document w3cDoc = documentBuilder.parse(new
454                                 ByteArrayInputStream(
455                                 responseContainerStream.toByteArray()));
456                         XPathExpression statusExpression = XPATH_FACTORY.newXPath().compile
457                                 (XPATH_IS_SUCCESS);
458                         NodeList nodeList = (NodeList) statusExpression.evaluate(w3cDoc,
459                                 XPathConstants.NODESET);
460                         if (nodeList == null || nodeList.getLength() == 0) {
461                             writeError = true;
462                         } else {
463                             for (int i = 0; i < nodeList.getLength(); i++) {
464                                 Node node = nodeList.item(i);
465                                 if (!TRUE.equals(node.getNodeValue())) {
466                                     writeError = true;
467                                     break;
468                                 }
469                             }
470                         }
471                     }
472                     Document responseDocument = documentUtility.newDocumentFor(
473                             inputDocumentOriginal != null ? inputDocumentOriginal : inputDocument,
474                             responseJson);
475                     Document processedResponse = postProcessResponse(responseDocument);
476                     if (!checkStatusGetResults) {
477                         if (writeError) {
478                             throw new SnapDataException(
479                                     processedResponse,
480                                     ERROR_WHILE_EXECUTING_SOAP_REQUEST)
481                                     .withReason(responseJson.toString())
482                                     .withResolution(PLEASE_VERIFY_REQUEST_DATA);
483                         } else if (processedResponse != null) {
484                             outputViews.write(processedResponse, inputDocument);
485                         }
486                     }
487                     return response;
488                 } catch (SnapDataException e) {
489                     throw e;
490                 } catch (SOAPException | SOAPFaultException e) {
491                     throw new SnapDataException(e,
492                             ERROR_WHILE_EXECUTING_SOAP_REQUEST)
493                             .withReason(e.getMessage())
494                             .withResolution(VERIFY_REQUEST_DATA);
495                 } catch (XPathExpressionException e) {
496                     throw new SnapDataException(e,
497                             ERROR_WHILE_EVALUATING_XPATH_EXPRESSION)
498                             .withReason(e.getMessage())
499                             .withResolutionAsDefect();
500                 } catch (IOException e) {
501                     throw new SnapDataException(e,
502                             ERROR_WHILE_PARSING_RESPONSE_DOCUMENT)
503                             .withReason(e.getMessage())
504                             .withResolution(TRY_INCREASING_MAX);
505                 } catch (ParserConfigurationException e) {
506                     throw new SnapDataException(e,
507                             ERROR_WHILE_CREATING_DOCUMENT_BUILDER)
508                             .withReason(e.getMessage())
509                             .withResolutionAsDefect();
510                 } catch (SAXException e) {
511                     throw new SnapDataException(e,
512                             ERROR_WHILE_PARSING_RESPONSE_DOCUMENT)
513                             .withReason(e.getMessage())
514                             .withResolutionAsDefect();
515                 } catch (XMLStreamException e) {
516                     throw new SnapDataException(e,
517                             ERROR_WHILE_SERIALIZING_RESPONSE_DATA)
518                             .withReason(e.getMessage())
519                             .withResolutionAsDefect();
520                 } catch (WebServiceException e) {
521                     throw new SnapDataException(e,
522                             ERROR_WHILE_SENDING_THE_REQUEST)
523                             .withReason(e.getMessage())
524                             .withResolution(VERIFY_REQUEST_DATA);
525                 } catch (ExecutionException e) {
526                     throw new SnapDataException(e,
527                             ERROR_WHILE_EXECUTING_SOAP_REQUEST)
528                             .withReason(e.getReason())
529                             .withResolution(e.getResolution());
530                 } catch (SuggestViewAbortException e) {
531                     throw e;
532                 } catch (IndexOutOfBoundsException e) {
533                     RuntimeException runtimeException =
534                             new RuntimeException(ARRAY_INDEX_OUT_OF_RANGE, e);
535                     throw new SnapDataException(runtimeException,
536                             ERROR_WHILE_EXECUTING_SOAP_REQUEST)
537                             .withReason(runtimeException.getMessage())
538                             .withResolution(PLEASE_VERIFY_REQUEST_DATA);

567     /**
568      * Invokes the given callable to process a document, retrying on failure according to the retry
569      * properties set during the configuration of this snap.
570      *
571      * @param inputDocument The document to be referenced when writing an error to the error view.
572      * @param callable      An object with a single call method to be executed in the retry loop.
573      * @param <V>           The type of object returned by the callable.
574      * @return The object returned by the callable.
575      */
576     protected <V> V doWithRetry(Document inputDocument, SnapCallable<V> callable) {
577         for (int i = 1; i <= maxRequests; i++) {
578             try {
579                 // If a call succeeds, just immediately return the result.
580                 return callable.call();
581             } catch (SnapDataException e) {
582                 // Log every failure.
583                 String message = String.format(CAUGHT_EXCEPTION_ON_REQUEST, i, maxRequests);
584                 logError(message, e);
585                 if (skipRetry(e)) {
586                     errorViews.write(e, inputDocument);
587                     break;
588                 }
589                 if (i < maxRequests) {
590                     // If not the last allowed request, then sleep before retrying.
591                     try {
592                         threadHelper.sleep(retryIntervalInSecs * 1000);
593                     } catch (InterruptedException ie) {
594                         logger.warn("interrupted", ie);
595                         threadHelper.restoreInterrupt();
596                         break;
597                     }
598                 } else {
599                     // The last allowed request failed, so write the exception to the error view.
600                     errorViews.write(e, inputDocument);
601                 }
602             }
603         }
604         return null;
605     }

787         GetAsyncResultCustom getAsyncResult =
788                 injector.getInstance(GetAsyncResultCustom.class);
789         // init Get Async Result
790         getAsyncResult.account = account;

795         Map<String, String> requestData = ImmutableMap.of(JOB_ID, jobId);

799         Map<String, Object> responseJson = (Map<String, Object>) getAsyncResult.getResponseJson();

813                     boolean isSuccess = Boolean.parseBoolean((String) statusMap.get(IS_SUCCESS));
814                     if (inputDocIterator.hasNext()) {
815                         Map<String, Object> inputMap = inputDocIterator.next();
816                         if (isSuccess) {

834                                 writeError(ERR_RECORD_NOT_FOUND, REASON_UNEXPECTED_RESPONSE_SCHEMA,
