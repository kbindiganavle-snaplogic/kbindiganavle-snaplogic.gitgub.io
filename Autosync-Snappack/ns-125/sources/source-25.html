335     public void mergeCustomFieldData(Map<String, Object> valuesMap,
336             OperationType operationType,
337             List<? extends Map> customFieldProperties,
338             Document document) {
339         checkNotNull(customFieldProperties);
340         ListMultimap<String, Map<String, Object>> pathToFieldDataListMultimap = ArrayListMultimap
341                 .create();
342         for (Map customFieldDef : customFieldProperties) {
343             String label = PATH_LABEL.readStatic(customFieldDef);
344             String internalId = PATH_INTERNAL_ID.readStatic(customFieldDef);
345             String type = PATH_FIELD_TYPE.readStatic(customFieldDef);
346             String sublistName = PATH_SUBLIST_NAME.readStatic(customFieldDef, null);
347 
348             // label is either "customFieldName" or "sublistItemName.customFieldName"
349             int iDot = label.indexOf('.');
350             String customFieldName = iDot == -1 ? label : label.substring(iDot + 1);
351             String sublistItemName = iDot == -1 ? null : label.substring(0, iDot);
352 
353             LOGGER.debug("NetsuiteUtil " + label + " " + internalId + " " + type + " " +
354                     sublistName + " " + customFieldName + " " + sublistItemName);
355             switch (operationType) {
356                 case CREATE:
357                 case UPDATE:
358                 case UPSERT:
359                     ExpressionProperty customFieldProperty = (ExpressionProperty) customFieldDef
360                             .get(VALUE);
361                     Object customFieldValue = customFieldProperty.eval(document);
362                     if (sublistItemName != null) {
363                         // The custom field is for sublist items.
364                         // If customFieldValue is a List, associate each element with the sublist
365                         // item with the same index.
366                         // If the value not a List, associate this value with every sublist item.
367                         String sublistPath = checkNotNull(sublistName) + '.' + sublistItemName;
368                         List sublist = jsonPathUtil.nullableRead(sublistPath, valuesMap);
369                         int sublistSize = (sublist == null ? 0 : sublist.size());
370                         Object[] customValues; // array of custom field values corresponding to
371                         // each sublist item
372                         if (customFieldValue instanceof List) {
373                             List customValuesList = (List) customFieldValue;
374                             int customValuesSize = customValuesList.size();
375                             if (customValuesSize != sublistSize) {
376                                 throw new SnapDataException(SUBLIST_LENGTH_MISMATCH_MSG)
377                                         .formatWith(label, customValuesSize, sublistSize,
378                                                 sublistName)
379                                         .withReason(SUBLIST_LENGTH_MISMATCH_REASON)
380                                         .withResolution(SUBLIST_LENGTH_MISMATCH_RESOLUTION);
381                             }
382                             customValues = customValuesList.toArray();
383                         } else {
384                             customValues = new Object[sublistSize];
385                             Arrays.fill(customValues, customFieldValue);
386                         }
387                         String customValuesPath = CUSTOM_VALUES_PATH;
388                         String extraPath = customizationResolver.getSublistNameToExtraPath
389                                 (sublistName);
390                         if (extraPath != null) {
391                             customValuesPath = extraPath + '.' + customValuesPath;
392                         }
393                         for (int i = 0; i < sublistSize; i++) {
394                             Map<String, Object> fieldData = createFieldData(internalId, type,
395                                     customFieldName);
396                             fieldData.put(VALUE, customValues[i]);
397                             String path = String.format("%s[%d].%s", sublistPath, i,
398                                     customValuesPath);
399                             pathToFieldDataListMultimap.put(path, fieldData);
400                         }
401                     } else {
402                         // The custom field is for the root object, not a sublist.
403                         Map<String, Object> fieldData = createFieldData(internalId, type,
404                                 customFieldName);
405                         // If the custom field is a MultiSelectCustomFieldRef, then the value
406                         // needs to be converted into a list to match the changes in the
407                         // NetSuiteTemplatizerImpl.complexType() method.
408 
409                         // The format of the generated list in request SOAP message should be:
410                         // <ns3:value ns3:internalId="1"/> <ns3:value ns3:internalId="2"/>
411                         if (MULTI_SELECT_CUSTOM_FIELD_REF.equals(type)) {
412                             customFieldValue = setMultiSelectField(customFieldValue, label);
413                         }
414                         fieldData.put(VALUE, customFieldValue);
415                         pathToFieldDataListMultimap.put(CUSTOM_VALUES_PATH, fieldData);
416                     }
417                     break;
418                 case SEARCH:
419                     if (sublistItemName != null) {
420                         throw new UnsupportedOperationException
421                                 (SUBLIST_CUSTOM_FIELDS_NOT_SUPPORTED);
422                     }
423                     ExpressionProperty customFieldSearchProperty = (ExpressionProperty)
424                             customFieldDef.get(SEARCH_VALUE);
425                     Object searchValue = customFieldSearchProperty.eval(document);
426 
427                     // If the custom type is a list, then the search value has to be modified
428                     // to match the changes in the NetSuiteTemplatizerImpl.complexType() method.
429 
430                     // The searchValue will contain data looking like this if it is a list:
431                     // <ns4:searchValue>[{internalId=1}, {internalId=3}]</ns4:searchValue>
432 
433                     // The format for the list should be:
434                     // <ns4:searchValue ns4:internalId="1" /> <ns4:searchValue ns4:internalId="3" />
435                     if (SEARCH_MULTI_SELECT_CUSTOM_FIELD.equals(type)) {
436                         searchValue = setMultiSelectField(searchValue, label);
437                     }
438                     Map<String, Object> fieldData = createFieldData(internalId, type,
439                             customFieldName);
440                     fieldData.put(SEARCH_VALUE, searchValue);
441                     // Add the operator, unless the type is boolean, which doesn't take an operator.
442                     if (!SEARCH_BOOLEAN_CUSTOM_FIELD.equals(type)) {
443                         ExpressionProperty customFieldOperatorProperty = (ExpressionProperty)
444                                 customFieldDef.get(OPERATOR);
445                         String operator = customFieldOperatorProperty.eval(document);
446                         fieldData.put(OPERATOR, operator);
447                     }
448                     pathToFieldDataListMultimap.put(CUSTOM_VALUES_PATH, fieldData);
449                     break;
450                 default:
451                     throw new UnsupportedOperationException(UNEXPECTED_OPERATION_TYPE);
452             }
453         }
454         /* At this point, the multimap might look something like this (for create/update/upsert):
455         {
456              "customFieldList.customField"=[
457                 {scriptId=custbodymytextarea, internalId=71, value=abc, type=StringCustomFieldRef},
458                 {scriptId=custbodymycurrency, internalId=63, value=1.23, type=DoubleCustomFieldRef}
459                 ],
460              "itemList.item[0].customFieldList.customField"=[
461                 {scriptId=custcolmystring, internalId=54, value=xyz0, type=StringCustomFieldRef},
462                 {scriptId=custcolterm, internalId=127, value=0, type=LongCustomFieldRef}
463                 ],
464              "itemList.item[1].customFieldList.customField"=[
465                 {scriptId=custcolmystring, internalId=54, value=xyz1, type=StringCustomFieldRef},
466                 {scriptId=custcolterm, internalId=127, value=1, type=LongCustomFieldRef}
467                 ],
468              "itemList.item[2].customFieldList.customField"=[
469                 {scriptId=custcolmystring, internalId=54, value=xyz2, type=StringCustomFieldRef},
470                 {scriptId=custcolterm, internalId=127, value=2, type=LongCustomFieldRef}
471                 ]
472             }
473          */
474         for (String path : pathToFieldDataListMultimap.keySet()) {
475             List<Map<String, Object>> fieldDataList = pathToFieldDataListMultimap.get(path);
476             // make a serializable copy of the list
477             fieldDataList = new ArrayList<>(fieldDataList);
478             jsonPathUtil.write(valuesMap, path, fieldDataList);
479         }
480     }

482     private Object setMultiSelectField(Object customFieldValue, String label) {
483         if (customFieldValue instanceof String) {
484             String[] values = ((String) customFieldValue).split(",");
485             customFieldValue = new ArrayList<Map<String, String>>();
486             Map<String, String> m;
487             for (String v : values) {
488                 m = new HashMap<>();
489                 m.put(NetsuiteConstants.INTERNAL_ID, v);
490                 ((List) customFieldValue).add(m);
491             }
492         } else if (customFieldValue instanceof List) {
493             List customFieldValueList = (List) customFieldValue;
494             customFieldValue = new ArrayList<Map<String, String>>();
495             Map<String, String> m;
496             for (Object object : customFieldValueList) {
497                 m = new HashMap<>();
498                 m.put(NetsuiteConstants.INTERNAL_ID, object.toString());
499                 ((List) customFieldValue).add(m);
500             }
501         } else {
502             throw new SnapDataException(INPUT_ERROR)
503                     .withReason(String.format(WRONG_INPUT_TYPE_FOUND_FOR_S, label))
504                     .withResolution(PLEASE_CORRECT_INPUT_DATA_AND_RETRY);
505         }
506         return customFieldValue;
507     }

584     public void processNullFields(Map<String, Object> valuesMap) {
585         List<String> nullFieldList = new ArrayList<>();
586 
587         // Find any standard fields with a null value.
588         // Remove these fields and add their names to nullFieldList.
589         Iterator<Map.Entry<String, Object>> itEntry = valuesMap.entrySet().iterator();
590         while (itEntry.hasNext()) {
591             Map.Entry<String, Object> entry = itEntry.next();
592             if (entry.getValue() == null) {
593                 nullFieldList.add(entry.getKey());
594                 itEntry.remove();
595             }
596         }
597 
598         // Find any custom fields with a null value.
599         // Remove these fields and add their names to nullFieldList.
600         List<Map<String, Object>> customFieldList =
601                 jsonPathUtil.nullableRead(CUSTOM_VALUES_PATH, valuesMap);
602         if (customFieldList != null) {
603             Iterator<Map<String, Object>> itField = customFieldList.iterator();
604             while (itField.hasNext()) {
605                 Map<String, Object> fieldMap = itField.next();
606                 if (fieldMap.containsKey(VALUE) && fieldMap.get(VALUE) == null) {
607                     String scriptId = (String) fieldMap.get(SCRIPT_ID);
608                     if (scriptId != null) {
609                         nullFieldList.add(scriptId);
610                         itField.remove();
611                     }
612                 }
613             }
614             // If the customFieldList is now empty, remove it.
615             if (customFieldList.isEmpty()) {
616                 valuesMap.remove(CUSTOM_FIELD_LIST);
617             }
618         }
619 
620         // Add nullFieldList to the document if it's not empty.
621         if (!nullFieldList.isEmpty()) {
622             List<String> nullFieldNames =
623                     jsonPathUtil.nullableRead(NULL_FIELD_NAMES_PATH, valuesMap);
624             if (nullFieldNames == null) {
625                 jsonPathUtil.write(valuesMap, NULL_FIELD_NAMES_PATH, nullFieldList);
626             } else {
627                 // The document already contains nullFieldList, so add to the existing list.
628                 // It's not necessary to worry about potential duplicates; NS ignores them.
629                 nullFieldNames.addAll(nullFieldList);
630             }
631         }
632     }
