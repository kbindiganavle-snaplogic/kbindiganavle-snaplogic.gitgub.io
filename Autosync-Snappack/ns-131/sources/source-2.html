355     protected void configureSnapSettings(PropertyValues propertyValues) {
356         if (isStagingSupported()) {
357             String dataSourceValue = propertyValues.get(DATA_SOURCE_PROP);
358             if (StringUtils.isNotBlank(dataSourceValue)) {
359                 dataSource = dataSourceValue;
360             }
361             stagedFileUpload = dataSource.equals(DATA_SOURCE_STAGED_FILES);
362             s3PathExp = propertyValues.getAsExpression(S3_PATH_PROP);
363             delimiterExp = propertyValues.getAsExpression(S3_DELIMITER_PROP);
364         }
365         //inputview is must If datasource is set to input view
366         if(dataSource.equals(DATA_SOURCE_INPUT_VIEW) && inputViews.isEmpty()){
367             throw new ConfigurationException(ERROR_NO_INPUT_VIEW)
368                     .withReason(ERROR_NO_INPUT_VIEW_REASON)
369                     .withResolution(ERROR_NO_INPUT_VIEW_RESOLUTION);
370         }
371         awsAccessId = ((RedshiftDatabaseAccount) account).getAWSAccessId();
372         awsSecretKey = ((RedshiftDatabaseAccount) account).getAWSSecretKey();
373         awsSecurityToken = ((RedshiftDatabaseAccount) account).getAwsSecurityToken();
374         s3Bucket = ((RedshiftDatabaseAccount) account).getBucketName();
375         s3Folder = ((RedshiftDatabaseAccount) account).getFolderPath();
376         regionName = ((RedshiftDatabaseAccount) account).getRegionName();
377         serverless = ((RedshiftDatabaseAccount) account).isServerless();
378         crossAccountIamRoleAccount = (account instanceof RedshiftCrossAccountIAMRoleAccount);
379         ((RedshiftDatabaseAccount) account).setStaging(stagedFileUpload);
380         if (crossAccountIamRoleAccount) {
381             ((RedshiftCrossAccountIAMRoleAccount) account).validateAccountCustom();
382         }
383         // if Access ID and Secret key are empty, EC2 IAM credential can be used in EC2 Groundplex
384         // do this check only if the account is not a cross-account IAM role one
385         // for cross-account iam role accounts, checks are done in the account itself
386         if (!(crossAccountIamRoleAccount && stagedFileUpload)) {
387             if (StringUtils.isEmpty(s3Bucket)) {
388                 throw new ExecutionException(ERROR_MISSING_S3_BUCKET)
389                         .withReason(ERROR_S3_BUCKET_REQUIRED)
390                         .withResolution(ERROR_PROVIDE_S3_BUCKET);
391             }
392         }
393         validateInputData = Boolean.TRUE.equals(propertyValues.get(VALIDATE_INPUT_DATA_PROP));
394         truncate = propertyValues.get(TRUNCATE_PROP);
395         vacuum = Boolean.TRUE.equals(propertyValues.get(VACUUM_PROP));
396         vacuumType = propertyValues.get(RedshiftUtils.VACUUM_TYPE_PROP);
397         vacuumThreshold = propertyValues.get(RedshiftUtils.VACUUM_THRESHOLD_PROP);
398         updateStats = propertyValues.get(UPDATE_STATS_PROP);
399         acceptInvalidChars = propertyValues.get(ACCEPT_INVALID_CHARS_PROP);
400         maxErrors = ((BigInteger) propertyValues.get(MAX_ERROR_PROP)).intValue();
401         truncateColumns = propertyValues.get(TRUNCATE_COLUMNS_PROP);
402         additionalOptions = propertyValues.get(ADDITIONAL_OPTIONS_PROP);
403         nullAsInAdditionalOptions = StringUtils.containsIgnoreCase(additionalOptions, NULL);
404         emptyAsInAdditionalOptions = StringUtils.containsIgnoreCase(additionalOptions,
405                 EMPTY_AS_NULL.trim());
406         blankAsInAdditionalOptions = StringUtils.containsIgnoreCase(additionalOptions,
407                 BLANK_AS_NULL.trim());
408         loadEmptyStrings = Boolean.TRUE.equals(propertyValues.get(LOAD_EMPTY_STRINGS_PROP));
409         String emptyNullOptions = StringUtils.EMPTY;
410         if (!loadEmptyStrings) {
411             if (!emptyAsInAdditionalOptions) {
412                 emptyNullOptions = emptyNullOptions.concat(EMPTY_AS_NULL);
413             }
414             if (!blankAsInAdditionalOptions) {
415                 emptyNullOptions = emptyNullOptions.concat(BLANK_AS_NULL);
416             }
417         } else if (!nullAsInAdditionalOptions) {
418             emptyNullOptions = NULL_AS;
419         }
420         additionalOptions = additionalOptions == null ? emptyNullOptions :
421                 emptyNullOptions.concat(additionalOptions);
422         disableCompression = propertyValues.get(DISABLE_COMPRESSION_PROP);
423         // check if the instance type is R6a or equivalent
424         isHighPerformanceS3 = StringUtils.equalsIgnoreCase(propertyValues.get(INSTANCE_TYPE_PROP),
425                 HIGH_PERFORMANCE_S3);
426         parallelism = propertyValues.get(PARALLELISM);
427         if (parallelism == null) {
428             parallelism = BigInteger.ONE;
429         } else if (isHighPerformanceS3 && parallelism.compareTo(MAX_PARALLELISM) > 0) {
430             // For parallelism > 4, the Snap performs lower in R6a instance.
431             parallelism = MAX_PARALLELISM;
432             statusReporter.notify(PARALLELISM_SET_4);
433         }
434         iamRoleEnabled = Boolean.TRUE.equals(propertyValues.get(IAM_ROLE_PROP));
435         serverSideEncryptionEnabled = Boolean.TRUE.equals(propertyValues.get(
436                 SERVER_SIDE_ENCRYPTION_PROP));
437         if (!crossAccountIamRoleAccount) {
438             iamRoleName = ((RedshiftDatabaseAccount) account).getIAMRoleNameForRedshiftType();
439             awsAccountID = ((RedshiftDatabaseAccount) account).getAwsAccountID();
440             // check for invalid configuration where IAM role is given along with Access-Key ID and
441             // Secret-Key
442             loadUnloadUtils.performInvalidIamConfigurationCheckOne(iamRoleEnabled, awsAccessId,
443                     awsSecretKey, awsSecurityToken);
444             // check for invalid configuration where IAM role is given but one of the properties
445             // (role name, AWS account ID) is missing
446             loadUnloadUtils.performInvalidIamConfigurationCheckTwo(iamRoleEnabled, awsAccountID,
447                     iamRoleName);
448         }
449         // TODO tstack - look into adding a migration script to change existing view names to be
450         // row/schema.
451         List<String> viewOrder = viewNameUtil.sortViewNames(inputViews.names(),
452                 INPUT_ROWS_VIEW_NAME);
453         if (!viewOrder.isEmpty()) {
454             rowViewName = viewOrder.get(0);
455             if (viewOrder.size() > 1) {
456                 schemaViewName = viewOrder.get(1);
457             }
458         }
459         if (serverSideEncryptionEnabled) {
460             if (encryptionType == S3EncryptionType.NONE) {
461                 encryptionType = SSE;
462             } else if (encryptionType.isKms()) {
463                 throw new ConfigurationException(DISABLE_SSE_FOR_KMS)
464                         .withReason(DISABLE_SSE_FOR_KMS_REASON);
465             }
466         }
467     }

471         if (stagedFileUpload) {
472             stagedS3Path = s3PathExp.eval(document);
473             if (StringUtils.isBlank(stagedS3Path)) {
474                 throw new SnapDataException(ERR_SNAP_CONFIG)
475                         .withReason(S3_PATH_MISSING)
476                         .withResolution(RESOLUTION_ADDRESS_ISSUE);
477             }
478             stagedS3Delimiter = delimiterExp.eval(document);
479             if (StringUtils.isBlank(stagedS3Delimiter)) {
480                 stagedS3Delimiter = String.valueOf(COMMA);
481             }
482             try {
483                 loadStagedFiles(document);
484                 return;
485             } catch (SQLException e1) {
486                 throw new SnapDataException(e1, ERR_STAGED_LOAD_FAILED)
487                         .withReason(e1.getMessage())
488                        .withResolution(ERR_STAGED_LOAD_FAILED_RESOLUTION);
489             }
490         }
491         // Check if this is the schema document and save it for later.
492         if (inputViewName != null && inputViewName.equals(schemaViewName)) {
493             if (document != null) {
494                 schema = document;
495                 columnTranslation = databaseUtils.buildColumnTranslation(schema);
496             }
497             return;
498         }
499         // Validate input document
500         if (!validateInputDocument(document)) {
501             return;
502         }
503 
504         databaseUtils.translateColumnNamesInDoc(columnTranslation, document);
505 
506         if (firstRecord) {
507             firstRecord = false;
508             if (autoCreateTable) {
509                 createTableIfAbsent(document);
510             }
511 
512             LOG.debug("Getting the metadata of the table");
513             long startTime = System.currentTimeMillis();
514             Connection conn = jdbcOperations.acquireConnection(account);
515             tableMetaData = jdbcOperations.getTableMetaData(conn,
516                     account.getDatabaseConfig().getDatabaseName(), tableName);
517             logExecutionTime(startTime, "DONE Getting the metadata of the table in {} ms");
518 
519             /*
520             Invoking getColumns() method on DatabaseMetaData instance for Redshift doesn't throw
521             error if the provided table doesn't exist. This seems to be a drawback of the
522             PostgreSQL database JDBC driver which is eventually changing the behaviour of
523             Redshift Bulk Load snap during execution. To overcome the above limitation and to
524             correct the behaviour of Redshift Bulk Load snap during the execution,
525             the below if block is added.
526             */
527             if (tableMetaData == null || tableMetaData.keySet() == null ||
528                     tableMetaData.keySet().size() == 0) {
529                 throw new ConfigurationException(ERR_FETCH_TABLE_METADATA_MSG)
530                         .formatWith(tableName)
531                         .withReason(ERR_FETCH_TABLE_METADATA_REASON)
532                         .withResolution(ERR_FETCH_TABLE_METADATA_RESOLUTION);
533             }
534             bulkLoadWrapper.setTableMetaData(tableMetaData);
535             Map<String, Object> data = document.get(Map.class);
536             if (data == null) {
537                 return;
538             }
539             for (String column : data.keySet()) {
540                 String normalizedColumn = jdbcOperations.normalize(column);
541                 if (tableMetaData.containsKey(normalizedColumn)) {
542                     copyColumns.add(column);
543                     quotedCopyColumns.add(String.format(ID_QUOTE_FORMAT, column));
544 
545                     ColumnMetaData columnMeta = tableMetaData.get(normalizedColumn);
546                     if (columnMeta.isIdentity()) {
547                         // An auto increment column is being passed in,
548                         // add the EXPLICIT_IDS option to the copy command
549                         explictIds = EXPLICIT_IDS;
550                     }
551                 }
552             }
553             if (copyColumns.isEmpty()) {
554                 throw new ExecutionException(ERROR_INVALID_INPUT)
555                         .withReason(ERROR_INVALID_INPUT_REASON)
556                         .withResolution(ERROR_INVALID_INPUT_RESOLUTION);
557             }
558             // If parallelism > 1, initialize multi-threads to CSV-format, GZIP-compress and
559             // upload data to S3 in parallel.
560             int threadCount = parallelism.intValue();
561             if (threadCount > 1 && isHighPerformanceS3) {
562                 initTasks(threadCount);
563             }
564         }
565         inputCount++;
566         Map<String, Object> data = document.get(Map.class);
567         if (data == null) {
568             return;
569         } else {
570             if (isAllCopyColumnsNull(data)) {
571                 // for RS Upsert Snap, write to error view if all values for copy columns are null
572                 SnapDataException ex = new SnapDataException(ERR_ALL_COLUMNS_NULL)
573                         .withReason(REASON_ALL_COLUMNS_NULL)
574                         .withResolution(RESOLUTION_ADDRESS_ISSUE);
575                 errorViews.write(ex, document);
576                 return;
577             }
578         }
579         if (loadEmptyStrings && !nullAsInAdditionalOptions) {
580             data = replaceNullWithMark(data);
581         }
582         if (!validateInputData || validateInputBulkLoad(document)) {
583             // if validateInputData is true and any invalid input document is found, return without
584             // writing any output document, since the error count from the error table is irrelevant
585             document.acknowledge();
586             writeCount++;
587             if (CollectionUtils.isEmpty(s3Uploads)) {
588                 bulkLoadWrapper.write(data, null);
589             } else {
590                 try {
591                     dataQueue.put(data);
592                 } catch (InterruptedException e) {
593                     LOG.debug("Interrupted while inserting data into dataQueue", e);
594                 }
595             }
596         }
597     }

745     /**
746      * The load action.
747      */
748     protected void loadAction(String targetTable) throws SQLException {
749         // Connection might have been created already if truncate was done
750         redshiftConnection = getRedshiftConnection(redshiftConnection);
751 
752         String acceptInvalidCharStr = ACCEPT_INVALID_CHARS;
753         if (!acceptInvalidChars) {
754             acceptInvalidCharStr = StringUtils.EMPTY;
755         }
756 
757         String truncateColumnsStr = TRUNCATE_COLUMNS;
758         if (!truncateColumns) {
759             truncateColumnsStr = StringUtils.EMPTY;
760         }
761 
762         String additionalOptionsStr = StringUtils.EMPTY;
763         if (!StringUtils.isEmpty(additionalOptions)) {
764             additionalOptionsStr = additionalOptions;
765         }
766 
767         String columns = StringUtils.join(quotedCopyColumns, COMMA);
768         if ((!stagedFileUpload) && StringUtils.isBlank(columns)) {
769                 return;
770         }
771         String copyCommand = bulkLoadWrapper.createCopyCommand(targetTable, columns,
772                 acceptInvalidCharStr, maxErrors, explictIds, truncateColumnsStr,
773                 additionalOptionsStr, stagedS3Path, stagedS3Delimiter);
774         String copyCommandObfuscated = null;
775         try {
776             copyCommandObfuscated = getObfuscatedCommand(copyCommand);
777             if (!additionalOptionsStr.toUpperCase().contains(DATE_FORMAT)) {
778                 copyCommand = copyCommand.concat(COPY_DATE_FORMAT);
779                 copyCommandObfuscated = copyCommandObfuscated.concat(COPY_DATE_FORMAT);
780             }
781             if (!additionalOptionsStr.toUpperCase().contains(TIME_FORMAT)) {
782                 copyCommand = copyCommand.concat(COPY_TIME_FORMAT);
783                 copyCommandObfuscated = copyCommandObfuscated.concat(COPY_TIME_FORMAT);
784             }
785         } catch (Exception e) {
786             //NO OP
787             //Ignoring errors raised while generating log message
788             LOG.warn(LOG_ERROR_COPYCOMMANDOBFUSCATED_GENERATION, e);
789         }
790         try (Statement stmt = redshiftConnection.createStatement()) {
791             LOG.debug(LOG_RUNNING_COPY, copyCommandObfuscated);
792             long startTime = System.currentTimeMillis();
793             stmt.execute(RedshiftUtils.addApnTaggingPrefix(copyCommand));
794             logExecutionTime(startTime, "DONE Creating the copy operation in {} ms");
795             LOG.debug(LOG_COPY_DONE);
796             commitSilently();
797         } catch (SQLException se) {
798             if (isDifferentRegionException(se.getMessage())) {
799                 isDiffRegionException = true;
800                 throwDiffRegionException(se);
801             }
802             handleException(se);
803         } finally {
804             writeErrorView();
805         }
806     }

897     /**
898      * Write all error entries from the stl_load_errors table into the error view.
899      */
900     private void writeErrorView() throws SQLException {
901         if (errorViews.isEmpty() || isDiffRegionException) {
902             return;
903         }
904 
905         Set<Integer> lineSet = Sets.newHashSet();
906         try (PreparedStatement stmt = redshiftConnection.prepareStatement(
907                 getLoadErrorSql(serverless, false))) {
908             stmt.setString(1, bulkLoadWrapper.getCopyCommandPath());
909             try (ResultSet rs = stmt.executeQuery()) {
910                 while (rs.next()) {
911                     int lineNumber = rs.getInt(1);
912                     if (lineSet.contains(lineNumber)) {
913                         // Only one error view entry per input record even if multiple
914                         // errors encountered on that record
915                         continue;
916                     }
917                     lineSet.add(lineNumber);
918                     String colName = StringUtils.trim(rs.getString(2));
919                     int errorCode = rs.getInt(3);
920                     String errorMessage = StringUtils.trim(rs.getString(4));
921 
922                     Map<String, Object> errorInfo = Maps.newLinkedHashMap();
923                     errorInfo.put(KEY_LINE_NUMBER, lineNumber);
924                     errorInfo.put(KEY_COLUMN_NAME, colName);
925                     errorInfo.put(KEY_ERROR_CODE, errorCode);
926                     errorInfo.put(KEY_ERROR_MESSAGE, errorMessage);
927 
928                     String reason = null;
929                     if (serverless) {
930                         reason = String.format(SERVERLESS_ERROR_LOAD_DATA_ROW, lineNumber,
931                             colName, errorCode, errorMessage);
932                     } else {
933                         String column_value = StringUtils.trim(rs.getString(6));
934                         String row_value = StringUtils.trim(rs.getString(7));
935                         errorInfo.put(KEY_COLUMN_VALUE, column_value);
936                         errorInfo.put(KEY_ROW_VALUE, row_value);
937                         reason = String.format(ERROR_LOAD_DATA_ROW, lineNumber, colName,
938                             column_value, row_value, errorCode, errorMessage);
939                     }
940 
941                     Document doc = documentUtility.newDocument(errorInfo);
942                     // Unlike regular error views, the complete input document is not
943                     // available, just whatever is provided by Redshift
944                     SnapDataException exception = new SnapDataException(doc, errorMessage)
945                             .withReason(reason)
946                             .withResolution(ERR_DATA_LOAD_ROW_RESOLUTION);
947                     errorViews.write(exception);
948                 }
949             }
950         }
951     }

971     private void handleException(Exception se) throws ExecutionException, SQLException {
972         StringBuilder error = new StringBuilder();
973         try (PreparedStatement stmt = redshiftConnection.prepareStatement(
974                 getLoadErrorSql(serverless, false))) {
975             if (!account.getDatabaseConfig().getAutoCommit()) {
976                 // if 'Auto commit' is false, Redshift won't let the error table read.
977                 // commit() should be executed here.
978                 // ToDo - psung: this workacount should be re-visited when we change the JDBC driver
979                 // to Redshift JDBC driver in the future.
980                 redshiftConnection.commit();
981             }
982             stmt.setString(1, bulkLoadWrapper.getCopyCommandPath());
983             try (ResultSet rs = stmt.executeQuery()) {
984                 int count = 0;
985                 while (rs.next()) {
986                     error.append(String.format(ERROR_COPY_DOCUMENT, rs.getInt(1),
987                             StringUtils.trim(rs.getString(2)), rs.getInt(3), StringUtils.trim(rs.getString(4))));
988                     count++;
989                     if (count >= ERROR_MESSAGE_ENTRY_COUNT) {
990                         // Just put first five errors in error message, more
991                         // can go to error view if created
992                         break;
993                     }
994                 }
995             }
996 
997             String errorStr = error.toString();
998             if (errorStr.isEmpty()) {
999                 errorStr = se.getLocalizedMessage();
1000             }
1001             if (errorStr.contains("UnauthorizedException")) {
1002                 // an iam role not associated with a cluster will raise this exception
1003                 unauthorizedException = true;
1004             }
1005             errorViews.write((SnapDataException)
1006                     new SnapDataException(se, ERROR_COPYING_S3_REDSHIFT)
1007                             .withReason(errorStr)
1008                             .withResolution(ERROR_FIX_DATA_ISSUES));
1009         }
1010     }

1012     @Override
1013     public void execute() {
1014         if (stagedFileUpload) {
1015             if (crossAccountIamRoleAccount) {
1016                 super.execute();
1017                 return;
1018             } else {
1019                  throw new SnapDataException(ERR_SNAP_CONFIG)
1020                          .withReason(ERR_STAGED_NOT_SUPPORTED)
1021                          .withResolution(RESOLUTION_ADDRESS_ISSUE);
1022             }
1023         }
1024         if (!hasInputDocuments(inputViews)) {
1025             return;
1026         }
1027         LOG.debug("Writing data to S3");
1028         long startTime = System.currentTimeMillis();
1029         try {
1030             bulkLoadWrapper = createBulkLoadWrapperInstance();
1031             ((BulkLoadWrapperS3) bulkLoadWrapper).configureWrapper(
1032                     serverSideEncryptionEnabled, encryptionType, kmsKey);
1033         } catch (Exception e) {
1034             String reason = e.getCause() != null ? e.getCause().getMessage() : null;
1035             reason = reason == null ? String.format(ERR_EXECUTE_SNAP, getName()) : reason;
1036             Map<String, Object> wrapper = Maps.newHashMapWithExpectedSize(5);
1037             wrapper.put(ERROR, e.getMessage());
1038             wrapper.put(STACKTRACE, Throwables.getStackTraceAsString(e.getCause() != null ?
1039                     e.getCause() : e));
1040             wrapper.put(REASON, reason);
1041             wrapper.put(RESOLUTION, RESOLUTION_ADDRESS_ISSUE_WITH_SEC_TOKEN);
1042             SnapDataException ex = new SnapDataException(documentUtility.newDocument(wrapper), e,
1043                     e.getMessage())
1044                     .withReason(reason)
1045                     .withResolution(RESOLUTION_ADDRESS_ISSUE_WITH_SEC_TOKEN);
1046             errorViews.write(ex);
1047             return;
1048         }
1049         super.execute();
1050         // Wait until all S3 upload tasks are completed.
1051         if (CollectionUtils.isNotEmpty(s3Uploads)) {
1052             s3Uploads.forEach(p -> p.setHasMoreRecords(false));
