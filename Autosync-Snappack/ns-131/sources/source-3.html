231     protected void loadAction(String targetTable) throws SQLException {
232         String[] parts = quotationHandler.splitFullyQualifiedName(targetTable, 0);
233         String unescapedTableName = quotationHandler.unescape(parts[parts.length - 1]);
234         parts[parts.length - 1] = quotationHandler.escape(String.format(TEMP_TABLE_FORMAT,
235                 unescapedTableName, UUID.randomUUID().toString().replaceAll(HYPHEN, UNDERSCORE)));
236         StringBuilder builder = new StringBuilder(parts[0]);
237         for (int i = 1; i < parts.length; i++) {
238             builder.append(DOT_STRING).append(parts[i]);
239         }
240         String tempTable = builder.toString();
241         // Connection might have been created already if truncate was done
242         redshiftConnection = getRedshiftConnection(redshiftConnection);
243         // The steps followed are as documented at
244         // http://docs.aws.amazon.com/
245         //        redshift/latest/dg/t_updating-inserting-using-staging-tables-.html
246         try (Statement stmt = redshiftConnection.createStatement()) {
247             try {
248                 // Create temp table
249                 String createTempTableCommand = String.format(
250                         CREATE_TEMP_TABLE_COMMAND, tempTable, targetTable);
251                 log.debug(LOG_CREATING_TEMP_TABLE, createTempTableCommand);
252                 long startTime = System.currentTimeMillis();
253                 stmt.execute(RedshiftUtils.addApnTaggingPrefix(createTempTableCommand));
254                 logExecutionTime(startTime, "DONE Creating the temp table in {} ms");
255 
256                 // Run the copy from S3 to Redshift
257                 log.debug("Called the super loadAction table");
258                 startTime = System.currentTimeMillis();
259                 super.loadAction(tempTable);
260                 logExecutionTime(startTime, "DONE Called the super loadAction table in {} ms");
261 
262                 // Check for duplicates in the temp table
263                 log.debug("Checking for duplicates");
264                 if (validateInputKeyValue && checkDuplicates(stmt, tempTable)) {
265                     return;
266                 }
267                 logExecutionTime(startTime, "DONE Checking for duplicates in {} ms");
268 
269                 // Update from temp table to target table
270                 String updateStmt = createUpdateStmt(targetTable, tempTable);
271                 if (updateStmt == null) {
272                     // No update
273                     log.debug(LOG_NO_UPDATE, tempTable, targetTable);
274                 } else {
275                     log.debug(LOG_RUNNING_UPDATE, updateStmt);
276                     startTime = System.currentTimeMillis();
277                     stmt.execute(RedshiftUtils.addApnTaggingPrefix(updateStmt));
278                     logExecutionTime(startTime, "DONE Running the update from temp to target in {} ms");
279                     updateCount = stmt.getUpdateCount();
280                 }
281 
282                 // Insert from temp table to target table
283                 String insertStmt = createInsertStmt(targetTable, tempTable);
284                 log.debug(LOG_RUNNING_INSERT, insertStmt);
285                 startTime = System.currentTimeMillis();
286                 stmt.execute(RedshiftUtils.addApnTaggingPrefix(insertStmt));
287                 logExecutionTime(startTime, "DONE Running the insert from temp to target in {} ms");
288                 insertCount = stmt.getUpdateCount();
289             } finally {
290                 String dropTableQuery = String.format(DROP_TABLE_COMMAND, tempTable);
291                 try {
292                     log.debug("Started executing the drop table query: {}", dropTableQuery);
293                     long startTime = System.currentTimeMillis();
294                     stmt.execute(RedshiftUtils.addApnTaggingPrefix(dropTableQuery));
295                     logExecutionTime(startTime, "DONE Dropping the table in {} ms");
296                 } catch (SQLException e) {
297                     log.debug("Retrying the drop table command");
298                     long startTime = System.currentTimeMillis();
299                     retryDropTempTable(dropTableQuery, tempTable, e);
300                     logExecutionTime(startTime, "DONE Retrying the drop table command in {} ms");
301                 }
302             }
303         }
304     }

306     protected int retryDropTempTable(String dropTableQuery, String tempTable, SQLException e)
307             throws SQLException {
308         Connection connection = null;
309         int retried = 1;
310         boolean done = false;
311         Statement stmt = null;
312 
313         try {
314             // In SWAT-467, it's been reported the dropping temp table failed due to 'connection
315             // reset'. Since it failed once, we use a new connection to retry to drop the table.
316             try {
317                 connection = jdbcOperations.acquireConnection(account);
318                 stmt = connection.createStatement();
319             } catch (SQLException e1) {
320                 // if fails to get a new connection, fall back to the existing connection
321                 log.info(LOG_CONN_RETRY_DROP_TABLE, tempTable);
322                 stmt = redshiftConnection.createStatement();
323             }
324             while (!done) {
325                 try {
326                     log.info(FORMAT_RETRY_DROP_TABLE, tempTable,
327                             RETRY_INTERVAL_IN_SEC * retried, e);
328                     // the sleep interval to be increased in every retry
329                     DatabaseUtils.delay(RETRY_INTERVAL_IN_SEC * retried++ * 1000);
330                     stmt.execute(RedshiftUtils.addApnTaggingPrefix(dropTableQuery));
331                     // Drop table worked so we are done trying.
332                     done = true;
333                 } catch (SQLException e1) {
334                     if (retried >= MAX_DROP_TABLE_ATTEMPT) {
335                         SnapDataException ex =
336                                 new SnapDataException(e1, ERR_REACHED_MAX_DROP_TABLE_ATTEMPTS)
337                                 .formatWith(tempTable)
338                                 .withReason(e.getMessage())
339                                 .withResolution(ERR_REACHED_MAX_DROP_TABLE_ATTEMPTS_RESOLUTION);
340                         errorViews.write(ex);
341                         // Ran out of retries so we are done trying.
342                         done = true;
343                     } else {
344                         e = e1;
345                     }
346                 }
347             }
348         } finally {
349             JDBCUtils.safeClose(stmt);
350             JDBCUtils.safeClose(connection);
351         }
352 
353         return retried;
354     }

370     private boolean checkDuplicates(Statement stmt, String tableName) {
371         boolean duplicateFound = false;
372         ResultSet resultSet = null;
373         try {
374             // create a query to find duplicates in temp table:
375             // "SELECT *, COUNT(*) duplicate_count FROM temp_table
376             // GROUP BY column1, column2, column3, ... HAVING COUNT(*) > 1
377             StringBuilder keyColumnBuilder = new StringBuilder();
378             boolean firstColumn = true;
379             for (String name : keyColumnNames) {
380                 if (firstColumn) {
381                     firstColumn = false;
382                 } else {
383                     keyColumnBuilder.append(COMMA_DELIMITER);
384                 }
385                 keyColumnBuilder.append(quotationHandler.escape(name));
386             }
387             String duplicateStmt = String.format(DUPLICATE_COMMAND, keyColumnBuilder, tableName,
388                     keyColumnBuilder);
389             log.debug(LOG_DUPLICATES, duplicateStmt);
390             // execute query
391             resultSet = stmt.executeQuery(RedshiftUtils.addApnTaggingPrefix(duplicateStmt));
392             while (resultSet.next()) {
393                 // get duplicate row and count
394                 duplicateFound = true;
395                 Map<String, Object> map = Maps.newHashMapWithExpectedSize(copyColumns.size() + 1);
396                 for (String column : keyColumnNames) {
397                     map.put(column, resultSet.getObject(column));
398                 }
399                 map.put(DUPLICATE_COUNT, resultSet.getInt(DUPLICATE_COUNT));
400                 // write to error view
401                 SnapDataException ex = new SnapDataException(ERR_DUPLICATES)
402                         .withReason(REASON_DUPLICATES)
403                         .withResolution(RESOLUTION_ADDRESS_ISSUE);
404                 errorViews.write(ex, map);
405             }
406         } catch (SQLException e) {
407             SnapDataException ex = new SnapDataException(ERR_CHECKING_DUPLICATES)
408                     .withReason(e.getMessage())
409                     .withResolution(RESOLUTION_ADDRESS_ISSUE);
410             errorViews.write(ex);
411             return true;
412         } finally {
413             JDBCUtils.safeClose(resultSet);
414         }
415         return duplicateFound;
416     }

419         StringBuilder updateList = new StringBuilder();
420         boolean firstColumn = true;
421         for (String column : copyColumns) {
422             if (!keyColumnNames.contains(jdbcOperations.normalize(column))) {
423                 if (!firstColumn) {
424                     updateList.append(COMMA_DELIMITER);
425                 }
426                 firstColumn = false;
427                 updateList.append(String.format(UPDATE_ENTRY, column, column));
428             }
429         }

436         String updateCommand = String.format(UPDATE_COMMAND,
437                 targetTable, updateList, tempTable, predicateList);

441     private String createInsertStmt(final String targetTable, final String tempTable) {
442         StringBuilder predicateList = getPredicateList(TARGET_ALIAS, keyColumnNames);
443         // prepare a string for checking target table's non-key-column values are null, e.g.
444         // t.name1 is null AND t.name2 is null AND ...
445         StringBuilder targetValueIsNull = new StringBuilder();
446         boolean firstColumn = true;
447         for (String column : tableMetaData.keySet()) {
448             if (!keyColumnNames.contains(column)) {
449                 if (!firstColumn) {
450                     targetValueIsNull.append(AND_DELIMITER);
451                 }
452                 firstColumn = false;
453                 targetValueIsNull.append(String.format(TARGET_VALUE_IS_NULL, column));
454             }
455         }
456         // prepare all input columns list
457         List<String> allInputColumns = Lists.newArrayList();
458         for (String columnName : copyColumns) {
459             if (!tableMetaData.get(jdbcOperations.normalize(columnName)).isIdentity()) {
460                 allInputColumns.add(String.format(ID_QUOTE_FORMAT, columnName));
461             }
462         }
463         List<String> transformedList = Lists.transform(allInputColumns, new Function<String,
464                 String>() {
465             @Nullable
466             @Override
467             public String apply(@Nullable final String input) {
468                 return String.format(SELECT_COLUMN, input);
469             }
470         });
471         String insertColumns = Joiner.on(COMMA_DELIMITER).join(allInputColumns);
472         String selectColumns = Joiner.on(COMMA_DELIMITER).join(transformedList);
473         if (StringUtils.isNotBlank(targetValueIsNull)) {
474             return String.format(INSERT_COMMAND, targetTable, insertColumns,
475                     selectColumns, tempTable, targetTable, predicateList, targetValueIsNull,
476                     getAllTargetKeyColumnsNull(), getAllInputValuesNotNull(allInputColumns));
477         } else {
478             return String.format(INSERT_COMMAND_NO_TARGET_VALUE_IS_NULL, targetTable, insertColumns,
479                     selectColumns, tempTable, targetTable, predicateList,
480                     getAllTargetKeyColumnsNull(), getAllInputValuesNotNull(allInputColumns));
481         }
482     }
