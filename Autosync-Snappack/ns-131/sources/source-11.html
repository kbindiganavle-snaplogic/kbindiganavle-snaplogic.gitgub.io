365     @Override
366     protected void processDocument(final Document document, final String inputViewName) {
367         // evaluate property values
368         String selectQuery = selectQueryProp.eval(document);
369         String s3FilePrefix = s3FilePrefixProp.eval(document);
370         String delimiter = delimiterProp.eval(document);
371         String fixedWidth = fixedWidthProp.eval(document);
372         String nullAs = nullAsProp.eval(document);
373         String key;
374         String s3Bucket;
375 
376         configureRetry(document);
377         // validate SELECT query property value
378         if (StringUtils.isBlank(selectQuery)) {
379             SnapDataException ex = new SnapDataException(ERR_EMPTY_PROPERTY)
380                     .formatWith(SELECT_QUERY_LABEL)
381                     .withReason(REASON_PROPERTY_REQUIRED)
382                     .withResolution(RESOLUTION_ADDRESS_ISSUE);
383             errorViews.write(ex, document);
384             return;
385         }
386         selectQuery = handleEscapeQuote(selectQuery);
387         RedshiftDatabaseAccount redshiftAccount = (RedshiftDatabaseAccount) account;
388         // get bucket name, key, and S3 file prefix path
389         s3Bucket = StringUtils.strip(redshiftAccount.getBucketName(), SLASH);
390         if (StringUtils.isBlank(s3Bucket)) {
391             SnapDataException ex = new SnapDataException(ERR_EMPTY_PROPERTY).formatWith(S3_BUCKET)
392                     .withReason(REASON_PROPERTY_REQUIRED)
393                     .withResolution(RESOLUTION_S3_BUCKET);
394             errorViews.write(ex, document);
395             return;
396         }
397         s3FilePrefix = s3FilePrefix == null ? StringUtils.EMPTY : s3FilePrefix;
398         String s3Folder = StringUtils.strip(redshiftAccount.getFolderPath(), SLASH);
399         key = StringUtils.isBlank(s3Folder) ? s3FilePrefix :
400                 String.format(FORMAT_KEY, s3Folder, s3FilePrefix);
401         s3FilePrefix = String.format(FORMAT_S3_PATH, s3Bucket, key);
402         s3FilePrefix = s3FilePrefix == null ? StringUtils.EMPTY : s3FilePrefix;
403         // init query builder
404         StringBuilder queryBuilder = new StringBuilder(
405                 String.format(FORMAT_UNLOAD, selectQuery, s3FilePrefix));
406 
407         addEncryption(queryBuilder);
408 
409         // append DELIMITER
410         if (StringUtils.isNotBlank(delimiter) && !DELIMITER_DEFAULT.equalsIgnoreCase(delimiter)) {
411             queryBuilder.append(String.format(FORMAT_OPTION, DELIMITER_PROP.toUpperCase(),
412                     delimiter));
413         }
414         // append FIXEDWIDTH
415         if (StringUtils.isNotBlank(fixedWidth)) {
416             queryBuilder.append(String.format(FORMAT_OPTION, FIXED_WIDTH_PROP.toUpperCase(),
417                     fixedWidth));
418         }
419         // append GZIP
420         if (gzip) {
421             queryBuilder.append(String.format(FORMAT_SPACE, GZIP_PROP.toUpperCase()));
422         }
423         // append ADDQUOTES
424         if (addQuotes) {
425             queryBuilder.append(String.format(FORMAT_SPACE, ADD_QUOTES_PROP.toUpperCase()));
426         }
427         // append NULL AS
428         if (StringUtils.isNotBlank(nullAs)) {
429             queryBuilder.append(String.format(FORMAT_OPTION, NULL_AS, nullAs));
430         }
431         // append ESCAPE
432         if (escape) {
433             queryBuilder.append(String.format(FORMAT_SPACE, ESCAPE_PROP.toUpperCase()));
434         }
435         // append ALLOWOVERWRITE
436         if (allowOverwrite) {
437             queryBuilder.append(String.format(FORMAT_SPACE, ALLOW_OVERWRITE_PROP.toUpperCase()));
438         }
439         // append PARALLEL
440         if (!parallel) {
441             queryBuilder.append(String.format(FORMAT_OPTION_OFF, PARALLEL_PROP.toUpperCase()));
442         }
443         // append region if provided
444         if (StringUtils.isNotBlank(regionName)) {
445             queryBuilder.append(String.format(REGION_OPTION, regionName));
446         }
447         String unloadQuery = queryBuilder.toString();
448         String trackingTaggedUnloadQuery = RedshiftUtils.addApnTaggingPrefix(unloadQuery);
449         try {
450             if (!isPreview) {
451                 jdbcOperations.executeQuery(trackingTaggedUnloadQuery, null, account,
452                     maxRetries, retryInterval);
453             }
454         } catch (SnapDataException e) {
455             checkAndThrowDiffRegionException(e, e.getReason());
456 
457             // we want to skip this exception since its stack trace may contain S3 credential
458             SnapDataException ex = new SnapDataException(ERR_UNLOAD)
459                     .formatWith(removeS3Credential(unloadQuery, getCredentialsToBeMasked()))
460                     .withReason(e.getReason())
461                     .withResolution(RESOLUTION_ADDRESS_ISSUE);
462             errorViews.write(ex, document);
463             return;
464         } catch (ExecutionException e) {
465             SnapDataException ex = new SnapDataException(e,
466                     removeS3Credential(e.getMessage(), getCredentialsToBeMasked()))
467                     .withReason(e.getReason())
468                     .withResolution(e.getResolution());
469             errorViews.write(ex, document);
470             return;
471         } catch (SuggestViewAbortException e) {
472             throw e;

588     private String handleEscapeQuote(String string) {
589         // remove enclosing single or double quotes
590         if (StringUtils.startsWith(string, QUOTE)) {
591             string = string.substring(1);
592             if (StringUtils.endsWith(string, QUOTE)) {
593                 string = string.substring(0, string.length() - 1);
594             }
595         } else if (StringUtils.startsWith(string, DOUBLE_QUOTE)) {
596             string = string.substring(1);
597             if (StringUtils.endsWith(string, DOUBLE_QUOTE)) {
598                 string = string.substring(0, string.length() - 1);
599             }
600         }
601         // escape single quote and backslash
602         if (!StringUtils.contains(string, ESCAPED_QUOTE) &&
603                 !StringUtils.contains(string, ESCAPED_DOUBLE_QUOTE) &&
604                 !StringUtils.contains(string, ESCAPED_BACKSLASH)) {
605             string = StringUtils.replace(string, BACKSLASH, ESCAPED_BACKSLASH);
606             string = StringUtils.replace(string, QUOTE, ESCAPED_QUOTE);
607         }
608         return string;
609     }
