432     @Override
433     protected void processDocument(final Document document, final String inputViewName) {
434         try {
435             if(firstRecord) {
436                 Connection conn = jdbcOperations.acquireConnection(account);
437                 tableMetaData = jdbcOperations.getTableMetaData(conn,
438                         account.getDatabaseConfig().getDatabaseName(), tableName);
439                 firstRecord = false;
440             }
441             // Evaluate key columns
442             keyColumnNames = new ArrayList<>();
443             for (ExpressionProperty columnExpression : keyColumnExpressions) {
444                 String column = columnExpression.eval(document);
445                 keyColumnNames.add(column);
446             }
447             List<String> normalizeKeyColumnNames = Lists.newArrayList();
448             for (String column : keyColumnNames) {
449                 String normalizedColumn = jdbcOperations.normalize(column);
450                 if (StringUtils.isBlank(normalizedColumn)) {
451                     continue;
452                 } else if (!tableMetaData.containsKey(normalizedColumn)) {
453                     SnapDataException exception = new SnapDataException(
454                         String.format(ERR_INVALID_COLUMN, normalizedColumn))
455                         .withReason(ERR_INVALID_COLUMN_REASON)
456                         .withResolution(ERR_INVALID_COLUMN_RESOLUTION);
457                     errorViews.write(exception);
458                     return;
459                 }
460                 normalizeKeyColumnNames.add(normalizedColumn);
461             }
462             keyColumnNames = normalizeKeyColumnNames;
463 
464             // Evaluate s3 file paths
465             S3FilePaths = new ArrayList<>();
466             for (ExpressionProperty pathExpression : s3FilePathExpressions) {
467                 String path = pathExpression.eval(document);
468                 if (StringUtils.isBlank(path)) {
469                     continue;
470                 } else if (StringUtils.startsWith(path, S3_FILE_PREFIX_WITH_EXTRA_SLASH)) {
471                     path = S3_FILE_PREFIX.concat(
472                         path.substring(S3_FILE_PREFIX_WITH_EXTRA_SLASH.length()));
473                 } else if (!StringUtils.startsWith(path, S3_FILE_PREFIX)) {
474                     // Add S3 file path prefix if absent.
475                     path = S3_FILE_PREFIX.concat(path);
476                 }
477                 S3FilePaths.add(path);
478             }
479 
480             bulkLoadWrapper = createBulkLoadFromS3WrapperInstance();
481             bulkLoadWrapper.close();
482 
483             // set temp table name
484             String[] parts = quotationHandler.splitFullyQualifiedName(tableName, 0);
485             String unescapedTableName = quotationHandler.unescape(parts[parts.length - 1]);
486             String tempTable = quotationHandler.escape(String.format(TEMP_TABLE_FORMAT,
487                 unescapedTableName, UUID.randomUUID().toString().replaceAll(HYPHEN, UNDERSCORE)));
488 
489             preLoadAction();
490             loadAction(tableName, tempTable);
491             postLoadAction();
492             writeOutputView();

540     protected void loadAction(String targetTable, String tempTable) throws SQLException {
541         if (redshiftConnection == null) {
542             // Connection might have been created already if truncate was done
543             redshiftConnection = jdbcOperations.acquireConnection(account);
544         }
545         // The steps followed are as documented at
546         // http://docs.aws.amazon.com/redshift/latest/dg/
547         //  t_updating-inserting-using-staging-tables-.html#merge-method-replace-existing-rows
548         try (Statement stmt = redshiftConnection.createStatement()) {
549             try {
550                 // Create temp table
551                 createTempTable(stmt, tempTable, targetTable);
552                 // Run the copy from S3 to Redshift
553                 copyToTempTable(tempTable);
554                 // Delete the records that will be updated from target table
555                 String deleteStmt = createDeleteStmt(targetTable, tempTable);
556                 log.debug(LOG_RUNNING_DELETE, deleteStmt);
557                 stmt.execute(RedshiftUtils.addApnTaggingPrefix(deleteStmt));
558                 deleteCount = stmt.getUpdateCount();
559 
560                 // Insert from temp table to target table
561                 String insertStmt = String.format(INSERT_ALL_COMMAND, targetTable, tempTable);
562                 log.debug(LOG_RUNNING_INSERT, insertStmt);
563                 stmt.execute(RedshiftUtils.addApnTaggingPrefix(insertStmt));
564                 insertCount = stmt.getUpdateCount();
565             } finally {
566                 // No reason to attempt to delete the table, temporary tables
567                 //  are deleted when the session goes away
568                 //  see https://docs.aws.amazon.com/redshift/latest/dg/r_CREATE_TABLE_NEW.html
569                 log.debug("loadAction completed target = {} temp = {}.", targetTable, tempTable);
570             }
571         }
572     }

590         if (redshiftConnection == null) {
591             // Connection might have been created already if truncate was done
592             redshiftConnection = jdbcOperations.acquireConnection(account);
593         }
594 
595         String acceptInvalidCharStr = ACCEPT_INVALID_CHARS;
596         if (!acceptInvalidChars) {

598         }

602             truncateColumnsStr = StringUtils.EMPTY;
603         }

606         if (!StringUtils.isEmpty(additionalOptions)) {
607             additionalOptionsStr = additionalOptions;
608         }

614         String copyCommandObfuscated = bulkLoadWrapper.createCopyCommand(targetTable,
615             acceptInvalidCharStr, maxErrors, explictIds,
616             truncateColumnsStr, delimiter, fileType, ignoreHeader, additionalOptionsStr, true);
617         if (!additionalOptionsStr.toUpperCase().contains(DATE_FORMAT)) {

619             copyCommandObfuscated = copyCommandObfuscated.concat(COPY_DATE_FORMAT);

621         if (!additionalOptionsStr.toUpperCase().contains(TIME_FORMAT)) {
622             copyCommand = copyCommand.concat(COPY_TIME_FORMAT);

683     /**
684      * Write all error entries from the the stl_load_errors table into the error view.
685      */
686     private void writeErrorView() throws SQLException {
687         String endTime = DateTime.now(DateTimeZone.UTC).toString();
688         if (errorViews.isEmpty()) {
689             return;
690         }
691 
692         for (String path : S3FilePaths) {
693             Set<Integer> lineSet = Sets.newHashSet();
694             try (PreparedStatement stmt = redshiftConnection.prepareStatement(
695                     getLoadErrorSql(serverless, true))) {
696                 stmt.setString(1, path);
697                 stmt.setTimestamp(2, parseTimestamp(startTime));
698                 stmt.setTimestamp(3, parseTimestamp(endTime));
699                 try (ResultSet rs = stmt.executeQuery()) {
700                     while (rs.next()) {
701                         int lineNumber = rs.getInt(1);
702                         if (lineSet.contains(lineNumber)) {
703                             // Only one error view entry per input record even if multiple
704                             // errors encountered on that record
705                             continue;
706                         }
707                         lineSet.add(lineNumber);
708                         String colName = StringUtils.trim(rs.getString(2));
709                         int errorCode = rs.getInt(3);
710                         String errorMessage = StringUtils.trim(rs.getString(4));
711                         String fileName = StringUtils.trim(rs.getString(5));
712 
713                         Map<String, Object> errorInfo = Maps.newLinkedHashMap();
714                         errorInfo.put(KEY_FILE_NAME, fileName);
715                         errorInfo.put(KEY_LINE_NUMBER, lineNumber);
716                         errorInfo.put(KEY_COLUMN_NAME, colName);
717                         errorInfo.put(KEY_ERROR_CODE, errorCode);
718                         errorInfo.put(KEY_ERROR_MESSAGE, errorMessage);
719 
720                         String reason = null;
721                         if (serverless) {
722                             reason = String.format(SERVERLESS_ERROR_LOAD_DATA_ROW, lineNumber,
723                                 colName, errorCode, errorMessage);
724                         } else {
725                             String column_value = StringUtils.trim(rs.getString(6));
726                             String row_value = StringUtils.trim(rs.getString(7));
727                             errorInfo.put(KEY_COLUMN_VALUE, column_value);
728                             errorInfo.put(KEY_ROW_VALUE, row_value);
729                             reason = String.format(ERROR_LOAD_DATA_ROW, lineNumber, colName,
730                                 column_value, row_value, errorCode, errorMessage);
731                         }
732 
733                         Document doc = documentUtility.newDocument(errorInfo);
734                         // Unlike regular error views, the complete input document is not
735                         // available, just whatever is provided by Redshift
736                         SnapDataException exception = new SnapDataException(doc, errorMessage)
737                             .withReason(reason)
738                             .withResolution(ERR_DATA_LOAD_ROW_RESOLUTION);
739                         errorViews.write(exception);
740                     }
741                 }
742             }
743         }
744     }

763     private void handleException(Exception se) throws ExecutionException, SQLException {
764         String endTime = DateTime.now(DateTimeZone.UTC).toString();
765         StringBuilder error = new StringBuilder();
766 
767         for (String path : S3FilePaths) {
768             try (PreparedStatement stmt = redshiftConnection.prepareStatement(
769                     getLoadErrorSql(serverless, true))) {
770                 stmt.setString(1, path);
771                 stmt.setTimestamp(2, parseTimestamp(startTime));
772                 stmt.setTimestamp(3, parseTimestamp(endTime));
773                 try (ResultSet rs = stmt.executeQuery()) {
774                     int count = 0;
775                     while (rs.next()) {
776                         error.append(String.format(ERROR_COPY_S3_FILE,
777                             StringUtils.trim(rs.getString(5)), rs.getInt(1),
778                             StringUtils.trim(rs.getString(2)), rs.getInt(3),
779                             StringUtils.trim(rs.getString(4))));
780                         count++;
781                         if (count >= ERROR_MESSAGE_ENTRY_COUNT) {
782                             // Just put first five errors in error message, more
783                             // can go to error view if created
784                             break;
785                         }
786                     }
787                 }
788             }
789         }
790         String errorStr = error.toString();
791         if (errorStr.isEmpty()) {
792             errorStr = se.getMessage();
793         }
794         String resolutionStr;
795         if (StringUtils.containsIgnoreCase(errorStr, MANDATORY_URL_IS_NOT_PRESENT)) {
796             resolutionStr = ERROR_CHECK_S3_FILE;
797         } else {
798             resolutionStr = ERROR_FIX_DATA_ISSUES;
799         }
800         throw new ExecutionException(se, ERROR_COPYING_S3_REDSHIFT)
801             .withReason(errorStr)
802             .withResolution(resolutionStr);
803     }
