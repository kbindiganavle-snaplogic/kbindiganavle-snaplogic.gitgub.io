118             }

122         /**
123          * Constructs a new {@code ParentRunner} that will run {@code @TestClass}
124          *
125          * @param testClass
126          * @param classFixture
127          */
128         protected TestMethodRunner(final Class<?> testClass, final TestFixture classFixture,
129                 FrameworkMethod method) throws InitializationError {
130             super(testClass);
131             this.method = method;
132 
133             TestFixture methodFixture = method.getAnnotation(TestFixture.class);
134             TestFixture fixture = wrapTestFixtures(methodFixture, classFixture);
135             PropertyRow[] rows = fixture.propertyMatrix();
136             List<PropertySetting> settings = new ArrayList<>();
137             List<Throwable> errors = new ArrayList<>();
138 
139             validateDataFile(method, fixture.input(), errors);
140             validateMethodArgument(method, errors);
141             for (PropertyRow row : rows) {
142                 for (int valueIndex = 0; valueIndex < row.values().length; valueIndex++) {
143                     try {
144                         String name = row.name();
145 
146                         if (valueIndex > 0) {
147                             name += valueIndex;
148                         }
149 
150                         settings.add(new PropertySetting(name, JsonPaths.compile(row.path()),
151                                 row.values()[valueIndex]));
152                     } catch (InvalidPathException e) {
153                         errors.add(e);
154                     }
155                 }
156             }
157 
158             configs.add(new SnapConfiguration(method, fixture, Collections.emptyList()));
159             for (int k = 1; k <= settings.size(); k++) {
160                 Combinations combinations = new Combinations(settings.size(), k);
161 
162                 for (int[] comb : combinations) {
163                     SnapConfiguration config = new SnapConfiguration(method, fixture,
164                             Arrays.stream(comb)
165                                     .mapToObj(settings::get)
166                                     .distinct()
167                                     .collect(Collectors.toList()));
168                     if (configs.contains(config)) {
169                         continue;
170                     }
171                     configs.add(config);
172                 }
173             }
174 
175             if (!errors.isEmpty()) {
176                 throw new InitializationError(errors);
177             }
178         }

191         @Override
192         protected void runChild(final SnapConfiguration child, final RunNotifier notifier) {
193             Description description = describeChild(child);
194             if (child.method.getAnnotation(Ignore.class) != null) {
195                 notifier.fireTestIgnored(description);
196             } else {
197                 try {
198                     Object test = getTestClass().getOnlyConstructor().newInstance();
199                     Statement statement = withSnapExecutor(child, test);
200 
201                     statement = possiblyExpectingExceptions(method, test, statement);
202                     statement = withPotentialTimeout(method, test, statement);
203                     statement = withBefores(method, test, statement);
204                     statement = withAfters(method, test, statement);
205                     statement = withRules(child, test, statement);
206                     runLeaf(statement, description, notifier);
207                 } catch (Exception e) {
208                     runLeaf(new Fail(e), description, notifier);
209                 }
210             }
211         }

374     public SnapTestRunner(final Class<?> klass) throws Throwable {
375         super(klass, Collections.emptyList());
376 
377         TestClass testClass = getTestClass();
378         TestFixture classFixture = testClass.getAnnotation(TestFixture.class);
379         List<FrameworkMethod> testMethods = testClass.getAnnotatedMethods(TestFixture.class);
380         runners = new ArrayList<>();
381         try {
382             runners.add(new BlockJUnit4ClassRunner(klass));
383         } catch (InitializationError e) {
384             if (e.getCauses().size() == 1 &&
385                     "No runnable methods".equals(e.getCauses().get(0).getMessage())) {
386                 // ignore
387             } else {
388                 throw e;
389             }
390         }
391         for (FrameworkMethod method : testMethods) {
392             runners.add(new TestMethodRunner(klass, classFixture, method));
393         }
394     }

408     private void validateDataFile(FrameworkMethod method,
409             final String file, final List<Throwable> errors) {
410         ClassLoader classLoader = Thread.currentThread().getContextClassLoader();
411         if (!EMPTY.equals(file)) {
412             InputStream inputStream = classLoader.getResourceAsStream(file);
413             if (inputStream == null) {
414                 errors.add(new Throwable(String.format(
415                         INPUT_FILE_IS_NOT_ACCESSIBLE, file, method.getName())));
416             } else {
417                 IOUtils.closeQuietly(inputStream);
418             }
419         }

422     private void validateMethodArgument(final FrameworkMethod method,
423             final List<Throwable> errors) {
424         Class<?>[] parameterTypes = method.getMethod().getParameterTypes();
425         if (parameterTypes.length == 0) {
426             // Suggest tests don't take a parameter
427         } else if (parameterTypes.length != 1) {
428             errors.add(new Throwable(String.format(METHOD_SHOULD_HAVE_EXACTLY_ONE_ARGUMENT,
429                     method.getName())));
430         } else if (!TEST_RESULT_CLASS.equals(parameterTypes[0].getName()) &&
431                 !TEST_SETUP_CLASS.equals(parameterTypes[0].getName())) {
432             errors.add(new Throwable(String.format(METHOD_SHOULD_HAVE_VALID_ARGUMENT,
433                     method.getName())));
434         }
435     }

437     /**
438      * Create a TestFixture proxy that delegates methods to the given set of fixtures.  The first
439      * fixture to return a non-default value will be used as the result for any given method in the
440      * fixture.
441      *
442      * @param fixtures The array of fixtures to delegate to.
443      * @return A new TestFixture proxy
444      */
445     private static TestFixture wrapTestFixtures(final TestFixture ...fixtures) {
446         return (TestFixture) Proxy.newProxyInstance(Thread.currentThread().getContextClassLoader(),
447                 new Class[]{TestFixture.class},
448                 (proxy, method, args) -> {
449                     Object annotationDefaultValue = method.getDefaultValue();
450 
451                     for (TestFixture fixture : fixtures) {
452                         if (fixture == null) {
453                             continue;
454                         }
455 
456                         Object fixtureValue = method.invoke(fixture);
457                         boolean isDefault = Objects.deepEquals(fixtureValue,
458                                 annotationDefaultValue);
459 
460                         if (!isDefault) {
461                             return fixtureValue;
462                         }
463                     }
464 
465                     return annotationDefaultValue;
466                 });
467     }
