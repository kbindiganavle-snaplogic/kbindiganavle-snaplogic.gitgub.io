258     private void process(Document document) {
259         // If the upstream pipeline provides the first input document long after the
260         // pipeline initialization, the Box connection may time out. Therefore, it is
261         // better to connect here than in configure() since Box service drops the
262         // connection if it idles for a long time.
263         if (connection == null) {
264             connection = account.connect();
265         }
266 
267         final File[] tmpFile = {null};
268         final Document[] header = new Document[1];
269         File tmpDir = BoxUtils.getTmpDir(expressionUtils, propertyValues, errorViews);
270         final String filename = getFilename(document);
271         Fallback<Object> fallback = errorConsumer(String.format(ERR_DOWNLOAD_FILE, filename),
272                 ERR_GENERIC_RESOLUTION, document, errorViews);
273         Failsafe.with(fallback, snapRetryPolicy, oauthRetryPolicy).run(() -> {
274             BoxFile.Info readFileInfo = getBoxFileInfo(document);

297             }
298         });

315     private void multiDownload(Iterable<Document> documents) {
316         File tmpDir = BoxUtils.getTmpDir(expressionUtils, propertyValues, errorViews);
317         if (tmpDir == null) {
318             return;
319         }
320         executor = Executors.newFixedThreadPool(maxConcurrency);
321         threads = new HashMap<>(maxConcurrency);
322         for (Document document : documents) {
323             // If the upstream pipeline provides the first input document long after the
324             // pipeline initialization, the Box connection may time out. Therefore, it is
325             // better to connect here than in configure() since Box service drops the
326             // connection if it idles for a long time.
327             if (connection == null) {
328                 connection = account.connect();
329             }
330 
331             Fallback<Object> fallback = errorConsumer(
332                     String.format(ERR_START_DOWNLOAD, getFilename(document)),
333                     ERR_GENERIC_RESOLUTION, document, errorViews);
334             Failsafe.with(fallback, snapRetryPolicy, oauthRetryPolicy).run(() -> {
335                 // Check the status of all download threads.
336                 // If completed, copy temp files to the output view.
337                 checkDownloadTasks();
338 
339                 BoxFile.Info boxFileInfo = getBoxFileInfo(document);
340                 File tempFile = null;
341                 try {
342                     if (boxFileInfo != null) {
343                         while (threads.size() >= maxConcurrency) {
344                             // if no more thread is available, wait until a thread is completed
345                             BoxUtils.delay(POLLING_INTERVAL_MILLISECOND);
346                             checkDownloadTasks();
347                         }
348                         tempFile = createTempFile(tmpDir, boxFileInfo, document);
349                         if (tempFile != null) {
350                             // start a new thread to download a Box file to a temp file
351                             BoxFileDownload boxFileDownload = new BoxFileDownload(boxFileInfo,
352                                     tempFile, document);
353                             Future<Boolean> thread = executor.submit(boxFileDownload);
354                             threads.put(thread, boxFileDownload);
355                         }
356                     }
357                 } finally {
358                     BoxUtils.delete(tempFile, LOG);
359                 }
360             });
361         }
362     }

364     // Evaluates File ID and File path expressions and returns a BoxFile.Info object.
365     private BoxFile.Info getBoxFileInfo(Document document) {
366         String filePath = null;
367         String fileId = null;
368         if (fileIdExpression != null) {
369             fileId = fileIdExpression.eval(document);
370         }
371         if (filePathExpression != null) {
372             filePath = filePathExpression.eval(document);
373         }
374 
375         if (StringUtils.isNotBlank(fileId)) {
376             return new BoxFile(connection, fileId).getInfo();
377         } else if (StringUtils.isNotBlank(filePath)) {
378             BoxFolder parentFolder = boxCommon.traverseFolders(connection, filePath, false);
379             String fileName = Iterables.getLast(boxCommon.splitPath(filePath));
380             BoxFile.Info boxFileInfo = boxCommon.findFile(parentFolder, fileName);
381             if (boxFileInfo != null) {
382                 return boxFileInfo;
383             }
384             SnapDataException ex = new SnapDataException(ERR_FILE_NOT_FOUND)
385                     .formatWith(fileName)
386                     .withReason(REASON_FILE_NOT_FOUND)
387                     .withResolution(RESOLTION_ADDRESS_ISSUE);
388             errorViews.write(ex, document);
389         } else {
390             SnapDataException ex = new SnapDataException(FILE_PATH_NOT_AVAILABLE)
391                     .withReason(FILE_PATH_REQUIRED)
392                     .withResolution(FILE_PATH_DATA_RESOLUTION);
393             errorViews.write(ex, document);
394         }
395         return null;
396     }

398     private void checkDownloadTasks() {
399         // Check all download threads.
400         List<Future<Boolean>> threadsToBeRemoved = new ArrayList<>(maxConcurrency);
401         for (Map.Entry<Future<Boolean>, BoxFileDownload> entry : threads.entrySet()) {
402             Future<Boolean> thread = entry.getKey();
403             // if completed, write temp file to the output view
404             if (thread.isDone()) {
405                 BoxFileDownload boxFileDownload = entry.getValue();
406                 File tempFile = boxFileDownload.getTempFile();
407                 final BoxFile.Info boxFileInfo = boxFileDownload.getBoxFileInfo();
408                 String filename = boxFileInfo.getName();
409                 final Document document = boxFileDownload.getDocument();
410                 if (isDownloadSuccess(thread, filename) && tempFile != null) {
411                     // If the thread has completed successfully, copy temp file to the output view.
412                     try {
413                         Document header = documentUtility.newDocumentFor(document,
414                                 (new BoxCommon()).createHeader(boxFileInfo));
415                         outputViews.write(new BoxBinaryOutput(tempFile, header, filename,
416                                 document, true));
417                     } finally {
418                         BoxUtils.delete(tempFile, LOG);
419                         threadsToBeRemoved.add(thread);
420                     }
421                 } else {
422                     // Box file download failed, the error has already been written to error view
423                     // delete temp file and remove the thread from the active thread list
424                     BoxUtils.delete(tempFile, LOG);
425                     threadsToBeRemoved.add(thread);
426                 }
427             }
428         }
429         // remove completed threads
430         for (Future<Boolean> thread : threadsToBeRemoved) {
431             threads.remove(thread);
432         }
433     }

435     private Boolean isDownloadSuccess(Future<Boolean> thread, String filename) {
436         try {
437             return thread.get(THREAD_GET_TIMEOUT_SECOND, TimeUnit.SECONDS);
438         } catch (TimeoutException e) {
439             LOG.warn("Timeout while download Box file: {}, {}", filename, e.getMessage());
440         } catch (InterruptedException e) {
441             LOG.warn("Interrupted while downloading Box file: {}, {}", filename, e.getMessage());
442         } catch (java.util.concurrent.ExecutionException e) {
443             LOG.warn("Failed to download Box file, {}", e.getMessage());
444         }
445         return false;
446     }

570     private Fallback<Object> errorFunction(String fileName, Document document) {
571         CheckedFunction<ExecutionAttemptedEvent<?>, ?> function = event -> {
572             Throwable t = event.getLastFailure();
573             SnapDataException sde = new SnapDataException(t, ERR_BOX_FILE_DOWNLOAD)
574                     .formatWith(fileName)
575                     .withResolution(ERR_GENERIC_RESOLUTION);
576             if (t instanceof BoxAPIException) {
577                 sde.withReason(toReason((BoxAPIException) t));
578             } else {
579                 sde.withReason(Throwables.getRootCause(t).getMessage());
580             }

583         };
584         return Fallback.of(function).handle(BoxAPIException.class);

620         @Override
621         public void write(WritableByteChannel writableByteChannel) throws IOException {
622             try {
623                 InputStream inputStream = new FileInputStream(inputFile);
624                 IOUtils.copyLarge(inputStream, Channels.newOutputStream(writableByteChannel));
625             } catch (SuggestViewAbortException e) {
626                 if (multidownload) {
627                     // If the end of preview reached during copyLarge(), the
628                     // outer try-catch block won't be able to catch
629                     // SuggestViewAbortException. It has to be caught here and
630                     // return quietly without throwing it back. Otherwise, the
631                     // platform will throw SnapStreamException, which will
632                     // cause the outer try-catch writes it to the error view.
633                 } else {
634                     throw e;
635                 }
636             } finally {
637                 BoxUtils.delete(inputFile, LOG);
638             }
639         }
