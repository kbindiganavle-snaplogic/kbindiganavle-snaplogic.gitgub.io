214     public void configureRetry() {
215         int numOfRetries = evalIntProperty(numOfRetriesExpr, null, MIN_RETRIES, MIN_RETRIES,
216                 errorViews);
217         int retryInterval = evalIntProperty(retryIntervalExpr, null, MIN_RETRY_INTERVAL,
218                 MIN_RETRY_INTERVAL, errorViews);
219         if (numOfRetries > 0) {
220             enableStaging = true;
221         } else if (enableStaging) {
222             // For maintaining backward compatibility, if enableStaging is checked we do
223             // one retry by default though number of retries is configured as 0.
224             numOfRetries = 1;
225         }
226         snapRetryPolicy = getSnapRetryPolicy(numOfRetries, retryInterval)
227                 .onFailedAttempt(e -> LOGGER.warn("box_write_retry - failed to " +
228                         "write Box file: {}, {}", filePath, e.getLastFailure()));
229         oauthRetryPolicy = getBaseOAuthRetryPolicy().onRetry(e -> {
230             LOGGER.warn("box_account_retry: attempt {}", e.getAttemptCount());
231             if (((BoxAccount) account).tryReloadAccount((BoxAPIException) e.getLastFailure())) {
232                 connection = account.connect();
233             }
234         });
235         if (account instanceof BoxAccount) {
236             writeTimeout = ((BoxAccount) account).getWriteTimeout();
237         } else if (account instanceof BoxServiceAccount) {
238             writeTimeout = ((BoxServiceAccount) account).getWriteTimeout();
239         }
240         if (writeTimeout > 0) {
241             timeoutPolicy = Timeout.of(Duration.ofSeconds(writeTimeout)).withInterrupt(true);
242         }
243     }

245     @Override
246     protected void process(Document header, final ReadableByteChannel readChannel) {
247         configureRetry();
248         try {
249             connection = account.connect();
250             filePath = boxUtils.castAsString(filePathExpression.eval(header));
251             if (StringUtils.isBlank(filePath)) {
252                 throw new SnapDataException(ERR_EMPTY_FILEPATH)
253                         .withReason(ERR_BLANK_FILEPATH)
254                         .withResolution(RESOLUTION_EMPTY_FILEPATH);
255             }
256             if (enableStaging) {
257                 try (InputStream inputStream = Channels.newInputStream(readChannel); 
258                     InputStream is = getNonEmptyInputStream(inputStream)) {
259                     
260                     if (is != null) {
261                         writeToTempFile(is);
262                         final Document finalHeader = header;
263                         try {
264                             if (timeoutPolicy != null) {
265                                 Failsafe.with(snapRetryPolicy, oauthRetryPolicy, timeoutPolicy)
266                                         .run(() -> writeForRetry(finalHeader));
267                             } else {
268                                 Failsafe.with(snapRetryPolicy, oauthRetryPolicy)
269                                         .run(() -> writeForRetry(finalHeader));
270                             }
271                         } catch (Throwable e) {
272                             throw e instanceof FailsafeException ? e.getCause() : e;
273                         }
274                     }
275                 } finally {
276                     cleanup();
277                 }
278             } else if (timeoutPolicy != null) {
279                 Document finalHeader = header;
280                 Failsafe.with(timeoutPolicy).run(() -> writeFile(finalHeader, readChannel));
281             } else {
282                 writeFile(header, readChannel);
283             }
284         } catch (SuggestViewAbortException e) {
285             throw e;
286         } catch (SnapDataException e) {
287             // This is based on ExecutionUtil.writeToErrorView, which is currently only
288             // used for snaps that extend SimpleSnap, which this one does not.
289             if (e.getErrorDocument() != null) {
290                 header = null;
291             }
292             errorViews.write(e, header);
293         } catch (TimeoutExceededException e) {
294             SnapDataException sde = new SnapDataException(e, ERR_WRITE_TIME_OUT)
295                     .withReason(String.format(REASON_WRITE_TIME_OUT, writeTimeout))
296                     .withResolution(RESOLUTION_WRITE_TIME_OUT);
297             errorViews.write(sde, header);
298         } catch (Throwable e) {
299             errorViews.write((SnapDataException) new SnapDataException(e, ERR_WRITE_FILE)
300                     .formatWith(filePath)
301                     .withResolution(RESOLUTION_ADDRESS_ISSUE), header);
302         } finally {
303             appendBuffer = null;
304         }
305     }

307     private void writeToTempFile(InputStream is) {
308         try {
309             // create a temp file
310             File tmpDir = BoxUtils.getTmpDir(expressionUtils, propertyValues, null);
311             String fileName = String.format(FORMAT_TEMP_FILE,
312                     RandomStringUtils.randomAlphabetic(SUFFIX_LENGTH));
313             tempFile = File.createTempFile(fileName, null, tmpDir);
314             LOGGER.debug("Temp file created: {}", tempFile);
315             // write input data into the temp file
316             try (FileOutputStream fileOut = FileUtils.openOutputStream(tempFile)) {
317                 IOUtils.copyLarge(is, fileOut);
318             }
319         } catch (IOException e) {
320             throw new SnapDataException(ERR_WRITE_TEMP_FILE)
321                     .withResolution(RESOLUTION_ADDRESS_ISSUE);
322         }
323     }

326      * @param inputStream the source InputStream to check for empty content.
327      * @return a usable InputStream, or {@code null} if the source input is empty. If the Snap is
328      * configured to allow empty file writing, the source InputStream will be returned as-is.
329      * @throws IOException if an error occurs reading the input.
330      */
331     private InputStream getNonEmptyInputStream(InputStream inputStream) throws IOException {
332         if (writeEmptyFiles) {
333             return inputStream;
334         }
335         PushbackInputStream pushbackInputStream = new PushbackInputStream(inputStream);
336         int firstByte = pushbackInputStream.read();
337         if (firstByte == -1) {
338             // File is empty.
339             LOGGER.warn("Empty file provided as input. Output may not be written.");
340             return null;
341         }
342         pushbackInputStream.unread(firstByte);
343         return pushbackInputStream;
344     }

346     @VisibleForTesting
347     void writeForRetry(Document header) throws IOException {
348         String fileName = Iterables.getLast(boxCommon.splitPath(filePath));
349         BoxFile.Info uploadedFileInfo;
350         try (InputStream inputStream = new FileInputStream(tempFile)) {
351             BoxFolder parentFolder = boxCommon.traverseFolders(connection, filePath, true);
352             BoxFile.Info writeFileInfo = boxCommon.findFile(parentFolder, fileName);
353             if (writeFileInfo == null) {
354                 // target file does not exists, create one
355                 uploadedFileInfo = parentFolder.uploadFile(inputStream, fileName);
356             } else {
357                 // target file exists
358                 BoxFile existingFile = new BoxFile(connection, writeFileInfo.getID());
359                 if (append.equals(APPEND)) {
360                     if (appendBuffer == null) {
361                         // download target file
362                         ByteArrayOutputStream outputStream = new ByteArrayOutputStream();
363                         existingFile.download(outputStream);
364                         appendBuffer = outputStream.toByteArray();
365                     }
366                     try (SequenceInputStream sequenceInputStream = new SequenceInputStream(
367                             new ByteArrayInputStream(appendBuffer), inputStream)) {
368                         existingFile.uploadVersion(sequenceInputStream);
369                         uploadedFileInfo = existingFile.getInfo();
370                     }
371                 } else {
372                     existingFile.uploadVersion(inputStream);
373                     uploadedFileInfo = existingFile.getInfo();
374                 }
375             }
376             writeToOutput(uploadedFileInfo, header);
377         }
378     }

380     /**
381      * Uploads a new copy of the file at the specified file path
382      */
383     private void writeFile(Document header, ReadableByteChannel readChannel) {
384         InputStream fileToUpload = null;
385         ByteArrayOutputStream existingContents = null;
386         String fileName = Iterables.getLast(boxCommon.splitPath(filePath));
387         InputStream in = null;
388         try {
389             in = getNonEmptyInputStream(Channels.newInputStream(readChannel));
390             if (in != null) {
391                 BoxFolder parentFolder = boxCommon.traverseFolders(connection, filePath, true);
392                 BoxFile.Info writeFileInfo = boxCommon.findFile(parentFolder, fileName);
393                 BoxFile.Info uploadedFileNewInfo;
394                 if (writeFileInfo != null) {
395                     // If the file exists, upload new version
396                     fileToUpload = in;
397                     BoxFile existingFile = new BoxFile(connection, writeFileInfo.getID());
398                     if (append.equals(APPEND)) {
399                         // If the append option is selected, download the current version of the
400                         // file from Box and append the inputStream to it
401                         // If retry is enabled, do not retry download
402                         if (appendBuffer == null) {
403                             existingContents = new ByteArrayOutputStream();
404                             existingFile.download(existingContents);
405                             appendBuffer = existingContents.toByteArray();
406                         }
407                         fileToUpload = new SequenceInputStream(
408                                 new ByteArrayInputStream(appendBuffer), in);
409                         existingFile.uploadVersion(fileToUpload);
410                         uploadedFileNewInfo = existingFile.getInfo();
411                     } else {
412                         // Overwriting was not handled earlier
413                         existingFile.uploadVersion(in);
414                         uploadedFileNewInfo = existingFile.getInfo();
415                     }
416                 } else {
417                     // file does not currently exist, creating file
418                     fileToUpload = in;
419                     uploadedFileNewInfo = parentFolder.uploadFile(fileToUpload, fileName);
420                 }
421                 writeToOutput(uploadedFileNewInfo, header);
422             }
423         } catch (BoxAPIException | IOException e) {
424             throw new SnapDataException(e, ERR_UPLOAD_FILE)
425                     .formatWith(fileName)
426                     .withReason(toReason(e))
427                     .withResolution(ERR_GENERIC_RESOLUTION);
428         } finally {
429             IOUtils.closeQuietly(fileToUpload, in, existingContents);
430         }
431     }
