309     private void connectAndCreateChannel(Document header) {
310         int attemptCount = 0;
311         while (attemptCount < maxConnectionAttempts) {
312             try {
313                 attemptCount++;
314                 if (connection == null || !connection.isOpen()) {
315                     connection = account.connect();
316                 }
317                 if (channel == null || !channel.isOpen()) {
318                     channel = connection.createChannel();
319                     channel.confirmSelect();
320                 }
321                 LOGGER.debug(RETRY_SUCC_LOG_MSG, attemptCount);
322                 return;
323             } catch (Exception e) {
324                 LOGGER.error(RETRY_FAILED_LOG_MSG, attemptCount, e.getMessage());
325                 if (attemptCount >= maxConnectionAttempts) {
326                     acknowledgeHeader(header);
327                     Throwable cause = e.getCause() != null ? e.getCause() : e;
328                     throw new ExecutionException(cause, ERROR_CREATING_CONNECTION)
329                             .withReason(cause.getMessage())
330                             .withResolution(ERROR_CREATING_CONNECTION_RES);
331                 }
332                 try {
333                     Thread.sleep(retryInterval);
334                 } catch (InterruptedException ie) {
335                     //no op
336                 }
337             }
338         }
339     }

359     private boolean prepareDestination(Document header, byte[] message) throws IOException {
360         if (channel == null || !channel.isOpen()) {
361             connectAndCreateChannel(header);
362         }
363         if (connection != null && !connection.isOpen()) {
364             if (isConnErrToErrorViewEnabled) {
365                 SnapDataException snapDataException =
366                         new SnapDataException(ERROR_WHILE_PREPARING_DESTINATION)
367                                 .withReason(ERR_RECONNECT_CHANNEL)
368                                 .withResolution(RES_VERIFY_INPUT_DATA);
369                 errorViews.write(snapDataException, documentUtility.newDocumentFor(header, getMessageObject
370                         (header, message)));
371             } else {
372                 throw new ExecutionException(ERROR_WHILE_PREPARING_DESTINATION)
373                         .withReason(ERR_RECONNECT_CHANNEL)
374                         .withResolution(RES_VERIFY_INPUT_DATA);
375             }
376             return false;
377         }
378         queueName = checkEmpty(queueName, StringUtils.EMPTY);
379         exchangeName = checkEmpty(exchangeName,
380                 String.format(DEFAULT_PREFIX_EXCHANGE_TYPE, exchangeType));
381         routingKey = checkEmpty(routingKey, StringUtils.EMPTY);
382         if (headerMessageProperties != null) {
383             headers = MessageUtils.populateHeaderProperties
384                     (headerMessageProperties,
385                             header, xmatch);
386         }
387         channel.exchangeDeclare(exchangeName, exchangeType, exchangeDurable);
388         if (StringUtils.isBlank(queueName)) {
389             if (createDynamicQueue && !isDynamicQueueCreated) {
390                 channel.queueDeclare();
391                 channel.queueBind(queueName, exchangeName, routingKey, headers);
392                 isDynamicQueueCreated = true;
393             }
394         } else {
395             Map argumentMap =  MessageUtils.populateArgumentProperties(argumentProperties, header);
396             channel.queueDeclare(queueName, true, false, autoDelete, argumentMap);
397             channel.queueBind(queueName, exchangeName, routingKey, headers);
398         }
399         return true;
400     }

431     @Override
432     protected void process(Document header, ReadableByteChannel readChannel) {
433         queueName = getValue(queueNameExp, header);
434         exchangeName = getValue(exchangeNameExp, header);
435         routingKey = getValue(routingKeyExp, header);
436         byte[] message = null;
437         BasicProperties messageProps = null;
438         try (InputStream inputStream = Channels.newInputStream(readChannel)) {
439             message = IOUtils.toByteArray(inputStream);
440         } catch (IOException e) {
441             if (isConnErrToErrorViewEnabled) {
442                 SnapDataException snapDataException =
443                         new SnapDataException(e, ERROR_WHILE_READING_STREAM_INTO_MESSAGE_BODY)
444                                 .withReason(e.getMessage())
445                                 .withResolution(RES_VERIFY_INPUT_DATA);
446                 errorViews.write(snapDataException, documentUtility.newDocumentFor(header, getMessageObject
447                         (header, message)));
448             } else {
449                 throw new ExecutionException(e, ERROR_WHILE_READING_STREAM_INTO_MESSAGE_BODY)
450                         .withReason(e.getMessage())
451                         .withResolution(RES_VERIFY_INPUT_DATA);
452             }
453             return;
454         }
455         try {
456             if (prepareDestination(header, message)) {
457                 if (messageHeaderProperties != null) {
458                     messageProps = populateMessageProperties(messageHeaderProperties, header);
459                 }
460                 if (headers != null) {
461                     if (messageProps == null) {
462                         messageProps = new AMQP.BasicProperties();
463                     }
464                     messageProps = messageProps.builder().headers(headers).build();
465                 }
466                 channel.basicPublish(exchangeName, routingKey, messageProps, message);
467                 /**
468                  * Wait until all messages published since the last call have been either ack'd or
469                  * nack'd by the broker; or until timeout elapses.
470                  **/
471                 if (waitForConfirmsTimeout > 0) {
472                     channel.waitForConfirms(waitForConfirmsTimeout);
473                 } else {
474                     channel.waitForConfirms();
475                 }
476                 messageCounter.inc();
477                 Map<String, Object> data = new LinkedHashMap<>();
478                 data.put(STATUS, SUCCESS);
479                 data.put(ORIGINAL_OBJ, getMessageObject(header, message));
480                 outputViews.write(documentUtility.newDocumentFor(header, data));
481             }
482         } catch (Exception e) {
483             if (isConnErrToErrorViewEnabled) {
484                 SnapDataException snapDataException;
485                 snapDataException = (SnapDataException) handleQueueTypeException(e, queueName, true,
486                         ERROR_WHILE_SENDING_THE_MESSAGE, RES_CHECK_NETWORK_CONNECTION);
487                 errorViews.write(snapDataException, documentUtility.newDocumentFor(header, getMessageObject
488                         (header, message)));
489             } else {
490                 throw MessageUtils.handleQueueTypeException(e, queueName, false,
491                         ERROR_WHILE_SENDING_THE_MESSAGE, RES_CHECK_NETWORK_CONNECTION);
492             }
493         }
494     }
