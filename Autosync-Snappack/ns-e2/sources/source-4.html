433     @Override
434     protected void process(final Document document, final String inputViewName) {
435         try {
436             Object objDimensions = dimensionsExpression.eval(document);
437             if (objDimensions != null) {
438                 dimensions = objDimensions.toString();
439             }
440             Object objStartDate = startdateExpression.eval(document);
441             if (validatePropertyValue(START_DATE_LABEL, objStartDate)) {
442                 startDate = objStartDate.toString();
443             }
444             Object objEndDate = enddateExpression.eval(document);
445             if (validatePropertyValue(END_DATE_LABEL, objEndDate)) {
446                 endDate = objEndDate.toString();
447             }
448             Object objMetrics = metricsExpression.eval(document);
449             if (validatePropertyValue(METRICS_LABEL, objMetrics)) {
450                 metrics = objMetrics.toString();
451             }
452             if (invalidPropertyList != null) {
453                 String error = String.format(ERR_EMPTY_PROPERTY, StringUtils.join(
454                         invalidPropertyList, ","));
455                 SnapDataException ex = new SnapDataException(error)
456                         .withReason(REASON_INVALID_PROPERTY)
457                         .withResolution(RESOLUTION_INVALID_PROPERTY);
458                 errorViews.write(ex, document);
459                 return;
460             }
461             boolean dataFound = false;
462             do {
463                 Get query = getGAQuery();
464                 GaData data = getGaData(query);
465                 if (!processGaData(data, document, dataFound)) {
466                     break;
467                 }
468                 if (!dataFound) {
469                     dataFound = true;
470                 }
471                 start_index += max_results;
472             } while (autoIterate);
473         } catch (SuggestViewAbortException e) {
474             throw e;
475         } catch (Exception e) {
476             SnapException ex;
477             if (e instanceof GoogleJsonResponseException) {
478                 GoogleJsonResponseException gex = (GoogleJsonResponseException) e;
479                 String message = StringUtils.EMPTY;
480                 String reason = StringUtils.EMPTY;
481                 GoogleJsonError err = gex.getDetails();
482                 if (err != null && err.getErrors() != null) {
483                     for (ErrorInfo errorInfo : err.getErrors()) {
484                         message = StringUtils.join(message, errorInfo.getMessage(),
485                                 COMMA_SEPARATOR);
486                         reason = StringUtils.join(reason, errorInfo.getReason(), COMMA_SEPARATOR);
487                     }
488                     ex = new SnapDataException(e, message)
489                             .withReason(reason)
490                             .withResolution(QUERY_EXECUTION_FAILED_RESOLUTION);
491                 } else {
492                     ex = new SnapDataException(e, ERR_EXECUTE_QUERY)
493                             .formatWith(gex.getStatusCode(), gex.getStatusMessage())
494                             .withResolution(QUERY_EXECUTION_FAILED_RESOLUTION);
495                 }
496 
497             } else if (e instanceof ExecutionException) {
498                 ex = new SnapDataException(e, e.getMessage())
499                         .withReason(((ExecutionException) e).getReason())
500                         .withResolution(((ExecutionException) e).getResolution());
501             } else {
502                 ex = new SnapDataException(e, EXECUTION_FAILED)
503                         .withReason(QUERY_EXECUTION_FAILED_REASON)
504                         .withResolution(QUERY_EXECUTION_FAILED_RESOLUTION);
505             }
506             errorViews.write((SnapDataException) ex, document);
507         }
508     }

510     /**
511      * Writes data generated as a result of the query to the output view
512      *
513      * @param  data                  the response data
514      * @param  document              the input document
515      * @param  foundPreviousData     tells if 'no data found' message to be written
516      * @return                       true if data is not empty; false otherwise
517      */
518     private boolean processGaData(GaData data, Document document, boolean foundPreviousData) {
519         String[] headerNames;
520         Map<String, Object> row;
521         Boolean containsSampledData = data.getContainsSampledData();
522         Long sampleSize = data.getSampleSize();
523         Long sampleSpace = data.getSampleSpace();
524 
525         if (!data.isEmpty() && data.getRows() != null) {
526             headerNames = new String[data.getColumnHeaders().size()];
527             int i = 0;
528             for (GaData.ColumnHeaders header : data.getColumnHeaders()) {
529                 headerNames[i++] = header.getName();
530             }
531             for (List<String> rowValues : data.getRows()) {
532                 i = 0;
533                 row = Maps.newLinkedHashMap();
534                 for (String rowValue : rowValues) {
535                     row.put(headerNames[i++], rowValue);
536                 }
537                 if (Boolean.TRUE.equals(containsSampledData)) {
538                     row.put(SAMPLE_SIZE, sampleSize);
539                     row.put(SAMPLE_SPACE, sampleSpace);
540                 }
541                 outputViews.write(documentUtility.newDocumentFor(document, row));
542             }
543             return true;
544         } else {
545             if (!foundPreviousData) {
546                 outputViews.write(documentUtility.newDocumentFor(document,
547                         NO_DATA_FOUND));
548             }
549             return false;
550         }
551     }
