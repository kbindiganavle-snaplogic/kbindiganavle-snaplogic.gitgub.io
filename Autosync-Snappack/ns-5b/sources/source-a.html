427     @Override
428     public DatabaseCursor executeQuery(final String query, final List<Object> bindValues,
429                                        final DatabaseAccount account,
430                                        int maxRetries, long retryInterval,
431                                        boolean elt) throws SQLException {
432         try {
433             DatabaseConfig databaseConfig = account.getDatabaseConfig();
434             SQLDialect dialect = databaseConfig.getSQLDialect();
435             return Failsafe.with(getDefaultRetryPolicy(maxRetries, retryInterval))
436                 .onFailedAttempt(e -> LOG.warn("jdbc_operations_impl_retry - failed to " +
437                     "executeQuery: {}", e.getMessage()))
438                 .onRetry((result, failure, context) -> {
439                     logRetryAttempt(context, LOG);
440                     handleOnRetry(dialect.getName(), failure);
441                 })
442                 .get(cntx -> {
443                     acquireConnection(account);
444                     handleAutoCommit();
445                     DSLContext context = DSL.using(conn, dialect);
446                     return executeQuery(query, bindValues, context,
447                         databaseConfig.getFetchSize(), dialect);
448                 });
449         } catch (ExecutionException | SnapDataException e) {
450             throw e;
451         } catch (FailsafeException | DataAccessException e) {
452             acquireConnection(account);
453             handleAutoCommit();
454             SQLDialect dialect = account.getDatabaseConfig().getSQLDialect();
455             DSLContext context = DSL.using(conn, dialect);
456             switch (getName()) {
457                 case POSTGRES:
458                     // This will check if the exception is unmaskable to run query
459                     if (!(e instanceof  DataAccessException) ||
460                             getEnumExceptionFromDataAccessException((DataAccessException) e) !=
461                                     exceptionsEnum.JOOQPostGresIncompatibility) {
462                         // This will try executing the SQL query without JOOQ.
463                         // Will execute the SQL with JDBC directly. JOOQ FAQ:
464                         // https://www.jooq.org/doc/latest/manual/getting-started/use-cases/
465                         // jooq-as-a-sql-executor/
466 
467                         if (bindValues != null) {
468                             return executeQueryWithoutjOOQ(account, context,
469                                     context.resultQuery(query, bindValues.toArray()),
470                                     maxRetries, retryInterval);
471                         } else {
472                             return executeQueryWithoutjOOQ(account, context,
473                                     context.resultQuery(query, null), maxRetries, retryInterval);
474                         }
475                     }
476                 case SNOWFLAKE:
477                     if (elt) {
478                         if (!e.getCause().toString().contains(SNOWFLAKE_SQL_EXECUTION_CANCELED)) {
479                             return executeQueryWithoutjOOQ(account, context,
480                                     context.resultQuery(query, bindValues.toArray()),
481                                     maxRetries, retryInterval);
482                         }
483                     }
484             }
485             Throwable cause = e.getCause();
486             if (DatabaseUtils.isDataException(cause)) {
487                 throw (SQLException) cause;
488             }
489             throw new SnapDataException(EXECUTE_QUERY_FAILED)
490                     .formatWith(getMaskedErrorMessages(query))
491                 .withReason(getMaskedErrorMessages(getSQLErrorMessage(e)))
492                     .withResolution(CHECK_QUERY)

535     private ResultSet getResultSet(final ResultQuery<Record> query, ResultSet rs)
536             throws SQLException {
537         if (StringUtils.equalsIgnoreCase(runtimeDBName, JdbcOperations.SNOWFLAKE)) {
538             ResultSetMetaData resultSetMetaData = rs.getMetaData();
539             Map<Integer, String> colIndexTimestamp = new HashMap<>();
540             int totalCols = resultSetMetaData.getColumnCount();
541             for (int i = 1; i <= totalCols; i++) {
542                 String colName = resultSetMetaData.getColumnName(i);
543                 String colClassName = resultSetMetaData.getColumnClassName(i);
544                 if (StringUtils.containsIgnoreCase(colClassName, TIMESTAMP)) {
545                     storeTimestampValues.put(colName, new ArrayList());
546                     colIndexTimestamp.put(i, colName);
547                 }
548             }
549             if (storeTimestampValues.size() > 0) {
550                 while (rs.next()) {
551                     for (int index: colIndexTimestamp.keySet()) {
552                         String value = rs.getString(index);
553                         if (value != null) {
554                             value = value.replace(UTC_Z, UTC_SNOWFLAKE);
555                         }
556                         String colName = colIndexTimestamp.get(index);
557                         List list = storeTimestampValues.get(colName);
558                         list.add(value);
559                         storeTimestampValues.put(colName, list);
560                     }
561                 }
562                 // Execute the select query to revert the cursor to first place as we have
563                 // modified the cursor pointer to fetch the timestamp values from server
564                 rs = conn.createStatement().executeQuery(query.toString());
565             }
566         }
567         return rs;
568     }

580      * Execute the query
581      *
582      * @param query      as the original query defined in the snap
583      * @param bindValues as the values to bind if any parameters existed in the snap
584      * @param context    as the context
585      * @param fetchSize  as the fetch size if any (this does not seem to work properly on some JDBC
586      *                   drivers, such as SQL server or Netezza)
587      * @param dialect    as the SQL dialect
588      * @return the cursor
589      */
590     private DatabaseCursor executeQuery(final String query, List<Object> bindValues,
591                                         final DSLContext context, int fetchSize,
592                                         SQLDialect dialect) {
593         String selectQuery;
594         if (maxRows > -1) {
595             // During suggest we append a LIMIT clause with maxRows records. We want to keep the
596             // sample size small.
597             String limitQuery = query;
598             switch (dialect) {
599                 case ORACLE:
600                 case ORACLE10G:
601                 case ORACLE11G:
602                 case ORACLE12C:
603                     // Oracle uses ROWNUM in a where clause, this would involve query parsing
604                     LOG.warn(NO_LIMIT_SUPPORT_FOR_ORACLE);
605                     break;
606                 case SQLSERVER:
607                     // lower then SQL Server 2012 support row fetch where clause only,
608                     LOG.warn(NO_LIMIT_SUPPORT_FOR_SQL_SERVER);
609                     break;
610                 default:
611                     // TODO - MK: Inject the right parser for the class as part of the the
612                     // instantiation. suggestQueryProvider will be null for every
613                     // vendor but generic and mysql besides
614                     if (suggestQueryProvider != null) {
615                         limitQuery = suggestQueryProvider.provideFor(query, maxRows);
616                     }
617             }
618             selectQuery = limitQuery;
619         } else {
620             selectQuery = query;
621         }
622 
623         fetchSize = adjustFetchSizeForProcedureCall(fetchSize, query);
624         String finalSelectQuery = selectQuery;
625         if (REDSHIFT.equals(getName())) {
626             finalSelectQuery = addApnTaggingPrefix(finalSelectQuery);
627         }
628         if (Objects.nonNull(queryQueue)) {
629             DatabaseUtils.submitQueryToThread(finalSelectQuery, queryQueue);
630         }
631         try {
632             Cursor<Record> cursor;
633             if (bindValues == null || bindValues.isEmpty()) {
634                 cursor = fetchLazy(context, context.resultQuery(finalSelectQuery), fetchSize);
635             } else {
636                 bindValues = convertValues(bindValues, dialect);
637                 cursor = fetchLazy(context, context.resultQuery(finalSelectQuery,
638                         bindValues.toArray()), fetchSize);
639             }
640             return getDatabaseCursor(conn, cursor, fetchSize);
641         } catch (Exception e) {
642             if (maxRows > -1) {
643                 LOG.warn(
644                     String.format(FALLING_BACK_TO_ORIGINAL_QUERY, selectQuery, e.getMessage()));
645                 LOG.warn(String.format(EXECUTING_OLD_QUERY, query));
646                 // reset it, to execute the original query, not all databases support limit query.
647                 maxRows = -1;
648                 return executeQuery(query, bindValues, context, fetchSize, dialect);
649             }
650             throw e;
651         }
652     }

670                                   final DatabaseAccount account,
671                                   boolean isQueryWithoutBindingParameter, final int maxRetries,
672                                   final long retryInterval) throws BatchUpdateException {
673 
674         final DatabaseConfig databaseConfig = account.getDatabaseConfig();
675         final SQLDialect dialect = databaseConfig.getSQLDialect();
676         final Settings settings = databaseConfig.getSettings();
677

706                 queryWithBindingParameters.add(query);
707             }
708             ((BatchBindStep) batch).bind(updatedValues.toArray());
709             if (batch.size() == databaseConfig.getBatchSize()) {
710                 return flushBatch(account, maxRetries, retryInterval);
711             }
712         }
713         return null;

807     public String convertLimitOffsetToTop(String selectQuery) {
808         StringBuilder stringBuilder = new StringBuilder();
809         // split the select query by any white space character (' ', \n, etc.)
810         String[] parts = StringUtils.split(selectQuery, null);
811         boolean isLimit = false;
812         boolean isOffset = false;
813         String limitValue = null;
814         for (String part : parts) {
815             if (isOffset) {
816                 isOffset = false; // offset is ignored
817                 continue;
818             } else if (isLimit) {
819                 limitValue = part;
820                 isLimit = false;
821                 continue;
822             }
823             switch (part) {
824                 case DatabaseUtils.SELECT:
825                     break;
826                 case DatabaseUtils.LIMIT:
827                     isLimit = true;
828                     break;
829                 case OFFSET:
830                     isOffset = true;
831                     break;
832                 default:
833                     stringBuilder.append(part).append(SPACE);
834             }
835         }
836         return StringUtils.isBlank(limitValue)
837             ? String.format(FORMAT_SELECT, stringBuilder.toString())
838             : String.format(FORMAT_SELECT_TOP, limitValue, stringBuilder.toString());

896                         ">" + offsetValue;

899             }
900         }

951         if (document != null) {
952             DatabaseConfig databaseConfig = account.getDatabaseConfig();
953             SQLDialect dialect = databaseConfig.getSQLDialect();
954             Settings settings = databaseConfig.getSettings();
955             Map<String, Object> data = document.get(Map.class);
956             if (MapUtils.isEmpty(data)) {
957                 return null;
958             }
959             acquireConnection(account);
960             Map<Field<?>, Object> fieldMap = convertValues(data, tableMetaData, dialect,
961                 preserveCaseSensitiveness);
962             List<Field<?>> fields = new ArrayList<>(fieldMap.keySet());
963             // If different fields are being inserted then flush the batch to create a new one
964             if (previousFields != null) {
965                 if (!fields.equals(new ArrayList<>(previousFields))) {
966                     inputDocuments.remove(document);
967                     BatchUpdateException batchUpdateException = null;
968                     int[] rowsModified;
969                     try {
970                         rowsModified = flushBatch(account, maxRetries, retryInterval);

979                         documentUtility, inputDocuments, batchUpdateException, runtimeDBName);
980                     batch = null;
981                     // Read the differing document and continue
982                     inputDocuments.add(document);
983                 }
984             }
985 
986             if (batch == null) {
987                 final Query query = createInsertQuery(account, tableName,
988                         duplicateKeyAction, fieldMap);
989                 if (Objects.nonNull(queryQueue) && Objects.isNull(batchQuery)) {
990                     // Retrieve batch query with bind parameters represented as '?' character
991                     // which is same as the query that hits Snowflake server with bind values
992                     batchQuery = query.getSQL(ParamType.INDEXED);
993                 }
994                 if (!isDataTypeSupportedOnjOOQ) {
995                     queryWithoutBindingParameters.add(query.toString());
996                 } else {
997                     DSLContext context = DSL.using(conn, dialect, settings);
998                     context.configuration()
999                             .set(new DefaultExecuteListenerProvider(new BatchExecuteListener()));
1000                     batch = context.batch(query);
1001                     queryWithBindingParameters.add(query);
1002                 }
1003             }
1004

1006                 ((BatchBindStep) batch).bind(fieldMap.values().toArray());
1007             }

1010                 && queryWithoutBindingParameters.size() == databaseConfig.getBatchSize()
1011                 || batch != null && batch.size() == databaseConfig.getBatchSize()) {
1012                 try {
1013                     return flushBatch(account, maxRetries, retryInterval);

1016                 } catch (Throwable e) {
1017                     DatabaseUtils.writeErrorView(errorViews, inputDocuments, e);
1018                     return null;
1019                 }
1020             }
1021         }
