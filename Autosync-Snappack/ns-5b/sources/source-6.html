145         try {

210      *
211      * @param maxRetries    The number of retries the customer has choosen in the UI
212      * @param retryInterval The number of seconds to wait between retries

216     @Override
217     public RetryPolicy getDefaultRetryPolicy(int maxRetries, long retryInterval)
218             throws SQLException {
219         return new RetryPolicy()
220                 .retryIf((o, throwable) -> {
221                     if (throwable instanceof SnapDataException) {
222                         Throwable cause = throwable.getCause();
223                         // If query was canceled by Lifecycle rules then we don't need to
224                         // retry it.
225                         return !StringUtils.containsIgnoreCase(cause.getMessage(),
226                                 "Operation is canceled.");
227                     } else if (throwable instanceof DataAccessException) {
228                         return true;
229                     }
230                     return false;
231                 })
232                 .withMaxRetries(maxRetries)
233                 .withDelay(retryInterval, TimeUnit.SECONDS);
234     }

238     public int[] delete(final String tableName, final String whereClause,
239                         List<Object> whereClauseValues, final DatabaseAccount account,
240                         final boolean isDynamicWhereClause,
241                         final int maxRetries, final long retryInterval)
242             throws BatchUpdateException {
243         int currentBatchSize;
244         DatabaseConfig databaseConfig = account.getDatabaseConfig();
245 
246         // Databricks JDBC driver (2.6.25) does not support batches.
247         // To speed up deletes we create one sql to imitate batches
248         // Example:
249         //  sql: delete from tableName where id = ?
250         //  bind values: 1,2,3
251         //  final query: delete from tableName where (id = 1) or (id = 2) or (id = 3)
252         if (deleteQueryBuilder == null) {
253             SQLDialect dialect = databaseConfig.getSQLDialect();
254             Settings settings = databaseConfig.getSettings();
255             DSLContext context = DSL.using(conn, dialect, settings);
256 
257             deleteQueryBuilder = new DeleteQueryBuilder(context);
258             deleteQueryBuilder.setTableName(tableName);
259             acquireConnection(account);
260         }
261 
262         if (StringUtils.isNotEmpty(whereClause)) {
263             deleteQueryBuilder.addClause(DSL.condition(whereClause, whereClauseValues.toArray()));
264         }
265 
266         currentBatchSize = deleteQueryBuilder.getClauses().size();
267         if (currentBatchSize == databaseConfig.getBatchSize()) {
268             return flushBatch(account, maxRetries, retryInterval);
269         }
270 
271         return null;
272     }

275     public int[] flushBatch(DatabaseAccount account, int maxRetry, long retryInterval)
276             throws BatchUpdateException {
277 
278         if (deleteQueryBuilder != null) {
279             DatabaseConfig databaseConfig = account.getDatabaseConfig();
280             SQLDialect dialect = databaseConfig.getSQLDialect();
281             Settings settings = databaseConfig.getSettings();
282             DSLContext context = DSL.using(conn, dialect, settings);
283 
284             DeleteQuery<Record> deleteQuery = deleteQueryBuilder.build();
285             int[] result = handleDeleteResult(executeQueryDirectly(context, deleteQuery,
286                     deleteQuery.getSQL(), maxRetry, retryInterval));
287             deleteQueryBuilder = null;
288             return result;
289         }
290         return super.flushBatch(account, maxRetry, retryInterval);
291     }

293     private int[] handleDeleteResult(int[] result) {
294         // Databricks returns -1 in case of truncation table. In handleUpdateResult we handle
295         // -2 as SUCCESS_NO_INFO. So we change -1 to -2 to show users a nice message.
296         return result[0] == -1 ? new int[]{-2} : result;
297     }

300     public int[] insert(final String tableName, final Document document,
301                         final DatabaseAccount account, final TableMetaData tableMetaData,
302                         final DuplicateKeyAction duplicateKeyAction,
303                         final boolean preserveCaseSensitiveness,
304                         final OutputViews outputViews, final ErrorViews errorViews,
305                         final DocumentUtility documentUtility, final List<Document> inputDocuments,
306                         final int maxRetries, final long retryInterval)
307             throws BatchUpdateException {
308         Map<String, Object> docData = (Map<String, Object>) document.get();
309         semiStructuredColumns = Lists.newArrayList();
310         for (String key : docData.keySet()) {
311 
312             Optional<Map.Entry<String, ColumnMetaData>> column = tableMetaData.entrySet()
313                     .stream()
314                     .filter(e -> e.getKey().equalsIgnoreCase(key))
315                     .findFirst();
316             // If any of Semi-Structured data type is there then we need to have a different
317             // Insert query and also need to keep track of key, so we can use proper converter
318             if (column.isPresent()) {
319                 String columnTypeName = column.get().getValue().getColumnTypeName();
320                 if (StringUtils.startsWithIgnoreCase(columnTypeName,STRUCT) ||
321                         StringUtils.startsWithIgnoreCase(columnTypeName, ARRAY) ||
322                         StringUtils.startsWithIgnoreCase(columnTypeName, MAP)) {
323                     semiStructuredColumns.add(key);
324                 }
325             }
326         }
327         if (!semiStructuredColumns.isEmpty()) {
328             return insertWithSemiStructuredColumns(tableName, document, account, tableMetaData,
329                     preserveCaseSensitiveness);
330         } else {
331             return super.insert(tableName, document, account, tableMetaData, duplicateKeyAction,
332                     preserveCaseSensitiveness, outputViews, errorViews, documentUtility,
333                     inputDocuments, maxRetries, retryInterval);
334         }
335     }

337     private int[] insertWithSemiStructuredColumns(final String tableName, final Document document,
338                                                   final DatabaseAccount account,
339                                                   final TableMetaData tableMetaData,
340                                                   final boolean preserveCaseSensitiveness) {
341         Map<String, Object> docData = (Map<String, Object>) document.get();
342         final DatabaseConfig databaseConfig = account.getDatabaseConfig();
343         final SQLDialect dialect = databaseConfig.getSQLDialect();
344         final Settings settings = databaseConfig.getSettings();
345         DSLContext context = DSL.using(conn, dialect, settings);
346         context.configuration()
347                 .set(new DefaultExecuteListenerProvider(new BatchExecuteListener()));
348         final Map<Field<?>, Object> fieldMap = convertValues(docData, tableMetaData, dialect,
349                 preserveCaseSensitiveness);
350         acquireConnection(account);
351         validateValues(fieldMap);
352         Set<Field<?>> fields = fieldMap.keySet();
353         List<String> selectExpressionList = new ArrayList<>();
354         List<Object> queryParams = new ArrayList<>();
355         List<String> columnNamesList = new ArrayList<>();
356 
357         for (Field field : fields) {
358             columnNamesList.add(field.getName());
359             boolean isSemiStructuredColumn = semiStructuredColumns.stream()
360                     .anyMatch(e -> e.equalsIgnoreCase(field.getName()));
361             if (isSemiStructuredColumn) {
362                 String dataType = tableMetaData.getColumnTypeName(field.getName());
363                 selectExpressionList.add(
364                         buildInsertValues(dataType, fieldMap.get(field),
365                                 queryParams));
366             } else {
367                 selectExpressionList.add("?");
368                 queryParams.add(fieldMap.get(field));
369             }
370         }
371         String insert = String.format(INSERT_INTO_TABLE_FMT,
372                 quoteIdentifier(tableName), String.join(", ", columnNamesList));
373         String query = String.format(SELECT_FMT, insert,
374                 String.join(", ", selectExpressionList));
375         LOG.debug("Inserting query: {}", query);
376 
377         final int[] result = {0};
378         context.connection(connection -> {
379             try (PreparedStatement ps = conn.prepareStatement(query)) {
380                 for (int i = 0; i < queryParams.size(); i++) {
381                     bindValueToStatement(ps, i + 1, queryParams.get(i));
382                 }
383 
384                 if (!ps.execute()) {
385                     result[0] = ps.getUpdateCount();
386                 }
387             }
388         });
389         return result;
390     }

392     void bindValueToStatement(PreparedStatement ps, int colno, Object obj) throws SQLException {
393         if (obj == null) {
394             ps.setObject(colno, null);
395         } else if (obj instanceof String) {
396             ps.setString(colno, (String) obj);
397         } else if (obj instanceof Short) {
398             ps.setShort(colno, (Short) obj);
399         } else if (obj instanceof Integer) {
400             ps.setInt(colno, (Integer) obj);
401         } else if (obj instanceof Long) {
402             ps.setLong(colno, (Long) obj);
403         } else if (obj instanceof Float) {
404             ps.setFloat(colno, (Float) obj);
405         } else if (obj instanceof Double) {
406             ps.setDouble(colno, (Double) obj);
407         } else if (obj instanceof java.sql.Date) {
408             ps.setDate(colno, (java.sql.Date) obj);
409         } else if (obj instanceof Timestamp) {
410             ps.setTimestamp(colno, (Timestamp) obj);
411         } else {
412             ps.setString(colno, String.valueOf(obj));
413         }
414     }

416     public String buildInsertValues(String dataType, Object data,
417                                     List<Object> queryParams) {
418         String jsonData;
419         if (data == null) {
420             return "null";
421         } else if (data instanceof Map || data instanceof List) {
422             jsonData = new Gson().toJson(data);
423         } else {
424             jsonData = data.toString();
425         }
426         queryParams.add(jsonData);
427         return String.format("from_json(?, '%s')", dataType);
428     }
