278     private final String escapeUrlProp(Object urlProp) {
279         String stringValue = String.valueOf(urlProp);
280         boolean containsRightBrace = false;
281         // See: https://learn.microsoft.com/en-us/sql/connect/jdbc/building-the-connection-url?
282         // view=sql-server-ver16#escaping-values-in-the-connection-url
283         if (stringValue.contains("}")) {
284             stringValue = stringValue.replaceAll("}", "}}");
285             containsRightBrace = true;
286         }
287         if (containsRightBrace || SPECIAL_CHARS_PATTERN.matcher(stringValue).find()) {
288             return "{" + stringValue + "}";

290             return stringValue;

295         /**
296          * Invoking thread to process future tasks from the blocking queue once its in instead of
297          * storing them in internal memory and processing in cleanup.
298          */
299 
300         outputWriterThread = threadFactory.newThread(() -> {

302             while (true) {
303                 try {
304                     future = futureBlockingQueue.poll(POLL_WAIT_TIME_IN_MS, TimeUnit

359     protected void shutdownThreadPool(final ThreadPoolExecutor threadPool) {
360         // first, no more new tasks accepted
361         threadPool.shutdown();
362         try {
363             // wait for some time for the scheduled tasks to complete
364             if (!threadPool.awaitTermination(DEFAULT_THREAD_POOL_COMPLETE_TIME,
365                     DEFAULT_THREAD_POOL_COMPLETE_TIME_UNIT)) {
366                 // still tasks didn't complete, so cancel them
367                 threadPool.shutdownNow();
368                 // wait for some time for cancelling the scheduled tasks
369                 if (!threadPool.awaitTermination(DEFAULT_THREAD_POOL_CANCEL_TIME,
370                         DEFAULT_THREAD_POOL_CANCEL_TIME_UNIT)) {
371                     LOG.error(THREAD_POOL_SHUTDOWN_ERROR_MESSAGE);
372                 }
373             }
374 
375         } catch (InterruptedException e1) {
376             // attempt to cancel the tasks again as the thread got interrupted
377             threadPool.shutdownNow();
378         }

494     /**
495      * Connects to Azure SQL database.
496      *
497      * Note: the SQLServerBulkCopy class is very particular about the type of
498      * Connection it receives - it MUST be a SQLServerConnection.
499      * <p>This includes matching ClassLoader - if we use a jarlist in the future it's possible that
500      * we'll load the driver via SLDBClassPathLoader and get an object that can't
501      * be cast to the required class in the SQLServerBulkCopy method.</p>
502      *
503      * @return Connection object
504      */
505     protected Connection connect() {
506         Connection conn = jdbcOperations.acquireConnection(account);
507         try {
508             // if we used HikariCP (or another connection manager) we'll get a wrapped connection.
509             // try unwrapping it.
510             // Note: We are using getSimpleName() here instead of instance of because
511             //  the connection might be obtained from a driver using a different class loader
512             if (!(conn instanceof SQLServerConnection) && (conn.unwrap(Connection.class) != null)) {
513                 Connection actualConn = conn.unwrap(Connection.class);
514 
515                 if (actualConn.getClass().getSimpleName().startsWith(SQL_SERVER_CONNECTION)) {
516                     conn = actualConn;
517                 }
518             }
519 
520             // if it's still not right type - release back to connection pool and
521             // create the connection ourselves. Note: this only supports standard
522             // authentication.
523             if (!(conn.getClass().getSimpleName().startsWith(SQL_SERVER_CONNECTION))) {
524                 conn.close();
525                 usingAcquiredConnection = false;
526 
527                 DatabaseConfig databaseConfig = account.getDatabaseConfig();
528                 SQLServerDataSource ds = new SQLServerDataSource();
529                 ds.setServerName(databaseConfig.getHostName());
530                 ds.setDatabaseName(databaseConfig.getDatabaseName());
531                 ds.setUser(databaseConfig.getUsername());
532                 ds.setPassword(databaseConfig.getPassword());
533                 conn = ds.getConnection();
534             }
535         } catch (Exception e) {
536             throw new ExecutionException(ERR_CONNECTION_REFUSED_MSG)
537                     .withReason(Throwables.getRootCause(e).getMessage())
538                     .withResolution(ERR_CONNECTION_REFUSED_RESOLUTION);
539         }
540         return conn;
541     }
