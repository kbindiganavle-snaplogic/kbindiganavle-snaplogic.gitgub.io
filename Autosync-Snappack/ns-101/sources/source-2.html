285      *
286      * @param document incoming document
287      * @return RetryPolicy object
288      */
289     public RetryPolicy createRetryPolicy(Document document) {
290         BigInteger retryVal = expNumRetry.eval(document);
291         if (Objects.nonNull(retryVal) && retryVal.intValue() >= MIN_RETRY) {
292             numRetry = retryVal.intValue();
293         }
294         BigInteger retryIntVal = expRetryInt.eval(document);
295         if (Objects.nonNull(retryIntVal) && retryIntVal.intValue() >= MIN_RETRY_INT) {
296             retryInt = retryIntVal.intValue();
297         }
298         return new RetryPolicy()
299                 .withMaxRetries(numRetry)
300                 .withDelay(retryInt, TimeUnit.SECONDS)
301                 .retryOn(Exception.class)
302                 .abortOn(SuggestViewAbortException.class);
303     }

322      * @param serviceEx Exception from Graah SDK API call
323      * @return true, if new token retrieved. false otherwise.

325     protected boolean refreshTokenOn401Error(GraphServiceException serviceEx) {
326         if (serviceEx.getResponseCode() == 401 || (serviceEx.getResponseCode() == 400 &&
327                 serviceEx.getMessage().contains("Error authenticating with resource"))) {
328             authenticationRetry++;
329             LOG.warn(String.format("Received %s error", serviceEx.getResponseCode()), serviceEx);
330             if (snapControl.reloadSnapAccount() || (authenticationRetry == 1 &&
331                     snapControl.reloadSnapAccount(true))) {
332                 LOG.info("Account reloaded, retrieving new access token");
333                 graphClient = utils.initializeGraphClient(account);
334                 return true;
335             }
336             LOG.warn("Account is not reloaded, failed to get new access token");
337         }
338         return false;
339     }

341     /**
342      * Handle ApplicationThrottled Error 429 gracefully. Refer below doc for best practices
343      * https://docs.microsoft.com/en-us/graph/throttling?context=graph%2Fapi%2F1.0
344      * &view=graph-rest-1.0
345      *
346      * @param serviceEx Exception from Graah SDK API call
347      * @return true, if needs to be retried
348      */
349     protected boolean handleThrottleError(GraphServiceException serviceEx) {
350         if (serviceEx.getResponseCode() == 429) {
351             throttleRetry++;
352             if (throttleRetry > Math.max(numRetry, MAX_THROTTLE_RETRY_LIMIT)) {
353                 // reset throttle count to 0 after limit is reached.
354                 throttleRetry = 0;
355                 return false;
356             }
357             LOG.warn("Received ApplicationThrottled Error, Error code: 429", serviceEx);
358             List<String> responseHeaders = serviceEx.getResponseHeaders();
359             int waitTime = 0;
360             if (Objects.nonNull(responseHeaders)) {
361                 for (String header : responseHeaders) {
362                     if (header.startsWith("Retry-After")) {
363                         String[] headerVal = header.split(":");
364                         try {
365                             waitTime = (headerVal.length > 1) ?
366                                     Double.valueOf(headerVal[1].trim()).intValue() : 0;
367                             // avoid cases if negative number is passed.
368                             waitTime = (waitTime < 0) ? 0 : waitTime;
369                         } catch (NumberFormatException nfe) {
370                             throw new SnapDataException(ERR_CONVERT_FAIL_MSG)
371                                     .withReason(ERR_CONVERT_FAIL_REASON)
372                                     .withResolutionAsDefect();
373                         }
374                         break;
375                     }
376                 }
377             }
378             try {
379                 // compute exponential backoff time
380                 int backoffTime = throttleRetry * (waitTime + 10);
381                 LOG.info("Recommended wait time is: {}, retry count is: {} and Sleeping for {} " +
382                                 "seconds", waitTime, throttleRetry, backoffTime);
383                 // Wait for backoff time
384                 Thread.sleep(TimeUnit.SECONDS.toMillis(backoffTime));
385             } catch (InterruptedException ie) {
386                 LOG.error("Wait Thread is interrupted", ie);
387                 Thread.currentThread().interrupt();
388             }
389             return true;
390         }
391         return false;
392     }

417     private void prepareMembersList(Document document, List<ExpressionProperty> membersListExpr,
418             List<ConversationMember> membersList, LinkedList<String> rolesList) {
419         membersListExpr.stream().forEach(memberExpr -> {
420             String memberID = memberExpr.eval(document);
421             if (StringUtils.isNotBlank(memberID)) {
422                 StringTokenizer tokenizer = new StringTokenizer(memberID, ",");
423                 while (tokenizer.hasMoreTokens()) {
424                     memberID = utils.extractID(tokenizer.nextToken().trim());
425                     if (StringUtils.isNotBlank(memberID)) {
426                         AadUserConversationMember member = new AadUserConversationMember();
427                         member.additionalDataManager().put(ODATA_TYPE,
428                                 new JsonPrimitive(CONVERSATION_MEMBER));
429                         member.additionalDataManager().put(USER_ODATA_BIND,
430                                 new JsonPrimitive(USER_PREFIX + memberID + USER_SUFFIX));
431                         if (Objects.nonNull(rolesList)) {
432                             member.roles = rolesList;
433                         }
434                         membersList.add(member);
435                     }
436                 }
437             }
438         });
439     }
