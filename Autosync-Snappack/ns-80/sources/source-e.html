154     private Set<String> parseResults(NamingEnumeration<SearchResult> searchResult)
155             throws NamingException {
156         Set<String> suggestedAttributes = new LinkedHashSet<>();
157         while (searchResult.hasMore()) {
158             SearchResult record = searchResult.next();
159             Enumeration<? extends Attribute> attributes = record.getAttributes().getAll();
160             while (attributes.hasMoreElements()) {
161                 String attribute = attributes.nextElement().toString();
162                 String[] keyValuePair = attribute.split(COLON);
163                 String fieldKey = keyValuePair[0];
164                 if (StringUtils.isBlank(fieldKey)) {
165                     continue;
166                 } else {
167                     suggestedAttributes.add(fieldKey);
168                 }
169             }
170         }
171         return suggestedAttributes;
172     }

183      * Updates an entry in Active Directory instance
184      */
185     protected void updateEntry(DirContext ldapContext, String distName,
186             List<String> attributeKeys, List<Object> attributeValues, String valueDelimiter)
187             throws NamingException {
188         ModificationItem[] modifications;
189         int numOfAttributes = attributeKeys.size();
190         modifications = new ModificationItem[numOfAttributes];
191         /**
192          * Submitting all attributes for modification with out skipping cn and dn attributes.
193          */
194         for (int i = 0; i < numOfAttributes; i++) {
195             String attrKey = attributeKeys.get(i);
196             BasicAttribute entry;
197             if (UNICODE_PWD.equalsIgnoreCase(attrKey)) {
198                 entry = new BasicAttribute(UNICODE_PWD, getUnicodePwdBytes(
199                         (String) attributeValues.get(i)));
200             } else {
201                 String valueCollection = (attributeValues.get(i) == null) ? null
202                         : attributeValues.get(i).toString();
203                 if (StringUtils.isBlank(valueDelimiter) || StringUtils.isBlank(valueCollection)) {
204                     entry = new BasicAttribute(attrKey, valueCollection);
205                 } else {
206                     String[] multipleValues = valueCollection.split(valueDelimiter);
207                     entry = new BasicAttribute(attrKey);
208                     for (String value : multipleValues) {
209                         entry.add(value);
210                     }
211                 }
212             }
213             modifications[i] = new ModificationItem(DirContext.REPLACE_ATTRIBUTE, entry);
214         }
215         ldapContext.modifyAttributes(distName, modifications);
216     }

298      * Returns all possible attributes based on the given entry
299      *
300      * @param ldapContext
301      * @param dn
302      * @param util
303      * @param cachedAttributesMap
304      * @return set containing all possible attributes
305      */
306     Set<String> getAllPossibleAttributes(LdapContext ldapContext, String dn,
307             ADUtilities util, Map<String, Set<String>> cachedAttributesMap) throws
308             NamingException {
309         try {
310             // Distinguished name here we receive is already having escape characters added
311             // by JNDI library for special characters except forward slash. I am adding escape
312             // character here only for forward slash and an extra level for backward slash.
313             String[] slashes = new String[] {"/", "\\"};
314             String[] escapedSlashes = new String[] {"\\/", "\\\\"};
315             dn = StringUtils.replaceEach(dn, slashes, escapedSlashes);
316             SearchControls controls = new SearchControls();
317             controls.setSearchScope(SearchControls.ONELEVEL_SCOPE);
318             controls.setReturningAttributes(new String[] {ATTR_ADMIN_DISPLAY_NAME});
319             Set<String> attributesToBeShown = new TreeSet<>();
320             Attributes attributesFetched = ldapContext.getAttributes(dn,
321                     new String[]{ATTR_OBJECT_CLASS, ATTR_OBJECT_CATEGORY});
322             Attribute objectClassAttr = attributesFetched.get(ATTR_OBJECT_CLASS);
323             Attribute objectCategoryAttr = attributesFetched.get(ATTR_OBJECT_CATEGORY);
324             if (objectCategoryAttr != null) {
325                 String objectCategory = objectCategoryAttr.get().toString();
326                 NamingEnumeration objectClassValues = objectClassAttr.getAll();
327                 String schemaDn = objectCategory.substring(objectCategory.toLowerCase()
328                         .indexOf(CN_SCHEMA.toLowerCase()));
329                 while (objectClassValues.hasMore()) {
330                     String objectClass = objectClassValues.next().toString();
331                     if (cachedAttributesMap != null) {
332                         Set<String> cachedAttributes = cachedAttributesMap.get(objectClass);
333                         if (cachedAttributes != null) {
334                             attributesToBeShown.addAll(cachedAttributes);
335                             continue;
336                         }
337                     }
338                     Set<String> newAttributes = new TreeSet<>();
339                     NamingEnumeration<SearchResult> results = ldapContext.search(schemaDn,
340                             String.format(ATTR_LDAP_DISPLAY_NAME_CONDITION, objectClass),
341                             controls);
342                     while (results.hasMore()) {
343                         SearchResult result = results.next();
344                         String adminDisplayName = result.getAttributes()
345                                 .get(ATTR_ADMIN_DISPLAY_NAME).get().toString();
346                         String objectClassDn = String.format(OBJECT_CLASS_DN, adminDisplayName,
347                                 schemaDn);
348                         attributesFetched = ldapContext.getAttributes(objectClassDn,
349                                 ATTRIBUTES_TO_BE_LOOKED_FOR.toArray(new String[0]));
350                         for (String attributeToBeLookedFor : ATTRIBUTES_TO_BE_LOOKED_FOR) {
351                             Attribute attribute = attributesFetched.get(attributeToBeLookedFor);
352                             if (attribute == null) {
353                                 continue;
354                             }
355                             NamingEnumeration attributeValues = attribute.getAll();
356                             while (attributeValues.hasMore()) {
357                                 String newAttribute = attributeValues.next().toString();
358                                 attributesToBeShown.add(newAttribute);
359                                 newAttributes.add(newAttribute);
360                             }
361                         }
362                         cachedAttributesMap.put(objectClass, newAttributes);
363                     }
364                 }
365             } else {
366                 attributesToBeShown = util.getSuggestedAttributes(ldapContext, dn);
367             }
368             return attributesToBeShown;
369         } catch (NamingException e1) {
370             throw e1;
371         }
372     }

375         /* Attribute parameters passed from search snap would need another level of escaping
376         because FILTER parameter of below format
377         (&(distinguishedName=CN=man\\\\+eesh,CN=Users,DC=ad1,DC=clouddev,DC=snaplogic,DC=com)

379         Which needs to be escaped as well. So, distinghuished name inside a FILTER

381         https://docs.oracle.com/javase/jndi/tutorial/basics/directory/filter.html
382         */
383         return StringUtils.replaceEach(dn, SPECIAL_CHARS, ESCAPED_SPECIAL_CHARS_FOR_SEARCH);

397         return StringUtils.replaceEach(dn, SPECIAL_CHARS, ESCAPED_SPECIAL_CHARS);

414     public static void putAttributeToMap(Map<String, Object> recordMap, Attribute attribute)
415             throws NamingException {
416         String fieldName = attribute.getID();
417         NamingEnumeration<?> fieldValue = attribute.getAll();
418         List<String> valueList = Lists.newArrayList();
419         while (fieldValue.hasMore()) {
420             Object value = fieldValue.next();
421             if (value instanceof byte[]) {
422                 // convert byte array to hex-string
423                 byte[] bytes = (byte[]) value;
424                 List<String> hexNumbers = new ArrayList<>(bytes.length);
425                 for (byte b : bytes) {
426                     hexNumbers.add(String.format(FORMAT_HEX, b));
427                 }
428                 value = String.join(StringUtils.SPACE, hexNumbers);
429             }
430             valueList.add(value.toString());
431         }
432         if (valueList.isEmpty()) {
433             // Value might be nothing so we still process that attribute with blank value
434             LOGGER.warn("NoSuchElement for {}", fieldName);
435             recordMap.put(fieldName, StringUtils.EMPTY);
436         } else {
437             recordMap.put(fieldName, String.join(", ", valueList));
438         }
439     }
