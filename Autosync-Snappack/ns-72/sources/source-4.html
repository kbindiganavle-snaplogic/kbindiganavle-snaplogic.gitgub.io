161      * Processes HttpResponse, and writes output document to the output view.
162      *
163      * @param httpResponse   - HttpResponse object
164      * @param requestHeaders - headers sent to the server to execute HTTP request
165      * @param document       - input document to be passed through
166      * @param responsesSpec  - response part of OpenAPI spec
167      *
168      * @return output document written to the output view
169      * @throws IOException if it fails while processing response entity
170      */
171     public Document processHttpResponse(HttpResponse httpResponse, List<Header> requestHeaders,
172             Document document, Map<String, Object> responsesSpec) throws IOException {
173         // get status line
174         StatusLine statusLine = httpResponse.getStatusLine();
175         Map<String, Object> statusMap = new LinkedHashMap<>();
176         int statusCode = statusLine.getStatusCode();
177         statusMap.put(STATUS_CODE, statusCode);
178         statusMap.put(STATUS_PHRASE, statusLine.getReasonPhrase());
179         statusMap.put(PROTO_VERSION, statusLine.getProtocolVersion().toString());
180         // get response headers
181         Header[] headers = httpResponse.getAllHeaders();
182         Map<String, Object> headerMap = new LinkedHashMap<>(headers.length);
183         for (Header header : headers) {
184             headerMap.put(header.getName(), header.getValue());
185         }
186         // prepare output document
187         Map<String, Object> outMap = new LinkedHashMap<>();
188         outMap.put(STATUS_LINE, statusMap);
189         outMap.put(HEADERS, headerMap);
190         // get response http entity
191         HttpEntity httpEntity = httpResponse.getEntity();
192         Object entity;
193         if (httpEntity == null) {
194             entity = NO_CONTENT;
195         } else {
196             // get Content-Type
197             Header header = httpEntity.getContentType();
198             String contentType = header == null ? null : header.getValue();
199             boolean isJsonContentType = StringUtils.containsIgnoreCase(contentType, JSON) ||
200                     StringUtils.containsIgnoreCase(contentType, JAVASCRIPT);
201             boolean isXmlContentType = StringUtils.containsIgnoreCase(contentType, XML);
202             // get Content-Encoding
203             header = httpEntity.getContentEncoding();
204             String contentEncoding = header == null ? null : header.getValue().toLowerCase();
205             // if response entity type is array, parse the array and stream documents to output view
206             if (isJsonContentType && enableProcessArray &&
207                     isProcessArray(statusCode, responsesSpec) ||
208                     StringUtils.containsIgnoreCase(contentType, CSV) ||
209                     StringUtils.containsIgnoreCase(contentType, SSV) ||
210                     StringUtils.containsIgnoreCase(contentType, TSV)) {
211                 Object data = null;
212                 if (displayHeadersGet) {
213                     data = collectData(httpEntity, contentType, contentEncoding);
214                     outMap.put(ENTITY, data);
215                     Document outDoc = documentUtility.newDocument(outMap);
216                     if (writeOutputView) {
217                         if (firstOutputView == null) {
218                             writeOutput(outputViews, outMap, document, passThrough);
219                         } else {
220                             firstOutputView.write(documentUtility.newDocument(outDoc, document));
221                         }
222                     }
223                 } else {
224                     data = streamContent(httpEntity, contentType, contentEncoding);
225                     outMap.put(ENTITY, data);
226                 }
227                 return documentUtility.newDocument(outMap);
228             }
229             // process http entity
230             entity = getContent(httpEntity, contentEncoding);
231             if (entity instanceof String) {
232                 String entityStr = (String) entity;
233                 if (entityStr.isEmpty()) {
234                     entity = NO_CONTENT;
235                 } else {
236                     try {
237                         // attempt to parse json or xml
238                         if (isJsonContentType) {
239                             entity = OBJECT_MAPPER.readValue(entityStr, OBJECT_TYPE_REFERENCE);
240                         } else if (isXmlContentType) {
241                             entity = parseXML(entityStr, xmlUtils);
242                         }
243                     } catch (SuggestViewAbortException e) {
244                         throw e;
245                     } catch (Exception e) {
246                         // if it fails to parse json or xml, leave text as is
247                         LOG.warn("Content-Type header: {}, Content: {}, failed to parse",
248                                 contentType, entityStr);
249                     }
250                 }
251             } else if (entity == null ||
252                     entity instanceof byte[] && ((byte[]) entity).length == 0) {
253                 entity = NO_CONTENT;
254             }
255         }
256         if (statusCode >= SC_OK && statusCode <= SC_TEMPORARY_REDIRECT) {
257             if (entityHasErrors(entity)) {
258                 processError(entity,document);
259                 return null;
260             } else {
261                 if (entityHasWarnings(entity)) {
262                     processWarnings(entity);
263                 }
264                 // write output document
265                 outMap.put(ENTITY, entity);
266                 // passThrough
267                 Document outDoc = documentUtility.newDocument(outMap);
268                 if (writeOutputView) {
269                     if (firstOutputView == null) {
270                         writeOutput(outputViews, outMap, document, passThrough);
271                     } else {
272                         firstOutputView.write(documentUtility.newDocument(outDoc, document));
273                     }
274                 }
275                 return outDoc;
276             }
277         } else {
278             outMap.put(ERROR_ENTITY, entity);
279             outMap.put(REQUEST_HEADERS, getHeaderMap(requestHeaders.toArray(new Header[0])));
280             throw new SnapDataException(documentUtility.newDocumentFor(document, outMap),
281                     ERR_OPENAPI)
282                 .withReason(REASON_ERROR_ENTITY)
283                 .withResolution(RESOLUTION_ADDRESS_ISSUE);
284         }
285     }

335      * Checks the response part of OpenAPI spec to find if the response entity is an array type
336      * so that it can be parsed into a document stream.
337      *
338      * @param statusCode    - status code in HttpResponse
339      * @param responsesSpec - response part of OpenAPI spec
340      *
341      * @return true if the response entity is an array type, false otherwise
342      */
343     protected boolean isProcessArray(Integer statusCode, Map<String, Object> responsesSpec) {
344         for (Map.Entry<String, Object> entry : responsesSpec.entrySet()) {
345             if (entry.getKey().equals(statusCode.toString())) {
346                 Map<String, Object> value = (Map<String, Object>) entry.getValue();
347                 Object type = isVersion2 ? readJsonPath(value, SCHEMA, TYPE) :
348                     readJsonPath(value, CONTENT, APPLICATION_JSON, SCHEMA, TYPE);
349                 if (type == null) {
350                     type = readJsonPath(value, CONTENT, APPLICATION_XML, SCHEMA, TYPE);
351                 }
352                 if (type != null) {
353                     return ARRAY.equalsIgnoreCase((String) type);
354                 } else {
355                     Object object = isVersion2 ? readJsonPath(value, SCHEMA, REF) :
356                         readJsonPath(value, CONTENT, APPLICATION_JSON, SCHEMA, REF);
357                     if (object instanceof String) {
358                         Map<String, Object> refSpec = openApiHelper.getRefSpec((String) object,
359                                 false);
360                         object = refSpec.get(TYPE);
361                         return object instanceof String && ARRAY.equalsIgnoreCase((String) object);
362                     }
363                 }
364             }
365         }
366         return false;
367     }

371      *
372      * @param httpEntity      - response entity
373      * @param contentType     - content-type of entity
374      * @param contentEncoding - content-encoding of entity
375      *
376      * @return data in the last document
377      * @throws IOException if it fails in processing stream

379     protected Object streamContent(HttpEntity httpEntity, String contentType,
380             String contentEncoding) throws IOException {
381         try (InputStream is = httpEntity.getContent()) {
382             InputStream inputStream = StringUtils.containsIgnoreCase(contentEncoding, GZIP) ?
383                 new GZIPInputStream(is) : is;
384             if (StringUtils.containsIgnoreCase(contentType, CSV)) {
385                 return parseDelimitedData(inputStream, getCsvFormat());
386             } else if (StringUtils.containsIgnoreCase(contentType, SSV)) {
387                 return parseDelimitedData(inputStream, getSsvFormat());
388             } else if (StringUtils.containsIgnoreCase(contentType, TSV)) {
389                 return parseDelimitedData(inputStream, getTsvFormat());
390             }
391             Iterator<Object> iterator =
392                 StringUtils.containsIgnoreCase(contentType, APPLICATION_JSON) ?
393                     new JsonDataIterator(inputStream) : getXmlDataIterator(inputStream);
394             Object data = null;
395             while (iterator.hasNext()) {
396                 data = iterator.next();
397                 Document outDoc = documentUtility.newDocument(data);
398                 if (firstOutputView == null) {
399                     outputViews.write(outDoc);
400                 } else {
401                     firstOutputView.write(outDoc);
402                 }
403             }
404             return data;
405         }
406     }

408     private Object parseDelimitedData(InputStream inputStream, CSVFormat format)
409             throws IOException {
410         // init input stream reader
411         InputStreamReader inputStreamReader = new InputStreamReader(
412                 new BufferedInputStream(inputStream), StandardCharsets.UTF_8);
413         // init delimited-document parser iterator
414         Iterator<CSVRecord> it = format.parse(inputStreamReader).iterator();
415         // init delimited-document headers
416         CSVRecord headerRecord = it.hasNext() ? it.next() : null;
417         if (headerRecord == null) {
418             throw new SnapDataException(ERR_PARSING_DATA)
419                     .withReason(REASON_EMPTY_ENTITY)
420                     .withReason(RESOLUTION_EMPTY_ENTITY);
421         }
422         List<String> headers = Lists.newArrayListWithExpectedSize(headerRecord.size());
423         Iterator<String> headerIterator = headerRecord.iterator();
424         while (headerIterator.hasNext()) {
425             headers.add(headerIterator.next());
426         }
427         // parse and write output view
428         Map<String, Object> data = null;
429         while (it.hasNext()) {
430             CSVRecord record = it.next();
431             data = new LinkedHashMap<>(headers.size());
432             int headerIndex = 0;
433             for (String header : headers) {
434                 Object value = record.get(headerIndex++);
435                 data.put(header, value);
436             }
437             if (additionalOutputData != null) {
438                 data.putAll(additionalOutputData);
439             }
440             outputViews.write(documentUtility.newDocument(data));
441         }
442         return data;
443     }

468      * Converts HTTP entity to string or byte array depending on the content-type.
469      *
470      * @param httpEntity      - HTTP entity
471      * @param contentEncoding - content-encoding of the entity
472      *
473      * @return string or byte array
474      * @throws IOException if it fails in processing HTTP entity
475      */
476     protected static Object getContent(HttpEntity httpEntity, String contentEncoding)
477             throws IOException {
478         // get content-type header
479         Header header = httpEntity.getContentType();
480         String contentType = header == null ? null : header.getValue().toLowerCase();
481         // try to get character set from content-type with default UTF-8
482         String charset = StandardCharsets.UTF_8.name();
483         if (contentType != null) {
484             int charsetEqualToIndex = contentType.indexOf(CHARSET_EQUAL_TO);
485             if (charsetEqualToIndex > -1) {
486                 charset = contentType.substring(charsetEqualToIndex + CHARSET_EQUAL_TO.length());
487                 // the content-type may be of pattern "...;charset=utf-8;..."
488                 charset = substringTo(charset, ';');
489                 // the content-type may be of pattern "..., charset=utf-8, ..."
490                 charset = substringTo(charset, ',');
491             }
492         }
493         try (InputStream is = httpEntity.getContent()) {
494             if (is == null) {
495                 return null;
496             }
497             // if Content-Encoding contains 'gzip', unzip the content
498             InputStream inputStream = StringUtils.containsIgnoreCase(contentEncoding, GZIP) ?
499                     new GZIPInputStream(is) : is;
500             if (StringUtils.containsAny(contentType, TEXT_TYPES)) {
501                 // convert input stream to string
502                 try {
503                     return IOUtils.toString(inputStream, charset);
504                 } catch (IllegalCharsetNameException | UnsupportedEncodingException e) {
505                     return IOUtils.toString(inputStream, StandardCharsets.UTF_8);
506                 }
507             } else {
508                 // convert input stream to byte array
509                 return IOUtils.toByteArray(inputStream);
510             }
511         }
512     }
