478     public void configure(PropertyValues propertyValues) {
479         this.propertyValues = propertyValues;
480         if (account == null) {
481             throw new ConfigurationException(ERR_NULL_ACCOUNT)
482                     .withReason(REASON_ACCOUNT_REQUIRED)
483                     .withResolution(RESOLUTION_ADDRESS_ISSUE);
484         }
485         // initialize output views if the secon output view is open
486         TreeMap<String, OutputView> views = new TreeMap<>(outputViews.getAll());
487         if (views.size() > 1) {
488             firstOutputView = (DocumentOutputView) views.firstEntry().getValue();
489             secondOutputView = (DocumentOutputView) views.lastEntry().getValue();
490         }
491         openApiUrlExpr = propertyValues.getAsExpression(OPEN_API_URL_PROP);
492         openApiBaseExpr = propertyValues.getAsExpression(OPEN_API_BASE_PROP);
493         openApiPathExpr = propertyValues.getAsExpression(OPEN_API_PATH_PROP);
494         operationExpr = propertyValues.getAsExpression(OPERATION_PROP);
495         trustAllCerts = propertyValues.getBoolean(TRUST_ALL_CERTS_PROP, false);
496         try {
497             // attempt to evaluate expression properties in case it contains pipeline parameters
498             if (openApiBaseExpr == null || openApiUrlExpr == null) {
499                 openApiUrl = validateAndGetOpenApiUrl(propertyValues);
500             } else {
501                 openApiUrl = getString(openApiUrlExpr.eval(null));
502                 openApiBase = getString(openApiBaseExpr.eval(null));
503             }
504             openApiPath = getString(openApiPathExpr.eval(null));
505             operation = getString(operationExpr.eval(null));
506         } catch (Exception e) {
507             // expression properties may need values from input document and will be evaluated
508             // again in process()

556     protected Map<String, Object> initOpenAPI(Document document) throws IOException {
557         // get OpenAPI spec
558         getOpenApiSpec();
559         // init openApiHelper
560         openApiHelper = openApiHelper == null ? new OpenApiHelper(resolvedV3OpenApiSpec,
561                 v2CompatibleOpenApiSpec)
562                 : openApiHelper;
563         // get input document as map
564         Object input = document == null ? null : document.get();
565         Map<String, Object> inputMap = openApiHelper.getAsMap(input);
566         // evaluate expression properties
567         if (openApiUrlExpr != null) {
568             openApiUrl = getString(openApiUrlExpr.eval(document));
569         }
570         openApiPath = getString(openApiPathExpr.eval(document));
571         operation = getString(operationExpr.eval(document));
572         if (openApiBaseExpr != null) {
573             String baseUrl = getString(openApiBaseExpr.eval(document));
574             openApiBase = StringUtils.isBlank(baseUrl) ? openApiBase : baseUrl;
575         }
576         if (advPropMap != null) {
577             evalAdvancedProperties(errorViews, document);
578         }
579         queryParams = new LinkedHashMap<>();
580         for (Pair<ExpressionProperty, ExpressionProperty> queryParamProp : queryParamProps) {
581             String key = evalToString(queryParamProp.getKey(), document);
582             if (StringUtils.isNotBlank(key)) {
583                 queryParams.put(key, evalToString(queryParamProp.getValue(), document));
584             }
585         }
586 
587         // init request headers
588         requestHeaders = buildHeaders(httpHeaderProps, document,
589                 new BasicHeader(ACCEPT, ACCEPT_ALL));
590         // get auth header
591         authHeader = authHeader == null ? account.connect() : authHeader;
592         // if an auth header does not exists in the list from Snap property, add it to the list
593         if (authHeader != null &&
594                 !openApiHelper.containsHeader(requestHeaders, authHeader.getName())) {
595             requestHeaders.add(authHeader);
596         }
597         // init HttpUtils
598         httpUtils = new HttpUtils(outputViews, errorViews, documentUtility, openApiHelper,
599                 passThrough, displayHeadersGet)
600                 .withFirstOutputView(firstOutputView)
601                 .withWriteOutputView(writeOutputView)
602                 .withXmlUtils(xmlUtils);
603         httpUtils.setEnableProcessArray(enableProcessArray);
604         httpUtils.setIsVersion2(isVersion2);
605         httpUtils.setCsvFormat(csvFormat);
606         httpUtils.setSsvFormat(ssvFormat);
607         httpUtils.setTsvFormat(tsvFormat);
608         return inputMap;
609     }

611     protected Document process(Map<String, Object> inputMap, Document document) throws Exception {
612         // get service URL
613         if (StringUtils.isNotBlank(openApiBase) && StringUtils.isNotBlank(openApiPath)) {
614             serviceUrl = concatPath(openApiBase, openApiPath);
615         } else {
616             throw new SnapDataException(ERR_SERVICE_URL)
617                     .formatWith(openApiBase, openApiPath)
618                     .withReason(REASON_INVALID_OPENAPI_URL)
619                     .withResolution(RESOLUTION_ADDRESS_ISSUE);
620         }
621         // get operation spec
622         Map<String, Object> operationSpec = openApiHelper.getSpec(resolvedV3OpenApiSpec,
623                 PATHS, openApiPath, operation);
624         if (MapUtils.isEmpty(operationSpec)) {
625             throw new SnapDataException(ERR_SPEC_NOT_FOUND)
626                     .formatWith(openApiPath, operation)
627                     .withReason(REASON_SPEC_NOT_FOUND)
628                     .withResolution(RESOLUTION_ADDRESS_ISSUE);
629         }
630         // get common parameter spec
631         List<Map<String, Object>> commonParameters = (List<Map<String, Object>>)
632                 openApiHelper.readJsonPath(resolvedV3OpenApiSpec, PATHS, openApiPath,
633                         PARAMETERS);
634         // write operationSpec to the second output view
635         openApiHelper.outputOperationSpec(openApiPath, operation, secondOutputView,
636                 documentUtility);
637         // execute http request and process response
638         int pageCount = 0;
639         do {
640             serviceUrl = openApiHelper.processParameters(operationSpec, inputMap, serviceUrl,
641                     requestHeaders, queryParams, commonParameters);
642             if (isUrlEncodingEnabled){
643                 serviceUrl = String.valueOf(urlEncoder.encodeURI(serviceUrl));
644             }
645             // prepare and execute http request with retry
646             executeRequestWithRetry(operationSpec, inputMap, document);
647             // check for pagination
648             pageCount = maxPages == -1 ? Integer.MIN_VALUE : ++pageCount;
649         } while (hasNextPage(operation, inputMap, outDoc) && pageCount < maxPages);
650         return outDoc;
651     }

653     protected void executeRequestWithRetry(Map<String, Object> operationSpec,
654             Map<String, Object> inputMap, Document document) throws IOException {
655         try (RestHttpClient restHttpClient = new RestHttpClient()) {
656             HttpResponse httpResponse = Failsafe.with(retryPolicy)
657                     .onFailedAttempt(e -> LOG.warn("{} - Failed to execute Request: {}",
658                             getRetrySignature(), e.getMessage()))
659                     .onRetry(retryLogListener(getRetrySignature()))
660                     .get(context -> prepareAndExecuteHttpRequest(operationSpec, inputMap,
661                             requestHeaders, queryParams, restHttpClient));
662             // process http response
663             outDoc = httpUtils.processHttpResponse(httpResponse, requestHeaders,
664                     document, openApiHelper.getSpec(operationSpec, RESPONSES),
665                     additionalOutputData);
666         }
667     }

728     protected void getOpenApiSpec() throws IOException {
729         if (MapUtils.isNotEmpty(resolvedV3OpenApiSpec)) {
730             return;
731         }
732         // initialize input views
733         if (firstInputView == null && secondInputView == null) {
734             TreeMap<String, InputView> views = new TreeMap<>(inputViews.getAll());
735             int size = views.size();
736             if (size == 2) {
737                 secondInputView = views.lastEntry().getValue();
738                 firstInputView = views.firstEntry().getValue();
739             } else if (size == 1) {
740                 firstInputView = views.firstEntry().getValue();
741             }
742         }
743         // get OpenAPI spec from the second input view if any
744         if (secondInputView != null) {
745             Iterator<Document> iterator = inputViews.
746                 getDocumentsFrom(secondInputView);
747             if (iterator.hasNext()) {
748                 Document document = iterator.next();
749                 if (document != null) {
750                     resolvedV3OpenApiSpec = documentUtility
751                         .getAsMap(document, errorViews);
752                 }
753             }
754         }
755         // download OpenAPI spec from OpenAPI URL
756         if (MapUtils.isEmpty(resolvedV3OpenApiSpec) && StringUtils.isNotBlank(openApiUrl)) {
757             // normalize sldb URL
758             if (!openApiUrl.contains(PROTOCOL_SEPARATOR)) {
759                 openApiUrl = String.format(FORMAT_SLDB, openApiUrl);
760             }
761             // get authorization header from account if any
762             if (authHeader == null) {
763                 authHeader = account.connect();
764             }
765             // Download OpenAPI spec with retries
766             URI uri = encodeUriWithQuery(openApiUrl);
767             retryPolicy = retryPolicy == null ? initRetryPolicy(propertyValues, null) : retryPolicy;
768             Object result = Failsafe.with(retryPolicy)
769                     .onFailedAttempt(e -> LOG.warn("{} - Failed to download OpenAPI spec: {}",
770                             getRetrySignature(), e.getMessage()))
771                     .onRetry(retryLogListener(getRetrySignature()))
772                     .get(context -> downloadOpenApiSpec(uri));
773             // OpenAPI spec should be a Map object
774             if (result instanceof Map) {
775                 resolvedV3OpenApiSpec = (Map<String, Object>) result;
776             } else if (result instanceof List) {
777                 resolvedV3OpenApiSpec = ((List<Map<String, Object>>) result).get(0);
778             } else if (result instanceof Iterator) {
779                 Iterator<Object> iterator = (Iterator<Object>) result;
780                 if (iterator.hasNext()) {
781                     Object object = iterator.next();
782                     if (object instanceof Map) {
783                         resolvedV3OpenApiSpec = (Map<String, Object>) object;
784                     }
785                 }
786             } else {
787                 throw new SnapDataException(ERR_OPENAPI_SPEC)
788                     .withReason(REASON_OPENAPI_SPEC)
789                     .withResolution(RESOLUTION_ADDRESS_ISSUE);
790             }
791             updateOpenApiSpec();
792         }
793         // get default OpenAPI spec if any
794         if (MapUtils.isEmpty(resolvedV3OpenApiSpec)) {
795             getDefaultOpenApiSpec();
796         }
797         if (MapUtils.isEmpty(resolvedV3OpenApiSpec)) {
798             throw new SnapDataException(ERR_OPENAPI_SPEC_NOT_FOUND)
799                 .withReason(REASON_OPENAPI_SPEC_NOT_FOUND)
800                 .withResolution(RESOLUTION_ADDRESS_ISSUE);
801         }
802         // check if it's OpenAPI 2.0
803         Object openapiVersion = resolvedV3OpenApiSpec.get(SWAGGER);
804         openapiVersion = openapiVersion == null ? resolvedV3OpenApiSpec.get(OPENAPI) :
805                 openapiVersion;
806         isVersion2 = openapiVersion instanceof String &&
807                 ((String) openapiVersion).startsWith(VERSION2);
808     }

838     @VisibleForTesting
839     protected Object downloadOpenApiSpec(URI uri) throws IOException {
840         computeOtherOpenApiSpec(uri);
841         switch (uri.getScheme()) {
842             case PROTOCOL_SLDB:
843                 // download spec through URLConnection
844                 URLConnection urlConnection = uri.toURL().openConnection();
845                 urlConnection.connect();
846                 try (InputStream inputStream = urlConnection.getInputStream()) {
847                     if (inputStream != null) {
848                         ParseOptions parseOptions = new ParseOptions();
849                         parseOptions.setResolve(true);
850                         parseOptions.setResolveFully(true);
851                         String text = IOUtils.toString(inputStream, StandardCharsets.UTF_8);
852                         SwaggerParseResult result = new OpenAPIParser()
853                                 .readContents(text, null, parseOptions);
854                         return parsedResultToMap(result);
855                     }
856                 }
857                 break;
858             case PROTOCOL_HTTPS:
859             case PROTOCOL_HTTP:
860                 String currentSetting = "";
861                 if (trustAllCerts) {
862                     currentSetting = System.getProperty(TRUST_ALL) != null
863                             ? System.getProperty(TRUST_ALL) : "";
864                     System.setProperty(TRUST_ALL , "true");
865                 }
866                 ParseOptions parseOptions = new ParseOptions();
867                 parseOptions.setResolve(true);
868                 parseOptions.setResolveFully(true);
869                 SwaggerParseResult result = new OpenAPIParser()
870                         .readLocation(openApiUrl, null, parseOptions);
871                 if (trustAllCerts && StringUtils.isEmpty(currentSetting)) {
872                     System.clearProperty(TRUST_ALL);
873                 } else if (trustAllCerts){
874                     System.setProperty(TRUST_ALL, currentSetting);
875                 }
876                 return parsedResultToMap(result);
877             default:
878                 return null;
879             // No-Op
880         }
881         return null;
882     }

898     private void computeOtherOpenApiSpec(URI uri) throws IOException {
899         Object result = null;
900         String text = "";
901         switch (uri.getScheme()) {
902             case PROTOCOL_SLDB:
903                 // download spec through URLConnection
904                 URLConnection urlConnection = uri.toURL().openConnection();
905                 urlConnection.connect();
906                 try (InputStream inputStream = urlConnection.getInputStream()) {
907                     if (inputStream != null) {
908                         text = IOUtils.toString(inputStream, StandardCharsets.UTF_8);
909                         result = OBJECT_MAPPER.readValue(text, OBJECT_TYPE_REFERENCE);
910                     }
911                 } catch (IOException e) {
912                     if (text != null && !StringUtils.isEmpty(text)) {
913                         result = parseYaml(text);
914                     }
915                 }
916                 break;
917             case PROTOCOL_HTTPS:
918             case PROTOCOL_HTTP:
919                 // download spec through ProxyHttpClient
920                 HttpUriRequest httpUriRequest = new HttpGet(encodeUriWithQuery(openApiUrl));
921                 httpUriRequest.setHeader(new BasicHeader(ACCEPT, ACCEPT_ALL));
922                 if (authHeader != null) {
923                     httpUriRequest.setHeader(authHeader);
924                 }
925                 HttpResponse httpResponse = null;
926                 try (RestHttpClient restHttpClient = new RestHttpClient()){
927                     httpResponse = restHttpClient.executeRequest(httpUriRequest, trustAllCerts);
928                     if (SC_OK == httpResponse.getStatusLine().getStatusCode()) {
929                         HttpEntity httpEntity = httpResponse.getEntity();
930                         if (httpEntity != null) {
931                             try (InputStream inputStream = httpEntity.getContent()) {
932                                 if (inputStream != null) {
933                                     text = IOUtils.toString(inputStream,
934                                             StandardCharsets.UTF_8);
935                                     result = OBJECT_MAPPER.readValue(text, OBJECT_TYPE_REFERENCE);
936                                 }
937                             }
938                         }
939                     }
940                 } catch (IOException e) {
941                     if (text != null && !StringUtils.isEmpty(text)) {
942                         result = parseYaml(text);
943                     }
944                 }
945             default:
946                 // No-Op
947         }
948         if (result instanceof Map) {
949             v2CompatibleOpenApiSpec = (Map<String, Object>) result;
950         } else if (result instanceof List) {
951             v2CompatibleOpenApiSpec = ((List<Map<String, Object>>) result).get(0);
952         } else if (result instanceof Iterator) {
953             Iterator<Object> iterator = (Iterator<Object>) result;
954             if (iterator.hasNext()) {
955                 Object object = iterator.next();
956                 if (object instanceof Map) {
957                     v2CompatibleOpenApiSpec = (Map<String, Object>) object;
958                 }
959             }
960         }
961     }

977         for (Map.Entry<String, ExpressionProperty> entry : advPropMap.entrySet()) {
978             ExpressionProperty expressionProperty = entry.getValue();
979             switch (entry.getKey()) {

985                     maxPages = evalIntProperty(expressionProperty, document,
986                             DEFAULT_MAX_PAGES, -1, errorViews);
987                     break;
988                 case ENABLE_PROCESS_ARRAY_LABEL:
989                     enableProcessArray = getBoolean(expressionProperty.eval(document));
990                     break;
991                 case MULTIPART_FILE_CONTENT_KEY_LABEL:
992                     String multipartFileContentKey = getString(expressionProperty.eval(document));
993                     if (StringUtils.isNotBlank(multipartFileContentKey)) {

1000                         openApiHelper.withMultipartFilenameKey(multipartFilenameKey);
1001                     }
1002                     break;
1003                 default:
1004                     LOG.warn("Unexpected entry key.");
1005             }

1009     /**
1010      * OpenAPI does not specify how the pagination should be implemented. Subclass may override
1011      * this method to match the server's pagination implementation.
1012      *
1013      * @param operation - http method used in the request
1014      * @param inputMap  - some server implementations require the next http entity to include
1015      *                    data from the last http response entity
1016      * @param outDoc    - output document with the last http response
1017      * @return true if a next page exists
1018      */
1019     protected boolean hasNextPage(String operation, Map<String, Object> inputMap, Document outDoc) {
1020         outDoc = documentUtility.newDocument(outDoc, inputMap);
1021         boolean hasNext = false;
1022         if (StringUtils.containsAny(operation.toUpperCase(),
1023                 HttpGet.METHOD_NAME, HttpHead.METHOD_NAME)) {
1024             for (Map.Entry<String, ExpressionProperty> entry : advPropMap.entrySet()) {
1025                 ExpressionProperty expressionProperty = entry.getValue();
1026                 Object value = expressionProperty.eval(outDoc, outDoc.get(), NULL_VALUE_HANDLER);
1027                 switch (entry.getKey()) {
1028                     case HAS_NEXT_LABEL:
1029                         hasNext = getBoolean(value);
1030                         break;
1031                     case NEXT_URL_LABEL:
1032                         serviceUrl = getString(value);
1033                         break;
1034                     default:
1035                         LOG.warn("Unexpected entry key.");
1036                 }
1037             }
1038             if (hasNext) {
1039                 updateOffset(inputMap);
1040                 if (paginationInterval > 0) {
1041                     sleepNoInterrupt(paginationInterval, TimeUnit.SECONDS, LOG);
1042                 }
1043             }
1044         }
1045         return hasNext;
1046     }

1056     private Map<String, ExpressionProperty> prepareAdvancedProperties(
1057             PropertyValues propertyValues) {
1058         List<Map<String, Object>> advPropList = propertyValues.get(ADVANCED_PROPERTIES_PROP);
1059         Map<String, ExpressionProperty> expressionPropMap = new LinkedHashMap<>();
1060         if (isNotEmpty(advPropList)) {
1061             for (Map<String, Object> propMap : advPropList) {
1062                 String propertyLabel = jsonPathUtil.nullableRead(PROPERTIES_PATH, propMap);
1063                 boolean isExpression = getBoolean(readJsonPath(propMap, VALUES_PROP, EXPRESSION));
1064                 boolean enableExpr = false;
1065                 if (StringUtils.containsAny(propertyLabel, HAS_NEXT_LABEL, NEXT_URL_LABEL) &&
