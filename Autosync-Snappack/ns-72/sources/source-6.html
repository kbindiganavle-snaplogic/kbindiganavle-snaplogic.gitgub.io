477     private void mergeMaps(Map<String, Object> map1, Map<String, Object> map2){
478         for (String elem : map1.keySet()){
479             if (!map2.containsKey(elem)){
480                 map2.put(elem, map1.get(elem));
481             } else {
482                 Map<String, Object> value1 = (Map<String, Object>) map1.get(elem);
483                 Map<String, Object> value2 = (Map<String, Object>) map2.get(elem);
484                 if (value1.size() > value2.size()){
485                     map2.put(elem, value1);
486                 }
487             }
488         }
489         for (String elem : map2.keySet()){
490             if (!map1.containsKey(elem)){
491                 map1.put(elem, map2.get(elem));
492             }
493         }

520     protected Map<String, Object> getRefSpec(String ref, boolean getFromOtherSpec) {
521         try {
522             if (ref != null) {
523                 ref = convertRefToJsonPath(ref);
524             } else {
525                 return MapUtils.EMPTY_MAP;
526             }
527             JsonPath jsonPath = JsonPathImpl.compile(ref);
528             if (getFromOtherSpec) {
529                 if (jsonPath.read(v2CompatibleOpenApiSpec) != null) {
530                     return JsonPathImpl.compile(ref).read(v2CompatibleOpenApiSpec);
531                 } else {
532                     return MapUtils.EMPTY_MAP;
533                 }
534             } else {
535                 if (jsonPath.read(resolvedV3OpenApiSpec) != null) {

546     private ObjectSchema createRequestBodySchemaV2(SchemaProvider provider,
547             Map<String, Object> spec, boolean required, ObjectSchema requestBody)
548         throws InvalidPathException {
549         JsonPath refPath = buildJsonPath(SCHEMA, REF);
550         if (refPath.existsIn(spec) && refPath.read(spec) != null) {
551                 // get content schema reference
552                 String schemaRef = refPath.read(spec);
553                 // ceate request body schema from #Ref
554                 requestBody = createSchemaFromRef(provider,
555                         REQUEST_BODY, required, schemaRef, null);
556         } else {
557             JsonPath typePath = buildJsonPath(SCHEMA, TYPE);
558             if (!typePath.existsIn(spec)) {
559                 typePath = buildJsonPath(TYPE);
560             }
561             if (typePath.existsIn(spec)) {
562                 String type = typePath.read(spec);
563                 SnapType snapType = getSnapType(type);
564                 // request body may be a list of maps
565                 if (snapType.equals(SnapType.TABLE)) {
566                     Map<String, Object> itemsSpec = getSpec(spec, SCHEMA, ITEMS);
567                     if (itemsSpec != null) {
568                         Object refValue = itemsSpec.get(REF);
569                         if (refValue != null) {
570                             requestBody = provider.createSchema(snapType, REQUEST_BODY);
571                             Schema child = createSchemaFromRef(provider, REQUEST_BODY,
572                                     required, (String) refValue, null);
573                             requestBody.addChild(child);
574                         } else if (itemsSpec.containsKey(TYPE)) {
575                             requestBody = provider.createSchema(getSnapType(OBJECT), REQUEST_BODY);
576                             Schema child = provider
577                                     .createSchema(snapType, spec.get(NAME).toString());
578                             type = itemsSpec.get(TYPE).toString();
579                             Schema grandchild = provider.createSchema(getSnapType(type),
580                                     StringUtils.EMPTY);
581                             ((ObjectSchema) child).addChild(grandchild);
582                             requestBody.addChild(child);
583                         }
584                     }
585                 } else {
586                     Object name = spec.get(NAME);
587                     if (name instanceof String && StringUtils.isNotBlank((String) name)) {
588                         if (BODY.equalsIgnoreCase((String) name)) {
589                             if (requestBody == null) {
590                                 requestBody = provider.createSchema(snapType, REQUEST_BODY);
591                                 requestBody.setRequired(required);
592                             }
593                         } else {
594                             if (requestBody == null) {
595                                 requestBody = provider.createSchema(getSnapType(OBJECT),
596                                         REQUEST_BODY);
597                             }
598                             Schema child = provider.createSchema(snapType, (String) name);
599                             child.setRequired(required);
600                             requestBody.addChild(child);
601                         }
602                     }
603                 }
604             }
605         }
606         return requestBody;
607     }

609     /**
610      * Builds input schema by traversing OpenAPI spec.
611      *
612      * @param schemaProvider - schema provider object to build schema tree
613      * @param name           - schema name
614      * @param required       - true if it is a required field
615      * @param ref            - value of '#ref' key in the spec
616      * @param refChain       - chain of references in recursive calls, needed to avoid stack
617      *                      overflow, format: |ref1||ref2||re3||...||refN|
618      *
619      * @return schema object
620      * @throws InvalidPathException if it fails to build schema
621      */
622     private ObjectSchema createSchemaFromRef(SchemaProvider schemaProvider, String name,
623             boolean required, String ref, String refChain)
624             throws InvalidPathException {
625         // create object schema
626         ObjectSchema schema = schemaProvider.createSchema(SnapType.COMPOSITE, name);
627         schema.setRequired(required);
628         return createChildSchemaFromRef(schemaProvider, schema, ref, refChain);
629     }

631     private ObjectSchema createChildSchemaFromRef(SchemaProvider schemaProvider,
632             ObjectSchema schema, String ref, String refChain) throws InvalidPathException {
633         // get schema spec by using #ref from OpenAPI spec
634         ref = convertRefToJsonPath(ref);
635         Map<String, Object> refSpec = getMapAt(resolvedV3OpenApiSpec, ref);
636         String refName = ref.substring(ref.lastIndexOf('.') + 1);
637         String refNameWrapped = String.format(FORMAT_REF_NAME, refName);
638         referenceMap.put(refName, refSpec);
639         // If a loop is found in the ref chain, it will end recursive call.
640         boolean isLoop = StringUtils.contains(refChain, refNameWrapped);
641         refChain = StringUtils.join(refChain, refNameWrapped);
642         List<String> requiredFields = refSpec.containsKey(REQUIRED) ?
643                 (List<String>) refSpec.get(REQUIRED) : new ArrayList<>(0);
644         Map<String, Object> propertiesSpec =
645                 (Map<String, Object>) refSpec.get(PROPERTIES);
646         if (propertiesSpec != null) {
647             return createSchemaFromSpec(schemaProvider, propertiesSpec, requiredFields, schema,
648                     isLoop, refChain);
649         } else if (refSpec.get(TYPE) instanceof String) {
650             SnapType snapType = getSnapType((String) refSpec.get(TYPE));
651             return schemaProvider.createSchema(snapType, schema.getName());
652         } else if (refSpec.get(ALL_OF) instanceof List) {
653             for (Map<String, Object> spec : (List<Map<String, Object>>) refSpec.get(ALL_OF)) {
654                 if (spec.containsKey(REF) && spec.get(REF) != null) {
655                         createChildSchemaFromRef(schemaProvider, schema, (String) spec.get(REF),
656                                 refChain);
657                 } else if (spec.containsKey(PROPERTIES)) {
658                     Object value = spec.get(REQUIRED);
659                     if (value instanceof List) {
660                         requiredFields.addAll((List<String>) value);
661                     }
662                     createSchemaFromSpec(schemaProvider, (Map<String, Object>) spec.get(PROPERTIES),
663                             requiredFields, schema, isLoop, refChain);
664                 }
665             }
666             return schema;
667         }
668         return null;
669     }

671     private ObjectSchema createSchemaFromSpec(SchemaProvider schemaProvider,
672             Map<String, Object> spec, List<String> requiredFields, ObjectSchema schema,
673             boolean isLoop, String refChain) throws InvalidPathException {
674         for (Map.Entry<String, Object> entry : spec.entrySet()) {
675             String name = entry.getKey();
676             boolean required = requiredFields.contains(name);
677             Map<String, Object> value = (Map<String, Object>) entry.getValue();
678             ObjectSchema child = null;
679             Object refValue = value.get(REF);
680             if (refValue == null) {
681                 SnapType snapType = getSnapType((String) value.get(TYPE));
682                 child = schemaProvider.createSchema(snapType, name);
683                 child.setRequired(required);
684                 if (snapType.equals(SnapType.TABLE)) {
685                     Map<String, Object> itemsSpec = (Map<String, Object>) value.get(ITEMS);
686                     if (itemsSpec != null) {
687                         refValue = itemsSpec.get(REF);
688                     }
689                     Schema grandchild = null;
690                     if (refValue == null) {
691                         String type = (String) itemsSpec.get(TYPE);
692                         grandchild = schemaProvider.createSchema(getSnapType(type), name);
693                     } else if (!isLoop) {
694                         grandchild = createSchemaFromRef(schemaProvider, name, required,
695                                 (String) refValue, refChain);
696                     }
697                     if (grandchild != null) {
698                         child.addChild(grandchild);
699                     }
700                 }
701             } else {
702                 if (!isLoop) {
703                     child = createSchemaFromRef(schemaProvider, name, required,
704                             (String) refValue, refChain);
705                 }
706             }
707             if (child != null) {
708                 schema.addChild(child);
709             }
710         }
711         return schema;
712     }

840     /**
841      * Prepares the path and queries of service URL by processing the parameter part of OpenAPI
842      * spec and input data.
843      *
844      * @param operationSpec - part of the OpenAPI spec for selected path and operation
845      * @param data          - input data
846      * @param serviceUrl    - service URL
847      * @param headers       - reuqest headers
848      *
849      * @return updated service URL
850      * @throws InvalidPathException if it fails in handling JSON paths
851      *         URISyntaxException   if there is any syntax error in URL
852      */
853     public String processParameters(Map<String, Object> operationSpec, Map<String, Object> data,
854             String serviceUrl, List<Header> headers)
855             throws InvalidPathException, URISyntaxException {
856         if (operationSpec.containsKey(PARAMETERS)) {
857             List<Map<String, Object>> params = buildJsonPath(PARAMETERS).read(operationSpec);
858             Map<String, String> pathParams = new LinkedHashMap<>();
859             Map<String, String> queryParams = new LinkedHashMap<>();
860             Map<String, String> headerParams = new LinkedHashMap<>();
861             for (Map<String, Object> parameter : params) {
862                 if (parameter.containsKey(REF)) {
863                     // it is possible the parameter spec has get from $ref
864                     parameter = getRefSpec((String) parameter.get(REF), false);
865                     if (MapUtils.isEmpty(parameter)){
866                         parameter = getRefSpec((String) parameter.get(REF), true);
867                     }
868                 }
869                 String paramName = (String) parameter.get(NAME);
870                 String in = (String) parameter.get(IN);
871                 boolean required = getBoolean(parameter.get(REQUIRED));
872                 JsonPath jsonPath = buildJsonPath(PARAMETERS, in, paramName);
873                 if (BODY.equals(in) || FORM_DATA.equals(in) ||
874                         !required && !jsonPath.existsIn(data)) {
875                     // if 'in' is "body", it is OpenAPI 2.0 and should skip
876                     continue;
877                 }
878                 String value = getValueFromInput(data, jsonPath, required);
879                 if (PATH.equalsIgnoreCase(in)) {
880                     pathParams.put(paramName, value);
881                 } else if (QUERY.equalsIgnoreCase(in)) {
882                     queryParams.put(paramName, value);
883                 } else if (HEADER.equalsIgnoreCase(in)) {
884                     headerParams.put(paramName, value);
885                 }
886             }
887             if (MapUtils.isNotEmpty(pathParams)) {
888                 serviceUrl = evalPathParameters(serviceUrl, pathParams);
889             }
890             if (MapUtils.isNotEmpty(queryParams)) {
891                 URIBuilder builder = new URIBuilder(serviceUrl);
892                 for (Map.Entry<String, String> entry : queryParams.entrySet()) {
893                     builder.addParameter(entry.getKey(), entry.getValue());
894                 }
895                 serviceUrl = builder.build()
896                         .toString();
897             }
898             if (MapUtils.isNotEmpty(headerParams)) {
899                 for (Map.Entry<String, String> entry : headerParams.entrySet()) {
900                     headers.add(new BasicHeader(entry.getKey(), entry.getValue()));
901                 }
902             }
903         }
904         return serviceUrl;
905     }

907     /**
908      * Prepares the path and queries of service URL by processing the parameter part of OpenAPI
909      * spec and input data.
910      *
911      * @param operationSpec    - part of the OpenAPI spec for selected path and operation
912      * @param data             - input data
913      * @param serviceUrl       - service URL
914      * @param headers          - reuqest headers
915      * @param queryParams      - query params acquired from request
916      * @param commonParameters - param names common for spec
917      *
918      * @return updated service URL
919      * @throws InvalidPathException if it fails in handling JSON paths
920      */
921     public String processParameters(Map<String, Object> operationSpec, Map<String, Object> data,
922             String serviceUrl, List<Header> headers, Map<String, String> queryParams,
923             List<Map<String, Object>> commonParameters) throws InvalidPathException {
924         // combine common parameter spec and the operation parameter spec
925         Object paramSpec = readJsonPath(operationSpec, PARAMETERS);
926         List<Object> parameters = union(paramSpec, commonParameters);
927         // process schema for parameters in path, query and header
928         Map<String, String> pathParams = new LinkedHashMap<>();
929         Map<String, String> headerParams = new LinkedHashMap<>();
930         for (Object object : parameters) {
931             Map<String, Object> parameter;
932             if (object instanceof Map) {
933                 parameter = (Map<String, Object>) object;
934             } else {
935                 continue;
936             }
937             if (parameter.containsKey(REF)) {
938                 // it is possible the parameter spec has get from $ref
939                 parameter = getRefSpec((String) parameter.get(REF), false);
940                 if (MapUtils.isEmpty(parameter)){
941                     parameter = getRefSpec((String) parameter.get(REF), true);
942                 }
943             }
944             String paramName = (String) parameter.get(NAME);
945             String in = (String) parameter.get(IN);
946             if (AUTHORIZATION.equalsIgnoreCase(paramName) && HEADER.equalsIgnoreCase(in)) {
947                 // Authorization header is not processed from the input data. It is taken
948                 // care of the Snap pack account or 'HTTP Headers' property.
949                 continue;
950             }
951             boolean required = getBoolean(parameter.get(REQUIRED));
952             JsonPath jsonPath = buildJsonPath(PARAMETERS, in, paramName);
953             if (BODY.equals(in) || FORM_DATA.equals(in) ||
954                     !required && !jsonPath.existsIn(data)) {
955                 // if 'in' is "body", it is OpenAPI 2.0 and should skip
956                 continue;
957             }
958             String value = getValueFromInput(data, jsonPath, required);
959             if (PATH.equalsIgnoreCase(in)) {
960                 pathParams.put(paramName, value);
961             } else if (QUERY.equalsIgnoreCase(in)) {
962                 // the query parameters from Snap property takes higher priority over the
963                 // parameters automatically-generated from spec & input data
964                 if (!queryParams.containsKey(paramName)) {
965                     queryParams.put(paramName, value);
966                 }
967             } else if (HEADER.equalsIgnoreCase(in)) {
968                 headerParams.put(paramName, value);
969             }
970         }
971         if (MapUtils.isNotEmpty(pathParams)) {
972             serviceUrl = evalPathParameters(serviceUrl, pathParams);
973         }
974         if (MapUtils.isNotEmpty(headerParams)) {
975             for (Map.Entry<String, String> entry : headerParams.entrySet()) {
976                 String name = entry.getKey();
977                 // the headers from Snap property take higher priority over the
978                 // headers automatically-generated from spec & input data
979                 if (!containsHeader(headers, name)) {
980                     headers.add(new BasicHeader(name, entry.getValue()));
981                 }
982             }
983         }
984         return serviceUrl;
985     }

987     private String evalPathParameters(String serviceUrl, Map<String, String> params) {
988         Set<String> patterns = new LinkedHashSet<>(params.size());
989         for (String key : params.keySet()) {
990             patterns.add(String.format(PATH_PARAM_PATTERN, key));
991         }
992         return StringUtils.replaceEach(serviceUrl, patterns.toArray(new String[0]),
993                 params.values().toArray(new String[0]));
994     }

996     /**
997      * prepares HTTP entity and request headers by processing operation spec and input data.
998      *
999      * @param operationSpec - part of the OpenAPI spec for selected path and operation
1000      * @param input         - input data
1001      * @param headers       - request headers
1002      *
1003      * @return HttpEntity object
1004      * @throws JsonProcessingException if it fails to convert input data to JSON or XML
1005      */
1006     public HttpEntity getHttpEntity(Map<String, Object> operationSpec, Object input,
1007             List<Header> headers) throws JsonProcessingException {
1008         String contentType = APPLICATION_JSON;
1009         if (operationSpec.containsKey(REQUEST_BODY)) {
1010             // get requestBody content spec
1011             Map<String, Object> contentSpec = getSpec(operationSpec, REQUEST_BODY, CONTENT);
1012             // get content type
1013             if (contentSpec.containsKey(APPLICATION_JSON)) {
1014                 contentType = APPLICATION_JSON;
1015             } else if (contentSpec.containsKey(APPLICATION_XML)) {
1016                 contentType = APPLICATION_XML;
1017             } else if (contentSpec.containsKey(APPLICATION_OCTET_STREAM)) {
1018                 contentType = APPLICATION_OCTET_STREAM;
1019             } else if (contentSpec.containsKey(MULTIPART_FORM_DATA)) {
1020                 contentType = MULTIPART_FORM_DATA;
1021             } else if (contentSpec.containsKey(APPLICATION_FORM_URLENCODED)) {
1022                 contentType = APPLICATION_FORM_URLENCODED;
1023             } else {
1024                 Set<String> types = contentSpec.keySet();
1025                 if (LOG.isWarnEnabled()) {
1026                     LOG.warn("Unsupported content type: {}",
1027                         StringUtils.join(types, COMMA_SEPARATOR));
1028                 }
1029             }
1030         } else if (operationSpec.containsKey(CONSUMES)) {
1031             // it is OpenAPI 2.0, the "consumes" field contains the expected content types
1032             Object consumes = operationSpec.get(CONSUMES);
1033             if (consumes instanceof List) {
1034                 List<String> contentTypes = (List<String>) consumes;
1035                 if (contentTypes.contains(APPLICATION_JSON)) {
1036                     contentType = APPLICATION_JSON;
1037                 } else if (contentTypes.contains(APPLICATION_XML)) {
1038                     contentType = APPLICATION_XML;
1039                 } else if (contentTypes.contains(APPLICATION_OCTET_STREAM)) {
1040                     contentType = APPLICATION_OCTET_STREAM;
1041                 } else if (contentTypes.contains(MULTIPART_FORM_DATA)) {
1042                     contentType = MULTIPART_FORM_DATA;
1043                 } else if (contentTypes.contains(APPLICATION_FORM_URLENCODED)) {
1044                     contentType = APPLICATION_FORM_URLENCODED;
1045                 } else {
1046                     if (LOG.isWarnEnabled()) {
1047                         LOG.warn("Unsupported content type: {}",
1048                             StringUtils.join(contentTypes, COMMA_SEPARATOR));
1049                     }
1050                 }
1051             }
1052         }
1053         // get content object
1054         Object content = input instanceof Map ?
1055             ((Map<String, Object>) input).get(REQUEST_BODY) : input;
1056         if (content instanceof Map &&
1057             ((Map<String, Object>) content).containsKey(CONTENT)) {
1058             content = ((Map<String, Object>) content).get(CONTENT);
1059         }
1060         if (content != null) {
1061             switch (contentType) {
1062                 case APPLICATION_JSON:
1063                     content = OBJECT_MAPPER.writeValueAsString(content);
1064                     break;
1065                 case APPLICATION_XML:
1066                     content = (new XmlMapper()).writeValueAsString(content);
1067                     break;
1068                 case MULTIPART_FORM_DATA:
1069                     content = generateMultipart(operationSpec, input, headers);
1070                     break;
1071                 case APPLICATION_FORM_URLENCODED:
1072                     content = generateFormURLEncoded(input);
1073                     break;
1074                 default:
1075                     // leave content 'as is' if none of the above
1076             }
1077             // append charset UTF-8 by default to Content-Type
1078             contentType = contentType.concat(CHARSET_UTF_8);
1079             // add content-type header
1080             if (!MULTIPART_FORM_DATA.equalsIgnoreCase(contentType) &&
1081                     !containsHeader(headers, CONTENT_TYPE)) {
1082                 headers.add(new BasicHeader(CONTENT_TYPE, contentType));
1083             }
1084         } else {
1085             return null;
1086         }
1087         // convert content to byte array
1088         byte[] bytes;
1089         if (content instanceof String) {
1090             bytes = ((String) content).getBytes(StandardCharsets.UTF_8);
1091         } else if (content instanceof byte[]) {
1092             bytes = (byte[]) content;
1093         } else if (content instanceof java.io.Serializable) {
1094             bytes = SerializationUtils.serialize((Serializable) content);
1095         } else {
1096             throw new SnapDataException(ERR_INVALID_CONTENT_TYPE)
1097                 .formatWith(content.getClass()
1098                     .getSimpleName())
1099                 .withReason(REASON_NOT_SERIALIZABLE)
1100                 .withResolution(RESOLUTION_BYTE_ARRAY);
1101         }
