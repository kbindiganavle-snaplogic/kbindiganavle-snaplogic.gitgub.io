289     protected void process(Document document, String inputViewName) {
290         Object objDir = dirProp.eval(document);
291         String dir;
292         if (!(objDir instanceof String) || StringUtils.isEmpty((String) objDir)) {
293             String filePath = fileProp.eval(document);
294             if (filePath != null && UrlUtils.getProtocolPattern().matcher(filePath).find()) {
295                 // Full path is defined in the "File" field
296                 dir = StringUtils.EMPTY;
297             } else {
298                 dir = DEFAULT_DIR;
299             }
300         } else {
301             dir = (String) objDir;
302             if (!dir.endsWith(URL_SEPARATOR)) {
303                 dir = dir.concat(URL_SEPARATOR);
304             }
305         }
306         if (fileProp.isEmpty()) {
307             // if file prop is empty, read all files matching the glob filter under the directory
308             // get FileSystem & Path objects
309             String dirWithUserInfo = urlUtils.insertAuthority(dir, accountType, userInfo,
310                     UrlUtils.getProtocolPattern(), abfsRestUtility);
311             URI pathUri = URI.create(dirWithUserInfo);
312             if (StringUtils.isBlank(pathUri.getScheme())) {
313                 SnapDataException ex = new SnapDataException(ERR_INVALID_DIR)
314                         .formatWith(dir)
315                         .withReason(REASON_EMPTY_PROTOCOL)
316                         .withResolution(RESOLUTION_INVALID_DIR);
317                 errorViews.write(ex, document);
318                 return;
319             }
320             FileSystem fileSystem = null;
321             if (StringUtils.startsWithIgnoreCase(dir, PROTOCOL_ABFS)) {
322                 dirWithUserInfo = dir;
323             } else {
324                 fileSystem = getFileSystem(pathUri);
325                 if (fileSystem == null) {
326                     SnapDataException ex = new SnapDataException(ERR_DIR_NOT_FOUND).formatWith(dir)
327                             .withReason(REASON_DIR_NOT_FOUND)
328                             .withResolution(RESOLUTION_DIR_NOT_FOUND);
329                     errorViews.write(ex, document);
330                     return;
331                 }
332             }
333             try {
334                 // get path matcher
335                 String pattern;
336                 if (UrlUtils.URL_PROTOCOL_FILE.equals(pathUri.getScheme()) ||
337                         UrlUtils.URL_PROTOCOL_HDFS.equals(pathUri.getScheme())) {
338                     pattern = StringUtils.join(GLOB_PATTERN, pathUri.getPath(), filter);
339                 } else {
340                     pattern = StringUtils.join(GLOB_PATTERN, dirWithUserInfo, filter);
341                 }
342                 PathMatcher matcher = FileSystems.getDefault().getPathMatcher(pattern);
343                 Set<String> files;
344                 // find all files under the current directory
345                 if (dir.startsWith(UrlUtils.URL_PROTOCOL_S3)) {
346                     files = findS3Files(dirWithUserInfo, fileSystem, matcher, false, true);
347                 } else if (StringUtils.startsWithIgnoreCase(dir, PROTOCOL_ABFS)) {
348                     initializeAbfsCallableRestUtility();
349                     List<String> filesFound = abfsCallableRestUtility.executeListFiles(1, 0, dir,
350                             true, filter, true, document);
351                     files = new HashSet<>(filesFound);
352 
353                 } else if (dir.startsWith(UrlUtils.URL_PROTOCOL_HDFS)) {
354                     files = findHadoopFiles(fileSystem, pathUri, matcher, false, true,true);
355                 } else {
356                     Path dirPath = FileSystemUtils.createPath(fileSystem, pathUri, false);
357                     files = findFiles(dirPath, matcher, false, filter.contains(DOUBLE_STARS));
358                     // "file://" should be prepended for local files
359                     if (dir.startsWith(UrlUtils.URL_PROTOCOL_FILE)) {
360                         Set<String> newFiles = Sets.newHashSetWithExpectedSize(files.size());
361                         for (String file : files) {
362                             file = String.format(FORMAT_PATH, UrlUtils.URL_PROTOCOL_FILE, file);
363                             newFiles.add(file);
364                         }
365                         files = newFiles;
366                     }
367                 }
368                 // read files
369                 for (String file : files) {
370                     readFile(document, file);
371                 }
372             } catch (SnapDataException e) {
373                 errorViews.write(e, document);
374             } catch (UnsupportedEncodingException e) {
375                 SnapDataException ex = new SnapDataException(e, ERROR_INVALID_ENCODING)
376                         .withReason(e.getMessage())
377                         .withResolution(RESOLUTION_ADDRESS_ISSUE);
378                 errorViews.write(ex, document);
379             } catch (IOException e) {
380                 SnapDataException ex = new SnapDataException(e, IO_ERROR)
381                         .withReason(e.getMessage())
382                         .withResolution(RESOLUTION_ADDRESS_ISSUE);
383                 errorViews.write(ex, document);
384             } finally {
385                 FileSystemUtils.closeFileSystem(fileSystem, dir);
386             }
387         } else {
388             // read a single file
389             String file = fileProp.eval(document);
390             if (file != null) {
391                 readFile(document, dir.concat(file));
392             } else {
393                 SnapDataException ex = new SnapDataException(ERR_FILE_NOT_FOUND)
394                         .withReason(REASON_FILEEXP_NULL)
395                         .withResolution(RESOLUTION_FILEEXP_NULL);
396                 errorViews.write(ex, document);
397             }
398         }

419     protected void readFile(final Document document, final String filename) {
420         // insert username and password into filename
421         actualFilename = filename;
422         LOGGER.debug(FILENAME_LOG, filename);
423         filePathURI = urlUtils.insertAuthority(filename, accountType, userInfo,
424                 UrlUtils.getProtocolPattern(), abfsRestUtility);
425         boolean isDebugEnabled = LOGGER.isDebugEnabled();
426         if (isDebugEnabled) {
427             String logFilePathURI = obfuscateCredentialsFromURI(filePathURI);
428             LOGGER.debug(FILEPATHURI_LOG, logFilePathURI);
429         }
430         URI pathUri = URI.create(filePathURI); // filePathURI is encoded
431         if (isDebugEnabled) {
432             String logPathUri = obfuscateCredentialsFromURI(pathUri.toString());
433             LOGGER.debug(PATHURI_LOG, logPathUri);
434         }
435 
436         // perform the file read with retries
437         try {
438             ContextualRunnable runnable = context ->  doRead(document, filename, pathUri);
439             CheckedConsumer<Throwable> failedAttemptListener = e -> LOGGER.warn(
440                     "{} - Failed to read file: {}, {}",
441                     HADOOP_RETRY_SIGNATURE, filename, e.getMessage());
442             failsafeExecute(runnable, failedAttemptListener, retries, retryInterval);
443         } catch (Exception e) {
444             Throwable ex = e instanceof FailsafeException ? e.getCause() : e;
445             if (ex instanceof SuggestViewAbortException) {
446                 throw (SuggestViewAbortException) ex;
447             } else if (ex instanceof SnapDataException) {
448                 errorViews.write((SnapDataException) ex, document);
449             } else if (ex instanceof ExecutionException) {
450                 errorViews.write((SnapDataException) new SnapDataException(ex, ex.getMessage())
451                         .withReason(((SnapException) ex).getReason())
452                         .withResolution(((SnapException) ex).getResolution()), document);
453             } else {
454                 handleException(e, ERR_CANT_READ, filename, e.getMessage(),
455                         RESOLUTION_CANT_ACCESS, documentUtility, errorViews, document);
456             }
457         }
458     }

461         StringBuilder result = new StringBuilder();
462         if (StringUtils.isNotEmpty(filePathURI) && filePathURI.contains("@")

464             String[] splits = filePathURI.split("@",2);
465             String[] splits1 = splits[0].split("://",2);
466             if (splits1[1].contains(":")) {
467                 String[] splits2 = splits1[1].split(":");
468                 if (splits2.length == 3) {
469                     result.append(splits1[0]).append("://").append(OBFUSCATED_CREDS)
470                             .append(":").append(OBFUSCATED_CREDS)
471                             .append(":").append(OBFUSCATED_CREDS)
472                             .append("@").append(splits[1]);
473                 } else if (splits2.length == 2) {
474                     result.append(splits1[0]).append("://").append(OBFUSCATED_CREDS)
475                             .append(":").append(OBFUSCATED_CREDS)

484     protected void doRead(Document document, final String filename, URI pathUri) throws Exception{
485         URLConnection urlConnection = null;
486         try {
487             statusReporter.notify(OPEN_FILE, filename);
488             urlConnection = createUrlConnection(pathUri);
489             setFilesystemProperties(urlConnection, account, retries, retryInterval, null);
490             if (StringUtils.startsWith(filename, PROTOCOL_ABFS)) {
491                 initializeAbfsCallableRestUtility();
492                 String token = abfsCallableRestUtility.getAccessToken();
493                 urlConnection.setRequestProperty(AUTHORIZATION_HEADER, String.format(
494                         AUTHORIZATION_HEADER_VALUE, token));
495             }
496             // connect to URL
497             urlConnection.connect();
498             statusReporter.notify(READ_FILE, filename);
499             LOGGER.debug(DEBUG_URL_CONNECTED, filename);
500             // get input stream
501             GetInputStream getInputStream = new GetInputStream(urlConnection);
502             ExecutorService executor = Executors.newSingleThreadExecutor();
503             Future<InputStream> submitted = executor.submit(getInputStream);
504             try (final InputStream inputStream =
505                          submitted.get(CONNECTION_TIMEOUT, TimeUnit.SECONDS)) {
506                 LOGGER.debug(DEBUG_INPUTSTREAM_OPEN, filename);
507 
508                 final long contentLength = urlConnection.getContentLengthLong();
509                 if (contentLength >= 0) {
510                     statusReporter.begin(contentLength, "bytes");
511                 } else {
512                     if (inputStream.available() >= 0) {
513                         statusReporter.begin(inputStream.available(), "bytes");
514                     } else {
515                         statusReporter.beginUnknown();
516                     }
517                 }
518 
519                 try {
520                     final Document header = documentUtility.newDocumentFor(document,
521                             createHeader(pathUri, urlConnection.getHeaderFields()));
522                     outputViews.write(new BinaryOutput() {
523                         @Override
524                         public Document getHeader() {
525                             return header;
526                         }
527 
528                         @Override
529                         public void write(final WritableByteChannel writeChannel)
530                                 throws IOException {
531                             // copy data to the output
532                             OutputStream outputStream = Channels.newOutputStream(writeChannel);
533                             long limit = maxSuggestValue == null ? Long.MAX_VALUE :
534                                     maxSuggestValue.longValue();
535                             StatusReportingUtils.copy(statusReporter, inputStream, outputStream,
536                                     limit);
537                             LOGGER.debug(DEBUG_COPY_IO_STREAM, filename);
538                         }
539                     });
540                 } catch (Exception e) {
541                     if (urlConnection instanceof S3UrlConnection) {
542                         // S3 InputStream does not get closed until its fully read.
543                         // Disconnect on the connection will
544                         ((S3UrlConnection) urlConnection).disconnect();
545                     }
546                     throw e;
547                 }
548             } catch (TimeoutException e) {
549                 throw new IOException(
550                         String.format(ERR_CONNECTION_TIMEOUT, CONNECTION_TIMEOUT));
551             } finally {
552                 submitted.cancel(true);
553                 executor.shutdown();
554             }
555         } catch (SuggestViewAbortException e) {
556             throw e;
557         } catch (SnapDataException e) {
558             errorViews.write(e);
559         } catch (Exception e) {
560             // check if it is caused by SuggestViewAbortException
561             if (Iterables.filter(Throwables.getCausalChain(e),
562                     SuggestViewAbortException.class).iterator().hasNext()) {
563                 return;
564             }
565            throw e;
566         } finally {
567             statusReporter.reset();
568             urlUtils.disconnect(urlConnection);
569             LOGGER.debug(DEBUG_DISCONNECTED, filename);
570         }
571     }

593     protected Map<String, Object> createHeader(
594             URI sensitiveUri, final Map<String, List<String>> headerFields) {
595         try {
596             // Rebuild the URI without the user info so we don't leak any private info.
597             URI uri = new URI(sensitiveUri.getScheme(),
598                     null,
599                     sensitiveUri.getHost(),
600                     sensitiveUri.getPort(),
601                     sensitiveUri.getPath(),
602                     sensitiveUri.getQuery(),
603                     sensitiveUri.getFragment());
604             Map<String, Object> headerMap = new LinkedHashMap<>();
605             String filename = uri.getPath();
606             // Guess the Content-Type, but allow the URLConnection to overwrite it.
607             String contentType = URLConnection.guessContentTypeFromName(filename);
608             if (contentType == null) {
609                 // XXX The URLConnection method isn't very good, it'll return null for things
610                 // like '.csv'.  Sigh...
611                 contentType = MediaType.OCTET_STREAM.toString();
612             }
613             headerMap.put(HttpHeaders.CONTENT_TYPE, contentType);
614             for (Map.Entry<String, List<String>> entry : headerFields.entrySet()) {
615                 // Skip any entry whose key is null. HttpURLConnection returns a header
616                 // representing the status line of the response, which has a null key.
617                 if (entry.getKey() != null) {
618                     switch (entry.getValue().size()) {
619                         case 0:
620                             break;
621                         case 1:
622                             headerMap.put(entry.getKey(), entry.getValue().get(0));
623                             break;
624                         default:
625                             headerMap.put(entry.getKey(), entry.getValue());
626                             break;
627                     }
628                 }
629             }
630             // Always set the Content-Location and Content-Disposition.
631             if (StringUtils.startsWithIgnoreCase(actualFilename, PROTOCOL_ABFS)) {
632                 headerMap.put(HttpHeaders.CONTENT_LOCATION, actualFilename);
633             } else {
634                 headerMap.put(HttpHeaders.CONTENT_LOCATION, UrlUtils.getFullPathNoAuth(uri));
635             }
636             // Find the base name from the filename, we'll use that as the name in the disposition.
637             int slashIndex = filename.lastIndexOf('/');
638             if (slashIndex == -1) {
639                 slashIndex = filename.lastIndexOf('\\');
640             }
641             String basename;
642             if (slashIndex != -1) {
643                 basename = filename.substring(slashIndex + 1);
644             } else {
645                 basename = filename;
646             }
647             if (!basename.isEmpty()) {
648                 // Make sure the filename in the disposition is simple and sane.
649                 headerMap.put(HttpHeaders.CONTENT_DISPOSITION, String.format(DISPOSITION_FORMAT,
650                         basename.replaceAll(DISPOSITION_FILENAME_SCRUBBER, "_")));
651             }
652             return headerMap;
653         } catch (URISyntaxException e) {
654             throw new ExecutionException(e, UNABLE_TO_REBUILD_URI)
655                     .withReason(e.getMessage())
656                     .withResolutionAsDefect();
657         }
658     }
