194      * Perform only the configuration required for Kerberos property suggestions.
195      *
196      * @param propertyValues
197      * @throws ConfigurationException
198      */
199     protected void configureForKerberosProperties(final PropertyValues propertyValues) {
200         this.userImpersonationEnabled = HdfsUtils.checkForUserImpersonationEnabled(propertyValues);
201         this.pipelineUser = BinaryUtils.getPipeLineUserName(expressionUtils, propertyValues);
202         this.useHdfs = HdfsUtils.checkForHdfs(propertyValues);
203         if (account instanceof HadoopFSAccount) {
204             ((HadoopClientConfig) configurationSupplier).setConfigDirectoryPath(
205                     ((HadoopClientConfigProvider) account).getConfigDirectoryPath());
206             this.hadoopConfig = ((HadoopFSAccount) account).getConfiguration();
207         } else {
208             this.hadoopConfig = configurationSupplier.get();
209         }
210         this.useKerberos = KerberosUtilities.isAuthenticatedWithKerberos(useKerberos, hadoopConfig);
211         switch (KerberosUtilities.getSourceOfKerberosCredentials(propertyValues)) {
212             case ACCOUNT:
213                 if (useHdfs) {
214                     // ensure configured authentication and account match.
215                     HdfsUtils.checkAccountMatchesUrl(account, useKerberos, hadoopConfig);
216                 }
217                 break;
218             case PLATFORM:
219                 // nothing required
220                 break;
221         }
222         kerberosConfig = HdfsUtils.getEffectiveKerberosConfig(propertyValues, account, useKerberos,
223                 userImpersonationEnabled, pipelineUser);
224         // we should always have a default login user (snaplogic: SYSTEM).
225         try {
226             user = KerberosUtilities.getEffectiveUser(kerberosConfig);
227         } catch (IOException | LoginException e) {
228             throw new ConfigurationException(e, ERR_UNABLE_TO_VALIDATE_USER_CREDENTIALS_MSG)
229                     .withReason(ERR_UNABLE_TO_VALIDATE_USER_CREDENTIALS_REASON)
230                     .withResolution(ERR_UNABLE_TO_VALIDATE_USER_CREDENTIALS_RESOLUTION);
231         }
232         HdfsUtils.setDefaultHandler(propertyValues, configurationSupplier, useHdfs, user);
233     }

260     /**
261      * @see AbstractSimpleDelete#process(Document, String)
262      */
263     @Override
264     protected void process(Document document, String inputViewName) {
265         this.useHdfs = HdfsUtils.checkForHdfs(dirProp, fileProp, document);
266         if (useHdfs) {
267             user.doAs((PrivilegedAction<Void>) () -> {
268                 super.process(document, inputViewName);
269                 return null;
270             });
271             return;
272         }
273         String directoryPath = dirProp.eval(document);
274         if (StringUtils.isEmpty(directoryPath)) {
275             String filePath = fileProp.eval(document);
276             if (!StringUtils.isEmpty(filePath) && UrlUtils.getProtocolPattern().matcher(filePath).find()) {
277                 directoryPath = FilenameUtils.getFullPath(filePath);
278             } else {
279                 writeErrorDocToErrorView(document, directoryPath, REASON_EMPTY_PROPERTY);
280                 return;
281             }
282         } else {
283             if (!StringUtils.endsWith(directoryPath, SEPARATOR)) {
284                 directoryPath = directoryPath.concat(SEPARATOR);
285             }
286         }
287         String dirWithUserInfo = urlUtils.insertAuthority(directoryPath, accountType, userInfo,
288                 UrlUtils.getProtocolPattern(), abfsRestUtility);
289         java.net.URI pathUri = java.net.URI.create(dirWithUserInfo);
290         if (StringUtils.isBlank(pathUri.getScheme())) {
291             writeErrorDocToErrorView(document, directoryPath, REASON_EMPTY_PROTOCOL);
292             return;
293         }
294         super.process(document, inputViewName);
295     }

316     public String getCurrentUser() throws IOException {
317         String currentUser = "";
318         // SNAP is authenticated with Kerberos
319         if (isAuthenticatedWithKerberos()) {
320             // Get the User name part from the client principal
321             String clientPrincipal = HdfsUtils.getPrimaryFromPrincipal(
322                     kerberosConfig.getClientPrincipal());
323             currentUser = (userImpersonationEnabled) ?
324                     pipelineUser :
325                     clientPrincipal;
326         } else {
327             // Non Kerberos, but impersonation can be enabled.
328             currentUser = (userImpersonationEnabled) ?
329                     pipelineUser :
330                     UserGroupInformation.getCurrentUser().getUserName();
331         }
332         LOG.debug("Current User Name is: {}", currentUser);
333         return currentUser;
334     }
