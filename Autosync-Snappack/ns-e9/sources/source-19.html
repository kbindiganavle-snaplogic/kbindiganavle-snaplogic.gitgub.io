259                         .withReason(INVALID_PATH_FOR_COLUMN_NAME_PROPERTY)

442      * Converts hive lazy objects to java objects.
443      *
444      * @param value
445      * @return convertedObject
446      */
447     public static Object typeCast(Object value) {
448         Object result = null;
449 
450         if (value instanceof Writable) {
451             if (value instanceof IntWritable) {
452                 result = ((IntWritable) value).get();
453             } else if (value instanceof ShortWritable) {
454                 result = ((ShortWritable) value).get();
455             } else if (value instanceof LongWritable) {
456                 result = ((LongWritable) value).get();
457             } else if (value instanceof FloatWritable) {
458                 result = ((FloatWritable) value).get();
459             } else if (value instanceof DoubleWritable) {
460                 result = ((DoubleWritable) value).get();
461             } else if (value instanceof ByteWritable) {
462                 result = ((ByteWritable) value).get();
463             } else if (value instanceof BytesWritable) {
464                 result = ((BytesWritable) value).getBytes();
465             } else if (value instanceof BooleanWritable) {
466                 result = ((BooleanWritable) value).get();
467             } else if (value instanceof Text) {
468                 result = value.toString();
469             } else if (value instanceof DateWritable) {
470                 result = ((DateWritable) value).get();
471             } else if (value instanceof TimestampWritable) {
472                 result = ((TimestampWritable) value).getTimestamp();
473             } else if (value instanceof ObjectWritable) {
474                 result = ((ObjectWritable) value).get();
475             } else if (value instanceof MapWritable) {
476                 MapWritable map = (MapWritable) value;
477                 // Lets convert each of the object inside the map into snaplogic data types
478                 Map<Object, Object> convertedMap = new HashMap<>(map.size());
479                 for (Map.Entry<Writable, Writable> entry : map.entrySet()) {
480                     convertedMap.put(typeCast(entry.getKey()), typeCast(entry.getValue()));
481                 }
482                 result = convertedMap;
483             } else if (value instanceof ArrayWritable) {
484                 return ((ArrayWritable) value).toArray();
485             } else if (value != null) {
486                 // We just pass through the values as this is unrecognized data type.
487                 result = value;
488             }
489         } else if (value instanceof LazyObject) {
490             if (value instanceof LazyString) {
491                 result = ((LazyString) value).getWritableObject().toString();
492             } else if (value instanceof LazyInteger) {
493                 result = BigInteger.valueOf(((LazyInteger) value).getWritableObject().get());
494             } else if (value instanceof LazyLong) {
495                 result = BigInteger.valueOf(((LazyLong) value).getWritableObject().get());
496             } else if (value instanceof LazyFloat) {
497                 result = BigDecimal.valueOf(((LazyFloat) value).getWritableObject().get());
498             } else if (value instanceof LazyDouble) {
499                 result = BigDecimal.valueOf(((LazyDouble) value).getWritableObject().get());
500             } else if (value instanceof LazyBoolean) {
501                 result = Boolean.valueOf(((LazyBoolean) value).getWritableObject().get());
502             } else if (value instanceof LazyByte) {
503                 result = Byte.valueOf(((LazyByte) value).getWritableObject().get());
504             } else if (value instanceof LazyShort) {
505                 result = BigInteger.valueOf(((LazyShort) value).getWritableObject().get());
506             } else if (value instanceof LazyArray) {
507                 List<Object> list = ((LazyArray) value).getList();
508                 // Lets convert each of the object inside the array into snaplogic data types
509                 List<Object> convertedList = new ArrayList<>(list.size());
510                 for (Object object : list) {
511                     convertedList.add(typeCast(object));
512                 }
513                 result = convertedList;
514             } else if (value instanceof LazyBinary) {
515                 result = ((LazyBinary) value).getWritableObject().getBytes();
516             } else if (value instanceof LazyDate) {
517                 result = new DateTime(((LazyDate) value).getWritableObject().get());
518             } else if (value instanceof LazyMap) {
519                 Map<Object, Object> map = ((LazyMap) value).getMap();
520                 // Lets convert each of the object inside the map into snaplogic data types
521                 Map<Object, Object> convertedMap = new HashMap<>(map.size());
522                 for (Map.Entry<Object, Object> entry : map.entrySet()) {
523                     convertedMap.put(typeCast(entry.getKey()), typeCast(entry.getValue()));
524                 }
525                 result = convertedMap;
526             } else if (value instanceof LazyStruct) {
527                 result = ((LazyStruct) value).getObject();
528             } else if (value instanceof LazyUnion) {
529                 result = ((LazyUnion) value).getField();
530             } else if (value instanceof LazyTimestamp) {
531                 result = new DateTime(((LazyTimestamp) value).getWritableObject().getTimestamp()
532                         .getTime());
533             } else if (value instanceof LazyObject) {
534                 result = ((LazyObject) value).getObject();
535             } else if (value != null) {
536                 // We just pass through the values as this is unrecognized data type.
537                 result = value;
538             }
539         } else if (value != null) {
540             result = value;
541         }
542 
543         return result;
544     }

628         Configuration configuration = new Configuration();
629         // Lets disable the cache, so that each run (with potentially new URI will get new client)
630         configuration.setBoolean(HCatConstants.HCAT_HIVE_CLIENT_DISABLE_CACHE, true);
631         // Metastore uris - can be comma separated list of uris.

686     private static void addMetastorePropsTo(final PropertyBuilder propBuilder,
687             String dbDescription, String tableDescription) {
688         propBuilder.describe(PROP_HIVE_METASTORE, LBL_HIVE_METASTORE, DESC_HIVE_METASTORE)
689                 .expression(SnapProperty.DecoratorType.ACCEPTS_SCHEMA)
690                 .optional()
691                 .add();
692         // When the Hcatalog server and user properties are configured,
693         // we can introspect and provide more details about the catalog.
694         propBuilder.describe(PROP_DATABASE, LBL_DATABASE, dbDescription)
695                 .withSuggestions(new Suggestions() {
696                     @Override
697                     public void suggest(SuggestionBuilder suggestionBuilder,
698                             PropertyValues propertyValues) {
699                         try {
700                             HCatClient hCatClient = getHCatClient(propertyValues);
701                             if (hCatClient != null) {
702                                 List<String> databaseNames = hCatClient
703                                         .listDatabaseNamesByPattern(STAR);
704                                 suggestionBuilder.node(PROP_DATABASE)
705                                         .suggestions(databaseNames.toArray(new String[0]));
706                             } else {
707                                 suggestionBuilder.node(PROP_DATABASE)
708                                         .suggestions(new String[]{HIVE_METASTORE_URI_NOT_PROVIDED});
709                             }
710                         } catch (Exception e) {
711                             // Unable to use hive metastore client. Log the error.
712                             LOG.error(UNABLE_TO_CONNECT_TO_HIVE_METASTORE, e);
713                             String error = String.format(ERROR_TEMPLATE,
714                                     UNABLE_TO_CONNECT_TO_HIVE_METASTORE, e.getMessage());
715                             suggestionBuilder.node(PROP_DATABASE)
716                                     .suggestions(new String[]{error});
717                         }
718                     }
719                 }).add();
720         propBuilder.describe(PROP_TABLE, LBL_TABLE, tableDescription)
721                 .withSuggestions(new Suggestions() {
722                     @Override
723                     public void suggest(SuggestionBuilder suggestionBuilder,
724                             PropertyValues propertyValues) {
725                         try {
726                             HCatClient hCatClient = getHCatClient(propertyValues);
727                             if (hCatClient != null) {
728                                 String database = propertyValues.get(PROP_DATABASE);
729                                 if (StringUtils.isNotBlank(database)) {
730                                     List<String> tableNames = hCatClient.listTableNamesByPattern(
731                                             database, STAR);
732                                     if (tableNames != null && !tableNames.isEmpty()) {
733                                         suggestionBuilder.node(PROP_TABLE)
734                                                 .suggestions(tableNames.toArray(new String[0]));
735                                     } else {
736                                         if (tableNames == null) {
737                                             suggestionBuilder.node(PROP_TABLE)
738                                                     .suggestions(new String[]{NO_DATABASE_RESPONSE});
739                                         } else {
740                                             suggestionBuilder.node(PROP_TABLE)
741                                                     .suggestions(new String[]{NO_TABLES_ERROR});
742                                         }
743                                     }
744                                 } else {
745                                     suggestionBuilder.node(PROP_TABLE)
746                                             .suggestions(new String[]{NO_DATABASE_SELECTED});
747                                 }
748                             } else {
749                                 suggestionBuilder.node(PROP_TABLE)
750                                         .suggestions(new String[]{HIVE_METASTORE_URI_NOT_PROVIDED});
751                             }
752                         } catch (Exception e) {
753                             // Unable to use hive metastore client. Log the error.
754                             LOG.error(UNABLE_TO_CONNECT_TO_HIVE_METASTORE, e);
755                             String error = String.format(ERROR_TEMPLATE,
756                                     UNABLE_TO_CONNECT_TO_HIVE_METASTORE, e.getMessage());
757                             suggestionBuilder.node(PROP_TABLE)
758                                     .suggestions(new String[]{error});

760                     }
