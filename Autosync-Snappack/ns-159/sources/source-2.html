253     protected void invokeBcpCommand() {
254         StringBuilder output = new StringBuilder();
255         try {
256             Runtime runtime = Runtime.getRuntime();
257             Process process = runtime.exec(String.format(EXECUTE_COMMAND,
258                     IS_WIN_PLEX ? BAT_COMMAND : SH_COMMAND, scriptFileName));
259             BufferedReader reader = new BufferedReader(new InputStreamReader(
260                     process.getInputStream()));
261             String line;
262             int count = 2;
263             boolean isNotFirstLine = false;
264             while ((line = reader.readLine()) != null) {
265                 if (IS_WIN_PLEX && count-- > 0) {
266                     continue;
267                 }
268                 if (isNotFirstLine) {
269                     output.append(TAB);
270                 } else {
271                     isNotFirstLine = true;
272                 }
273                 output.append(line);
274             }
275             reader.close();
276 
277             // Mask a password from the log info it is there
278             String logInfo = getMaskedInfo(output.toString());
279 
280             reader = new BufferedReader(new InputStreamReader(process.getErrorStream()));
281             output = new StringBuilder();
282             isNotFirstLine = false;
283             while ((line = reader.readLine()) != null) {
284                 if (isNotFirstLine) {
285                     output.append(TAB);
286                 } else {
287                     isNotFirstLine = true;
288                 }
289                 output.append(line);
290             }
291             reader.close();
292             String errorInfo = output.toString();
293 
294             File errorFile = new File(errorFileName);
295             if (errorFile.exists()) {
296                 reader = new BufferedReader(new FileReader(errorFile));
297                 isNotFirstLine = false;
298                 Pattern errorRecordPattern = Pattern.compile(getErrorRecordRegExp());
299                 String errorMessage = null;
300                 String errorRecord = null;
301                 while ((line = reader.readLine()) != null) {
302                     if (errorRecordPattern.matcher(line).matches()) {
303                         if (isNotFirstLine) {
304                             writeToErrorView(errorMessage, errorRecord);
305                             errorMessage = line;
306                         } else {
307                             isNotFirstLine = true;
308                             errorMessage = line;
309                         }
310                     } else {
311                         errorRecord = line;
312                     }
313                 }
314                 reader.close();
315                 if (errorMessage != null && errorRecord != null) {
316                     // Error file is not empty, handle the last invalid input record in error file
317                     writeToErrorView(errorMessage, errorRecord);
318                 }
319             }
320             // NOTE: Before invoking exitValue() on Process object, its recommended to invoke
321             // waitFor() first. REF follow link for more details.
322             // http://stackoverflow.com/questions/25979336/handling-the-illegalthreadstateexception
323             process.waitFor();
324             if (isErrorOccurred || StringUtils.equalsIgnoreCase(logInfo,
325                     DATABASE_NAME_OPTION_IS_NOT_SUPPORTED_ERROR_MSG)) {
326                 SnapDataException dataException = new SnapDataException(
327                         ERR_EXECUTE_BCP_CMD_MSG)
328                         .withReason(logInfo)
329                         .withResolution(ERR_EXECUTE_BCP_CMD_RESOLUTION);
330                 errorViews.write(dataException);
331             } else if (StringUtils.equalsIgnoreCase(logInfo, BCP_COPY_IN_FAILED_ERROR_MSG)) {
332                 // BCP Reached the maximum error limit and stopped
333                 SnapDataException dataException = new SnapDataException(
334                         ERR_REACHED_MAXIMUM_ERROR_LIMIT_MSG)
335                         .withReason(logInfo)
336                         .withResolution(ERR_REACHED_MAXIMUM_ERROR_LIMIT_RESOLUTION);
337                 errorViews.write(dataException);
338             } else if (StringUtils.contains(logInfo.toLowerCase(), BCP_COPY_DIRECTION_ERROR_MSG)) {
339                 // BCP Syntax error (possible due to wrongly placed spaces in the command)
340                 SnapDataException dataException = new SnapDataException(
341                         ERR_BCP_DIRECTION_MSG)
342                         .withReason(logInfo)
343                         .withResolutionAsDefect();
344                 errorViews.write(dataException);
345             } else if (StringUtils.contains(logInfo.toLowerCase(), INVALID_OBJECT)) {
346                 SnapDataException dataException = new SnapDataException(
347                         ERR_BCP_INVALID_OBJECT)
348                         .withReason(logInfo)
349                         .withResolution(PLEASE_ADDRESS_REPORTED_ISSUE);
350                 errorViews.write(dataException);
351             } else if (process.exitValue() != 0 || (StringUtils.isEmpty(logInfo) &&
352                     StringUtils.isBlank(errorInfo))) {
353                 SnapDataException dataException = new SnapDataException(
354                         ERR_EXECUTE_BCP_CMD_MSG)
355                         .withReason(logInfo)
356                         .withResolution(ERR_EXECUTE_BCP_CMD_RESOLUTION);
357                 errorViews.write(dataException);
358             } else {
359                 if (StringUtils.isBlank(errorInfo)) {
360                     writeToOutputView(logInfo);
361                 } else {
362                     SnapDataException dataException = new SnapDataException(
363                             ERR_EXECUTE_BCP_CMD_MSG)
364                             .withReason(errorInfo)
365                             .withResolution(ERR_EXECUTE_BCP_CMD_RESOLUTION);
366                     errorViews.write(dataException);
367                 }
368             }
369         } catch (ExecutionException ee) {
370             if (ee.getCause() instanceof SnapDataException) {
371                 errorViews.write((SnapDataException) ee.getCause());
372             } else {
373                 SnapDataException dataException = new SnapDataException(
374                         ee, ERR_EXECUTE_BCP_CMD_MSG)
375                         .withReason(ee.getMessage())
376                         .withResolution(ee.getResolution());
377                 errorViews.write(dataException);
378             }
379         } catch (Exception e2) {
380             SnapDataException dataException = new SnapDataException(e2, ERR_EXECUTE_BCP_CMD_MSG)
381                     .withReason(e2.getMessage())
382                     .withResolution(ERR_EXECUTE_BCP_CMD_RESOLUTION);
383             errorViews.write(dataException);
384         }

430     private String doLinuxEscape(String value) {
431         String newValue = value;
432         String newSymbol;
433         for (String symbol : UNIX_SPECIAL_SYMBOLS) {
434             switch (symbol) {
435                 case CARET:
436                     newValue = newValue.replaceAll(SOURCE_CARET, TARGET_CARET);
437                     break;
438                 case ESCAPE_SYMBOL:
439                     newValue = newValue.replaceAll(SOURCE_ESCAPE, TARGET_ESCAPE);
440                     break;
441                 case QUESTION:
442                     newValue = newValue.replaceAll(SOURCE_QUESTION, TARGET_QUESTION);
443                     break;
444                 case ASTERISK:
445                     newValue = newValue.replaceAll(SOURCE_ASTERISK, TARGET_ASTERISK);
446                     break;
447                 case DOLLAR:
448                     newValue = newValue.replaceAll(SOURCE_DOLLAR, TARGET_DOLLAR);
449                     break;
450                 case PIPE:
451                     newValue = newValue.replaceAll(SOURCE_PIPE, TARGET_PIPE);
452                     break;
453                 case LEFT_PAREN:
454                     newValue = newValue.replaceAll(SOURCE_LEFT_PAREN, TARGET_LEFT_PAREN);
455                     break;
456                 case RIGHT_PAREN:
457                     newValue = newValue.replaceAll(SOURCE_RIGHT_PAREN, TARGET_RIGHT_PAREN);
458                     break;
459                 case LESS_THAN:
460                     newValue = newValue.replaceAll(SOURCE_LESS_THAN, TARGET_LESS_THAN);
461                     break;
462                 case GREATER_THAN:
463                     newValue = newValue.replaceAll(SOURCE_GREATER_THAN, TARGET_GREATER_THAN);
464                     break;
465                 default:
466                     newSymbol = ESCAPE_SYMBOL + ESCAPE_SYMBOL + symbol;
467                     newValue = newValue.replaceAll(symbol, newSymbol);
468                     break;
469             }
470         }
471         return newValue;
472     }

482     private void lookForBcp() {
483         // a caution sysadmin will either drop the PATH environment or
484         // set it to an empty string. In these cases we should return
485         // immediately.
486         if (!System.getenv().containsKey(PATH_ENV_VARIABLE)) {
487             return;
488         }
489 
490         // see above.
491         final String path = System.getenv(PATH_ENV_VARIABLE).trim();
492         if (path.isEmpty()) {
493             return;
494         }
495 
496         for (String prefix : path.split(File.pathSeparator)){
497             final File file = new File(new File(prefix), BCP_NAME);
498             if (file.exists() && file.canRead() && file.canExecute()) {
499                 bcpPath = file.getAbsolutePath();
500                 return;
501             }
502         }
503 
504         throwNoBcpException();
505     }
