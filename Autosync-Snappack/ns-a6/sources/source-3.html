187     /**
188      * Create URLResolver object using the HTTP request query parameters passed.
189      *
190      * @param queryParams key,value parameters to be passed in REST URI
191      * @return URLResolver object
192      */
193     protected URLResolver createURLResolver(final List<Pair<ExpressionProperty,
194             ExpressionProperty>> queryParams) {
195         return document -> {
196             try {
197                 String resourceURL = ((CoupaBaseAccount) account).getBaseUrl();
198                 uriBuilder = new URIBuilder(String.format("%s%s%s", resourceURL, API,
199                         currRestObject));
200                 // queryParams is null for Create and Update snaps
201                 if (Objects.nonNull(queryParams)) {
202                     // left is the param key, right is the value
203                     for (Pair<ExpressionProperty, ExpressionProperty> paramPair : queryParams) {
204                         ExpressionProperty paramKeyProp = paramPair.getLeft();
205                         Object objKey = paramKeyProp.eval(document);
206                         if (Objects.isNull(objKey)) {
207                             continue;
208                         }
209                         String paramKey = String.valueOf(objKey);
210                         if (StringUtils.isBlank(paramKey)) {
211                             continue;
212                         }
213                         ExpressionProperty paramValueProp = paramPair.getRight();
214                         Object objValue = paramValueProp.eval(document);
215                         String paramValue = String.valueOf(objValue);
216                         //record_offset and record_limit are applicable only for Read snap
217                         if (record_offset == 0 && OFFSET.equalsIgnoreCase(paramKey)) {
218                             record_offset = Integer.parseInt(paramValue);
219                         }
220                         if (LIMIT.equalsIgnoreCase(paramKey)) {
221                             limitParamProvided = true;
222                             param_limit = Integer.parseInt(paramValue);
223                         }
224                         uriBuilder.addParameter(paramKey, paramValue);
225                     }
226                 }
227                 if (!responseType.equalsIgnoreCase(RESPONSE_TYPE_DEFAULT)) {
228                     uriBuilder.addParameter(RESPONSE_TYPE_PARAM, responseType);
229                 }
230                 // isExport can be true only for update snaps
231                 if (isExport) {
232                     uriBuilder.addParameter(EXPORTED, Boolean.TRUE.toString());
233                 }
234                 LOGGER.info("URI Builder after adding all query parameters:: {}", uriBuilder);
235                 return uriBuilder.build().toString();
236             } catch (URISyntaxException e) {
237                 LOGGER.error("In createURLResolver, exception:: {}", e.getMessage());
238                 throw new SnapDataException(e, CANNOT_MAKE_REQUEST)
239                         .withReason(String.format("%s: %s", INVALID_URI, e.getMessage()))
240                         .withResolution(PLEASE_VERIFY_URL_ON_ACCOUNT);
241             }
242         };
243     }

267      * Submits HTTP GET request to Coupa server
268      *
269      * @param methodName             HTTP request method name GET
270      * @param document               document object
271      * @param urlResolver            URLResolver object
272      * @param payLoad                HTTP request body
273      * @param isExecuteDuringPreview true when snap is executing in preview mode
274      * @param pageCount              number of records can be retrieved in preview mode
275      */
276     public void executeRequestForRead(String methodName, Document document, URLResolver urlResolver,
277             RequestDataHandler payLoad, boolean isExecuteDuringPreview, int pageCount) {
278         final RestResponseProcessor restResponseProcessor = new ProcessResponse(null,
279                 methodName, outputViews, errorViews, documentUtility);
280         int totalRecords = 0;
281         int recordsSize;
282         restRequestExecutor = new RestRequestExecutor(restHttpClient, errorViews)
283                 .withHeaders(headers)
284                 .withMethod(methodName)
285                 .withResponseProcessor(restResponseProcessor)
286                 .withRetryDecider(this::retryOnAccountReload);
287         do { // continue sending rest GET request until number of records reached the limit
288             // mentioned or all records retrieved
289             result = restRequestExecutor.execute(document, urlResolver, payLoad);
290             //break when no result
291             if (Objects.isNull(result)) {
292                 break;
293             }
294             if (result instanceof ArrayList) {
295                 recordsSize = ((List) result).size();
296                 if (recordsSize == 0) {
297                     break;
298                 }
299             } else {
300                 break;
301             }
302             totalRecords = totalRecords + recordsSize;
303             LOGGER.debug("Result of REST read request contain {} of records", recordsSize);
304             //break when preview page count or requested limit reaches
305             if (isExecuteDuringPreview && totalRecords >= pageCount) {
306                 break;
307             }
308             record_offset += DEFAULT_RECORDS_PER_PAGE_COUNT;
309             if (limitParamProvided) {
310                 param_limit -= recordsSize;
311                 if (param_limit <= 0) {
312                     break;
313                 }
314                 if (param_limit < DEFAULT_RECORDS_PER_PAGE_COUNT) {
315                     uriBuilder.setParameter(LIMIT, String.valueOf(param_limit));
316                 }
317             }
318             uriBuilder.setParameter(OFFSET, String.valueOf(record_offset));
319             urlResolver = createURLResolver();
320         } while (true);
321     }

355      * Creates a schema that can be used by upstream snaps like Mapper. Reads the properties from
356      * json file and adds field name and data type of the field to root schema as child elements.
357      *
358      * @param propertiesMap  REST object properties map
359      * @param rootSchema     REST object schema template
360      * @param definitionsMap All objects definitions map for lookup
361      * @param parentKeyName  parent of the current field, to avoid infinite loop lookup     *
362      * @return schema object that can be used for validating the object format we receive for the
363      * REST operation
364      */
365     protected ObjectSchema createSchemaTemplate(Map propertiesMap, ObjectSchema rootSchema,
366             Map<String, Map<String, Object>> definitionsMap, String parentKeyName) {
367         for (Map.Entry entry : (Iterable<Map.Entry>) propertiesMap.entrySet()) {
368             String fieldKey = entry.getKey().toString();
369             // skip for key name :: parent
370             if (StringUtils.isEmpty(fieldKey) || StringUtils.equalsIgnoreCase(fieldKey, PARENT) ||
371                     StringUtils.equalsIgnoreCase(fieldKey, OWNER)) {
372                 continue;
373             }
374             // add values for the fields
375             if (entry.getValue() instanceof Map) {
376                 Map<String, Object> fieldMap = (Map<String, Object>) entry.getValue();
377                 if (fieldMap.containsKey(REF)) {
378                     // logic for $ref fields
379                     String keyName = fieldMap.get(REF).toString();
380                     keyName = keyName.substring(keyName.lastIndexOf('/') + 1);
381                     if (keyName.equalsIgnoreCase(parentKeyName)) {
382                         continue;
383                     }
384                     if (keyName.equalsIgnoreCase(PARENT)) {
385                         continue;
386                     }
387                     //load properties map for related key name
388                     if (definitionsMap.containsKey(keyName) &&
389                             definitionsMap.get(keyName).containsKey(PROPERTIES)) {
390                         if (summaryMap.containsKey(keyName)) {
391                             ObjectSchema summaryObjectSchema =
392                                     (ObjectSchema) summaryMap.get(keyName);
393                             summaryObjectSchema.setName(fieldKey);//reset the name to the current
394                             // key name
395                             rootSchema.addChild(summaryObjectSchema);
396                         } else {
397                             ObjectSchema summaryObjectSchema =
398                                     schemaFactory.createSchema(SnapType.COMPOSITE, fieldKey);
399                             Map objectPropertiesMap = (Map) definitionsMap.get(keyName).get(
400                                     PROPERTIES);
401                             summaryObjectSchema = createSchemaTemplate(objectPropertiesMap,
402                                     summaryObjectSchema, definitionsMap, keyName);
403                             //process schema for the keyname
404                             summaryMap.put(keyName, summaryObjectSchema);
405                             rootSchema.addChild(summaryObjectSchema);
406                         }
407                     }
408                 } else {
409                     //load datatype for the field
410                     if (Objects.isNull(fieldMap.get(TYPE))) {
411                         continue;
412                     }
413                     String fieldType = fieldMap.get(TYPE).toString();
414                     rootSchema.addChild(createFieldSchema(fieldType, fieldKey));
415                 }
416             }
417         }
418         return rootSchema;
419     }

422      * Search and returns matching rest object path in json file
423      *
424      * @param pathsMap   paths json object represented as a Map
425      * @param restObject REST object name e.g. /suppliers
426      * @return corresponding json object name in given map.
427      */
428     private String matchObjectPath(Map pathsMap, String restObject) {
429         Object[] PATHS_KEY_NAMES_SET = pathsMap.keySet().toArray();
430         String[] restObjectArray = restObject.split(PATH_SEPARATOR);
431         for (Object path : PATHS_KEY_NAMES_SET) {
432             String fieldKey = String.valueOf(path);
433             // REST object and path in map matched exactly
434             if (StringUtils.equals(fieldKey, restObject)) {
435                 return fieldKey;
436             }
437             if (fieldKey.contains(PATH_SEPARATOR)) {
438                 String[] fieldKeyArray = fieldKey.split(PATH_SEPARATOR);
439                 if (fieldKeyArray.length == restObjectArray.length) {
440                     boolean found = true;
441                     for (int restPathInd = 0; restPathInd < restObjectArray.length; restPathInd++) {
442                         Object value = restObjectArray[restPathInd];
443                         String fieldKeyArrayFieldName = fieldKeyArray[restPathInd];
444                         // The places where id is found are place holders, so REST object passed
445                         // from Snap can have different name
446                         if (!Objects.equals(value, fieldKeyArrayFieldName) &&
447                                 !(fieldKeyArrayFieldName.equals(ID_PLACE_HOLDER) ||
448                                         fieldKeyArrayFieldName.contains(ID_PLACE_HOLDER_SUFFIX))) {
449                             found = false;
450                             break;
451                         }
452                     }
453                     if (found) {
454                         return fieldKey;
455                     }
456                 }
457             }
458         }
459         return null;
460     }

622      * Replace the place holders in restObject with id field values coming from the input document.
623      *
624      * @param originalMap input document data in map format
625      */
626     protected void replaceObjectFieldsWithInputFields(Map originalMap) {
627         String[] restObjectArray = currRestObject.split(PATH_SEPARATOR);
628         String regex = "<[a-z]+>";
629         int len = restObjectArray.length;
630         // Requested REST API object does not have any place holders e.g. /suppliers
631         if (len <= 1) {
632             return;
633         }
634         // REST APIs of this format e.g. suppliers/:id, id should be received from input view.
635         for (int restPathInd = 0; restPathInd < len; restPathInd++) {
636             String keyname = restObjectArray[restPathInd];
637             if (keyname.indexOf(":") == 0) {
638                 keyname = keyname.substring(1);
639                 if (Objects.nonNull(originalMap) && Objects.nonNull(originalMap.get(keyname))) {
640                     restObjectArray[restPathInd] = originalMap.get(keyname).toString();
641                     originalMap.remove(keyname); //may be removed if not works
642                 } else {
643                     throw new SnapDataException(String.format(FIELD_NOT_FOUND, keyname))
644                             .withReason(String.format(FIELD_NOT_FOUND, keyname))
645                             .withResolution(ERROR_NOT_FOUND_RESOLUTION);
646                 }
647             } else if (keyname.matches(regex)) {
648                 keyname = keyname.substring(1, keyname.length() - 1);
649                 if (Objects.nonNull(originalMap) && Objects.nonNull(originalMap.get(keyname))) {
650                     restObjectArray[restPathInd] = originalMap.get(keyname).toString();
651                     originalMap.remove(keyname); //may be removed if not works
652                 } else {
653                     throw new SnapDataException(String.format(FIELD_NOT_FOUND, keyname))
654                             .withReason(String.format(FIELD_NOT_FOUND, keyname))
655                             .withResolution(ERROR_NOT_FOUND_RESOLUTION);
656                 }
657             }
658             // put '/' at end for all except last one
659             if (restPathInd < len - 1) {
660                 restObjectArray[restPathInd] += PATH_SEPARATOR;
661             }
662         }
663         currRestObject = String.join("", restObjectArray);
664         LOGGER.debug("After replacing id fields with id fields mapped in the input document, " +
665                 "restObject:: {}", currRestObject);
666     }

689             }

696                             refKey);

698                 } else {

723                     .withResolution(ERROR_NOT_FOUND_RESOLUTION);
