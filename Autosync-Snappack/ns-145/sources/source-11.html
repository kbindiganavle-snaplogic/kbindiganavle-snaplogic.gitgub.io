298      * Trigger a child pipeline with the given import parameters
299      *
300      *
301      * @param jCoFunction
302      * @param importParams
303      */
304     private void triggerChildPipeline(JCoFunction jCoFunction, Map<String, Object> importParams)
305             throws AbapClassException {
306         // Increase activity account for the snap's own input view.
307         LinkedPipeline linkedPipeline;
308         Map<ViewType, Number> inputViewSummary;
309         Map<ViewType, Number> outputViewSummary;
310         if (pipeExecProperties.reuse && initialLinkedPipeline != null) {
311             linkedPipeline = initialLinkedPipeline;
312             inputViewSummary = initialInputViewSummary;
313             outputViewSummary = initialOutputViewSummary;
314         } else {
315             LOGGER.debug("[RFC Server] Attempting to link-and-load pipeline from path: {}, " +
316                     "for function: {} ", pipelinePath, jCoFunction.getName());
317             linkedPipeline = pipelineFacade.linkAndLoad(pipelinePath);
318             inputViewSummary =
319                     initialLinkedPipeline.getViewSummary(ViewType.ViewDirection.INPUT);
320             outputViewSummary =
321                     initialLinkedPipeline.getViewSummary(ViewType.ViewDirection.OUTPUT);
322         }
323         Map<String, String> envMap = Maps.newLinkedHashMap();
324         if (!pipeExecProperties.reuse) {
325             // Parameter mapping doesn't work for reuse
326             importParams.forEach((name, value) -> {
327                 if (paramMapping.containsKey(name)) {
328                     String envKey = paramMapping.get(name);
329                     envMap.put(envKey, String.valueOf(value));
330                 }
331             });
332         }
333         Document document = documentUtility.newDocument(importParams);
334         PipelineExecutionBuilder builder = linkedPipeline
335                 .prepare()
336                 .withLabel(jCoFunction.getName())
337                 .withEnvMap(envMap);
338         snaplexPath.ifPresent(builder::onSnaplex);
339         PipeCompletionHandler completionHandler = null;
340         PipelineRun pipelineRun;
341         if (pipeExecProperties.reuse) {
342             builder.isQueueable(false);
343             List<ExecutionTracker> existingRuns = Lists.newArrayList();
344             boolean tryNewExecution = false;
345             for (ExecutionTracker executionTracker : executions.values()) {
346                 if (executionTracker.isClosed()) {
347                     continue;
348                 }
349                 if (!snaplexPath.isPresent()) {
350                     executionTracker.updateActiveCount();
351                 }
352                 existingRuns.add(executionTracker);
353             }
354             // If the snapplex for the child pipeline is not the current one, we randomly
355             // pick a pipeline to send document. Otherwise, we sort the runtimes and pick up
356             // the pipeline that has lease load
357             if (snaplexPath.isPresent()) {
358                 Collections.shuffle(existingRuns);
359             } else {
360                 Collections.sort(existingRuns);
361             }
362             if (existingRuns.isEmpty() || snaplexPath.isPresent()) {
363                 // Nothing is running yet, need to start up something.
364                 tryNewExecution = true;
365             } else if (existingRuns.get(0).getLastActiveCount() > 0) {
366                 // The least-loaded run is still processing docs, so start up a new one.
367                 tryNewExecution = true;
368             }
369             // Try to start up a new execution.  If we're already at our max pool size, the
370             // pool.tryAcquire() should return false and we'll just reuse an existing execution.
371             if (tryNewExecution && pool.tryAcquire()) {
372                 int childIndex = childCount;
373                 childCount += 1;
374                 completionHandler = new PipeCompletionHandler(
375                         builder, null, linkedPipeline,
376                         retryLimitValue, childIndex,
377                         documentUtility);
378                 pipelineRun = executePipeline(completionHandler, existingRuns);
379 
380                 if (pipelineRun == null) {
381                     pipelineRun = existingRuns.get(0).useExecution();
382                 }
383             } else {
384                 pipelineRun = existingRuns.get(0).useExecution();
385                 completionHandler = executions.get(pipelineRun.getId()).getCompletionHandler();
386                 //statusReporter.reset();
387             }
388         } else {
389             builder.withTimeout(pipeExecProperties.timeout);
390             LOG.info("Waiting for execution slot: {}", pool.availablePermits());
391             try {
392                 pool.acquire();
393             } catch (InterruptedException e) {
394                 LOG.debug("Interrupted while trying to acquire pool semaphore", e);
395                 looping = false;
396                 return;
397             }
398 
399             LOG.info("Acquired execution slot: {}", pool.availablePermits());
400             int childIndex = childCount;
401             childCount += 1;
402             completionHandler = new PipeCompletionHandler(builder, document,
403                     linkedPipeline, retryLimitValue, childIndex, documentUtility);
404             pipelineRun = executePipeline(completionHandler, Collections.emptyList());
405         }
406 
407         if (hasUnlinkedView(outputViewSummary)) {
408             Thread writerThread = spawnHelperThread(RFC_OUTPUT_COLLECTOR_THREAD_NAME,
409                     new OutputDocumentCollector(pipelineRun, completionHandler));
410             writers.put(pipelineRun.getId(), writerThread);
411             writerThread.start();
412         }
413 
414         if (document != null) {
415             Optional<OutputViews> views = getChildUnlinkedInputView(pipelineRun, inputViewSummary);
416             if (views.isPresent()) {
417                 writeDocumentToChild(document, views.get());
418             }
419         }
420 
421         try {
422             while (looping && !aborted && !completionHandler.isAborted()) {
423                 if (completionHandler.isPipelineComplete()) {
424                     pool.release();
425                     executions.remove(pipelineRun.getId());
426                     PipeCompletionHandler retryHandler = completionHandler.retryIfNeeded();
427                     if (retryHandler != null) {
428                         // Retry
429                         completionHandler = retryHandler;
430                         if (retryIntervalValue > 0) {
431                             TimeUnit.SECONDS.sleep(retryIntervalValue);
432                         }
433                         pipelineRun = executePipeline(retryHandler, Collections.emptyList());
434                         if (hasUnlinkedView(
435                                 linkedPipeline.getViewSummary(ViewType.ViewDirection.OUTPUT))) {
436                             Thread writerThread = spawnHelperThread(
437                                     RFC_OUTPUT_COLLECTOR_THREAD_NAME,
438                                     new OutputDocumentCollector(pipelineRun, completionHandler));
439                             writers.put(pipelineRun.getId(), writerThread);
440                             writerThread.start();
441                         }
442                         if (document != null) {
443                             Optional<OutputViews> optionalOutputViews =
444                                     getChildUnlinkedInputView(pipelineRun,
445                                     linkedPipeline.getViewSummary(ViewType.ViewDirection.INPUT));
446                             if (optionalOutputViews.isPresent()) {
447                                     writeDocumentToChild(document, optionalOutputViews.get());
448                             }
449                         }
450 
451                     } else {
452                         // Handle output and exceptions
453                         Throwable th = completionHandler.getExceptionQueue().poll();
454                         if (th != null) {
455                             throw handleUnexpectedException(th);
456                         }
457                         List<Document> outputDocuments = completionHandler.getDocsFromChild();
458                         if (!outputDocuments.isEmpty()) {
459                             handleExportParameters(jCoFunction, outputDocuments.get(0));
460                             synchronized (outputViews) {
461                                 outputDocuments.forEach(outputViews::write);
462                             }
463                         }
464                         break;
465                     }
466                 }
467                 TimeUnit.MILLISECONDS.sleep(checkoutIntervalMs);
468             }
469         } catch (InterruptedException e) {
470             LOG.debug("The handler got interrupted while waiting for the child pipeline to " +
471                     "complete");
472         }
473     }

509     private void startCompletionReceiver() {
510         execActivityThread = spawnHelperThread(RFC_PIPELINE_COMPLETION_RECEIVER_THREAD_NAME,
511                 () -> {
512                     while (looping) {
513                         try {
514                             LOGGER.debug("Waiting for completions");
515                             AbstractActivityHandler handler =
516                                     executionActivity.poll(MAX_WAIT_TIME, MAX_WAIT_UNIT);
517                             if (handler != null) {
518                                 LOGGER.debug("Received completion: {}", handler);
519                                 handler.run();
520                             }
521                         } catch (InterruptedException e) {
522                             LOGGER.debug("Interrupted while waiting for pipeline completions!", e);
523                             return;
524                         } catch (Throwable th) {
525                             handleUnexpectedException(th);
526 
527                         }
528                     }
529                 });
530         execActivityThread.start();
531     }

550      * Helper function for handling an exception in a helper thread.  We want to queue the
551      * exception and let the main thread pick it up so that the snap will fail.
552      *
553      * @param th The unexpected exception.
554      */
555     private AbapClassException handleUnexpectedException(Throwable th) {
556         LOGGER.error("Received an unexpected exception, passing to the main thread", th);
557         aborted = true;
558         looping = false;
559         exceptionQueue.offer(th);
560         if (th instanceof Exception) {
561             return new AbapClassException((Exception) th);
562         } else {
563             return new AbapClassException(new ExecutionException(th, th.getMessage()));
564         }

619     /**
620      * Execute a pipeline as specified by the builder.
621      *
622      * @param completionHandler The handler to be invoked when the pipeline finishes.
623      * @param existingRuns
624      * @return
625      */
626     private PipelineRun executePipeline(PipeCompletionHandler completionHandler,
627                                         final List<ExecutionTracker> existingRuns)
628             throws AbapClassException {
629         PipelineExecutionBuilder builder = completionHandler.getBuilder()
630                 .withInitializationHandler(new AbstractActivityHandler() {
631                     @Override
632                     public void run() {
633                         executions.put(pipelineRun.getId(),
634                                 new ExecutionTracker(pipelineRun, completionHandler));
635                         if (pipeExecProperties.reuse) {
636                             initQueue.offer(pipelineRun.getId());
637                         }
638                     }
639                 })
640                 .withCompletionHandler(completionHandler);
641 
642         final long deadline = System.currentTimeMillis() + OVERLOAD_RETRY_TIMEOUT;
643 
644         while (true) {
645             try {
646                 final PipelineRun pipelineRun = builder.execute();
647                 LOG.info("Executed pipeline: {}", pipelineRun.getId());
648                 if (pipeExecProperties.reuse) {
649                     try {
650                         // When reusing executions, we expect the 'executions' map to be updated,
651                         // so we need to wait until the initialization handler has been run before
652                         // we can return.
653                         initQueue.poll(1, TimeUnit.MINUTES);
654                     } catch (InterruptedException e) {
655                         LOG.debug("Interrupted while waiting for new pipeline execution ID", e);
656                     }
657                 }
658                 return pipelineRun;
659             } catch (NoCCSnapiException e) {
660                 LOG.warn("Plex is overloaded", e);
661 
662                 if (pipeExecProperties.reuse && !existingRuns.isEmpty()) {
663                     LOG.info("Unable to start a new execution while below pool size, need to " +
664                             "reuse an existing execution");
665                     pool.release();
666                     return null;
667                 }
668 
669                 if (System.currentTimeMillis() >= deadline) {
670                     pool.release();
671                     throw handleUnexpectedException(
672                             new SnapDataException(e, ERROR_UNABLE_EXECUTE_ON_OVERLOADED_SNAPLEX)
673                             .withResolution(RESOLUTION_ADD_MORE_RESOURCES));
674                 } else {
675                     try {
676                         Thread.sleep(OVERLOAD_RETRY_DELAY);
677                     } catch (InterruptedException e1) {
678                         throw handleUnexpectedException(
679                                 new ExecutionException(e1,
680                                         ERROR_INTERRUPTED_WHILE_ATTEMPTING_TO_EXECUTE_PIPELINE));
681                     }
682                 }
683             } catch (RuntimeException e) {
684                 LOG.error("Failed to execute pipeline", e);
685                 pool.release();
686                 if (e instanceof NotFoundSnapiException) {
687                     if (snaplexPath.isPresent()){
688                         throw handleUnexpectedException(
689                                 new SnapDataException(e, ERROR_UNABLE_TO_EXECUTE_PIPELINE_ON_PLEX)
690                                         .formatWith(snaplexPath.get())
691                                         .withResolution(RESOLUTION_VERIFY_SNAPLEX_PATH));
692                     }
693 
694                 }
695                 throw handleUnexpectedException(
696                         new SnapDataException(e, ERROR_EXECUTE_PIPELINE));
697             }
698         }
699     }

705     public void shutdown() {
706         Thread.interrupted(); // Clear the interrupt state
707         looping = false;
708         if (execActivityThread != null) {
709             execActivityThread.interrupt();
710         }
711         LOG.info("Shutting down any remaining child executions");
712         for (ExecutionTracker executionTracker : executions.values()) {
713             executionTracker.stop();
714         }
715         executions.clear();
716         if (aborted) {
717             // On a failure, we want to encourage the writer threads to exit.
718             for (Thread thread : writers.values()) {
719                 thread.interrupt();
720             }
721         }
722     }
