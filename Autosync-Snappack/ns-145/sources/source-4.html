343     @Override
344     protected void process(final Document document, final String viewName) {
345         /**If exception caught in configure method and routeErrors enabled then no
346          * need to continue process().*/
347         if (caughtError && routeErrors) {
348             return;
349         }
350         LOG.debug(STARTING_SERVER);
351         mainThreadContextMap = MDC.getCopyOfContextMap();
352         JCoServerState state = server.getState();
353         if (state == JCoServerState.STOPPED || state == JCoServerState.DEAD) {
354             server.start();
355         } else if (state == JCoServerState.STOPPING) {
356             // Don't delete this line, or the server running elsewhere with the same
357             // configuration will be stopped!
358             server = null;
359             throw new ExecutionException(EXCEPTION_STARTING_SEVER)
360                     .withReason(EXISTING_SERVER)
361                     .withResolution(RESOLUTION_SHUT_DOWN_EXISTING_SERVER);
362         }
363         if (maxRetryAttempts == 0) {
364             /**
365              * Default behaviour of SAP JCO listener..
366              * If maxAttempts = 0 then it does infinite retries.
367              */
368             interrupt = true;
369             try {
370                 Thread.currentThread().join();
371             } catch (InterruptedException e) {
372                 // Ignore
373             }
374         } else {
375             while (!interrupt) {
376                 try {
377                     Thread.sleep(WAIT_TIME);
378                 } catch (InterruptedException e) {
379                     // Ignore
380                 }
381             }
382             if (interruptedException != null && interruptedException instanceof SnapDataException) {
383                 errorViews.write((SnapDataException) interruptedException);
384             } else if (lastIgnoredException != null &&
385                     lastIgnoredException instanceof SnapDataException) {
386                 errorViews.write((SnapDataException) lastIgnoredException);
387             }
388         }
389     }

397     void stopAndReleaseJCoServer() {
398         interrupt = true;
399         if (server != null) {

401             if (!(state == JCoServerState.STOPPED) && !(state == JCoServerState.STOPPING)) {
402                 server.stop();

404             releaseJCoServer(server);
405         }
406         jcoConnectionManager.getJcoServerDataProvider().deregisterListener(this);
407         retryCount = 0;

478     private synchronized void updateRetryAttemptOnJCoError(Throwable error, JCoServer jCoServer) {
479         error = Throwables.getRootCause(error);
480         int group = SAPUtils.getErrorGroup(error);
481         if (isConnectionError(group)) {
482             retryCount++;
483             LOG.error(String.format(JCO_SERVER_CONNECTION_ERROR_OCCURRED, retryCount), error);
484             if (maxRetryAttempts != 0 && retryCount >= maxRetryAttempts) {
485                 SnapDataException sde = new SnapDataException(error,
486                         String.format(JCO_SERVER_ERROR_OCCURRED, maxRetryAttempts))
487                         .withReason(error.getMessage())
488                         .withResolution(RESOLUTION_USE_VALID_PROPERTIES);
489                 interruptedException = sde;
490                 interrupt = true;
491             }
492         } else {
493             // Resets retryCount to 0 when non network/connection error occurred.
494             retryCount = 0;
495             SnapDataException sde = new SnapDataException(error, JCO_SERVER_EXCEPTION_OCCURRED)
496                     .withReason(error.getMessage())
497                     .withResolution(RESOLUTION_USE_VALID_PROPERTIES);
498             if (haltOnSapErrorEnabled && isInterruptibleError(group)) {
499                 LOG.error(String.format("%s. Stopping server", JCO_SERVER_EXCEPTION_OCCURRED),
500                     error);
501                 interruptedException = sde;
502                 interrupt = true;
503             } else {
504                 lastIgnoredException = sde;
505             }
506         }
507     }

509     private void releaseJCoServer(JCoServer server) throws ExecutionException {
510         server.removeServerErrorListener(listener);
511         server.removeServerExceptionListener(listener);
512         try {
513             int waitTime = 0;
514             while (server.getState() != JCoServerState.STOPPED) {
515                 Thread.sleep(POLLING_INTERVAL);
516                 waitTime += POLLING_INTERVAL;
517                 if (waitTime >= POLLING_TIME_OUT_IN_MS) {
518                     throw new ExecutionException(EXCEPTION_STOPPING_SERVER)
519                             .withReason(TIME_OUT_WAITING)
520                             .withResolution(RESOLUTION_STOP_SERVER_MANUALLY);
521                 }
522             }
523         } catch (InterruptedException e) {
524             throw new ExecutionException(EXCEPTION_STOPPING_SERVER)
525                     .withReason(INTERRUPTED_WHILE_WAITING_FOR_SERVER_TO_STOP)
526                     .withResolution(RESOLUTION_STOP_SERVER_MANUALLY);
527         }
528         server.release();
529         server = null;
530     }
