319     protected JCoDestination getJCoDestination(String sessionID) {
320         JCoDestination jCoDestination = null;
321         for (int i = 1; i <= maxAttempts; i++) {
322             try {
323                 if (StringUtils.isNotEmpty(sessionID)) {
324                     jCoDestination = jCoAccount.connect(sessionID);
325                 } else {
326                     jCoDestination = jCoAccount.connect();
327                 }
328                 log.debug(LOG_CONNECTION_ATTEMPT_MSG);
329                 break;
330             } catch (ConfigurationException e) {
331                 JCoException jcoe = (JCoException) e.getCause();
332                 log.error(String.format(LOG_CONNECTION_RETRY_MSG, i,
333                         jcoe.getLocalizedMessage(), retryInterval));
334                 /**
335                  * Break the loop If login failed due to invalid credentials.
336                  * */
337                 if (jcoe.getGroup() == JCoException.JCO_ERROR_LOGON_FAILURE || jcoe.getGroup() ==
338                         JCoException.JCO_ERROR_PASSWORD_CHANGE_REQUIRED) {
339                     handleExceptionOnFailure(errorHandler, jcoe, String.format
340                                     (CONNECTION_FAIL_MSG, jCoAccount.getUsername()),
341                             encodeMessage(jcoe.getLocalizedMessage()),
342                             CONNECTION_FAIL_RESOLUTION);
343                     break;
344                 }
345                 if (i == maxAttempts) {
346                     handleExceptionOnFailure(errorHandler, jcoe, String.format
347                                     (CONNECTION_FAIL_MSG, jCoAccount.getUsername()),
348                             encodeMessage(jcoe.getLocalizedMessage()),
349                             CONNECTION_FAIL_RESOLUTION);
350                     break;
351                 }
352                 try {
353                     Thread.sleep(retryInterval * 1000);
354                 } catch (InterruptedException ie) {
355                     //NO OP
356                 }
357             }
358         }
359         return jCoDestination;
360     }

377     protected void defineSchema(final SchemaProvider provider, IDocDocument iDocDocument) {
378         for (String inputViewName : provider.getRegisteredViewNames()) {
379             SchemaBuilder schemaBuilder = provider.getSchemaBuilder(inputViewName);
380             ObjectSchema objectSchemaControlRecord = getIDocControlRecordSchema(provider);
381             schemaBuilder.withChildSchema(objectSchemaControlRecord);
382 
383             if (iDocDocument.getRootSegment() != null) {
384                 IDocSegmentMetaData segmentMetaData =
385                         iDocDocument.getRootSegment().getSegmentMetaData();
386                 if (segmentMetaData == null) {
387                     return;
388                 }
389                 long maxOccurrences = segmentMetaData.getMaxOccurrence();
390                 IDocRecordMetaData recordMetaData = segmentMetaData.getRecordMetaData();
391                 String recordName = recordMetaData.getName();
392                 ObjectSchema objectSchema;
393                 if (maxOccurrences > 1) {
394                     ObjectSchema objectSchemaTable =
395                             provider.createSchema(SnapType.TABLE, recordName);
396                     schemaBuilder.withChildSchema(objectSchemaTable);
397                     objectSchema = provider.createSchema(SnapType.COMPOSITE, recordName);
398                     objectSchemaTable.addChild(objectSchema);
399                 } else {
400                     objectSchema = provider.createSchema(SnapType.COMPOSITE, recordName);
401                     schemaBuilder.withChildSchema(objectSchema);
402                 }
403                 buildSchemaFromIDocSegment(provider, objectSchema, segmentMetaData);
404                 schemaBuilder.build();
405             }
406         }
407     }

506     /**
507      * Build SnapLogic schema based upon SAP IDocWrite segment
508      *
509      * @param provider        schema provider
510      * @param objectSchema    SnapLogic object schema
511      * @param segmentMetaData sap segment metadata
512      */
513     private void buildSchemaFromIDocSegment(SchemaProvider provider, ObjectSchema objectSchema,
514             IDocSegmentMetaData segmentMetaData) {
515         if (segmentMetaData == null) {
516             return;
517         }
518         IDocRecordMetaData recordMetaData = segmentMetaData.getRecordMetaData();
519         int numFields = recordMetaData.getNumFields();
520         for (int i = 0; i < numFields; ++i) {
521             SnapType snapType =
522                     SAP_TYPE_TO_JSON_SCHEMA_MAPPING.getOrDefault(recordMetaData.getType(i),
523                             SnapType.STRING);
524             ObjectSchema childSchema = provider.createSchema(snapType, recordMetaData.getName(i));
525             childSchema.setDescription(String.format(DESCR_FORMAT, recordMetaData.getLength(i),
526                     recordMetaData.getDescription(i)));
527
