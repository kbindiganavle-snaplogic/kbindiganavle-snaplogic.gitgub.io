268     public void configure(PropertyValues propertyValues) {
269         executor = Executors.newSingleThreadExecutor();
270         bapiExpr = propertyValues.getAsExpression(BAPI);
271         isTransaction = Boolean.TRUE.equals(propertyValues.get(COMMIT));
272         isOutputDateTimeAsText = Boolean.TRUE.equals(propertyValues.get(OUTPUT_DATE_TIME_AS_TEXT));
273         dateConversionExpr = propertyValues.getAsExpression(DATE_CONVERSION);
274         List<Map<String, String>> errorHandling = propertyValues.get(ERROR_HANDLING);
275         if (CollectionUtils.isNotEmpty(errorHandling)) {
276             errorHandlings = Lists.newArrayListWithExpectedSize(errorHandling.size());
277             for (Map<String, String> errorHandlingRow : errorHandling) {
278                 ExpressionProperty errorPathProperty = propertyValues.getExpressionPropertyFor(
279                         errorHandlingRow, ERROR_PATH);
280                 ExpressionProperty errorValueProperty = propertyValues.getExpressionPropertyFor(
281                         errorHandlingRow, ERROR_VALUE);
282                 errorHandlings.add(Pair.of(errorPathProperty, errorValueProperty));
283             }
284         }
285         configureRetryProperties(propertyValues);
286         configureRouteErrorProperty(propertyValues);
287         // setup the pipeline runtimeID to the account instance, so that a dedicated session could
288         // be assigned to the execution.
289         ExpressionProperty pipRuuidExpr = expressionUtils.createExpressionProperty(propertyValues
290                 , true, PIPE_RUUID_EXPRESSION);
291         runtimeID = pipRuuidExpr.eval(null);
292         log.debug("Get RuntimeID = {}", runtimeID);
293         jCoAccount.setRuntimeID(runtimeID);
294         sessionID = propertyValues.getAsExpression(SESSION_ID_PROP).eval(null);
295         jCoDestination = getJCoDestination(sessionID);
296         commitTimeout = propertyValues.getAsExpression(COMMIT_TIMEOUT_PROP).eval(null);
297         if (commitTimeout == null) {
298             commitTimeout = BigInteger.valueOf(DEFAULT_SEQUENCE_TIMEOUT);
299         }
300         startCallSequence = Boolean.TRUE.equals(propertyValues.get(START_CALL_SEQUENCE_PROP));
301         reloadFunctionMetadata = Boolean.TRUE.equals(propertyValues.get(RELOAD_BAPI_METADATA));
302         try {
303             bapiName = bapiExpr.eval(null);
304         } catch (Exception e) {
305             log.debug(e.getMessage());
306         }
307         if (bapiName != null) {

309                     isTransaction, startCallSequence, reloadFunctionMetadata);

313     @Override
314     protected void process(Document document, String inputViewName) {
315         try {
316             DateTimeFormatter dateTimeFormatter =
317                     getDateTimeFormatter(dateConversionExpr.eval(document));
318             if (StringUtils.isNotBlank(bapiName) && context == null) {
319                 return;
320             }
321             if (!isConnected) {
322                 throw new SnapDataException(String.format(CONNECTION_FAILED, jCoDestination
323                                 .getDestinationName()))
324                                 .withReason(String.format(CONNECTION_FAIL_REASON,
325                                         JCoException.JCO_ERROR_COMMUNICATION))
326                                 .withResolution(CONNECTION_FAIL_RESOLUTION);
327             }
328             if ((super.getDocumentContentAsMap(document) == null && inputViewName != null)
329                     || !checkSAPContext(document, isTransaction, startCallSequence)) {
330                 // if not a map: will write to error view if enabled otherwise throw exception
331                 return;
332             }
333             SAPDocumentWrapper documentWrapper = new SAPDocumentWrapper(document);
334             try {
335                 if (startCallSequence) {
336                     AbstractJCoAccount.localSessionRef.get()
337                             .awaitTransactionStart(commitTimeout.intValue());
338                 }
339                 sapExecutor.process(documentWrapper, context,
340                         dateTimeFormatter, isOutputDateTimeAsText);
341             } catch (ExecutionException e) {
342                 JCoException jCoException = null;
343                 if (e.getCause() instanceof JCoException) {
344                     jCoException = (JCoException) e.getCause();
345                 }
346                 if (jCoException != null && isAConnectionError(jCoException)) {
347                     try {
348                         checkSAPConnection(executor, jCoDestination);
349                     } catch (JCoException jcoe) {
350                         isConnected = false;
351                         throw new SnapDataException(jcoe, String.format(CONNECTION_FAILED,
352                                 jCoDestination))
353                                 .withResolution(CONNECTION_FAIL_RESOLUTION);
354                     }
355                 }
356                 throw new SnapDataException(e.getCause(), EXCEPTION_OCCURRED)
357                         .withResolution(PLEASE_VERIFY_CONFIGURATION_AND_INPUT_DOCUMENT);
358             }
359             if (documentWrapper.hasErrors()) {
360                 throw documentWrapper.createException(documentUtility);
361             } else {
362                 Map<String, Object> outputDocumentMap = documentWrapper.getDocumentMap();
363                 if (CollectionUtils.isNotEmpty(errorHandlings)) {
364                     for (Pair<ExpressionProperty, ExpressionProperty> errorHandlingRow :
365                             errorHandlings) {
366                         Object path = errorHandlingRow.getLeft().eval(document);
367                         if (!(path instanceof String)) {
368                             throw new SnapDataException(INVALID_ERROR_PATH)
369                                     .withReason(INVALID_ERROR_PATH)
370                                     .withResolution(INVALID_ERROR_PATH_RESOLUTION);
371                         }
372                         try {
373                             JsonPath jsonPath = JsonPathImpl.compile((String) path);
374                             // a null value seems valid for error condition
375                             Object value = errorHandlingRow.getRight().eval(document);
376                             Object documentValue = jsonPath.readStatic(outputDocumentMap, null);
377                             boolean foundError = false;
378                             if (documentValue instanceof Collection) {
379                                 // check if any of the values returned by the path match the error
380                                 // value, if it does, we route it to the error view.
381                                 for (Object entry : (Collection) documentValue) {
382                                     foundError = entry.equals(value);
383                                     if (foundError) {
384                                         break;
385                                     }
386                                 }
387                             }
388                             if (value.equals(documentValue) || foundError) {
389                                 throw new SnapDataException(FOUND_ERROR)
390                                         .withReason(FOUND_ERROR)
391                                         .withResolution(String.format(FOUND_ERROR_RESOLUTION, value,
392                                                 path));
393                             }
394                         } catch (InvalidPathException e) {
395                             throw new SnapDataException(INVALID_ERROR_JSON_PATH)
396                                     .withReason(INVALID_ERROR_JSON_PATH)
397                                     .withResolution(INVALID_ERROR_JSON_RESOLUTION);
398                         }
399                     }
400                 }
401                 Document documentCreated = documentWrapper.createOutputDocument(documentUtility);
402                 outputViews.write(documentCreated,document);
403                 Document original = documentWrapper.getInputDocument();
404                 if (original != null) {
405                     original.acknowledge();
406                 }
407             }
408         } catch (SnapDataException sde) {
409             errorViews.write(sde, document);
410         }
411     }

413     private Set<String> getBAPISuggestions(PropertyValues propertyValues) {
414         Set<String> bapiNames = Sets.newHashSet();
415         JCoRepository jCoRepositorySuggest;
416         try {
417             JCoDestination jCoSuggestDestination = jCoAccount.connect();
418             JCo.setProperty(Messages.JCO_AUTO_COMMIT_OPTION, Messages.JCO_AUTO_COMMIT_OPTION_TRUE);
419             jCoRepositorySuggest = jCoSuggestDestination.getRepository();
420             JCoFunctionTemplate jCoFunctionTemplateBapiGetList = jCoRepositorySuggest
421                     .getFunctionTemplate(BAPI_MONITOR_GETLIST);
422             JCoFunction jCoFunctionBapiGetList = null;
423             if (jCoFunctionTemplateBapiGetList != null) {
424                 jCoFunctionBapiGetList = jCoFunctionTemplateBapiGetList.getFunction();
425             }
426             if (jCoFunctionBapiGetList != null) {
427                 JCoParameterList exportParamList = jCoFunctionBapiGetList.getImportParameterList();
428                 // configure BAPI List call
429                 List<Map<String, String>> config = propertyValues.get(BAPI_CONFIG);
430                 if (CollectionUtils.isNotEmpty(config)) {
431                     for (Map<String, String> configRow : config) {
432                         String configKey = BAPI_CONFIG_KEY_PATH.read(configRow);
433                         if (configKey == null) {
434                             continue;
435                         }
436                         String configValue = BAPI_CONFIG_VALUE_PATH.read(configRow);
437                         if (configValue == null) {
438                             configValue = SAP_EMPTY_VALUE;
439                         }
440                         exportParamList.setValue(configKey, configValue);
441                     }
442                 }
443                 // return all BAPI's
444                 jCoFunctionBapiGetList.execute(jCoSuggestDestination);
445                 JCoParameterList bapis = jCoFunctionBapiGetList.getTableParameterList();
446                 addBAPIs(bapis, bapiNames);
447             }
448         } catch (Exception e) {
449             bapiNames.clear();
450             bapiNames.add(Throwables.getRootCause(e).getMessage());
451         }
452         return bapiNames;
453     }

455     private void addBAPIs(JCoParameterList bapis, Set<String> bapiNames) {
456         if (bapis != null) {
457             JCoListMetaData jCoListMetaData = bapis.getListMetaData();
458             int fieldCount = jCoListMetaData.getFieldCount();
459             for (int i = 0; i < fieldCount; ++i) {
460                 String name = jCoListMetaData.getName(i);
461                 int type = jCoListMetaData.getType(i);
462                 if (type == JCoMetaData.TYPE_TABLE) {
463                     JCoRecord record = bapis.getTable(name);
464                     JCoTable tableRecord = (JCoTable) record;
465                     tableRecord.firstRow();
466                     int numRows = tableRecord.getNumRows();
467                     for (int j = 0; j < numRows; ++j, tableRecord.nextRow()) {
468                         try {
469                             String abapName = tableRecord.getValue(ABAP_FIELDNAME).toString();
470                             if (StringUtils.isNotEmpty(abapName)) {
471                                 bapiNames.add(abapName);
472                             }
473                         } catch (Throwable e) {
474                             // ignore
475                         }
476                     }
477                 }
478             }
479         }
