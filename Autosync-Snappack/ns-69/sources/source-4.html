18 /**
19  * A schema collector which collects the JSON schema information from <code>XSComponent</code>s
20  *
21  * @author slad@snaplogic.com (Samarth Lad)
22  */
23 public class SchemaCollector {
24     private final XSTypeResolver xsTypeResolver;
25     private final JsonSchema schema;
26 
27     public SchemaCollector() {
28         xsTypeResolver = XSTypeResolver.getInstance();
29         schema = new JsonSchema();
30     }
31 
32     /**
33      * Collects JSON schema information
34      *
35      * @param elementPath
36      * @param xsType
37      * @param isElemRepeated
38      */
39     public void collect(String elementPath, XSType xsType, boolean isElemRepeated) {
40         SnapType type;
41         if (isElemRepeated) {
42             type = SnapType.TABLE;
43         } else {
44             type = xsTypeResolver.resolve(xsType);
45         }
46         schema.add(elementPath, type);
47     }
48 
49     /**
50      * Returns the collected schema
51      *
52      * @return
53      */
54     public JsonSchema getCollectedSchema() {
55         return schema;
56     }
57 }
