336         // initlialize table name

354                 uriBuilder = new StringBuilder(String.format(PRECOMPILED_URI,
355                         account.getInstance(), table, jsonWSVersion));
356             }
357             attachDisplayValue(uriBuilder);
358             attachDisplayVariables(uriBuilder);
359             uriBuilder.append(String.format(SYSPARM_RECORD_COUNT, batchSize));
360         }
361         fetchQuery = uriBuilder.toString();
362         // process query

370     private void processThroughDefaultAPI(final Document document) {
371         // Initially the sys id is null. We fetch a batch and return the
372         // last sys_id from the batch. Then in the next batch we fetch
373         // anything that is larger then the current sys_id.
374         // @see http://wiki.servicenow.com/index.php?title=Exporting_Data
375         String sysId = null;
376         int batchNum = 1;
377         try {
378             do {
379                 URI fullUri;
380                 StringBuilder builder = new StringBuilder(fetchQuery);
381                 attachQuery(query, builder, document);
382                 Iterator<Map<String, Object>> recordIterator;
383                 if (sysId == null) {
384                     fullUri = new URI(builder.toString());
385                 } else {
386                     try {
387                         String encodedQuery = URLEncoder.encode(String.format(SYS_ID_S, sysId),
388                                 DEFAULT_CHARSET);
389                         builder.append(encodedQuery);
390                         fullUri = new URI(builder.toString());
391                     } catch (UnsupportedEncodingException e) {
392                         throw new ConfigurationException(e, ENCODING_NOT_SUPPORTED);
393                     }
394                 }
395                 recordIterator = failsafeExecute(fullUri, batchNum);
396                 if (recordIterator != null) {
397                     if (!recordIterator.hasNext()) {
398                         if (sysId == null) {
399                             //The query fetched nothing.
400                             if (allowEmptyResult) {
401                                 //Return empty document without pass through
402                                 outputViews.write(documentUtility.newDocumentFor(document,
403                                         new LinkedHashMap<String, Object>()));
404                             } else {
405                                 SnapDataException sde = new SnapDataException(ERR_NO_RECORDS_FOUND)
406                                         .withReason(ERR_NO_RECORDS_FOUND_REASON)
407                                         .withResolution(ERR_NO_RECORDS_FOUND_RESOLUTION);
408                                 errorViews.write(sde, document);
409                             }
410                         }
411                         break;
412                     }
413                     while (recordIterator.hasNext()) {
414                         recordCounter.inc();
415                         Map<String, Object> data = recordIterator.next();
416                         writeToOutputView(documentUtility.newDocumentFor(document, data),
417                                 document);
418                         sysId = JsonPath.read(data, SYS_ID);
419                     }
420                 }
421                 batchNum++;
422             } while (sysId != null && !sysId.isEmpty() && !ignorePagination);
423         } catch (URISyntaxException e) {
424             throw new ConfigurationException(e, UNABLE_TO_CREATE_FULL_URI);
425         } catch (SuggestViewAbortException e) {
426             // We have read enough for preview/suggest.
427             // Do nothing.
428         } catch (SnapException e) {
429             // We have already written SnapDataException to error view.
430             // That's why skipping here.
431             if(e.getCause() instanceof SnapDataException) {
432                 throw e;
433             }
434             errorViews.write(serviceNowUtil.getSnapDataException(e, document, documentUtility),
435                     document);
436         }
437     }

439     private void processThroughTableAPI(final Document document) {
440         int offset = 0;
441         int batchNum = 1;
442         try {
443             do {
444                 URI fullUri;
445                 StringBuilder builder = new StringBuilder(fetchQuery);
446                 attachQuery(query, builder, document);
447                 attachFields(fields, builder, document);
448                 attachLimit(offset, builder, document);
449                 Iterator<Map<String, Object>> recordIterator;
450                 fullUri = new URI(builder.toString());
451                 recordIterator = failsafeExecute(fullUri, batchNum);
452                 if (recordIterator != null) {
453                     if (!recordIterator.hasNext()) {
454                         if (allowEmptyResult) {
455                             outputViews.write(documentUtility.newDocumentFor(document,
456                                     new HashMap<String, Object>()));
457                             break;
458                         } else {
459                             SnapDataException sde = new SnapDataException(ERR_NO_RECORDS_FOUND)
460                                     .withReason(ERR_NO_RECORDS_FOUND_REASON)
461                                     .withResolution(ERR_NO_RECORDS_FOUND_RESOLUTION);
462                             errorViews.write(sde, document);
463                         }
464                     }
465                     while (recordIterator.hasNext()) {
466                         recordCounter.inc();
467                         Map<String, Object> data = recordIterator.next();
468                         writeToOutputView(documentUtility.newDocumentFor(document, data),
469                                 document);
470                     }
471                     offset += batchSize;
472                 }
473                 batchNum++;
474             } while (offset < totalRecordCount);
475         } catch (URISyntaxException e) {
476             throw new ConfigurationException(e, UNABLE_TO_CREATE_FULL_URI);
477         } catch (SuggestViewAbortException e) {
478             // We have read enough for preview/suggest.
479             // Do nothing.
480         } catch (SnapException e) {
481             // We have already written SnapDataException to error view.
482             // That's why skipping here.
483             if(e.getCause() instanceof SnapDataException) {
484                 throw e;
485             }
486             errorViews.write(serviceNowUtil.getSnapDataException(e, document, documentUtility),
487                     document);
488         }

491     /**
492      * Prepares the <code>HttpGet</code> request and executes it. It also
493      * sets the iterator object associated with this class if everything goes well.
494      *
495      * @param fullUri A URI to send the request to
496      */
497     private Iterator<Map<String, Object>> prepareAndExecute(URI fullUri) {
498         HttpResponse response = null;
499         HttpGet getRequest = new HttpGet(fullUri);
500         serviceNowRequest.initialize(getRequest, account);
501         try {
502             StatusLine statusLine;
503             int responseCode;
504             HttpEntity entity;
505             int oauthTokenRetryCount = 0;
506             boolean refreshed;
507             do {
508                 refreshed = false;
509                 response = serviceNowRequest.execute();
510                 statusLine = response.getStatusLine();
511                 responseCode = statusLine.getStatusCode();
512                 if (responseCode == SC_UNAUTHORIZED &&
513                         (account instanceof ServiceNowOAuth2Account) &&
514                         (oauthTokenRetryCount++ < MAX_OAUTH_RETRY_COUNT)) {
515                     refreshed = ServiceNowHelper.refreshOAuth2Account(account, LOG);
516                     LOG.debug("Able to refresh oauth2 token after 401 error? {}", refreshed);
517                     if (refreshed) {
518                         serviceNowRequest.updateHeaders(account);
519                     }
520                 }
521             } while (refreshed);
522             entity = response.getEntity();
523             if (entity != null) {
524                 try (InputStream is = entity.getContent()) {
525                     if (responseCode != SC_OK) {
