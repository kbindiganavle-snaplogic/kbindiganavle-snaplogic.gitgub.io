345      * Executes an <code>httpPost</code> request
346      *
347      * Generic response format:
348      * {
349      *   "records": [{
350      *     "active": "true",
351      *     ...
352      *     "category": "Inquiry / Help",
353      *     ...
354      *     "state": "Open",
355      *     ...
356      *   }, ...]
357      * }
358      */
359     protected void executeRequest() {
360         HttpResponse response = null;
361         try {
362             StatusLine statusLine;
363             int responseCode;
364             HttpEntity entity;
365             int oauthTokenRetryCount = 0;
366             boolean refreshed;
367             do {
368                 refreshed = false;
369                 response = serviceNowRequest.execute();
370                 statusLine = response.getStatusLine();
371                 responseCode = statusLine.getStatusCode();
372                 if (responseCode == SC_UNAUTHORIZED &&
373                         (account instanceof ServiceNowOAuth2Account) &&
374                         (oauthTokenRetryCount++ < MAX_OAUTH_RETRY_COUNT)) {
375                     refreshed = ServiceNowHelper.refreshOAuth2Account(account, LOG);
376                     getLogger().debug("Able to refresh oauth2 token after 401 error? {}",
377                             refreshed);
378                 }
379                 if (refreshed) {
380                     serviceNowRequest.updateHeaders(account);
381                 }
382             } while (refreshed);
383             entity = response.getEntity();
384             if (entity != null) {
385                 try (InputStream is = entity.getContent()) {
386                     if (responseCode != SC_OK) {
387                         handleHttpNotOK(responseCode, statusLine);
388                     } else {
389                         String entityStr = IOUtils.toString(is, CharEncoding.UTF_8);
390                         Map<String, Object> responseMap = objectMapper.readValue(entityStr);
391                         responseRecords = (List<Map<String, Object>>) responseMap.get(RECORDS_KEY);
392                         if (responseRecords != null) {
393                             handleRecords(responseRecords, inputDocList);
394                         } else {
395                             String error = (String) responseMap.get(QUERY_ERROR_KEY);
396                             if (error != null) {
397                                 throw new SnapDataException(ERR_FETCHTING_DATA)
398                                         .withReason(error)
399                                         .withResolution(RESOLUTION_ADDRESS_ISSUE);
400                             }
401                         }
402                     }
403                 }
404             }
405         } catch (ClientProtocolException e) {
406             throw new ExecutionException(e, HTTP_PROTOCOL_ERROR);
407         } catch (IOException e) {
408             throw new ExecutionException(e, IO_ERROR_REQUEST_EXEC);
409         } finally {
410             if (response != null) {
411                 try {
412                     EntityUtils.consume(response.getEntity());
413                 } catch (IOException e) {
414                     LOG.error(Throwables.getRootCause(e).getMessage(), e);
415                 }
416             }
417         }
418     }

433     protected void handleRecords(List<Map<String, Object>> records, List<Document> inputDocList) {
434         filterErrorDataFromResponse(records);
435         int index = 0;
436         for (Object record : records) {
437             Document responseDoc = documentUtility.newDocument(record);
438              if (inputDocList != null && inputDocList.size() == records.size()) {
439                  if (Action.DELETE == this.action) {
440                      responseDoc = documentUtility.newDocumentFor
441                              (inputDocList.get(index++), record);
442                      outputViews.write(responseDoc);
443                  } else {
444                      outputViews.write(responseDoc, inputDocList.get(index++));
445                  }
446             } else {
447                 outputViews.write(responseDoc);
448             }
449             recordCounter.inc();
450         }
451     }

453     protected void filterErrorDataFromResponse(List<Map<String, Object>> responseRecords) {
454         List<Map<String,Object>> tempList = Lists.newArrayList();
455         for (Map<String, Object> record: responseRecords) {
456             if (record.containsKey(STATUS_KEY) && record.get(STATUS_KEY).equals(FAILURE)) {
457                 String errorMessage = ERR_INSERTING_RECORDS_REASON;
458                 String errorReason = ERR_INSERTING_RECORDS_REASON;
459                 tempList.add(record);
460                 Map<String, String> errorInfo = (Map)record.get(ERROR_KEY);
461                 if(errorInfo != null ) {
462                     if (errorInfo.get(REASON_KEY) != null) {
463                         errorReason = errorMessage + SPACE + errorInfo.get(REASON_KEY);
464                     }
465 
466                     if (errorInfo.get(MESSAGE_KEY) != null) {
467                         errorMessage = errorMessage + SPACE + errorInfo.get(MESSAGE_KEY);
468                     }
469                 }
470                 SnapDataException insertException = new SnapDataException(
471                         errorMessage)
472                         .withReason(errorReason)
473                         .withResolution(ERR_INSERTING_RECORDS_RESOLUTION);
474                 if (errorViews.getErrorBehaviour() == PropertyCategory.ErrorBehaviourType.FAIL) {
475                     throw insertException;
476                 } else {
477                     errorViews.write(insertException);
478                 }
479             }
480         }
481         responseRecords.removeAll(tempList);
482     }
