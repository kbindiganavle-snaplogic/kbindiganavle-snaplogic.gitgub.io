105 /**
106  * Snap to perform bulk load onto oracle server using sqlldr.
107  * <p>
108  * Implementation note: the original implementation created a one-line shell script
109  * and executed that. This required a fair amount of code to escape special characters
110  * and resulted in unexpected behavior if the max. number of errors occurred but an
111  * error view was not defined. (See SNAP-2443.)
112  * <p>
113  * The code was changed to call Runtime.exec(String[], String[], File). This eliminates
114  * the need to escape special characters and, somehow, resulted in error described in
115  * SNAP-2443 going away.
116  * <p>
117  * Implementation note: all of the temporary files needed by the snap are put into a single
118  * subdirectory. This offers improved security if we need it in the future. (E.g., we might
119  * want to change the directory permissions to limit access to us in case the discard file
120  * contains sensitive information.) That's not done at this time.
121  * <p>
122  * Implementation note: the temporary directory name no longer contains a random string.
123  * Instead it contains the timestamp and pipeline's RUUID. The former makes it much easier
124  * to identify stale files if they're not properly deleted and the latter makes it easy to
125  * associate the temp files with the logs (if they were not deleted).
126  */

400     protected void configureSnap(PropertyValues propertyValues) {
401         super.configureSnap(propertyValues);
402         loaderState = LoaderState.DEFAULT;
403         ExpressionProperty sqlldrPathExpression =
404                 propertyValues.getAsExpression(SQLLDR_ABSOLUTE_PATH_PROP);
405         if (sqlldrPathExpression != null) {
406             sqlldrPath = sqlldrPathExpression.eval(null);
407         }
408         insertMode = propertyValues.get(INSERT_MODE_PROP);
409         maxError = ((BigInteger) propertyValues.get(MAX_ERROR_PROP)).intValue();
410         useDirectPathLoad = Boolean.TRUE.equals(propertyValues.get(DIRECT_PATH_LOAD_PROP));
411         skipVirtualColumns = Boolean.TRUE.equals(propertyValues.get(SKIP_VIRTUAL_COLUMNS_PROP));
412         tempDir = JfsUtils.getTmpDir(expressionUtils, propertyValues, errorViews);
413 
414         // put all of the temporary files in a subdirectory. This is a little more secure
415         // than putting them in under /tmp directly since we can control file permissions
416         // easier (if necessary). Directory contains timestamp so it's easier to identify
417         // stale files and the thread name so we can support more than one BulkLoad snap
418         // per pipeline.
419         DateFormat df = new SimpleDateFormat(SQLLDR_DATE_FORMAT);
420         timestamp = df.format(new Date());
421         String threadName = Thread.currentThread().getName().replaceAll(" ", "_");
422         rootDir = new File(tempDir, String.format("%s-%s", timestamp, threadName));
423         if (!rootDir.mkdirs()) {
424             throw new ConfigurationException("Unable to create rootDir=%s").formatWith(rootDir);
425         }
426 
427         // sqlldrPath is mandatory in new snaps but legacy pipelines might have
428         // used the 'which'/'where' feature. Check for a blank field and use a default
429         // value if possible. (Use the first match if there is more than one - this matches
430         // the behavior of 'which'/'where'. This isn't 100% equivalent since an 'alias'
431         // could change the behavior but that is unlikely on a server.
432         if (sqlldrPath == null || sqlldrPath.isEmpty()) {
433             if (SqlLoaderPathSuggestions.SUGGESTIONS.length > 0) {
434                 sqlldrPath = SqlLoaderPathSuggestions.SUGGESTIONS[0];
435             } else {
436                 // Throw exception here since we'll get a NPE if we continue.
437                 throw new ConfigurationException(ERR_SQLLDR_PATH_NOT_EXECUTABLE);
438             }
439         }
440 
441         // verify sqlldrPath points to executable file.
442         File sqlldr = new File(sqlldrPath);
443         if (!sqlldr.exists() || !sqlldr.canExecute() || (sqlldr.length() == 0)) {
444             throw new ConfigurationException(ERR_SQLLDR_PATH_NOT_EXECUTABLE);
445         }
446 
447         List<String> viewOrder = viewNameUtil.sortViewNames(inputViews.names(),
448                 INPUT_ROWS_VIEW_NAME);
449         if (!viewOrder.isEmpty()) {
450             rowViewName = viewOrder.get(0);
451             if (viewOrder.size() > 1) {
452                 schemaViewName = viewOrder.get(1);
453             }
454         }
455         params = new HashSet<>();
456         params.add(String.format(PARAM_PATTERN, ERRORS_PARAM, maxError));
457         // get file permission set data from properties
458         List<Map<String, Object>> additonalParams = propertyValues
459                 .get(ADDITIONAL_PARAM_PROP);
460         if (additonalParams != null && !additonalParams.isEmpty()) {
461             for (Map<String, Object> entry : additonalParams) {
462                 String key = (String) (((Map<String, String>) entry.get(PARAM_PROP)).get(VALUE));
463                 String value = null;
464                 ExpressionProperty valueExpression =
465                         propertyValues.getExpressionPropertyFor(entry, VALUE_PROP);
466                 if (valueExpression != null) {
467                     value = valueExpression.eval(null);
468                 }
469                 if (key.equals(RESUMABLE_NAME_PARAM) & !value.startsWith(QUOTE_CHAR)) {
470                     // single quote is required for RESUMABLE_NAME param
471                     value = QUOTE_CHAR + value + QUOTE_CHAR;
472                 }
473                 if (StringUtils.isEmpty(key) || StringUtils.isEmpty(value)) {
474                     throw new ConfigurationException(String.format(ERR_INVALID_ENTRY, key))
475                             .withReason(REASON_INVALID_ENTRY)
476                             .withResolution(RESOLUTION_INVALID_ENTRY);
477                 } else if (key.equals(SILENT_PARAM) && !SILENT_PARAMS_OPT.contains(value)) {
478                     throw new ConfigurationException(String.format(ERR_INVALID_ENTRY, key))
479                             .withReason(String.format(REASON_INVALID_VALUE, SILENT_PARAMS_OPT))
480                             .withResolution(RESOLUTION_INVALID_ENTRY);
481                 }
482                 params.add(String.format(PARAM_PATTERN, key, value));
483             }
484         }
485 
486         configureColumnDefinitions(propertyValues);
487     }

513     @Override
514     public void execute() {
515         if (inputViews.isEmpty()) {
516             return;
517         }
518         Connection conn = jdbcOperations.acquireConnection(account);
519         if (jdbcOperations.tableExists(conn, account.getDatabaseConfig().getDatabaseName(),
520                 tableName)) {
521             tableMetaData = jdbcOperations.getTableMetaData(conn,
522                     account.getDatabaseConfig().getDatabaseName(), tableName);
523         }
524         for (String viewName : getInputViewOrder()) {
525             for (Document document : inputViews.documents(inputViews.get(viewName))) {
526                 process(document, viewName);
527             }
528         }
529 
530         if (!firstRecord && autoCreateTable && schema != null) {
531             // We want to create the table even if there are no rows to be loaded.
532             createTableIfAbsent(null);
533         }
534 
535         if (IS_WIN_PLEX && successful && dataWriter != null) {
536             loadAction();
537         }
538 
539         // Wait for sqlldr to finish if loading was started successfully.
540         if (loaderState == LoaderState.STARTED) {
541             // Wait for the sqlldr to finish its job in another thread so that the snap can be
542             // stopped immediately when manually stopping the pipeline.
543             final CountDownLatch latch = new CountDownLatch(1);
544             Thread waitThread = new Thread(new Runnable() {
545                 @Override
546                 public void run() {
547                     try {
548                         loaderProcess.waitFor();
549                     } catch (InterruptedException e) {
550                         // ignore
551                     } finally {
552                         latch.countDown();
553                     }
554                 }
555             });
556             waitThread.start();
557             if (!IS_WIN_PLEX) {
558                 // Close writer of the named pipe so that sqlldr won't wait for input forever.
559                 IOUtils.closeQuietly(dataWriter);
560             }
561             try {
562                 latch.await();
563                 loaderState = LoaderState.SUCCEEDED;
564             } catch (InterruptedException e) {
565                 successful = false;
566             }
567         }
568     }

570     @Override
571     protected void processDocument(final Document document, final String inputViewName) {
572         // Check if this is the schema document and save it for later.
573         if (inputViewName != null && inputViewName.equals(schemaViewName)) {
574             if (document != null) {
575                 schema = document;
576                 columnTranslation = databaseUtils.buildColumnTranslation(schema);
577             }
578             return;
579         }
580         if (document == null || !successful) {
581             return;
582         }
583         if (firstRecord) {
584             firstRecord = false;
585             if (autoCreateTable) {
586                 createTableIfAbsent(document);
587             }
588             if (!setInputSchemaFields()) {
589                 successful = false;
590                 return;
591             }
592             preLoadAction();
593         }
594         if (loaderState == LoaderState.FAILED) {
595             return;
596         }
597         try {
598             StringBuilder dataBuilder = new StringBuilder();
599             if (document.get() instanceof Map) {
600                 @SuppressWarnings("unchecked")
601                 Map<String, Object> record = (Map<String, Object>) document.get();
602                 if (keySet == null) {
603                     keySet = record.keySet();
604                 }
605                 Object value;
606                 for (int i = 0; i < inputViewFields.size(); i++) {
607                     String actualKey = inputViewFields.get(i).getKey();
608                     for (String key : keySet) {
609                         if (key.equalsIgnoreCase(actualKey)) {
610                             value = record.get(key);
611                             dataBuilder.append((value == null) ? StringUtils.EMPTY :
612                                     convertValue(value, actualKey));
613                             break;
614                         }
615                     }
616                     if (i < (inputViewFields.size() - 1)) {
617                         dataBuilder.append(FIELD_DELIMITER);
618                     }
619                 }
620             }
621             dataBuilder.append(RECORD_SEPARATOR);
622             dataWriter.write(dataBuilder.toString());
623             dataWriter.flush();
624         } catch (IOException e1) {
625             SnapDataException dataException = new SnapDataException(e1,
626                     ERR_WRITTING_INPUT_TO_DATA_FILE)
627                     .withReason(e1.getMessage());
628             if (StringUtils.containsIgnoreCase(e1.getMessage(), NOT_ENOUGH_SPACE)) {
629                 dataException.withResolution(ERR_NOT_ENOUGH_SPACE_RESOLUTION);
630             } else if (StringUtils.containsIgnoreCase(e1.getMessage(), ERR_BROKEN_PIPE)) {
631                 dataException.withResolution(RESOLUTION_BROKEN_PIPE);
632             } else  {
633                 dataException.withResolutionAsDefect();
634             }
635             errorViews.write(dataException);
636         }
637     }

651     /**
652      * Convert the input object to string
653      */
654     String convertValue(Object object, String actualKey) {
655         String ret;
656         if (object instanceof DateTime) {
657             if (MapUtils.isNotEmpty(tableMetaData) &&
658                     (tableMetaData.get(actualKey).getDataType()
659                             == TYPE_TIMESTAMP_WITH_LOCAL_TIME_ZONE
660                             || tableMetaData.get(actualKey).getDataType() == Types.TIMESTAMP
661                             || tableMetaData.get(actualKey).getDataType() == Types.DATE)) {
662                 ret = ((DateTime) object).toString(DateTimeFormat.forPattern
663                         (PATTERN_TIMESTAMP_JODA));
664             } else {
665                 ret = ((DateTime) object).toString(DateTimeFormat.forPattern
666                         (PATTERN_TIMESTAMP_WITH_TIME_ZONE_JODA));
667             }
668         } else if (object instanceof LocalDateTime) {
669             ret = ((LocalDateTime) object).toString(DateTimeFormat.forPattern
670                     (PATTERN_TIMESTAMP_JODA));
671         } else if (object instanceof Map) {
672             ret = convertDateTime((Map) object);
673         } else if (MapUtils.isNotEmpty(tableMetaData)) {
674             int dataType = tableMetaData.get(actualKey).getDataType();
675             // If the dataType is 1111, then we need to check with columnTypeName
676             String columnTypeName = tableMetaData.get(actualKey).getColumnTypeName();
677             if (dataType == Types.DATE) {
678                 ret = formatter.parseDateTime(object.toString())
679                         .toString(DateTimeFormat.forPattern(PATTERN_TIMESTAMP_JODA));
680             } else if (dataType == Types.TIMESTAMP || (dataType == OTHER_DATA_TYPE
681                     && columnTypeName.matches(TIMESTAMP_PATTERN))) {
682                 ret = JavaTimeUtils.parseToLocalDateTimeString(object.toString());
683             } else if (dataType == TYPE_TIMESTAMP_WITH_TIME_ZONE
684                     || (dataType == OTHER_DATA_TYPE
685                     && columnTypeName.matches(TIMESTAMP_TZ_PATTERN))) {
686                 ret = parseDateTimeWithTimeZone(object.toString());
687             } else if (dataType == TYPE_TIMESTAMP_WITH_LOCAL_TIME_ZONE
688                     || (dataType == OTHER_DATA_TYPE
689                     && columnTypeName.matches(TIMESTAMP_LTZ_PATTERN))) {
690                 ret = parseDateTimeToLocalDateTime(object.toString());
691             } else {
692                 ret = object.toString();
693             }
694         } else {
695             ret = object.toString();
696         }
697         return escapeInput(ret);
698     }

700     String convertDateTime(Map<Object, Object> map) {
701         for (Map.Entry<Object, Object> entry : map.entrySet()) {
702             Object key = entry.getKey();
703             Object val = entry.getValue();
704             if (key instanceof String && val instanceof String) {
705                 String value = (String) val;
706                 switch ((String) key) {
707                     case SNAP_TYPE_DATE_TIME:
708                         return formatter.parseDateTime(value).toString(DateTimeFormat.forPattern
709                                 (PATTERN_TIMESTAMP_WITH_TEXT_TIME_ZONE_JODA));
710                     case SNAP_TYPE_LOCAL_DATE_TIME:
711                         return formatter.parseLocalDateTime(value)
712                                 .toString(DateTimeFormat.forPattern(PATTERN_TIMESTAMP_JODA));
713                     case SNAP_TYPE_LOCAL_DATE:
714                         return formatter.parseLocalDate(value)
715                                 .toString(DateTimeFormat.forPattern(PATTERN_DATE_JODA));
716                     case SNAP_TYPE_LOCAL_TIME:
717                         return formatter.parseLocalTime(value).toString(DateTimeFormat.forPattern
718                                 (PATTERN_TIME_JODA));
719                 }
720             }
721         }
722         return map.toString();
723     }

768     private void preLoadAction() {
769         // Creating data file
770         dataFile = new File(rootDir, DATA_FILENAME);
771         // generate Discard file
772         discardFile = new File(rootDir, DISCARD_FILENAME);
773         // generate bad file name
774         badFile = new File(rootDir, BAD_FILENAME);
775         if (!IS_WIN_PLEX) {
776             // Note: 0600 = owner read/write, nothing else
777             JnaLib.mkfifo(dataFile.getAbsolutePath(), 0600);
778         }
779 
780         // creating control file
781         controlFile = new File(rootDir, CONTROL_FILENAME);
782         BufferedWriter controlWriter = null;
783         try {
784             String options = params.toString();
785             String sqlldrCmd = String.format(LOAD_COMMAND,
786                     options.substring(1, options.length() - 1)
787                     , dataFile, badFile, discardFile, insertMode, tableName, FIELD_DELIMITER,
788                     FIELD_ENCLOSING_CHAR, joinInputViewFields());
789             LOGGER.debug(sqlldrCmd);
790             controlWriter = new BufferedWriter(new FileWriter(controlFile));
791             controlWriter.write(sqlldrCmd);
792             controlWriter.flush();
793         } catch (IOException e2) {
794             SnapDataException dataException = new SnapDataException(e2,
795                     ERR_CREATE_CONTROL_FILE_MSG)
796                     .withReason(e2.getMessage());
797             if (e2.getMessage().toLowerCase().contains(NOT_ENOUGH_SPACE.toLowerCase())) {
798                 dataException = dataException.withResolution(ERR_NOT_ENOUGH_SPACE_RESOLUTION);
799             } else {
800                 dataException = dataException.withResolutionAsDefect();
801             }
802             errorViews.write(dataException);
803         } finally {
804             IOUtils.closeQuietly(controlWriter);
805         }
806 
807         // setting up log file
808         logFile = new File(rootDir, LOG_FILENAME);
809         // generate bad file name
810         badFile = new File(rootDir, BAD_FILENAME);
811         DatabaseConfig databaseConfig = account.getDatabaseConfig();
812 
813         // SQLLDR format used for loading the data::
814         // sqlldr userid=user/pass@"(DESCRIPTION\=(ADDRESS\=(PROTOCOL\=TCP)(HOST\=host)(PORT\=port))
815         // (CONNECT_DATA\=(SERVER\=DEDICATED)(SERVICE_NAME\=sname)))" control=ctl_filename
816         // log=file_name
817         String jdbcURLSQLLDRFormat = databaseConfig.getJdbcUrl().substring(OracleThinDatabaseAccount
818                 .JDBC_PREFIX.length());
819         jdbcURLSQLLDRFormat = jdbcURLSQLLDRFormat.replace(EQUALS, REPLACE_EQUALS);
820         LOGGER.debug(String.format("jdbc url format for sqlldr: %s", jdbcURLSQLLDRFormat));
821         sqlldrArgs.add(sqlldrPath);
822         sqlldrArgs.add(String.format("userid=%s/\"%s\"@\"%s\"", databaseConfig.getUsername(),
823                 databaseConfig.getPassword(), jdbcURLSQLLDRFormat));
824         sqlldrArgs.add(String.format("control=\"%s\"", controlFile));
825         sqlldrArgs.add(String.format("log=\"%s\"", logFile));
826         if (useDirectPathLoad) {
827             sqlldrArgs.add(SQLLDR_COMMAND_DIRECT);
828         }
829         if (IS_WIN_PLEX) {
830             try {
831                 dataWriter = new BufferedWriter(new OutputStreamWriter(new FileOutputStream
832                         (dataFile), StandardCharsets.UTF_8));
833             } catch (IOException e) {
834                 throw new ExecutionException(e, ERR_CREATE_DATA_FILE_MSG)
835                         .withReason(e.getMessage())
836                         .withResolution(PLEASE_ADDRESS_REPORTED_ISSUE);
837             }
838         } else {
839             // Reader of the named pipe must be created before writer.
840             loadAction();
841 
842             // The call to open a writer of a named pipe will block the thread until the named pipe
843             // has a reader, so we do it in a separate thread, give sqlldr time to initialize and
844             // close the thread when thing goes wrong.
845             Thread waitThread = new Thread(() -> {
846                 try {
847                     dataWriter = new BufferedWriter(new OutputStreamWriter(new FileOutputStream
848                             (dataFile), StandardCharsets.UTF_8));
849                 } catch (IOException e) {
850                     throw new ExecutionException(e, ERR_CREATE_DATA_FILE_MSG)
851                             .withReason(e.getMessage())
852                             .withResolution(PLEASE_ADDRESS_REPORTED_ISSUE);
853                 }
854             });
855             waitThread.start();
856             long waitTime = 0;
857             while (waitTime < TIMEOUT_FOR_OPENING_WRITTER_IN_MS) {
858                 try {
859                     Thread.sleep(INTERVAL_IN_MS);
860                     try {
861                         loaderProcess.exitValue();
862                         // Loader didn't run successfully, stop running
863                         loaderState = LoaderState.FAILED;
864                         break;
865                     } catch (IllegalThreadStateException e) {
866                         // No op
867                     }
868                     if (!waitThread.isAlive()) {
869                         break;
870                     }
871                     waitTime += INTERVAL_IN_MS;
872                 } catch (InterruptedException e) {
873                     LOGGER.error(INTERRUPTED_EXCEPTION_MESSAGE, e);
874                 }
875             }
876             if (waitTime >= TIMEOUT_FOR_OPENING_WRITTER_IN_MS) {
877                 // Timeout exceeded , stop the snap
878                 waitThread.interrupt();
879                 if (loaderState != LoaderState.FAILED) {
880                     successful = false;
881                 }
882             } else {
883                 if (loaderState == LoaderState.DEFAULT) {
884                     // success
885                     loaderState = LoaderState.STARTED;
886                 } else {
887                     // Loader failed initializing
888                     waitThread.interrupt();
889                 }
890             }
891         }
892     }

894     private void loadAction() {
895         try {
896             Runtime runtime = Runtime.getRuntime();
897             loaderProcess = runtime.exec(sqlldrArgs.toArray(new String[0]), null, rootDir);
898             if (IS_WIN_PLEX) {
899                 loaderState = LoaderState.STARTED;
900             }
901             if (loaderProcess != null) {
902                 // Runtime.exec() has limited buffer size for stdin and stdout streams, and
903                 // the external process will get stuck when the buffer is full, so we keep
904                 // cleaning up the stdout buffer to keep sqlldr from hanging.
905                 new Thread(() -> {
906                     try (BufferedReader reader = new BufferedReader(new InputStreamReader(
907                             loaderProcess.getInputStream()))) {
908                         String line;
909                         while ((line = reader.readLine()) != null) {
910                             LOGGER.debug(SQLLDR_OUTPUT_FMT, line);
911                         }
912                     } catch (IOException e) {
913                         LOGGER.error(e.getLocalizedMessage(), e);
914                     }
915                 }).start();
916             }
917         } catch (Exception e5) {
918             SnapDataException dataException = new SnapDataException(e5,
919                     ERR_EXECUTE_SQLLDR_CMD_MSG)
920                     .withReason(e5.getMessage())
921                     .withResolution(ERR_EXECUTE_SQLLDR_CMD_RESOLUTION);
922             errorViews.write(dataException);
923         }
924     }

926     private void postLoadAction() {
927         // check if we actually ran...
928         if (loaderProcess == null) {
929             return;
930         }
931 
932         if (loaderState != LoaderState.DEFAULT) {
933             if (loaderState != LoaderState.FAILED && (dataWriter == null || !successful)) {
934                 loaderProcess.destroy();
935                 return;
936             }
937             StringBuilder output = new StringBuilder();
938             try (BufferedReader reader = new BufferedReader(
939                     new InputStreamReader(loaderProcess.getInputStream()))) {
940                 String line;
941                 while ((line = reader.readLine()) != null) {
942                     LOGGER.debug(SQLLDR_OUTPUT_FMT, line);
943                 }
944             } catch (IOException e) {
945                 // safe to ignore - just copying logging info.
946                 LOGGER.error(e.getLocalizedMessage(), e);
947             }
948 
949             try (BufferedReader reader = new BufferedReader(
950                     new InputStreamReader(loaderProcess.getErrorStream()))) {
951                 String line;
952                 boolean isNotFirstLine = false;
953                 while ((line = reader.readLine()) != null) {
954                     if (isNotFirstLine) {
955                         output.append(TAB);
956                     } else {
957                         isNotFirstLine = true;
958                     }
959                     output.append(line);
960                 }
961                 reader.close();
962                 String errorInfo = output.toString();
963 
964                 boolean isErrorOccurred = false;
965                 if (StringUtils.isNotBlank(errorInfo)) {
966                     isErrorOccurred = true;
967                 }
968 
969                 /**
970                  * Log file parsing states:
971                  * INIT -->[DISCARDED_RECORDS]-->[REJECTED_RECORDS]--->SUMMARY
972                  */
973                 if (!isErrorOccurred && loaderState == LoaderState.SUCCEEDED) {
974                     String summaryString;
975                     LogParsingState state = LogParsingState.INIT;
976                     BufferedReader discardReader = null;
977                     BufferedReader badReader = null;
978                     try (BufferedReader br = new BufferedReader(new FileReader(logFile))) {
979                         StringBuilder builder = new StringBuilder();
980                         String logLine = br.readLine();
981                         Pattern discardPattern = Pattern.compile(DISCARDED_LOG_REGEX);
982                         Pattern rejectPattern = Pattern.compile(REJECTED_LOG_REGEX);
983                         Pattern summaryPattern = Pattern.compile(SUMMARY_LOG_REGEX);
984                         while (logLine != null) {
985                             if (StringUtils.isBlank(logLine)) {
986                                 logLine = br.readLine();
987                                 continue;
988                             }
989                             switch (state) {
990                                 case INIT:
991                                     if (discardPattern.matcher(logLine).matches()) {
992                                         state = LogParsingState.DISCARDED;
993                                         builder.append(logLine);
994                                         discardReader = new BufferedReader(new FileReader(
995                                                 discardFile));
996                                     } else if (rejectPattern.matcher(logLine).matches()) {
997                                         state = LogParsingState.REJECTED;
998                                         builder.append(logLine);
999                                         badReader = new BufferedReader(new FileReader(badFile));
1000                                     } else if (summaryPattern.matcher(logLine).matches()) {
1001                                         state = LogParsingState.SUMMARY;
1002                                         builder.append(logLine);
1003                                     }
1004                                     break;
1005                                 case DISCARDED:
1006                                     if (discardPattern.matcher(logLine).matches()) {
1007                                         // new discarded record message
1008                                         writeToErrorView(builder.toString(),
1009                                                 discardReader.readLine());
1010                                         builder.setLength(0);
1011                                         builder.append(logLine);
1012                                     } else if (rejectPattern.matcher(logLine).matches()) {
1013                                         // new rejected record message
1014                                         writeToErrorView(builder.toString(),
1015                                                 discardReader.readLine());
1016                                         builder.setLength(0);
1017                                         builder.append(logLine);
1018                                         state = LogParsingState.REJECTED;
1019                                         badReader = new BufferedReader(new FileReader(badFile));
1020                                     } else if (summaryPattern.matcher(logLine).matches()) {
1021                                         // no more error messages
1022                                         writeToErrorView(builder.toString(),
1023                                                 discardReader.readLine());
1024                                         builder.setLength(0);
1025                                         builder.append(logLine);
1026                                         state = LogParsingState.SUMMARY;
1027                                     } else {
1028                                         // multiple-line error message
1029                                         builder.append(LINE_BREAK).append(logLine);
1030                                     }
1031                                     break;
1032                                 case REJECTED:
1033                                     if (rejectPattern.matcher(logLine).matches()) {
1034                                         // new rejected record message
1035                                         writeToErrorView(builder.toString(), badReader.readLine());

1239     /**
1240      * 1. Duplicate enclose characters in input string.<br>
1241      * http://docs.oracle.com/cd/B19306_01/server.102/b14215/ldr_field_list.htm#sthref1092
1242      * <p>
1243      * 2. Quote the input string that contains field delimiter.
1244      * 3. Quote any string that has leading or trailing spaces. If we don't SQL*Loader will
1245      * trim the strings and we won't have the expected results. This is esp. critical since
1246      * Oracle treats an empty string as equivalent to a NULL value and it's common for
1247      * developers to use a string containing a single space as a replacement for an empty
1248      * string. Without quotes it will truncate that string and we'll get an unwanted result.
1249      * <p>
1250      * Note: it's also valid to simply always use quotes. If so we need to be careful to verify
1251      * that the parser will recognize ,"", as an empty string and not try to interpret it as ,",
1252      *
1253      * @param s input string
1254      * @return escaped string
1255      */
1256     String escapeInput(final String s) {
1257         if (s.isEmpty()) {
1258             return StringUtils.EMPTY;
1259         }
1260         String newStr = StringUtils.replace(s, FIELD_ENCLOSING_CHAR, FIELD_ENCLOSING_CHAR +
1261                 FIELD_ENCLOSING_CHAR);
1262         if (StringUtils.containsAny(newStr, FIELD_DELIMITER + FIELD_ENCLOSING_CHAR)
1263                 || newStr.startsWith(SPACE) || newStr.endsWith(SPACE)) {
1264             newStr = String.format(ENCLOSE_FMT, newStr);
1265         }
1266         return newStr;
1267     }

1293         private static boolean checkFile(File file) {
1294             try {
1295                 return file.exists() && file.canExecute() && (file.length() > 0);
1296             } catch (Throwable throwable) {
1297                 // The sqlldr executable utility pointed to by the @file parameter here may not be
1298                 //  accessible due to no required permissions set, or a security policy in place.
1299                 // In that case, it's better to just suppress this exception instead of floating
1300                 //  it up which will eventually lead to the failure of the loading of this class
1301                 //  thereby resulting in not loading of the default bundled driver JAR.  Though
1302                 //  the default JAR is bundled in the snappack, and sqlldr utility's path is also
1303                 //  provided by the user, still the snap fails saying
1304                 //  "unable to load the default driver".  This leads to a poor user experience.
1305                 // And, this suppressing of the exception is done as part of the SNAP-11353 bug.
1306                 LOGGER.debug("Encountered error '{}' while trying to check the execution " +
1307                         "ability on the utility {}", throwable.getMessage(),
1308                         file.getAbsolutePath());
1309                 return false;
1310             }
1311         }

1323                         list.add(file.getAbsolutePath());

1328                 File file = new File(ORACLE_HOME + FILE_SEPARATOR + "bin" + FILE_SEPARATOR +
1329                         SQLLDR_NAME);
1330                 if (checkFile(file) && !list.contains(file.getAbsolutePath())) {
1331                     list.add(file.getAbsolutePath());
1332                 }
1333             }
1334             return list.toArray(new String[0]);
