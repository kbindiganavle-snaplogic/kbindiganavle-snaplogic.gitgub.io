446     public void suggestFileFormatOptions(SuggestionBuilder suggestionBuilder,
447                                          PropertyValues propertyValues) {
448         String fileFormatType = propertyValues.getAsExpression(FILE_FORMAT_TYPE_PROP).eval(null);
449         Set<String> fileFormatOpts;
450         switch (fileFormatType) {
451             case "CSV":
452                 fileFormatOpts = DLP_FILE_FORMAT_CSV_OPTIONS;
453                 break;
454             case "JSON":
455                 fileFormatOpts = DLP_FILE_FORMAT_JSON_OPTIONS;
456                 break;
457             case "PARQUET":
458                 fileFormatOpts = DLP_FILE_FORMAT_PARQUET_OPTIONS;
459                 break;
460             case "ORC":
461                 fileFormatOpts = DLP_FILE_FORMAT_ORC_OPTIONS;
462                 break;
463             case "TEXT":
464                 fileFormatOpts = DLP_FILE_FORMAT_TEXT_OPTIONS;
465                 break;
466             case "AVRO":
467                 fileFormatOpts = DLP_FILE_FORMAT_AVRO_OPTIONS;
468                 break;
469             default:
470                 throw new ConfigurationException(ERR_SUGGESTING_FILE_FORMAT_OPT)
471                         .withReason(ERR_SUGGESTING_FILE_FORMAT_OPT_REASON)
472                         .withResolution(ERR_SUGGESTING_FILE_FORMAT_OPT_RESOLUTION);
473         }
474 
475         suggestionBuilder.node(FILE_FORMAT_OPTIONS_LIST_PROP)
476                 .over(FILE_FORMAT_OPTION_PROP)
477                 .suggestions(fileFormatOpts.toArray(
478                         new String[0]));
479     }
