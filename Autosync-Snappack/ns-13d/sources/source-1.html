171             if (StringUtils.equals(httpMethod, HttpPost.METHOD_NAME)
172                     || StringUtils.equals(httpMethod, HttpGet.METHOD_NAME)) {

178                             .withReason(e1.getMessage())
179                             .withResolution(RESOLUTION_CUSTOMER_SUPPORT);
180                 }
181             } else if (StringUtils.equals(httpMethod, HttpPut.METHOD_NAME) ||
182                     StringUtils.equals(httpMethod, HttpPatch.METHOD_NAME)) {
183                 return new RestResponseObject(httpResponse.getStatusLine());
184             }
185             return new RestResponseObject(map, httpResponse.getStatusLine());
186         } else {
187             LOGGER.error(ERR_EXECUTING_HTTP_REQUEST);
188             throw new ExecutionException(ERR_EXECUTING_HTTP_REQUEST)
189                     .withResolution(RESOLUTION_CUSTOMER_SUPPORT);

568      * validateAndProcessResults
569      *
570      * @param url
571      * @param startTimeInMs
572      * @throws InterruptedException
573      */
574     private void validateAndProcessResults(String url, long startTimeInMs)
575             throws InterruptedException {
576         switch (bulkJobInfo.getState()) {
577             case INPROGRESS:
578             case UPLOAD_COMPLETE:
579                 TimeUnit.SECONDS.sleep(bulkJobInfo.getPollingIntervalSec());
580                 if ((System.currentTimeMillis() - startTimeInMs) >
581                         (bulkJobInfo.getPollingTimeoutSec() * 1000)) {
582                     abortJob(url);
583                     processError(Options.State.TIMEOUT, bulkJobInfo);
584                 } else {
585                     verifyJobStatus(url, startTimeInMs);
586                 }
587                 break;
588             case JOB_COMPLETE:
589                 if (bulkJobInfo.getOperation().isQuery()) {
590                     bulkJobInfo.setContentUrl(url);
591                 } else {
592                     bulkJobInfo.getResultsMap().put(SUCCESSRECORDS,
593                             getResults(url + SUCCESS_RESULTS));
594                     bulkJobInfo.getResultsMap().put(FAILEDRECORDS,
595                             getResults(url + FAILED_RESULTS));
596                 }
597                 break;
598             case FAILED:
599                 if (!bulkJobInfo.getOperation().isQuery()) {
600                     bulkJobInfo.getResultsMap().put(UNPROCESSEDRECORDS,
601                             getResults(url + UNPROCESSED_RESULTS));
602                 }
603                 break;
604             default:
605         }
606     }

742      * @return
743      */
744     public List<Map<String, String>> convertResponseToMap(String resultsString) {
745         List<Map<String, String>> outputRecordsList = Lists.newArrayList();
746         CSVFormat.Builder csvFormatBuilder =
747                 CSVFormat.Builder.create(CSVFormat.DEFAULT).setHeader();
748         if (isDisplayEmptyAsNull) {
749             //salesforce Bulk API return for both empty string and null value as empty String ("").
750             // so,here setting the empty string as null value in csv formatter to display
751             // empty string as null in response.
752             csvFormatBuilder.setNullString("");
753         }
754         CSVFormat csvFormat = csvFormatBuilder.build();
755         StringReader reader = new StringReader(resultsString);
756         try (CSVParser parser = new CSVParser(reader, csvFormat)) {
757             Map<String, Integer> headers = parser.getHeaderMap();
758             for (CSVRecord record : parser) {
759                 Map<String, String> outputMap = getMapFrom(headers, record);
760                 if (outputMap != null) {
761                     outputRecordsList.add(outputMap);
762                 }
763             }
764         } catch (SuggestViewAbortException e) {
765             throw e;
766         } catch (Exception e) {
767             LOGGER.error(ERR_CONSTRUCTING_RESULTS, e);
768             throw new SnapDataException(e, ERR_CONSTRUCTING_RESULTS)
769                     .withReason(e.getMessage())
770                     .withResolution(RESOLUTION_CUSTOMER_SUPPORT);
771         }

775     public Map<String, String> getMapFrom(Map<String, Integer> headers, CSVRecord record) {
776         if (headers != null && !headers.isEmpty() && record.isConsistent()) {
777             // CSVRecord#toMap() returns a HashMap object, where the columns are in a
778             // random order. This could be a defect in the current version of CSVRecord.
779             // It should be converted to LinkedHashMap as a workaround.
780             Map<String, String> recordMap = record.toMap();
781             Map<String, String> outputMap = new LinkedHashMap<>(recordMap.size());
782             for (String header : headers.keySet()) {
783 
784                 if (header.startsWith(BULK2_FIELD_PREFIX)) {
785                     outputMap.put(header.replace(BULK2_FIELD_PREFIX, StringUtils.EMPTY),
786                             recordMap.get(header));
787                 } else {
788                     outputMap.put(header, recordMap.get(header));
789                 }
790 
791             }
792             return outputMap;
793         }
794         return null;
795     }
