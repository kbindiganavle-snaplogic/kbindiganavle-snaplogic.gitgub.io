347                                 ODataClient client) {

350             Document outDoc = documentUtility.newDocumentFor(inputDocument, record);

352         }
353         if (map.get(NEXT_PAGE) != null) {

355                 return;
356             } else {
357                 startPage++;
358                 String nextLink = (String) map.get(NEXT_PAGE);

360                 Map<String, Object> nextMap = oDataUtils.executeRequestWithRetry(
361                         uri.build(), account, pageSize, httpHeaders, retryCount,
362                         retryInterval);
363                 processRecords(nextMap, inputDocument, client);
364             }
365         }

437     private void setFilter(Document document) {
438         if (filterExprList != null) {
439             StringJoiner builder = new StringJoiner(SPACE_CHAR);
440             String lastFilterType = "";
441             for (Map<String, ExpressionProperty> mapFilter : filterExprList) {
442                 String filterType = mapFilter.get(FILTER_TYPE_PROP).eval(document);
443                 String filterAttribute = mapFilter.get(ATTRIBUTE_PROP).eval(document);
444                 String filterOperator = mapFilter.get(FILTER_OPERATOR_PROP).eval(document);
445                 filterOperator = SearchComparisonOperator.valueOf(filterOperator)
446                         .getAbbreviation();
447                 Object filterValue = mapFilter.get(FILTER_VALUE_PROP).eval(document);
448                 if (StringUtils.isBlank(filterAttribute)) {
449                     throw new SnapDataException(INVALID_SEARCH_CONDITION)
450                             .withReason(INVALID_FILTER_ATTRIBUTE)
451                             .withResolution(PLEASE_CORRECT_AND_RETRY);
452                 }
453                 if (filterValue == null) {
454                     throw new SnapDataException(INVALID_SEARCH_CONDITION)
455                             .withReason(INVALID_FILTER_VALUE)
456                             .withResolution(PLEASE_CORRECT_AND_RETRY);
457                 }
458                 builder.add(filterAttribute).add(filterOperator)
459                         .add(formatFilterValue(filterAttribute, filterValue))
460                         .add(filterType);
461                 lastFilterType = filterType;
462             }
463             if (builder.length() > 0) {
464                 filter = oDataUtils.truncateLastFilterType(builder.toString(), lastFilterType);
465             }
466         }
