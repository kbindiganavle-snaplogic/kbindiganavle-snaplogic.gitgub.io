116     public void initODataClient(int timeout) {
117         if (client != null) {
118             return;
119         }
120         client = ODataClientFactory.getClient();
121         HttpClientFactory httpClientFactory = new CustomHttpClientFactory(timeout);
122         final ProxySettings proxySettings = RequestClientUtil.getProxySettings();
123         if (proxySettings.isProxied()) {
124             URI uri;
125             try {
126                 uri = new URI("http", null,
127                         proxySettings.getProxyHost(),
128                         proxySettings.getProxyPort(), null, null, null);
129             } catch (URISyntaxException ex) {
130                 throw new ExecutionException(ex, ERR_PROXY_SETUP)
131                         .withReason(ex.getMessage())
132                         .withResolution(ERR_PROXY_SETUP_RESOLUTION);
133             }
134             if (proxySettings.isProxyAuthenticated()) {
135                 httpClientFactory = new ProxyWrappingHttpClientFactory(uri,
136                         proxySettings.getProxyUsername(),
137                         proxySettings.getProxyPassword(),
138                         (CustomHttpClientFactory) httpClientFactory);
139             } else {
140                 httpClientFactory = new ProxyWrappingHttpClientFactory(uri,
141                         (CustomHttpClientFactory) httpClientFactory);
142             }
143         }
144         client.getConfiguration().setHttpClientFactory(httpClientFactory);
145         client.getConfiguration().setDefaultPubFormat(ContentType.APPLICATION_JSON);
146         if (oDataUtils == null) {
147             oDataUtils = new ODataUtils(account, client);
148         }
149     }

400     protected void evaluateKeys(Document document) {
401         if (MapUtils.isNotEmpty(keysExpr)) {
402             keys = new LinkedHashMap<>();
403             for (Map.Entry<ExpressionProperty, ExpressionProperty> entry :
404                     keysExpr.entrySet()) {
405                 String key = entry.getKey().eval(null);
406                 Object value = entry.getValue().eval(document);
407                 if (StringUtils.isBlank(key) && (value != null &&
408                         StringUtils.isNotBlank(value.toString()))) {
409                     throw new SnapDataException(ERR_PROPERTY_INVALID)
410                             .formatWith(PRIMARY_KEY_LABEL)
411                             .withReason(String.format(ERR_PROPERTY_EMPTY, PRIMARY_KEY_LABEL))
412                             .withResolution(String.format(ERR_PROPERTY_EMPTY_RESOLUTION,
413                                     PRIMARY_KEY_LABEL));
414                 } else if (StringUtils.isNotBlank(key) && (value == null
415                         || StringUtils.isBlank(value.toString()))) {
416                     String label = PRIMARY_KEY_LABEL + " " + VALUE_LABEL
417                             .toLowerCase();
418                     throw new SnapDataException(ERR_PROPERTY_INVALID).formatWith(label)
419                             .withReason(String.format(ERR_PROPERTY_EMPTY, label))
420                             .withResolution(String.format(ERR_PROPERTY_EMPTY_RESOLUTION,
421                                     label));
422                 } else if (StringUtils.isNotBlank(key) && value != null ) {
423                     keys.put(key, value);
424                 }
425             }
426         }
427 
428         if (MapUtils.isNotEmpty(keys)) {
429             if (MapUtils.isEmpty(properties)) {
430                 properties = getPropertiesWithType(entity);
431                 if (MapUtils.isEmpty(properties)) {
432                     handleInvalidEntity();
433                 }
434             }
435             for (Map.Entry<String, Object> key : keys.entrySet()) {
436                 if (key.getValue() instanceof String) {
437                     String dataType = properties.get(key.getKey());
438                     if (StringUtils.isEmpty(dataType)) {
439                         handleInvalidKey(key.getKey());
440                     }
441                     keys.put(key.getKey(), oDataUtils.convertValue(dataType,
442                             (String) key.getValue()));
443                 }
444             }
445         }
446     }
