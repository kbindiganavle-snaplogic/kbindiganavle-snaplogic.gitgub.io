442                 .appendValueElement()
443                 .appendCurrentElement(DELETE_FILES_PROP)
444                 .build();
445         deleteFiles = Boolean.TRUE.equals(propertyValues.get(deleteFilesPath));
446 
447         String fileListPath = new JsonPathBuilder(EXISTING_FILES_UPLOAD_PROPS)
448                 .appendValueElement()
449                 .appendCurrentElement(FILE_PATHS_PROP)
450                 .build();
451         List<Map<String, Map<String, Object>>> fileList = propertyValues.get(fileListPath);
452         if (fileList.isEmpty()) {
453             throw new ConfigurationException(ERR_INVALID_SNAP_CONFIGURATION)
454                     .withReason(ERR_NO_FILE_PROVIDED_REASON)
455                     .withResolution(ERR_NO_FILE_PROVIDED_RESOLUTION);
456         }
457         filePaths = new ArrayList<>();
458         for (Map<String, Map<String, Object>> file : fileList) {
459             String format = (String) file.get(FILE_FORMAT_PROP).get(VALUE);
460             boolean header = Boolean.TRUE.equals(file.get(CSV_FILE_HEADER_PROP).get(VALUE));
461             ExpressionProperty filePathExp = propertyValues.getExpressionPropertyFor(
462                     file, FILE_PATH_PROP);
463             Map<String, Object> delimiterMapObject = file.get(CSV_FILE_DELIMITER_PROP);
464             String csvDelimiter = null;
465             if (delimiterMapObject != null) {
466                 Object delimiter = delimiterMapObject.get(VALUE);
467                 if (delimiter != null) {
468                     csvDelimiter = (String) delimiter;
469                 }
470             }
471 
472             if (StringUtils.isBlank(csvDelimiter)) {
473                 filePaths.add(new FilePathObject(format, header, filePathExp));
474             } else {
475                 filePaths.add(new FilePathObject(format, header, filePathExp, csvDelimiter));
476             }
477         }
478     }

595     private void uploadExistingFiles(Table targetTable, Document document) {
596         List<String> avroFiles = new ArrayList<>();
597         List<String> jsonFiles = new ArrayList<>();
598         Map<String, List<String>> csvFilesWithHeader = new LinkedHashMap<>();
599         Map<String, List<String>> csvFilesWithoutHeader = new LinkedHashMap<>();
600         // The variable allCsvFiles is used to store all CSV files (irrespective of header and
601         //  delimiter) to delete them easily
602         List<String> allCsvFiles = new ArrayList<>();
603         String fileName;
604 
605         for (FilePathObject filePath : filePaths) {
606             fileName = filePath.filePathExp.eval(document);
607             if (StringUtils.isBlank(fileName)) {
608                 SnapDataException snapDataException =
609                         new SnapDataException(ERR_INVALID_SNAP_CONFIGURATION)
610                                 .withReason(ERR_FILE_PATH_NULL_REASON)
611                                 .withResolution(ERR_FILE_PATH_NULL_RESOLUTION);
612                 errorViews.write(snapDataException, document);
613                 return;
614             }
615             switch (filePath.fileFormat) {
616                 case FILE_FORMAT_CSV:
617                     allCsvFiles.add(fileName);
618                     if (filePath.containsHeader) {
619                         if (csvFilesWithHeader.get(filePath.csvDelimiter) == null) {
620                             csvFilesWithHeader.put(filePath.csvDelimiter, new ArrayList<>());
621                         }
622                         csvFilesWithHeader.get(filePath.csvDelimiter).add(fileName);
623                     } else {
624                         if (csvFilesWithoutHeader.get(filePath.csvDelimiter) == null) {
625                             csvFilesWithoutHeader.put(filePath.csvDelimiter, new ArrayList<>());
626                         }
627                         csvFilesWithoutHeader.get(filePath.csvDelimiter).add(fileName);
628                     }
629                     break;
630                 case FILE_FORMAT_JSON:
631                     jsonFiles.add(fileName);
632                     break;
633                 default:
634                     avroFiles.add(fileName);
635                     break;
636             }
637         }
638 
639         try {
640             // load CSV files (that contain header) grouped based on the delimiter
641             for (String csvDelimiter : csvFilesWithHeader.keySet()) {
642                 doBulkLoad(csvFilesWithHeader.get(csvDelimiter), targetTable, document,
643                         CsvOptions
644                                 .newBuilder()
645                                 .setSkipLeadingRows(1)
646                                 .setFieldDelimiter(csvDelimiter)
647                                 .setAllowQuotedNewLines(true)
648                                 .build(),
649                         String.format(FILES_TYPE_CSV_WITH_HEADER, csvDelimiter));
650             }
651             // load CSV files (with no header) grouped based on the delimiter
652             for (String csvDelimiter : csvFilesWithoutHeader.keySet()) {
653                 doBulkLoad(csvFilesWithoutHeader.get(csvDelimiter), targetTable, document,
654                         CsvOptions
655                                 .newBuilder()
656                                 .setFieldDelimiter(csvDelimiter)
657                                 .setAllowQuotedNewLines(true)
658                                 .build(),
659                         String.format(FILES_TYPE_CSV_WITHOUT_HEADER, csvDelimiter));
660             }
661             // load JSON files
662             doBulkLoad(jsonFiles, targetTable, document, FormatOptions.json(), FILES_TYPE_JSON);
663             // load AVRO files
664             doBulkLoad(avroFiles, targetTable, document, FormatOptions.avro(), FILES_TYPE_AVRO);
665         } finally {
666             if (deleteFiles) {
667                 // delete AVRO files
668                 deleteFiles(document, avroFiles);
669                 // delete JSON files
670                 deleteFiles(document, jsonFiles);
671                 // delete CSV files
672                 deleteFiles(document, allCsvFiles);
673             }
674         }
675     }

746     private void doBulkLoad(List<String> sourceURIs, com.google.cloud.bigquery.Table targetTable,
747             Document document, FormatOptions formatOptions, String filesType) {
748         if (sourceURIs.isEmpty()) {
749             return;
750         }
751         try {
752             JobInfo jobInfo = createJobInfo(targetTable, sourceURIs, formatOptions);
753             Job bulkLoadJob = bigQuery.create(jobInfo);
754             bulkLoadJob = bulkLoadJob.waitFor(
755                     RetryOption.initialRetryDelay(
756                             Duration.ofSeconds(BULK_LOAD_STATUS_CHECK_PERIOD_SECONDS)),
757                     RetryOption.totalTimeout(
758                             Duration.ofMinutes(BULK_LOAD_PROCESS_TIMEOUT_MINUTES)));
759             if (bulkLoadJob.getStatus().getError() == null) {
760                 Map<String, Object> result = new LinkedHashMap<>();
761                 if (StringUtils.isNotBlank(filesType)) {
762                     result.put(FILES_TYPE, filesType);
763                 }
764                 result.put(STATUS, COMPLETED);
765                 result.put(STATISTICS, createMapObjectForLoadStatistics(
766                         bulkLoadJob.getStatistics()));
767                 outputViews.write(documentUtility.newDocumentFor(document, result));
768             } else {
769                 String reasons = StringUtils.EMPTY;
770                 for (BigQueryError error : bulkLoadJob.getStatus().getExecutionErrors()) {
771                     reasons = reasons + error.getMessage() + SEMI_COLON;
772                 }
773                 reasons = StringUtils.removeEnd(reasons, SEMI_COLON);
774                 SnapDataException snapDataException =
775                         new SnapDataException(ERR_BULK_LOAD_GCS_MSG)
776                                 .formatWith(sourceURIs)
777                                 .withReason(reasons)
778                                 .withResolution(ERR_BULK_LOAD_RESOLUTION);
779                 errorViews.write(snapDataException, document);
780             }
781         } catch (SuggestViewAbortException e) {
782             throw e;
783         } catch (Exception e) {
784             SnapDataException snapDataException =
785                     new SnapDataException(e, ERR_BULK_LOAD_GCS_MSG)
786                             .formatWith(sourceURIs)
787                             .withResolution(RESOLUTION_ADDRESS_ISSUE);
788             errorViews.write(snapDataException, document);
789         }
790     }

828     protected void writeDocumentToTempFile(Document document) throws IOException {

836         switch (fileFormat) {

841                 csvLine = StringUtils.removeEnd(csvLine, COMMA_SEPARATOR);
842                 csvLine += NEW_LINE;

844                 break;

854     private void writeCsvRecord(StringBuffer csvStringBuffer, Map<String, Object> record) {
855         Collection<Object> values = record.values();
856         for (Object value : values) {
857             if (value instanceof Map) {
858                 writeCsvRecord(csvStringBuffer, (Map) value);
859             } else if (value instanceof List) {
860                 throw new ExecutionException(ERR_CSV_CANNOT_SUPPORT_LIST_MSG)
861                         .withReason(ERR_CSV_CANNOT_SUPPORT_LIST_REASON)
862                         .withResolution(ERR_CSV_CANNOT_SUPPORT_LIST_RESOLUTION);
863             } else {
864                 String valueAsString;
865                 if (value instanceof DateTime) {
866                     valueAsString = ((DateTime) value).toString(DATE_TIME_FORMATTER);
867                 } else if (value instanceof LocalDateTime) {
868                     valueAsString = ((LocalDateTime) value).toString(DATE_TIME_FORMATTER);
869                 } else if (value instanceof LocalDate) {
870                     valueAsString = ((LocalDate) value).toString(DATE_FORMATTER);
871                 } else if (value instanceof LocalTime) {
872                     valueAsString = ((LocalTime) value).toString(TIME_FORMATTER);
873                 } else if (value instanceof String) {
874                     // For CSV, its better to quote the string value to avoid errors if the value
875                     //  contains comma.
876                     // Further, if the value contains double quote as part of its data then we need
877                     //  to escape the double quote by putting an additional double quote
878                     //  before/after it.
879                     valueAsString = String.format(QUOTED_STRING,
880                             ((String) value).replaceAll(DOUBLE_QUOTE, DOUBLE_QUOTE_TWICE));
881                 } else {
882                     valueAsString = (value == null ? StringUtils.EMPTY : value.toString());
883                 }
884                 csvStringBuffer = csvStringBuffer.append(valueAsString).append(COMMA_SEPARATOR);
885             }
886         }
887     }

908     private void generateAvroSchema(Map<String, Object> record,
909             List<org.apache.avro.Schema.Field> fieldList) {
910         Set<Map.Entry<String, Object>> entries = record.entrySet();
911         for (Map.Entry<String, Object> entry : entries) {
912             String key = entry.getKey();
913             Object value = entry.getValue();
914             if (value instanceof Map) {
915                 List<org.apache.avro.Schema.Field> innerFields = new ArrayList<>();
916                 generateAvroSchema((Map<String, Object>) value, innerFields);
917                 fieldList.add(new org.apache.avro.Schema.Field(key,
918                         org.apache.avro.Schema.createRecord(
919                                 AVRO_SCHEMA_INNER_RECORD_NAME + avroInnerRecordIndex, null,
920                                 AVRO_SCHEMA_INNER_RECORD_NAMESPACE + avroInnerRecordIndex, false,
921                                 innerFields), null, (Object) null));
922                 avroInnerRecordIndex++;
923                 continue;
924             } else if (value instanceof List) {
925                 Object firstObject = ((List) value).get(0);
926                 if (firstObject instanceof Map) {
927                     List<org.apache.avro.Schema.Field> innerFields = new ArrayList<>();
928                     generateAvroSchema((Map<String, Object>) firstObject, innerFields);
929                     fieldList.add(new org.apache.avro.Schema.Field(key,
930                             org.apache.avro.Schema.createArray(
931                                     org.apache.avro.Schema.createRecord(
932                                             AVRO_SCHEMA_INNER_RECORD_NAME + avroInnerRecordIndex,
933                                             null,
934                                             AVRO_SCHEMA_INNER_RECORD_NAMESPACE +
935                                                     avroInnerRecordIndex,
936                                             false, innerFields)), null,
937                             (Object) null));
938                     avroInnerRecordIndex++;
939                 } else {
940                     fieldList.add(new org.apache.avro.Schema.Field(key,
941                             org.apache.avro.Schema.createArray(getAvroPrimitiveType(firstObject)),
942                             null, (Object) null));
943                 }
944                 continue;
945             } else {
946                 fieldList.add(new org.apache.avro.Schema.Field(key, getAvroPrimitiveType(value),
947                         null, (Object) null));
948             }
949         }
950     }

952     private org.apache.avro.Schema getAvroPrimitiveType(Object value) {
953         if (value == null || value instanceof String) {
954             return org.apache.avro.Schema.create(org.apache.avro.Schema.Type.STRING);
955         } else if (value instanceof BigInteger || value instanceof Integer ||
956                 value instanceof Long) {
957             return org.apache.avro.Schema.create(org.apache.avro.Schema.Type.LONG);
958         } else if (value instanceof BigDecimal || value instanceof Double ||
959                 value instanceof Float) {

961         } else if (value instanceof Boolean) {

963         } else if (value instanceof byte[]) {
964             return org.apache.avro.Schema.create(org.apache.avro.Schema.Type.BYTES);
965         } else {
966             throw new ExecutionException(ERR_UNSUPPORTED_DATA_TYPE_MSG)
967                     .withReason(String.format(ERR_UNSUPPORTED_DATA_TYPE_AVRO_REASON,
968                             value.getClass().getSimpleName()))

1041     @Override
1042     protected void flushBatch() {
1043         try {
1044             Failsafe
1045                     .with(retryPolicy)
1046                     .run(context -> {
1047                         if (context.getAttemptCount() > 0) {
1048                             // its  retry
1049                             // recreate the temp file
1050                             deleteTempFile();
1051                             initWriteChannel();
1052                             for (Document document : documents) {
1053                                 uploadDocument(document);
1054                             }
1055                         }
1056                         // upload the temp file
1057                         closeWriter();
1058                         performBulkLoad();
1059                     });
1060         } catch (Throwable cause) {
1061            SnapDataException exception = new SnapDataException(cause,
1062                     ERR_BATCH_PROCESS)
1063                     .withReason(cause.getMessage())
1064                     .withResolution(ERR_BATCH_RESOLUTION);
1065             for (Document document : documents) {
1066                 errorViews.write(exception, document);
1067             }
1068         } finally {
1069             documents.clear();
1070         }
1071     }

1073     @Override
1074     public void handle(LifecycleEvent lifecycleEvent) {
1075         switch (lifecycleEvent) {
1076             case SUCCESS:
1077                 // If user has unchecked "Execute during preview" checkbox then simply return
1078                 if (!executeDuringPreview) {
1079                     return;
1080                 }
1081                 if (uploadType.equals(UPLOAD_TYPE_INCOMING_DOCS)) {
1082                     if (batching) {
1083                         synchronized (documents) {
1084                             if (!documents.isEmpty()) {
1085                                 flushBatch();
1086                             }
1087                             flushBatchTask.setStopFlag();
1088                             flushBatchTimer.cancel();
1089                         }
1090                     } else {
1091                         if (writeChannel == null) {
1092                             // Do not perform any subsequent operation for no input documents
1093                             // The write channel will be null in that scenario
1094                             // Simply return and do nothing
1095                             return;
1096                         }
1097                         closeWriter();
1098                         performBulkLoad();
1099                         deleteTempFile();
1100                     }
1101                 }
1102                 break;
1103             case FAILURE:
1104             case CLOSE:
1105             case STOP:
