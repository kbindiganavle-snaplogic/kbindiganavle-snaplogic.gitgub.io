369     protected void processQueryResponse(GetQueryResults queryResults, boolean bulkRequest) {
370         String pageToken = null;
371         int errorCount = 0;
372         // This loop polls until results are present, then loops over result pages.
373         do {
374             GetQueryResultsResponse queryResultsResponse = extractResults(queryResults);
375             boolean jobComplete = false;
376 
377             // checking if query results are returned and job is completed
378             if (queryResultsResponse != null) {
379                 jobComplete = queryResultsResponse.getJobComplete();
380                 int retryCounter = 0;
381                 if (!jobComplete) {
382                     boolean apiError = false;
383                     // Iterating loop till we get job status as Done from GBQ API
384                     // and there was no api error
385                     while (!apiError){
386                         try {
387                             /*
388                              * Polling the GBQ job api till it returns job status as Done.
389                              */
390                             Get getJobRequest = bigquery.jobs().get(projectId, queryResults.getJobId());
391                             if (Objects.nonNull(location)) {
392                                 getJobRequest.setLocation(location);
393                             }
394                             pollJob(getJobRequest, POLLING_INTERVAL);
395                             // extracting new results again from the API.
396                             queryResultsResponse = extractResults(queryResults);
397                             jobComplete = queryResultsResponse != null && queryResultsResponse.getJobComplete();
398                             break;
399                         } catch (InterruptedException ex) {
400                             LOG.error(POLLING_JOB_FAILED);
401                             SnapDataException e = new SnapDataException(ex, POLLING_JOB_FAILED)
402                                 .withReason(POLLING_JOB_FAILED_REASON)
403                                 .withResolutionAsDefect();
404                             errorViews.write(e);
405                             apiError = true;
406                         } catch (IOException e) {
407                            apiError = reconnectIfUnauthorizedOtherwiseHandleError(e, retryCounter);
408                            retryCounter++;
409                         }
410                     }
411                 }
412             }
413             // processing only if job is complete
414             if (jobComplete) {
415                 TableSchema schemaObj = queryResultsResponse.getSchema();
416                 List<TableRow> rows = queryResultsResponse.getRows();
417                 List<TableFieldSchema> fields = schemaObj == null ? null : schemaObj.getFields();
418                 List<ErrorProto> errorList = queryResultsResponse.getErrors();
419                 if (errorList != null && !errorList.isEmpty()) {
420                     errorCount = errorList.size();
421                     writeToErrorView(errorList);
422                 }
423                 /** For select query, TableRow/TableFieldSchema will be returned other wise it will be null*/
424                 pageToken = queryResultsResponse.getPageToken();
425                 if (rows != null && fields != null) {
426                     writeToOutputView(rows, fields);
427                     queryResults.setPageToken(pageToken);
428                 } else {
429                     writeToOutputView(queryResultsResponse, errorCount);
430                 }
431             }
432         } while (!StringUtils.isEmpty(pageToken));
433     }

451     private void writeToOutputView(List<TableRow> tableRows, List<TableFieldSchema> fieldSchemas) {
452         Object obj = null;
453         Map<String, Object> docAsMap = null;
454         TableFieldSchema tableFieldSchema = null;
455         String fieldName = null, fieldType = null;
456         for (TableRow row : tableRows) {
457             List<TableCell> cell = row.getF();
458             docAsMap = new LinkedHashMap<>();
459             for (int i = 0; i < fieldSchemas.size(); i++) {
460                 tableFieldSchema = fieldSchemas.get(i);
461                 fieldName = tableFieldSchema.getName();
462                 fieldType = tableFieldSchema.getType();
463                 obj = cell.get(i).getV();
464                 try {
465                     BigQueryUtils.parseSchema(obj, tableFieldSchema, docAsMap);
466                 } catch (IllegalArgumentException | NullPointerException e) {
467                     /**
468                      * If any bad values found in resultset then we are writing error to errorview and
469                      * writing the input value to output to not to break functionality.
470                      * */
471                     SnapDataException snapDataException = new SnapDataException
472                             (String.format(ERR_DATA_CONVERTION, fieldName, fieldType, obj))
473                             .withReason(ERR_BAD_RECORD)
474                             .withResolution(ERR_DATA_CONVERTION_RES);
475                     errorViews.write(snapDataException);
476                 }
477             }
478             outputViews.write(documentUtility.newDocument(docAsMap));
479         }
480     }
