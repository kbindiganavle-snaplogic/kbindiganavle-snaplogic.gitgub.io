256             return;
257         }

260             // Processing the original input document to remove null entries if any
261             ((Map) documentData).values().removeIf(Objects::isNull);
262             // Due to platform enhancement that displays numbers as json objects, we need to

266         } catch (IOException e2) {

270             row = (Map) documentData;
271         }
272         // This snap inserts the batch of documents either when the batch reaches the full size or
273         // when the batch timeout occurs (whichever is earlier).
274         // The batch size condition is handled in this main thread.
275         // This task should not proceed if the child thread is already processing the batch.
276         // This is achieved by synchronizing the code block using the shared object "documents".
277         synchronized (documents) {

289                     documents.clear();

296     private void doStream() {
297         InsertAllResponse response = null;
298         RetryPolicy<Object> retryPolicy = new RetryPolicy<>()
299                 .handleIf(exception -> exception instanceof BigQueryException)
300                 .withDelay(Duration.ofMillis(retryDelay))
301                 .withMaxRetries(retryCount)
302                 .onFailedAttempt(e -> {
303                     Throwable throwable = e.getLastFailure();
304                     LOGGER.debug("BulkUpsertWithStreaming failed due to the error " +
305                             throwable.getMessage());
306                     if (throwable instanceof BigQueryException) {
307                         LOGGER.debug("Failure HTTP code: " +
308                                 ((BigQueryException) throwable).getCode());
309                     }
310                 })
311                 .onRetry(e -> {
312                     LOGGER.debug("BulkUpsertWithStreaming retry attempt " +
313                             e.getAttemptCount());
314                     reloadBigQueryIf401(e.getLastFailure());
315                 });
316         try {
317             response = Failsafe
318                     .with(retryPolicy)
319                     .get(() -> bigQuery.insertAll(insertAllRequestBuilder.build()));
320         } catch (BigQueryException e) {
321             LOGGER.error(e.getMessage(), e);
322             deleteTemp();
323             SnapDataException exception = new SnapDataException(e, ERR_BULK_LOAD_STREAM_MSG)
324                     .withReason(e.getMessage())
325                     .withResolution(ERR_BULK_LOAD_STREAMING_RESOLUTION);
326             writeToErrorView(exception, documents);
327             failedRecordsCount += documents.size();
328         }
329         if (response.hasErrors()) {
330             deleteTemp();
331             String reasons = StringUtils.EMPTY;
332             for (Map.Entry<Long, List<BigQueryError>> errorEntry : response.getInsertErrors()
333                     .entrySet()) {
334                 List<BigQueryError> errors = errorEntry.getValue();
335                 for (BigQueryError error : errors) {
336                     reasons = reasons + error + SEMI_COLON;
337                 }
338             }
339             reasons = StringUtils.removeEnd(reasons, SEMI_COLON);
340             SnapDataException exception = new SnapDataException(ERR_BULK_LOAD_STREAM_MSG)
341                     .withReason(reasons)
342                     .withResolution(ERR_BULK_LOAD_STREAMING_RESOLUTION);
343             writeToErrorView(exception, documents);
344             failedRecordsCount += documents.size();
345         } else {
346             successfulRecordsCount += documents.size();
347         }
348     }

437         String pageToken = null;
438         int errorCount = 0;
439         // This loop polls until results are present, then loops over result pages.
440         do {
441             GetQueryResultsResponse queryResultsResponse = extractResults(queryResults);
442             boolean jobComplete = false;
443 
444             // checking if query results are returned and job is completed
445             if (queryResultsResponse != null) {
446                 jobComplete = queryResultsResponse.getJobComplete();
447                 int retryCounter = 0;
448                 if (!jobComplete) {
449                     boolean apiError = false;
450                     // Iterating loop till we get job status as Done from GBQ API
451                     // and there was no api error
452                     while (!apiError){
453                         try {
454                             /*
455                              * Polling the GBQ job api till it returns job status as Done.
456                              */
457                             Bigquery.Jobs.Get getJobRequest = bigquery.jobs().get(projectId, queryResults.getJobId());
458                             pollJob(getJobRequest, POLLING_INTERVAL);
459                             // extracting new results again from the API.
460                             queryResultsResponse = extractResults(queryResults);
461                             jobComplete = queryResultsResponse != null && queryResultsResponse.getJobComplete();
462                             break;
463                         } catch (InterruptedException ex) {
464                             deleteTemp();
465                             LOGGER.error(POLLING_JOB_FAILED);
466                             SnapDataException e = new SnapDataException(ex, POLLING_JOB_FAILED)
467                                     .withReason(POLLING_JOB_FAILED_REASON)
468                                     .withResolutionAsDefect();
469                             errorViews.write(e);
470                             apiError = true;
471                         } catch (IOException e) {
472                             apiError = reconnectIfUnauthorizedOtherwiseHandleError(e, retryCounter);
473                             retryCounter++;
474                         }
475                     }
476                 }
477             }
478             // processing only if job is complete
479             if (jobComplete) {
480                 List<ErrorProto> errorList = queryResultsResponse.getErrors();
481                 if (errorList != null && !errorList.isEmpty()) {
482                     errorCount = errorList.size();
483                     writeToErrorView(errorList);
484                 }
485                 /** For select query, TableRow/TableFieldSchema will be returned otherwise it will be null*/
486                 pageToken = queryResultsResponse.getPageToken();
487                 writeToOutputView(queryResultsResponse, errorCount);
488             }
489         } while (!StringUtils.isEmpty(pageToken));
490     }

590     protected Set<String> getColumns(String projectId, String datasetName, String tableName) {
591         try {
592             // Initialize client that will be used to send requests. This client only needs to be created
593             // once, and can be reused for multiple requests.
594             TableId tableId = TableId.of(projectId, datasetName, tableName);
595             TableInfo table = bigQuery.getTable(tableId);
596             Set<String> columnNames = new HashSet<>();
597             // Retrieve the schema and iterate through the fields
598             Schema schema = table.getDefinition().getSchema();
599             for (Field field : schema.getFields()) {
600                columnNames.add(field.getName());
601             }
602             return columnNames;
603         } catch (BigQueryException e) {
604             LOGGER.error("Table not retrieved. \n" + e.toString());
605         } catch (NullPointerException e) {
606             LOGGER.error("Schema not retrieved. \n" + e.toString());
607         }
608         return new HashSet<>();
609     }

611     protected void suggestTableColumns(SuggestionBuilder suggestionBuilder,
612                                        PropertyValues propertyValues) {
613         //Configure table related properties to initialize table metadata
614         super.configure(propertyValues);
615         Set<String> columnsToSuggest = getColumns(projectId,datasetId,tableId);
616         if (columnsToSuggest.isEmpty()) {
617             throw new ExecutionException(NO_COLUMNS_AVAILABLE);
618         }
619 
620         //Filter columns that have already been selected
621         List<ExpressionProperty> rows = getTableColumns(propertyValues);
622         rows.forEach(row -> {
623             String inputColumn = row.eval(null);
624             if (inputColumn != null) {
625                 columnsToSuggest.remove(inputColumn);
626             }
627         });
628         if (columnsToSuggest.isEmpty()) {
629             throw new ExecutionException(ERR_ALL_COLUMNS_SELECTED)
630                     .withReason(ERR_ALL_COLUMNS_SELECTED_REASON)
631                     .withResolution(ERR_ALL_COLUMNS_SELECTED_RESOLUTION);
632         }
633 
634         suggestionBuilder.node(COLUMN_NAMES_PROP)
635                 .over(COLUMN_PROP)
636                 .suggestions(columnsToSuggest.toArray(new String[columnsToSuggest.size()]));
637     }

639     protected List<ExpressionProperty> getTableColumns(PropertyValues propertyValues) {
640         List<Map<String, Object>> table = propertyValues.getAsExpression(COLUMN_NAMES_PROP)
641                 .eval(null);
642         final List<ExpressionProperty> rows = new ArrayList<>();
643         if (table != null) {
644             table.forEach(tableRow -> {
645                 ExpressionProperty row = propertyValues.getExpressionPropertyFor(tableRow,
646                         COLUMN_PROP);
647                 rows.add(row);
648             });
649         }
650         return rows;
