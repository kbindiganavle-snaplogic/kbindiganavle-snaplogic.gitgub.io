223     /**
224      * Retrieves supported data types for a specified db.
225      *
226      * @param db the Database
227      * @return the string [ ]
228      */
229     public static String[] getSupportedDataTypes(String db) {
230         switch (db) {
231             case JdbcOperations.SNOWFLAKE:
232                 return SnowflakeDataTypes.getSupportedTypes();
233             case JdbcOperations.REDSHIFT:
234                 return RedshiftDataTypes.getSupportedTypes();
235             case JdbcOperations.AZURE_SYNAPSE:
236                 return AzureSynapseDataTypes.getSupportedTypes();
237             case JdbcOperations.DATABRICKS_LAKEHOUSE_PLATFORM:
238                 return DLPDataTypes.getSupportedTypes();
239             case JdbcOperations.BIGQUERY:
240                 return BigQueryDataTypes.getSupportedTypes();
241             default:
242                 throw new ConfigurationException(INVALID_ACCOUNT_CONFIGURATION)
243                         .withReason(INVALID_DATABASE_TYPE_SPECIFIED)
244                         .withResolution(SELECT_SUPPORTED_DATABASE_TYPE);
245         }
246     }
