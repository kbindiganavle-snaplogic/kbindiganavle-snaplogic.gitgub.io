371     private static String getWithClauseAzureSynapse(String hashColumn,
372                                                     List<String> tableOptionList) {
373         StringBuilder withClauseBuilder = new StringBuilder();
374         boolean first = true;
375         boolean distributionTableOptionExists = false;
376         for (String tableOption : tableOptionList) {
377             if (first) {
378                 first = false;
379             } else {
380                 withClauseBuilder.append(COMMA_SPACE);
381             }
382 
383             tableOption = tableOption.trim();
384             if (tableOption.startsWith(PREFIX_DISTRIBUTION)) {
385                 distributionTableOptionExists = true;
386             }
387             withClauseBuilder.append(tableOption);
388         }
389         // Append Hash distribution with the given hashColumn if
390         // no distribution table option is given.
391         if (!distributionTableOptionExists) {
392             if (!first) {
393                 withClauseBuilder.append(COMMA_SPACE);
394             }
395             withClauseBuilder.append(
396                     String.format(AZURE_SYNAPSE_HASH_DISTRIBUTION_CLAUSE, hashColumn));
397         }
398         return withClauseBuilder.toString();
399     }

538     public static String getAzureSynapseTableConstraints(Connection conn, String targetTablePath) {
539         try (PreparedStatement stmt =
540                      conn.prepareStatement(String.format(GET_TABLE_CONSTRAINTS, targetTablePath))) {
541             StringBuilder keyConstraints = new StringBuilder();
542             try (ResultSet rs = stmt.executeQuery()) {
543                 int prevId = 0;
544                 while (rs.next()) {
545                     int currentId = rs.getInt("object_id");
546                     String type = rs.getString("type");
547                     String column = rs.getString("column_name");
548                     if (currentId == prevId) {
549                         keyConstraints.append(", ").append(column);
550                     } else {
551                         if (prevId == 0) {
552                             keyConstraints.append(type).append("(").append(column);
553                         } else {
554                             keyConstraints.append(") ").append(AZURE_SYNAPSE_NOT_ENFORCED)
555                                     .append(", ").append(type).append(" (").append(column);
556                         }
557                         prevId = currentId;
558                     }
559 
560                 }
561             }
562             String keyConstraintsString = keyConstraints.toString();
563             keyConstraintsString =
564                     keyConstraintsString.replace("PK", AZURE_SYNAPSE_PK_CONSTRAINT);
565             keyConstraintsString =
566                     keyConstraintsString.replace("UQ", AZURE_SYNAPSE_UQ_CONSTRAINT);
567             return keyConstraintsString.isEmpty() ? "" :
568                     keyConstraintsString + " ) " + AZURE_SYNAPSE_NOT_ENFORCED;
569         } catch (SQLException e) {
570             throw new SnapDataException(e, ERR_BULK_LOAD)
571                     .withReason(ERROR_PERFORMING_RETRIEVAL_OF_TARGET_TABLE_DDL)
572                     .withResolution(RESOLUTION_ADDRESS_ISSUE);
573         }
