185     @Override
186     protected void process(Document document, String inputViewName) {
187         if (store == null) {
188             if (document != null) {
189                 document.acknowledge();
190             }
191             return;
192         }
193         mailFolder = mailFolderExpression.eval(document);
194         if (StringUtils.isBlank(mailFolder)) {
195             throw new SnapDataException(String.format(ERR_FOLDER_NAME, SOURCE))
196                     .withReason(REASON_FOLDER_NAME)
197                     .withResolution(String.format(RESOLUTION_FOLDER_NAME, SOURCE.toLowerCase()));
198         }
199         if (numberOfMessages == 0) {
200             return;
201         }
202         from = fromExpression.eval(document);
203         subject = subjectExpression.eval(document);
204         Message[] messages = getMessages(document);
205         if (messages == null || messages.length == 0) {
206             return;
207         }
208         List<Message> filteredMessages = new ArrayList<>();
209         CollectionUtils.addAll(filteredMessages, messages);
210         if (StringUtils.isNotBlank(from) &&
211                 !EmailUtils.filterListByFrom(filteredMessages, from, document, errorViews)) {
212             return;
213         }
214         if (!EmailUtils.filterListBySubject(filteredMessages, subject, document, errorViews)) {
215             return;
216         }
217         if (numberOfMessages > 0 && numberOfMessages < filteredMessages.size()) {
218             filteredMessages.subList(numberOfMessages, filteredMessages.size()).clear();
219         }
220         for (Message message : filteredMessages) {
221             deleteMessage(message, document);
222         }
223     }

225     @VisibleForTesting
226     protected Message[] getMessages(Document document) {
227         Message[] messages = null;
228         try {
229             folder = store.getFolder(mailFolder);
230             folder.open(Folder.READ_WRITE);
231             if (account.getClass() != Pop3Account.class) {
232                 switch (Enum.valueOf(MessageType.class, typeOfMessages)) {
233                     case All:
234                         messages = folder.getMessages();
235                         break;
236                     case READ:
237                         messages = folder.search(FLAG_TERM_SEEN);
238                         break;
239                     case UNREAD:
240                         messages = folder.search(FLAG_TERM_UNSEEN);
241                         break;
242                 }
243             } else {
244                 messages = folder.getMessages();
245             }
246             FetchProfile fetchProfile = new FetchProfile();
247             fetchProfile.add(FetchProfile.Item.ENVELOPE);
248             folder.fetch(messages, fetchProfile);
249             EmailUtils.sortMessages(folder, messages);
250         } catch (MessagingException e) {
251             SnapDataException ex = new SnapDataException(e,
252                     ERR_RETRIEVE_MESSAGE_LIST).formatWith(mailFolder)
253                     .withResolution(ERR_RETRIEVE_MESSAGE_LIST_RESOLUTION);
254             errorViews.write(ex, document);
255         }
256         return messages;
257     }

259     private void deleteMessage(Message message, Document originalDocument) {
260         Map<String, Object> messageMap = new LinkedHashMap<>();
261         try {
262             messageMap.put(EmailUtils.FROM_PROP,
263                     EmailUtils.convertEmailAddressesToString(message.getFrom()));
264             messageMap.put(EmailUtils.TO_PROP,
265                     EmailUtils.convertEmailAddressesToString(
266                             message.getRecipients(Message.RecipientType.TO)));
267             messageMap.put(EmailUtils.CC_PROP,
268                     EmailUtils.convertEmailAddressesToString(
269                             message.getRecipients(Message.RecipientType.CC)));
270             messageMap.put(EmailUtils.SUBJECT_PROP, message.getSubject());
271             Date sentDate = message.getSentDate();
272             messageMap.put(DATE_SENT, sentDate == null ? null : sentDate.toString());
273             message.setFlag(Flags.Flag.DELETED, true);
274             outputViews.write(documentUtility.newDocument(messageMap), originalDocument);
275         } catch (MessagingException e) {
276             SnapDataException ex = new SnapDataException(documentUtility.newDocument(messageMap),
277                     ERR_DELETE_MESSAGE)
278                     .withReason(e.getMessage())
279                     .withResolution(ERR_CUSTOMER_SUPPORT);
280             errorViews.write(ex);
281         }
282     }
