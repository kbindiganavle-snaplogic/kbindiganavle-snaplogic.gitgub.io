292     @Override
293     protected void process(final Document document, final String inputViewName) {
294         if (store == null) {
295             if (document != null) {
296                 document.acknowledge();
297             }
298             return;
299         }
300         mailFolder = mailFolderExpression.eval(document);
301         if (StringUtils.isBlank(mailFolder)) {
302             throw new SnapDataException(String.format(ERR_FOLDER_NAME, SOURCE))
303                     .withReason(REASON_FOLDER_NAME)
304                     .withResolution(String.format(RESOLUTION_FOLDER_NAME, SOURCE.toLowerCase()));
305         }
306         if (archive) {
307             archiveFolder = archiveFolderExpression.eval(document);
308             if (StringUtils.isBlank(archiveFolder)) {
309                 throw new SnapDataException(ERR_READER_CONFIG_MSG)
310                         .withReason(ERR_ARCHIVE_FOLDER_EMPTY_REASON)
311                         .withResolution(ERR_ARCHIVE_FOLDER_EMPTY_RESOLUTION);
312             }
313             if (StringUtils.equals(mailFolder, archiveFolder)) {
314                 throw new SnapDataException(ERR_READER_CONFIG_MSG)
315                         .withReason(ERR_SOURCE_AND_ARCHIVE_FOLDERS_SAME_REASON)
316                         .withResolution(ERR_SOURCE_AND_ARCHIVE_FOLDERS_SAME_RESOLUTION);
317             }
318             archiveList = new ArrayList<>();
319         }
320         initFolders();
321         if (attachmentFolderExpression != null) {
322             attachmentFolder = attachmentFolderExpression.eval(document);
323             ignoreAttachment = StringUtils.isBlank(attachmentFolder);
324             if (!ignoreAttachment) {
325                 if (!attachmentFolder.endsWith(URL_SEPARATOR)) {
326                     attachmentFolder = attachmentFolder.concat(URL_SEPARATOR);
327                 }
328             }
329         }
330         try {
331             // get message meta data list
332             Message[] messages = getMessages();
333 
334             // determine the number of messages to be retrieved
335             if (messages == null || messages.length == 0) {
336                 return;
337             }
338             int numMessages = messages.length;
339             int lastMessage;
340             if (olderMessageFirst) {
341                 lastMessage = numMessages - 1;
342                 if (numberOfMessages > 0 && numMessages > numberOfMessages) {
343                     lastMessage = numberOfMessages - 1;
344                 }
345                 for (int i = 0; i <= lastMessage; i++) {
346                     readMessage(messages[i], document, i);
347                 }
348             } else {
349                 lastMessage = 0;
350                 if (numberOfMessages > 0 && numMessages > numberOfMessages) {
351                     lastMessage = numMessages - numberOfMessages;
352                 }
353                 // retrieve messages, the most recent message first, the oldest message last
354                 for (int i = numMessages - 1; i >= lastMessage; i--) {
355                     readMessage(messages[i], document, i);
356                 }
357             }
358             if (archiveList != null && !archiveList.isEmpty()) {
359                 EmailArchive.archiveMessages(sourceFolder, targetFolder,
360                         archiveList.toArray(new Message[0]), outputViews, errorViews,
361                         documentUtility, document, false);
362             }
363         } catch (SnapDataException e) {
364             errorViews.write(e, document);
365         } catch (SuggestViewAbortException e) {
366             throw e;
367         } catch (Exception e) {
368             errorViews.write((SnapDataException) new SnapDataException(e, ERR_RETRIEVE_MESSAGES)
369                     .withReason(e.getMessage())
370                     .withResolution(RESOLUTION_ADDRESS_ISSUE), document);
371         }
372     }

429     private void readMessage(Message message, Map<String, Object> messageMap)
430             throws MessagingException, IOException {
431         // get 'from' address
432         Address[] addresses = message.getFrom();
433         messageMap.put(FROM_PROP, EmailUtils.convertEmailAddressesToString(addresses));
434         // get 'to' addresses
435         addresses = message.getRecipients(Message.RecipientType.TO);
436         messageMap.put(TO_PROP, EmailUtils.convertEmailAddressesToString(addresses));
437         // get 'cc' addresses
438         addresses = message.getRecipients(Message.RecipientType.CC);
439         messageMap.put(CC_PROP, EmailUtils.convertEmailAddressesToString(addresses));
440         // get subject
441         messageMap.put(SUBJECT_PROP, message.getSubject());
442         // get data sent
443         Date sentDate = message.getSentDate();
444         messageMap.put(DATE_SENT, sentDate != null ? sentDate.toString() : StringUtils.EMPTY);
445         // get content type
446         String contentType = null;
447         try {
448             contentType = message.getContentType();
449         } catch (MessagingException e1) {
450             LOG.error("Error fetching content-type for a message", e1);
451         }
452         messageMap.put(EMAIL_TYPE_PROP, contentType);
453         Object object = message.getContent();
454         if (object instanceof Multipart) {
455             Multipart multipart = (Multipart) object;
456             int count = multipart.getCount();
457             for (int i = 0; i < count; i++) {
458                 readPart(multipart.getBodyPart(i), messageMap);
459             }
460         } else {
461             readPart(message, messageMap);
462         }
463         if (!messageMap.containsKey(ATTACHMENTS)) {
464             messageMap.put(ATTACHMENTS, Collections.EMPTY_LIST);
465         }
466     }

468     @SuppressWarnings("unchecked")
469     private void readPart(Part part, Map<String, Object> messageMap)
470             throws MessagingException, IOException {
471         if (Part.ATTACHMENT.equalsIgnoreCase(part.getDisposition())) {
472             // email has attachment
473             addAttachment(part, messageMap);
474         } else if (Part.INLINE.equalsIgnoreCase(part.getDisposition())) {
475             // inline attachment
476             addAttachment(part, messageMap);
477         } else if (part.isMimeType(MediaType.PLAIN_TEXT_UTF_8.toString())) {
478             // plain text content
479             appendValue(messageMap, TEXT_BODY, part.getContent());
480         } else if (part.isMimeType(MediaType.HTML_UTF_8.toString())) {
481             // HTML content
482             appendValue(messageMap, HTML_BODY, part.getContent());
483         } else if (part.isMimeType(MULTIPART)) {
484             // multipart content
485             Multipart multipart = (Multipart) part.getContent();
486             int count = multipart.getCount();
487             for (int i = 0; i < count; i++) {
488                 readPart(multipart.getBodyPart(i), messageMap);
489             }
490         } else if (part.isMimeType(RFC822) || part instanceof Message) {
491             // nested message part
492             readPart((Part) part.getContent(), messageMap);
493         }
494     }

510     private Map<String, String> writeAttachment(Part part) throws MessagingException, IOException {
511         Map<String, String> attachmentMap = new HashMap<>(3);
512         String filename = getAttachmentFrom(part, attachmentMap);
513         if (ignoreAttachment) {
514             attachmentMap.put(STATUS, WriteResult.IGNORED.toString());
515             return attachmentMap;
516         }
517         // check if the file already exists
518         if (StringUtils.isBlank(filename)) {
519             filename = ATTACHMENT;
520             attachmentMap.put(FILENAME, filename);
521         }
522         String strPathNoPassword = attachmentFolder.concat(filename);
523         URI uri = getUriForSenderReader(strPathNoPassword, account.getAttachmentPassword());
524         FileSystem fileSystem = EmailUtils.getFileSystemFor(uri);
525         try {
526             Path path = EmailUtils.createPath(fileSystem, uri);
527             if (Files.exists(path, LinkOption.NOFOLLOW_LINKS)) {
528                 switch (writeOption) {
529                     case OVERWRITE:
530                         attachmentMap.put(STATUS, WriteResult.OVERWRITTEN.toString().toLowerCase());
531                         break;
532                     case RENAME:
533                         uri = rename(fileSystem, strPathNoPassword);
534                         strPathNoPassword = URIUtil.decode(uri.toString());
535                         attachmentMap.put(STATUS, WriteResult.RENAMED.toString().toLowerCase());
536                         break;
537                     case IGNORE:
538                         attachmentMap.put(STATUS, WriteResult.IGNORED.toString().toLowerCase());
539                         return attachmentMap;
540                     case ERROR:
541                         String error = String.format(ERR_FILE_EXISTS_REASON, strPathNoPassword);
542                         throw new SnapDataException(ERR_FILE_EXISTS)
543                                 .withReason(error)
544                                 .withResolution(ERR_DUPLICATE_RESOLUTION);
545                     default:
546                         error = String.format(ERR_UNSUPPORTED_WRITE_OPTION, writeOption.toString());
547                         throw new SnapDataException(error)
548                                 .withReason(error)
549                                 .withResolution(ERR_CUSTOMER_SUPPORT);
550                 }
551             } else {
552                 attachmentMap.put(STATUS, WriteResult.CREATED.toString());
553             }
554         } finally {
555             closeFileSystem(fileSystem, strPathNoPassword);
556         }
557         attachmentMap.put(FILENAME, getFullPath(strPathNoPassword));
558         InputStream inputStream = null;
559         OutputStream outputStream = null;
560         URLConnection urlConnection = null;
561         try {
562             // get input stream for attachment content
563             Object object = part.getContent();
564             if (object instanceof InputStream) {
565                 inputStream = (InputStream) object;
566             } else if (object instanceof String) {
567                 String string = (String) object;
568                 if (StringUtils.isBlank(string)) {
569                     attachmentMap.put(STATUS, WriteResult.EMPTY.toString().toLowerCase());
570                     return attachmentMap;
571                 } else {
572                     // SWAT-1237 - this properly handles UTF-8 attachments wrongly
573                     // labeled with Content-Type: text/xml; charset="US-ASCII". Using
574                     // IOUtils.toInputStream() results in corrupted data with extended chars.
575                     inputStream = new BOMInputStream(part.getInputStream());
576                 }
577             } else {
578                 throw new SnapDataException(ERR_UNKNOWN_CONTENT_TYPE)
579                         .formatWith(attachmentMap.get(CONTENT_TYPE))
580                         .withReason(ERR_UNKNOWN_CONTENT_TYPE_REASON)
581                         .withResolution(ERR_CUSTOMER_SUPPORT);
582             }
583             int bytesRead = inputStream.read(buffer);
584             if (bytesRead == -1) {
585                 attachmentMap.put(STATUS, WriteResult.EMPTY.toString().toLowerCase());
586                 IOUtils.closeQuietly(inputStream);
587                 return attachmentMap;
588             }
589             // get output stream
590             urlConnection = jfsUtils.openURLConnection(uri);
591             if (urlConnection != null) {
592                 try {
593                     urlConnection.connect();
594                     outputStream = urlConnection.getOutputStream();
595                 } catch (IOException e) {
596                     throwExceptionFailToWrite(filename, e);
597                 }
598             }
599             if (outputStream == null) {
600                 throwExceptionFailToWrite(filename);
601             }
602             // write to output stream
603             while (bytesRead != -1) {
604                 outputStream.write(buffer, 0, bytesRead);
605                 bytesRead = inputStream.read(buffer);
606             }
607         } finally {
608             IOUtils.closeQuietly(inputStream);
609             try {
610                 // if writing to sldb fails, it happens when closing output stream
611                 if (outputStream != null) {
612                     outputStream.close();
613                 }
614             } catch (IOException e) {
615                 throwExceptionFailToWrite(filename, e);
616             } finally {
617                 if (urlConnection instanceof Disconnectable) {
618                     ((Disconnectable) urlConnection).disconnect();;
619                 }
620             }
621         }
622         return attachmentMap;
623     }

649     private URI rename(FileSystem fileSystem, String existingFile) {
650         // find the index where the suffix should be inserted
651         int index = FilenameUtils.indexOfExtension(existingFile);
652         if (index == -1) {
653             index = existingFile.length();
654         }
655         // find a unique file name in the directory
656         URI uri;
657         boolean firstTime = true;
658         while (true) {
659             StringBuilder newBuilder = new StringBuilder(existingFile);
660             if (!firstTime) {
661                 newBuilder.insert(index, String.format(FORMAT_SUFFIX, System.currentTimeMillis()));
662             } else {
663                 firstTime = false;
664             }
665             uri = getUriForSenderReader(newBuilder.toString(), account.getAttachmentPassword());
666             Path path = EmailUtils.createPath(fileSystem, uri);
667             if (!Files.exists(path, LinkOption.NOFOLLOW_LINKS)) {
668                 break;
669             }
670         }
671         return uri;
672     }
