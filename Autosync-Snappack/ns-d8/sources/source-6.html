349         this.document = document;
350         Object object = bodyTemplate.eval(document);
351         if (object instanceof String) {
352             body = (String) object;

364             Object data = JsonPath.read(document.get(), tableJsonPath);
365             if (data instanceof Map) {
366                 tableDataList.add((Map<String, Object>) data);
367                 rows++;
368             } else if (data instanceof List) {
369                 for (Object value : (List) data) {
370                     if (value instanceof Map) {
371                         tableDataList.add((Map<String, Object>) value);
372                         rows++;
373                     } else {
374                         writeErrorView(document, value);
375                         return;
376                     }
377                 }
378             } else {
379                 writeErrorView(document, data);
380                 return;
381             }
382             if (rows < batchSize) {
383                 return;
384             }
385             prepareTable();
386             rows = 0;
387         }
388         Map<String, Object> resultMap = substituteEmailContents(document);
389         sendMail(document, resultMap);

421     private Map<String, Object> substituteEmailContents(Document document) {
422         Map<String, Object> resultMap = new LinkedHashMap<>(RESULT_MAP_SIZE);
423         resultMap.put(TO_PROP, to.eval(document));
424         resultMap.put(CC_PROP, cc.eval(document));
425         resultMap.put(BCC_PROP, bcc.eval(document));
426         resultMap.put(FROM_PROP, from.eval(document));
427         resultMap.put(SUBJECT_PROP, subject.eval(document));
428         resultMap.put(EMAIL_TYPE_PROP, emailType.toString());
429         resultMap.put(TEMPLATE_BODY_PROP, body);
430         if (CollectionUtils.isNotEmpty(attachments)) {
431             attachmentList = new ArrayList<>();
432             for (ExpressionProperty attachment : attachments) {
433                 Object filePath = attachment.eval(document);
434                 if (filePath instanceof List) {
435                     for (Object file : ((List) filePath)) {
436                         populateAttachment(file);
437                     }
438                 } else {
439                     populateAttachment(filePath);
440                 }
441             }
442             // remove username and password from attachment URL's if any in the output document
443             List<Map<String, String>> attachmentNoAuthList = new ArrayList<>(attachmentList.size());
444             if (CollectionUtils.isNotEmpty(attachmentList)) {
445                 for (Map<String, String> attachmentMap : attachmentList) {
446                     Map<String, String> attachmentNoAuthMap = new LinkedHashMap<>(2);
447                     attachmentNoAuthMap.put(FILENAME, attachmentMap.get(FILENAME));
448                     // encode '+' chars to '%2B', otherwise URIUtils.decde() will change it to
449                     // space char
450                     String attachmentUrl = StringUtils.replace(attachmentMap.get(URL_PROP), PLUS,
451                             PLUS_CODE);
452                     try {
453                         attachmentUrl = URIUtil.decode(attachmentUrl);
454                     } catch (URIException e1) {
455                         LOG.error("Exception while decoding the attachment url {}",
456                                 URLUtils.getFileNoAuth(attachmentMap.get(URL_PROP)));
457                     }
458                     attachmentNoAuthMap.put(URL_PROP, URLUtils.getFileNoAuth(attachmentUrl));
459                     attachmentNoAuthList.add(attachmentNoAuthMap);
460                 }
461             }
462             resultMap.put(ATTACHMENTS_PROP, attachmentNoAuthList);
463         }
464         return resultMap;
465     }

483     private void sendMail(Document document, Map<String, Object> resultMap) {
484         Document documentOut = documentUtility.newDocumentFor(document, resultMap);
485         try {
486             // initialize email message with From, CC, To and Subject
487             MimeMessage message = new MimeMessage(session);
488             String value = (String) resultMap.get(FROM_PROP);
489             if (StringUtils.isBlank(value)) {
490                 message.setFrom();
491             } else {
492                 message.setFrom(new InternetAddress(value));
493             }
494             value = (String) resultMap.get(TO_PROP);
495             message.addRecipients(RecipientType.TO, value);
496             value = (String) resultMap.get(CC_PROP);
497             if (StringUtils.isNotBlank(value)) {
498                 message.addRecipients(RecipientType.CC, value);
499             }
500             value = (String) resultMap.get(BCC_PROP);
501             if (StringUtils.isNotBlank(value)) {
502                 message.addRecipients(RecipientType.BCC, value);
503             }
504             message.setSubject((String) resultMap.get(SUBJECT_PROP), StandardCharsets.UTF_8.name());
505             // prepare the email body part
506             BodyPart messageBodyPart = new MimeBodyPart();
507             value = (String) resultMap.get(TEMPLATE_BODY_PROP);
508             messageBodyPart.setContent(value, String.format(FORMAT_EMAIL_TYPE, emailType));
509             Multipart multipart = new MimeMultipart();
510             multipart.addBodyPart(messageBodyPart);
511             // attach files
512             if (attachmentList != null && !attachmentList.isEmpty()) {
513                 for (Map<String, String> attachmentMap : attachmentList) {
514                     messageBodyPart = new MimeBodyPart();
515                     // Previously the attachment filename had been encoded as following:
516                     //     messageBodyPart.setFileName(
517                     //             MimeUtility.encodeText(attachmentMap.get(FILENAME),
518                     //             StandardCharsets.UTF_8.name(), null));
519                     // It was found that MimeUtility.encodeText() method failed to encode the
520                     // filename with European UTF-8 characters like äâêîôûñç¡¿ÉÙË. It caused
521                     // Email Sender Snap to fail silently by attaching a file with a wrong name
522                     // which contains '?' char.
523                     // '?' char is an illegal char in SMB protocol and this attached file
524                     // caused Email Reader Snap to fail to write the file to SMB server.
525                     // It was fixed by removing the encoder and setting the filename as is.
526                     // The unit test "testSpecialCharsInAttachmentFilename()" covers this case.
527                     messageBodyPart.setFileName(attachmentMap.get(FILENAME));
528                     URI uri = new URI(attachmentMap.get(URL_PROP));
529                     DataHandler handler = handleAttachment(uri);
530                     messageBodyPart.setDataHandler(handler);
531                     multipart.addBodyPart(messageBodyPart);
532                 }
533             }
534             // Set the complete message parts
535             message.setContent(multipart);
536             // Send message
537             sendMessage(message, retries);
538             // write result to output view
539             if (passThrough) {
540                 outputViews.write(documentOut, document);
541             } else {
542                 outputViews.write(documentOut);
543             }
544         } catch (SuggestViewAbortException e) {
545             throw e;
546         } catch (Throwable e) {
547             LOG.error(ERR_MESSAGE_SEND, e);
548             writeErrorView(documentOut, e);
549         }

552     /**
553      * The Attachments property in Email Sender Snap supports SFTP & SMB. These
554      * attachment files should be downloaded to local temp files and URL's of these temp
555      * files should be passed to javax.mail library as URLDataSource. Otherwise, each
556      * email send will cause one connection leak per each attachment and per each
557      * recipient because javax.mail library can close input streams but not SFTP & SMB
558      * connection resources after the email is sent. For file:/// and sldb:///, there is
559      * no connection resources to be closed other than input streams.
560      *
561      * @param uri - URI object of attachment file
562      *
563      * @return data hanlder
564      * @throws IOException if it fails to generate data handler from the attachment URI
565      */
566     private DataHandler handleAttachment(URI uri) throws IOException {
567         URL url;
568         switch(uri.getScheme()) {
569             case PROTOCOL_FILE:
570             case PROTOCOL_SLDB:
571                 url = uri.toURL();
572                 break;
573             default:
574                 File tmpFile = File.createTempFile(this.getClass().getSimpleName(), null, tmpDir);
575                 copyAttachmentToTempFile(uri, tmpFile);
576                 tmpFiles.add(tmpFile);
577                 url = tmpFile.toURI().toURL();
578         }
579         URLDataSource dataSource = new URLDataSource(url);
580         return new DataHandler(dataSource);
581     }

583     @VisibleForTesting
584     protected void copyAttachmentToTempFile(URI uri, File tmpFile) {
585         try {
586             Failsafe.with(getRetryPolicy(retries, 1))
587                     .onFailedAttempt(e -> LOG.warn("{} - Failed to copy attachment file: {}, {}",
588                             RETRY_SIGNATURE, uri.getPath(), e.getMessage()))
589                     .run(context -> {
590                         logRetryAttempt(context, LOG, RETRY_SIGNATURE);
591                         URLConnection urlConnection = null;
592                         try {
593                             urlConnection = jfsUtils.openURLConnection(uri);
594                             urlConnection.connect();
595                             try (InputStream inputStream = urlConnection.getInputStream()) {
596                                 FileUtils.copyInputStreamToFile(inputStream, tmpFile);
597                             }
598                         } finally {
599                             if (urlConnection instanceof Disconnectable) {
600                                 ((Disconnectable) urlConnection).disconnect();;
601                             }
602                         }
603                     });
604         } catch (Exception e) {
605             Throwable ex = e instanceof FailsafeException ? e.getCause() : e;
606             if (ex instanceof SuggestViewAbortException) {
607                 throw (SuggestViewAbortException) ex;
608             } else {
609                 errorViews.write((SnapDataException)
610                         new SnapDataException(ex, ERR_COPY_ATTACHMENT)
611                                 .formatWith(uri.getPath())
612                                 .withResolution(RESOLUTION_ADDRESS_ISSUE), document);
613             }
614         }
615     }

617     /**
618      * Sends email message.
619      *
620      * @param message - email message
621      * @param retries - max number of email send attempts
622      * @throws Exception if fails in all attempts
623      */
624     private void sendMessage(MimeMessage message, int retries) throws Exception {
625         try {
626             transportSend(message);
627         } catch (MessagingException e) {
628             LOG.info("Email Sender: Error occurred while sending mail " + e.getMessage());
629             int delay = INITIAL_RETRY_DELAY_IN_MS;
630             boolean success = false;
631             // recursive method call is not used here since stack may overflow if retries is large
632             while (!success && retries-- > 0) {
633                 TimeUnit.MILLISECONDS.sleep(delay);
634                 try {
635                     transportSend(message);
636                     success = true;
637                 } catch (MessagingException ex) {
638                     LOG.info("Email Sender: Error occurred while sending mail in a retry attempt" +
639                             e.getMessage());
640                     TimeUnit.MILLISECONDS.sleep(delay);
641                     if ((delay = delay * 2) > MAX_RETRY_DELAY_IN_MS) {
642                         delay = MAX_RETRY_DELAY_IN_MS;
643                     }
644                 }
645             }
646             if (!success) {
647                 throw e;
648             }
649         }
650     }

728     private void writeErrorView(Document document, Throwable e) {
729         // check if any of the attachment files does not exist
730         List<String> notExistingFiles = null;
731         if (attachmentList != null && !attachmentList.isEmpty()) {
732             notExistingFiles = new ArrayList<>(attachmentList.size());
733             for (Map<String, String> attachmentMap : attachmentList) {
734                 // encode '+' chars to '%2B', otherwise URIUtils.decde() will change it to
735                 // space char
736                 String filename = StringUtils.replace(attachmentMap.get(URL_PROP), PLUS, PLUS_CODE);
737                 try {
738                     filename = URIUtil.decode(filename);
739                 } catch (URIException e1) {
740                     String uriNoAuth = URLUtils.getFileNoAuth(attachmentMap.get(URL_PROP));
741                     LOG.error("Error while decoding the URI: {}", uriNoAuth, e1);
742                     filename = attachmentMap.get(FILENAME);
743                     LOG.info("Using just filename instead {}", filename);
744                 }
745                 if (!fileExists(filename)) {
746                     notExistingFiles.add(filename);
747                 }
748             }
749         }
750         // prepare error, reason and resolution messages
751         String error, reason, resolution;
752         if (notExistingFiles != null && !notExistingFiles.isEmpty()) {
753             // Snap failed due to not-existing attachment file(s)
754             error = ERR_FILE_NOT_FOUND;
755             String files = Joiner.on(COMMA_SPACE).join(notExistingFiles);
756             reason = String.format(REASON_ATTACHMENT_NOT_FOUND, URLUtils.getFileNoAuth(files));
757             resolution = ERR_NOT_FOUND_RESOLUTION;
758         } else {
759             // Snap failed due to another reason;
760             error = ERR_MESSAGE_SEND;
761             reason = String.format(ERR_MESSAGE_SEND_REASON, e.getMessage());
762             resolution = ERR_MESSAGE_SEND_RESOLUTION;
763         }
764         // write to error view
765         SnapDataException snapDataException = new SnapDataException(e, error)
766                 .withReason(reason)
767                 .withResolution(resolution);
768         errorViews.write(snapDataException, document);
769     }
