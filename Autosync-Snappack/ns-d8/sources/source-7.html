133     static void populate(String url, List<Map<String, String>> selectedFiles, String password) {
134         url = StringUtils.trim(url);
135         String filename = getFilename(url);
136         if (!hasWildcard(filename)) {
137             selectedFiles.add(newAttachmentMap(filename,
138                     getUriForSenderReader(url, password).toString()));
139             return;
140         }
141         String folder = url.substring(0, url.lastIndexOf(filename));
142         // get URI for file path and check its syntax
143         URI pathUri = getUriForSenderReader(folder, password);
144         // get file system object
145         FileSystem fileSystem = getFileSystemFor(pathUri);
146         if (fileSystem == null) {
147             // file path has wildcard, but has no FileSystemProvider
148             throw new ExecutionException(ERR_SUPPORT_WILDCARD).formatWith(url)
149                     .withReason(ERR_SUPPORT_WILDCARD_REASON)
150                     .withResolution(ERR_SUPPORT_WILDCARD_RESOLUTION);
151         }
152         try {
153             // Get a Pattern object from wildcard/regex
154             Pattern pattern = getRegexPattern(filename);
155             // get Path object
156             Path path = createPath(fileSystem, pathUri);
157             findFiles(path, pattern, selectedFiles);

229      * Converts a wildcard pattern to a regex pattern.
230      *
231      * @param wildcard - wildcard pattern
232      *
233      * @return - regex pattern that is equivalent to the wildcard pattern
234      */
235     private static String wildcardToRegex(String wildcard) {
236         if (wildcard == null) {
237             return null;
238         }
239         StringBuilder buffer = new StringBuilder(STRING_CARET);
240         char[] chars = wildcard.toCharArray();
241         for (char character : chars) {
242             if (character == CHAR_WILDCARD) {
243                 buffer.append(DOT_STAR);
244             } else if (character == CHAR_QUESTION_MARK) {
245                 buffer.append('.');
246             } else if (REGEX_PATTERN_WILDCARD.indexOf(character) != -1) {
247                 // prefix all meta-characters with backslash
248                 buffer.append(CHAR_BACKSLASH).append(character);
249             } else {
250                 buffer.append(character);
251             }
252         }
253         buffer.append(CHAR_DOLLAR);
254         return buffer.toString().toLowerCase();
255     }

332         try {
333             // SWAT-834: ensure we have sldb:/// if not otherwise specified.
334             Pattern pattern = Pattern.compile(SCHEME_PATTERN);
335             if (!pattern.matcher(file).matches()) {
336                 if (file.startsWith(URL_SEPARATOR)) {
337                     //Remove the first / if the file path is an absolute one
338                     file = file.substring(1);
339                 }
340                 file = URI_SCHEME_SLDB + file;
341             }
342 
343             // Check if the file path needs to be encoded or not
344             // For this, we will decode the given file path
345             // If exception occurs, we will encode it

347             URI uri;
348             boolean encode = false;
349             try {

351                 if (!StringUtils.equals(decodedPath, file)) {
352                     encode = true;

354             } catch (IOException e) {
355                 LOG.error("Exception while decoding the given file path {}", file, e);
356                 encode = true;
357             }

360                 LOG.info("The given file path {} will be encoded", file);
361                 file = URIUtil.encodePath(file);
362                 LOG.info("Encoded path is {}", file);
363             }

366             String username = uri.getUserInfo();
367             if (StringUtils.isNotBlank(password) && StringUtils.isNotBlank(username) &&

389         try {
390             file = getFullPath(file);
391             URI uri = new URI(URIUtil.encodePath(file, StandardCharsets.UTF_8.name()));
392             String username = uri.getUserInfo();
393             if (StringUtils.isNotBlank(password) && StringUtils.isNotBlank(username) &&

395                 String userInfo = String.join(COLON, username, password);
396                 // If folder or filename contains special chars like äâêîôûñç¡¿ÉÙËÇ¨°ðø©¢, the
397                 // following way to encode path to URI does not encode these special chars and
