263     protected AmazonSQS getSqsClient() {
264         if (sqsClient == null) {
265             ClientConfiguration configuration = new ClientConfiguration();
266             if (maxErrorRetry != null) {
267                 configuration.setMaxErrorRetry(maxErrorRetry.intValue());
268             }
269             String prefix = SQSUtils.getApnUserAgentHeader();
270             configuration.setUserAgentPrefix(prefix);
271             account.setRegion(region);
272             AWSCredentialsProvider credentialsProvider = account.connect();
273             AmazonSQSAsyncClientBuilder builder = AmazonSQSAsyncClientBuilder.standard()
274                     .withClientConfiguration(configuration)
275                     .withCredentials(credentialsProvider);
276             if (StringUtils.isNotBlank(region)) {
277                 builder.withRegion(region);
278             } else {
279                 builder.withRegion(Regions.DEFAULT_REGION);
280             }
281             if (clientSideBuffering) {
282                 QueueBufferConfig config = getQueueBufferConfig();
283                 sqsClient = new AmazonSQSBufferedAsyncClient(builder.build(), config);
284             } else {
285                 sqsClient = builder.build();
286             }
287             if(account.largeMessageSupportEnabled()) {
288                 // Extend the current sqs client with the large message support
289                 sqsClient = new AmazonSQSExtendedClient(sqsClient,
290                         account.getExtendedClientConfiguration(credentialsProvider));
291             }
292         }
293 
294         return sqsClient;
295     }

325      * Decide and return the corresponding queue url. If the queueUrl is given from the property,
326      * then use it. Otherwise, get the url from the given queue name.
327      */
328     protected String updateQueueUrl(String region, String queueName, String queueUrl) {
329         if (StringUtils.isNotBlank(queueUrl)) {
330             return queueUrl;
331         } else if (StringUtils.isBlank(queueName)) {
332             throw new SnapDataException(ERROR_MUST_PROVIDE_QUEUE_URL_OR_NAME)
333                     .withReason(ERROR_MUST_PROVIDE_QUEUE_URL_OR_NAME_REASON)
334                     .withResolution(RESOLUTION_ADDRESS_ISSUE);
335         } else {
336             String key = String.format(QUEUE_URL_KEY_FMT, region, queueName);
337             if (!queueUrls.containsKey(key)) {
338                 LOG.debug("Getting new queueUrl for queueName: {}", queueName);
339                 queueUrls.put(key, getSqsClient().getQueueUrl(queueName).getQueueUrl());
340             }
341             return queueUrls.get(key);
342         }
343     }

346      * Define properties for Region, Queue name and Queue URL, and specify whether the properties
347      * can be get from upstream document input or not
348      *
349      * @param propertyBuilder
350      */
351     protected void defineQueueDestinationProps(final PropertyBuilder propertyBuilder) {
352         // Region
353         PropertyBuilder regionPropBuilder = propertyBuilder.describe(REGION_PROP, REGION_LABEL,
354                 REGION_DESC)
355                 .type(SnapType.STRING);
356 
357         regionPropBuilder.expression(SnapProperty.DecoratorType.ACCEPTS_SCHEMA);
358 
359         regionPropBuilder.withSuggestions((suggestionBuilder, propertyValues) -> {
360             Regions[] regions = Regions.values();
361             List<String> regionStrs = Arrays.stream(regions)
362                     .map(Regions::getName)
363                     .collect(Collectors.toList());
364             suggestionBuilder
365                     .node(REGION_PROP)
366                     .suggestions(regionStrs.toArray(new String[regionStrs.size()]));
367         });
368         regionPropBuilder.add();
369 
370         // Queue Name
371         PropertyBuilder queueNamePropBuilder = propertyBuilder.describe(QUEUE_NAME_PROP,
372                 QUEUE_NAME_LABEL, QUEUE_NAME_DESC)
373                 .type(SnapType.STRING);
374 
375         queueNamePropBuilder.expression(SnapProperty.DecoratorType.ACCEPTS_SCHEMA);
376 
377         queueNamePropBuilder.withSuggestions(((suggestionBuilder, propertyValues) -> {
378             account.setRegion(region);
379             AWSCredentialsProvider credentialsProvider = account.connect();
380             AmazonSQS sqs = null;
381             try {
382                 AmazonSQSAsyncClientBuilder builder = AmazonSQSAsyncClientBuilder.standard()
383                         .withCredentials(credentialsProvider);
384                 String region = propertyValues.getAsExpression(REGION_PROP).eval(null);
385                 if (StringUtils.isNotBlank(region)) {
386                     builder.withRegion(region);
387                 }
388                 sqs = builder.build();
389                 List<String> queueList = sqs.listQueues().getQueueUrls();
390                 List<String> nameList = Lists.newArrayList();
391                 queueList.forEach(url -> {
392                     String[] urlParts = url.split(SLASH);
393                     String name = url;
394                     if (urlParts.length > 0) {
395                         name = urlParts[urlParts.length - 1];
396                     }
397                     nameList.add(name);
398                 });
399                 suggestionBuilder.node(QUEUE_NAME_PROP)
400                         .suggestions(nameList.toArray(new String[nameList.size()]));
401 
402             } finally {
403                 if (sqs != null) {
404                     sqs.shutdown();
405                 }
406             }
407         }));
408         queueNamePropBuilder.add();
409 
410         // Queue Url
411         PropertyBuilder queueUrlPropBuilder = propertyBuilder.describe(QUEUE_URL_PROP,
412                 QUEUE_URL_LABEL, QUEUE_URL_DESC)
413                 .type(SnapType.STRING);
414 
415 
416         queueUrlPropBuilder.expression(SnapProperty.DecoratorType.ACCEPTS_SCHEMA);
417 
418         queueUrlPropBuilder.add();
419     }
