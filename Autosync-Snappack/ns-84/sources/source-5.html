220     @Override
221     protected void process(final Document document, final String inputViewName) {
222         try {
223             region = regionExpr.eval(document);
224             queueName = queueNameExpr.eval(document);
225             queueUrl = updateQueueUrl(region, queueName, queueUrlExpr.eval(document));
226             isFIFO = isFifo(queueUrl);
227             AmazonSQS client = getSqsClient();
228             if (isValidation) {
229                     List<Message> messages = consumeOnceSync(client, MAX_MESSAGES_PER_REQUEST);
230                         for (Message message : messages) {
231                             writeMessageToOutputView(message, queueUrl);
232                             if (autoAcknowledge) {
233                                 acknowledge(client, message.getReceiptHandle(), queueUrl);
234                             }
235                         }
236             } else {
237                 if (messageCount < 0) {
238                     // Infinite polling
239                     poll(client, false);
240                 } else if (messageCount > 0) {
241                     // Consume specific number of messages
242                     consumeMessages(messageCount, client);
243                 } else {
244                     // Poll for all available messages
245                     poll(client, true);
246                 }
247             }
248         } catch (QueueDoesNotExistException exception) {
249             SnapDataException dataException = new SnapDataException(exception,
250                     ERR_FAIL_TO_PROCESS_MESSAGE)
251                     .withReason(exception.getMessage())
252                     .withResolution(RESOLUTION_ADDRESS_ISSUE);
253             errorViews.write(dataException);
254         } catch (InterruptedException | java.util.concurrent.ExecutionException e) {
255             SnapDataException snapDataException = new SnapDataException
256                     (e, ERROR_FAIL_TO_FETCH_MESSAGE)
257                     .withReason(e.getMessage())
258                     .withResolution(RESOLUTION_ADDRESS_ISSUE);
259             synchronized (errorViews) {
260                 errorViews.write(snapDataException, document);
261             }
262         }
263     }

296      * Consume messages based on the message count
297      */
298     private void consumeMessages(int messageCount, AmazonSQS client)
299             throws java.util.concurrent.ExecutionException, InterruptedException {
300         int messagesRemaining = messageCount;
301         while (messagesRemaining > 0) {
302             // Each response can return 10 messages at maximum
303             int maxNumMsg = messagesRemaining > MAX_MESSAGES_PER_REQUEST ?
304                     MAX_MESSAGES_PER_REQUEST : messagesRemaining;
305             List<Message> messages = consumeOnceSync(client, maxNumMsg);
306             //Message processing and commit
307             for (final Message message : messages) {
308                 writeMessageToOutputView(message, queueUrl);
309                 --messagesRemaining;
310                 if (autoAcknowledge) {
311                     acknowledge(client, message.getReceiptHandle(), queueUrl);
312                 }
313             }
314         }
315     }

325     private ReceiveMessageRequest generateFetchRequest(int maxNumberOfMessages) {
326         ReceiveMessageRequest request = new ReceiveMessageRequest()
327                 .withQueueUrl(queueUrl)
328                 .withMaxNumberOfMessages(maxNumberOfMessages)
329                 .withMessageAttributeNames(ALL_MESSAGE_ATTRIBUTES);
330         if (waitTimeoutSec != null && waitTimeoutSec > 0) {
331             request.setWaitTimeSeconds(waitTimeoutSec);
332         }
333         // For client-side buffering, we have to keep the visibility timeout to null in the
334         // request, this is defined in the QueueBuffer. The visibilityTimeout has already set in
335         // the buffer configuration, though.
336         // refer:
337         // https://github.com/aws/aws-sdk-java/blob/1.11.1010/aws-java-sdk-sqs/src/main/java/
338         // com/amazonaws/services/sqs/buffered/QueueBuffer.java#L282-L295
339         if (visibilityTimeout != null && visibilityTimeout > 0 && !clientSideBuffering) {
340             request.setVisibilityTimeout(visibilityTimeout);
341         }
342         return request;
343     }

363      * Keep polling messages from the SQS. This should be called when message count is set to -1
364      *
365      * @param client
366      */
367     private void poll(AmazonSQS client, final boolean exitWhenNoneAvailable) throws InterruptedException, java.util.concurrent.ExecutionException {
368         final AtomicReference<AmazonSQS> clientReference = new AtomicReference<>(client);
369         isPolling.set(true);
370 
371         while (isPolling.get()) {
372             String requestID = UUID.randomUUID().toString();
373             ReceiveMessageRequest request = generateFetchRequest(MAX_MESSAGES_PER_REQUEST);
374             pendingFetchRequests.add(requestID);
375 
376             if(clientReference.get() instanceof AmazonSQSAsync) {
377                 final AmazonSQSAsync sqsAsyncClient = (AmazonSQSAsync) clientReference.get();
378                 try {
379                     // Use AsyncHandler here since the client-side Buffering has some optimization
380                     // on the approach
381                     Future<ReceiveMessageResult> result =
382                         sqsAsyncClient.receiveMessageAsync(request,
383                                 new AsyncHandler<ReceiveMessageRequest, ReceiveMessageResult>() {
384                             @Override
385                             public void onError(final Exception exception) {
386                                 try {
387                                     if (!isTokenExpiredException(exception)) {
388                                         SnapDataException snapDataException = new SnapDataException
389                                                 (exception, ERROR_FAIL_TO_FETCH_MESSAGE)
390                                                 .withReason(exception.getMessage())
391                                                 .withResolution(RESOLUTION_ADDRESS_ISSUE);
392                                         synchronized (errorViews) {
393                                             errorViews.write(snapDataException);
394                                         }
395                                     }
396                                 } finally {
397                                     pendingFetchRequests.remove(requestID);
398                                 }
399                             }
400 
401                             @Override
402                             public void onSuccess(final ReceiveMessageRequest request,
403                                     final ReceiveMessageResult receiveMessageResult) {
404                                 try {
405                                     handleMessages(receiveMessageResult,
406                                         sqsAsyncClient, exitWhenNoneAvailable, request);
407                                 } catch (InterruptedException e) {
408 
409                                 } finally {
410                                     pendingFetchRequests.remove(requestID);
411                                 }
412                             }
413                         });
414                     result.get();
415                 } catch (Exception exception) {
416                     Throwable rootException = ExceptionUtils.getRootCause(exception);
417                     // Security token can't be refreshed
418                     if (isTokenExpiredException(rootException) && !account.usesSecurityToken()) {
419                         resetSqsClient(clientReference);
420                     } else {
421                         throw exception;
422                     }
423                 }
424             } else {
425                 try {
426                     ReceiveMessageResult result = clientReference.get().receiveMessage(request);
427                     //Message processing and commit
428                     handleMessages(result, clientReference.get(), exitWhenNoneAvailable, request);
429                 } catch (Exception exception) {
430                     // Security token can't be refreshed
431                     if (isTokenExpiredException(exception) && !account.usesSecurityToken()) {
432                         resetSqsClient(clientReference);
433                     } else {
434                         SnapDataException snapDataException = new SnapDataException
435                                 (exception, ERROR_FAIL_TO_FETCH_MESSAGE)
436                                 .withReason(exception.getMessage())
437                                 .withResolution(RESOLUTION_ADDRESS_ISSUE);
438                         errorViews.write(snapDataException);
439                     }
440                 } finally {
441                     pendingFetchRequests.remove(requestID);
442                 }
443             }
444         }
445     }

447     private void handleMessages(ReceiveMessageResult result, AmazonSQS client,
448                                 boolean exitWhenNoneAvailable, ReceiveMessageRequest request)
449             throws InterruptedException {
450         List<Message> messages = result.getMessages();
451         //Message processing and commit
452         if (messages.isEmpty() && exitWhenNoneAvailable) {
453             // If no messages is fetched during this fetch, we need consider to stop
454             // polling when messageCount is set to 0;
455             if (getApproximateMessagesNumber(request.getQueueUrl(), client) == 0) {
456                 isPolling.set(false);
457             }
458         }
459         for (final Message message : messages) {
460             writeMessageToOutputView(message, request.getQueueUrl());
461             if (autoAcknowledge) {
462                 acknowledge(client, message.getReceiptHandle(), request.getQueueUrl());
463             }
464         }
465     }

469     protected void writeMessageToOutputView(Message message, String queueUrl) {
470         Map<String, Object> messageMetadata = Maps.newLinkedHashMap();
471         messageMetadata.put(MESSAGE_ID_KEY, message.getMessageId());
472         messageMetadata.put(RECEIPT_HANDLE_OUTPUT_KEY, message.getReceiptHandle());
473         messageMetadata.put(QUEUE_URL_OUTPUT_KEY, queueUrl);
474         Map<String, MessageAttributeValue> messageAttributes = message.getMessageAttributes();
475         messageAttributes.keySet().forEach(key -> {
476             messageMetadata.put(key, messageAttributes.get(key).getStringValue());
477         });
478         Document metadataDocument = documentUtility.newDocument(messageMetadata);
479         synchronized (outputViews) {
480             outputViews.write(new BinaryOutput() {
481                 @Override
482                 public Document getHeader() {
483                     return metadataDocument;
484                 }
485 
486                 @Override
487                 public void write(final WritableByteChannel writeChannel) throws IOException {
488                     try (OutputStream os = Channels.newOutputStream(writeChannel)) {
489                         os.write(message.getBody().getBytes(CharEncoding.UTF_8));
490                     }
491                 }
492             });
493         }
494     }

507     /**
508      * Assemble the configuration for the QueueBuffer
509      *
510      * In order to make a message retrievable from the buffer, we should match the following
511      * settings in the receiveMessage request:
512      *  1. The receiveMessageAttribute in the request and in the bufferConfig should be identical
513      *  2. The receiveAttribute in the request and bufferConfig should be identical (Not
514      *  implemented in the Snap, yet)
515      *  3. The visibilityTimeout in the request should NOT be set.
516      *
517      *  The Client-Side buffering would not work if any of the above doesn't match.
518      *
519      *  Refer: https://github.com/aws/aws-sdk-java/blob/1.11.1010/aws-java-sdk-sqs/src/main/java/
520      *  com/amazonaws/services/sqs/buffered/QueueBuffer.java#L282-L295
521      *
522      * @return The config object for the buffer
523      */
524     @Override
525     protected QueueBufferConfig getQueueBufferConfig() {
526         QueueBufferConfig config = super.getQueueBufferConfig();
527         if (waitTimeoutSec != null) {
528             if (waitTimeoutSec > 0) {
529                 config.withLongPoll(true);
530                 config.withLongPollWaitTimeoutSeconds(waitTimeoutSec);
531             } else {
532                 config.withLongPoll(false);
533             }
534         }
535         if (maxDoneReceiveBatches != null) {
536             config.withMaxDoneReceiveBatches(maxDoneReceiveBatches);
537         }
538         if (maxInflightReceiveBatches != null) {
539             config.withMaxInflightReceiveBatches(maxInflightReceiveBatches);
540         }
541         if (visibilityTimeout != null && visibilityTimeout > 0) {
542             config.withVisibilityTimeoutSeconds(visibilityTimeout);
543         }
544         config.withReceiveMessageAttributeNames(Arrays.asList(ALL_MESSAGE_ATTRIBUTES));
545         return config;
546     }
