97      *
98      * @param graphClient
99      * @param driveType
100      * @return

102     public List<String> fetchUsers(GraphServiceClient graphClient, String driveType) {
103         List<String> usersList = new ArrayList<>();
104         if (ONEDRIVE.equals(driveType) || ALL.equalsIgnoreCase(driveType)) {
105             List<QueryOption> options = new ArrayList<>();
106             // Microsoft API offers max page size of 999
107             // Refer to https://docs.microsoft.com/en-us/graph/api/user-list?view=graph-rest-1.0&tabs=http
108             //   for more details
109             options.add(new QueryOption("$top", "999"));
110             options.add(new QueryOption(SELECT, "id, displayName, userPrincipalName"));
111             UserCollectionRequest request = graphClient.users().buildRequest(options);
112             try {
113                 while (request != null) {
114                     UserCollectionPage collection = request.get();
115                     List<User> users = collection.getCurrentPage();
116                     users.stream().forEach(user -> {
117                         if (StringUtils.isBlank(user.displayName)) {
118                             usersList.add(user.userPrincipalName);
119                         } else {
120                             usersList.add(user.displayName + ":" + user.userPrincipalName);
121                         }
122                     });
123                     if (collection.getNextPage() != null) {
124                         request = collection.getNextPage().buildRequest(getUserAgentHeader());
125                     } else {
126                         request = null;
127                     }
128                 }
129             } catch (GraphServiceException ex) {
130                 if (handleThrottleError(ex)) {
131                     return fetchUsers(graphClient, driveType);
132                 }
133                 throw new SnapDataException(ex, FETCH_USERS_ERROR)
134                         .withReason(ex.getResponseCode() + ":" + ex.getResponseMessage())
135                         .withResolution(SUGGESTIONS_ERROR_RESOLUTION);
136             }
137         } else {
138             LOGGER.error(FETCH_USERS_ERROR + ":" + FETCH_SUGGESIONS_REASON);
139             throw new SnapDataException(FETCH_USERS_ERROR)
140                     .withReason(FETCH_SUGGESIONS_REASON)
141                     .withResolution(FETCH_USERS_RESOLUTION);
142         }
143         return usersList;
144     }

215         try {
216             Map<String, String> results = new LinkedHashMap<>();
217             DirectoryObject memberOf;
218             List<Option> requestOptions = mergeOptions(queryOptions, getUserAgentHeader());
219             if (!StringUtils.isBlank(userName)) {
220                 String userPrincipalName = extractUserPrincipalNameFromUser(userName);
221                 memberOf = graphClient.users(userPrincipalName).memberOf("microsoft.graph.group")
222                         .buildRequest(requestOptions).get();
223             } else {
224                 memberOf = graphClient.me().memberOf("microsoft.graph.group")
225                         .buildRequest(requestOptions).get();
226             }
227             Iterator iterator = memberOf.additionalDataManager().get("value").getAsJsonArray().iterator();
228             while (iterator.hasNext()) {
229                 JsonObject group = (JsonObject) iterator.next();
230                 if (group.get("displayName") == null) {
231                     throw new SnapDataException(FETCH_GROUPS_ERROR)
232                             .withReason("Not enough permissions")
233                             .withResolution(SUGGESTIONS_ERROR_RESOLUTION);
234                 }
235                 JsonElement displayNameElement = group.get("displayName");
236                 JsonElement idElement = group.get("id");
237                 if (!displayNameElement.isJsonNull()) {
238                     String displayName = displayNameElement.getAsString();
239                     String id = idElement.getAsString();
240                     results.put(displayName, id);
241                 }
242             }
243             return results;
244         } catch (GraphServiceException ex) {
245             if (handleThrottleError(ex)) {
246                 return buildUsersGroupIdNameMap(graphClient, queryOptions, userName);
247             }
248             throw new SnapDataException(ex, FETCH_GROUPS_ERROR)
249                     .withReason(ex.getResponseCode() + ":" + ex.getResponseMessage())
250                     .withResolution(SUGGESTIONS_ERROR_RESOLUTION);

638     /**
639      * Handle ApplicationThrottled Error 429 gracefully. Refer below doc for best practices
640      * https://docs.microsoft.com/en-us/graph/throttling?context=graph%2Fapi%2F1.0
641      * &view=graph-rest-1.0
642      * @param serviceEx GraphServiceException
643      * @return true, if needs to be retried
644      */
645     public boolean handleThrottleError(GraphServiceException serviceEx) {
646         if (serviceEx.getResponseCode() == 429) {
647             LOGGER.warn("Received ApplicationThrottled Error, Error code: 429", serviceEx);
648             List<String> responseHeaders = serviceEx.getResponseHeaders();
649             int waitTime = 0;
650             if (Objects.nonNull(responseHeaders)) {
651                 for (String header : responseHeaders) {
652                     if (header.startsWith("Retry-After")) {
653                         String[] headerVal = header.split(":");
654                         try {
655                             waitTime = (headerVal.length > 1) ?
656                                     Double.valueOf(headerVal[1].trim()).intValue() : 0;
657                             // avoid cases if negative number is passed.
658                             waitTime = (waitTime < 0) ? 0 : waitTime;
659                         } catch (NumberFormatException nfe) {
660                             throw new SnapDataException(ERR_CONVERT_FAIL_MSG)
661                                     .withReason(ERR_CONVERT_FAIL_REASON)
662                                     .withResolutionAsDefect();
663                         }
664                         break;
665                     }
666                 }
667             }
668             try {
669                 // Wait for suggested time plus extra 5 seconds before retrying.
670                 Thread.sleep(TimeUnit.SECONDS.toMillis(waitTime + 5));
671             } catch (InterruptedException ie) {
672                 LOGGER.error("Wait Thread is interrupted", ie);
673                 Thread.currentThread().interrupt();
674             }
675             return true;
676         }
677         return false;
678     }
