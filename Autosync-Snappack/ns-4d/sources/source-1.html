35     public void configure(Dispatch<SOAPMessage> dispatch, long timeoutInSecs) {

37 
38         HTTPClientPolicy httpClientPolicy = new HTTPClientPolicy();

40         httpClientPolicy.setReceiveTimeout(timeoutInMillis);
41         httpClientPolicy.setAutoRedirect(true);
42 
43         HTTPConduit httpConduit = (HTTPConduit) ((DispatchImpl) dispatch).getClient().getConduit();
44         httpConduit.setClient(httpClientPolicy);
