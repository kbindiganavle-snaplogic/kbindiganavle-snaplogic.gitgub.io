184     /**
185      * Returns the absolute path of the element being visited
186      * (NOTE: This method doesn't work if a macro is being generated)
187      *
188      * @return an absolute JSON path followed by a '.'
189      */
190     public String getAbsolutePath() {
191         checkIfDisposed();
192         if (StringUtils.isEmpty(currentCollectionPath)) {
193             return peekPath();
194         }
195         // Check if the collection path is the same as the accessor variable
196         if (currentCollectionPath.equals(collectionPathToAccessorMap.get(currentCollectionPath))) {
197             return peekPath();
198         }
199 
200         // If not, then attach the relative path to the collection path
201         String relativePath = peekPath();
202         int firstIndexOfDot = relativePath.indexOf(JsonPathBuilder.CURRENT_ELEMENT);
203         if (firstIndexOfDot != -1) {
204             return currentCollectionPath.concat(relativePath.substring(firstIndexOfDot));
205         }
206         return currentCollectionPath.concat(JsonPathBuilder.CURRENT_ELEMENT);
207     }

272      * Includes a foreach directive in the given element
273      *
274      * @param element         A SOAP element
275      * @param elementVarName  A name for accessor variable
276      * @param collectionPath  A path for collection
277      * @throws SOAPException
278      */
279     public void includeForeachDirective(SOAPElement element, String elementVarName,
280             String collectionPath) throws SOAPException {
281         checkIfDisposed();
282         String forEachDirective = String.format(FOREACH_DIRECTIVE_FORMAT, elementVarName,
283                 collectionPath);
284         element.addTextNode(forEachDirective);
285         previousCollectionPathLengthStack.push(currentCollectionPath.length());
286         currentCollectionPath = StringUtils.EMPTY.equals(currentCollectionPath) ? collectionPath :
287                 currentCollectionPath
288                         .concat(collectionPath.substring(collectionPath.indexOf(
289                                 JsonPathBuilder.CURRENT_ELEMENT)));
290         collectionPathToAccessorMap.put(currentCollectionPath, elementVarName);
291     }

309      *

311      * @return     A prefix for the given namespace
312      * @throws SOAPException
313      */
314     public String addNamespaceDeclIfAbsent(String tns) throws SOAPException {
315         checkIfDisposed();
316         if (namespaceToPrefixCache.containsKey(tns)) {
317             return namespaceToPrefixCache.get(tns);
318         }
319         String prefixForTns = prefixGenerator.generatePrefix();
320         namespaceToPrefixCache.put(tns, prefixForTns);
321         SOAPEnvelope soapEnvelope = soapMessageHolder.getSoapEnvelope();
322         soapEnvelope.addNamespaceDeclaration(prefixForTns, tns);
323         return prefixForTns;
324     }
