181         // if empty S3 credential, check if IAM-Credential-for-S3 mode is enabled.
182         if (StringUtils.isBlank(awsAccessId) || StringUtils.isBlank(awsSecretKey)) {
183             String ec2CredentialForS3 = System.getProperty(IAM_CREDENTIAL_FOR_S3);
184             if (iamRoleEnabled || Boolean.parseBoolean(ec2CredentialForS3)) {
185                 // let AWS SDK get S3 credential from EC2 metadata
186                 s3Client = createS3ClientWithIam();
187             } else {
188                 throw new ConfigurationException(ERR_MISSING_S3_CREDENTIAL)
189                         .withReason(REASON_MISSING_S3_CREDENTIAL)
190                         .withResolution(RESOLUTION_ADDRESS_ISSUE);
191             }
192         } else {
193             int colonIndex = awsSecretKey.indexOf(COLON);
194             AWSCredentials awsCreds;
195             if (awsToken != null) {
196                 awsCreds = new BasicSessionCredentials(awsAccessId, awsSecretKey, awsToken);
197             } else if (colonIndex > -1) {
198                 awsCreds = new BasicSessionCredentials(awsAccessId,
199                         awsSecretKey.substring(0, colonIndex),
200                         awsSecretKey.substring(colonIndex + 1));
201             } else {
202                 awsCreds = new BasicAWSCredentials(awsAccessId, awsSecretKey);
203             }
204             s3Client = new AmazonS3Client(awsCreds, UrlUtils.createClientConfiguration());
205         }
206         initAndVerifyBucket();

305      * Serializes data and writes to given output stream.
306      * @param data         - Map data
307      * @param outputStream - output stream
308      */
309     public void write(Map<String, Object> data, OutputStream outputStream) {
310         try {
311             boolean firstColumn = true;
312             for (String column : copyColumns) {
313                 Object value = data.get(column);
314                 if (!firstColumn) {
315                     outputStream.write(getDelimiter());
316                 }
317                 firstColumn = false;
318                 if (value != null) {
319                     if (value instanceof String) {
320                         writeEncodedString((String) value, outputStream);
321                     } else if (value instanceof ReadableInstant ||
322                             value instanceof ReadablePartial) {
323                         writeDateField(column, value, outputStream);
324                     } else {
325                         outputStream.write(value.toString().getBytes());
326                     }
327                 } else if (snowflakeLoadEmptyStrings) {
328                     outputStream.write(snowflakeSubstituteValueForNull);
329                 }
330             }
331             outputStream.write(NEWLINE);
332         } catch (IOException ioe) {
333             throw new StagingException(ioe, ERROR_STREAMING_DATA).
334                     withReason(ioe.getLocalizedMessage());
335         }

343     public OutputStream getOutputStream() {
344         if (max == 1) {
345             return outputStreams.get(0);
346         }
347         CountingOutputStream outputStream = outputStreams.get(curIndex - 1);
348         if (outputStream.getCount() >= BUFFER_SIZE) {
349             // move to the next stream we have reached the max buffer
350             curIndex++;
351             outputStream.resetByteCount();
352             if (curIndex > max) {
353                 curIndex = 1;
354             }
355             return outputStreams.get(curIndex - 1);
356         }
357         return outputStream;
358     }

427     protected static String encodeString(String value) {
428         int extraCharCount = 0;
429         StringBuilder returnValue = null;
430         for (int index = 0; index < value.length(); index++) {
431             switch (value.charAt(index)) {
432                 case DELIMITER:
433                 case NEWLINE:
434                 case CARRIAGE_RETURN:
435                 case ESCAPE_CHARACTER:
436                     if (returnValue == null) {
437                         returnValue = new StringBuilder(value);
438                     }
439                     returnValue.insert(extraCharCount + index, ESCAPE_CHARACTER);
440                     extraCharCount++;
441                     break;
442                 case REDSHIFT_NULL_CHARACTER:
443                     // '\0' can only be used as a single-character string which represents a NULL
444                     // value for Redshift, otherwise it's illegal character and should be skipped.
445                     if (value.length() > 1) {
446                         if (returnValue == null) {
447                             returnValue = new StringBuilder(value);
448                         }
449                         returnValue.deleteCharAt(extraCharCount + index);
450                     }
451                     extraCharCount--;
452                     break;
453                 default:
454                     // No-Op
455             }
456         }
457         return returnValue == null ? value : returnValue.toString();
458     }
