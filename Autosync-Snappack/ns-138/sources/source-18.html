176     private void provideInputViewSchema(SchemaProvider provider) {
177         loginInfo = getLoginInfo(false);
178         if (loginException != null) {
179             LOG.error("Unable to generate input view schema {}", loginException.getMessage(),
180                     loginException);
181             return;
182         }
183         try (MetadataRead metadataRead = injector.getInstance(MetadataRead.class)) {
184             metadataRead.initialize(options, loginInfo);
185             Operation operation = options.getOperation();
186             for (String inputViewName : provider.getRegisteredViewNames()) {
187                 SchemaBuilder schemaBuilder = provider.getSchemaBuilder(inputViewName);
188                 setSchemaForRelatedObject(provider, schemaBuilder);
189                 if (operation == Operation.DELETE) {
190                     Schema childSchema = createChildSchema(DELETE_SCHEMA, provider);
191                     schemaBuilder.withChildSchema(childSchema);
192                 } else {
193                     Iterator<Map<String, Object>> results = null;
194                     try {
195                         results = metadataRead.execute();
196                     } catch (RetryException e) {
197                         if (retriesRemaining-- > 0) {
198                             loginInfo = getLoginInfo(true);
199                             provideInputViewSchema(provider);
200                             resetRetry();
201                             return;
202                         } else {
203                             LOG.info(ERR_READ_META_DATA, e);
204                         }
205                     } catch (IOException e) {
206                         LOG.info(ERR_READ_META_DATA, e);
207                     }
208                     if (results == null) {
209                         return;
210                     }
211                     while (results.hasNext()) {
212                         Map<String, Object> entry = results.next();
213                         Boolean isCreatable = jsonPathUtil.nullableRead(CREATEABLE, entry);
214                         Boolean isUpdateable = jsonPathUtil.nullableRead(UPDATEABLE, entry);
215                         Boolean isIdLookup = jsonPathUtil.nullableRead(ID_LOOKUP, entry);
216                         String name = String.valueOf(entry.get(NAME));
217                         switch (operation) {
218                             case INSERT:
219                                 if (ID.equalsIgnoreCase(name) ||
220                                         isCreatable != null && !isCreatable) {
221                                     continue;
222                                 }
223                                 break;
224                             case UPDATE:
225                                 if ((isIdLookup != null && !isIdLookup) && (isUpdateable != null &&
226                                         !isUpdateable)) {
227                                     continue;
228                                 }
229                                 break;
230                             case UPSERT:
231                                 if (ID.equalsIgnoreCase(name) || (isIdLookup != null &&
232                                         !isIdLookup) && (isUpdateable != null && !isUpdateable) &&
233                                         (isCreatable != null && !isCreatable)) {
234                                     continue;
235                                 }
236                                 break;
237                             // TODO: add external ID information if any
238                             default:
239                                 String error = String.format(ERR_SCHEMA_PROVIDER_UNSUPPORTED,
240                                         operation.toString());
241                                 throw new ExecutionException(error)
242                                         .withReason(error)
243                                         .withResolution(ERR_CUSTOMER_SUPPORT);
244                         }
245                         Schema childSchema = createChildSchema(entry, provider);
246                         schemaBuilder.withChildSchema(childSchema);

248                 }
249                 schemaBuilder.build();

389     public void execute() throws ExecutionException {
390         if (batchTimeoutMs <= 0 || inputViews.size() != 1) {
391             super.execute();
392             return;
393         }
394         // if batch timeout is positive and input view is connected, handle batch timeout
395         lastBatchWriteTime = System.currentTimeMillis();
396         Document lastDocument = documentUtility.newDocument(new HashMap());
397         String inputViewName = inputViews.get().getName();
398         Map<String, Document> map;
399         while (true) {
400             // poll input view for a document
401             map = inputViews.select();
402             if (map.isEmpty()) {
403                 // input document has not arrived yet
404                 if (batchTimeoutMs < (System.currentTimeMillis() - lastBatchWriteTime)) {
405                     // batch timeout has occurred
406                     if (!sObjectList.isEmpty()) {
407                         // submit the batch
408                         LOG.info("Write batch request is sent due to a batch timeout");
409                         writeBatch(lastDocument);
410                     } else {
411                         // the batch buffer is empty, just reset timeout timer
412                         lastBatchWriteTime = System.currentTimeMillis();
413                         LOG.info("Batch timeout has occurred, but batch buffer is empty");
414                     }
415                 } else {
416                     // batch timeout has not occurred yet, pause here, otherwise cpu may hit 100%
417                     sleep(1, TimeUnit.SECONDS);
418                 }
419             } else {
420                 // input document is present
421                 Document document = map.get(inputViewName);
422                 if (document != null) {
423                     validateServiceVersion(options.getServiceVersion());
424                     process(document, inputViewName);
425                     lastDocument = document;
426                 } else {
427                     // it is the end of the input document stream
428                     return;
429                 }
430             }
431         }
432     }

435     protected void process(Document document, String inputViewName) {
436         if (bulkAPI) {
437             docsList.add(document);
438             sObjectList.add(getSObject(document));
439             if (sObjectList.size() == batchSize) {
440                 writeBatch(document);
441                 lastBatchWriteTime = System.currentTimeMillis();
442                 docsList.clear();
443             }
444         } else {
445             loginInfo = getLoginInfo(false); // make sure access token is not expired
446             if (loginException != null) {
447                 errorViews.write(loginException, document);
448                 return;
449             }
450             try {
451                 syncWrite.initialize(options, loginInfo);
452                 RestWriteResult restWriteResult = syncWrite.execute(getSObject(document));
453                 writeObjectToView(document, restWriteResult);
454             } catch (RetryException e) {
455                 if (retriesRemaining-- > 0) {
456                     loginInfo = getLoginInfo(true);
457                     LOG.info("Retrying REST API request in {}, retires remaining: {}",
458                             snapTitle, retriesRemaining, e);
459                     process(document, inputViewName);
460                     LOG.info("Retry of REST API request in {} was successful", snapTitle, e);
461                     resetRetry();
462                 } else {
463                     LOG.info("All retries of REST API request in {} have failed, routing input " +
464                             "data to error view", snapTitle, e);
465                     throwSnapDataException(e, ERR_REQUEST, document);
466                 }
467             } catch (SnapDataException e) {
468                 errorViews.write(e, document);
469             } catch (ExecutionException | SuggestViewAbortException e) {
470                 throw e;
471             } catch (Throwable e) {
472                 throwSnapDataException(e, ERR_REQUEST, document);
473             } finally {
474                 syncWrite.close();
475             }
476         }
477     }

479     private void writeBatch(Document document) {
480         try {
481             loginInfo = getLoginInfo(false); // make sure access token is not expired
482             if (loginException != null) {
483                 if (docsList.isEmpty()) {
484                     errorViews.write(loginException, document);
485                 } else {
486                     for (Document d : docsList) {
487                         errorViews.write(loginException, d);
488                     }
489                 }
490                 return;
491             }
492             bulkWrite.setLoginInfo(loginInfo);
493             bulkWrite.setPayloadContentType(bulkContentType);
494             Map<String, Object> batchJobInfo = bulkWrite.execute(sObjectList);
495             if (isPublisher) {
496                 getBatchResults(batchJobInfo, document);
497             } else {
498                 writeResultsToOutput(document, batchJobInfo);
499             }
500         } catch (RetryException e) {
501             retryBulkWrite(e, document, ERR_REQUEST);
502         } catch (Throwable e) {
503             throwSnapDataException(e, ERR_REQUEST);
504         } finally {
505             sObjectList.clear();
506             bulkWrite.close();
507         }
508     }

520         JfsUtils.sleepNoInterrupt(retryInterval, TimeUnit.SECONDS, LOG);
521         Map<String, Object> batchJobInfo = null;
522         if (retriesRemaining-- > 0) {
523             loginInfo = getLoginInfo(true);
524             if (loginException != null) {
525                 if (docsList.isEmpty()) {
526                     errorViews.write(loginException, document);
527                 } else {
528                     for (Document d : docsList) {
529                         errorViews.write(loginException, d);
530                     }
531                 }
532                 return;
533             }
534             try {
535                 LOG.info("Retrying Bulk API request in {}, retires remaining: {}",
536                         snapTitle, retriesRemaining, exception);
537                 batchJobInfo = bulkWrite.execute(sObjectList);
538                 LOG.info("Retry of Bulk API request in {} was successful", snapTitle, exception);
539                 resetRetry();
540             } catch (SuggestViewAbortException e) {
541                 throw e;
542             } catch (Exception ex) {
543                 retryBulkWrite(exception, document, error);
544             }
545         } else {
546             LOG.info("All retries of Bulk API request in {} have failed, routing input data to" +
547                     " error view", snapTitle, exception);
548             for (SObject sObject : sObjectList) {
549                 throwSnapDataException(exception, error, sObject.getAllFields());
550             }
551         }
552         if (MapUtils.isNotEmpty(batchJobInfo)) {
553             if (isPublisher) {
554                 getBatchResults(batchJobInfo, document);
555             } else {
556                 writeResultsToOutput(document, batchJobInfo);
557             }
558         }
559     }

616         try {
617             if (bulkAPI && CollectionUtils.isNotEmpty(sObjectList)) {
618                 Document document = getDocumentFrom(sObjectList);
619                 try {
620                     loginInfo = getLoginInfo(false); // make sure access token is not expired
621                     if (loginException != null) {
622                         errorViews.write(loginException, document);
623                         return;
624                     }
625                     bulkWrite.setLoginInfo(loginInfo);
626                     bulkWrite.setPayloadContentType(bulkContentType);
627                     Map<String, Object> batchJobInfo = bulkWrite.execute(sObjectList);
628                     if (isPublisher) {
629                         getBatchResults(batchJobInfo, getDocumentFrom(sObjectList));
630                     } else {
631                         writeResultsToOutput(document, batchJobInfo);
632                     }
633                 } catch (RetryException | ConnectException e) {
634                     retryBulkWrite(e, document, ERR_LAST_BATCH);
635                 } catch (IOException e) {
636                     String error = e.getMessage();
637                     if (StringUtils.containsIgnoreCase(error, ERR_NETWORK_UNREACHABLE) ||
638                             StringUtils.containsIgnoreCase(error, ERR_UNKNOWN)) {
639                         retryBulkWrite(e, document, ERR_LAST_BATCH);
640                     } else {
641                         throwSnapDataException(e, ERR_LAST_BATCH, document);
642                     }
643                 } catch (Exception e) {
644                     throwSnapDataException(e, ERR_LAST_BATCH, document);
645                 } finally {
646                     sObjectList.clear();
647                 }
648             }
649         } finally {
650             if (bulkWrite != null) {
651                 try {
652                     bulkWrite.closeBatchJob();
653                 } finally {
654                     bulkWrite.close();
655                 }
656             }
657             if (syncWrite != null) {

703     private void suggestRelatedNames(final SuggestionBuilder suggestionBuilder,
704             final PropertyValues propertyValues, final String suggestProperty,
705             final String objectTypeProperty) {
706         Login loginInfo = getLoginInfo(false);
707         if (loginException != null) {
708             throw new ExecutionException(loginException, loginException.getMessage());
709         }
710         String[] suggestedNames;
711         String compositeKey = null;
712         String objectType = propertyValues.get(objectTypeProperty);
713         if (StringUtils.isNotBlank(objectType)) {
714             compositeKey = new StringBuilder(suggestProperty)
715                     .append(loginInfo.getSessionId())
716                     .append(objectType)
717                     .toString();
718             suggestedNames = FIELD_NAME_CACHE.getIfPresent(compositeKey);
719         } else {
720             suggestedNames = EMPTY_ARRAY;
721         }
722         if (suggestedNames == null) {
723             MetadataRead metadataRead = injector.getInstance(MetadataRead.class);
724             options.withObjectType(objectType);
725             options.withServiceVersion(propertyValues.get(SERVICE_VERSION_PROP));
726             String instanceUrl = ((SalesforceLogin) loginInfo).getInstanceUrl();
727             versionedBaseServiceUrl = String.format(FORMAT_REST_SERVICE_URL,
728                     makeSureEndWithSeparator(instanceUrl), options.getServiceVersion());
729             options.withServiceUrl(versionedBaseServiceUrl);
730             metadataRead.initialize(options, loginInfo);
731             Iterator<Map<String, Object>> results;
732             try {
733                 results = metadataRead.execute();
734             } catch (RetryException e) {
735                 if (retriesRemaining-- > 0) {
736                     suggestRelatedNames(suggestionBuilder, propertyValues, suggestProperty,
737                             objectTypeProperty);
738                     resetRetry();
739                     return;
740                 } else {
741                     throw new ExecutionException(e, ERR_READ_META_DATA)
742                             .withReason(e.getMessage())
743                             .withResolution(ERR_LOGIN_RESOLUTION);
744                 }
745             } catch (IOException e) {
746                 throw new ExecutionException(e, ERR_READ_META_DATA)
747                         .withReason(e.getMessage())
748                         .withResolution(ERR_LOGIN_RESOLUTION);
749             }
750             if (results == null) {
751                 return;
752             }
753             Set<String> suggestedNameSet = new HashSet<>();
754             while (results.hasNext()) {
755                 Map<String, Object> entry = results.next();
756                 Object object;
757                 switch (suggestProperty) {
758                     case RELATED_OBJECT_PROP:
759                         object = entry.get(REFERENCE_TO);
760                         if (object instanceof List && !((List<?>) object).isEmpty()) {
761                             suggestedNameSet.addAll((List<String>) object);
762                         }
763                         break;
764                     case RELATED_EXTERNAL_ID_PROP:
765                         object = entry.get(EXTERNAL_ID);
766                         if (object instanceof Boolean && (Boolean) object) {
767                             suggestedNameSet.add((String) entry.get(NAME));
768                         }
769                         break;
770                     default:
771                         suggestionBuilder.node(suggestProperty)
772                                 .suggestions(EMPTY_ARRAY);
773                         return;
774                 }
775             }
776             if (suggestedNameSet.isEmpty()) {
777                 // If we pass an empty array to the suggestion builder, it gives a wrong error
778                 // message. We should pass an array with one empty string.
779                 suggestedNames = EMPTY_ARRAY;
780             } else {
781                 suggestedNames = suggestedNameSet.toArray(new String[0]);
782             }
783             FIELD_NAME_CACHE.put(compositeKey, suggestedNames);
784         }
785         suggestionBuilder.node(suggestProperty)
786                 .suggestions(suggestedNames);
787     }
