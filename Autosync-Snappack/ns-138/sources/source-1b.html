315     /**
316      * Login to Salesforce.com:
317      * If it has successfully logged into Salesforce.com with given credentials, it
318      * reuses the SalesforceLogin object without login again, or else, logs in to Salesforce.com
319      * and creates a SalesforceLogin object. The SalesforceLogin object contains an access token
320      * and an instance URL.
321      *
322      * @param invalidate - if true, invalidate cached Login object
323      *
324      * @return SalesforceLogin object
325      */
326     protected Login getLoginInfo(boolean invalidate) {
327         if (account == null) {
328             throw new ConfigurationException(ERR_ACCOUNT_NOT_SPECIFIED)
329                 .withReason(REASON_ACCOUNT_NOT_SPECIFIED)
330                 .withResolution(RESOLUTION_ACCOUNT_NOT_SPECIFIED);
331         }
332 
333         if (account instanceof SalesforceAccount) {
334             ((SalesforceAccount) account).setInvalidate(invalidate);
335         }
336         try {
337             Login login = account.connect();
338             if (account instanceof SalesforceAccount ||
339                     account instanceof SalesforceOauth2Account) {
340                 // [SWAT-1984] If dynamic account, resetRetry() should not be executed. Otherwise,
341                 // it may cause an infinite recursive call to process() for SOQL & Read Snaps.
342                 resetRetry();
343             }
344             statusReporter.reset();
345             LOGGER.debug("Successfully logged in: {}", login);
346             return login;
347         } catch (ExecutionException e) {
348             LOGGER.error("Failed to connect {}", e.getMessage(), e);
349             if (!isInvalidCredential(e) && retriesRemaining > 0) {
350                 statusReporter.notify(RETRYING_LOGIN_MESSAGE, retryInterval, retriesRemaining,
351                         e.getCause());
352                 JfsUtils.sleepNoInterrupt(retryInterval, TimeUnit.SECONDS, LOGGER);
353                 LOGGER.info("Retrying login in {}, retires remaining: {}",
354                         snapTitle, retriesRemaining, e);
355                 retriesRemaining--;
356                 return getLoginInfo(true);
357             } else {
358                 loginException = new SnapDataException(e, e.getMessage())
359                         .withReason(e.getReason())
360                         .withResolution(e.getResolution());
361                 return null;
362             }
363         }
364     }

482     protected static void suggestServiceVersions(final SuggestionBuilder suggestionBuilder,
483             final String startVersion) {
484         try {
485             List<String> versions = SERVICE_VERSION_CACHE.get(SERVICE_VERSION_PROP);
486             versions = versions == null ? new ArrayList<>() : versions;
487             List<String> filteredVersions = new ArrayList<>(versions.size());
488             for (String version : versions) {
489                 if (StringUtils.isNotBlank(version) &&
490                         Float.valueOf(version) >= Float.valueOf(startVersion)) {
491                     filteredVersions.add(version);
492                 }
493             }
494             if (filteredVersions.isEmpty()) {
495                 filteredVersions.add(ERR_GET_VERSIONS);
496             }
497             suggestionBuilder.node(SERVICE_VERSION_PROP)
498                     .suggestions(filteredVersions.toArray(new String[0]));
499         } catch (Exception e) {
500             suggestionBuilder.node(SERVICE_VERSION_PROP)
501                     .suggestions(ERR_GET_VERSIONS);
502         }
503     }

505     @VisibleForTesting
506     protected static List<String> getServiceVersion() {
507         List<String> versions = null;
508         HttpGet httpGet = new HttpGet(URL_SERVICE_VERSIONS);
509         HttpClient client = new ProxyHttpClient();
510         try {
511             HttpResponse httpResponse = client.execute(httpGet);
512             if (SC_OK == httpResponse.getStatusLine().getStatusCode()) {
513                 HttpEntity entity = httpResponse.getEntity();
514                 String versionsStr = EntityUtils.toString(entity);
515                 List<Map<String, String>> versionList =
516                         OBJECT_MAPPER.readValue(versionsStr, LIST_MAP_STRING_REF);
517                 versions = Lists.newArrayListWithExpectedSize(versionList.size());
518                 for (Map<String, String> versionMap : versionList) {
519                     String version = versionMap.get(VERSION);
520                     if (StringUtils.isNotBlank(version)) {
521                         versions.add(version);
522                     }
523                 }
524                 // the latest version on top
525                 Collections.reverse(versions);
526             } else {
527                 StatusLine statusLine = httpResponse.getStatusLine();
528                 int statusCode = statusLine.getStatusCode();
529                 String reasonPhrase = statusLine.getReasonPhrase();
530                 HttpEntity entity = httpResponse.getEntity();
531                 String errorEntity = StringUtils.EMPTY;
532                 if (entity != null) {
533                     errorEntity = EntityUtils.toString(entity);
534                 }
535                 LOGGER.error(LOG_FAILED_GET_VERSIONS, String.format(FORMAT_ERR_VERSIONS,
536                         statusCode, reasonPhrase, errorEntity));
537             }
538         } catch (IOException e) {
539             LOGGER.error(LOG_FAILED_GET_VERSIONS, e.getMessage());
540         }
541         if (CollectionUtils.isEmpty(versions)) {
542             if (LAST_SERVICE_VERSIONS.isEmpty()) {
543                 throw new SnapDataException(ERR_GET_VERSIONS)
544                         .withReason(String.format(REASON_ACCESS_SFDC, URL_SERVICE_VERSIONS))
545                         .withResolution(RESOLUTION_NETWORK_STABLE);
546             } else {
547                 return LAST_SERVICE_VERSIONS;
548             }
549         } else {
550             LAST_SERVICE_VERSIONS.clear();
551             LAST_SERVICE_VERSIONS.addAll(versions);
552         }
553         return versions;

557      * Validates the evaluated value of the 'Service Version' expression.
558      *
559      * @param version - evaluated value of the 'Service Version' expression
560      * @return - service version string if it is part of versions supported by SFDC
561      * @throws SnapDataException if it is invalid version
562      */
563     protected static String validateServiceVersion(Object version) {
564         List<String> versions = null;
565         boolean versionsFound = false;
566         try {
567             versions = SERVICE_VERSION_CACHE.get(SERVICE_VERSION_PROP);
568             versionsFound = true;
569         } catch (SuggestViewAbortException e) {
570             throw e;
571         } catch (Exception e) {
572             LOGGER.warn(String.format(REASON_ACCESS_SFDC, URL_SERVICE_VERSIONS), e);
573         }
574         String versionStr = version instanceof String ? (String) version : version.toString();
575         versionStr = versionStr.startsWith(V) ? versionStr.substring(1) : versionStr;
576         if (versionsFound && !versions.contains(versionStr)) {
577             throw new ConfigurationException(ERR_INVALID_VERSION)
578                     .formatWith(version)
579                     .withReason(REASON_VERSION_NOT_IN_SERVICE)
580                     .withResolution(RESOLUTION_ADDRESS_ISSUE);
581         }
582         return versionStr;
583     }
