219     private ObjectSchema prepareSchema(SchemaProvider provider, Options.Operation operation) {
220         ObjectSchema rootSchema = schemaFactory.createSchema(SnapType.COMPOSITE, "rootSchema");
221         setSchemaForRelatedObject(provider, rootSchema);
222         if (operation == Options.Operation.DELETE) {
223             Schema childSchema = createChildSchema(DELETE_SCHEMA, provider);
224             rootSchema.addChild(childSchema);
225 
226         } else {
227             List<Map<String, Object>> fieldsMap = getFields(null,
228                     options.getObjectType(), options.getServiceVersion());
229             for (Map<String, Object> fieldMap : fieldsMap) {
230                 Boolean isCreatable = jsonPathUtil.nullableRead(CREATEABLE, fieldMap);
231                 Boolean isUpdateable = jsonPathUtil.nullableRead(UPDATEABLE, fieldMap);
232                 Boolean isIdLookup = jsonPathUtil.nullableRead(ID_LOOKUP, fieldMap);
233                 String name = String.valueOf(fieldMap.get(NAME));
234                 switch (operation) {
235                     case INSERT:
236                         if (ID.equalsIgnoreCase(name) || isCreatable != null && !isCreatable) {
237                             continue;
238                         }
239                         break;
240                     case UPDATE:
241                         if ((Objects.nonNull(isIdLookup) && !isIdLookup) &&
242                                 (Objects.nonNull(isUpdateable) && !isUpdateable)) {
243                             continue;
244                         }
245                         break;
246                     case UPSERT:
247                         if (ID.equalsIgnoreCase(name) || (isIdLookup != null && !isIdLookup) &&
248                                 (isUpdateable != null && !isUpdateable) &&
249                                 (isCreatable != null && !isCreatable)) {
250                             continue;
251                         }
252                         break;
253                     default:
254                         String error = String.format(ERR_SCHEMA_PROVIDER_UNSUPPORTED,
255                                 operation.toString());
256                         throw new ExecutionException(error)
257                                 .withReason(error)
258                                 .withResolution(ERR_CUSTOMER_SUPPORT);
259                 }
260                 Schema childSchema = createChildSchema(fieldMap, provider);
261                 rootSchema.addChild(childSchema);
262             }
263         }
264         return rootSchema;
265     }

274     private Schema createChildSchema(Map<String, Object> entry, SchemaProvider provider) {
275         String name = jsonPathUtil.nullableRead(NAME, entry);
276         String type = jsonPathUtil.nullableRead(TYPE, entry);
277         Object lengthObj = jsonPathUtil.nullableRead(LENGTH, entry);
278         int length = 0;
279         if (lengthObj instanceof BigInteger) {
280             length = ((BigInteger) lengthObj).intValue();
281         } else {
282             length = (Integer) lengthObj;
283         }
284         Boolean isNullable = jsonPathUtil.nullableRead(NILLABLE, entry);
285         SalesforceSchemaUtil.SnapTypeInfo snapTypeInfo = schemaUtil
286                 .getSnapTypeForSQLType(type);
287         SnapType snapType = snapTypeInfo.getSnapType();
288         Schema childSchema = provider.createSchema(snapType,
289                 snapTypeInfo.getSnapFormatType(), name);
290         if (ObjectUtils.equals(SnapType.STRING, snapType)) {
291             childSchema.setMaximumLength(length);
292         }
293         if (Objects.nonNull(isNullable) && !isNullable) {
294             childSchema.setRequired(true);
295         }
296         return childSchema;
297     }

490      *
491      * @param suggestionBuilder
492      * @param propertyValues

494     protected void suggestSObjects(final SuggestionBuilder suggestionBuilder,
495             final PropertyValues propertyValues) {
496         if (account instanceof SalesforceMutualAuthAccount) {
497             account.connect();
498         }
499         String serviceUrl =
500                 makeSureEndWithSeparator(getServiceUrlFromAccount());
501         options.withServiceUrl(serviceUrl)
502                 .withServiceVersion(propertyValues.get(SERVICE_VERSION_PROP));
503         String url = createFullUrl(FORMAT_SOBJECTS_URL);
504         try {
505             String entityString = processHttpResponse(url);
506             if (StringUtils.isBlank(entityString)) {
507                 suggestionBuilder.node(OBJECT_TYPE_PROP).suggestions(RETRIVE_DATA_ERROR);
508                 return;
509             }
510             Map<String, Object> sObjectMap = OBJECT_MAPPER.readValue(entityString, MAP_REF);
511             @SuppressWarnings("unchecked")
512             List<Map<String, Object>> sObjectsList =

700         Map<String, Object> data = document.get(Map.class);
701         if (MapUtils.isEmpty(data)) {

725                 writeDataToCsvPrinter(csvData, currentCsvDataSize);
726             }
727         } catch (SnapException e) {

729                     .withReason(e.getReason())
730                     .withResolution(Messages.RESOLUTION_ADDRESS_ISSUE);

810     private void writeResultsToView(Map<String, List<Map<String, String>>> resultsMap,
811             Document document) {
812         String reason = bulkJobInfo.getSfapiError().getResponse();
813         Map<String, List<Map<String, String>>> originalMap = new LinkedHashMap<>();
814 
815         if (Objects.nonNull(resultsMap.get(SUCCESSRECORDS))) {
816             resultsMap.get(SUCCESSRECORDS).forEach(record -> {
817                 if (document == null) {
818                     outputViews.write(documentUtility.newDocument(record));
819                 } else {
820                     outputViews.write(documentUtility.newDocumentFor(document, record));
821                 }
822             });
823         }
824         if (Objects.nonNull(resultsMap.get(FAILEDRECORDS)) &&
825                 resultsMap.get(FAILEDRECORDS).size() > 0) {
826             SnapDataException ex = logAndCreateException(reason);
827             if (document != null) {
828                 errorViews.write(ex, document);
829             } else {

831                         getFailedOriginalRecords(resultsMap.get(FAILEDRECORDS)));
832                 if (MapUtils.isNotEmpty(originalMap)) {
833                     errorViews.write(ex, documentUtility.newDocument(originalMap));
834                 }
835                 return;
836             }
837         }
838         if (Objects.nonNull(resultsMap.get(UNPROCESSEDRECORDS)) &&
839                 resultsMap.get(UNPROCESSEDRECORDS).size() > 0) {
840             originalMap.put(KEY_RECORDS, resultsMap.get(UNPROCESSEDRECORDS));
841             SnapDataException ex = logAndCreateException(reason);
842             if (MapUtils.isNotEmpty(originalMap)) {
843                 if (document == null) {
844                     errorViews.write(ex, documentUtility.newDocument(originalMap));
845                 } else {
