407     @Override
408     protected void process(Document document, String inputViewName) {
409         LOG.info("{}.process(): count={}", this, ++processCount);
410         initOptions(document);
411         // make sure access token is not expired
412         loginInfo = getLoginInfo(false);
413         if (loginException != null) {
414             errorViews.write(loginException, document);
415             return;
416         }
417         evalRetryProperties(document);
418         pkChunking = forcePKChunking && batchSize >= MIN_CHUNK_SIZE;
419         evalExpressionProperties(document);
420         prepareQueryStatement(document);
421         // check if there are related object fields
422         List<Map<String, String>> relatedObjectData = null;
423         if (isPKChunkingSuggest) {
424             relatedObjectData = checkRelatedObjectFields();
425         }
426         Iterator<Object> iterator = null;
427         try (AbstractRequest<Iterator<Object>, Void> requester = bulkAPI ?
428                 getBulkRead() : getSyncRead()) {
429             if (bulkAPI) {
430                 if (attemptPKChunking) {
431                     if (StringUtils.isNotBlank(options.getLimitClause()) || StringUtils.isNotBlank(options.getOrderbyClause())) {
432                         pkChunking = false;
433                         UNABLE_TO_PK_CHUNK_SETTINGS.report();
434                     } else {
435                         if (forcePKChunking) {
436                             if (PK_CHUNKING_TYPES.stream().noneMatch(v -> v.equalsIgnoreCase(objectType)) && !objectType.toLowerCase().endsWith("__c")) {
437                                 AbstractResponse bulkResponse = ((BulkRequest<Iterator<Object>, Void>) requester).getBulkResponse();
438                                 String parentEntityName = requester.getSalesforceFields().getParentEntity(bulkResponse, options, loginInfo);
439                                 options.withParentEntityName(parentEntityName);
440                                 if (StringUtils.isNotBlank(parentEntityName) &&
441                                         PK_CHUNKING_TYPES.stream().anyMatch(
442                                                 v -> v.equalsIgnoreCase(parentEntityName))) {
443                                     pkChunking = true;
444                                 } else {
445                                     pkChunking = false;
446                                 }
447                             } else {
448                                 pkChunking = true;
449                             }
450                             if (!pkChunking) {
451                                 UNABLE_TO_PK_CHUNK_TYPE.report(() -> new Object[]{objectType});
452                             }
453                         } else {
454                             pkChunking = false;
455                         }
456                     }
457                 }
458                 BulkRead bulkRead;
459                 if (requester instanceof BulkRead) {
460                     bulkRead = (BulkRead) requester;
461                     bulkRead.setErrorViews(errorViews, documentUtility);
462                 } else {
463                     throw new SnapDataException(ERR_SFDC_REQUEST_CLASS)
464                             .formatWith(requester.getClass().getSimpleName())
465                             .withReason(REASON_BULK_READ)
466                             .withResolutionAsDefect();
467                 }
468                 if (passThrough && document != null) {
469                     options.getSnapContext().withOriginal(document);
470                 }
471                 if (pkChunking) {
472                     if (StringUtils.isNotBlank(options.getLimitClause()) ||
473                             StringUtils.isNotBlank(options.getOrderbyClause())) {
474                         errorViews.write((SnapDataException)
475                                 new SnapDataException(ERR_MALFORMED_QUERY)
476                                 .withReason(REASON_ORDERBY_LIMIT_CLAUSES)
477                                 .withResolution(RESOLUTION_ORDERBY_LIMIT_CLAUSES), document);
478                         return;
479                     }
480                     bulkRead.execute(dateTimeFields, matchDataType, nonStringFields);
481                 } else {
482                     iterator = bulkRead.execute();
483                     long recordsDownloaded = 0;
484                     boolean nonStringFieldsPresent = MapUtils.isNotEmpty(nonStringFields);
485                     Document original = passThrough ? document : null;
486                     if (iterator == null) {
487                         // The Bulk API SOQL query length has exceeded the max limit.
488                         // Join the output documents resulted from the vertical slicing
489                         recordsDownloaded = joinVerticalSlicingBulkApi(bulkRead, nonStringFieldsPresent,
490                                 original);
491                     } else {
492                         while (iterator.hasNext()) {
493                             Object object = iterator.next();
494                             object = SalesforceSchemaUtil.convertDateTime(object, dateTimeFields);
495                             if (nonStringFieldsPresent) {
496                                 object = SalesforceSchemaUtil.matchDataType(object,
497                                         nonStringFields);
498                             }
499                             writeToOutputView(document, object, original);
500                             recordsDownloaded++;
501                         }
502                     }
503                     long recordsProcessed = bulkRead.getRecordsProcessed();
504                     if (recordsDownloaded != recordsProcessed) {
505                         String jobId = bulkRead.getJobId();
506                         if (validateRecordCountDownloaded) {
507                             errorViews.write((SnapDataException)
508                                     new SnapDataException(ERR_MATCH_NUMBER_RECORDS)
509                                     .formatWith(jobId, recordsDownloaded, recordsProcessed)
510                                     .withReason(REASON_DOWNLOAD_FAILED)
511                                     .withResolutionAsDefect());
512                         } else {
513                             LOG.info("The number of records downloaded does not match the number " +
514                                     "of records processed for job ID: {}, " +
515                                     "downloaded records: {}, processed records: {}",
516                                     jobId, recordsDownloaded, recordsProcessed);
517                         }
518                     }
519                 }
520             } else {
521                 iterator = requester.execute();
522                 Document original = passThrough ? document : null;
523                 boolean emptyResult = true;
524                 if (iterator == null) {
525                     // The REST API SOQL query length has exceeded the max limit.
526                     // Join the output documents resulted from the vertical slicing
527                     joinVerticalSlicesRestApi((SyncRead) requester, document);
528                 } else {
529                     while (iterator.hasNext()) {
530                         Object object = iterator.next();
531                         object = SalesforceSchemaUtil.convertDateTime(object, dateTimeFields);
532                         if (relatedObjectData != null) {
533                             object = modifyRelatedFields(object, relatedObjectData);
534                         }
535                         writeToOutputView(document, object, original);
536                         emptyResult = false;
537                     }
538                     if (emptyResult && !ignoreEmptyResult) {
539                         writeToOutputView(document, new HashMap<>(0), original);
540                     }
541                 }
542             }
543             // describe SObject for the second output view
544             describeSObject(requester);
545         } catch (RetryException e) {
546             if (retriesRemaining-- > 0) {
547                 loginInfo = getLoginInfo(true);
548                 process(document, inputViewName);
549                 resetRetry();
550             } else {
551                 Throwable cause = e.getCause();
552                 handleException(cause == null ? e : cause, document);
553             }
554         } catch (SuggestViewAbortException e) {
555             // for a long query result, the input stream should be flushed out if it is in preview
556             if (iterator instanceof RestResponseIterator) {
557                 ((RestResponseIterator) iterator).close();
558             }
559             throw e;
560         } catch (IOException ioe) {
561             String message = ioe.getMessage();
562             if (containsIgnoreCase(message, CONNECTION_RESET) &&
563                     length(options.getServiceUrl()) > REQUEST_URI_LENGTH_LIMIT) {
564                 // If the http request URI in REST API is larger than approximately 32,350,
565                 // it was found that the http request execution throws IOException with
566                 // "Connection reset" error. Therefore, REQUEST_URI_LENGTH_LIMIT was chosen to be
567                 // 32,000 to safely catch this case and improve the reason message.
568                 message = getStatusText(SC_REQUEST_URI_TOO_LONG);
569             }
570             Exception exception = ioe;
571             if (contains(message, STATUS_CODE_431) || contains(message, STATUS_CODE_414) ||
572                     getStatusText(SC_REQUEST_URI_TOO_LONG).equals(message)) {
573                 // From the post here:
574                 // https://salesforce.stackexchange.com/questions/195449/what-is-the-
575                 // longest-uri-that-salesforce-will-accept-through-the-rest-api/195450
576                 // An 431 or 414 status code would be returned if the URI is to large in
577                 // REST API
578                 exception = new SnapDataException(ioe, ERR_REQUEST_URI_TOO_LONG)
579                         .withReason(message)
580                         .withResolution(RESOLUTION_REQUEST_URI_TOO_LONG);
581             }
582             handleException(exception, document);
583         } catch (Exception e) {
584             handleException(e, document);
585         } finally {
586             if (iterator instanceof BulkResponseIterator) {
587                 ((BulkResponseIterator) iterator).close();
588             }
589         }
590     }

652     private void handleException(Throwable e, Document document) {
653         String error;
654         SnapDataException snapDataException;
655         if (e instanceof SnapDataException) {
656             errorViews.write((SnapDataException) e, document);
657         } else if (e instanceof SnapException) {
658             SnapException se = (SnapException) e;
659             Throwable cause = Throwables.getRootCause(e);
660             snapDataException = new SnapDataException(cause, cause.getMessage())
661                     .withReason(se.getReason())
662                     .withResolution(se.getResolution());
663             errorViews.write(snapDataException, document);
664         } else {
665             switch (snapName) {
666                 case READ:
667                     snapDataException = new SnapDataException(e,
668                             String.format(ERR_READ, objectType))
669                             .withResolution(checkForPollTimeout(e) ? RESOLUTION_POLL_TIMEOUT :
670                                     ERR_RESOLUTION);
671                     break;
672                 case QUERY:
673                     error = String.format(ERR_QUERY, options.getQuery());
674                     snapDataException = new SnapDataException(e, error)
675                             .withResolution(checkForPollTimeout(e) ? RESOLUTION_POLL_TIMEOUT :
676                                     ERR_RESOLUTION);
677                     break;
678                 case SEARCH:
679                     error = String.format(ERR_SEARCH, options.getQuery());
680                     snapDataException = new SnapDataException(e, error)
681                             .withResolution(ERR_RESOLUTION);
682                     break;
683                 default:
684                     snapDataException = new SnapDataException(e, ERR_UNSUPPORTED_SNAP)
685                             .formatWith(snapName.name())
686                             .withResolution(ERR_RESOLUTION);
687             }
688             Throwable t = Throwables.getRootCause(e);
689             snapDataException.withReason(t.getMessage());
690             errorViews.write(snapDataException, document);
691         }
692     }

895     // If there are related-object fields in the SELECT query, prepare metadata here so that
896     // each output document can be modified during the validation. The purpose is to make the
897     // output document schema the same in both validation and execution in PK-Chunking.
898     private List<Map<String, String>> checkRelatedObjectFields() {
899         String query = options.getQuery();
900         if (StringUtils.startsWithIgnoreCase(query, SELECT)) {
901             int index = StringUtils.indexOfIgnoreCase(query, FROM);
902             if (index > 0) {
903                 List<Map<String, String>> relatedObjectData = new ArrayList<>();
904                 String fieldsPart = query.substring(SELECT.length(), index);
905                 String[] fields = fieldsPart.split(COMMA);
906                 for (String field : fields) {
907                     if (field.contains(DOT)) {
908                         String[] parts = field.split(DOT_REGEX);
909                         if (parts.length == 2) {
910                             Map<String, String> relatedFieldData = new HashMap<>();
911                             relatedFieldData.put(INSERT_KEY, field.trim());
912                             relatedFieldData.put(REMOVE_KEY_1, parts[0].trim());
913                             relatedFieldData.put(REMOVE_KEY_2, parts[1].trim());
914                             relatedObjectData.add(relatedFieldData);
915                         }
916                     }
917                 }
918                 if (!relatedObjectData.isEmpty()) {
919                     return relatedObjectData;
920                 }
921             }
922         }
923         return null;
924     }

927     // in PK-Chunking

929         @SuppressWarnings("unchecked")

939                         if (StringUtils.isNotBlank(removeKey2)) {
940                             @SuppressWarnings("unchecked")
941                             Object value = ((Map<String, Object>) valueObject).get(removeKey2);
942                             map.put(insertKey, value);
943                         }
944                     } else {
945                         map.put(insertKey, null);
946                     }
947                 }
948             }
949         }

996     @SuppressWarnings("unchecked")
997     private Map<String, Object> joinParts(List<Iterator<Object>> iteratorList) {
998         List<Map<String, Object>> parts = new ArrayList<>(iteratorList.size());
999         for (Iterator<Object> iterator : iteratorList) {
1000             Object value = iterator.next();
1001             if (value instanceof Map) {
1002                 parts.add((Map<String, Object>) value);
1003             } else {
1004                 throw new SnapDataException(ERR_UNEXPECTED_TYPE)
1005                         .withReason(String.format(REASON_MAP_EXPECTED,
1006                                 value.getClass().getSimpleName()))
1007                         .withResolutionAsDefect();
1008             }
1009         }
1010         Map<String, Object> joinedMap = parts.get(0);
1011         if (allEqualId(parts)) {
1012             for (int i = 1; i < parts.size(); i++) {
1013                 Map<String, Object> part = parts.get(i);
1014                 part.remove(ID);
1015                 joinedMap.putAll(part);
1016             }
1017         } else {
1018             throw new SnapDataException(ERR_UNEXPECTED_ORDER)
1019                     .withReason(REASON_SAME_ID)
1020                     .withResolutionAsDefect();
1021         }
1022         return (Map<String, Object>) SalesforceSchemaUtil.convertDateTime(joinedMap,
1023                 dateTimeFields);
1024     }
