443     private void verifyJobCompleteStatusAndProcessQueryResults(final Document document) {
444         String offset = bulkJobInfo.getLocator();
445         bulkJobInfo = bulk2Utils.getQueryJobStatus(
446                 createFullUrl(FORMAT_CREATE_OR_CLOSE_JOB_URL) + bulkJobInfo.getId());
447         if (StringUtils.equalsIgnoreCase(bulkJobInfo.getState(), JOB_COMPLETE)) {
448             if (maxRecords > MAX_RECORDS_DEFAULT) {
449                 multiThreadDownload(offset);
450             } else {
451                 do {
452                     List<Map<String, String>>
453                             resultsList = bulk2Utils.getQueryResults(constructQueryUrl(offset,
454                             bulkJobInfo.getContentUrl(), maxRecords));
455                     writeResultsToView(resultsList, document);
456                     offset = bulkJobInfo.getLocator();
457                 } while (Objects.nonNull(offset));
458             }
459         } else {
460             String reason = "";

462                 reason = bulkJobInfo.getSfapiError().getResponse();
463             }
464             SnapDataException ex = new SnapDataException(ERROR_MSG)
465                     .withReason(reason)
466                     .withResolution(RESOLUTION_ADDRESS_ISSUE);
467             errorViews.write(ex, document);
468         }

471     private void multiThreadDownload(String locator) {
472         if (executor == null) {
473             // initialize thread pool
474             executor = (ThreadPoolExecutor) Executors.newFixedThreadPool(maxThreads, threadFactory);
475             tasks = new ArrayList<>();
476         }
477         do {
478             // send a request to retrieve a set of records
479             String url = constructQueryUrl(locator, bulkJobInfo.getContentUrl(), maxRecords);
480             HttpEntity entity = bulk2Utils.getNextQueryResults(url);
481             if (entity == null) {
482                 throw new SnapDataException(ERR_DOWNLOAD_QUERY_RESULT)
483                         .withReason(REASON_ENTITY_NULL)
484                         .withResolutionAsDefect();
485             }
486             // invoke a new thread to download an httpEntity
487             tasks.add(CompletableFuture.supplyAsync(new Download(entity, tempDir), executor));
488             // check tasks for any completed task and limit the size of task queue
489             checkDownloadsWithLimit(maxThreads * 2);
490             // get the next locator
491             locator = bulkJobInfo.getLocator();
492         } while (Objects.nonNull(locator));
493         // process all remaining tasks and results
494         checkDownloadsWithLimit(0);
495     }
496 
497     private void checkDownloadsWithLimit(int maxThreadSize) {
498         // limit the size of the task queue
499         while (tasks.size() > maxThreadSize) {
500             // process the result of the completed task if any, and remove them from the task list
501             tasks.removeIf(this::processResult);
502             sleepNoInterrupt(1, TimeUnit.MILLISECONDS, LOG);
503         }
504     }

506     private boolean processResult(CompletableFuture<CSVParser> task) {
507         if (task.isDone()) {
508             CSVParser csvParser = task.join();
509             writeResultsToView(csvParser, document);
510             return true;
511         }
512         return false;
513     }

590         private InputStream downloadToTempFile(HttpEntity httpEntity) throws IOException {
591             // create a temp file
592             File tempFile = SalesforceUtils.getTempFile(tempDir, ERR_TEMP_FILE);
593             // Stage data to a temporary file, compressed using gzip.
594             // If the data is already compressed, write the raw data to the file.
595             // If the data is not compressed, compress it with gzip when writing to the file.
596             SalesforceUtils.downloadToTempFile(new ResponseBody(httpEntity), tempFile);
597             return new GZIPInputStream(new FileInputStream(tempFile), BUFFER_SIZE);
598         }
