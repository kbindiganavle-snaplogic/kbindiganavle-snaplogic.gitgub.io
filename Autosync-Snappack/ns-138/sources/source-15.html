280     protected void initCustomProperties(PropertyValues propertyValues) {
281         int pollingIntervalSec = ((BigInteger) propertyValues.get(POLLING_INTERVAL_PROP))
282                 .intValue();
283         int pollingTimeoutSec = ((BigInteger) propertyValues.get(POLLING_TIMEOUT_PROP)).intValue();
284         options.withOperation(Operation.QUERY)
285                 .withPollingIntervalSec(pollingIntervalSec)
286                 .withPollingTimeoutSec(pollingTimeoutSec);
287         // only attempt pk chunking if using the bulk API (default to the previous behavior where
288         // if the batch size >= 100,000 it's on, otherwise off)
289         attemptPKChunking = bulkAPI && propertyValues.getBoolean(USE_PK_CHUNKING_PROP,
290                 batchSize >= MIN_CHUNK_SIZE);
291         // get the WHERE clause expression property
292         whereClauseExpr = propertyValues.getAsExpression(WHERE_CLAUSE_PROP);
293         orderbyClauseExpr = propertyValues.getAsExpression(ORDERBY_CLAUSE_PROP);
294         limitClauseExpr = propertyValues.getAsExpression(LIMIT_CLAUSE_PROP);
295         columnLimitExpr = propertyValues.getAsExpression(COLUMN_LIMIT_PROP);
296         columnOffsetExpr = propertyValues.getAsExpression(COLUMN_OFFSET_PROP);
297 
298         // get a list of the output field expression properties
299         List<Map<String, Object>> outputFieldsProp = propertyValues.get(OUTPUT_FIELDS_PROP);
300         if (CollectionUtils.isNotEmpty(outputFieldsProp)) {
301             if (!columnOffsetExpr.isEmpty() || !columnLimitExpr.isEmpty()) {
302                 throw new ConfigurationException(ERR_INVALID_PROPERTIES_COMBO)
303                         .withReason(REASON_OUTPUT_FIELDS_AND_OFFSET)
304                         .withResolution(RES_OUTPUT_FIELDS_AND_OFFSET);
305             }
306 
307             outputFieldExprList = Lists.newArrayListWithExpectedSize(outputFieldsProp.size());
308             for (Map<String, Object> outputFieldProp : outputFieldsProp) {
309                 ExpressionProperty outputFieldExpression = propertyValues.getExpressionPropertyFor(
310                         outputFieldProp, OUTPUT_FIELD_PROP);
311                 outputFieldExprList.add(outputFieldExpression);
312             }
313         }

316     @Override
317     protected void evalExpressionProperties(Document document) {
318         if (CollectionUtils.isNotEmpty(outputFieldExprList)) {
319             List<String> outputFields = new ArrayList<>(outputFieldExprList.size());
320             for (ExpressionProperty outputFieldExpr : outputFieldExprList) {
321                 String outputField = outputFieldExpr.eval(document);
322                 if (StringUtils.isNotBlank(outputField)) {
323                     outputFields.add(outputField);
324                 }
325             }
326             if (CollectionUtils.isNotEmpty(outputFields)) {
327                 options.withOutputFields(Joiner.on(COMMA).join(outputFields));
328             }
329         }
330         if (whereClauseExpr != null) {
331             options.withWhereClause(whereClauseExpr.eval(document));
332         }
333         if (orderbyClauseExpr != null) {
334             options.withOrderbyClause(orderbyClauseExpr.eval(document));
335         }
336         if (limitClauseExpr != null) {
337             options.withLimitClause(limitClauseExpr.eval(document));
338         }
339         Object offsetObj = columnOffsetExpr.eval(document);
340         columnOffset = getIntValue(offsetObj);
341         if (columnOffset.intValue() < 0) {
342             throw new ExecutionException(PROPERTY_VALUE_ERROR)
343                     .withReason(String.format(
344                             THE_PROVIDED_VALUE_S_IS_NOT_A_POSITIVE_VALUE, columnOffset))
345                     .withResolution(PLEASE_RETRY_WITH_A_POSITIVE_INTEGER_VALUE);
346         }
347         columnLimit = getIntValue(columnLimitExpr.eval(document));
348         if (columnLimit.intValue() < 0) {
349             throw new ExecutionException(PROPERTY_VALUE_ERROR)
350                     .withReason(String.format(
351                             THE_PROVIDED_VALUE_S_IS_NOT_A_POSITIVE_VALUE, columnLimit))
352                     .withResolution(PLEASE_RETRY_WITH_A_POSITIVE_INTEGER_VALUE);
353         } else if (columnLimit.intValue() == 0) {
354             //0 represents no limit set. therefore set the limit to the max possible value.
355             columnLimit = BigInteger.valueOf(Integer.MAX_VALUE);
356         }
357         if (!attemptPKChunking) {
358             if (batchSize > MAX_BATCH_SIZE) {
359                 if (StringUtils.isBlank(options.getLimitClause()) &&
360                         StringUtils.isBlank(options.getOrderbyClause()) &&
361                         (PK_CHUNKING_TYPES.stream().anyMatch(v -> v.equalsIgnoreCase(objectType))
362                 || objectType.toLowerCase().endsWith("__c"))) {
363                     PK_CHUNKING_SUGGESTED.report();
364                 } else{
365                     BATCH_SIZE_LIMITED.report();
366                 }
367             }
368             pkChunking = false;
369         }
370     }

372     private BigInteger getIntValue(Object value) {
373         if (value != null) {
374             switch (value.getClass().getSimpleName()) {
375                 case BIG_INTEGER:
376                     return ((BigInteger) value);
377                 case STRING:
378                     String str = ((String) value).trim();
379                     if (StringUtils.isNotBlank(str)) {
380                         try {
381                             return new BigInteger(str);
382                         } catch (NumberFormatException e) {
383                             throw new ExecutionException(e, e.getMessage())
384                                     .withReason(String.format(
385                                             THE_PROVIDED_VALUE_S_IS_NOT_AN_INTEGER_VALUE, str))
386                                     .withResolution(PLEASE_RETRY_WITH_AN_INTEGER_VALUE);
387                         }
388                     }
389                     break;
390                 case LONG:
391                     return BigInteger.valueOf((Long) value);
392                 case INTEGER:
393                     return new BigInteger(value.toString());
394             }
395         }
396         return BigInteger.ZERO;
397     }
