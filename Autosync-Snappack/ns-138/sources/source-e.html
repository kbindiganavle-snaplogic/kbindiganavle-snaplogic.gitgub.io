222         for (Map<String, Object> outputFieldProp : outputFieldsProp) {

285     @SuppressWarnings("unchecked")
286     protected void process(Document document, String inputViewName) {
287         if (document == null) {
288             return;
289         }
290         Object data = document.get();
291         if (data == null) {
292             return;
293         }
294         if (!(data instanceof Map)) {
295             String error = String.format(ERR_MAP_DATA, data.getClass().getSimpleName());
296             hasExecErr = true;
297             SnapDataException snapDataException = new SnapDataException(error)
298                     .withReason(error)
299                     .withResolution(RESOLUTION_MAP_DATA);
300             errorViews.write(snapDataException);
301             return;
302         }
303         Map<String, Object> map = (Map<String, Object>) data;
304         // format an and-clause with input map data
305         String andClause = formatAndClause(map, document);
306         // in single lookup request mode, process one lookup request
307         if (singleLookup) {
308             loginInfo = getLoginInfo(false); // make sure access token is not expired
309             if (loginException != null) {
310                 errorViews.write(loginException, document);
311                 return;
312             }
313             processSingleLookup(document, andClause, map);
314             return;
315         }
316         // if the SOQL statement approaches the maximum size limit, process the lookup operation
317         if (queryBuilder.length() + andClause.length() > MAX_SOQL) {
318             loginInfo = getLoginInfo(false); // make sure access token is not expired
319             if (loginException != null) {
320                 for (Document d : docsList) {
321                     errorViews.write(loginException, d);
322                 }
323                 return;
324             }
325             processLookup();
326             // re-initialize query string builder and input records storage
327             int index = queryBuilder.indexOf(WHERE) + WHERE.length();
328             queryBuilder.delete(index, queryBuilder.length());
329             inputRecords.clear();
330             resultFlags.clear();
331             docsList.clear();
332         }
333         docsList.add(document);
334         // append an and-clause to the query builder
335         queryBuilder.append(andClause).append(OR);
336         // store input map data so that it can be merged into output records later
337         if (mergeRecord) {
338             Object id = map.get(correlationId);
339             if (id instanceof String && StringUtils.isNotBlank((String) id)) {
340                 inputRecords.put((String) id, map);
341                 resultFlags.put((String) id, Boolean.FALSE);
342             } else {
343                 hasExecErr = true;
344                 throw new ExecutionException(ERR_CORRELATION_ID)
345                         .withReason(ERR_CORRELATION_ID)
346                         .withResolution(RESOLUTION_CORRELATION_ID);
347             }
348         }
349     }

369     private String formatAndClause(Map<String, Object> data, Document document) {
370         Object dataCopy = makeDeepCopy(document);
371         List<String> predicates = new ArrayList<>(data.size());
372         for (ExpressionProperty condition : conditions) {
373             String strCondition = condition.eval(document, dataCopy, new SingleQuoteHandler());
374             if (!StringUtils.isBlank(strCondition)) {
375                 predicates.add(strCondition);
376             }
377         }
378         //return an and-clause string: key1='value1' AND key2='value2' AND ...
379         String andClause = Joiner.on(AND).join(predicates);
380         return String.format(FORMAT_PARENTHESIS, andClause);
381     }

405     @SuppressWarnings("unchecked")
406     private void processSingleLookup(final Document document, String andClause, Map<String,
407             Object> map) {
408         // append an and-clause to the query builder
409         queryBuilder.append(andClause);
410         options.withQuery(queryBuilder.toString());
411         // if REST API, include SOQL statement in the service URL
412         if (!bulkAPI) {
413             String serviceUrl = getServiceEndpoint();
414             options.withServiceUrl(serviceUrl);
415         }
416         boolean result = false;
417         try (AbstractRequest<Iterator<Object>, Void> requester = bulkAPI ?
418                 getBulkRead() : getSyncRead()) {
419             // execute SOQL request
420             Iterator<Object> iterator = requester.execute();
421             while (iterator.hasNext()) {
422                 // get the output record
423                 Map<String, Object> output = (Map<String, Object>) iterator.next();
424                 output = (Map<String, Object>)
425                         SalesforceSchemaUtil.convertDateTime(output, dateTimeFields);
426                 if (matchDataType) {
427                     output = (Map<String, Object>) SalesforceSchemaUtil.matchDataType(output,
428                             nonStringFields);
429                 }
430                 // send output record to output view
431                 outputViews.write(documentUtility.newDocumentFor(document, output), map);
432                 result = true;
433             }
434         } catch (SuggestViewAbortException e) {
435             throw e;
436         } catch (SnapDataException e) {
437             errorViews.write(e);
438         } catch (Throwable e) {
439             SnapDataException ex = new SnapDataException(e, ERR_PROCESSING_QUERY)
440                     .withReason(e.getMessage())
441                     .withResolution(RESOLUTION_LOOKUP);
442             errorViews.write(ex, document);
443             return;
444         } finally {
445             // re-initialize query string builder and input records storage
446             int index = queryBuilder.indexOf(WHERE) + WHERE.length();
447             queryBuilder.delete(index, queryBuilder.length());
448         }
449         // if no result is received, send input data to error view
450         if (!result) {
451             if (passThrough) {
452                 outputViews.write(documentUtility.newDocumentFor(document, new LinkedHashMap<>()),
453                         map);
454             } else {
455                 writeErrorView(map);
456             }
457         }
458     }

460     @SuppressWarnings("unchecked")
461     private void processLookup() {
462         // format SOQL statement
463         int index = queryBuilder.lastIndexOf(OR);
464         options.withQuery(queryBuilder.substring(0, index));
465         // if REST API, include SOQL statement in the service URL
466         if (!bulkAPI) {
467             String serviceUrl = getServiceEndpoint();
468             options.withServiceUrl(serviceUrl);
469         }
470         try (AbstractRequest<Iterator<Object>, Void> requester = bulkAPI ?
471                 getBulkRead() : getSyncRead()) {
472             Document inputDoc = getLastFrom(docsList);
473             // execute SOQL request
474             Iterator<Object> iterator = requester.execute();
475             while (iterator.hasNext()) {
476                 // get the output record
477                 Map<String, Object> output = (Map<String, Object>) iterator.next();
478                 output = (Map<String, Object>)
479                         SalesforceSchemaUtil.convertDateTime(output, dateTimeFields);
480                 if (matchDataType) {
481                     output = (Map<String, Object>) SalesforceSchemaUtil.matchDataType(output,
482                             nonStringFields);
483                 }
484                 // send output record to output view
485                 if (mergeRecord) {
486                     // pass through the original data to the output
487                     String id = (String) output.get(correlationId);
488                     Map<String, Object> inputMap = inputRecords.get(id);
489                     if (inputMap != null) {
490                         outputViews.write(documentUtility.newDocumentFor(inputDoc, output),
491                                 inputMap);
492                         resultFlags.put(id, Boolean.TRUE);
493                     }
494                 } else {
495                     outputViews.write(documentUtility.newDocumentFor(inputDoc, output));
496                 }
497             }
498         } catch (SuggestViewAbortException e) {
499             throw e;
500         } catch (ExecutionException e) {
501             SnapDataException ex = new SnapDataException(e, e.getMessage())
502                     .withReason(e.getReason())
503                     .withResolution(e.getResolution());
504             errorViews.write(ex);
505         } catch (SnapDataException e) {
506             errorViews.write(e);
507         } catch (Throwable e) {
508             SnapDataException ex = new SnapDataException(e, ERR_PROCESSING_QUERY)
509                     .withReason(e.getMessage())
510                     .withResolution(RESOLUTION_LOOKUP);
511             errorViews.write(ex);
512             return;
513         }
514         // send input data to output view or error view if it did not produce any result
515         for (Map.Entry<String, Boolean> entry : resultFlags.entrySet()) {
516             if (!entry.getValue()) {
517                 Map<String, Object> original = inputRecords.get(entry.getKey());
518                 if (passThrough) {
519                     outputViews.write(documentUtility.newDocument(new LinkedHashMap<>()), original);
520                 } else {
521                     writeErrorView(original);
522                 }
523             }
524         }
525     }
