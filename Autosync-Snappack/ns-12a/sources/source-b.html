704     /**
705      * This method converts the OpenAir Date object into appropriate Joda DateTime object.
706      *
707      * @param openAirDate
708      * @param retainNullDateValues tells if date objects with zero values to be treated as null
709      * @return Joda DateTime object
710      */
711     private Object getDateTimeObject(Map<String, Object> openAirDate, boolean retainNullDateValues) {
712         boolean isTimeNotAvailable = false;
713         boolean isDateNotAvailable = false;
714         boolean isTimeZoneNotAvailable = false;
715         String hour, minute, second, timezone;
716         int year, day, month;
717 
718         // read date and time values
719         //// if hour, minute, second and timezone are not available then they are null
720         hour = (String) (openAirDate.get(HOUR_KEY));
721         minute = (String) (openAirDate.get(MINUTE_KEY));
722         second = (String) (openAirDate.get(SECOND_KEY));
723         timezone = (String) (openAirDate.get(TIMEZONE_KEY));
724         //// if year, month and day are not available then they are "00"
725         Object yearObj = openAirDate.get(YEAR_KEY);
726         Object monthObj = openAirDate.get(MONTH_KEY);
727         Object dayObj = openAirDate.get(DAY_KEY);
728         year = yearObj != null ? Integer.parseInt(yearObj.toString()) : 0;
729         month = monthObj != null ? Integer.parseInt(monthObj.toString()) : 0;
730         day = dayObj != null ? Integer.parseInt(dayObj.toString()) : 0;
731 
732         // perform checks for non-availability of date, time, and timezone
733         if (hour == null || minute == null || second == null) {
734             isTimeNotAvailable = true;
735         }
736         if (timezone == null) {
737             isTimeZoneNotAvailable = true;
738         }
739         if (year == 0 || month == 0 || day == 0) {
740             isDateNotAvailable = true;
741         }
742 
743         // based on above checks, create respective DateTime object
744         if (isDateNotAvailable) {
745             // Date is not present
746             if (isTimeNotAvailable) {
747                 // Time is not present -> create LocalDateTime object
748                 if (retainNullDateValues) {
749                     return null;
750                 } else {
751                     return new LocalDateTime();
752                 }
753             } else {
754                 // Time is present -> create LocalTime object
755                 return new LocalTime(Integer.parseInt(hour),
756                         Integer.parseInt(minute),
757                         Integer.parseInt(second));
758             }
759         } else {
760             // Date is present
761             if (isTimeNotAvailable) {
762                 // Time is not present -> create LocalDate object
763                 return new LocalDate(year, month, day);
764             } else {
765                 // Time is present
766                 if (isTimeZoneNotAvailable) {
767                     // TimeZone is not present -> create DateTime object without TimeZone
768                     return new DateTime(year, month, day, Integer.parseInt(hour),
769                             Integer.parseInt(minute),
770                             Integer.parseInt(second));
771                 } else {
772                     // TimeZone is present -> create DateTime object with TimeZone
773                     return new DateTime(year, month, day, Integer.parseInt(hour),
774                             Integer.parseInt(minute),
775                             Integer.parseInt(second), DateTimeZone.forID(timezone));
776                 }
777             }
778         }
779     }
