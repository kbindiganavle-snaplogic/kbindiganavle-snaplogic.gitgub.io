58 /**
59  * The OpenAirDelete snap is used to delete the open air data for given entities.
60  *
61  * @author sprasad, smudassir
62  */
63 @General(title = OPEN_AIR_DELETE_TITLE, purpose = OPEN_AIR_DELETE_PURPOSE)
64 @Inputs(min = 0, max = 1, accepts = { ViewType.DOCUMENT })
65 @Outputs(min = 0, max = 1, offers = { ViewType.DOCUMENT })
66 @Version(snap = 1)
67 @Category(snap = SnapCategory.WRITE)
68 @Accounts(provides = OpenAirAccount.class)
69 public class OpenAirDelete extends SimpleSnap implements InputSchemaProvider,
70         LifecycleCallback, DependencyManager {
71     private String openAirDeleteType;
72     private StringBuffer writerDeleteRequest;
73     private int count;
74     private ExpressionProperty recordIdExpression;
75     @Inject
76     private Utilities utilities;
77     @Inject
78     private OpenAirAccount openAirAccount;
79     @Inject
80     private DocumentUtility documentUtility;
81     private UserCredentials userCredentials;
82     private static final String CMP_DELETE_CONTENT_TYPE_PROP = "OpenAirDeleteTypeProperty";
83     private static final String CMP_DELETE_RECORD_ID_PROP = "RecordIdProperty";
84 
85     @Override
86     public void defineInputSchema(SchemaProvider provider) {
87         for (String inputViewName : provider.getRegisteredInputViewNames()) {
88             Schema deleteRecordIdSchema = provider.createSchema(
89                     SnapType.STRING, CMP_DELETE_RECORD_ID_PROP);

91                     .withChildSchema(deleteRecordIdSchema)
92                     .build();
93         }
94     }
95 
96     @Override
97     public Module getManagedModule() {
98         return new AbstractModule() {
99             @Override
100             protected void configure() {
101                 bind(XmlUtils.class).to(XmlUtilsImpl.class);
102             }
103         };
104     }
105 
106     @Override
107     public void cleanup() throws ExecutionException {
108         // NO OP
109     }

111     @Override
112     public void configure(PropertyValues propertyValues)
113             throws ConfigurationException {
114         userCredentials = openAirAccount.connect();
115         openAirDeleteType = propertyValues.get(CMP_DELETE_CONTENT_TYPE_PROP);
116         recordIdExpression = propertyValues
117                 .getAsExpression(CMP_DELETE_RECORD_ID_PROP);
118         writerDeleteRequest = new StringBuffer();
119     }
120

122     public void defineProperties(PropertyBuilder propertyBuilder) {
123         propertyBuilder
124                 .describe(CMP_DELETE_CONTENT_TYPE_PROP, CMP_DELETE_CONTENT_TYPE_LABEL,
125                         CMP_DELETE_CONTENT_TYPE_DESC)
126                 .withSuggestions(new Suggestions() {
127                     @Override
128                     public void suggest(
129                             final SuggestionBuilder suggestionBuilder,
130                             final PropertyValues propertyValues) {
131                         suggestionBuilder.node(CMP_DELETE_CONTENT_TYPE_PROP)
132                                 .suggestions(OPENAIR_DELETE_TYPES.toArray(new String[0]));
133                     }
134                 })
135                 .required()
136                 .add();
137 
138         propertyBuilder
139                 .describe(CMP_DELETE_RECORD_ID_PROP, CMP_DELETE_RECORD_ID_PROP_LABEL,
140                         CMP_DELETE_RECORD_ID_PROP_DESC)
141                 .required()
142                 .schemaAware(SnapProperty.DecoratorType.ACCEPTS_SCHEMA)
143                 .expression()
144                 .add();
145     }
146 
147     @Override
148     protected void process(Document document, String inputViewName) {
149         String recordIdValue = utilities.castAsString(recordIdExpression.eval(document));
150         if (StringUtils.isNotBlank(recordIdValue)) {
151             writerDeleteRequest.append(String.format(DELETE_REQUEST_IDS,
152                     openAirDeleteType, openAirDeleteType, recordIdValue,
153                     openAirDeleteType));
154             count++;
155         }
156         if (count == OPENAIR_READ_LIMIT) {
157             utilities.executeOpenAirRequest(
158                     writerDeleteRequest.toString(), userCredentials,
159                     documentUtility, outputViews, errorViews, DELETE_TYPE);
160             count = 0;
161             writerDeleteRequest = new StringBuffer();
162         }
163     }
164 
165     @Override
166     public void handle(LifecycleEvent event) {
167         switch (event) {
168             case SUCCESS:
169                 if (StringUtils.isNotBlank(writerDeleteRequest)) {
170                     utilities.executeOpenAirRequest(
171                             writerDeleteRequest.toString(), userCredentials,
172                             documentUtility, outputViews, errorViews, DELETE_TYPE);
173                 }
174                 break;
175         }
176     }
