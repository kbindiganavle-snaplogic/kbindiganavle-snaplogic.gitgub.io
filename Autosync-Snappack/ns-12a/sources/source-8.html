105                 }

107                     filterValues.setSearchFieldValueAvailable(((Map) row).keySet().contains(key));
108                 }
109                 if (key.equals(CMP_READ_CUSTOM_READ_TYPE_PROP)) {
110                     filterValues.setCustomReadTypeAvailable(((Map) row).keySet().contains(key));
111                 }
112                 Object field = ((Map) row).get(key);
113                 if (field != null) {
114                     fieldString = field.toString();
115                 }
116             }
117         }
118         return fieldString;
