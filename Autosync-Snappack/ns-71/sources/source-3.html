378                             props.put(ALLOWEDIT_ON_CREATE, a.getExpression()

380                         } else if (name.contains(REQUIRED)) {
381                             props.put(REQUIRED, a.getExpression()
382                                     .asConstant().getValue());
383                         } else if (name.contains(ALLOW_EDIT)) {
384                             props.put(ALLOW_EDIT, a.getExpression()
385                                     .asConstant().getValue());
386                         }
387                     } else {
388                         props.put(term.getName(), term.getDefaultValue());

441         Throwable t = Throwables.getRootCause(ex);

444         if (reason != null && (reason.contains("404") || reason.contains("400"))) {
445             resolution = ERR_EXECUTING_REQUEST_400_RESOLUTION;
446         }
447         if (reason != null && (reason.contains("401"))) {
448             resolution = ERR_EXECUTING_REQUEST_401_RESOLUTION;
449         }
450         if (ex instanceof ODataClientErrorException) {
451             ODataError error = ((ODataClientErrorException) ex).getODataError();
452             if (error != null && error.getInnerError() != null) {
453                 reason = error.getInnerError().get("message");
454             }
455         }
456         return new SnapDataException(t, ERR_EXECUTING_REQUEST)
457                 .withReason(reason)
458                 .withResolution(resolution);

518         AccountInfo accountInfo = account.connect();
519         ODataRawRequest request = oDataClient.getRetrieveRequestFactory().getRawRequest(uri);
520         request = (ODataRawRequest) prepareRequest(request, accountInfo, headers);
521         if (pageSize > 0) {

523                     String.format(ODATA_MAX_PAGE_SIZE, pageSize));
524         }
525         ODataRawResponse response = request.execute();
526         int statusCode = response.getStatusCode();
527         if (response.getContentType() != null &&

530                     .withReason(HttpStatus.SC_UNAUTHORIZED + " : " + ERROR_ACCOUNT_EXPIRED)

532         } else if (statusCode >= HttpStatus.SC_OK && statusCode <= HttpStatus.SC_ACCEPTED) {
533             Map<String, Object> map = readRawResponse(response);
534             return map;
535         } else {
536             throw new SnapDataException(ERR_EXECUTING_REQUEST_MSG)
537                     .formatWith(response.getStatusCode())
538                     .withReason(statusCode + " : " + response.getStatusMessage())
539                     .withResolution(ERR_ADDRESS_ISSUE_RESOLUTION);

543     protected boolean handleThrottleError(ODataClientErrorException exception) {
544         LOGGER.warn("Received ApplicationThrottled Error, Error code: 429");
545         int waitTime = 0;
546         try {
547             Header[] list = exception.getHeaderInfo();
548             for (Header header : list) {
549                 if ("Retry-After".equalsIgnoreCase(header.getName())) {
550                     waitTime = Integer.parseInt(header.getValue());
551                 }
552             }
553             waitTime = waitTime < 0 ? 0 : waitTime;
554         } catch (NumberFormatException nfe) {

595     public boolean retryChecker(Throwable throwable) {
596         String msg = throwable.getMessage();
597         if (throwable instanceof ODataClientErrorException) {
598             int statusCode = ((ODataClientErrorException) throwable).getStatusLine()
599                     .getStatusCode();
600             if (HttpStatus.SC_UNAUTHORIZED == statusCode
601                     || HttpStatus.SC_FORBIDDEN == statusCode) {
602                 return account.reloadOnAuthError();
603             }
604             if (statusCode == 429) {
605                 return handleThrottleError(((ODataClientErrorException) throwable));
606             }
607             return false;
608         } else if (throwable instanceof SnapDataException) {
609             msg = ((SnapDataException) throwable).getReason();
610             if (checkForAuthCode(msg)) {
611                 return account.reloadOnAuthError();
612             }
613             return false;
614         } else if (throwable instanceof ODataServerErrorException && checkForAuthCode(msg)) {
615             account.reloadOnAuthError();
616         }
617         return true;
618     }

659     public ODataInvokeResponse<ClientProperty> invokeRequest(AccountInfo accountInfo,
660                                     URIBuilder uri, Map<String, Object> parameters) {
661         Map<String, ClientValue> requestParams = new LinkedHashMap<>();
662         if (parameters != null && parameters.size() > 0) {
663             for (Map.Entry<String, Object> param : parameters.entrySet()) {
664                 ClientValue value = null;
665                 if (param.getValue() == null) {
666                     value = oDataClient.getObjectFactory()
667                             .newPrimitiveValueBuilder().buildString(null);
668                 } else if (param.getValue() instanceof Boolean) {
669                     value = oDataClient.getObjectFactory()
670                             .newPrimitiveValueBuilder().buildBoolean((boolean) param.getValue());
671                 } else {
672                     // by default, considering the value is of type String
673                     value = oDataClient.getObjectFactory()
674                             .newPrimitiveValueBuilder().buildString(
675                                     String.valueOf(param.getValue()));
676                 }
677                 requestParams.put(param.getKey(), value);
678             }
679         }
680         ODataInvokeRequest<ClientProperty> request = oDataClient.getInvokeRequestFactory()
681                 .getActionInvokeRequest(uri.build(), ClientProperty.class, requestParams);
682         request.addCustomHeader(AUTHORIZATION, accountInfo.getBearerToken());
683         return request.execute();
684     }
