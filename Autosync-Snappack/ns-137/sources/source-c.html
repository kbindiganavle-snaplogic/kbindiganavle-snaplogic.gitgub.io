116     @Override
117     protected void processBinary(@Nullable Document header, String inputViewName,
118             ReadableByteChannel channel) {
119         super.processBinary(header, inputViewName, channel);
120         s3CommonConfig.process(header);
121         s3UploadConfig.process(header);
122         if (isS3Express()) {
123             if (s3UploadV2 == null) {
124                 s3UploadV2 = new S3UploadV2(s3ClientConfig, s3CommonConfig, s3UploadConfig,
125                         isUltraPipeline, outputViews, errorViews, documentUtility);
126             }
127             if (s3ServiceV2 != null) {
128                 s3UploadV2.setS3ServiceV2(s3ServiceV2);
129             }
130             s3UploadV2.process(header, channel);
131             return;
132         }
133         if (s3Service == null) {
134             s3Service = S3Service.builder()
135                     .s3ClientConfig(s3ClientConfig)
136                     .threadPoolSize(s3CommonConfig.getThreadPoolSize())
137                     .multipartSizeInMB(s3UploadConfig.getMultipartSizeInMB())
138                     .build();
139             amazonS3 = s3Service.getAmazonS3();
140         }
141         try {
142             // InputStream should not be closed here. It is closed when S3Service completes the
143             // upload of the S3 object requested.
144             InputStream inputStream = Channels.newInputStream(channel);
145             Upload.UploadRequest uploadRequest = new Upload.UploadRequest(
146                     s3CommonConfig.getBucket(), s3CommonConfig.getObjectKey())
147                     .withInputStream(inputStream)
148                     .withHeaderMap(header.get(Map.class))
149                     .withAcl(s3UploadConfig.getAcl())
150                     .withCannedACL(s3UploadConfig.getCannedACL())
151                     .withWriteEmptyFile(s3UploadConfig.isWriteEmptyFile())
152                     .withUserDefinedMetadataMap(s3UploadConfig.getUserDefinedMetadata())
153                     .withObjectTagMap(s3UploadConfig.getObjectTags());
154             CompletableFuture<Upload> task = s3Service.submitUpload(uploadRequest);
155             if (isUltraPipeline) {
156                 processResult(task, header);
157             } else {
158                 uploads.add(Pair.of(task, header));
159                 checkUploads();
160             }
161         } catch (SuggestViewAbortException e) {
162             throw e;
163         } catch (Exception e) {
164             SnapDataException ex = new SnapDataException(e, ERR_UPLOAD_S3)
165                     .withReason(e.getMessage())
166                     .withResolution(RESOLUTION_ADDRESS_ISSUE);
167             errorViews.write(ex, header);
168         }
169     }

204     private boolean checkUploads() {
205         List<Pair<CompletableFuture<Upload>, Document>> doneList = uploads.stream()
206                 .filter(p -> p.getLeft().isDone())
207                 .collect(Collectors.toList());
208         doneList.forEach(p -> processResult(p.getLeft(), p.getRight()));
209         return uploads.removeAll(doneList);
210     }

212     private void processResult(CompletableFuture<Upload> task, Document document) {
213         Upload upload;
214         try {
215             upload = task.join();
216             Map<String, Object> data = new LinkedHashMap() {{
217                 put(BUCKET_PROP, s3CommonConfig.getOriginalBucket());
218                 put(OBJECT_KEY_PROP, upload.getObjectKey());
219             }};
220             if (s3CommonConfig.isPassThough()) {
221                 outputViews.write(documentUtility.newDocument(data), document);
222             } else {
223                 outputViews.write(documentUtility.newDocumentFor(document, data));
224             }
225         } catch (CompletionException e) {
226             errorViews.write((SnapDataException) new SnapDataException(e, e.getMessage())
227                     .withReason(ExceptionUtils.getRootCauseMessage(e))
228                     .withResolution(RESOLUTION_ADDRESS_ISSUE), document);
229         }

461         /**
462          * Creates a set of grantee/permission pairs.
463          *
464          * @param aclProperties - access control list properties
465          * @param header        - input header document
466          * @return a set of grantee/permission pairs
467          */
468         private Set<Grant> getGrantList(List<Map<String, Map<String, Object>>> aclProperties,
469                 Document header) {
470             Set<Grant> grants = new HashSet<>();
471             for (Map<String, Map<String, Object>> property : aclProperties) {
472                 // get Grantee object
473                 ExpressionProperty granteeExpr =
474                         propertyValues.getExpressionPropertyFor(property, GRANTEE_PROP);
475                 String granteeStr = granteeExpr.eval(header);
476                 if (StringUtils.isBlank(granteeStr)) {
477                     throw new ConfigurationException(ERR_INVALID_VALUE)
478                             .formatWith(granteeStr, GRANTEE_LABEL)
479                             .withReason(REASON_BLANK_PROPERTY)
480                             .withResolution(RESOLUTION_ADDRESS_ISSUE);
481                 }
482                 Grantee grantee;
483                 if (EVERYONE.equalsIgnoreCase(granteeStr)) {
484                     grantee = GroupGrantee.AllUsers;
485                 } else if (StringUtils.containsIgnoreCase(granteeStr, AUTHENTICATED)) {
486                     grantee = GroupGrantee.AuthenticatedUsers;
487                 } else {
488                     if (EmailValidator.getInstance().isValid(granteeStr)) {
489                         grantee = new EmailAddressGrantee(granteeStr);
490                     } else {
491                         grantee = new CanonicalGrantee(granteeStr);
492                     }
493                 }
494                 // add permissions to the list of grants
495                 if (Boolean.TRUE.equals(property.get(READ_PROP).get(VALUE))) {
496                     grants.add(new Grant(grantee, Permission.Read));
497                 }
498                 if (Boolean.TRUE.equals(property.get(VIEW_PERMISSIONS_PROP).get(VALUE))) {
499                     grants.add(new Grant(grantee, Permission.ReadAcp));
500                 }
501                 if (Boolean.TRUE.equals(property.get(FULL_CONTROL_PROP).get(VALUE))) {
502                     grants.add(new Grant(grantee, Permission.FullControl));
503                 }
504             }
505             return grants;
506         }
