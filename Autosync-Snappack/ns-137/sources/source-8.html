246     protected void writeBinaryOutput(final Download download, Document document) {
247         final long limit = maxSuggestValue == null ? Long.MAX_VALUE :
248                 maxSuggestValue.longValue();
249         Map<String, Object> headerMap = getHeaderMap(download);
250         long contentLength = Long.valueOf((String) headerMap.get(CONTENT_LENGTH.toLowerCase()));
251         final Document header = documentUtility.newDocumentFor(document, headerMap);
252         statusReporter.begin(contentLength, BYTES);
253         statusReporter.notify(DOWNLOADING, download.getBucket(),
254                 download.getObjectKey());
255         outputViews.write(new BinaryOutput() {
256             @Override
257             public Document getHeader() {
258                 return header;
259             }
260 
261             @Override
262             public void write(final WritableByteChannel writeChannel) throws IOException {
263                 try (InputStream inputStream = download.getInputStream()) {
264                     try (OutputStream outputStream = Channels.newOutputStream(writeChannel)) {
265                         statusReporter.notify(DOWNLOADING, download.getBucket(),
266                                 download.getObjectKey());
267                         // If a multipart file is being downloaded, this input stream is
268                         // initially an empty input stream with a first group of parts
269                         // are being downloaded and the following copy() method is blocked
270                         // until the entire file content is copied to the output stream.
271                         long copied = StatusReportingUtils.copy(statusReporter, inputStream,
272                                 outputStream, limit);
273                         if ((limit < 0 || copied < limit) &&
274                                 contentLength >= 0 && copied != contentLength) {
275                             // If the content was client-side-encrypted before read from the
276                             // inputStream, contentLength is -1. (see getHeaderMap() below)
277                             // If the actual length of data copied from input stream to output
278                             // stream is less than the content-length, one or more part
279                             // downloads may have failed.
280                             Throwable t = s3Service.getThrowableFor(
281                                     download.getDownloadId());
282                             throw t == null ? new IOException(String.format(ERR_RECEIVED_BYTES,
283                                     copied, contentLength)) : t;
284                         }
285                         LOG.debug("Copied from input stream to output stream for s3:///{}/{}",
286                                 download.getBucket(), download.getObjectKey());
287                     }
288                 } catch (SuggestViewAbortException e){
289                     throw e;
290                 } catch (Throwable e){
291                     errorViews.write((SnapDataException)
292                             new SnapDataException(e, ERR_DOWNLOAD_FILE)
293                                     .formatWith(download.getBucket(), download.getObjectKey())
294                                     .withReason(e.getMessage())
295                                     .withResolutionAsDefect(), document);
296                 } finally {
297                     IOUtils.closeQuietly(download);
298                     statusReporter.done();
299                 }
300             }
301         });
302     }

304     private boolean processDownload(Pair<CompletableFuture<Download>, Document> pair) {
305         CompletableFuture<Download> submitDownload = pair.getLeft();
306         if (submitDownload.isDone() || isUltraPipeline) {
307             try {
308                 Download download = submitDownload.join();
309                 writeBinaryOutput(download, pair.getRight());
310             } catch (CompletionException e) {
311                 Throwable cause = e.getCause();
312                 errorViews.write((SnapDataException) new SnapDataException(e, e.getMessage())
313                         .withReason(cause == null ? REASON_REFER_TO_STACKTRACE : cause.getMessage())
314                         .withResolution(RESOLUTION_ADDRESS_ISSUE), pair.getRight());
315             } catch (SuggestViewAbortException e) {
316                 throw e;
317             } catch (Exception e) {
318                 errorViews.write((SnapDataException)
319                         new SnapDataException(e, ERR_PROCESS_DOWNLOAD_RESULT)
320                                 .withResolution(RESOLUTION_ADDRESS_ISSUE), pair.getRight());
321             }
322             return true;
323         }
324         return false;
325     }

327     private Map<String, Object> getHeaderMap(Download download) {
328         String bucket = download.getBucket();
329         String objectKey = download.getObjectKey();
330         ObjectMetadata objectMetadata = download.getObjectMetadata();
331         Map<String, Object> headerMap = new LinkedHashMap<>();
332         if (objectMetadata != null) {
333             String filePath = String.format(S3_PATH_FORMAT, bucket, objectKey);
334             addHeader(headerMap, CONTENT_LOCATION, filePath);
335             if (StringUtils.isBlank(objectMetadata.getContentType())) {
336                 addHeader(headerMap, CONTENT_TYPE, guessContentTypeFromName(objectKey));
337             }
338             String unencrypted = objectMetadata.getUserMetaDataOf(UNENCRYPTED_CONTENT_LENGTH);
339             String cryptoKeymapAlgorithm =
340                     objectMetadata.getUserMetaDataOf(CRYPTO_KEYWRAP_ALGORITHM);
341             // if unencrypted content length is available, use it as content length
342             String contentLength = StringUtils.isNotBlank(unencrypted) ? unencrypted :
343                     // if client-side encrypted, but unencrypted content length is not available,
344                     // set content length negative to disable file size check
345                     String.valueOf(StringUtils.isBlank(cryptoKeymapAlgorithm) ?
346                             objectMetadata.getContentLength() : -1);
347             addHeader(headerMap, CONTENT_LENGTH, contentLength);
348 
349             if (StringUtils.isBlank(objectMetadata.getContentDisposition())) {
350                 String filename = objectKey.substring(objectKey.lastIndexOf(URL_SEPARATOR) + 1);
351                 addHeader(headerMap, CONTENT_DISPOSITION,
352                         String.format(DISPOSITION_FORMAT, filename));
353             }
354 
355             addHeader(headerMap, CONTENT_ENCODING, objectMetadata.getContentEncoding());
356             addHeader(headerMap, LAST_MODIFIED,
357                     new DateTime(objectMetadata.getLastModified(), UTC));
358             addHeader(headerMap, CACHE_CONTROL, objectMetadata.getCacheControl());
359 
360             // add user metadata to header map
361             headerMap.putAll(objectMetadata.getUserMetadata());
362 
363             // add raw metadata (e.g. server-side encryption) to header map
364             objectMetadata.getRawMetadata().forEach((key, value) ->
365                     headerMap.putIfAbsent(key.toLowerCase(), value.toString()));
366 
367             if (configuration.isGetObjectTags()) {
368                 getObjectTagging(headerMap, bucket, objectKey);
369             }
370 
371             return headerMap;
372         }
373         return new LinkedHashMap<>(0);
374     }

378             headerMap.putIfAbsent(headerName.toLowerCase(), value);

543                     int year = validateBigInteger(p, YEAR_PROP);
544                     int month = validateBigInteger(p, MONTH_PROP);
545                     int date = validateBigInteger(p, DATE_PROP);

547                     DateTimeZone dateTimeZone = StringUtils.isBlank(zone) ? UTC :
548                             DateTimeZone.forID(zone);

551                 switch (dates.size()) {
552                     case 0:
553                         break;
554                     case 1:

556                         break;
557                     default:
558                         interval.add(dates.first().toDate());

560                         interval.add(dates.last().plusDays(1).toDate());

567                 String fileKey, List<Date> interval) {
568             ListVersionsRequest request = new ListVersionsRequest()
569                     .withBucketName(bucketName)
570                     .withPrefix(fileKey);
571             final List<String> suggestionList = new ArrayList<>();
572             VersionListing versionListing;
573             do {
574                 versionListing = amazonS3.listVersions(request);
575                 versionListing.getVersionSummaries()
576                         .stream()
577                         // Do not suggest an S3 object version ID of 'Delete Marker' resource type
578                         // since it cannot be downloaded
579                         .filter(s -> !s.isDeleteMarker())
580                         .forEach(s -> {
581                             String versionId = s.getVersionId();
582                             if (!NULL.equalsIgnoreCase(versionId)) {
583                                 String size = humanReadableByteCount(s.getSize(), true);
584                                 Date date = s.getLastModified();
585                                 if (isWithinInterval(date, interval)) {
586                                     suggestionList.add(
587                                         String.format(FORMAT_VERSION, date, size, versionId));
588                                 }
589                             }
590                         });
591                 request.setKeyMarker(versionListing.getNextKeyMarker());
592                 request.setVersionIdMarker(versionListing.getNextVersionIdMarker());

595                 // there should be at leat one empty entry in the suggestions
596                 suggestionList.add(StringUtils.EMPTY);

612         private String humanReadableByteCount(long bytes, boolean si) {
613             int unit = si ? 1000 : 1024;
614             if (bytes < unit) {
615                 return bytes + BYTES;
616             }
617             int exp = (int) (Math.log(bytes) / Math.log(unit));
618             String pre = (si ? SI_KMGTPE : KMGTPE).charAt(exp - 1) + (si ? StringUtils.EMPTY : "i");
619             return String.format(FORMAT_BYTE_COUNT, bytes / Math.pow(unit, exp), pre);
620         }
