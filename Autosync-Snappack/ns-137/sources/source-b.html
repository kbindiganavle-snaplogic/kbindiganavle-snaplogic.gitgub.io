142     public void process(Document document, String inputViewName) {
143         try {
144             super.process(document, inputViewName);
145             s3SelectConfig.process(document);
146             if (s3Service == null) {
147                 s3Service = S3Service.builder()
148                         .s3ClientConfig(s3ClientConfig)
149                         .threadPoolSize(s3CommonConfig.getThreadPoolSize())
150                         .build();
151                 amazonS3 = s3Service.getAmazonS3();
152             }
153             InputSerialization inputSerialization = new InputSerialization()
154                     .withCompressionType(s3SelectConfig.compressionType);
155             OutputSerialization outputSerialization = new OutputSerialization();
156             switch(s3SelectConfig.dataFormat) {
157                 case CSV:
158                     inputSerialization.withCsv(new CSVInput()
159                             .withFieldDelimiter(s3SelectConfig.fieldDelimiter)
160                             .withRecordDelimiter(s3SelectConfig.recordDelimiter)
161                             .withQuoteCharacter(s3SelectConfig.quoteChar)
162                             .withQuoteEscapeCharacter(s3SelectConfig.quoteEscapeChar)
163                             .withAllowQuotedRecordDelimiter(true)
164                             .withFileHeaderInfo(s3SelectConfig.fileHeaderInfo));
165                     outputSerialization.withCsv(new CSVOutput());
166                     break;
167                 case JSON:
168                     inputSerialization.withJson(new JSONInput().withType(JSONType.DOCUMENT));
169                     outputSerialization.withJson(new JSONOutput());
170                     break;
171                 case PARQUET:
172                     inputSerialization.withParquet(new ParquetInput());
173                     outputSerialization.withJson(new JSONOutput());
174                     break;
175                 default:
176                     throw new SnapDataException(ERR_UNSUPPORTED_INPUT_DATA_FORMAT)
177                             .formatWith(s3SelectConfig.dataFormat)
178                             .withReason(REASON_INVALID_INPUT_DATA_FORMAT)
179                             .withResolution(DATA_FORMAT_DESC);
180             }
181             SelectObjectContentRequest request = new SelectObjectContentRequest()
182                     .withBucketName(s3CommonConfig.getBucket())
183                     .withKey(s3CommonConfig.getObjectKey())
184                     .withExpression(s3SelectConfig.query)
185                     .withExpressionType(ExpressionType.SQL)
186                     .withInputSerialization(inputSerialization)
187                     .withOutputSerialization(outputSerialization);
188             // execute the select query
189             statusReporter.beginUnknown();
190             statusReporter.notify(SELECT_QUERY_EXECUTING, s3CommonConfig.getBucket(),
191                     s3CommonConfig.getObjectKey());
192             if (s3Service.refreshAmazonS3IfNeeded()) {
193                 amazonS3 = s3Service.getAmazonS3();
194             }
195             SelectObjectContentResult result = amazonS3.selectObjectContent(request);
196             processResult(result, document);
197         } catch (SuggestViewAbortException e) {
198             throw e;
199         } catch (SnapDataException e) {
200             errorViews.write(e, document);
201         } catch (Exception e) {
202             boolean isCseKms = e instanceof AmazonS3Exception &&
203                     INVALID_TEXT_ENCODING.equals(((AmazonS3Exception) e).getErrorCode()) &&
204                     CSE_KMS.equals(s3Service.getS3ClientConfig().getEncryptionType());
205             errorViews.write((SnapDataException) new SnapDataException(e, ERR_SELECT_QUERY)
206                     .formatWith(s3SelectConfig.query, s3CommonConfig.getBucket(),
207                             s3CommonConfig.getObjectKey())
208                     .withReason(isCseKms ? REASON_CSE_KMS : e.getMessage())
209                     .withResolution(isCseKms ? RESOLUTION_CSE_KMS : RESOLUTION_CHECK_PROPERTIES),
210                     document);
211         }
212     }

230     private void processResult(SelectObjectContentResult result, Document document)
231             throws IOException {
232         // Currently, SelectObjectContentResult doesn't contain any response header.
233         // The header document should be outside "new BinaryOutput()" to be idempotent in
234         // ultra-pipeline mode where getHeader() can be called more than once.
235         final Document header = documentUtility.newDocumentFor(document, new LinkedHashMap<>());
236         try (InputStream inputStream = getInputStream(result)) {
237             outputViews.write(new BinaryOutput() {
238                 @Override
239                 public Document getHeader() {
240                     return header;
241                 }
242 
243                 @Override
244                 public void write(WritableByteChannel writeChannel) throws IOException {
245                     try (OutputStream outputStream = Channels.newOutputStream(writeChannel)) {
246                         statusReporter.notify(SELECT_QUERY_RESULT_DOWNLOAD,
247                                 s3CommonConfig.getBucket(), s3CommonConfig.getObjectKey());
248                         long copied = StatusReportingUtils.copy(statusReporter, inputStream,
249                                 outputStream, limit);
250                         LOG.debug("{} bytes copied from the select query result to output view " +
251                                 "for {}/{}", copied, s3CommonConfig.getBucket(),
252                                 s3CommonConfig.getObjectKey());
253                     } finally {
254                         statusReporter.done();
255                     }
256                 }
257             });
258         } catch (IOException e) {
259             throw new SnapDataException(e, ERR_DOWNLOAD_SELECT_RESULT)
260                     .withResolution(RESOLUTION_CHECK_PROPERTIES);
261         } finally {
262             result.close();
263         }
264     }

344         S3SelectConfig process(final Document document) {
345             query = selectQueryExpr.eval(document);
346             if (StringUtils.isBlank(query)) {
347                 throw new SnapDataException(ERR_EMPTY_QUERY)
348                         .withReason(REASON_EMPTY_QUERY)
349                         .withResolution(RESOLUTION_VALID_QUERY);
350             }
351             String type = compressionTypeExpr.eval(document);
352             compressionType = StringUtils.isBlank(type) ? NONE : CompressionType.valueOf(type);
353             dataFormat = dataFormatExpr.eval(document);
354             if (StringUtils.isBlank(dataFormat)) {
355                 dataFormat = CSV;
356             }
357             fieldDelimiter = fieldDelimiterExpr.eval(document);
358             if (TAB_STRING.equalsIgnoreCase(fieldDelimiter)) {
359                 fieldDelimiter = TAB_CHAR;
360             }
361             recordDelimiter = recordDelimiterExpr.eval(document);
362             if (StringUtils.isBlank(recordDelimiter)) {
363                 recordDelimiter = null;
364             } else {
365                 switch (recordDelimiter) {
366                     case LF:
367                     case CRLF:
368                     case LFCR:
369                         recordDelimiter = null;
370                         break;
371                     case CR:
372                         recordDelimiter = ESC_R;
373                         break;
374                     default:
375                         // No-Op
376                 }
377             }
378             quoteChar = quoteCharExpr.eval(document);
379             if (StringUtils.isBlank(quoteChar)) {
380                 quoteChar = DOUBLE_QUOTE;
381             }
382             quoteEscapeChar = quoteEscapeCharExpr.eval(document);
383             if (StringUtils.isBlank(quoteEscapeChar)) {
384                 quoteEscapeChar = BACK_SLASH;
385             }
386             fileHeaderInfo = fileHeaderInfoExpr.eval(document);
387             return this;
388         }
