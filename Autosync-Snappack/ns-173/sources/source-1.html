91         protected final Comparator<Object> comparator;
92         protected final Scanner leftScan;
93         protected final Scanner rightScan;
94         private Object leftRecord;
95         private Object rightRecord;
96 
97         public PairWiseMerger(Scanner leftScanner, Scanner rightScanner,
98                 Comparator<Object> comparator) throws IOException {
99             this.leftScan = leftScanner;
100             this.rightScan = rightScanner;
101             this.comparator = comparator;
102             leftRecord = leftScan.next();
103             rightRecord = rightScan.next();
104         }
105 
106         protected int compare() {
107             return comparator.compare(leftRecord, rightRecord);
108         }
109 
110         public Object next() throws IOException {
111             if (leftRecord == null && rightRecord == null) {
112                 return null;
113             }
114             Object winRecord;
115             if (rightRecord == null || (leftRecord != null && compare() < 0)) {
116                 winRecord = leftRecord;
117                 leftRecord = leftScan.next();
118                 return winRecord;
119             }
120             winRecord = rightRecord;
121             rightRecord = rightScan.next();
122             return winRecord;
123         }
124 
125         @Override
126         public void close() throws IOException {
127             leftScan.close();
128             rightScan.close();
129         }
130     }

197     public void add(final Object data) throws IOException {
198         inMemoryTable.add(data);
199 
200         //Update the object size estimation
201         if (objectSizeEstimator.addSizeObservation(data)) {
202             maxBufferSize = adjustMaxBufferSize(maxBufferSize);
203             //The object size estimation has changed. Recalculate the number of objects that can be
204             //buffered in memory before persisting the objects to disk.
205             bufferedObjThreshold = objectSizeEstimator.getDocCountEstimate(maxBufferSize);
206         }
207 
208         try {
209             if (++bufferedObjCount > bufferedObjThreshold) {
210                 LOG.info("Sorting a chunk and writing to temp file, object count: {}",
211                         bufferedObjCount);
212                 memoryResident = false;
213                 chunkPaths.add(sortAndStoreChunk(chunkId++, inMemoryTable));
214                 bufferedObjCount = 0;
215 
216                 //Check to see if a previous merge thread has finished
217                 checkMergeFutures(false);
218 
219                 //Check to see if there are enough chunks present to start a merge now
220                 if (chunkPaths.size() > DEFAULT_FANOUT) {
221                     // This means there will need to be intermediate merge files. Start sorting and
222                     // combining into an intermediate file while the data is still being added to
223                     // save time. Sort the chunks from smallest files first to reduce data
224                     // movement between intermediate files.
225                     sortChunksBySize();
226                     startMerge();
227                 }
228             }
229         } catch (IOException e) {
230             throw new SnapDataException(e, ERR_SORT_AND_STORE)
231                     .withReason(e.getMessage())
232                     .withResolution(RESOLUTION_ADDRESS_ISSUE);
233         } catch (ExecutionError | NoClassDefFoundError e) {
234             throw new SnapDataException(e, ERR_NON_JAVA_OBJECT)
235                     .withReason(e.getMessage())
236                     .withResolution(RESOLUTION_NON_JAVA);
237         }
238     }

276      * Schedules a merge to occur on the merge thread pool.
277      */
278     private void startMerge() {
279         //Identify files to merge
280         List<SortFile> filesToMerge = new ArrayList<>();
281         for (SortFile chunk : chunkPaths) {
282             filesToMerge.add(chunk);
283             if (filesToMerge.size() == DEFAULT_FANOUT) {
284                 break;
285             }
286         }
287         chunkPaths.removeAll(filesToMerge);
288 
289         //Define path for merged file
290         String outputPath = String.format("%s/out_%s", sortTmpDir, chunkId++);
291 
292         //Start merge thread
293         Future<SortFile> mergeFuture = mergeThreadPool.submit(() -> {
294             try {
295                 //Merge files
296                 LOG.info("{} files are being merged to output file {}", filesToMerge.size() , outputPath);
297 
298                 SortFile outputSortFile;
299                 try (Scanner merger = createKWayMerger(filesToMerge, 0, filesToMerge.size());
300                      final SortFileWriter output = new SortFileWriter(outputPath)) {
301                     Object mergedObject;
302                     while((mergedObject = merger.next()) != null) {
303                         output.append(mergedObject);
304                     }
305 
306                     outputSortFile = new SortFile(output.count(), outputPath);
307                 }
308                 LOG.info("{} is written to a disk.", outputPath);
309 
310                 //Delete merged input files
311                 for (SortFile mergedFile : filesToMerge) {
312                     new File(mergedFile.getPath()).delete();
313                 }
314 
315                 return outputSortFile;
316             } catch (IOException e) {
317                 throw new com.snaplogic.api.ExecutionException(e, "There was an error merging files to %s")
318                         .formatWith(outputPath);
319             }
320         });
321 
322         //Add merge thread to tracking collection
323         mergeFutures.add(mergeFuture);
324     }

327      * The entire records are inserted and now it begins an external merge sort.
328      *
329      * @return scanner
330      */
331     public Scanner sort() {
332         Scanner scanner;
333         try {
334             if (!memoryResident && !inMemoryTable.isEmpty()) {
335                 chunkPaths.add(sortAndStoreChunk(chunkId, inMemoryTable));
336             }
337             if (memoryResident) { // if all sorted data reside in a main-memory table.
338                 scanner = new MemTableScanner(inMemoryTable, inMemoryTable.size(), sortComparator);
339             } else { // if input data exceeds main-memory at least once
340                 scanner = externalMergeAndSort();
341             }
342         } catch (IOException | InterruptedException | ExecutionException e) {
343             throw new SnapDataException(e, ERR_EXTERNAL_MERGE_SORT)
344                     .withReason(e.getMessage())
345                     .withResolution(RESOLUTION_ADDRESS_ISSUE);
346         }
347         return scanner;
348     }

384     private SortFile sortAndStoreChunk(int chunkId, List<Object> datas)
385             throws IOException {
386         Collections.sort(datas, sortComparator);
387         String outputPath = String.format("%s/%s", sortTmpDir, chunkId);
388         long startTime = System.currentTimeMillis();
389         SortFile sortFile;
390         try (final SortFileWriter writer = new SortFileWriter(outputPath)) {
391             for (Object eachObject : datas) {
392                 writer.append(eachObject);
393             }
394 
395             sortFile = new SortFile(writer.count(), outputPath);
396         }
397 
398         datas.clear();
399         LOG.info("Finished writing to sorted chunk: {} to {}, {} ms",
400                 chunkId, outputPath, (System.currentTimeMillis() - startTime));
401         printMemory();
402 
403         return sortFile;
404     }

439     /**
440      * This sorts the chunks by file size. This is used so that the smallest files are always merged first.
441      * Merging the smallest files first reduces the total work because it means less data is moved between intermediate
442      * files.
443      *
444      * For example, if three chunks need to be merged to two chunks before the final merge.
445      * chunk1 (10 bytes)
446      * chunk3 (1000 bytes)
447      * chunk2 (100 bytes)
448      *
449      * Merging chunk 1 and 3 results in 1010 bytes of intermediate processing
450      * chunkMerged (1010 bytes)
451      * chunk2 (100 bytes)
452      *
453      * Merging chunk 1 and 2 results in 110 bytes of intermediate processing. This is optimal.
454      * chunkMerged (110 bytes)
455      * chunk3 (1000 bytes)
456      */
457     private void sortChunksBySize() {
458         Collections.sort(chunkPaths, (file1, file2) -> {
459             long length1 = new File(file1.getPath()).length();
460             long length2 = new File(file2.getPath()).length();
461             return Long.compare(length1, length2);
462         });
463     }

465     private Scanner externalMergeAndSort() throws IOException, ExecutionException,
466             InterruptedException {
467         LOG.info("Start externalMergeAndSort: default fan-out: {}", DEFAULT_FANOUT);
468         long mergeStart = System.currentTimeMillis();
469 
470         //Wait for merges that were scheduled during the add phase to finish.
471         checkMergeFutures(true);
472 
473         //Merge the chunks until the number of files is small enough to create the final merge.
474         while (chunkPaths.size() > DEFAULT_FANOUT) {
475             //Sort the merge chunks so that the minimum amount of data is moved between intermediate files.
476             sortChunksBySize();
477 
478             //Schedule a merge for each group of files
479             while (chunkPaths.size() > DEFAULT_FANOUT) {
480                 startMerge();
481             }
482 
483             //Wait for the scheduled merge work to finish.
484             //The output files will be merged in the next loop, if required.
485             checkMergeFutures(true);
486         }
487 
488         long mergeEnd = System.currentTimeMillis();
489         LOG.info("Total merge time: {} ms", mergeEnd - mergeStart);
490         return createKWayMerger(chunkPaths, 0, chunkPaths.size());
491     }

493     private Scanner createKWayMerger(List<SortFile> inputs, final int startChunkId,
494             final int num) throws IOException {
495         final Scanner[] sources = new Scanner[num];
496         for (int i = 0; i < num; i++) {
497             SortFile sortFile = inputs.get(startChunkId + i);
498             sources[i] = new FileScanner(sortFile.getPath(), sortFile.getCount());
499         }
500         return createKWayMergerInternal(sources, 0, num);
501     }
502 
503     private Scanner createKWayMergerInternal(final Scanner[] sources, final int startIdx,
504             final int num) throws IOException {
505         if (num > 1) {
506             final int mid = (int) Math.ceil((float)num / 2);
507             Scanner left = createKWayMergerInternal(sources, startIdx, mid);
508             Scanner right = createKWayMergerInternal(sources, startIdx + mid, num - mid);
509             return new PairWiseMerger(left, right, sortComparator);
510         } else {
511             return sources[startIdx];
512         }
513     }
