114     @Override
115     protected void process(Document inputDocument, String inputViewName) {
116         Object object = inputSearchTerm.eval(inputDocument);
117         if (object instanceof String) {
118             searchTerm = (String) object;
119         } else {
120             try {
121                 searchTerm = FacebookUtil.OBJECT_MAPPER.writeValueAsString(object);
122             } catch (IOException ex) {
123                 String reason = String.format(ERROR_JSON_REASON, ex.getMessage());
124                 throw new ExecutionException(ex, ERROR_JSON)
125                         .withReason(reason)
126                         .withResolution(ERROR_JSON_RESOLUTION);
127             }
128         }
129         searchResult(inputDocument);
130     }

132     private void searchResult(Document inputDocument) {
133 
134 
135         try {
136             if (typeValue.equals(TYPE_USER)) {
137                 Parameter fields = Parameter.with("fields",
138                         USER_FIELDS);
139                 map = facebookUtil.getMapData(searchTerm, accessToken, typeValue, fields);
140             } else {
141                 map = facebookUtil.getMapData(searchTerm, accessToken, typeValue);
142             }
143 
144             List<Map> listMap = (List<Map>) map.get(DATA_KEY);
145             for (Map recordMap : listMap) {
146                 outputViews.write(documentUtility.newDocumentFor(inputDocument, recordMap));
147             }
148         } catch (IOException ex) {
149             SnapDataException e =
150                     new SnapDataException(ex, ex.getMessage())
151                             .withReason(IO_ERROR_REASON)
152                             .withResolution(IO_RESOLUTION);
153             errorViews.write(e, inputDocument);
154         }
155     }
