20 /**
21  * GetExecutor for Maps with custom support for EMPTY_DATA placeholders. If a Map entry has a null
22  * or empty value, it is replaced with the EMPTY_DATA placeholder.
23  */
24 public class EmptyDataMapGetExecutor extends MapGetExecutor {
25     private final String property;
26 
27     public EmptyDataMapGetExecutor(final Log log, final Class clazz, final String property) {
28         super(log, clazz, property);
29         this.property = property;
30     }
31 
32     @Override
33     public Object execute(final Object o) {
34         if (!(o instanceof Map)) {
35             throw new IllegalArgumentException("for property=" + property +
36                     " expected a Map, got " + o);
37         }
38         if (((Map) o).containsKey(property)) {
39             Object value = ((Map) o).get(property);
40             if (value == null || value.toString().isEmpty()) {
41                 return XmlUtilsImpl.EMPTY_DATA;
42             } else {
43                 return value;
44             }
45         }
46 
47         return XmlUtilsImpl.NO_DATA;
48     }
49 }
