191     protected void process(Document document, String inputViewName) {
192         try {
193             initRequestExecutor(httpMethod)

195                     .execute(document, createURLResolver(), new RequestDataHandler<StringEntity>() {
196                         @Override
197                         public StringEntity createEntity(final Document document) {
198                             try {
199                                 String contents = httpMethod.equals(HttpGet.METHOD_NAME) ? null :
200                                         NO_CONTENT;
201                                 Map<String, Object> data = documentUtility.getAsMap(document,
202                                         errorViews);
203                                 if (data != null) {
204                                     data = processDocument(document, data);
205                                     if (data == null) {
206                                         return null; // an error was detected in the input data
207                                     }
208                                     contents = objectMapper.writeValueAsString(data);
209                                 }
210                                 return createStringEntity(contents);
211                             } catch (IOException e) {
212                                 throw new ExecutionException(e, ERR_INPUT_DATA)
213                                         .withReason(ERR_INPUT_DATA)
214                                         .withResolution(ERR_INPUT_DATA_RESOLUTION);
215                             }

217                         }
218                     });
219         } catch (SnapDataException e) {
220             errorViews.write(e, document);

300     @SuppressWarnings("unchecked")
301     protected String formatRestUriWithIssueId(Document document) throws SnapDataException {
302         Map<String, Object> data = null;
303         if (document != null) {
304             data = document.get(Map.class);
305             if (data != null) {
306                 String id = (String) data.get(ISSUE_ID_KEY);
307                 if (StringUtils.isNotBlank(id)) {
308                     return String.format(FORMAT_REST_URI, hostname, version, id);
309                 }
310             }
311         }
312         data = Maps.newHashMapWithExpectedSize(2);
313         data.put(ERROR_KEY, ERR_ISSUE_ID);
314         throw new SnapDataException(documentUtility.newDocumentFor(document, data), ERR_ISSUE_ID)
315                 .withReason(ERR_ISSUE_ID)
316                 .withResolution(ERR_ISSUE_ID_RESOLUTION);
317     }

319     protected static void checkHostnameSyntax(String hostname) {
320         URI uri;
321         try {
322             uri = new URI(hostname);
323         } catch (URISyntaxException e) {
324             throw new ExecutionException(e, ERR_HOSTNAME_SYNTAX_REASON).formatWith(hostname)
325                     .withReason(String.format(ERR_HOSTNAME_SYNTAX_REASON, hostname))
326                     .withResolution(HOSTNAME_DESC);
327         }
328         String scheme = uri.getScheme();
329         if (StringUtils.isBlank(scheme)) {
330             throw new ExecutionException(ERR_URI_SCHEME).formatWith(hostname)
331                     .withReason(String.format(ERR_URI_SCHEME, hostname))
332                     .withResolution(HOSTNAME_DESC);
333         }
