676     /**
677      * Formats SOQL query.
678      *
679      * @param fields    : output fields of SELECT statement
680      * @param orderById : ORDER BY Id clause. If not null, this method is called by vertical slicing
681      *
682      * @return SOQL query
683      */
684     public String toSOQL(String fields, String orderById) {
685         // format SOQL query
686         String soql = String.format(FORMAT_SOQL, fields, getObjectType());
687         // append WHERE clause if any
688         if (StringUtils.isNotBlank(whereClause)) {
689             soql = String.format(FORMAT_WHERE, soql, whereClause);
690         }
691         // append ORDER BY if any
692         if (StringUtils.isNotBlank(orderbyClause)) {
693             // if order by clause contains 'Id' field, ignore orderById argument
694             String[] columns = orderbyClause.split(COMMA);
695             for (String column : columns) {
696                 if (StringUtils.trimToEmpty(column).equalsIgnoreCase(ID)) {
697                     orderById = null;
698                 }
699             }
700             String orderby = StringUtils.isBlank(orderbyClause) ? orderById :
701                     String.format(ORDER_BY, orderbyClause);
702             if (StringUtils.isNotBlank(orderById)) {
703                 orderby = orderby.concat(CONCAT_ID);
704             }
705             soql = soql.concat(orderby);
706         } else {
707             if (StringUtils.isNotBlank(orderById)) {
708                 soql = soql.concat(orderById);
709             }
710         }
711         // append LIMIT clause if any
712         int limit = StringUtils.isBlank(limitClause) ? Integer.MAX_VALUE :
713                 Integer.valueOf(limitClause);
714         limit = isPreview ? Integer.min(maxSuggestValue.intValue(), limit) : limit;
715         return limit == Integer.MAX_VALUE ? soql : soql.concat(String.format(LIMIT_N, limit));
716     }
