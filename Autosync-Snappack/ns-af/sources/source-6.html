331     protected void process(Document document, String inputViewName) {
332         try {
333             super.process(document, inputViewName);
334             currentDocumentReference = document;
335             int pageCount = pageSize;
336             if (isExecuteDuringPreview) {
337                 pageCount = maxSuggestValue.intValue();
338             } else if (pageCount <= 0) {
339                 pageCount = DEFAULT_PAGESIZE;
340             }
341             if (null != objectIdExpression) {
342                 objectId = objectIdExpression.eval(document);
343             }
344             queryStringParts = new HashMap<>();
345             setOutputAttributes(document);
346             setFilterObjects(document);
347             setOrderByParameters(document);
348             setAdditionalQueryParameters(document);
349 
350             //Removing logical operator If any present at the end of filter condition.
351             StringBuilder builder = queryStringParts.get(FILTER_OPTION);
352             if (builder != null) {
353                 String filterCond = builder.toString();
354                 if (filterCond.endsWith(AND) ||
355                         filterCond.endsWith(OR) ||
356                         filterCond.endsWith(NOT)) {
357                     queryStringParts.put(FILTER_OPTION, builder.delete(builder.length() - 4,
358                             builder.length()));
359                 }
360             }
361             boolean pagingNotRequired = isExecuteDuringPreview || StringUtils.isNotBlank(objectId) ||
362                     (maxPage == ONE_PAGE);
363             Map<String, Object> responseMap =
364                     communicateWithDynamics365ForSales(objectType, objectId,
365                             Integer.valueOf(pageCount), null);
366             if (null != responseMap) {
367                 if (pagingNotRequired) {
368                     writeSingleOrMultipleOutputDocuments(responseMap, document);
369                 } else {
370                     //first check if we NEED to page.
371                     Object nextLink = responseMap.get(ODATA_NEXT_LINK);
372                     writeSingleOrMultipleOutputDocuments(responseMap, document);
373                     if (nextLink != null) {
374                         handleRequiredPaging(((String) nextLink),
375                                 pageCount, maxPage, document);
376                     }
377                 }
378             }
379         } finally {
380             currentDocumentReference = null;
381         }
382     }

398     private void setFilterObjects(Document document) {
399         if (filterExpressions != null) {
400             StringBuilder builder = new StringBuilder();
401             for (Map<String, ExpressionProperty> mapFilter : filterExpressions) {
402                 String filterType =
403                         mapFilter.get(FILTER_TYPE_PROP).eval(document);
404                 String filterAttribute =
405                         mapFilter.get(FILTER_ATTRIBUTE_PROP).eval(document);
406                 String filterOperator =
407                         mapFilter.get(FILTER_OPERATOR_PROP).eval(document);
408                 filterOperator = SearchComparisonOperator.valueOf(filterOperator)
409                         .getAbbreviation();
410                 String filterValue =
411                         mapFilter.get(FILTER_VALUE_PROP).eval(document);
412 
413                 if (StringUtils.isBlank(filterAttribute)) {
414                     throw new SnapDataException(Messages.INVALID_SEARCH_CONDITION)
415                             .withReason(Messages.INVALID_FILTER_ATTRIBUTE)
416                             .withResolution(Messages.PLEASE_CORRECT_AND_RETRY);
417                 }
418                 if (StringUtils.isBlank(filterValue)) {
419                     throw new SnapDataException(Messages.INVALID_SEARCH_CONDITION)
420                             .withReason(Messages.INVALID_FILTER_VALUE)
421                             .withResolution(Messages.PLEASE_CORRECT_AND_RETRY);
422                 }
423                 builder.append(filterAttribute).append(SPACE_CHAR)
424                         .append(filterOperator).append(SPACE_CHAR)
425                         .append(filterValue).append(SPACE_CHAR)
426                         .append(filterType).append(SPACE_CHAR);

428             }

435     private void setAdditionalQueryParameters(Document document) {
436         if (queryParams != null) {
437             StringBuilder builder;
438             for (Pair<String, ExpressionProperty> paramPair : queryParams) {
439                 // left is the param key, right is the value
440                 String paramKey = paramPair.getLeft();
441                 if (StringUtils.isBlank(paramKey)) {
442                     throw new SnapDataException(Messages.INVALID_SEARCH_CONDITION)
443                             .withReason(Messages.INVALID_FILTER_ATTRIBUTE)
444                             .withResolution(Messages.PLEASE_CORRECT_AND_RETRY);
445                 }
446                 String paramValue = paramPair.getRight().eval(document);
447                 if (StringUtils.isBlank(paramValue)) {
448                     throw new SnapDataException(Messages.INVALID_SEARCH_CONDITION)
449                             .withReason(Messages.INVALID_FILTER_VALUE)
450                             .withResolution(Messages.PLEASE_CORRECT_AND_RETRY);
451                 }
452                 paramKey = PARAM_PREFIX + paramKey;
453                 if ((builder = queryStringParts.get(paramKey)) != null) {
454                     builder.append(paramValue);
455                 } else {
456                     if (StringUtils.isNotEmpty(paramValue)) {
457                         builder = new StringBuilder(paramValue);
458                     }
459                 }
460                 queryStringParts.put(paramKey, builder);
461             }
462             // Clearing queryParams after setting queryStringParts
463             queryParams.clear();
464         }
465     }
