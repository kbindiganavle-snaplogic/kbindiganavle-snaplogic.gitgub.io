302     /**
303      * Method that invokes API call to read emails
304      *
305      * @param document incoming document
306      */
307     private void invokeReadEmailAPI(Document document) {
308         MessageCollectionPage messageCollection =
309                 (MessageCollectionPage) performTokenRefreshableOperation(() ->
310                         userRequestBuilder.mailFolders(
311                                 EOUtils.retrieveMailFolderID(mailFoldersMap, emailFolder))
312                         .messages().buildRequest(options).get(), ERR_RETRIEVE_EMAILS_MSG);
313         boolean firstPage = true;
314         int msgCount = 0;
315         do {
316             if (firstPage) {
317                 firstPage = false;
318             } else {
319                 messageCollection = messageCollection.getNextPage().buildRequest().get();
320             }
321             List<Message> messages = messageCollection.getCurrentPage();
322             for (Message msg : messages) {
323                 Map<String, Object> data = EOUtils.writeToDocOutputView(documentUtility,
324                         outputViews, document, msg, SERIALIZER);
325                 if (BooleanUtils.isTrue(markRead)) {
326                     Message readMessage = new Message();
327                     readMessage.isRead = true;
328                     userRequestBuilder.messages(msg.id).buildRequest().patch(readMessage);
329                     msg.isRead = true;
330                 }
331                 String bodyContent = msg.body.content;
332                 Pattern pattern = Pattern.compile("src=\"(.*?)\"");
333                 boolean isHasInlineAttachments = pattern.matcher(bodyContent).find();
334                 boolean isHasAttachments = data.containsKey(HAS_ATTACHMENTS) &&
335                         ((Boolean) data.get(HAS_ATTACHMENTS));
336                 if (downloadAttach && (isHasAttachments || isHasInlineAttachments) ) {
337                     if (data.get(ID) != null) {
338                         downloadAttachments(data.get(ID).toString());
339                     } else {
340                         LOG.warn(MSG_ID_NULL);
341                         throw new SnapDataException(MSG_ID_NULL)
342                                 .withReason(MSG_ID_NULL_REASON)
343                                 .withResolutionAsDefect();
344                     }
345                 }
346                 msgCount++;
347                 if (EOUtils.countReached(numMsgs, numMsgsProvided, msgCount)) {
348                     break;
349                 }
350             }
351             if (EOUtils.countReached(numMsgs, numMsgsProvided, msgCount)) {
352                 break;
353             }
354         } while (messageCollection.getNextPage() != null);
355         LOG.info("Fetched total {} messages", msgCount);
356     }

358     /**
359      * Add Query parameters provided in Filters setting in the snap
360      *
361      * @param filters Map of Key value pairs
362      * @param doc     incoming document
363      */
364     private void addFilters(List<Map<String, Object>> filters, Document doc) {
365         options = new LinkedList<>();
366         String filterCondition = StringUtils.EMPTY;
367         // receivedDate filter is needed always to sort the results.
368         // If not added by user, we have to add explicitly.
369         boolean receivedDateFilterAdded = false;
370         if (CollectionUtils.isNotEmpty(filters)) {
371             for (Map<String, Object> filterRow : filters) {
372                 if (StringUtils.isNotEmpty(filterCondition)) {
373                     filterCondition += " and ";
374                 }
375                 String operator = filterRow.get("rowOp").toString();
376                 String leftOperand = FIELD_MAP.get(filterRow.get("rowKey").toString());
377                 Object rightOperand =
378                         ((ExpressionProperty) filterRow.get("rowValue")).eval(doc);
379                 if (RECEIVED_DATE_FIELD_PATH.equals(leftOperand)) {
380                     receivedDateFilterAdded = true;
381                 }
382                 switch (operator) {
383                     case EQ_OP:
384                     case NE_OP:
385                         filterCondition += leftOperand + StringUtils.SPACE
386                                 + operator + StringUtils.SPACE + SINGLE_QUOTE_CHAR
387                                 + rightOperand + SINGLE_QUOTE_CHAR;
388                         break;
389                     case GT_OP:
390                     case GE_OP:
391                     case LT_OP:
392                     case LE_OP:
393                         filterCondition += leftOperand + StringUtils.SPACE
394                                 + operator + StringUtils.SPACE + rightOperand;
395                         break;
396                     case STARTSWITH_OP:
397                         filterCondition +=
398                                 operator + "(" + leftOperand + ",'" + rightOperand + "')";
399                         break;
400                     default:
401                         throw new SnapDataException("Unsupported Filter Operation")
402                                 .withReason("Selected operation is not supportes")
403                                 .withResolution("Select operation from the provided list");
404                 }
405             }
406         }
407         if (focusedOnly) {
408             if (StringUtils.isNotEmpty(filterCondition)) {
409                 filterCondition += " and ";
410             }
411             filterCondition += "inferenceClassification eq 'focused'";
412         }
413         if (unreadOnly) {
414             if (StringUtils.isNotEmpty(filterCondition)) {
415                 filterCondition += " and ";
416             }
417             filterCondition += "isRead eq false";
418         }
419         if (StringUtils.isNotEmpty(filterCondition)) {
420             if (BooleanUtils.isNotTrue(receivedDateFilterAdded)) {
421                 filterCondition = "receivedDateTime ge 1970-01-01 and " + filterCondition;
422             }
423             options.add(new QueryOption("filter", filterCondition));
424         }
425         if (oldOnTop) {
426             options.add(new QueryOption("orderby", "receivedDateTime asc"));
427         } else {
428             options.add(new QueryOption("orderby", "receivedDateTime desc"));
429         }
430     }
