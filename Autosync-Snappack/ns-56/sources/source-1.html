152     @Override
153     public void close() throws SQLException {
154         if (hostedConnection != null) {
155             try {
156                 boolean evicted = false;
157                 if (evictConnectionOnClose) {
158                     evicted = account.evictConnection(hostedConnection);
159                     if (LOG.isDebugEnabled() && evicted) {
160                         LOG.debug("(+) evicted connection @{}",
161                             Integer.toHexString(hostedConnection.hashCode()));
162                     }
163                 }
164                 if (!evicted) {
165                     hostedConnection.close();
166                 }
167             } finally {
168                 if (LOG.isDebugEnabled() && hostedConnection != null) {
169                     LOG.debug("(+) closing connection id: @{} {}",
170                             Integer.toHexString(hostedConnection
171                                     .hashCode()), snapID);
172                 }
173                 SnapConnection.remove(this);
174                 hostedConnection = null;
175             }
176         } else {
177             throw new SQLException(CONNECTION_CLOSED);
178         }
179     }
