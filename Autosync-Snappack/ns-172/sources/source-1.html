117                                 final SnapInternals snapInternals,
118                                 File tmpDir,
119                                 ThreadFactory threadFactory) {
120         this.orderValidator = orderValidator;
121         this.threadFactory = threadFactory;
122         this.snapInternals = snapInternals;
123         inputViews = snapInternals.getInputViews();
124         outputViews = snapInternals.getOutputViews();
125         errorViews = snapInternals.getErrorViews();
126         isErrorViewClosed =  PropertyCategory.ErrorBehaviourType.FAIL ==
127                 errorViews.getErrorBehaviour();

129         statusReporter = snapInternals.getStatusReporter();
130         diskMemoryDocSize = snapInternals.getDiskMemoryDocSize();
131         diskMemorySize = snapInternals.getDiskMemorySize();
132         unmatchedDataToErrorView = Boolean.TRUE.equals(propertyValues.get(UNMATCHED_DATA_PROP));
133         nullGreater = Boolean.TRUE.equals(propertyValues.get(NULL_GREATER_PROP));
134         sortOrder = SortOrder.fromString((String) propertyValues.get(SORTED_STREAMS_PROP));
135         nullSafeAccess = Boolean.TRUE.equals(propertyValues.get(NULL_SAFE_ACCESS_PROP));
136         this.tmpDir = tmpDir;
137         // Initialize input view ID's and labels.
138         // inputViews.getAll() returns a HashMap<{input view ID}, InputView>.
139         // Note that ID's are called as 'names' in the InputViews class.
140         // The input view labels can be obtained from inputView.getLabel().
141         // Labels may be changed by users in the Views pane of the Snap settings dialog box.
142         Map<String, InputView> viewMap = inputViews.getAll();
143         if (viewMap != null) {
144             inputViewCount = viewMap.size();
145             if (inputViewCount > 1) {
146                 // get inputViewNames(Map<inputViewLabel, inputViewId>) sorted by inputViewId
147                 inputViewNames = new LinkedHashMap<>(inputViewCount);
148                 List<String> inputViewIds = new ArrayList<>(viewMap.keySet());
149                 Collections.sort(inputViewIds);
150                 for (String inputViewId : inputViewIds) {
151                     String inputVewLabel = viewMap.get(inputViewId).getLabel();
152                     if (inputViewNames.containsKey(inputVewLabel)) {
153                         throw new ConfigurationException(ERR_DUPLICATE_INPUT_VIEW_LABEL)
154                                 .formatWith(inputVewLabel)
155                                 .withReason(REASON_DUPLICATE_INPUT_VIEW_LABEL)
156                                 .withResolution(RESOLUTION_ADDRESS_ISSUE);
157                     }
158                     inputViewNames.put(inputVewLabel, inputViewId);
159                 }
160                 // initialize other data for join
161                 joinDataMap = new LinkedHashMap<>(inputViewCount);
162                 lastJoinDataMap = new HashMap<>(inputViewCount);
163                 List<String> inputViewLabels = new ArrayList<>(inputViewNames.keySet());
164                 isStarting = new HashMap<>(inputViewLabels.size());
165                 for (String inputViewLabel : inputViewLabels) {
166                     isStarting.put(inputViewLabel, true);
167                 }
168                 leftInputViewLabel = inputViewLabels.remove(0);
169                 rightInputViewLabels = inputViewLabels;
170                 remainingInputViews = new ArrayList<>(rightInputViewLabels);
171                 endedInputViews = new ArrayList<>(rightInputViewLabels.size());
172                 String joinType = propertyValues.get(JOIN_TYPE_PROP);
173                 if (JOIN_TYPE_MERGE.equals(joinType)) {
174                     // ignore join paths for merge join type
175                     return;
176                 }
177                 // process join paths table property
178                 List<Map<String, Object>> joinPathsTable = propertyValues.get(JOIN_PATHS_PROP);
179                 if (joinPathsTable.isEmpty()) {
180                     throwNoJoinPathException(joinType);
181                 }
182                 Map<String, Map<String, ExpressionProperty>> pathRelation = new LinkedHashMap<>();
183                 for (Map<String, Object> joinPathRow : joinPathsTable) {
184                     ExpressionProperty leftPath =
185                             propertyValues.getExpressionPropertyFor(joinPathRow, LEFT_PATH_PROP);
186                     String leftPathStr = leftPath.getExpression();

188                     if (!pathRelation.containsKey(leftPathStr)) {
189                         Map<String, ExpressionProperty> pathMap = new HashMap<>(inputViewCount);

200                                 .withReason(e.getMessage())
201                                 .withResolutionAsDefect();
202                     }
203                     ExpressionProperty rightPath =
204                             propertyValues.getExpressionPropertyFor(joinPathRow, RIGHT_PATH_PROP);
205                     pathRelation.get(leftPathStr).put(rightInputView, rightPath);
206                 }
207                 // validate join paths property and init join paths
208                 joinPaths = new LinkedHashMap<>(inputViewCount);
209                 for (String inputViewLabel : inputViewNames.keySet()) {
210                     joinPaths.put(inputViewLabel,
211                             new ArrayList<ExpressionProperty>(pathRelation.size()));
212                 }
213                 for (Map<String, ExpressionProperty> paths : pathRelation.values()) {
214                     for (String inputViewLabel : inputViewNames.keySet()) {
215                         ExpressionProperty path = paths.get(inputViewLabel);
216                         if (path == null) {
217                             throw new ConfigurationException(ERR_RIGHT_PATH_NOT_FOUND)
218                                     .formatWith(inputViewLabel)
219                                     .withReason(REASON_RIGHT_PATH_NOT_FOUND)
220                                     .withResolution(RESOLUTION_ADDRESS_ISSUE);
221                         }
222                         checkForEmptyJoinPath(path.getExpression(), joinType);
223                         joinPaths.get(inputViewLabel).add(path);
224                     }
225                 }
226                 leftJoinPaths = joinPaths.get(leftInputViewLabel);
227                 return;
228             }
229         }

232                 .withResolution(RESOLUTION_ADDRESS_ISSUE);

245      * Gets the next one document or all documents with the same join-paths value for a given
246      * input view. Left view needs one document at a time while right views need all so that
247      * join operation can be performed.
248      *
249      * @param inputViewLabel  - input view label to get the next document from
250      * @param getOne - if true, gets one next document,
251      *                 if false, gets all documents with same join-paths value
252      *
253      * @return true if next document is same as the last one in join-paths value,
254      *         false if next document is not same as the last one, but in correct order
255      *         null if there is no more document
256      */
257     protected Boolean getNextDocumentsFor(String inputViewLabel, boolean getOne) {
258         List<JoinData> joinDataList = joinDataMap.get(inputViewLabel);
259         joinDataList.clear();
260         JoinData lastJoinData = lastJoinDataMap.get(inputViewLabel);
261         if (lastJoinData == null) {
262             if (isStarting.get(inputViewLabel)) {
263                 isStarting.put(inputViewLabel, false);
264             } else {
265                 // end of the document stream
266                 endedInputViews.add(inputViewLabel);
267                 return null;
268             }
269         } else if (!getOne) {
270             joinDataList.add(lastJoinData);
271         }
272         List<ExpressionProperty> joinPath = joinPaths.get(inputViewLabel);
273         String inputViewId = inputViewNames.get(inputViewLabel);
274         while (true) {
275             Document document = getNextDocument(inputViewId);
276             if (document == null) {
277                 // end of the document stream
278                 lastJoinDataMap.put(inputViewLabel, null);
279                 return null;
280             }
281             document.acknowledge();
282             JoinData joinData = nullGreater ?
283                     new JoinDataNullGreater(document, joinPath, nullSafeAccess) :
284                     new JoinData(document, joinPath, nullSafeAccess);
285             if (getOne) {
286                 joinDataList.add(joinData);
287                 lastJoinDataMap.put(inputViewLabel, joinData);
288                 if (lastJoinData != null) {
289                     return orderValidator.compareAndValidate(lastJoinData, joinData);
290                 } else {
291                     return false;
292                 }
293             } else {
294                 if (lastJoinData != null) {
295                     boolean areSame = orderValidator.compareAndValidate(lastJoinData, joinData);
296                     if (!areSame) {
297                         lastJoinDataMap.put(inputViewLabel, joinData);
298                         return false;
299                     }
300                 }
301                 lastJoinData = joinData;
302                 joinDataList.add(joinData);
303             }
304         }
305     }

308      * Gets one document from a give input view ID.
309      *
310      * @param inputViewId - input view ID to get the document from
311      *
312      * @return document object
313      */
314     protected Document getNextDocument(String inputViewId) {
315         //Check to see if scanners is null to see if the inputs have been sorted.
316         //For the merge join a sort is not required. It just combines documents in order and
317         //the sort configuration is ignored.
318         if (sortOrder == SortOrder.UNSORTED && scanners != null) {
319             try {
320                 Object data;
321                 if ((data = scanners.get(inputViewId).next()) != null) {
322                     return documentUtility.newDocument(data);
323                 }
324                 return null; // end of the document stream
325             } catch (IOException e) {
326                 SnapDataException ex = new SnapDataException(e, ERR_SORT_OUT)
327                         .withReason(e.getMessage())
328                         .withResolution(RESOLUTION_ADDRESS_ISSUE);
329                 errorViews.write(ex);
330             }
331         } else {
332             Iterator<Document> it = inputIterators.computeIfAbsent(
333                     inputViewId,
334                     viewId -> inputViews.documents(inputViews.get(viewId)).iterator());
335 
336             if (it.hasNext()) {
337                 return it.next();
338             }
339         }
340         return null;
341     }

344      * Sorts the document streams if unsorted. Each input stream is sorted in a separate thread.
345      */
346     protected void sortDocumentStreams() {
347         if (sortOrder == SortOrder.UNSORTED) {
348             // invoke a separate sorter thread for each input view
349             ExecutorService executor = Executors.newCachedThreadPool(threadFactory);
350             Map<String, Future<Scanner>> threads = new LinkedHashMap<>(inputViewCount);
351             joinSorters = new ArrayList<>(inputViewCount);
352             for (Map.Entry<String, String> entry : inputViewNames.entrySet()) {
353                 String inputViewLabel = entry.getKey();
354                 List<String> sortPaths = new ArrayList<>(joinPaths.size());
355                 List<ExpressionProperty> sortPathExpressions = new ArrayList<>(joinPaths.size());
356                 for (ExpressionProperty expressionProperty : joinPaths.get(inputViewLabel)) {
357                     sortPaths.add(expressionProperty.getExpression());
358                     sortPathExpressions.add(expressionProperty);
359                 }
360                 JoinSorter joinSorter = new JoinSorter(inputViews, entry.getValue(), sortPaths,
361                         nullGreater, tmpDir, threadFactory)
362                         .withSortPathExpressions(sortPathExpressions)
363                         .withNullSafeAccess(nullSafeAccess)
364                         .withStatusReporter(statusReporter, diskMemoryDocSize)
365                         .withSnapInternals(snapInternals);
366                 joinSorters.add(joinSorter);
367                 Future<Scanner> thread = executor.submit(joinSorter);
368                 threads.put(inputViewLabel, thread);
369             }
370             // get the list of scanners when sortings are completed
371             scanners = new HashMap<>(inputViewCount);
372             for (Map.Entry<String, Future<Scanner>> entry : threads.entrySet()) {
373                 String inputViewLabel = entry.getKey();
374                 Future<Scanner> thread = entry.getValue();
375                 try {
376                     Scanner scanner = thread.get();
377                     scanners.put(inputViewNames.get(inputViewLabel), scanner);
378                 } catch (SuggestViewAbortException e) {
379                     throw e;
380                 } catch (Exception e) {
381                     // This is an execution halting issue. The input must be sorted before joining
382                     // inputs.
383                     String reason;
384                     String resolution;
385                     Throwable cause = e.getCause();
386                     if (cause instanceof KryoException &&
387                             OUT_OF_SPACE.equalsIgnoreCase(cause.getMessage())) {
388                         reason = REASON_INSUFFICIENT_DISK_SPACE;
389                         resolution = RESOLUTION_INCREASE_DISK_SPACE;
390                     } else {
391                         reason = e.getMessage();
392                         resolution = RESOLUTION_ADDRESS_ISSUE;
393                     }
394                     ExecutionException ex = new ExecutionException(e, ERR_SORT)
395                             .formatWith(inputViewLabel)
396                             .withReason(reason)
397                             .withResolution(resolution);
398 
399                     throw ex;
400                 }
401             }
402         }
403     }

430         if (MapUtils.isNotEmpty(scanners)) {
431             for (Map.Entry<String, Scanner> entry : scanners.entrySet()) {
432                 Scanner scanner = entry.getValue();
433                 if (scanner != null) {
434                     try {
435                         scanner.close();

437                         LOG.debug("Failed to close Sort Scanner for input view: {}, detail: {}",

440                 }
441             }
442         }
443         if (CollectionUtils.isNotEmpty(joinSorters)) {
444             for (JoinSorter joinSorter : joinSorters) {
445                 if (joinSorter != null) {
446                     joinSorter.cleanup();
447                 }
448             }
449         }
450         // clear ArrayList and/or ExtendedArrayList in joinDataMap
451         for (List<JoinData> joinDataList : joinDataMap.values()) {
452             if (joinDataList != null) {
453                 joinDataList.clear();
454             }
