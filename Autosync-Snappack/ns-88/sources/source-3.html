234     protected static List<AutoSyncSpec> filterSpecs(String dbType, String tableName,
235                                                     List<AutoSyncSpec> specs) {
236         if (StringUtils.isNotBlank(dbType) && StringUtils.isNotBlank(tableName)) {
237             List<AutoSyncSpec> results = specs.stream().filter(spec -> {
238                 if (spec.getSource().getType().equals(dbType)) {
239                     List<TableConfig> tables = spec.getTables();
240                     for (TableConfig table : tables) {
241                         if (table.getTableName().equals(tableName)) {
242                             return true;
243                         }
244                     }
245                 }
246                 return false;
247             }).collect(Collectors.toList());
248             if (results.isEmpty()) {
249                 throw new IllegalStateException("No match was found for database type: " + dbType +
250                         ", table name: " + tableName);
251             }
252             return results;
253         } else {
254             throw new IllegalStateException("Database type and table name arguments are " +
255                     "required.");
256         }

259     private void start(AutoSyncSpec spec) {
260 
261         List<TableConfig> tables = spec.getTables();
262         tables = getTableIfNeeded(tableName, tables);
263         ArrayList<CompletableFuture<Void>> tasks = new ArrayList<>(tables.size());
264         ExecutorService executor = Executors.newFixedThreadPool(POOL_SIZE, threadFactory);
265         LOGGER.debug("Transformations enabled? {}", TRANSFORMATIONS_ENABLED);
266         try {
267             int idx = 0;
268 
269             for (TableConfig tableConfig : tables) {
270                 int tableIdx = idx;
271                 // Handle transformation flags
272                 handleTransformationFlags(tableConfig);
273                 CompletableFuture<Void> task = CompletableFuture.runAsync(() ->
274                         processTable(spec, tableConfig, tableIdx), executor).exceptionally(e -> {
275                     ErrorDTO errorDto;
276                     Throwable cause = e.getCause();
277                     if (cause instanceof PipelineExecutionException &&
278                             ((PipelineExecutionException) cause).getErrorDto() != null) {
279                         errorDto = ((PipelineExecutionException) cause).getErrorDto();
280                     } else {
281                         errorDto = new ErrorDTO();
282                         errorDto.setFailure("Failed to perform AutoSync for table " +
283                                 tableConfig.getName());
284                         errorDto.setReason(AN_AUTO_SYNC_INTERNAL_ERROR_OCCURRED);
285                         errorDto.setResolution(CONTACT_SUPPORT);
286                     }
287                     LOGGER.error("Error processing table {} = {}, {}",
288                             tableConfig.getName(), errorDto, cause);
289                     if (!previewMode) {
290                         jobStatusHandler.updateIntermediaryStatus(tableConfig.getTargetTableName(),
291                                 errorDto);
292                     } else {
293                         Map<String, Object> errorResponse =
294                                 AutoSyncUtils.getPreviewDataErrorResponse(e,
295                                         tableConfig.getSrcDbType());
296                         outputViews.write(documentUtility.newDocument(errorResponse));
297                     }
298                     return null;
299                 });
300                 idx++;
301                 tasks.add(task);
302             }
303             tasks.forEach(CompletableFuture::join);
304 
305         } finally {
306             executor.shutdown();
307         }
308     }

310     protected void handleTransformationFlags(TableConfig tableConfig){
311         if (TRANSFORMATIONS_ENABLED) {
312             String[] disabledTransformations = StringUtils.split(
313                     AutoSyncDriverSnap.DISABLED_TRANSFORMATIONS, ',');
314             // Check if disabledTransformations is null and set it to an empty array
315             if (disabledTransformations == null) {
316                 disabledTransformations = new String[0];
317             }
318             filterDisabledTransformations(tableConfig, disabledTransformations);
319         } else {
320             tableConfig.setTransformations(null);
321         }
322     }

324     protected void filterDisabledTransformations(TableConfig tableConfig,
325                                                         String[] disabledTransformations) {
326         final Map<String, List<ColumnTransformation>> transformations =
327                 tableConfig.getTransformations();
328 
329         for (Map.Entry<String, List<ColumnTransformation>> transformationsEntry :
330                 transformations.entrySet()) {
331             if (StringUtils.isNotBlank(transformationsEntry.getKey())) {
332                 List<ColumnTransformation> columnTransformations = transformationsEntry.getValue();
333 
334                 if (columnTransformations != null) {
335                     // Iterate through column transformations and remove disabled ones
336                     columnTransformations.removeIf(columnTransformation ->
337                             Arrays.asList(disabledTransformations)
338                                     .contains(columnTransformation.getTransformationType()));
339 
340                     // If all transformations are removed, set the value to null
341                     if (columnTransformations.isEmpty()) {
342                         transformationsEntry.setValue(null);
343                     }
344                 }
345             }
346         }
347 
348         // Remove entries with null values (columns with no transformations)
349         transformations.entrySet().removeIf(entry -> entry.getValue() == null);
350     }

504         setTableConfigFromLoadType(tableConfig, loadType);
505 
506         // preprocess for syncTable
507         if (!previewMode) {
508             preprocessForSyncData(spec, actionStatus, tgt, endpointService, tableConfig,
509                     inputToSyncData, nameHandler);
510         }
511 
512         // Get org name from path.
513         String orgName = spec.getPath().split("/")[0];
514 
515         if (!previewMode && tableConfig.isEnableErrorTable()) {
516             createErrorTable(spec, tgt, tableConfig, inputToSyncData, tgtTableDef, nameHandler);
517             createErrorIdTable(spec, tgt, tableConfig, inputToSyncData, loadType, nameHandler);
518         }
519 
520         if (previewMode) {
521             Map<String, Object> previewDataRes;
522             switch (jobType) {
523                 case PREVIEW_DATA:
524                     previewDataRes = previewData(tableConfig, src, tgt, inputToSyncData,
525                             spec.getPath(), orgName, nameHandler, spec.getIntegrationName());
526                     break;
527                 default:
528                     throw new UnsupportedOperationException(
529                             String.format("Preview job type not implemented: %s", jobType));
530             }
531             currentTableName = tableName;
532 
533             LOGGER.debug("AutoSyncExecutionJobHandler: preview complete for {}",
534                     tableConfig.getTableName());
535 
536             updateStatus((String) previewDataRes.get("status"));
537 
538             outputViews.write(documentUtility.newDocument(Map.of(tableName, previewDataRes)));
539         } else {
540 
541             Map<String, Object> syncDataRes = syncData(tableConfig, src, tgt,
542                     inputToSyncData, spec.getPath(), orgName, nameHandler,
543                     spec.getIntegrationName());
544 
545             Optional<String> latestModRecOptional = getLatestModifiedRecord(spec, inputToSyncData,
546                     tgt, tableConfig, nameHandler);
547 
548             this.tableViewStats = getTableStatsOnSyncDataCompletion(src, syncDataRes,
549                     latestModRecOptional.orElse(null), tableConfig.getLoadMethod());
550             jobStatusHandler.updateStatsAfterSyncDataCompletion(targetTableName, tableViewStats);
551             // later development - once we give an option to the user
552             // to have emailing at the table level per integration job
553             //emailUpdate("per_table_email");
554             LOGGER.debug("AutoSyncExecutionJobHandler: completed processTable for {}",
555                     tableConfig.getTargetTableName());
556 
557             if (tableConfig.isEnableErrorTable()) {
558                 dropErrorIdTable(spec, tgt, tableConfig, inputToSyncData, nameHandler);
559             }
560 
561             updateStatus((String) syncDataRes.get("status"));
562         }
563     }

739     /**
740      * Returns a {@link Duration} of delay calculated by the index of the given table<br/>
741      * - 1st table will have no delay.<br/>
742      * - 2nd to the {@value #POOL_SIZE}th will be equal to its index times {@value #DELAY}.<br/>
743      * - the rest will be {@value #DELAY} seconds.
744      */
745     public static Duration calculateDelay(int tableIdx) {

747             return Duration.ofSeconds(tableIdx * DELAY);

749         return Duration.ofSeconds(DELAY);

752     /**
753      * Add a delay before each execution to avoid the same plex scheduling issue.
754      *
755      * @param tableConfig {@link TableConfig} instance
756      * @param tableIdx index of the table
757      */
758     private void delay(TableConfig tableConfig, int tableIdx) {
759         String table = tableConfig.getTargetTableName();
760         Duration delay = calculateDelay(tableIdx);
761         if (delay.isZero()) {
762             return;
763         }
764         try {
765             Thread.sleep(delay.toMillis());
766         } catch (InterruptedException e) {
767             throw new ExecutionException(e, "Interrupted before loading {}", table);
768         }
769         LOGGER.debug("Table {}-{} started after {}s delay", tableIdx, table, delay.getSeconds());
770     }

772     private LoadType getLoadMethod(AutoSyncSpec spec, boolean isTgtTableExists,
773                                    List<Map<String, Object>> inputToSyncData,
774                                    Endpoint tgt,
775                                    TableConfig tableConfig,
776                                    NameHandler nameHandler) {
777         if (!isTgtTableExists) {
778             // target table does not exist, so do initial sync always irrespective of
779             // requested load method
780             return RELOAD;
781         }
782         if (tableConfig.getPrimaryKey() == null || tableConfig.getPrimaryKey().isEmpty()) {
783             return RELOAD;
784         }
785 
786         // target table exists, still if last modified record is not present, do
787         // initial sync always irrespective of requested load method
788         return getLatestModifiedRecord(spec, inputToSyncData, tgt, tableConfig, nameHandler)
789                 // we don't need to verify that requested load method is supported or not,
790                 // as UI wouldn't have given an option for user to select unsupported load method
791                 .map(latestModifiedRecord -> {
792                     tableConfig.setLastModifiedRecord(latestModifiedRecord);
793                     return fromString(tableConfig.getLoadMethod());
794                 })
795                 .orElse(RELOAD);
796     }

990     private void emailUpdate(List<AutoSyncSpec> specs, String message) {
991         LOGGER.info("Email update: {}", message);
992 
993         //extract org name from path
994         String org = getOrgName(specs.get(0).getPath());
995 
996         //get the integration job name
997         String pipelineName = "AutoSync " + assetDTO.getName();
998 
999         // set user notification setting options at every level of the job integration (GET req)
1000         boolean start = false;
1001         boolean end = false;
1002         boolean error = false;
1003 
1004         try {
1005             if (this.autosyncUserSettingsMap != null) {
1006                 start = getUserNotificationValue(this.autosyncUserSettingsMap, STATUS_START, org);
1007                 end = getUserNotificationValue(this.autosyncUserSettingsMap, STATUS_END, org);
1008                 error = getUserNotificationValue(this.autosyncUserSettingsMap, STATUS_ERROR, org);
1009             } else {
1010                 // Handle the case when autosyncUserSettingsMap is null
1011                 LOGGER.error("JCC version not updated");
1012             }
1013         } catch (Exception e) {
1014             // Handle any other exception that may occur during the method call
1015             LOGGER.error("Error occurred while processing user notification settings: "
1016                     + e.getMessage());
1017         }
1018 
1019 
1020         // for future development -- email notification per table in an integration job
1021         boolean breakdown = false;
1022 
1023         //get the owner of the integration job to receive the email notification
1024         List<String> toEmail = new ArrayList<>();
1025         toEmail.add(this.assetDTO.getOwner());
1026 
1027         //create an instance of the payload creator class for POST email service request
1028         AutoSyncMapTemplatePlaceholders payload = new AutoSyncMapTemplatePlaceholders();
1029 
1030         //create an instance of the html Java body email class
1031         AutoSyncEmailBody emailBody = new AutoSyncEmailBody();
1032 
1033         switch (message) {
1034 
1035             case STATUS_START:
1036 
1037                 if (!start) {
1038                     //return in case the user set false to get "start" email notification
1039                     return;
1040                 }
1041 
1042                 // Pass the data list to the Jinja2 template for rendering
1043                 StringBuilder templateStarts = new StringBuilder();
1044                 for (AutoSyncSpec spec: specs) {
1045                     Map<String, Object> templateMap =
1046                             payload.startTemplate(spec, assetDTO);
1047 
1048                     if (templateStarts.length() == 0) {
1049                         String templateStart = emailBody.startEmailBody(templateMap, true);
1050                         templateStarts = new StringBuilder(templateStart);
1051                     } else {
1052                         String templateStart = emailBody.startEmailBody(templateMap, false);
1053                         templateStarts.append("<br>").append(templateStart);
1054                     }
1055                 }
1056 
1057                 Map<String, Object> bodyMap =
1058                         createEmailData(templateStarts.toString(), pipelineName, EMAIL_START,
1059                                 toEmail);
1060 
1061                 try {
1062                     if (this.autosyncAssetFacade != null &&
1063                             hasCreateEmailRequestMethod(this.autosyncAssetFacade)) {
1064                         this.autosyncAssetFacade.createEmailRequest(
1065                                 this.assetDTO.getOrgSnodeId(), bodyMap);
1066                         LOGGER.info("Email notification sent: start integration job");
1067                     } else {
1068                         // Handle the case when the method does not exist
1069                         LOGGER.error("Create email request method unavailable.");
1070                     }
1071                 } catch (Exception e) {
1072                     LOGGER.debug("Email notification failed: " + e);
1073                 }
1074 
1075                 break;
1076 
1077             case STATUS_BREAKDOWN: //for future development of the email notification feature
1078 
1079                 if (!breakdown) {
1080                     return;
1081                 }
1082                 break;
1083 
1084             case STATUS_COMPLETE:
1085 
1086                 if (!end) {
1087                     return;
1088                 }
1089                 if (jobStatusHandler.hasErrors()) {
1090                     return;
