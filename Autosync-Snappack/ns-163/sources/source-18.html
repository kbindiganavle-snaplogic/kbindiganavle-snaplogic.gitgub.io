330     protected String formatUpdateScript(String filename, String logTableName, String
331                 errorTableName1, String errorTableName2, String workTable, DatabaseConfig
332                 databaseConfig, List<Pair<String, String>> inputViewFields, List<String>
333                 whereFileds, List<String> primaryColumns, boolean upsert, boolean isWinPlex) {
334         // iterate the table fields and prepare the define fields, insert fields and value fields
335         StringBuilder defineFields = new StringBuilder();
336         StringBuilder insertFields = new StringBuilder();
337         StringBuilder valueFields = new StringBuilder();
338         StringBuilder setFields = new StringBuilder();
339 
340         String fieldName;
341         for (int i = 0; i < inputViewFields.size(); ++i) {
342             fieldName = inputViewFields.get(i).getKey();
343             defineFields.append(String.format(FORMAT_DEFINE_FIELD, fieldName));
344             insertFields.append(String.format(FORMAT_INSERT_FIELD, fieldName));
345             valueFields.append(String.format(FORMAT_VALUE_FIELD, fieldName));
346             if (!primaryColumns.contains(fieldName) && !whereFileds.contains(fieldName)) {
347                 setFields.append(String.format(FORMAT_SET_FIELD, fieldName, fieldName));
348             }
349             if (i < (inputViewFields.size() - 1)) {
350                 defineFields.append(COMMA);
351                 insertFields.append(COMMA_SPACE);
352                 valueFields.append(COMMA_SPACE);
353                 if (!primaryColumns.contains(fieldName) && !whereFileds.contains(fieldName)) {
354                     setFields.append(COMMA_SPACE);
355                 }
356 
357             }
358         }
359         // get db host name
360         String url = databaseConfig.getJdbcUrl();
361         String host = url;
362         int hostStartPos = url.indexOf(JDBC_URL_HOST_PREFIX) + JDBC_URL_HOST_PREFIX.length();
363         if (hostStartPos > 0) {
364             int hostEndPos = url.indexOf(SLASH, hostStartPos);
365             if (hostEndPos > hostStartPos) {
366                 host = url.substring(hostStartPos, hostEndPos);
367             }
368         }
369         host = doEscape(host, isWinPlex);
370         // prepare username and password
371         String username = doEscape(databaseConfig.getUsername(), isWinPlex);
372         String password = doEscape(databaseConfig.getPassword(), isWinPlex);
373         //Construct the where clause from the whereCloumn list
374         StringBuilder whereClause = new StringBuilder();
375         String where;
376         if (whereFileds != null && whereFileds.size() > 0) {
377             for (int i = 0; i < whereFileds.size(); i++) {
378                 whereClause.append(String.format(FORMAT_WHERE_FIELD, whereFileds.get(i),
379                         whereFileds.get(i)));
380                 if (i < (whereFileds.size() - 1)) {
381                     whereClause.append(SPACE_AND_SPACE);
382                 }
383             }
384             where = whereClause.toString();
385         } else {
386             // if no whereclause update all the fields
387             where = setFields.toString().replace(COMMA_SPACE, SPACE_AND_SPACE);
388         }
389         //Upsert
390         if (upsert) {
391             return genUpsertScript(filename, logTableName, errorTableName1, errorTableName2,
392                     workTable, defineFields, setFields, host, username, password, where,
393                     insertFields, valueFields);
394         } else {
395             return genUpdateScript(filename, logTableName, errorTableName1, errorTableName2,
396                     workTable, defineFields, setFields, host, username, password, where);
397         }
398     }

400     private String genUpdateScript(final String filename, final String logTableName, final String
401                 errorTableName1, final String errorTableName2, final String workTable, final
402                 StringBuilder defineFields, final StringBuilder setFields, final String host, final

408                 errorTableName2, dbTableName, dbTableName, logTableName,

414                 errorTableName2, dbTableName, dbTableName, logTableName,
