240     private void lookForFastLoad() {
241         // Look for fastload executable in PATH if not defined.
242         Runtime runtime = Runtime.getRuntime();
243         String findPathcommand = isWinPlex ? FIND_FASTLOAD_COMMAND_WIN
244                 : FIND_FASTLOAD_COMMAND_LINUX;
245         BufferedReader reader = null;
246         StringBuilder output = new StringBuilder();
247         try {
248             Process process = runtime.exec(findPathcommand);
249             reader = new BufferedReader(new InputStreamReader(process.getInputStream()));
250             String line;
251             while ((line = reader.readLine()) != null) {
252                 output.append(line);
253             }
254             fastLoadPath = StringUtils.strip(output.toString());
255             if (StringUtils.isBlank(fastLoadPath)
256                     || StringUtils.containsIgnoreCase(fastLoadPath, NOT_FOUND)
257                     || StringUtils.containsIgnoreCase(fastLoadPath, COULD_NOT_FIND_FILES)) {
258                 throw new ConfigurationException(ERR_NO_FASTLOAD_FOUND_MSG)
259                         .withReason(ERR_NO_FASTLOAD_FOUND_REASON)
260                         .withResolution(ERR_NO_FASTLOAD_FOUND_RESOLUTION);

264                     .withReason(ERR_NO_FASTLOAD_FOUND_REASON)
265                     .withResolution(ERR_NO_FASTLOAD_FOUND_RESOLUTION);

439     private void preLoadAction() {
440         // Creating data file
441         String tempDir = System.getProperty(TEMP_DIR_PROPERTY);
442         String randomString = UUID.randomUUID().toString();
443         dataFileName = String.format(DATA_FILE_PATTERN, tempDir,
444                 tempDir.endsWith(File.separator) ? StringUtils.EMPTY : File.separator,
445                 randomString);
446         fastloadScript = tptScript.getFastloadScript(dataFileName, checkPointSize,errorTableName1,
447                 errorTableName2, account.getDatabaseConfig(), inputViewFields, isWinPlex, false);
448         fastloadMaskedScript = tptScript.getFastloadScript(dataFileName, checkPointSize,
449                 errorTableName1, errorTableName2, account.getDatabaseConfig(), inputViewFields,
450                 isWinPlex, true);
451         if (!isWinPlex) {
452             try {
453                 Process process = Runtime.getRuntime()
454                         .exec(String.format(CMD_MKFIFO_FMT, dataFileName));
455                 process.waitFor();
456             } catch (IOException | InterruptedException e) {
457                 throw new ExecutionException(e, ERR_CREATE_DATA_FILE_MSG).withReason(e.getMessage())
458                         .withResolution(RESOLUTION_ADDRESS_ISSUE);
459             }
460         }
461 
462         if (!isWinPlex) {
463             // Reader of the named pipe must be created before writer.
464             loadAction();
465         }
466 
467         if (isWinPlex) {
468             try {
469                 dataWriter = new BufferedWriter(new FileWriter(dataFileName));
470             } catch (IOException e) {
471                 throw new ExecutionException(e, ERR_CREATE_DATA_FILE_MSG)
472                         .withReason(e.getMessage())
473                         .withResolution(RESOLUTION_ADDRESS_ISSUE);
474             }
475         } else {
476             // The call to open a writer of a named pipe will block the thread until the named pipe
477             // has a reader, so we do it in a separate thread, give fastload time to initialize and
478             // close the thread when thing goes wrong.
479             // After initiating the fastload command in a separate process, wait for a given time
480             // and
481             // verify the status, if fastload could not be invoked or dataWriter can't be initiated
482             // in the given timeout, set the loader state to failed.
483 
484             Thread waitThread = new Thread(new Runnable() {
485                 @Override
486                 public void run() {
487                     try {
488                         dataWriter = new BufferedWriter(new FileWriter(dataFileName));
489                     } catch (IOException e) {
490                         throw new ExecutionException(e, ERR_CREATE_DATA_FILE_MSG)
491                                 .withReason(e.getMessage())
492                                 .withResolution(RESOLUTION_ADDRESS_ISSUE);
493                     }
494                 }
495             });
496             waitThread.start();
497             long waitTime = 0;
498             while (waitTime < TIMEOUT_FOR_OPENING_WRITTER_IN_MS) {
499                 try {
500                     Thread.sleep(INTERVAL_IN_MS);
501                     try {
502                         loaderProcess.exitValue();
503                         // Loader didn't run successfully, stop running
504                         loaderState = LoaderState.LOADER_STATE_FAILED;
505                         break;
506                     } catch (IllegalThreadStateException e) {
507                         // No op
508                     }
509                     if (!waitThread.isAlive()) {
510                         break;
511                     }
512                     waitTime += INTERVAL_IN_MS;
513                 } catch (InterruptedException e) {
514                     e.printStackTrace();
515                 }
516             }
517             if (waitTime >= TIMEOUT_FOR_OPENING_WRITTER_IN_MS) {
518                 // Timeout exceeded , stop the snap
519                 waitThread.interrupt();
520                 if (loaderState != LoaderState.LOADER_STATE_FAILED) {
521                     isSuccessful = false;
522                 }
523             } else {
524                 if (loaderState == LoaderState.LOADER_STATE_DEFAULT) {
525                     // success
526                     loaderState = LoaderState.LOADER_STATE_STARTED;
527                 } else {
528                     // Loader failed initializing
529                     waitThread.interrupt();
530                 }
531             }
532         }
533     }

612     private void loadAction() {
613         try {
614             Runtime runtime = Runtime.getRuntime();
615             loaderProcess = runtime.exec(fastLoadPath);
616             if (isWinPlex) {
617                 loaderState = LoaderState.LOADER_STATE_STARTED;
618             }
619 
620             if (loaderProcess != null) {
621                 // Runtime.exec() has limited buffer size for stdin and stdout streams, and
622                 // the external process will get stuck when the buffer is full, so we keep
623                 // cleaning up the stdout buffer to keep fastload from hanging.
624                 new Thread(new Runnable() {
625                     @Override
626                     public void run() {
627                         BufferedReader reader;
628                         try {
629                             reader = new BufferedReader(
630                                     new InputStreamReader(loaderProcess.getInputStream()));
631                             String line;
632                             while ((line = reader.readLine()) != null) {
633                                 LOGGER.debug(FASTLOAD_OUTPUT_FMT, line);
634                                 output.append(line).append(LINE_BREAK);
635                             }
636                             reader.close();
637                         } catch (IOException ex) {
638                             // no op
639                         }
640                     }
641                 }).start();
642 
643                 PrintWriter procOutput = new PrintWriter(loaderProcess.getOutputStream());
644                 procOutput.write(fastloadScript);
645                 procOutput.flush();
646             }
647         } catch (Exception e5) {
648             SnapDataException dataException = new SnapDataException(e5,
649                     ERR_EXECUTE_FASTLOAD_CMD_MSG).withReason(e5.getMessage())
650                     .withResolution(ERR_EXECUTE_FASTLOAD_CMD_RESOLUTION);
651             errorViews.write(dataException, errorDoc);
652         }

680     @Override
681     public void execute() {
682         if (inputViews.isEmpty()) {
683             return;
684         }
685 
686         for (String viewName : getInputViewOrder()) {
687             for (Document document : inputViews.documents(inputViews.get(viewName))) {
688                 process(document, viewName);
689             }
690         }
691 
692         if (!firstRecord && autoCreateTable && schema != null) {
693             // We want to create the table even if there are no rows to be loaded.
694             createTableIfAbsent(null);
695         }
696 
697         if (isWinPlex && isSuccessful && dataWriter != null) {
698             loadAction();
699         }
700 
701         // Wait for fastload to finish if loading was started successfully.
702         if (loaderState == LoaderState.LOADER_STATE_STARTED) {
703             final CountDownLatch latch = new CountDownLatch(1);
704             Thread waitThread = new Thread(new Runnable() {
705                 @Override
706                 public void run() {
707                     try {
708                         loaderProcess.waitFor();
709                     } catch (InterruptedException e) {
710                         // ignore
711                     } finally {
712                         latch.countDown();
713                     }
714                 }
715             });
716             waitThread.start();
717             if (!isWinPlex) {
718                 // Close writer of the named pipe so that fastload won't wait for input forever.
719                 IOUtils.closeQuietly(dataWriter);
720             }
721             try {
722                 latch.await();
723                 loaderState = LoaderState.LOADER_STATE_SUCCEEDED;
724             } catch (InterruptedException e) {
725                 isSuccessful = false;
726             }
727         }
728     }
