117     public static TreeSet<String> getVersions() {
118         LocalDate curdate = LocalDate.now();
119         //Findout current shopify API release quarter
120         NumberFormat decimalFormat = new DecimalFormat("00");
121         String curQtr =
122                 decimalFormat.format(curdate.getMonth().firstMonthOfQuarter().getValue());
123         int[] years = {curdate.getYear() - 1, curdate.getYear()};
124         TreeSet<String> versions = new TreeSet<>();
125         for (String month : RELEASE_QTRS) {
126             for (int year : years) {
127                 //Finding which versions are supported from last year release
128                 if (year == curdate.getYear() - 1 && Integer.parseInt(month) >= Integer.parseInt(curQtr)) {
129                     versions.add(String.format(VERSION_PATTERN, year, month));
130                     //Finding which versions are release for current year
131                 } else if (year == curdate.getYear() && Integer.parseInt(month) <= Integer.parseInt(curQtr)) {
132                     versions.add(String.format(VERSION_PATTERN, year, month));
133                 }
134             }
135         }
136         return versions;
137     }

264     static String getNextPageURL(String link) {
265         //Example Link header value: "<https://{shop}.myshopify.com/admin/api/{version}/products
266         // .json?page_info={page_info}&limit={limit}>; rel={next}, <https://{shop}.myshopify
267         // .com/admin/api/{version}/products.json?page_info={page_info}&limit={limit}>;
268         // rel={previous}"
269         String[] pages = link.split(COMMA);
270         for (String token : pages) {
271             String[] tokens = token.split(COLON);
272             if (tokens[1].endsWith(NEXT_PAGE_TAG)) {
273                 String pageURL = tokens[0].trim();
274                 return pageURL.substring(1, pageURL.length() - 1);
275             }
276         }
277         return null;
278     }

313         private void writeToOutput(List<Map<String, Object>> recordsList, Document document) {
314             for (Map<String, Object> result : recordsList) {
315                 outputViews.write(documentUtility.newDocumentFor(document, result),
316                         document);
317             }
318         }
319 
320         @Override
321         protected void writeToOutput(Object content, Document document,
322                 Map<String, Object> statusLine, Map<String, Object> headersResult) {
323             if (getType() == HttpGet.METHOD_NAME && headersResult.containsKey(LINK)) {
324                 nextPage = getNextPageURL((String) headersResult.get(LINK));
325             }
326             if (content instanceof List) {
327                 List<Map<String, Object>> recordsList = (List<Map<String, Object>>) content;
328                 writeToOutput(recordsList, document);
329             } else if (content instanceof Map) {
330                 Map<String, Object> resultSet = (Map<String, Object>) content;
331                 if (resultSet.containsKey(objectName)) {
332                     //Executes for Get with query resources
333                     List<Map<String, Object>> recordsList =
334                             (List<Map<String, Object>>) resultSet.get(objectName);
335                     writeToOutput(recordsList, document);
336                 } else if (resultSet.isEmpty()) {
337                     //Executes for delete operations
338                     Map<String, Object> data = new LinkedHashMap<>();
339                     data.put(STATUS, SUCCESS);
340                     outputViews.write(documentUtility.newDocumentFor(document, data),
341                             document);
342                 } else {
343                     //Executes for Get by resource id, create/update resource
344                     resultSet.forEach((key, value) -> {
345                         if (value instanceof Map) {
346                             outputViews.write(documentUtility.newDocumentFor(document,
347                                     value), document);
348                         }
349                     });
350                 }
351             } else {
352                 //Executes when we get HTML web page error
353                 Map<String, Object> data = new LinkedHashMap<>();
354                 data.put(CONTENT, content);
355                 outputViews.write(documentUtility.newDocumentFor(document, data),
356                         document);
357             }
358         }
359     }
