301             public Socket connectSocket(int connectTimeout, Socket socket, HttpHost host,
302                     InetSocketAddress remoteAddress, InetSocketAddress localAddress,
303                     HttpContext context) throws IOException {
304                 if (socket instanceof SSLSocket) {
305                     /*
306                     This adds SNI support for Oracle Java 7, by using Commons BeanUtils to invoke
307                      an Oracle private method via reflection
308                      https://wiki.apache.org/HttpComponents/SNISupport
309                      */
310                     try {
311                         PropertyUtils.setProperty(socket, "host", host.getHostName());
312                     } catch (IllegalAccessException | NoSuchMethodException |
313                             InvocationTargetException e) {
314                         LOG.debug(String.format(ERR_UNABLE_SET_HOST_ON_SOCKET_SNI,
315                                 host.getHostName(), e.getLocalizedMessage()));
316                     }
317 
318                     /*
319                     Java 7's Cryptography Extension (JCE) restricts Diffie-Hellman prime sizes to a
320                      maximum of 1024 bits. Java 7 also changed the default order of cipher suites
321                      the client presents to the server during the ClientHello phase of the SSL
322                      handshake.
323 
324                      The following disables cipher suites that use Ephemeral Diffie-Hellman (DHE),
325                      since those appear affected by the above JCE limitation. We are therefore
326                      relying on target servers supporting alternative algorithms (which they almost
327                      always do).
328 
329                      This could be re-visited when/if we require at least Java 8.
330 
331                      http://bugs.java.com/bugdatabase/view_bug.do?bug_id=6521495
332                      http://bugs.java.com/bugdatabase/view_bug.do?bug_id=7044060
333                      http://stackoverflow.com/a/6862383
334                      http://stackoverflow.com/q/10687200
335                      */
336                     List<String> limitedCipherSuites = new LinkedList<String>();
337                     for (String cipherSuite : ((SSLSocket) socket).getEnabledCipherSuites()) {
338                         if (!cipherSuite.contains("_DHE_")) {
339                             limitedCipherSuites.add(cipherSuite);
340                         }
341                     }
342                     ((SSLSocket) socket).setEnabledCipherSuites(
343                             limitedCipherSuites.toArray(new String[limitedCipherSuites.size()]));
344                 }
345 
346                 return super.connectSocket(connectTimeout, socket, host, remoteAddress,
347                         localAddress, context);

352     private void initConnManagerClient(ProxyConnectionManager connManager,
353             ProxyHttpClient client) {
354         // Determines the maximum connections per route.
355         connManager.setDefaultMaxPerRoute(maxPerRoute > 0 ? maxPerRoute :
356                 HTTP_DEFAULT_MAX_PER_ROUTE);
357         // Determines the maximum total connections.
358         connManager.setMaxTotal(maxConnections > 0 ? maxConnections : HTTP_MAX_CONN);
359         // Set Proxy credentials for http and https as well as http/socket timeouts
360         ProxyHttpClient.initializeConnection(client, HTTP_CONNECTION_TIMEOUT_SEC,
361                 HTTP_SOCKET_TIMEOUT_SEC);
362         // The SystemDefaultRoutePlanner with a null ProxySelector will default to a ProxySelector
363         // that leverages provided System properties in order to support nonProxyHosts capabilities
364         client.setRoutePlanner(new SystemDefaultRoutePlanner(null));
365         ScheduledFuture futute = execService.scheduleAtFixedRate(() -> {
366             /**
367              * One of the major shortcomings of the classic blocking I/O model is that the
368              * network socket can react to I/O events only when blocked in an I/O operation.
369              * When a connection is released back to the manager, it can be kept alive however
370              * it is unable to monitor the status of the socket and react to any I/O events.
371              * If the connection gets closed on the server side, the client side connection is
372              * unable to detect the change in the connection state (and react appropriately by
373              * closing the socket on its end).
374              *
375              * HttpClient tries to mitigate the problem by testing whether the connection is
376              * 'stale',
377              * that is no longer valid because it was closed on the server side, prior to using the
378              * connection for executing an HTTP request. The stale connection check is not 100%
379              * reliable.
380              * The only feasible solution that does not involve a one thread per socket model for
381              * idle connections is a dedicated monitor thread used to evict connections that are
382              * considered expired due to a long period of inactivity. The monitor thread can
383              * periodically
384              * call ClientConnectionManager#closeExpiredConnections() method to close all expired
385              * connections and evict closed connections from the pool. It can also optionally call
386              * ClientConnectionManager#closeIdleConnections() method to close all connections that
387              * have been idle over a given period of time.
388              */
389             checkStaleConnections(connManager);
390         }, 30, 30, TimeUnit.SECONDS);
391         futureList.add(futute);
392     }

441      * Executes http request. This is a method which physically accesses REST API service and may be
442      * overridden for unit tests to get a mocked HttpResponse object.
443      *
444      * @param httpUriRequest      - HttpUriRequest object
445      * @param trustAllCerts       - true if all certificates are trusted
446      * @param socketTimeoutInSecs - the read/socket timeout to wait before aborting the request
447      * @param connTimeoutInSecs   - the connection timeout to wait before aborting the request
448      * @return HttpResponse object
449      * @throws IOException if an error occurs while executing request
450      */
451     public HttpResponse executeRequest(HttpUriRequest httpUriRequest, boolean trustAllCerts,
452                                        int socketTimeoutInSecs, int connTimeoutInSecs,
453                                        boolean followRedirects)
454             throws IOException {
455         if (client == null || clientTrustAll == null) {
456             initHttpClient();
457         }
458         String scheme = httpUriRequest.getURI().getScheme();
459         if (StringUtils.isBlank(scheme)) {
460             throw new IOException(ERR_URL_SCHEME);
461         }
462         ProxyHttpClient requestClient = getRestClient(scheme.equals(SCHEME_HTTPS) &&
463                 trustAllCerts);
464 
465         setCookieSpecs(cookiePolicy, requestClient);
466 
467         HttpParams params = requestClient.getParams();
468         params
469                 .setParameter(ClientPNames.COOKIE_POLICY, cookiePolicy)
470                 .setParameter(ClientPNames.HANDLE_REDIRECTS, followRedirects);
471 
472         /**
473          * As a standard mostly no website can have more than 3 redirects.
474          *
475          * Limitations observed:
476          * More than 5 redirects in a chain (HTTP/1.0 Browser Limits)
477          * More than 3 redirects in a chain (Google Bot Limits)
478          * Long redirect chains with useless hubs (Bad Page Speed and Bad User Experience)
479          *
480          * Observer some serverside redirects which are implemented with 4 redirects. So to avoid
481          * such cases adding default MAX_REDIRECTS as 15 which is 5 multiple of 3.
482          *
483          * TODO: Has to implement table property to provide a provision to user to set custom
484          * values for HTTP params.
485          */
486 
487         params
488                 .setParameter(ClientPNames.MAX_REDIRECTS, DEFAULT_MAX_REDIRECTS)
489                 .setIntParameter(CoreConnectionPNames.SO_TIMEOUT, socketTimeoutInSecs * 1000)
490                 .setIntParameter(CoreConnectionPNames.CONNECTION_TIMEOUT,
491                         connTimeoutInSecs * 1000);
492         // adding any additional params to existing list,
493         // overrides old value, added to support circular redirects in salesforce
494         if (additionalParams != null) {
495             additionalParams.forEach(params::setParameter);
496         }
497         // overriding params with new one.
498         requestClient.setParams(params);
499         if (LOG.isDebugEnabled()) {
500             log(httpUriRequest);
501         }
502         return executeUriRequest(requestClient, httpUriRequest);
503     }

539     private void initPreemptiveProxyAuthentication() {
540         // Check if we have an authenticated proxy (this should probably be an easy getter in
541         // the ProxyHttpClient itself, or fully implemented in the respective execute method(s)
542         if ((System.getProperty(Constants.HTTP_PROXY_USER) != null &&
543                 System.getProperty(Constants.HTTP_PROXY_PASSWORD) != null &&
544                 System.getProperty(Constants.HTTP_PROXY_HOST) != null) ||
545                 (System.getProperty(Constants.HTTPS_PROXY_USER) != null &&
546                         System.getProperty(Constants.HTTPS_PROXY_PASSWORD) != null &&
547                         System.getProperty(Constants.HTTPS_PROXY_HOST) != null)) {
548             HttpClientContext preemptiveAuthHttpContext = this.httpContext != null ?
549                     HttpClientContext.adapt(this.httpContext) : HttpClientContext.create();
550             String httpHost = System.getProperty(Constants.HTTP_PROXY_HOST);
551             String httpPort = System.getProperty(Constants.HTTP_PROXY_PORT);
552             String httpsHost = System.getProperty(Constants.HTTPS_PROXY_HOST);
553             String httpsPort = System.getProperty(Constants.HTTPS_PROXY_PORT);
554 
555             BasicScheme proxyAuth = new BasicScheme();
556             try {
557                 proxyAuth.processChallenge(new BasicHeader(AUTH.PROXY_AUTH, "BASIC realm=default"));
558             } catch (MalformedChallengeException e) {
559                 LOG.warn("Unable to set challenge for proxy authentication: {}", e.getMessage());
560             }
561             BasicAuthCache authCache = new BasicAuthCache();
562             if (httpHost != null && httpPort != null) {
563                 authCache.put(new HttpHost(httpHost, Integer.parseInt(httpPort)), proxyAuth);
564             }
565             if (httpsHost != null && httpsPort != null) {
566                 authCache.put(new HttpHost(httpsHost, Integer.parseInt(httpsPort)), proxyAuth);
567             }
568             preemptiveAuthHttpContext.setAuthCache(authCache);
569 
570             this.httpContext = preemptiveAuthHttpContext;
571         }
572     }

591         if (httpUriRequest == null || !LOG.isDebugEnabled()) {

593         }

619     protected String obfuscatePassword(String string) {
620         if (StringUtils.isBlank(string)) {
621             return StringUtils.EMPTY;
622         }
623         int i = string.indexOf(PASSWORD);
624         if (i != -1) {
625             int j = i + PASSWORD.length();
626             int k = string.indexOf("\\\"", j);
627             if (k < 0) {    // if password does not begin with escape quote
628                 k = string.indexOf('&', j);
629                 if (k < 0) {
630                     k = string.indexOf(' ', j);
631                     if (k < 0) {
632                         k = string.indexOf('#', j);
633                         if (k < 0) {
634                             k = string.length();
635                         }
636                     }
637                 }
638             } else {    // Password does begin with esc quote so find the closing esc quote
639                 k = string.indexOf("\\\"", k + 1);
640                 if (k < 0) {    // If we didn't find a closing quote, just mask the end of string
641                     k = string.length();
642                 } else {    // found the closing esc quote so mask that also
643                     k += 2;
644                 }
645             }
646             string = string.replace(string.substring(i, k), PASSWORD_OBFUSCATED);
647         }
648         return string;
649     }

651     private void initHttpClient() {
652         final HttpParams httpParams = new BasicHttpParams();
653         // set to prevent from following a redirect
654         httpParams.setParameter(ClientPNames.HANDLE_REDIRECTS, false);
655         httpParams.setParameter(ClientPNames.MAX_REDIRECTS, 3);
656         httpParams.setParameter(ClientPNames.ALLOW_CIRCULAR_REDIRECTS, false);
657         httpParams.setParameter(ClientPNames.REJECT_RELATIVE_REDIRECT, false);
658 
659         SchemeRegistry schemeRegistry = new SchemeRegistry();
660         // register with a socket factory that supports SNI
661         registerSchemes(schemeRegistry,
662                 SSLSocketFactory.BROWSER_COMPATIBLE_HOSTNAME_VERIFIER, keyManagers, trustManagers,
663                 null, plainSocketFactory, sslSocketFactory);
664         // create a connection manager and default HTTP client
665         connManager = new ProxyConnectionManager(schemeRegistry);
666         client = new ProxyHttpClient(connManager, httpParams);
667         if (customRedirectStrategy != null) {
668             client.setRedirectStrategy(customRedirectStrategy);
669         }
670         LOG.trace("{}.connManager = {}, .client = {}", this, connManager, client);
671         // initialize default connection manager and http client
672         initConnManagerClient(connManager, client);
673 
674         SchemeRegistry schemeRegistryTrustAll = new SchemeRegistry();
675         // register with a socket factory that supports SNI and trusts all certificates
676         registerSchemes(schemeRegistryTrustAll,
677                 SSLSocketFactory.ALLOW_ALL_HOSTNAME_VERIFIER, keyManagers, TM_TRUST_ALL,
678                 new SecureRandom(), plainSocketFactory, trustAllSslSocketFactory);
679 
680         // create a connection manager and HTTP client which trusts all certificates
681         connManagerTrustAll = new ProxyConnectionManager(schemeRegistryTrustAll);
682         clientTrustAll = new ProxyHttpClient(connManagerTrustAll, httpParams);
683         LOG.trace("{}.connManagerTrustAll = {}, .clientTrustAll = {}",
684                 this, connManagerTrustAll, clientTrustAll);
685         // initialize connection manager and http client
686         initConnManagerClient(connManagerTrustAll, clientTrustAll);
687 
688         // Add NTLM and OAuth Schemes
689         client.getAuthSchemes().register(NTLM, new NTLMSchemeFactory());
690         clientTrustAll.getAuthSchemes().register(NTLM, new NTLMSchemeFactory());
691         // register OAuth (1.0a) as an auth scheme for authorization challenges
692         client.getAuthSchemes().register(OAuthSchemeFactory.SCHEME_NAME, new OAuthSchemeFactory());
693         clientTrustAll.getAuthSchemes().register(
694                 OAuthSchemeFactory.SCHEME_NAME, new OAuthSchemeFactory());
695         // NTLM and OAuth 1.0a require this
696         if (authScope != null && credentials != null) {
697             client.getCredentialsProvider().setCredentials(authScope, credentials);
698             clientTrustAll.getCredentialsProvider().setCredentials(authScope, credentials);
699         }
700 
701         // add any request interceptors at their desired indexes
702         if (requestInterceptors != null && !requestInterceptors.isEmpty()) {
703             Set<Map.Entry<Integer, HttpRequestInterceptor>> entrySet =
704                     requestInterceptors.entrySet();
705             for (Map.Entry<Integer, HttpRequestInterceptor> entry : entrySet) {
706                 HttpRequestInterceptor interceptor = entry.getValue();
707                 if (interceptor != null) {
708                     Integer index = entry.getKey();
709                     client.addRequestInterceptor(interceptor, index);
710                     clientTrustAll.addRequestInterceptor(interceptor, index);
711                 }
712             }
713         }
714 
715         // SWAT-1742: Since Jcc starts with 'http.keepAlive=false' after 4.15 release, the reuse
716         // strategy got changed to NoConnectionReuseStrategy in the client. Switch the setting
717         // back to default one to keep the old behavior.
718         if (!(client.getConnectionReuseStrategy() instanceof DefaultConnectionReuseStrategy)) {
719             client.setReuseStrategy(new DefaultConnectionReuseStrategy());
720         }
721         if (!(clientTrustAll.getConnectionReuseStrategy()
722                 instanceof DefaultConnectionReuseStrategy)) {
723             clientTrustAll.setReuseStrategy(new DefaultConnectionReuseStrategy());
724         }
725     }
