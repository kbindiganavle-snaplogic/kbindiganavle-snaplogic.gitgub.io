177      * Executes the request with a current document
178      *
179      * @param document        as the document to use
180      * @param payloadDocument the document to use as the payload for the request
181      * @param urlResolver     as the URL resolver
182      * @param dataHandler     as the handler that processes the incoming document
183      */
184     public Object execute(Document document, Document payloadDocument, URLResolver urlResolver,
185             @Nullable RequestDataHandler<? extends HttpEntity> dataHandler) {
186         Object result = null;
187         SnapDataException ex = null;
188         int retry = 0;
189         while (retry++ < maxRetries) {
190             result = executeOnce(document, payloadDocument, urlResolver, dataHandler);
191             ex = restResponseProcessor.getDownloadException();
192             if (ex == null) {
193                 return result;
194             } else {
195                 try {
196                     TimeUnit.SECONDS.sleep(retryInterval);
197                 } catch (InterruptedException e) {
198                     LOG.error("Interrupted while sleeping, but ignored", e);
199                 }
200                 restResponseProcessor.resetDownloadException();
201                 LOG.info("Failed to download HTTP entity content, retrying...");
202             }
203         }
204         if (ex != null) {
205             errorViews.write(ex, document);
206         }
207         return result;
208     }

210     private Object executeOnce(Document document, Document payloadDocument,
211             URLResolver urlResolver, RequestDataHandler<? extends HttpEntity> dataHandler) {
212         HttpEntity entity = null;
213         HttpResponse httpResponse = null;
214 
215         // Number of request attempts to the service endpoint
216         int retry = 0;
217 
218         // Number of retries for OAuth failures (401/403)
219         int oauthRetry = 0;
220 
221         // Boolean variable indicates whether the account instance is updated.
222         // It will be true only if "retryImmediately()" returns true.
223         // That means there is a newer account instance to be used for retry.
224         boolean isAccountReloaded = false;
225 
226         String url;
227         restResponseProcessor.resetDownloadException();
228         try {
229             url = urlResolver.resolveUrl(document);
230         } catch (SnapDataException ex) {
231             errorViews.write(ex, document);
232             return null;
233         }
234 
235         /*
236          This attempts to execute the request and will retry when either
237          1. 401/403 error, the account instance is updated and does not exceed OAuth retry limit.
238             This is separated from the Snap request attempts setting.
239          2. The error code matches the user select retry policy,
240             and the current request attempts does not exceed the max request attempts setting.
241          */
242         while ((isAccountReloaded && oauthRetry < maxOAuthRetries) || retry < maxRetries) {
243             try {
244                 if (retry + oauthRetry > 0) {
245                     LOG.debug(String.format(EXECUTING_REQUEST, retry + oauthRetry));
246                 }
247                 if (httpResponse != null) {
248                     EntityUtils.consumeQuietly(httpResponse.getEntity());
249                 }
250                 // always reset to false because the next attempt could be a different error
251                 // For example, get 401 -> reload account -> retry -> get 503
252                 isAccountReloaded = false;
253 
254                 // create entity if null or cannot produce its data more than once
255                 if (dataHandler != null && (entity == null || !entity.isRepeatable())) {
256                     entity = dataHandler.createEntity(payloadDocument);
257                 }
258                 // The request() method invocation below will throw exception for connection or
259                 //  network failures.
260                 // If so then retries are handled in catch block.
261                 // Else, if there is any client/server error then isRetryRequested() will decide
262                 //  whether to do retry or not based on user's retry policy
263                 httpResponse = request(url, entity);
264                 if (retryDecider != null && httpResponse != null) {
265                     /*
266                     "retryImmediately" will check if the account is OAuth, the response
267                      status code is 401 or 403, and that the server has a different account
268                      instance FQID than the one the Snap currently has; if so, it will reload the
269                      account instance from the Platform and rebuild the Authorization header.
270                      */
271                     if (oauthRetry < maxOAuthRetries
272                             && retryDecider.retryImmediately(httpResponse)) {
273                         oauthRetry++;
274                         isAccountReloaded = true;
275                         LOG.warn("jrest_retry - retrying request immediately");
276                         continue;
277                     }
278 
279                     /*
280                     Then, if the response is 429 "Too Many Requests", we will try to capture the
281                     value from the 'Retry-After' Header, and retry the request automatically
282                      */
283                     if (httpResponse.getStatusLine() != null
284                             && httpResponse.getStatusLine().getStatusCode() == 429) {
285                         long timeout = getIntervalFromRetryAfter(httpResponse);
286                         LOG.warn("jrest_retry - The endpoint returns {}, will retry after {} " +
287                                 "seconds", httpResponse.getStatusLine().getStatusCode(), timeout);
288                         retry++;
289                         sleepFor(timeout);
290                         continue;
291                     }
292 
293                     /*
294                     Otherwise, if the user decides to retry on 4xx/5xx errors then that is handled
295                     in isRetryRequired() method. It will check if the response code matches
296                     the select retry policy.
297                      */
298                     if (isRetryRequested(httpResponse)) {
299                         retry++;
300                         LOG.warn(String.format("jrest_retry - retrying request after %d seconds",
301                                 retryInterval));
302                         sleepFor(retryInterval);
303                         continue;
304                     }
305                 }
306                 // otherwise give up
307                 break;
308             } catch (ExecutionException e) {
309                 retry++;
310                 // By default, any exception occurred as a result of invocation is retried
311                 // So connection/network errors would be retried here
312                 if (retry == maxRetries) {
313                     SnapDataException ex = new SnapDataException(e, e.getMessage())
314                             .withReason(e.getReason())
315                             .withResolution(e.getResolution());
316                     errorViews.write(ex, document);
317                     return null;
318                 }
319                 sleepFor(retryInterval);
320                 LOG.error(String.format(REQUEST_EXECUTION_FAILED_RETRYING, maxRetries - retry), e);
321             }
322         }
323         return execute(url, httpResponse, document);
324     }

326     /**
327      * Try to get the retry interval from the Retry-After Header.
328      *
329      * If the Retry-After is a number,
330      * we will treat it as the number of seconds.
331      *
332      * Otherwise, it would be a datetime, and we will
333      * calculate the interval accordingly.
334      *
335      * The interval would not be longer than 15 mins
336      *
337      * If the Header is not present, we will follow the retry interval set from the Snap
338      * @param httpResponse
339      * @return
340      */
341     @VisibleForTesting
342     long getIntervalFromRetryAfter(HttpResponse httpResponse) {
343         long timeout = retryInterval;
344         Header retryAfterHeader = httpResponse.getFirstHeader(HttpHeaders.RETRY_AFTER);
345         if (retryAfterHeader != null) {
346             if (StringUtils.isNumeric(retryAfterHeader.getValue())) {
347                 timeout = Long.parseLong(retryAfterHeader.getValue());
348             } else {
349                 Date retryAfter = DateUtils.parseDate(retryAfterHeader.getValue());
350                 if (retryAfter != null) {
351                     timeout = TimeUnit.MILLISECONDS.toSeconds(
352                             retryAfter.getTime() - new Date().getTime());
353                 }
354             }
355             timeout = Math.min(timeout, 5 * 60); // We wait for 5 mins maximum
356         }
357         return timeout;
358     }

360     boolean isRetryRequested(HttpResponse httpResponse) {
361         if (httpResponse != null) {
362             StatusLine statusLine = httpResponse.getStatusLine();
363             if (statusLine != null) {
364                 int statusCode = statusLine.getStatusCode();
365                 return isUsersRetryChoiceMatched(statusCode);
366             }
367         }
368         return false;
369     }

371     private boolean isUsersRetryChoiceMatched(int statusCode) {
372         switch (retryPolicy) {
373             case RETRY_ON_CONNECTION_ERRORS:
374                 // If its a connection error then the code flow wouldn't come to this point.
375                 // Its not a connection error but 4xx or 5xx error but user chose to retry only on
376                 //  connection errors.  So, return false.
377                 return false;
378             case RETRY_ON_CONNECTION_AND_AUTH_ERRORS_401_AND_403:
379                 // User chose to retry if its 401 or 403 error status code
380                 return (statusCode == 401 || statusCode == 403);
381             case RETRY_ON_CONNECTION_AND_EVERY_400_ERROR:
382                 // User chose to retry if its any 4xx error
383                 return (statusCode >= 400 && statusCode < 500);
384             case RETRY_ON_CONNECTION_AND_EVERY_500_ERROR:
385                 // User chose to retry if its any 5xx error
386                 return (statusCode >= 500);
387             case RETRY_ON_CONNECTION_AND_EVERY_400_OR_500_ERROR:
388                 // User chose to retry if its any 4xx or 5xx error
389                 return (statusCode >= 400);
390             default:
391                 // I prefer to throw error in case in future someone just adds a new policy and
392                 //  didn't handle it then it should be brought to light instead of sinking it.
393                 throw new SnapDataException(ERR_UNSUPPORTED_RETRY_OPTION_MSG)
394                         .formatWith(retryPolicy)
395                         .withResolutionAsDefect();
396         }
397     }

429     /**
430      * Executes the request and returns the {@link HttpResponse}
431      *
432      * @param url    the url being executed
433      * @param entity the http entity if any
434      *
435      * @return the http response
436      */
437     protected HttpResponse request(String url, @Nullable HttpEntity entity) {
438         HttpUriRequest uriRequest = null;
439         try {
440             switch (method) {
441                 case HttpGet.METHOD_NAME:
442                     uriRequest = new HttpGet(url);
443                     break;
444                 case HttpPost.METHOD_NAME:
445                     uriRequest = setEntity(new HttpPost(url), entity);
446                     break;
447                 case HttpPut.METHOD_NAME:
448                     uriRequest = setEntity(new HttpPut(url), entity);
449                     break;
450                 case HttpDelete.METHOD_NAME:
451                     if (entity == null) {
452                         uriRequest = new HttpDelete(url);
453                     } else {
454                         uriRequest = new HttpDeleteWithEntity(url);
455                         ((HttpDeleteWithEntity) uriRequest).setEntity(entity);
456                     }
457                     break;
458                 case HttpPatch.METHOD_NAME:
459                     uriRequest = setEntity(new HttpPatch(url), entity);
460                     break;
461                 default:
462                     throw new IOException(String.format(ERR_HTTP_METHOD, method));
463             }
464             if (headers != null) {
465                 uriRequest.setHeaders(headers);
466             }
467 
468             return restHttpClient.executeRequest(uriRequest, trustCerts, readSocketTimeoutInSec,
469                     connTimeoutInSec, followRedirects);
470         } catch (IOException e) {
471             throw new ExecutionException(e, ERR_HTTP_REQUEST)
472                     .withReason(e.getMessage())
473                     .withResolution(ERR_PROPERTIES_RESOLUTION);
474         } catch (Exception e) {
475             if (uriRequest != null) {
476                 uriRequest.abort();
477             }
478             // TODO - MK: move this into SnapServiceImpl L300, we do not log the stacktrace
479             //            during suggest values.
480             LOG.error(REQUEST_EXECUTION_FAILED, e);
481             String reason;
482             try {
483                 String entityContent = StringUtils.EMPTY;
484                 if (entity != null) {
485                     entityContent = EntityUtils.toString(entity);
486                 }
487                 reason = String.format(ERR_HTTP_REQUEST_REASON, method, url,
488                         Arrays.toString(headers), entityContent, e.getMessage());
489             } catch (IOException e1) {
490                 throw new ExecutionException(e, ERR_PROVIDED_HTTP_ENTITY)
491                         .withResolutionAsDefect();
492             }
493             throw new ExecutionException(e, ERR_HTTP_REQUEST)
494                     .withReason(reason)
495                     .withResolution(ERR_PROPERTIES_RESOLUTION);
496         }
497     }
