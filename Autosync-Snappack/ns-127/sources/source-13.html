185     private List<String> suggestRecipientNames() {
186         if (graphClient == null) {
187             prepareGraphClient();
188         }
189         List<String> recipientsList = new ArrayList<>();
190         List<QueryOption> options = new ArrayList<>();
191         options.add(new QueryOption(SELECT, "mail"));
192         UserCollectionRequest request = graphClient.users().buildRequest(options);
193         while (request != null) {
194             UserCollectionPage collection = request.get();
195             List<User> users = collection.getCurrentPage();
196             users.stream().forEach(user -> {
197                 if (user.mail != null) {
198                     recipientsList.add(user.mail);
199                 }
200             });
201             if (collection.getNextPage() != null) {
202                 request = collection.getNextPage().buildRequest(getUserAgentHeader());
203             } else {
204                 request = null;
205             }
206         }
207         GroupCollectionRequest groupRequest = graphClient.groups().buildRequest(options);
208         while (groupRequest != null) {
209             GroupCollectionPage collection = groupRequest.get();
210             List<Group> groups = collection.getCurrentPage();
211             groups.stream().forEach(group -> {
212                 if (group.mail != null) {
213                     recipientsList.add(group.mail);
214                 }
215             });
216             if (collection.getNextPage() != null) {
217                 groupRequest = collection.getNextPage().buildRequest(getUserAgentHeader());
218             } else {
219                 groupRequest = null;
220             }
221         }
222         return recipientsList;
223     }

261     private String encodeShareLink(String url) {
262         String encodedURL = Base64.getUrlEncoder()
263                 .encodeToString(url.getBytes(StandardCharsets.UTF_8));
264         String result = StringUtils.stripEnd(encodedURL, "=").replaceAll("/", "_")
265                 .replaceAll("/+", "-");
266         return "u!" + result;

275     private List<DriveRecipient> evaluateRecipients(Document document) {
276         List<DriveRecipient> recipientsList = new LinkedList<DriveRecipient>();
277         recipientsExpr.stream().forEach(recipient -> {
278             String mail = recipient.eval(document);
279             if (StringUtils.isNotBlank(mail)) {
280                 StringTokenizer mails = new StringTokenizer(mail, ",");
281                 while (mails.hasMoreTokens()) {
282                     DriveRecipient recipients = new DriveRecipient();
283                     recipients.email = mails.nextToken();
284                     recipientsList.add(recipients);
285                 }
286             }
287         });
288         return recipientsList;
289     }

296     private void sendSharingLink(Document document) {
297         Calendar calender = null;
298         if (ANONYMOUS.equals(scope)) {
299             String expires = expireTimeExpr.eval(document);
300             if (StringUtils.isNotBlank(expires)) {
301                 DateTime eventStart = oneDriveCommonUtil.convertDateTimeField(expires);
302                 calender = eventStart.toGregorianCalendar();
303                 calender.setTimeZone(TimeZone.getTimeZone("UTC"));
304             }
305         }
306         String message = messageExpr.eval(document);
307         String role = rolesExpr.eval(document);
308         try {
309             DriveItemRequestBuilder request = buildDriveItemRequestBuilder(document);
310             OffsetDateTime offsetDateTime = calender != null ? calender.toInstant()
311                     .atOffset(ZoneOffset.UTC) : null;
312             DriveItemCreateLinkParameterSet createLinkParameterSet = DriveItemCreateLinkParameterSet.newBuilder()
313                     .withScope(scope)
314                     .withMessage(message)
315                     .withExpirationDateTime(offsetDateTime)
316                     .withType(WRITE.equalsIgnoreCase(role) ? "edit" : "view")
317                     .build();
318             Permission per = request.createLink(createLinkParameterSet)
319                     .buildRequest(getUserAgentHeader()).post();
320             List<DriveRecipient> recipientsList = evaluateRecipients(document);
321             if (CollectionUtils.isEmpty(recipientsList)) {
322                 processResults(per, document);
323             } else {
324                 String shareURL = per.link.webUrl;
325                 LinkedList<String> rolesList = new LinkedList<String>();
326                 rolesList.add(role);
327                 if (!StringUtils.isBlank(shareURL)) {
328                     if (graphClient == null) {
329                         prepareGraphClient();
330                     }
331                     grantAccessToLink(shareURL, rolesList, recipientsList, document);
332                 }
333             }
334         } catch (GraphServiceException ex) {
335             if (refreshToken401Exception(ex) || oneDriveCommonUtil
336                     .handleThrottleError(ex)) {
337                 sendSharingLink(document);
338                 return;
339             }
340             LOGGER.error(ERROR_CREATING_SHARE_LINK, ex);
341             throw new SnapDataException(ex, ERROR_CREATING_SHARE_LINK)
342                     .withReason(String.format(GRAPH_SERVICE_EXCEPTION_REASON,
343                             ex.getResponseCode(), ex.getResponseMessage(),
344                             ex.getServiceError() == null ? "" : ex.getServiceError()
345                                     .message)).withResolution(RESOLUTION_CUSTOMER_SUPPORT);
346         } catch (SnapDataException ex) {
347             throw ex;
348         } catch (Exception ex) {
349             LOGGER.error(ERROR_CREATING_SHARE_LINK, ex);
350             throw new SnapDataException(ex, ERROR_CREATING_SHARE_LINK)
351                     .withReason(ex.getMessage())
352                     .withResolution(RESOLUTION_CUSTOMER_SUPPORT);
353         }
354     }

363             while (requestBuilder != null) {

389         List<DriveRecipient> recipientsList = evaluateRecipients(document);
390         if (recipientsList.isEmpty()) {
391             throw new SnapDataException(ERROR_SENDING_INVITE)
392                     .withReason(RECIPIENT_EMPTY_FOR_INVITE)
393                     .withResolution(RESOLUTION_ADDRESS_ISSUE);
394         }
395         String message = messageExpr.eval(document);
396         String role = rolesExpr.eval(document);
397         if (!requireSignIn && !sendInvite) {
398             throw new SnapDataException(ERROR_SENDING_INVITE)
399                     .withReason(INVITE_OPTIONS_ERROR)
400                     .withResolution(RESOLUTION_ADDRESS_ISSUE);
401         }
402         LinkedList<String> rolesList = new LinkedList<String>();
403         rolesList.add(role);
404         try {
405             DriveItemRequestBuilder request = buildDriveItemRequestBuilder(document);
406             DriveItemInviteCollectionRequestBuilder requestBuilder = request.invite(
407                     DriveItemInviteParameterSet.newBuilder().withRoles(rolesList)
408                             .withRecipients(recipientsList).withRequireSignIn(requireSignIn)
409                             .withMessage(message).withSendInvitation(sendInvite).build());
410             while (requestBuilder != null) {
411                 DriveItemInviteCollectionPage collectionPage = requestBuilder.buildRequest(getUserAgentHeader())
412                         .post();
413                 printResults(collectionPage.getCurrentPage(), document);
414                 requestBuilder = collectionPage.getNextPage();
415             }
416         } catch (GraphServiceException ex) {
417             if (refreshToken401Exception(ex) || oneDriveCommonUtil.handleThrottleError(ex)) {
418                 sendInvitation(document);
419                 return;
420             }
421             LOGGER.error(ERROR_SENDING_INVITE, ex);
422             throw new SnapDataException(ex, ERROR_SENDING_INVITE)
423                     .withReason(String.format(GRAPH_SERVICE_EXCEPTION_REASON, ex.getResponseCode(),

429         } catch (Exception e) {
