269      * Method for copying a Folder or file
270      *
271      * @param document
272      */
273     private void doCopyOperation(Document document) {
274         DriveRequestBuilder request;
275         DriveItemRequestBuilder itemRequestBuilder = null;
276         ItemReference parentReference = new ItemReference();
277         String targetDriveId = null;
278         try {
279             String sourceDriveID = getDriveId(document);
280             if (StringUtils.isBlank(sourceDriveID)) {
281                 if (ONEDRIVE.equalsIgnoreCase(type)) {
282                     request = graphClient.me().drive();
283                 } else {
284                     throw new SnapDataException(COPY_ITEM_ERROR)
285                             .withReason(INVALID_SOURCE_DRIVE)
286                             .withResolution(INVALID_DRIVE_RESOLUTION);
287                 }
288             } else {
289                 request = graphClient.drives(sourceDriveID);
290             }
291             String targetRelativePath = targetRelativePathExp.eval(document);
292             String targetItemName = targetItemNameExp.eval(document);
293             targetDriveId = sourceDriveID;
294             if (!targetSameAsSource) {
295                 targetDriveId = getTargetDriveId(document);
296             }
297             if (!StringUtils.isBlank(targetRelativePath) &&
298                     !StringUtils.startsWith(targetRelativePath, ROOT)) {
299                 parentReference.driveId = targetDriveId;
300                 parentReference.id = targetRelativePath;
301             } else {
302                 String item_id = DRIVE_URL + ROOT;
303                 if (!StringUtils.isBlank(targetDriveId)) {
304                     item_id = DRIVES + targetDriveId + "/" + ROOT;
305                 }
306                 String targetPath = StringUtils.remove(targetRelativePath, ROOT);
307                 if (!StringUtils.isBlank(targetPath)) {
308                     item_id = item_id + targetPath;
309                     parentReference.path = item_id;
310                 } else {
311                     // when relative path is empty, we should find ID of the root element
312                     // to make copy operation work.
313                     DriveRequestBuilder target = null;
314                     if (!StringUtils.isBlank(targetDriveId)) {
315                         parentReference.driveId = targetDriveId;
316                         target = graphClient.drives(targetDriveId);
317                     } else {
318                         throw new SnapDataException(document, COPY_ITEM_ERROR)
319                                 .withReason(INVALID_TARGET_DRIVE)
320                                 .withResolution(INVALID_DRIVE_RESOLUTION);
321                     }
322                     item_id = target.root().buildRequest(ImmutableList.of(
323                             new QueryOption(SELECT, ID))).get().id;
324                     parentReference.id = item_id;
325                 }
326             }
327             itemRequestBuilder =
328                     buildDriveItemRequestBuilder(document, request);
329             DriveItemCopyRequest copyRequest = itemRequestBuilder
330                     .copy(DriveItemCopyParameterSet.newBuilder()
331                             .withParentReference(parentReference).withName(targetItemName).build())
332                     .buildRequest(getUserAgentHeader());
333             DriveItem item = copyRequest.post();
334             processOutput(item, targetDriveId, document, itemRequestBuilder, parentReference,
335                     graphClient.drives(targetDriveId));
336         } catch (GraphServiceException ex) {
337             String code = ex.getServiceError().code;
338             if (refreshToken401Exception(ex) || oneDriveCommonUtil.handleThrottleError(ex)) {
339                 doCopyOperation(document);
340                 return;
341             } else if (NAMEALREADYEXISTS.equals(code)) {
342                 throw new SnapDataException(ex, COPY_ITEM_ERROR)
343                         .withReason(code + " : " + ex.getServiceError().message)
344                         .withResolution(RESOLUTION_ADDRESS_ISSUE);
345             }
346 
347             throw ex;
348         }
349     }

387     /**
388      * Processing the output of copy sourceReq. we will process the output
389      * till we get completed/failed status
390      *
391      * @param item
392      * @param targetDriveID
393      */
394     private void processOutput(DriveItem item, String targetDriveID, Document document,
395             DriveItemRequestBuilder itemRequestBuilder, ItemReference parentRef,
396             DriveRequestBuilder targetReq) {
397         JsonElement jsonElement = item.additionalDataManager().get(GRAPH_RESPONSE_HEADERS);
398         try (RestHttpClient restHttpClient = new RestHttpClient()) {
399             if (jsonElement != null) {
400                 JsonObject obj = jsonElement.getAsJsonObject();
401                 JsonElement responseCodeJson = obj.get(RESPONSE_CODE);
402                 // We can assume 202 safely for most cases without bad side effects. 200 is no
403                 // longer mentioned in the API docs. We might consider removal of the other response
404                 // code blocks.
405                 // See: https://docs.microsoft.com/en-us/onedrive/developer/rest-api/api/driveitem_copy?view=odsp-graph-online
406                 int responseCode = responseCodeJson == null ? 202 : obj.get(RESPONSE_CODE).getAsInt();
407                 if (responseCode == HttpStatus.SC_ACCEPTED) {
408                     Map<String, Object> results = new LinkedHashMap<>();
409                     results = getStatusOfAsynCall(obj.get(LOCATION).getAsString(), results,
410                             restHttpClient);
411                     String status = (String) results.get(STATUS);
412                     if (FAILED.equalsIgnoreCase(status)) {
413                         Map<String, String> error = (Map<String, String>) results.get(ERROR);
414                         try {
415                             if (!NAMEALREADYEXISTS.equals(error.get(CODE))) {
416                                 // cleaning up partially copied item if error is other than
417                                 // "Name already exists"
418                                 DriveItem sourceItem = itemRequestBuilder.buildRequest(getUserAgentHeader()).get();
419                                 String sourceName = sourceItem.name;
420                                 sourceName = removeExtraQuotes(sourceName);
421                                 String targetPath = "";
422                                 String parentPath = parentRef.path;
423                                 if (parentPath != null) {
424                                     parentPath = parentPath.substring(parentPath.indexOf(ROOT));
425                                     if (!parentPath.endsWith("/")) {
426                                         targetPath = parentPath + "/" + sourceName;
427                                     } else {
428                                         targetPath = parentPath + sourceName;
429                                     }
430                                 } else {
431                                     DriveItemCollectionPage children = graphClient.me().drive().root().children()
432                                             .buildRequest(getUserAgentHeader())
433                                             .get();
434                                     List currentPage = children.getCurrentPage();
435                                     Iterator currentPageIterator = currentPage.iterator();
436                                     while (currentPageIterator.hasNext()) {
437                                         JsonObject child = (JsonObject) currentPageIterator.next();
438                                         if (sourceName != null &&
439                                                 sourceName.equals(removeExtraQuotes(child.get(NAME)
440                                                         .toString()))) {
441                                             targetPath = removeExtraQuotes(child.get(ID)
442                                                     .toString());
443                                             break;
444                                         }
445                                     }
446                                 }
447                                 if (!targetPath.isEmpty()) {
448                                     buildDriveItemRequestBuilder(targetReq, targetPath)
449                                             .buildRequest(getUserAgentHeader()).delete();
450                                 }
451                             }
452                         } catch (Exception ex) {
453                             // ignoring if we have encountered any issue in clean up
454                         }
455                         throw new SnapDataException(COPY_ITEM_ERROR)
456                                 .withReason(error.get(CODE) + " : " + error.get(MESSAGE))
457                                 .withResolution(RESOLUTION_ADDRESS_ISSUE);
458                     } else if (COMPLETED.equalsIgnoreCase(status)) {
459                         String resourceID = (String) results.get(RESOURCE_ID);
460                         if (resourceID == null) {
461                             LOGGER.warn("ResourceID is empty. So rechecking for resourceID");
462                             results = getStatusOfAsynCall(obj.get(LOCATION).getAsString(), results,
463                                     restHttpClient);
464                             resourceID = (String) results.get(RESOURCE_ID);
465                         }
466                         getResultItem(targetDriveID, resourceID, document, 0);
467                     } else {
468                         throw new SnapDataException(COPY_ITEM_ERROR)

502         HttpUriRequest httpRequest = new HttpGet(location);
503         try (CloseableHttpResponse httpResponse = (CloseableHttpResponse)restHttpClient.
504                 executeRequest(httpRequest, false,
505                         9000, 480, true)) {
506             StatusLine statusLine = httpResponse.getStatusLine();
507             if (statusLine.getStatusCode() >= HttpStatus.SC_OK &&
508                     statusLine.getStatusCode() <= HttpStatus.SC_ACCEPTED) {
509                 String entityString = EntityUtils.toString(httpResponse.getEntity());
510                 results = OBJECT_MAPPER.readValue(entityString, MAP_TYPE_REF);
511                 String status = (String) results.get(STATUS);
512                 if (INPROGRESS.equalsIgnoreCase(status) || NOT_STARTED.equalsIgnoreCase(status)) {
513                     Thread.sleep(2000);
514                     return getStatusOfAsynCall(location, results, restHttpClient);
515                 } else {
516                     return results;
517                 }
518             }
519         } catch (IOException | InterruptedException e) {
520             throw new SnapDataException(e, COPY_ITEM_ERROR)
521                     .withReason(e.getMessage())
522                     .withResolution(RESOLUTION_CUSTOMER_SUPPORT);
523         }
524         return results;

563     /**
564      * Process the exception based on type
565      *
566      * @param exception
567      * @param targetItemName
568      */
569     private void processException(Exception exception, String targetItemName) {
570         LOGGER.error(COPY_ITEM_ERROR, exception);
571         if (exception instanceof GraphServiceException) {
572             GraphServiceException ex = (GraphServiceException) exception;
573             String resolution = RESOLUTION_CUSTOMER_SUPPORT;
574             String reason = String.format(GRAPH_SERVICE_EXCEPTION_REASON, ex.getResponseCode(),
575                     ex.getResponseMessage(), ex.getServiceError().message);
576             if (ex.getResponseCode() == HttpStatus.SC_NOT_FOUND) {
577                 resolution = ITEM_ACCESS_ERROR_RESOLUTION;
578             } else if (ex.getResponseCode() == HttpStatus.SC_BAD_REQUEST
579                     && !StringUtils.isBlank(targetItemName)) {
580                 reason = ex.getResponseCode() + " : " + ex.getResponseMessage()
581                         + " : " + "Target Item Name is not empty";
582                 resolution = "Target Item Name should be empty."
583                         + RESOLUTION_ADDRESS_ISSUE;
584             }
585             throw new SnapDataException(ex, COPY_ITEM_ERROR)
586                     .withReason(reason)
587                     .withResolution(resolution);
588         } else if (exception instanceof SnapDataException) {
589             throw (SnapDataException) exception;
590         } else {
591             throw new SnapDataException(exception, COPY_ITEM_ERROR)
592                     .withReason(exception.getMessage())
593                     .withResolution(RESOLUTION_CUSTOMER_SUPPORT);
594         }
595     }
