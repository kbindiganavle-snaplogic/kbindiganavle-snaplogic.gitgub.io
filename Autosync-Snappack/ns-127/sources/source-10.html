362             suggestionBuilder.node(nodeName)

441     public String getDriveId(Document document) {
442         try {

444             String driveID = null;
445             if (DRIVE.equalsIgnoreCase(type)) {
446                 if (StringUtils.isBlank(driveName)) {
447                     throw new SnapDataException(FETCH_DRIVES_ERROR)
448                             .withReason(DRIVE_ID_NULL)
449                             .withResolution(DRIVE_ID_NULL_RESOLUTION);
450                 }
451                 driveID = driveName;
452             } else if (ALL.equalsIgnoreCase(type)) {
453                 String userName = userIdExpr == null ? null : userIdExpr.eval(document);
454                 driveID = oneDriveCommonUtil.evaluateFullDriveName(graphClient, grantType,
455                         driveName, userName);
456             } else if (DOCUMENT_LIBRARY.equalsIgnoreCase(type)) {
457                 String group = groupIdExpr == null ? null : groupIdExpr.eval(document);
458                 driveID = getDriveIdNameMap(type, group, "").get(driveName);
459             } else {
460                 String userName = userIdExpr == null ? null : userIdExpr.eval(document);
461                 driveID = getDriveIdNameMap(type, "", userName).get(driveName);
462             }
463             if (StringUtils.isBlank(driveID)) {
464                 throw new SnapDataException(INVALID_SOURCE_DRIVE)
465                         .withReason(INVALID_SOURCE_DRIVE)
466                         .withResolution(INVALID_DRIVE_RESOLUTION);
467             }
468             return driveID;
469         } catch (SnapDataException ex) {
470             if (refreshToken401Exception(ex)) {
471                 return getDriveId(document);
472             }
473             throw ex;
474         }
475     }

630                                                                     String relativePath) {
631         DriveItemRequestBuilder requestBuilder;
632         if (StringUtils.isBlank(relativePath)) {
633             throw new SnapDataException(String.format(PROPERTY_EMPTY_REASON, RELATIVE_PATH_LABEL))
634                     .withReason(String.format(PROPERTY_EMPTY_REASON, RELATIVE_PATH_LABEL))
635                     .withResolution(RELATIVE_PATH_RESOLUTION);
636         }
637         if (!StringUtils.isBlank(relativePath)) {
638             if (StringUtils.startsWith(relativePath, ROOT)) {
639                 relativePath = StringUtils.replace(relativePath, ROOT, "");
640                 if (StringUtils.isBlank(relativePath)) {
641                     requestBuilder = request.root();
642                 } else {
643                     DriveItemRequestBuilder rootBuilder = request.root().itemWithPath(relativePath);
644                     requestBuilder = new DriveItemRequestBuilder(
645                             // The graph SDK seems to incorrectly encode file separators.
646                             rootBuilder.getRequestUrl().replace("%2F", "/"),
647                             rootBuilder.getClient(), null);
648                 }
649             } else {
650                 requestBuilder = request.items(relativePath);
651             }
652         } else {
653             requestBuilder = request.root();
654         }
655         return requestBuilder;
