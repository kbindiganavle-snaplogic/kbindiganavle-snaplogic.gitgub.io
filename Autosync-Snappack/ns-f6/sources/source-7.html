58     @Override
59     protected void processAdditional(Document inDoc, String inView) {
60         // remove existing entry from LDAP server
61         try {
62             if (!ldapConnection.exists(dn)) {
63                 Map<String, Object> errorMap = new LinkedHashMap<>();
64                 errorMap.put(ERROR, LDAP_DN_EXISTS_ERROR_MSG);
65                 SnapDataException exception = new SnapDataException(
66                         documentUtility.newDocument(errorMap), LDAP_DN_EXISTS_ERROR_MSG)
67                         .withReason(LDAP_DN_EXISTS_ERROR_MSG)
68                         .withResolution(LDAP_DN_EXISTS_ERROR_RESOLUTION);
69                 errorViews.write(exception, inDoc);
70                 return;
71             }
72             ldapConnection.delete(dn);
73             Map<String, String> map = new LinkedHashMap<String, String>();
74             map.put(LDAP_OUT_VIEW_FIELD_ENTRY_NAME,
75                     dn);
76             map.put(LDAP_OUT_VIEW_FIELD_DEL_STATUS,
77                     SUCCESS);
78             Document outDoc = documentUtility.newDocumentFor(inDoc, map);
79             outputViews.write(outDoc, passThrough ? inDoc : null);
80         } catch (LdapException e1) {
81             Map<String, Object> errorMap = new LinkedHashMap<>();
82             errorMap.put(ERROR, LDAP_DELETE_ERROR_MSG);
83             SnapDataException exception = new SnapDataException(
84                     documentUtility.newDocument(errorMap), LDAP_DELETE_ERROR_MSG)
85                     .withReason(e1.getMessage())
86                     .withResolution(LDAP_DELETE_ERROR_RESOLUTION);
87             errorViews.write(exception, inDoc);
88         }
89     }
