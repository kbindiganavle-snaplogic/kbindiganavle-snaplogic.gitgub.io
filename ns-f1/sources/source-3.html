347     public Connection connect() throws ExecutionException {
348         // Set the SQL dialect and/or the jdbcOperations(unique db name) so the the Hikari Test
349         // connection string is set correctly.
350         // This is called before the actual connection cause the are connection which were made
351         // to fetch the database name, This has to be checked here as JdbcOperationsImpl class
352         // can invoke acquireConnection() which would require the appropriate JdbcOperation or
353         // Dialect to set the TestConnectionQuery.
354         String jdbcOpName = JdbcUtils.getDbName(this);
355         databaseConfig.setJdbcOperations(jdbcOpName);
356         if (isSshTunnel) {
357             int localPort = FreePortFinder.findFreeLocalPort();
358             String jdbcUrl = databaseConfig.getJdbcUrl();
359             String dbHost = "";
360             int dbPort = 0;
361             try {
362                 URI uri = new URI(jdbcUrl.substring(5));
363                 dbHost = uri.getHost();
364                 dbPort = uri.getPort();
365             } catch (URISyntaxException e) {
366                 e.printStackTrace();
367             }
368             jschConnectionAssistant.createSshTunnel(dbHost,
369                     localPort, urlEncoder, dbPort);
370             databaseConfig.setJdbcUrl(createJdbcUrl(jdbcUrl, LOCALHOST,
371                     localPort));
372         }
373         Connection connection = dataSourceRegistry.getConnection(databaseConfig);
374         connections.add(connection);
375         return connection;
376     }

399     /**
400      * Create a DB connection that is not managed by the HikariCP. The connection is used as a
401      * fix for SWAT-2653 to fetch the correct DB name from the database, and guarantee the
402      * correctness of Auto-Detect.
403      *
404      * Note: Remember to close the connection after usage in order to avoid connection leak
405      *
406      * @return  The Un-pooled db connection; null if exception encountered
407      */
408     protected Connection getUnPooledConnection() {
409         ClassLoader originalClassLoader = Thread.currentThread().getContextClassLoader();
410         Connection conn = null;
411         try {
412             if (databaseConfig.getClassLoader() != null) {
413                 Thread.currentThread().setContextClassLoader(databaseConfig.getClassLoader());
414                 Class.forName(databaseConfig.getDriver(), true, databaseConfig.getClassLoader());
415                 Properties properties = new Properties();
416                 properties.putAll(databaseConfig.getUrlProperties());
417                 properties.put("user", databaseConfig.getUsername());
418                 properties.put("password", databaseConfig.getPassword());
419                 conn = DriverManager.getConnection(databaseConfig.getJdbcUrl(), properties);
420                 if (LOGGER.isDebugEnabled() && conn != null) {
421                     LOGGER.debug("(+) created new un-pooled connection id = @{}",
422                             Integer.toHexString(conn.hashCode()));
423                 }
424             }
425         } catch (Exception e) {
426             LOGGER.warn("Failed to create an un-pooled connection", e);
427         } finally {
428             if(containsAbandonedConnectionCleanupThread(databaseConfig)) {
429                 // Shutdown the possibly started abandonedConnectionCleanupThread from Mysql JDBC
430                 shutdownAbandonedConnectionCleanupThread(databaseConfig.getClassLoader());
431             }
432             Thread.currentThread().setContextClassLoader(originalClassLoader);
433         }
434         return conn;
435     }

471     public ClassLoader getSldbClassLoader(List<String> jdbcDriversPath)

473         // our driver might be specified by the snap UI and located in SLDB so we need to
474         // create an appropriate classloader. Copied from JdbcDriverRegistryManager

476         ClassLoader loader = parentClassLoader;
477         if (!CollectionUtils.isEmpty(jdbcDriversPath)) {
478             List<URL> urlList = Lists.newArrayListWithExpectedSize(jdbcDriversPath.size());
479             for (String jdbcDriverPath : jdbcDriversPath) {
480                 if (StringUtils.isNotBlank(jdbcDriverPath)) {

482                     urlList.add(resolvedPath.toURL());
483                 }
484             }
485             loader = new SldbJarClassLoader(urlList.toArray(new URL[urlList.size()]),
486                     parentClassLoader);
487         }
488         return loader;
489     }

491     @Override
492     public void loadDriver(final PropertyValues propertyValues,
493                            final JdbcDriverManager jdbcDriverManager) {
494         List<String> jdbcDriversList = new ArrayList<>();
495         if (jdbcDriverPaths != null && !jdbcDriverPaths.isEmpty()) {
496             for (Map<String, Map<String, Object>> jdbcDriverMap : jdbcDriverPaths) {
497                 Object jdbcDriver = jdbcDriverMap.get(JDBC_DRIVER_PROP).get(
498                         SdkConstants.SNAP_PROPERTY_VALUE);
499                 if (jdbcDriver == null) {
500                     throw new ConfigurationException(JDBC_DRIVER_JAR_ERROR_MSG)
501                             .withReason(JDBC_DRIVER_JAR_ERROR_REASON)
502                             .withResolution(JDBC_DRIVER_JAR_ERROR_RESOLUTION);
503                 }
504                 jdbcDriversList.add(jdbcDriver.toString());
505             }
506         }
507 
508         ClassLoader originalClassLoader = Thread.currentThread().getContextClassLoader();
509         // we need this list for jars loaded in a privileged context, e.g., the hive drivers
510         // when using Kerberos.
511         try {
512             databaseConfig.setClassLoader(getSldbClassLoader(jdbcDriversList));
513             if (databaseConfig.getClassLoader() != null) {
514                 Thread.currentThread().setContextClassLoader(databaseConfig.getClassLoader());
515             }
516             // load driver
517             jdbcDriverManager.registerDriver(databaseConfig.getDriver(), jdbcDriversList);
518         } catch (MalformedURLException e) {
519             throw new ConfigurationException(JDBC_DRIVER_JAR_ERROR_MSG)
520                     .withReason(e.getLocalizedMessage())
521                     .withResolution(JDBC_DRIVER_JAR_ERROR_RESOLUTION);
522         } finally {
523             if(containsAbandonedConnectionCleanupThread(databaseConfig)) {

528         }

669     private BigInteger evalInt(ExpressionProperty prop, BigInteger defaultVal) {
670         Object val = prop.eval(null);
671         if (val instanceof Number) {
672             return BigInteger.valueOf(((Number) val).longValue());
673         } else if (val instanceof String) {
674             String s = ((String) val).trim();
675             if (s.isEmpty()) {
676                 return defaultVal;
677             } else {
678                 return new BigInteger(s);

680         } else {
681             return defaultVal;
