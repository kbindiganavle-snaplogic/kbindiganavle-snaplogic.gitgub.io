65     public Update() {

73     @Override

87     @Override
88     protected void process(final Document document, final String inputViewName) {
89         super.process(document, inputViewName);
90         Document formattedDocument = formatDocumentLookUpField(document,
91                 currentEntitySchemaLookUpMap);
92         if (batchSize == 1) {
93             processDocumentSingle(document, formattedDocument);
94         } else {
95             processDocumentBatch(document, formattedDocument);
96         }
97     }

101     }
