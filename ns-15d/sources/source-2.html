210         toDateTimeString = toExpression.eval(document);

226                     .withReason(TIME_ZONE_NOT_FOUND_IN_SUGGEST_LIST)
227                     .withResolution(PLEASE_CORRECT_TIME_ZONE_AND_RETRY);

250     @Override
251     protected void process(Document document, String inputViewName) {
252         String searchJobId = null;
253         try {
254             evaluateExpressionProperties(document);
255             searchJobId = sumoClient.createSearchJob(query,
256                     fromDateTimeString, toDateTimeString, timeZone);
257             int batchSize = Math.min(account.getBatchSize().intValue(), SUMOLOGIC_API_LIMIT);
258             int retryLimit = account.getRetryLimit().intValue();
259             int statusPolls = 0;
260             int messagesRead  = 0;
261             int messageCount = 0;
262             Map<String, String> fieldMap =  null;
263             final List<Map<String, Object>> messageList = new ArrayList<>();
264             GetSearchJobStatusResponse getSearchJobStatusResponse;
265 
266             do {
267                 LOG.debug(GETTING_JOB_STATUS);
268                 getSearchJobStatusResponse = sumoClient.getSearchJobStatus(searchJobId);
269                 LOG.debug("JobStatus: state: {}, statusPolls/retryLimit: {}/{}, msgRead/msgCount:" +
270                                 " {}/{}", getSearchJobStatusResponse.getState(), statusPolls,
271                         retryLimit, messagesRead, messageCount);
272                 messageCount = getSearchJobStatusResponse.getMessageCount();
273                 int messageLength = Math.min(messageCount - messagesRead, batchSize);
274                 if (messageCount > messagesRead) {
275                     GetMessagesForSearchJobResponse getMessagesForSearchJobResponse =
276                             sumoClient.getMessagesForSearchJob(searchJobId, messagesRead,
277                                     messageLength);
278                     fieldMap = (fieldMap != null) ?
279                             fieldMap : extractFieldTypes(getMessagesForSearchJobResponse.getFields());
280                     mapSearchLogMessages(document, fieldMap, messageList,
281                             getMessagesForSearchJobResponse.getMessages());
282                     messagesRead += messageLength;
283                     if (isPreview && messageList.size() >= maxSuggestValue.intValue()) {
284                         break;
285                     }
286                 } else {
287                     if (getSearchJobStatusResponse.getState().equals(CANCELLED)) {
288                         throw new ExecutionException(JOB_SEARCH_ERROR)
289                                 .withReason(JOB_WAS_CANCELLED)
290                                 .withResolution(JOB_WAS_CANCELLED_RES);
291                     }
292                     if (!getSearchJobStatusResponse.getState().equals(DONE_GATHERING_RESULTS)) {
293                         // According to SumoLogic example don't hammer the Sumo Logic service.
294                         Thread.sleep(account.getRetryDelay().multiply(SECOND_TO_MS).intValue());
295                     }
296                 }
297             } while (getSearchJobStatusResponse.getState().equals(GATHERING_RESULTS)
298                     ||(!getSearchJobStatusResponse.getState().equals(DONE_GATHERING_RESULTS)
299                     && ++statusPolls < retryLimit)
300                     || messagesRead < messageCount);
301 
302             if (statusPolls > retryLimit) {
303                 throw new ExecutionException(EXECUTION_ERROR)
304                         .withReason(String.format(RETRY_LIMIT_EXCEEDED, retryLimit))
305                         .withResolution(RETRY_LIMIT_EXCEEDED_RES);
306             }
307 
308             if (showAggregateRecs && hasAggregate()) {
309                 Map<String, Object> results = new LinkedHashMap<>();
310                 results.put(MESSAGES, messageList);
311                 final List<Map<String, Object>> recordList = new ArrayList<>();
312                 GetRecordsForSearchJobResponse getRecordsForSearchJobResponse =
313                         sumoClient.getRecordsForSearchJob(searchJobId, 0,
314                                 getSearchJobStatusResponse.getRecordCount());
315                 mapAggregateRecords(document,
316                         extractFieldTypes(getRecordsForSearchJobResponse.getFields()),
317                         recordList, getRecordsForSearchJobResponse.getRecords());
318                 results.put(RECORDS, recordList);
319                 Document d = documentUtility.newDocument(results);
320                 outputViews.write(d, document);
321             }
322         } catch (ExecutionException e) {
323             Map<String, Object> wrapper = Maps.newHashMapWithExpectedSize(5);
324             wrapper.put(ERROR, (StringUtils.isNotBlank(e.getMessage()) ?
325                     e.getMessage() : StringUtils.EMPTY));
326             wrapper.put(STACKTRACE, Throwables.getStackTraceAsString(e));
327             SnapDataException se = new SnapDataException(documentUtility.newDocumentFor(
328                     document, wrapper), e, e.getMessage())
329                     .withReason(e.getReason())
330                     .withResolution(e.getResolution());
331             errorViews.write(se, document);
332         } catch (Throwable t) {
333             SnapDataException se = new SnapDataException(t, EXECUTION_ERROR)
334                     .withReason(Throwables.getRootCause(t).getMessage())
335                     .withResolution(INVALID_INPUT_RES);
336             errorViews.write(se, document);
337         } finally {
338             try { // Proactively cancelling a search job to be a good citizen.
339                 if (searchJobId != null) {
340                     sumoClient.cancelSearchJob(searchJobId);
341                 }
342             } catch (Throwable t) {
343                   LOG.info(ERROR_CANCELLING_SEARCH_JOB, t.getMessage());
344             }
345         }
346     }

367     private void mapSearchLogMessages(Document document, Map<String, String> fieldMap,
368             List<Map<String, Object>> messageList, List<LogMessage> messages) {
369         for (LogMessage message : messages) {
370             Map<String, Object> messageMap = new LinkedHashMap<>();
371             Map<String, String> errorMap = null;
372             for (Map.Entry<String, String> mapEntry : message.getProperties().entrySet()) {
373                 if (useResponseFieldTypes) {
374                     errorMap = fieldMapper(fieldMap, messageMap, mapEntry.getKey(),
375                             mapEntry.getValue());
376                 } else {
377                     messageMap.put(mapEntry.getKey(), mapEntry.getValue());
378                 }
379             }
380             if (!MapUtils.isEmpty(errorMap)) {
381                 Document errorDocument = documentUtility.newDocument(errorMap);
382                 SnapDataException snapDataException = new SnapDataException(
383                         errorDocument, INVALID_DATA);
384                 errorViews.write(snapDataException, document);
385             } else {
386                 if (showAggregateRecs) {
387                     messageList.add(messageMap);
388                 } else {
389                     outputViews.write(documentUtility.newDocument(messageMap));
390                 }
391             }
392         }
393     }

395     private void mapAggregateRecords(Document document, Map<String, String> fieldMap,
396             List<Map<String, Object>> recordList, List<SearchJobRecord> records) {
397         for (SearchJobRecord record : records) {
398             if (useResponseFieldTypes) {
399                 Map<String, Object> convenienceMap = new LinkedHashMap<>();
400                 Map<String, String> errorMap = null;
401                 for (Map.Entry<String, String> mapEntry : record.getMap().entrySet()) {
402                     errorMap = fieldMapper(fieldMap, convenienceMap,
403                             mapEntry.getKey(), mapEntry.getValue());
404                 }
405                 if (errorMap != null && errorMap.size() > 0) {
406                     Document errorDocument = documentUtility.newDocument(errorMap);
407                     SnapDataException snapDataException = new SnapDataException(
408                             errorDocument, INVALID_DATA);
409                     errorViews.write(snapDataException, document);
410                 } else {
411                     recordList.add(convenienceMap);
412                 }
413             } else {
414                 recordList.add((Map) record.getMap());
415             }
416         }
417     }

419     private Map<String, String> fieldMapper(Map<String, String> fieldMap,
420             Map<String, Object> convenienceMap, String property, String value) {
421         Map<String, String> errorMap = new LinkedHashMap<>(2);
422         try {
423             String fieldType = fieldMap.get(property);
424             if (fieldType == null) {
425                 errorMap.put(property, String.format(
426                         MISSING_FIELD_TYPE,property));
427                 return errorMap;
428             }
429             switch (fieldMap.get(property)) {
430                 case STRING:
431                     convenienceMap.put(property, value);
432                     break;
433                 case LONG:
434                 case INT:
435                     convenienceMap.put(property, new BigInteger(value));
436                     break;
437                 case DOUBLE:
438                     convenienceMap.put(property, new BigDecimal(value));
439                     break;
440                 case BOOLEAN:
441                     convenienceMap.put(property, Boolean.valueOf(value));
442                     break;
443                 default:
444                     throw new ExecutionException(INVALID_MAPPING)
445                             .withReason(UNKNOWN_PROPERTY_FIELD_TYPE)
446                             .withResolution(UNKONWN_PROPERTY_RES);
447             }
448         } catch (NumberFormatException nfe) {
449             errorMap.put(property, String.format(
450                     FIELD_TYPE_VALUE,
451                     property, fieldMap.get(property), value));
452         }
453         return errorMap;
454     }
