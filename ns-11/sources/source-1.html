72     public static AutosyncAssetStatsDTO.TableStatsDTO convertAssetStatsFromMap(
73             Map<String, Object> syncDataResult, String loadMethod) {
74         AutosyncAssetStatsDTO.AutosyncAssetStatsResultDTO autosyncAssetStatsResultDTO =
75                 MAPPER.convertValue(syncDataResult,
76                         AutosyncAssetStatsDTO.AutosyncAssetStatsResultDTO.class);
77         return CollectionUtils.isEmpty(autosyncAssetStatsResultDTO.getResults()) ?
78 //              If result is empty -> there is no data in src table and
79 //              load was not invoked. We set 'Success' in this case manually
80                 getDefaultSuccessStatsWhenEmptySyncResult(loadMethod) :
81                 autosyncAssetStatsResultDTO.getResults().get(0);
82     }

153

155 
156         Map<String, Object> tgtExtraParamMap = Optional

267     private static String parseSourceAccount(EndpointAssetDTO source) {
268         String sourceAccount = EMPTY_STRING;
269         List<String> sourceAccounts = source.getAccounts();
270 
271         LOGGER.info("AutosyncAssetUtil: parseSourceAccount {}", prettyPrint(source));
272         if (sourceAccounts == null || sourceAccounts.isEmpty()) {
273             if (!StringUtils.equalsIgnoreCase(CSV_FILE_READER, source.getName())) {
274                 throw new ConfigurationException("Source account is empty")
275                         .withReason("Source account is not provided.")
276                         .withResolution("Ensure that source account is correct and provided");
277             }
278         } else {
279             sourceAccount = sourceAccounts.get(0);
280         }
281 
282         return sourceAccount;
283     }
