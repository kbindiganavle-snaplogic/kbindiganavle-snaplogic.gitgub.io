78      *
79      * Type 1 is "plain" Json format where there is a single json object for a Json file.
80      *
81      * For example, we have a single Json array object for the file:
82      * [
83      *    {
84      *       'first_name': 'Jane',
85      *       'last_name': 'Smith',
86      *       'phone_num': '111-222-3333'
87      *    },
88      *    {
89      *       'first_name': 'Jane',
90      *       'last_name': 'Smith',
91      *       'phone_num': '111-222-3333'
92      *    }
93      * ]
94      *
95      * Type 2 is "json lines" Json format. Here a Json file has multiple lines,
96      * one line per source file record.
97      *
98      * For example,
99      *
100      * {"FIRSTNAME":"Dare","DEPT":"dev","RANGE":0.0,"LOCAL":true,"QTY":7}
101      * {"FIRSTNAME":"Lia","DEPT":"sales","RANGE":16.4,"LOCAL":false,"QTY":3000}
102      * {"FIRSTNAME":"Bruke","DEPT":"dev","RANGE":1.54,"LOCAL":true,"QTY":100}
103      *
104      * Type 3 is "json objects" Json format where a Json file has multiple Json objects,
105      * one object per Json file record (but one record object typically spans multiple lines).
106      *
107      * For example,
108      * {
109      *     "catdesc": "Major League Baseball",
110      *     "catid": 1,
111      *     "catgroup": "Sports",
112      *     "catname": "MLB"
113      * }

195             BufferedReader reader, List<String> columnNames, int chunkSizeToRead)  {
196         List<JSONRecord> recordList = new ArrayList<>();
197 
198         try {
199             String jsonLine;
200             lastLine = EMPTY_STRING;
201             lastLinePos = 0;
202 
203             while ((jsonLine = readRecordLine(reader)) != null &&
204                     recordNumber <= chunkSizeToRead) {
205                 recordNumber++;
206                 // Skip a line with all white space
207                 if (StringUtils.isNotBlank(jsonLine)) {

211                         List<JSONRecord> recordListPerStream =

216                     } catch (final Throwable e) {
217                         SnapDataException ex = new SnapDataException(e, ERROR_PROCESSING_RECORD_AT)
218                                 .formatWith(recordNumber)
219                                 .withReason(e.getMessage())
220                                 .withResolution(String.format(ERROR_PROCESSING_JSON_FILE_RESOLUTION,

240      * Reteieve and return a JSON object string
241      * from the given "reader".
242      *
243      * @param reader    Reader to read
244      * @return          JSON object string
245      */
246     private String readRecordLine(BufferedReader reader) throws IOException {
247         StringBuilder sb = new StringBuilder();
248         Pair<Integer, Integer> foundRange;
249 
250         // If the "lastLine" has data to process, process.
251         if (StringUtils.isNotBlank(lastLine) && lastLinePos < lastLine.length()) {
252             if ((foundRange =
253                     findJsonObject(lastLine, lastLinePos)) != null) {
254                 return lastLine.substring(foundRange.getLeft(),
255                         foundRange.getRight() + 1);
256             }
257             sb.append(lastLine.substring(lastLinePos));
258         }
259 
260         while ((lastLine = reader.readLine()) != null) {
261             lastLinePos = 0;
262             lastLine = lastLine.replaceAll(System.lineSeparator(), "");
263             sb.append(lastLine);
264             if ((foundRange = findJsonObject(sb.toString(), 0)) != null) {
265                 // The lastLine should have RIGHT_CURLY_BRACKET to have foundRange = true.
266                 // Thus we set the position after RIGHT_CURLY_BRACKET to lastLinePos here.
267                 lastLinePos = lastLine.indexOf(RIGHT_CURLY_BRACKET) + 1;
268                 return sb.substring(foundRange.getLeft(), foundRange.getRight() + 1);
269             }
270         }
271 
272         return null;
273     }

276      * Find a JSON object from the given "str" starting from the given "begin"
277      * as search begin position. Then return, if found, the start position
278      * and the end position of the JSON object within "str".
279      * Returns null if not found.
280      *
281      * Also this function verifies the given "str" is a valid JSON string.
282      * by checking parentheses. If not, throws an exception.
283      *
284      * For example, this function returns
285      * "{ "catdesc": "Major League Baseball", "catid": 1 }" from
286      * the given input string "str" with
287      * "{ "catdesc": "Major League Baseball", "catid": 1 }{ "catdesc":"catdesc":" +
288      * ""National Hockey League", "catgroup": "Sports", "catid": 2 }"
289      *
290      * @param str       string to search with
291      *
292      * @param begin     search start position
293      * @return      Pair with start and end postion of the JSON object string
294      *              that is found within "str" on success.
295      *
296      *              null if not found.
297      */
298     private Pair<Integer, Integer> findJsonObject(String str, int begin) throws IOException {
299         String targetStr = str.substring(begin);
300         // Find left and right parenthesis from the given "str".
301         int start = targetStr.indexOf(LEFT_CURLY_BRACKET);
302         int end = targetStr.indexOf(RIGHT_CURLY_BRACKET);
303 
304         if (start == -1 || end == -1) {
305             return null;
306         }
307 
308         // Check the validity of the current JSON object string
309         if (start >= end) {
310             throw new IOException("Invalid JSON object: } is before {");
311         }
312 
313         // Check parentheses validity
314         // isValidParenthesese(str)
315 
316         return Pair.of(start, end);
317     }

369     private List<JSONRecord> readRecordFromInputStream(
370             InputStream inputStream, List<String> columnNames) {
371         // Build an iterator
372         JsonDataIterator dataIterator = createDataIterator(inputStream);
373 
374         List<JSONRecord> records = new ArrayList<>();
375         boolean isFirst = true;
376         // Read a JSON record
377         while (dataIterator.hasNext() &&
378                 recordNumber < detectorInputFileConfig.getChunkSizeToRead()) {
379             if (isFirst) {
380                 isFirst = false;
381             } else {
382                 recordNumber++;
383             }
384             Object data = dataIterator.next();
385             Map<String, String> dataMap = getMapOnData(data);
386 
387             // Set up columnNames if not done yet
388             if (CollectionUtils.isEmpty(columnNames)) {
389                 for (Object k : dataMap.keySet()) {
390                     columnNames.add(String.valueOf(k));
391                 }
392             }
393 
394             // Add a new JSONRecord with "values" in the insertion order
395             // of "this.columnNames".
396             String[] values = new String[dataMap.size()];
397             for (int i = 0; i < columnNames.size(); i++) {
398                 Object value = dataMap.get(columnNames.get(i));
399                 if (value != null) {
400                     values[i] = String.valueOf(value);
401                 }
402             }
403 
404             records.add(new JSONRecord(values, recordNumber));
405         }
406 
407         return records;
408     }
