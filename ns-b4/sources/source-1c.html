220     private void rewriteSqlQueryToUseSubqueryFromCte() {
221         // Rewrite the given sqlQuery to use subqueries instead of CTEs since
222         // Azure Synapse does not support
223         // a SELECT statement that uses CTE in WHERE clause:
224         // For example,
225         // "SELECT * FROM (WITH cte1 (col1, col2) AS (SELECT c1, c2 FROM t1)
226         // SELECT * FROM cte1) dt1"
227         // gets incorrect syntax error due to WITH clause.
228 
229         // Note that, for our example, 'sqlQuery' here will have
230         // "WITH cte1 (col1, col2) AS (SELECT c1, c2 FROM t1".
231         sqlQuery = sqlQuery.trim();
232         if (StringUtils.startsWithIgnoreCase(sqlQuery, WITH_STATEMENT)) {
233             sqlQuery = EltRewriterUtils.rewriteCteToSubquery(sqlQuery);
234         }
235     }

246     private void configureViewTableMap(final PropertyValues propertyValues) {
247         List<Map<String, String>> viewTableMappingProperty =
248                 propertyValues.get(VIEW_TABLE_MAPPING_PATH);
249         if (viewTableMappingProperty == null) {
250             viewTableMappingEmpty = true;
251         } else {
252             viewTableMappingEmpty = viewTableMappingProperty.isEmpty();
253         }
254 
255         if (!inputViewsSpecified && !viewTableMappingEmpty) {
256             throw new ConfigurationException(INVALID_CONFIGURATION_OF_THE_SNAP)
257                     .withReason(VIEW_TO_TABLE_MAP_SPECIFIED)
258                     .withResolution(DO_NOT_SPECIFY_VIEW_TO_TABLE_MAP);
259         }
260 
261         if (!viewTableMappingEmpty) {
262             for (Map<String, String> viewTable : viewTableMappingProperty) {
263                 ExpressionProperty tableNameExpr = propertyValues.getExpressionPropertyFor(
264                         viewTable, VIRTUAL_TABLE_NAME_PROP);
265                 String viewName = JsonPath.read(viewTable, INPUT_VIEW_PROP_PATH);
266                 if (StringUtils.isBlank(viewName)) {
267                     throw new ConfigurationException(INVALID_CONFIGURATION_OF_THE_SNAP)
268                             .withReason(VIEW_NAME_EMPTY_IN_VIEW_TABLE_MAP)
269                             .withResolution(SPECIFY_VALID_VIEW_NAME_IN_VIEW_TABLE_MAP);
270                 }
271                 if (!tableNameExpr.isEmpty()) {
272                     String virtualTableName = tableNameExpr.eval(null);
273                     if (virtualTableName.contains("\"") ||
274                             virtualTableName.contains("'")) {
275                         throw new ConfigurationException(INVALID_CONFIGURATION_OF_THE_SNAP)
276                                 .withReason(TABLE_NAME_CANNOT_CONTAIN_QUOTES)
277                                 .withResolution(REMOVE_QUOTES_FROM_TABLE_NAME);
278                     }
279                     viewTableMapping.put(viewName, tableNameExpr);
280                 } else {
281                     throw new ConfigurationException(INVALID_CONFIGURATION_OF_THE_SNAP)
282                             .withReason(TABLE_NAME_EMPTY_IN_VIEW_TABLE_MAP)
283                             .withResolution(SPECIFY_VALID_TABLE_NAME_IN_VIEW_TABLE_MAP);
284                 }
285             }
286         }

347     @Override
348     public void execute() {
349         if (inputViews.isEmpty()) {
350             processDocument(null, null);
351         } else {
352             Set<String> pendingViews = Sets.newHashSet(inputViews.names());
353             Map<String, Document> selectedDocuments =
354                     inputViews.select(TIMEOUT_MILLISECONDS, TimeUnit.MILLISECONDS);
355             // Keep reading till we complete all the input views.
356             int pollTime = TIMEOUT_MILLISECONDS;
357             String virtualTableName;
358 
359             while (!pendingViews.isEmpty()) {
360                 if (!selectedDocuments.isEmpty()) {
361                     for (Map.Entry<String, Document> documentEntry : selectedDocuments.entrySet()) {
362 
363                         String docInputViewName = documentEntry.getKey();
364                         Document doc = documentEntry.getValue();
365                         if (doc == null) {
366                             pendingViews.remove(docInputViewName);
367                             continue;
368                         }
369 
370                         String inputSql = extractQueryFromInputDocument(doc);
371                         if (StringUtils.isBlank(inputSql)) {
372                             pendingViews.remove(docInputViewName);
373                             continue;
374                         }
375 
376                         if (enableSQLQuery && isViewTableMappingNotEmpty()) {
377 
378                             // Get the corresponding virtualTableName defined by the user for the
379                             // given input view being processed.
380                             String docInputViewLabel = getInputViewLabel(docInputViewName);
381                             ExpressionProperty virtualTableNameExpr =
382                                     viewTableMapping.get(docInputViewLabel);
383                             virtualTableName = EMPTY_STRING;
384                             if (virtualTableNameExpr != null && !virtualTableNameExpr.isEmpty()) {
385                                 virtualTableName = virtualTableNameExpr.eval(null);
386                             }
387 
388                             // Attach the inputSql from the upstream snap with the virtualTableName
389                             // extracted from the mapping table.
390                             String inputSqlWithVirtualTableName =
391                                     String.format(INPUT_SQL_WITH_VIRTUAL_TABLE_NAME, inputSql,
392                                             virtualTableName);
393 
394                             /* Replace the virtualTableName in the original user specified SQL Query
395                              in the Query Text Box with the inputSql attached with the
396                              virtualTableName.
397                              We only want to replace the virtualTableName used in the FROM clause of
398                              a query expression and not from the SELECT list or any other places of
399                              the query expression.
400                              For example if the derived table from upstream inputSql comes as:
401                              "SELECT a1, b1 FROM t1"
402                              Let's say the user references that inputSql as dt1 in t
403                             he mapping table.
404                              Let's say the user writes a new query in the query text box as follows:
405                              "SELECT dt1.b1, t2.b2 FROM t2 JOIN dt1 ON dt1.a1 = t2.a2"
406 
407                              The replacement of dt1 reference in the above query should happen only
408                              in FROM clause and no where else. So the query would look like after
409                              replacement as:
410                              "SELECT dt1.b1, t2.b2 FROM t2 JOIN (SELECT a1, b1 FROM t1) dt1 ON
411                              dt1.a1 = t2.a2" */
412 
413                             String virtualTableNamePattern =
414                                     String.format(VIRTUAL_TABLE_NAME_PATTERN, virtualTableName);
415                             Pattern p = Pattern.compile(virtualTableNamePattern);
416                             Matcher m = p.matcher(transformedSql);
417                             transformedSql = m.replaceAll(inputSqlWithVirtualTableName);
418                         }
419                         pollTime = TIMEOUT_MILLISECONDS;
420                     }
421                 } else {
422                     // We are not receiving any data. Lets increment poll time.
423                     pollTime += INCREMENTS;
424                     if (pollTime > MAX_POLL_TIME) {
425                         // If the poll time exceeds the max allowed poll time,
426                         // lets fix the poll time as the max allowed poll time.
427                         pollTime = MAX_POLL_TIME;
428                     }
429                 }
430                 selectedDocuments = inputViews.select(pollTime, TimeUnit.MILLISECONDS);
431             }
432 
433             if (StringUtils.isBlank(transformedSql)) {
434                 processDocument(null, null);
435                 return;
436             }
437 
438             Map<String, Object> output = getOutputViewMap(null, transformedSql,
439                     singletonList(snapType));
440             if (handlePreview(documentUtility.newDocument(output), null, account,
441                     validationInProgress, previewRowCount, executionMode,
442                     executeSqlInPreview)) {
443                 return;
444             }
445             outputViews.write(documentUtility.newDocument(output));
446         }
447     }
