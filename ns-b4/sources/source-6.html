249     private List<Function> buildBinaryFunctions(PropertyValues propertyValues) {
250         List<Map<String, Object>> funcsTable = propertyValues.get(BINARY_FUNCTIONS_PROP);
251         List<Function> binaryFuncsList = new ArrayList<>();
252         if (funcsTable != null) {
253             for (Map<String, Object> arg : funcsTable) {
254                 BinaryFunction name = BinaryFunction.valueOf(
255                         jsonPathUtil.nullableRead(BINARY_FUNCTION_NAME_PATH, arg));
256                 ExpressionProperty firstArgExpression =
257                         propertyValues.getExpressionPropertyFor(arg, BINARY_FUNC_FIRST_ARG_PROP);
258                 String firstArgument = firstArgExpression.eval(null);
259                 ExpressionProperty secondArgExpression =
260                         propertyValues.getExpressionPropertyFor(arg, BINARY_FUNC_SECOND_ARG_PROP);
261                 String secondArgument = secondArgExpression.eval(null);
262                 String alias = jsonPathUtil.nullableRead(BINARY_FUNCTION_ALIAS_PATH, arg);
263                 binaryFuncsList.add(provider.provideBinaryFunction(name, firstArgument,
264                         secondArgument, alias));
265             }
266         }
267         return binaryFuncsList;
268     }

308     private Map<String, Object> getConditionalOutput(String inputSqlString, Document document,
309                                                      List<String> snapTypes) {
310         List<String> previousSnapTypes = extractSnapTypeFromInputDocument(document);
311         Map<String, Object> output;
312         String randomString = "Conditional_".concat(getDerivedTableName());
313         if (canPushdownFunctions(document, CONDITIONAL_PUSHDOWN_SNAP_LIST, specifiedFunctions,
314                 inputSqlString)) {
315             snapTypes.addAll(previousSnapTypes);
316             output = buildConditionalPushDownOutput(document, snapTypes, randomString,
317                     inputSqlString);
318         } else {
319             output = buildConditionalOutput(inputSqlString, snapTypes, document, randomString);
320         }
321         return output;
322     }

324     private Map<String, Object> buildConditionalPushDownOutput(Document document,
325                                                                List<String> snapTypes,
326                                                                String randomString,
327                                                                String inputSqlString) {
328         String resultSql = StringUtils.EMPTY;
329         PushdownContext ctx = PushdownContext.getOrCreate(document, randomString);
330         if (!specifiedFunctions.isEmpty()) {
331             List<Field> allFunctionsFieldsWithOriginalNames = buildAllFields(specifiedFunctions);
332             checkColumnsExists(document, allFunctionsFieldsWithOriginalNames, inputSqlString);
333 
334             replaceArgumentsIfRenamed(ctx, specifiedFunctions);
335             List<Field> allFunctionsFields = buildAllFields(specifiedFunctions);
336             SelectPart selectPart = ctx.getSelectPart();
337             if (passThroughConditional) {
338                 List<Field> expressions = new ArrayList<>(selectPart.getExpressions());
339                 expressions.addAll(0, allFunctionsFields);
340                 selectPart.setExpressions(expressions);
341             } else {
342                 selectPart.setExpressions(new ArrayList<>(allFunctionsFields));
343             }
344             resultSql = pushdownQueryBuilder.buildSql(ctx);
345         } else if (passThroughConditional) {
346             resultSql = pushdownQueryBuilder.buildSql(ctx);
347         } else {
348             throwInvalidConfig();
349         }
350         return getOutputViewMap(document, resultSql, snapTypes, ctx);
351 
352     }

354     private Map<String, Object> buildConditionalOutput(String inputSqlString,
355                                                        List<String> snapTypes,
356                                                        Document document, String randomString) {
357         PushdownContext ctx = PushdownContext.create(document, randomString);
358         String resultSql = StringUtils.EMPTY;
359         if (!specifiedFunctions.isEmpty()) {
360             List<Field> allFunctionsFields = buildAllFields(specifiedFunctions);
361             String allFunctionsString = buildAllFuncsString(specifiedFunctions);
362             SelectPart selectPart = ctx.getSelectPart();
363             if (passThroughConditional) {
364                 List<Field> expressions = new ArrayList<>(selectPart.getExpressions());
365                 expressions.addAll(0, allFunctionsFields);
366                 selectPart.setExpressions(expressions);
367                 resultSql = String.format(FUNCTIONS_AND_PASS_THROUGH_SQL,
368                         allFunctionsString, inputSqlString, randomString);
369             } else {
370                 selectPart.setExpressions(new ArrayList<>(allFunctionsFields));
371                 resultSql = String.format(
372                         FUNCTIONS_ONLY_SQL, allFunctionsString, inputSqlString, randomString);
373             }
374         } else if (passThroughConditional) {
375             if (isSubQueryPushdownEnabled()) {
376                 // if Pass through is enabled and there are no conditional
377                 // functions configured in the snap, subquery pushdown is enabled
378                 // then the Conditional Functions snap can simply forward the input
379                 // SQL without any processing.
380                 resultSql = inputSqlString;
381             } else {
382                 resultSql = String.format(
383                         PASS_THROUGH_ONLY_SQL_FUNCTIONS, inputSqlString, randomString);
384             }
385         } else {
386             throwInvalidConfig();
387         }
388         return getOutputViewMap(document, resultSql, snapTypes, ctx);
389     }
