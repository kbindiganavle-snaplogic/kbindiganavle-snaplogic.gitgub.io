710         String inputSqlString = extractQueryFromInputDocument(document);
711         if (StringUtils.isBlank(inputSqlString)) {
712             LOG.info("Input sql is empty, do nothing");
713             return;
714         }
715         String derivedTableName = "AgrGroupBy_" + getDerivedTableName();
716 
717         if (isGroupByAllFields) {
718             groupByClauseList = buildAllGroupByList(document, inputSqlString);
719         }
720 
721         if (validationInProgress) {
722             schemaSupportService.defineInputSchema(schemaProvider, inputSqlString, inputViewName);
723         }
724 
725         String sqlAggregate;
726 
727         List<String> snapTypes = new ArrayList<>();
728         snapTypes.add(ELT_AGGREGATE_TITLE);
729 
730         PushdownContext pushdownContext;
731 
732         if (!linearRegressionRewrites && isGroupByBlockPushdownPossible()
733                 && canPushdownFunctions(document, AGGREGATE_PUSHDOWN_SNAP_LIST,
734                 functionList, inputSqlString)) {
735             pushdownContext = PushdownContext.getOrCreate(document, derivedTableName);
736             sqlAggregate =
737                     doAggregateRewrite(document, snapTypes, pushdownContext, inputSqlString);
738         } else {
739             if (doesAggrFunctionsExprListExist()) {
740                 sqlAggregate = buildAggregateQuery(document, inputSqlString, derivedTableName);
741             } else {

744                         .withResolution(ERR_RES_INVALID_SNAP_CONFIGURATION_AGGR);

747                 //disable pushdown for the next snap
748                 pushdownContext = null;
749             } else {
750                 pushdownContext = PushdownContext.create(document, derivedTableName);
751                 populatePushdownCtx(pushdownContext);
752             }
753         }
754 
755         sqlAggregate = addUniqueQueryID(sqlAggregate);
756 
757         LOG.info("Resulted aggregate sql query : {}", sqlAggregate);
758 
759         Map<String, Object> output = getOutputViewMap(document, sqlAggregate,
760                 snapTypes, pushdownContext);
761 
762         if (handlePreview(documentUtility.newDocument(output), document,
763                 account, validationInProgress, previewRowCount, executionMode,
764                 executeSqlInPreview)) {
765             return;
766         }
767         outputViews.write(documentUtility.newDocument(output), document);
768     }

1517     private Map<String, RewriterFunction> compressAggrRegrFunctionsList(
1518             List<RewriterFunction> functions) {
1519         Map<String, RewriterFunction> compressedFunctionMap = new LinkedHashMap<>();
1520         // This set is for counting number of unique pairs of arguments to the given "functions".
1521         Set<String> argumentPairSet = new LinkedHashSet<>();
1522 
1523         int setIdx = 0;
1524         for (int i = 0; i < functions.size(); i++) {
1525             RewriterFunction current = functions.get(i);
1526             current.setOriginalArrayIdx(i);
1527             List<String> keys = getkeyForAggrFunction(current.getFunctionName(),
1528                     current.getField1Str(), current.getField2Str());
1529             argumentPairSet.add(keys.get(1));
1530 
1531             String keyToMap = keys.get(0);
1532             current.setKeyToMap(keyToMap);
1533             RewriterFunction value = compressedFunctionMap.get(keyToMap);
1534             if (value == null) {
1535                 compressedFunctionMap.put(keyToMap, current);
1536                 current.setCompressedArrayIdx(setIdx++);
1537                 current.setCompressedArrayEntry(true);
1538             } else {
1539                 // Two cases for current's <field1Str, field2Str>
1540                 // if current is commutative: <f1, f2>, <f2, f1>
1541                 current.setCompressedArrayIdx(value.getCompressedArrayIdx());
1542             }
1543         }
1544

1546             this.targetListType = RewriteTargetListTypeForAggr
1547                     .TYPE_ONE_WITHOUT_GROUP_BY;
