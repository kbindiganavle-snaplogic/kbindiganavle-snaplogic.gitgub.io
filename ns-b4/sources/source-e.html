877     protected List<String> getColumnWithTypeListFromInputSql(String inputSql,
878                                                              List<Object> sqlBindValues) {
879         List<String> colWithType = new ArrayList<>();
880         Connection connection = jdbcOperations.acquireConnection(account);
881         try (PreparedStatement statement = connection.prepareStatement(inputSql)) {
882             for (int colNo = 1; colNo <= sqlBindValues.size(); colNo++) {
883                 statement.setString(colNo, (String) sqlBindValues.get(colNo - 1));
884             }
885             try (ResultSet resultSet = statement.executeQuery()) {
886                 ResultSetMetaData resultSetMetaData = resultSet.getMetaData();
887                 int columnCount = resultSetMetaData.getColumnCount();
888                 for (int i = 1; i <= columnCount; i++) {
889                     colWithType.add(buildColumnWithTypeDefinition(resultSetMetaData, i));
890                 }
891             }
892         } catch (SQLException e) {
893             throw new SnapDataException(e, ERR_MSG_FAILED_TO_FIND_TABLE_COLUMNS)
894                     .withReason(e.getMessage())
895                     .withResolution(ERR_RESOLUTION_FAILED_TO_FIND_TABLE_COLUMNS);
896         }
897         return colWithType;
898     }

900     private String buildColumnWithTypeDefinition(ResultSetMetaData resultSetMetaData, int colNumber)
901             throws SQLException {
902         String colWithTypeStr = "";
903         final String columnName = resultSetMetaData.getColumnName(colNumber);
904         final String columnTypeName = resultSetMetaData.getColumnTypeName(colNumber);
905         final String columnClassName = resultSetMetaData.getColumnClassName(colNumber);
906         final boolean precisionAssignable =
907                 isPrecisionAssignable(columnClassName, columnTypeName);
908         if (precisionAssignable) {
909             final int scale = resultSetMetaData.getScale(colNumber);
910             int precision = resultSetMetaData.getPrecision(colNumber);
911 
912             if (BigDecimal.class.getName().equals(columnClassName) &&
913                     precision < DEFAULT_NUMERIC_PRECISION_REDSHIFT) {
914                 precision = DEFAULT_NUMERIC_PRECISION_REDSHIFT;
915             } else if (String.class.getName().equals(columnClassName) &&
916                     precision < DEFAULT_STRING_PRECISION_REDSHIFT) {
917                 precision = DEFAULT_STRING_PRECISION_REDSHIFT;
918             }
919             if (scale > 0) {
920                 colWithTypeStr =
921                         String.format(COLUMN_DATA_TYPE_WITH_PRECISION_AND_SCALE_RS,
922                                 columnName, columnTypeName, precision, scale);
923             } else {
924                 colWithTypeStr = String.format(COLUMN_DATA_TYPE_WITH_PRECISION_RS,
925                         columnName, columnTypeName, precision);
926             }
927         } else {
928             colWithTypeStr = String.format(COLUMN_DATA_TYPE_QUOTED_RS, columnName,
929                     columnTypeName);
930         }
931         return colWithTypeStr;
932     }

966         @Override
967         SrcTrgColumnsPair getSrcTrgColumnPair(Document document) {
968             if (MapUtils.isEmpty(insertValuesMap)) {
969                 Set<String> srcColumns = findTableColumns(document,
970                         extractQueryFromInputDocument(document));
971                 Set<String> trgColumns = findTableColumnsWithAutoCommitEnabled(targetTableName);
972 
973                 if (srcColumns.size() > trgColumns.size()) {
974                     throw new SnapDataException(
975                             "Source table has more columns than target table")
976                             .withResolution("Uncheck the advanced option and specify all the " +
977                                     "source columns and target columns manually using " +
978                                     "the insert expression list");
979                 }
980 
981                 String srcColumnsString = String.join(COMMA_SPACE, srcColumns);
982 
983                 String trgColumnsString = trgColumns.stream()
984                         .limit(srcColumns.size())
985                         .collect(Collectors.joining(COMMA_SPACE));
986

988             } else {
989                 Set<String> srcColumns = findTableColumns(document,
990                         extractQueryFromInputDocument(document));
991                 Set<String> trgColumns = findTableColumnsWithAutoCommitEnabled(targetTableName);
992 
993                 if (srcColumns.size() + insertValuesMap.size() > trgColumns.size()) {
994                     throw new SnapDataException("The sum total of source table columns " +
995                             "and columns in the expression list is more than " +
996                             "the number of target table columns")
997                             .withResolution("Uncheck the Advanced options and " +
998                                     "specify the expressions manually in the expression list.");
999                 }
1000 
1001                 if (!trgColumns.containsAll(insertValuesMap.keySet())) {
1002                     throw new SnapDataException("Target table does not contain " +
1003                             "one or more specified columns in the insert expression list")
1004                             .withResolution("Please specify only the existent " +
1005                                     "target columns in the insert expression list");
1006                 }
1007 
1008                 Map<String, String> srcTrgColumnsMap = replaceWithExpressionValues(
1009                         trgColumns, srcColumns, insertValuesMap);
1010 
1011                 String srcColumnsString = String.join(COMMA_SPACE, srcTrgColumnsMap.values());
1012                 String trgColumnsString = String.join(COMMA_SPACE, srcTrgColumnsMap.keySet());
1013 
1014                 return new SrcTrgColumnsPair(srcColumnsString, trgColumnsString);
1015             }
1016         }

1019     private Map<String, String> replaceWithExpressionValues(Set<String> trgColumns,
1020                                                             Set<String> srcColumns,
1021                                                             Map<String, String> insertValuesMap) {
1022         Map<String, String> tmp = new LinkedHashMap<>();
1023 
1024         Iterator<String> iterator = srcColumns.iterator();
1025 
1026         trgColumns.forEach(
1027                 col -> {
1028                     if (iterator.hasNext()) {
1029                         tmp.put(col, iterator.next());
1030                     } else {
1031                         tmp.put(col, null);
1032                     }
1033                 }
1034         );
1035 
1036         tmp.putAll(insertValuesMap);
1037 
1038         tmp.values().removeAll(Collections.singleton(null));
1039 
1040         return tmp;
1041     }

1091                 }
1092

1094                 intersect.retainAll(trgColumns);
1095

1097                     throw new SnapDataException("There were no matching source and " +

1109                         extractQueryFromInputDocument(document));
1110                 Set<String> trgColumns = findTableColumnsWithAutoCommitEnabled(targetTableName);
1111 
1112                 Set<String> intersect = new LinkedHashSet<>(srcColumns);
1113                 intersect.retainAll(trgColumns);
1114 
1115                 Map<String, String> matchedColumns = intersect.stream()
1116                         .collect(Collectors.toMap(
1117                                 Function.identity(),
1118                                 c -> insertValuesMap.getOrDefault(c, c),
1119                                 (col1, col2) -> col1,
1120                                 LinkedHashMap::new
1121                         ));
1122 
1123                 Set<String> diffSet = new LinkedHashSet<>(trgColumns);
1124                 diffSet.removeAll(intersect);
1125 
1126                 if (!diffSet.containsAll(insertValuesMap.keySet())) {
1127                     throw new SnapDataException("Target table does not contain one " +
1128                             "or more specified columns in the insert expression list")
1129                             .withResolution("Please specify only the existent " +
1130                                     "target columns in the insert expression list");
1131                 }
1132 
1133                 Map<String, String> matchedTrgExprList = new LinkedHashMap<>();
1134 
1135                 for (String col: diffSet) {
1136                     String expr = insertValuesMap.get(col);
1137                     if (expr != null) {
1138                         matchedTrgExprList.put(col, expr);
1139                     }
1140                 }
1141 
1142                 if (!matchedTrgExprList.isEmpty()) {
1143                     matchedColumns.putAll(matchedTrgExprList);
1144                 }
1145 
1146                 return new SrcTrgColumnsPair(String.join(COMMA_SPACE, matchedColumns.values()),
1147                         String.join(COMMA_SPACE, matchedColumns.keySet()));
