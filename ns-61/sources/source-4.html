188     /**
189      * Validate config for SSL encryption configuration.
190      *
191      * @param builder
192      * @param properties
193      */
194     public boolean validateConfigProperties(PropertyValues values, ConfigPropertyEnum[] properties,
195             HadoopConfig hadoopConfig) {
196         String truststore = values.get(getDatabasePropertyValuePath(Property.TRUSTSTORE));
197         String keystore = values.get(getDatabasePropertyValuePath(Property.KEYSTORE));
198 
199         if (hadoopConfig.useSSL()) {
200             if (isEmpty(truststore) && isEmpty(keystore)) {
201                 throw new ConfigurationExceptionWithBundle(ErrorCondition.KEYSTORE_REQUIRED);
202             }
203 
204             File truststoreFile = new File(truststore);
205             if (!truststoreFile.exists() || !truststoreFile.isFile() || !truststoreFile.canRead()) {
206                 throw new ConfigurationExceptionWithBundle(
207                         ErrorCondition.TRUSTSTORE_DOES_NOT_EXIST);
208             }
209 
210             File keystoreFile = new File(keystore);
211             if (!keystoreFile.exists() || !keystoreFile.isFile() || !keystoreFile.canRead()) {
212                 throw new ConfigurationExceptionWithBundle(ErrorCondition.KEYSTORE_DOES_NOT_EXIST);
213             }
214         }
215 
216         return true;
217     }
