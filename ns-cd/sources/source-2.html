324     /**
325      * Detect and return column data type for the given column values
326      * tab[][colIndex] with checkUnknownType.
327      *
328      * CF: https://docs.aws.amazon.com/redshift/latest/dg/r_Date_and_time_literals.html
329      *
330      * Please see the comments at TypeInferenceUtils::detectColumnDataType() for more.
331      *
332      * @param typePatternMap    Redshift column data type -> Pattern
333      * @param tab               2-dimensional array with rows being records,
334      *                          and columns being columns
335      * @param colIndex          column index of the column to detact
336      * @param checkUnknownType  If true, keep unknown column type as UNKNOWN_COLUMN_TYPE.
337      *                          If false, assign VARCHAR if nothing matches the column data values.
338      * @return      Detected Redshift data type
339      */
340     public static String detectColumnDataType(
341             Map<String, Pattern> typePatternMap, String[][] tab, int colIndex,
342             boolean checkUnknownType) {
343         if (ArrayUtils.isEmpty(tab)) {
344             return getReturnTypeIfNotMatchesOrTableEmpty(checkUnknownType);
345         }
346 
347         int[] startIndex = new int[1];
348         startIndex[0] = 0;
349         String typeToMatch = checkUnknownType ?
350                 UNKNOWN_COLUMN_TYPE :
351                 REDSHIFT_AUTODETECT_NEXT_TYPE_FOR_FIRST_MATCH.get(UNKNOWN_COLUMN_TYPE);
352         String prevTypeToMatch = EMPTY_STRING;
353         ColumnMatcherResult result = null;
354         boolean matching = false;
355         boolean matchingPartiallyHappened = false;
356 
357         while (!matching && !REDSHIFT_VARCHAR_COLUMN_TYPE.equalsIgnoreCase(prevTypeToMatch)) {
358             result = columnMatches(typeToMatch,
359                     typePatternMap, tab, colIndex, startIndex);
360             matching = result.matching;
361             // Prepare for the next span if necessary (when matching = false).
362             if (!matching) {
363                 prevTypeToMatch = typeToMatch;
364                 if (!matchingPartiallyHappened) {
365                     matchingPartiallyHappened = result.matchingPartially;
366                 }
367                 if (matchingPartiallyHappened) {
368                     typeToMatch = REDSHIFT_AUTODETECT_NEXT_TYPE_TO_MATCH.get(typeToMatch);
369                 } else {
370                     // No type has matches on any column value at tab[][colIndex].
371                     typeToMatch = REDSHIFT_AUTODETECT_NEXT_TYPE_FOR_FIRST_MATCH.get(typeToMatch);
372                 }
373             }
374         }
375 
376         // If nothing matches, return UNKNOWN if checkUnknownTYpe is true,
377         // or VARCHAR if it is false
378         if (!matching) {
379             return getReturnTypeIfNotMatchesOrTableEmpty(checkUnknownType);
380         }
381 
382         return result.typeWithSize;
383     }

387             return UNKNOWN_COLUMN_TYPE;
388         } else {

393     private static ColumnMatcherResult columnMatches(
394             final String type, final Map<String, Pattern> typePatternMap,
395             final String[][] tab, int colIndex, int[] startIndex) {
396         // Check the given "startIndex".
397         if (startIndex[0] < 0 || startIndex[0] >= tab.length) {
398             throw new SnapDataException(INTERNAL_ERROR_HAS_OCCURRED)
399                     .withReason(String.format(REASON_INVALID_INDEX_DETECTED, startIndex[0]))
400                     .withResolution(CONTACT_SUPPORT);
401         }
402         // Check the given "type" and its corresponding entry at "TypePatternMap".
403         Pattern pattern = typePatternMap.get(type);
404         if (pattern == null) {
405             throw new ConfigurationException(ERR_PATTERN_NOT_FOUND)
406                     .withReason(String.format(REASON_PATTERN_NOT_FOUND, type))
407                     .withResolution(CONTACT_SUPPORT);
408         }
409 
410         // Note that we set REDSHIFT_MAX_VARCHAR_SIZE_LIMIT
411         // (max size of Redshift varchar column type) to "sizeLimitForVarchar"
412         // since the schema detector reads only first 400 lines of input file.
413         // If full input file scan is used, we can start with
414         // REDSHIFT_DEFAULT_VARCHAR_SIZE_LIMIT (256).
415         int sizeLimitForVarchar = REDSHIFT_MAX_VARCHAR_SIZE_LIMIT;
416         boolean matchedValueExists = false;
417         for (int i = startIndex[0]; i < tab.length; i++) {
418             String currentValue = tab[i][colIndex];
419             if (StringUtils.isBlank(currentValue) ||
420                     FileSchemaDetectorUtils.isValueUnknownType(typePatternMap, currentValue)) {
421                 continue;
422             }
423             if (!pattern.matcher(currentValue).matches()) {
424                 updateStartIndexWhenNotMatches(type, startIndex, i);
425                 if (matchedValueExists) {
426                     return new ColumnMatcherResult(false, true, EMPTY_STRING);
427                 } else {
428                     return new ColumnMatcherResult(false, EMPTY_STRING);
429                 }
430             }
431             matchedValueExists = true;
432             if (StringUtils.equalsIgnoreCase(type, REDSHIFT_VARCHAR_COLUMN_TYPE) &&
433                     currentValue.length() > sizeLimitForVarchar &&
434                     sizeLimitForVarchar != REDSHIFT_MAX_VARCHAR_SIZE_LIMIT) {
435                     // Check just to prevent int overflow.
436                     if (sizeLimitForVarchar < Integer.MAX_VALUE / 2) {
437                         sizeLimitForVarchar =
438                                 sizeLimitForVarchar * 2 < REDSHIFT_MAX_VARCHAR_SIZE_LIMIT ?
439                                         sizeLimitForVarchar * 2 : REDSHIFT_MAX_VARCHAR_SIZE_LIMIT;
440                     } else {
441                         sizeLimitForVarchar = REDSHIFT_MAX_VARCHAR_SIZE_LIMIT;
442                     }
443             }
444         }
445         String typeWithSize = type;
446         if (StringUtils.equalsIgnoreCase(type, REDSHIFT_VARCHAR_COLUMN_TYPE)) {
447             typeWithSize = sizeLimitForVarchar == REDSHIFT_DEFAULT_VARCHAR_SIZE_LIMIT ?
448                     REDSHIFT_VARCHAR_COLUMN_TYPE :
449                     String.format(REDSHIFT_VARCHAR_DATA_TYPE_FORMAT, sizeLimitForVarchar);
450         }
451         // if type is not UNKNOWN and matchedValueExists is false,
452         // return false as ColumnMatcherResult.matching.
453         boolean matchesToReturn = true;
454         if (!StringUtils.equalsIgnoreCase(type, UNKNOWN_COLUMN_TYPE) &&
455                 !matchedValueExists) {
456             matchesToReturn = false;
457         }
458         return new ColumnMatcherResult(matchesToReturn, typeWithSize);
459     }
