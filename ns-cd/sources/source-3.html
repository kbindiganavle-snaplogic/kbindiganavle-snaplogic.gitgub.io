215     public static String detectColumnDataType(
216         Map<String, Pattern> typePatternMap, String[][] tab, int colIndex,
217         boolean checkUnknownType) {
218         if (ArrayUtils.isEmpty(tab)) {
219             return getReturnTypeIfNotMatchesOrTableEmpty(checkUnknownType);
220         }
221 
222         int[] startIndex = new int[1];
223         startIndex[0] = 0;
224         String typeToMatch = checkUnknownType ?
225                 UNKNOWN_COLUMN_TYPE :
226                 SNOWFLAKE_AUTODETECT_NEXT_TYPE_FOR_FIRST_MATCH.get(UNKNOWN_COLUMN_TYPE);
227         String prevTypeToMatch = EMPTY_STRING;
228         ColumnMatcherResult result = null;
229         boolean matching = false;
230         boolean matchingPartiallyHappened = false;
231 
232         while (!matching && !SNOWFLAKE_VARCHAR_COLUMN_TYPE.equalsIgnoreCase(prevTypeToMatch)) {
233             result = columnMatches(typeToMatch,
234                     typePatternMap, tab, colIndex, startIndex);
235             matching = result.matching;
236             // Prepare for the next span if necessary (when matching = false).
237             if (!matching) {
238                 prevTypeToMatch = typeToMatch;
239                 if (!matchingPartiallyHappened) {
240                     matchingPartiallyHappened = result.matchingPartially;
241                 }
242                 if (matchingPartiallyHappened) {
243                     typeToMatch = SNOWFLAKE_AUTODETECT_NEXT_TYPE_TO_MATCH.get(typeToMatch);
244                 } else {
245                     // No type has matches on any column value at tab[][colIndex].
246                     typeToMatch = SNOWFLAKE_AUTODETECT_NEXT_TYPE_FOR_FIRST_MATCH.get(typeToMatch);
247                 }
248             }
249         }
250 
251         // If nothing matches, return UNKNOWN if checkUnknownTYpe is true,
252         // or VARCHAR if it is false
253         if (!matching) {
254             return getReturnTypeIfNotMatchesOrTableEmpty(checkUnknownType);
255         }
256 
257         return result.typeWithSize;
258     }

260     private static String getReturnTypeIfNotMatchesOrTableEmpty(boolean checkUnknownType) {

262             return UNKNOWN_COLUMN_TYPE;
263         } else {
264             return SNOWFLAKE_VARCHAR_COLUMN_TYPE;
265         }

268     private static ColumnMatcherResult columnMatches(
269             final String type, final Map<String, Pattern> typePatternMap,
270             final String[][] tab, int colIndex, int[] startIndex) {
271         // Check the given "startIndex".
272         if (startIndex[0] < 0 || startIndex[0] >= tab.length) {
273             throw new SnapDataException(INTERNAL_ERROR_HAS_OCCURRED)
274                     .withReason(String.format(REASON_INVALID_INDEX_DETECTED, startIndex[0]))
275                     .withResolution(CONTACT_SUPPORT);
276         }
277         // Check the given "type" and its corresponding entry at "TypePatternMap".
278         Pattern pattern = typePatternMap.get(type);
279         if (pattern == null) {
280             throw new ConfigurationException(ERR_PATTERN_NOT_FOUND)
281                     .withReason(String.format(REASON_PATTERN_NOT_FOUND, type))
282                     .withResolution(CONTACT_SUPPORT);
283         }
284 
285         boolean matchedValueExists = false;
286         for (int i = startIndex[0]; i < tab.length; i++) {
287             String currentValue = tab[i][colIndex];
288             if (StringUtils.isBlank(currentValue) ||
289                     FileSchemaDetectorUtils.isValueUnknownType(typePatternMap, currentValue)) {
290                 continue;
291             }
292             if (!pattern.matcher(currentValue).matches()) {
293                 updateStartIndexWhenNotMatches(type, startIndex, i);
294                 if (matchedValueExists) {
295                     return new ColumnMatcherResult(false, true, EMPTY_STRING);
296                 } else {
297                     return new ColumnMatcherResult(false, EMPTY_STRING);
298                 }
299             }
300             matchedValueExists = true;
301         }
302         // if type is not UNKNOWN and matchedValueExists is false,
303         // return false as ColumnMatcherResult.matching.
304         boolean matchesToReturn = true;
305         if (!StringUtils.equalsIgnoreCase(type, UNKNOWN_COLUMN_TYPE) &&
306                 !matchedValueExists) {
307             matchesToReturn = false;
308         }
309         return new ColumnMatcherResult(matchesToReturn, type);
310     }

312     static void updateStartIndexWhenNotMatches(String type, int[] startIndex, int currentIndex) {
313         if (SNOWFLAKE_NUMERIC_COLUMN_TYPES.contains(type)) {
314             startIndex[0] = currentIndex;
315         } else {
316             startIndex[0] = 0;
317         }
318     }
