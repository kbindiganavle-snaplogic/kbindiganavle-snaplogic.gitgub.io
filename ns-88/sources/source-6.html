282     private void startCompletionReceiver() {
283         execActivityThread = spawnHelperThread(() -> {
284                     while (looping) {
285                         try {
286                             LOG.debug("Waiting for completions");
287                             AbstractActivityHandler handler = executionActivity.poll(MAX_WAIT_TIME,
288                                     MAX_WAIT_UNIT);
289                             if (handler != null) {
290                                 LOG.debug("  Received completion: {}", handler);
291                                 handler.run();
292                             }
293                         } catch (InterruptedException e) {
294                             LOG.debug("Interrupted while waiting for pipeline completions!", e);
295                             return;
296                         } catch (Throwable th) {
297                             handleUnexpectedException(th);
298                         }
299                     }
300                 });
301         execActivityThread.start();
302     }

462     private RuntimeException transformError(Map<String, String> error,
463                                             Map<String, Object> snapMap,
464                                             TypeEnum type, String source, String target) {
465         String pipeExecLabel = error.get("label");
466         String failureReason = error.get("reason");
467         String failureMessage = error.get("failure");
468         String failedSnapLabel = StringUtils.substringBetween(failureReason, "label=", ",");
469         if (StringUtils.isBlank(failedSnapLabel)){
470             failedSnapLabel = pipeExecLabel;
471         }
472         String snapClassFqId =
473                 getFailedSnapClassFqId(snapMap, failedSnapLabel);
474 
475         if (StringUtils.isBlank(snapClassFqId)) {
476             snapClassFqId =
477                     StringUtils.substringBetween(failureReason, "snap_fqid=", "}");
478         }
479 
480         if (StringUtils.isBlank(snapClassFqId)) {
481             snapClassFqId = getFailedSnapClassFqId(snapMap, pipeExecLabel);
482         }
483         ErrorHandler errorHandler =
484                 ErrorHandlerCreator.getErrorHandler(snapClassFqId);
485 
486         String stringError = failureMessage.contains(GENERIC_PIP_EXEC_ERROR_MESSAGE) ?
487                 failureReason : StringUtils.join(error);
488 
489         return errorHandler.handle(
490                 stringError, type, source, target);
491     }

530     protected List<Document> getDocumentsFromChild(PipelineRun pipeRun, TypeEnum type,

532         List<Document> documents = new ArrayList<>();
533         PipelineState state = pipeRun.getCurrentInfo().getPipelineState();
534         if (state != PipelineState.FAILED) {
535             // get data from child's output view
536             InputViews iVs = pipeRun.getPipelineOutputs();
537             try {
538                 if (iVs.size() >= 1) {
539                     for (Document document : (DocumentInputView) iVs.get()) {
540                         documents.add(document);
541                     }
542                 }
543             } catch (RuntimeException e) {
544                 LOG.error("Pipeline execution failed", e);
545                 handleErrors(pipeRun, type, source, target, action);
546                 throw e;
547             }
548         } else {
