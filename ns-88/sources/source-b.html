77     public Map<String, Object> actions(
78             Endpoint endpoint, String action, TableConfig tableConfig,
79             List<Map<String, Object>> srcDefValues,
80             Map<String, Object> columnSchema,
81             String endpointType,
82             String columnDefinition,
83             String path,
84             String query,
85             TargetAction targetAction,
86             NameHandler nameHandler) {
87         // prepare inputs
88         String tableName = nameHandler.enquoteIfRequired(NameType.TABLE_NAME,
89                 tableConfig.getTableName());
90         String sourceTableName = tableName;
91         String targetTableName = nameHandler.enquoteIfRequired(NameType.TABLE_NAME,
92                 tableConfig.getTargetTableName());
93         if (endpointType.equalsIgnoreCase("target")) {
94             tableName = targetTableName;
95         }
96         Map<String, String> parameters = new HashMap<>();
97         parameters.put("catalog", endpoint.getDatabaseName());
98         parameters.put("schema", endpoint.getSchemaName());
99         parameters.put("action", action);
100         parameters.put("account", endpoint.getAccountPath());
101         parameters.put("source_table", sourceTableName);
102         parameters.put("table", tableName);
103         String tgtExtraParams = tableConfig.getTgtExtraParams();
104         parameters.put("extra_params", tgtExtraParams == null ? "" : tgtExtraParams);
105         Map<String, Object> docFields =
106                 getActionsDocFields(tableConfig, columnDefinition,
107                         srcDefValues, columnSchema,
108                         endpointType, endpoint,
109                         query, nameHandler);
110 
111         final String actionsPipelinePath = String.format("%s/%s-%s-actions",
112                 path, endpoint.getType().toLowerCase(), endpointType);
113         PipelineConfig pipeConfig = new PipelineConfig(actionsPipelinePath, path);
114         BasePipeExecSnap.PipelineResponse pipelineResponse = null;
115         try {
116             pipelineResponse = snap.executeAndReturn(
117                 pipeConfig, new DocumentImpl(docFields), parameters, TypeEnum.TRANSFORM,
118                 endpointType.equalsIgnoreCase("source") ?
119                         endpoint.getType().toLowerCase() : "",
120                 endpointType.equalsIgnoreCase("target") ?
121                         endpoint.getType().toLowerCase() : "", action);
122         } catch (PipelineExecutionException e) {
123             LOGGER.debug("actions response: {} got PipelineExecutionException {}",
124                     targetAction.toString(), e.toString());
125 
126             if (StringUtils.equals(tableConfig.getTgtDbType(), BIGQUERY) &&
127                     e.getErrorDto() != null) {
128                 if (targetAction.equals(CHECK_TABLE_EXISTENCE_DIRECTLY) &&
129                         StringUtils.equalsIgnoreCase(TARGET_TABLE_NOT_EXIST_REASON,
130                                 e.getErrorDto().getReason())) {
131                     LOGGER.debug("actions returning emptyList result " +
132                             "after parsing table not exist error");
133                     return Map.of("results", Collections.emptyList(),
134                             "dbType", endpoint.getType());
135                 }
136             } else {
137                 throw e;
138             }
139         }
140         List<Document> documentList = pipelineResponse.getDocuments();
141         List<Map<String, Object>> maps = new ArrayList<>(documentList.size());
142         documentList.forEach(document -> maps.add(document.get(Map.class)));
143         LOGGER.debug("actions response: {}", maps);
144         return Map.of("results", maps, "dbType", endpoint.getType());
145     }

147     private Map<String, Object> getActionsDocFields(TableConfig tableConfig,
148                                                     String columnDefinition,
149                                                     List<Map<String, Object>> srcDefValues,
150                                                     Map<String, Object> columnSchema,
151                                                     String endpointType, Endpoint endpoint,
152                                                     String query,
153                                                     NameHandler nameHandler) {
154         Map<String, Object> docFields = new HashMap<>();
155 
156         docFields.put("db_type_name", endpoint.getType());
157         docFields.put("endpoint", endpoint.getType().toLowerCase());
158         docFields.put("endpointType", endpointType);
159         docFields.put("columnDefinition", columnDefinition);
160         docFields.put("srcDef", srcDefValues);
161         docFields.put("config", tableConfig);
162         docFields.put("query", query);
163         docFields.put("tmpColumn", enquoteColumnNameIfRequired(nameHandler,
164                 tableConfig.getTmpColumn()));
165         docFields.put("tmpColumnType", tableConfig.getTmpColumnType());
166         docFields.put("startQuote", nameHandler.getStartQuote());
167         docFields.put("endQuote", nameHandler.getEndQuote());
168         if (MapUtils.isNotEmpty(columnSchema)) {
169             List<String> includedFields = new ArrayList<>(List.of("dataType", "precision", "scale",
170                     "isNullable", "defaultValue", "isPrimaryKeyColumn", "isLastUpdateColumn",
171                     "srcType", "srcColumn", "srcPrecision", "cdt", "column"));
172             for (String field : includedFields) {
173                 docFields.put(field, columnSchema.get(field));
174             }
175             // We need to enquote column names per nameHandler instance here not in the pipeline
176             docFields.put("column", enquoteColumnNameIfRequired(nameHandler,
177                     docFields.get("column")));
178         }
179         return docFields;
180     }

197         // Edge cases
198         if (columnName == null) {
199             return columnName;
200         }
201         String columnNameAsString = getValueAsString(columnName);
202         // If blank, return the name as is.
203         if (StringUtils.isBlank(columnNameAsString)) {
204             return columnName;
205         }
206 
207         // Regular case
208         return nameHandler.enquoteIfRequired(NameType.COLUMN_NAME, columnNameAsString);

265             if (!StringUtils.isEmpty(config.getLastUpdateColumn())) {
