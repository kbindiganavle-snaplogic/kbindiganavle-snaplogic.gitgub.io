307     @Override
308     public void execute() throws ExecutionException {
309         InputView inputView = inputViews.get();
310         Set<String> uniqueTopics = new LinkedHashSet<>();
311         for (final Document inputDocument : inputViews.documents(inputView)) {
312             // Create the producer on the first document. This avoids the overhead of
313             // creating a producer when there's no input, like in an error pipeline
314             // that receives no errors. Don't do this inside the try. If createProducer throws
315             // an exception, we just want to let that exception be thrown from this method.
316             if (producer == null) {
317                 createProducer();
318             }
319             try {
320                 String topic = topicNameExpr.eval(inputDocument);
321                 if (StringUtils.isBlank(topic)) {
322                     throw new SnapDataException(ERR_TOPIC_NULL_MSG)
323                             .withReason(ERR_TOPIC_NULL_REASON)
324                             .withResolution(ERR_TOPIC_NULL_RESOLUTION);
325                 }
326                 uniqueTopics.add(topic);
327                 Integer partition = confluentKafkaUtil.getIntegerExpr(partitionExpr, inputDocument);
328                 Object messageKey = messageKeyExpr.eval(inputDocument);
329                 Object messageValue = messageValueExpr.eval(inputDocument);
330                 if (messageValue == null) {
331                     throw new SnapDataException(ERR_MSG_VAL_NULL_MSG)
332                             .withReason(ERR_MSG_VAL_NULL_REASON)
333                             .withResolution(ERR_MSG_VAL_NULL_RESOLUTION);
334                 }
335                 Long timestamp = PropertyUtil.evalTimestamp(inputDocument, timestampExpr, null);
336                 if (timestamp != null && timestamp < 0) {
337                     throw new SnapDataException(INVALID_TIMESTAMP).formatWith(timestamp)
338                             .withReason(INVALID_TIMESTAMP_REASON)
339                             .withResolution(RESOLUTION_ADDRESS_ISSUE);
340                 }
341 
342                 Map<String, Object> recordMap = new LinkedHashMap<>();
343                 recordMap.put("key", messageKey);
344                 recordMap.put("value", messageValue);
345 
346                 Map<String, Object> headerMap = new LinkedHashMap<>();
347                 List<Header> headers = new ArrayList<>();
348                 for (HeaderRow row : headerRows) {
349                     String headerKey = row.keyExp.eval(inputDocument);
350                     if (headerKey == null) {
351                         throw new SnapDataException(ERR_HEADER_KEY_NULL_MSG)
352                                 .withReason(ERR_HEADER_KEY_NULL_REASON)
353                                 .withResolution(ERR_HEADER_KEY_NULL_RESOLUTION);
354                     }
355                     Object headerVal = row.valExp.eval(inputDocument);
356                     if (headerMap.containsKey(headerKey)) {
357                         // Kafka allows multiple headers with the same key, and so does this snap.
358                         // But we can't have a "headers" object with duplicate keys in the output
359                         // document, so instead, we use a list with multiple values.
360                         List<Object> valueList;
361                         Object oldValue = headerMap.get(headerKey);
362                         if (oldValue instanceof List) {
363                             valueList = (List<Object>) oldValue;
364                         } else {
365                             valueList = new ArrayList<>();
366                             headerMap.put(headerKey, valueList);
367                             valueList.add(oldValue);
368                         }
369                         valueList.add(headerVal);
370                     } else {
371                         headerMap.put(headerKey, headerVal);
372                     }
373                     try {
374                         if (headerVal instanceof List<?> && !row.serdeChoice.supportsList()) {
375                             for (Object headerValElem : (List<?>) headerVal) {
376                                 byte[] bytes = row.valSer.serialize(topic, headerValElem);
377                                 headers.add(new RecordHeader(headerKey, bytes));
378                             }
379                         } else {
380                             byte[] bytes = row.valSer.serialize(topic, headerVal);
381                             headers.add(new RecordHeader(headerKey, bytes));
382                         }
383                     } catch (Exception e) {
384                         if (headerVal instanceof String) {
385                             headerVal = '"' + escapeEcmaScript((String) headerVal) + '"';
386                         }
387                         if (e instanceof ClassCastException) {
388                             e = new RuntimeException(CAST_ERROR, e);
389                         }
390                         throw new SnapDataException(e, UNABLE_TO_SERIALIZE_HEADER)
391                                 .formatWith(headerKey, row.serdeChoice.label(), headerVal)
392                                 .withReason(e.getMessage())
393                                 .withResolution(UNABLE_TO_SERIALIZE_RES);
394                     }
395                 }
396                 if (!headerMap.isEmpty()) {
397                     recordMap.put("headers", headerMap);
398                 }
399 
400                 final ProducerRecord<Object,Object> producerRecord =
401                         new ProducerRecord<>(topic, partition, timestamp, messageKey, messageValue, headers);
402                 if (modeSynchronous) {
403                     Future<RecordMetadata> futureMetadata = producer.send(producerRecord);
404                     try {
405                         // Block and wait for the metadata to pass
406                         RecordMetadata recordMetadata = futureMetadata.get();
407                         passMetadata(recordMap, recordMetadata, inputDocument);
408                     } catch (java.util.concurrent.ExecutionException ex) {
409                         throw (Exception) ex.getCause();
410                     }
411                 } else {
412                     producer.send(producerRecord, new
413                             Callback() {
414                                 @Override
415                                 public void onCompletion(final RecordMetadata recordMetadata,
416                                         final Exception e) {
417                                     if (e != null) {
418                                         SnapDataException ex;
419                                         if (e instanceof RecordTooLargeException) {
420                                             ex = new SnapDataException(e, MSG_TOO_LARGE_ERR)
421                                                     .withReason(e.getMessage())
422                                                     .withResolution(RESOLUTION_MSG_TOO_LARGE);
423                                         } else {
424                                             // Catch Exceptions in the sending process
425                                             ex = new SnapDataException(e, e.getLocalizedMessage())
426                                                     .withReason(REASON_KAFKA_EXCEPTION)
427                                                     .withResolution(RESOLUTION_SEEK_HELP);
428                                         }
429                                         errorViews.write(ex, inputDocument);
430                                     }
431                                     else if (recordMetadata != null) {
432                                         // Send metadata to the output
433                                         passMetadata(recordMap, recordMetadata, inputDocument);
434                                     }
435                                 }
436                             });
437                 }
438             } catch (InterruptedException | InterruptException | SuggestViewAbortException e) {
439                 // The InterruptException constructor calls Thread.currentThread().interrupt(),
440                 // so let's clear the interrupted flag by calling Thread.interrupted(). If this
441                 // isn't done, the producer.close() will sometimes throw InterruptException again
442                 // and not close the producer cleanly.
443                 LOG.info("aborted; interrupted={}", Thread.interrupted(), e);
444                 break;
445             } catch (SnapDataException e) {
446                 errorViews.write(e, inputDocument);
447             } catch (ExecutionException e) {
448                 SnapDataException ex = new SnapDataException(e, e.getMessage())
449                         .withReason(e.getReason())
450                         .withResolution(e.getResolution());
451                 errorViews.write(ex, inputDocument);
452             } catch (RecordTooLargeException rtle) {
453                 SnapDataException ex = new SnapDataException(rtle, MSG_TOO_LARGE_ERR)
454                         .withReason(rtle.getMessage())
455                         .withResolution(RESOLUTION_MSG_TOO_LARGE);
456                 errorViews.write(ex, inputDocument);
457             } catch (SerializationException e) {
458                 SnapDataException ex = new SnapDataException(e, ERR_SERIALIZATION)
459                             .withReason(e.getMessage())
460                             .withResolution(RESOLUTION_CUSTOMER_SUPPORT);
461                 errorViews.write(ex, inputDocument);
462             } catch (KafkaException e) {
463                 SnapDataException ex = new SnapDataException(e, e.getMessage())
464                             .withReason(REASON_KAFKA_EXCEPTION)
465                             .withResolution(RESOLUTION_SEEK_HELP);
466                 errorViews.write(ex, inputDocument);
467             } catch (Exception e) {
468                 SnapDataException ex = new SnapDataException(e, ERR_READ_INPUT)
469                         .withReason(e.getMessage())
470                         .withResolution(RESOLUTION_CUSTOMER_SUPPORT);
471                 errorViews.write(ex, inputDocument);
472             }
473         }
474         if (!uniqueTopics.isEmpty()) {
475             service.logTopicDiagnostics(producerConfigProperties, uniqueTopics);
476         }
477     }

490     protected void createProducer() {
491         checkConfigConstraints();
492         producerConfigProperties = new Properties();
493         if (StringUtils.isNotBlank(clientID)) {
494             producerConfigProperties.put(ProducerConfig.CLIENT_ID_CONFIG, clientID);
495         }
496         producerConfigProperties.put(ProducerConfig.ACKS_CONFIG, acks);
497         if (batchSize != -1) {
498             producerConfigProperties.put(ProducerConfig.BATCH_SIZE_CONFIG, batchSize);
499         }

527     private void checkConfigConstraints() {
528         if (messagePublishTimeout != -1) {
529             int requestTimeout = DEFAULT_REQUEST_TIMEOUT;
530             String requestTimeoutStr = account.getAdvancedProperty(REQUEST_TIMEOUT_MS_CONFIG);
531             if (requestTimeoutStr != null) {
532                 try {
533                     requestTimeout = Integer.parseInt(requestTimeoutStr);
534                 } catch (NumberFormatException e) {
535                     LOG.warn("can't parse {}={}", REQUEST_TIMEOUT_MS_CONFIG, requestTimeoutStr);
536                 }
537             }
538             int linger = Math.max(lingerTime, 0);
539             int minDeliveryTimeout = requestTimeout + linger;
540             if (messagePublishTimeout < minDeliveryTimeout) {
541                 DELIVERY_TIMEOUT_LINT.report();
542                 LOG.warn("increasing messagePublishTimeout ({}) to minDeliveryTimeout ({}) = requestTimeout ({}) + linger ({})",
543                         messagePublishTimeout, minDeliveryTimeout, requestTimeout, linger);
544                 messagePublishTimeout = minDeliveryTimeout;
545             }
546         }

612     private void defineInputSchemaForProperty(ExpressionProperty prop,
613                                               SerdeChoice serdeChoice,
614                                               SchemaProvider provider,
615                                               String topic,
616                                               boolean isKey) {
617         if (prop != null && prop.isExpression()) {
618             String expression = prop.getExpression();
619             JsonPath path;
620             try {
621                 path = JsonPaths.compile(expression);
622                 if (path.hasMultipleResults()) {
623                     return;
624                 }
625             } catch (InvalidPathException e) {
626                 // the expression isn't a JSON path, so we don't have a path for a schema
627                 return;
628             }
629             switch (serdeChoice) {
630                 case AVRO:
631                     if (account == null) {
632                         return;
633                     }
634                     SchemaRegistryClient schemaRegistry = account.getSchemaRegistryClient();
635                     AvroSchemaObjects schemaObjects =
636                             schemaRegistryHelper.getAvroSchemaObjects(schemaRegistry, topic, isKey);
637                     for (String inputViewName : provider.getRegisteredViewNames()) {
638                         SchemaBuilder sb = provider.getSchemaBuilder(inputViewName);
639                         ObjectSchema objectSchema = schemaConverter.buildViewSchema(sb, path, schemaObjects);
640                         if (objectSchema != null) {
641                             provider.setProvidedSchemaForViewName(inputViewName, objectSchema);
642                         }
643                     }
644                     break;
645                 default:
646                     for (String inputViewName : provider.getRegisteredViewNames()) {
647                         try {
648                             SchemaBuilder sb = provider.getSchemaBuilder(inputViewName);
649                             List<Object> resolvedPath = path.resolvedPath(null);
650                             Deque<String> segments = new ArrayDeque<>(resolvedPath.size());
651                             for (Object pathElem : resolvedPath) {
652                                 segments.add(pathElem.toString());
653                             }
654                             String segment = segments.pollLast();
655                             ObjectSchema objectSchema = sb.name(segment).type(serdeChoice.snapType()).create();
656                             while (segment != null) {
657                                 segment = segments.pollLast();
658                                 objectSchema = sb.name(segment).type(SnapType.COMPOSITE).withChildSchema(objectSchema).create();
659                             }
660                             provider.setProvidedSchemaForViewName(inputViewName, objectSchema);
661                         } catch (InvalidPathException e) {
662                             // ignore
663                         }
664                     }
665                     break;
666             }
667         }
668     }
