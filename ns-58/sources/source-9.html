115     public Object from(String columnName, int columnIndex, boolean bindByName, DataType dataType,
116             CallableStatement stmt, CustomTypeConverter customTypeConverter)
117             throws SQLException {
118         Object value;
119         if (bindByName) {
120             value = stmt.getObject(columnName);
121         } else {
122             value = stmt.getObject(columnIndex);
123         }
124         if (value == null) {
125             return value;
126         }
127         String valueClassName = value.getClass().getSimpleName();
128         byte[] bytes;
129         switch (valueClassName) {
130             case TIMESTAMPTZ:
131                 try {
132                     if (bindByName) {
133                         bytes = stmt.getBytes(columnName);
134                     } else {
135                         bytes = stmt.getBytes(columnIndex);
136                     }
137                     value = ((OracleCustomTypeConverter) customTypeConverter)
138                             .fromTimestampTZ(bytes);
139                     if (value == null) {
140                         String dateStr;
141                         if (bindByName) {
142                             dateStr = stmt.getString(columnName);
143                         } else {
144                             dateStr = stmt.getString(columnIndex);
145                         }
146                         value = handleTimestampTz(dateStr, bytes);
147                     }
148                 } catch (SQLException e) {
149                     throw new SnapDataException(e, PROCESS_TIMESTAMPTZ_FAILED)
150                             .withReason(Throwables.getRootCause(e).getMessage())
151                             .withResolution(RESOLUTION_ADDRESS_ISSUE);
152                 }
153             case TIMESTAMPLTZ:
154                 value = handleTimestampLtz(columnName, stmt);
155                 break;
156             case TIMESTAMP:
157                 try {
158                     if (bindByName) {
159                         bytes = stmt.getBytes(columnName);
160                     } else {
161                         bytes = stmt.getBytes(columnIndex);
162                     }
163                     int[] convertedBytes = convertBytes(bytes);
164                     value = handleTimestamp(convertedBytes);
165                 } catch (SQLException e) {
166                     throw new SnapDataException(e, PROCESS_TIMESTAMP_FAILED)
167                             .withReason(Throwables.getRootCause(e).getMessage())
168                             .withResolution(RESOLUTION_ADDRESS_ISSUE);
169                 }
170                 break;
171             case INTERVAL_YEAR_TO_MONTH:
172             case INTERVAL_DAY_TO_SECOND:
173                 value = String.valueOf(value);
174                 break;
175             case DATE:
176                 // Avoid using org.jooq.Record because it would cut off the hh:mm:ss part
177                 try {
178                     if (bindByName) {
179                         value = LOCAL_DATE_TIME_CONVERTER.from(stmt.getTimestamp(columnName));
180                     } else {
181                         value = LOCAL_DATE_TIME_CONVERTER.from(stmt.getTimestamp(columnIndex));
182                     }
183                 } catch (SQLException e) {
184                     throw new SnapDataException(e, PROCESS_DATE_FAILED)
185                             .withReason(Throwables.getRootCause(e).getMessage())
186                             .withResolution(RESOLUTION_ADDRESS_ISSUE);
187                 }
188                 break;
189             case DOUBLE:
190                 value = handleDouble(dataType, value);
191                 break;
192             case BYTE:
193             case INTEGER:
194             case LONG:
195             case SHORT:
196                 value = BIG_INTEGER_CONVERTER.from((Number) value);
197                 break;
198             default:
199                 Converter converter = from(dataType);
200                 if (converter != null) {
201                     value = converter.from(value);
202                 }
203         }

358     private int[] convertBytes(final byte[] bytes) {
359         int[] convertedBytes = new int[bytes.length];
360         for (int j = 0; j < bytes.length; j++) {
361             // Oracle uses unsigned bytes so convert to ints
362             // bit-wise and with 255
363             convertedBytes[j] = bytes[j] & 0xff;
364         }
365         return convertedBytes;
366     }

382     private Object handleTimestamp(final int[] convertedBytes) {
383         Object value;
384         int century = (convertedBytes[0] - 100) * 100;
385         int decade = convertedBytes[1] - 100;
386         int month = convertedBytes[2];
387         int day = convertedBytes[3];
388         int hour = convertedBytes[4] - 1;
389         int minute = convertedBytes[5] - 1;
390         int second = convertedBytes[6] - 1;
391         if (convertedBytes.length > 7) {
392             int nanoseconds = convertedBytes[7] << 24;
393             nanoseconds |= convertedBytes[8] << 16;
394             nanoseconds |= convertedBytes[9] << 8;
395             nanoseconds |= convertedBytes[10] & 0xff;
396             // nanosecond percision is lost because jodatime objects only store milliseconds
397             long milliseconds = TimeUnit.MILLISECONDS.convert(nanoseconds,
398                     TimeUnit.NANOSECONDS);
399             value = new LocalDateTime(century + decade, month, day,
400                     hour, minute, second, (int) milliseconds);
401         } else {
402             value = new LocalDateTime(century + decade, month, day, hour,
403                     minute, second);
404         }
405         return value;
406     }

445     /**
446      * Some notes on conversion for the future.
447      *
448      * <pre>
449      * http://www.velocityreviews.com/forums/t128223-oracle-sql-timestamptz-byte-array.html
450      *
451      * Byte 0: Century, offset is 100 (value - 100 is century)
452      * Byte 1: Decade, offset is 100 (value - 100 is decade)
453      * Byte 2: Month UTC
454      * Byte 3: Day UTC
455      * Byte 4: Hour UTC, offset is 1 (value-1 is UTC hour)
456      * Byte 5: Minute UTC, offset is 1 (value-1 is UTC Minute)
457      * Byte 6: Second, offset is 1 (value-1 is seconds)
458      * Byte 7: nanoseconds (most significant bit)
459      * Byte 8: nanoseconds
460      * Byte 9: nanoseconds
461      * Byte 10: nanoseconds (least significant bit)
462      * Byte 11: Hour UTC-offset of Timezone, offset is 20 (value-20 is UTC-hour offset)
463      * Byte 12: Minute UTC-offset of Timezone, offset is 60 (value-60 is UTC-minute offset)
464      *
465      *
466      * int nanoseconds = tz[7] << 24;
467      * nanoseconds |= tz[8] << 16;
468      * nanoseconds |= tz[9] << 8;
469      * nanoseconds |= tz[10] & 0xff;
470      * </pre>
471      */
472     private Object handleTimestampTz(final String dateStr, final byte[] bytes) {
473         int[] convertedBytes = convertBytes(bytes);
474         if ((convertedBytes[11] & -128) != 0) {
475             return ORACLE_TIMESTAMPTZ_FMT_WITH_ZONEID.parseDateTime(dateStr);
476         } else {
477             int century = (convertedBytes[0] - 100) * 100;
478             int decade = convertedBytes[1] - 100;
479             int month = convertedBytes[2];
480             int day = convertedBytes[3];
481             int hour = convertedBytes[4] - 1;
482             int minute = convertedBytes[5] - 1;
483             int second = convertedBytes[6] - 1;
484             int ns = convertedBytes[7] << 24;
485             ns |= convertedBytes[8] << 16;
486             ns |= convertedBytes[9] << 8;
487             ns |= convertedBytes[10] & 0xff;
488             int hoursOffset = convertedBytes[11] - 20;
489             int minutesOffset = convertedBytes[12] - 60;
490             // nanosecond precision is lost because joda time objects only store milliseconds
491             long milliseconds = TimeUnit.MILLISECONDS.convert(ns,
492                     TimeUnit.NANOSECONDS);
493             DateTime dateTime = new DateTime(century + decade, month, day, hour,
494                     minute, second, (int) milliseconds, DateTimeZone.UTC);
495             DateTimeZone dateTimeZone = DateTimeZone.forOffsetHoursMinutes(hoursOffset,
496                     minutesOffset);
497             return dateTime.toDateTime(dateTimeZone);
498         }
499     }
