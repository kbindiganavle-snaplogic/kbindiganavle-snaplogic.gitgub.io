219                     headers = headerLine.split(serverFile.getData().getSeparator());
220                 } else {

222                     return;
223                 }
224             }
225             for (String viewName : provider.getRegisteredViewNames()) {
226                 SchemaBuilder schemaBuilder = provider.getSchemaBuilder(viewName);
227                 for (String header : headers) {
228                     Schema childSchema = provider.createSchema(SnapType.ANY, header);
229                     schemaBuilder.withChildSchema(childSchema);

239     public final void execute() {
240         LOGGER.trace("Anaplan Write Snap - Beginning Write Snap Execution");
241         if (inputViews.isEmpty()) {
242             LOGGER.debug("inputViews was null, returning without executing snap");
243             return;
244         }
245         final Iterator<Document> documentIterator = inputViews.getDocumentsFrom(inputViews.get());
246         Document document;
247         Set<JsonPath> columnNames;
248         String serverFileName;
249         if (documentIterator.hasNext()) {
250             document = documentIterator.next();
251             columnNames = getColumnNamesFrom(document);
252             setWorkspaceModelFromExpressions(document);
253             serverFileName = getValidStringExpressionProperty(SERVER_FILENAME_PROP,
254                     serverFileExpression.eval(document));
255         } else {
256             LOGGER.debug("Anaplan Write Snap - documentIterator had no elements");
257             return; // no data to write
258         }
259 
260         try {
261             anaplanModel = anaplanUtil.getWorkspaceModel(service, workspace, model);
262             serverFile = anaplanModel.getServerFile(serverFileName);
263             if (!isOverwrite && ServerFileExists(anaplanModel.getServerFiles(), serverFileName)) {
264                 SnapDataException sde = new SnapDataException(Messages.UPLOAD_CANNOT_PROCEED)
265                         .withReason(Messages.FILE_ALREADY_EXISTS_AND_OVERWRITE_NOT_TRUE)
266                         .withResolution(Messages.REMOVE_FILE_OR_SET_FILE_OVERWRITE_SETTING);
267                 errorViews.write(sde, document);
268                 return;
269             }
270             if (serverFile == null) {
271                 serverFile = anaplanModel.createServerFileImportDataSource(serverFileName, UPLOAD);
272                 if (serverFile == null) {
273                     SnapDataException sde = new SnapDataException(ANAPLAN_API_ERROR)
274                             .withReason(ANAPLAN_API_TO_CREATE_SERVER_FILE_FAILED)
275                             .withResolution(PLEASE_VERIFY_ANAPLAN_CONFIGURATION_AND_RETRY);
276                     errorViews.write(sde, document);
277                     return;
278                 }
279             }
280         } catch (ExecutionException e) {
281             SnapDataException sde = new SnapDataException(Messages.ANAPLAN_ERROR)
282                     .withReason(e.getReason())
283                     .withResolution(e.getResolution());
284             errorViews.write(sde, document);
285             return;
286         } catch (AnaplanAPIException e) {
287             SnapDataException sde =  new SnapDataException(e, ANAPLAN_API_ERROR)
288                     .withReason(String.format(ANAPLAN_API_EXCEPTION_MESSAGE, e.getMessage()))
289                     .withResolution(PLEASE_REVIEW_AND_RESOLVE_MESSAGE);
290             errorViews.write(sde);
291             return;
292         }
293 
294         if (overrideServerSettings) {
295             LOGGER.debug("Overriding server settings");
296             serverFile.getData().setSeparator(separator.getString());
297             serverFile.getData().setDelimiter(delimiter.getString());
298         } else {
299             LOGGER.debug("Using server settings");
300         }
301 
302         File tempFile = null;
303         try {
304             File tempFileDir = JfsUtils.getTmpDir(expressionUtils, propertyValues, errorViews);
305             tempFile = File.createTempFile(ANAPLAN_WRITE, TEMP, tempFileDir);
306             writeServerFile(tempFile, documentIterator, document, serverFile
307                     , serverFile.getData(), columnNames
308                     , isOverwrite);
309         } catch (SnapDataException sde) {
310             errorViews.write(sde, document);
311             return;
312         } catch (IOException e) {
313             SnapDataException sde =  new SnapDataException(Messages.ANAPLAN_UPLOAD_ERROR)
314                     .withReason(COULD_NOT_CREATE_A_TEMP_FILE_MSG + e.getMessage())
315                     .withResolution(Messages.PLEASE_VERIFY_SNAPPLEX_MESSAGE);
316             errorViews.write(sde, document);
317             return;
318         }
319 
320         if (!isUploadOnly) {
321             try {
322                 LOGGER.trace("Anaplan Write Snap - Beginning upload");
323                 importFile(anaplanModel, serverFile.getId());
324             } catch (SnapDataException e) {
325                 errorViews.write(e, document);
326             }
327         } else {
328             LOGGER.trace("Anaplan Write Snap - Upload only");
329             Map<String, Object> messageMap = new LinkedHashMap<>();
330             messageMap.put(MESSAGE_LINES, FILE_UPLOADED);
331             outputViews.write(documentUtility.newDocument(messageMap), document);
332         }

335     public void writeServerFile(File file, Iterator<Document> documentIterator, Document document,
336                                 ServerFile serverFile, ServerFileData serverFileData,
337                                 Set<JsonPath> columnNames, boolean isOverwrite)
338             throws SnapDataException {
339         LOGGER.trace("Anaplan Write Snap - Beginning writeServerFile");
340         char textDelimiter = StringUtils.isNotEmpty(serverFileData.getDelimiter()) ?
341                 serverFileData.getDelimiter().charAt(0) : UNICODE_NULL;
342         char separator = StringUtils.isNotBlank(serverFileData.getSeparator()) ?
343                 serverFileData.getSeparator().charAt(0) : TAB_CHAR;
344         int headerRow = serverFileData.getHeaderRow() == null ? 0 : serverFileData.getHeaderRow();
345         try (FileChannel tempFileChannel = new FileOutputStream(file).getChannel();
346              Document2CSVConverter document2CSVConverter = new Document2CSVConverter(
347                      tempFileChannel, separator, textDelimiter, headerRow > 0, false, columnNames,
348                      StandardCharsets.UTF_8.name(), false)) {
349             processDocument(document, document2CSVConverter);
350             while (documentIterator.hasNext()) {
351                 document = documentIterator.next();
352                 if (document != null) {
353                     processDocument(document, document2CSVConverter);
354                 }
355             }
356             serverFile.upLoad(file, isOverwrite, CHUNK_SIZE);
357         } catch (IOException e) {
358             SnapDataException sde = new SnapDataException(e, ANAPLAN_EXCEPTION)
359                     .withReason(String.format(EXCEPTION_MESSAGE, e.getMessage()))
360                     .withResolution(PLEASE_REVIEW_AND_RESOLVE_MESSAGE);
361             errorViews.write(sde, document);
362         } catch (AnaplanAPIException e) {
363             SnapDataException sde = new SnapDataException(e, ANAPLAN_API_ERROR)
364                     .withReason(String.format(ANAPLAN_API_EXCEPTION_MESSAGE, e.getMessage()))
365                     .withResolution(PLEASE_REVIEW_AND_RESOLVE_MESSAGE);
366             errorViews.write(sde, document);
367         }

387     // evaluate expression and get map key set to use for column headers
388     private Set<JsonPath> getColumnNamesFrom(Document document) {
389         LOGGER.trace("Anaplan Write Snap - Beginning getColumnNamesFrom");
390         Set<JsonPath> columnNames = Collections.emptySet();
391         if (document != null) {
392             Object object = document.get();
393             if (object instanceof Map) {
394                 Set<String> mapKeys = ((Map) object).keySet();
395                 if (CollectionUtils.isEmpty(mapKeys)) {
396                     throw new ExecutionException(INPUT_ERROR).withReason(INPUT_DOCUMENT_MAP_EMPTY)
397                             .withResolution(INPUT_DOCUMENT_MAP_EMPTY_RESOLUTION);
398                 }
399                 columnNames = mapKeys.stream()
400                         .map(JsonPaths::compileStatic)
401                         .collect(Collectors.toCollection(LinkedHashSet::new));
402             } else {
403                 throw new ExecutionException(INPUT_ERROR).withReason(INPUT_DOCUMENT_NOT_MAP)
404                         .withResolution(INPUT_DOCUMENT_NOT_MAP_RESOLUTION);
405             }
406         }
407         return columnNames;
408     }
