285     protected void processServerTask(Map<String, Object> messages, Task task, String desc)
286             throws SnapDataException {
287         LOGGER.trace("AnaplanBase: Beginning processServerTask");
288         TaskStatus status = runTilFinished(task);
289         messages.put(desc, constructStatusLineMessage(desc, status));
290         TaskResult taskResult = status.getResult();
291         if (taskResult == null) {
292             LOGGER.debug("AnaplanBase: taskResult is null");
293             throw new SnapDataException(ANAPLAN_ERROR)
294                     .withReason(INVALID_SERVER_TASK_RESULT)
295                     .withResolution(INVALID_SERVER_TASK_RESULT_RESOLUTION);
296         }
297         messages.put(TASK_ID, taskResult.getObjectId());
298         messages.put(TASK_NAME, taskResult.getObjectName());
299         List<String> details = new ArrayList<>();
300         for (TaskResultDetail detail : taskResult.getDetails()) {
301             String[] detailedResult = StringUtils.split(detail.toString(), "\n");
302             if ((detailedResult != null) && (detailedResult.length != 0)) {
303                 details.addAll(Arrays.asList(detailedResult));
304             } else {
305                 details.add(detail.toString());
306             }
307         }
308         messages.put(DETAILS, details);
309         if (isStreamErrors) {
310             streamErrors(status);
311             return;
312         }
313         List<Map<String, Object>> listMessages = new ArrayList<>();
314         boolean hadDumpFile = false;
315         String firstErrorMsg = "";
316         for (TaskResult nestedResult : taskResult.getNestedResults()) {
317             Map<String, Object> nestedMessages = new LinkedHashMap<>();
318             nestedMessages.put(TASK_ID, nestedResult.getObjectId());
319             nestedMessages.put(TASK_NAME, nestedResult.getObjectName());
320             ServerFile nestedDumpServerFile = nestedResult.getFailureDump();
321             if (nestedDumpServerFile != null) {
322                 readErrorInformation(nestedMessages, nestedDumpServerFile);
323                 hadDumpFile = true;
324             }
325             List<String> nestedDetails = new ArrayList<>();
326             List<String> errors = new ArrayList<>();
327             for (TaskResultDetail detail : nestedResult.getDetails()) {
328                 if (ANAPLAN_FAILURES.contains(detail.getType())) {
329                     errors.add(detail.toString());
330                     if (details.size() == 0 && StringUtils.isEmpty(firstErrorMsg)) {
331                         firstErrorMsg = detail.toString();
332                     }
333                 }
334                 nestedDetails.add(detail.toString());
335             }
336             if (isDataErrorWarning && errors.size() > 0) {
337                 nestedMessages.put(ERRORS, errors);
338             }
339             nestedMessages.put(DETAILS, nestedDetails);
340             listMessages.add(nestedMessages);
341         }
342         messages.put(RESULTS, listMessages);
343         if (StringUtils.isEmpty(firstErrorMsg) && details.size() == 0) {
344             firstErrorMsg = DATA_ERROR_WARNING_REASON;
345         }
346         ServerFile serverDumpFile = taskResult.getFailureDump();
347         if (serverDumpFile != null) {
348             LOGGER.debug("AnaplanBase: serverDumpFile exists");
349             readErrorInformation(messages, serverDumpFile);
350             Document errorDocument = documentUtility.newDocument(messages);
351             throw new SnapDataException(errorDocument, ANAPLAN_ERROR)
352                     .withReason(details.size() == 0 ? firstErrorMsg : details.toString())
353                     .withResolution(PLEASE_REVIEW_ERROR_OUTPUT_AND_CORRECT);
354         }
355         if (hadDumpFile) {
356             LOGGER.debug("AnaplanBase: hadDumpFile is true");
357             Document errorDocument = documentUtility.newDocument(messages);
358             throw new SnapDataException(errorDocument, ANAPLAN_ERROR)
359                     .withReason(details.size() == 0 ? firstErrorMsg : details.toString())
360                     .withResolution(PLEASE_REVIEW_ERROR_OUTPUT_AND_CORRECT);
361         }
362         if (status.getCurrentStep().contains(FAIL)) {
363             LOGGER.debug("AnaplanBase: getCurrentStep contains FAIL");
364             Document errorDocument = documentUtility.newDocument(messages);
365             throw new SnapDataException(errorDocument, ANAPLAN_ERROR)
366                     .withReason(details.size() == 0 ? firstErrorMsg : details.toString())
367                     .withResolution(PLEASE_REVIEW_ERROR_OUTPUT_AND_CORRECT);
368         }
369     }

413         LOGGER.trace("AnaplanBase: Beginning readErrors");
414         String[] headerRow = cellReader.getHeaderRow();
415         String[] dataRow = cellReader.readDataRow();
416         List<Map> errorRows = new ArrayList<>();
417         while (dataRow != null && dataRow.length > 0) {
418             Map<String, Object> errorRowMap = new LinkedHashMap<>();
419             for (int i = 0; i < dataRow.length; i++) {
420                 String fieldValue = StringUtils.defaultString(dataRow[i]);
421                 if (StringUtils.isNotBlank(headerRow[i])) {
422                     errorRowMap.put(headerRow[i], fieldValue);
423                 } else {
424                     errorRowMap.put(String.format(COLUMN_D, i), fieldValue);
425                 }
426             }
427             errorRows.add(errorRowMap);
428             dataRow = cellReader.readDataRow();
429         }
430         return errorRows;
431     }

442     private void processTaskResult(TaskResult taskResult) {
443         LOGGER.trace("AnaplanBase: Beginning processTaskResult");
444         ServerFile serverDumpFile = taskResult.getFailureDump();
445         if (serverDumpFile != null) {
446             CellReader cellReader = null;
447             try {
448                 cellReader = serverDumpFile.getDownloadCellReader();
449                 if (cellReader == null) {
450                     throw new SnapDataException(ANAPLAN_ERROR)
451                             .withReason(INVALID_CELLREADER_OBJECT_RETURNED)
452                             .withResolution(RESOLUTION_ADDRESS_ISSUE);
453                 }
454                 // prepare error and reason messages
455                 String error = String.format(ERR_ANAPLAN_TASK, taskResult.getObjectName(),
456                         taskResult.getObjectId());
457                 List<String> details = new ArrayList<>();
458                 for (TaskResultDetail detail : taskResult.getDetails()) {
459                     details.add(detail.toString());
460                 }
461                 String reason = details.toString();
462                 // parse error CSV records
463                 String[] header = parseCsvRows(cellReader.getHeaderRow());
464                 if (ArrayUtils.isNotEmpty(header)) {
465                     int headerSize = header.length;
466                     String[] errorValues = parseCsvRows(cellReader.readDataRow());
467                     while (ArrayUtils.isNotEmpty(errorValues)) {
468                         int size = NumberUtils.min(headerSize, errorValues.length);
469                         Map<String, Object> errorMap = new LinkedHashMap<>(size + 2);
470                         errorMap.put(ERROR, error);
471                         errorMap.put(REASON, reason);
472                         for (int i = 0; i < size; i++) {
473                             errorMap.put(header[i], errorValues[i]);
474                         }
475                         if (isFailErrorBehaviour) {
476                             throw new SnapDataException(error)
477                                     .withReason(reason)
478                                     .withResolution(RESOLUTION_ADDRESS_ISSUE);
479                         }
480                         errorViews.write(documentUtility.newDocument(errorMap));
481                         // get the next error record
482                         errorValues = parseCsvRows(cellReader.readDataRow());
483                     }
484                 }
485             } catch (IOException e) {
486                 throw new SnapDataException(e, ANAPLAN_ERROR)
487                         .withReason(e.getMessage())
488                         .withResolution(RESOLUTION_ADDRESS_ISSUE);
489             } catch (AnaplanAPIException e) {
490                 throw new SnapDataException(e, ANAPLAN_API_ERROR)
491                         .withReason(String.format(ANAPLAN_API_EXCEPTION_MESSAGE, e.getMessage()))
492                         .withResolution(RESOLUTION_ADDRESS_ISSUE);
493             } finally {
494                 if (cellReader != null) {
495                     try {
496                         cellReader.close();
497                     } catch (AnaplanAPIException | IOException e) {
498                         // close quietly
499                     }
500                 }
501             }
502         }
503     }

505     private String[] parseCsvRows(String[] rows) throws IOException {
506         if (ArrayUtils.isNotEmpty(rows)) {
507             List<String> items = new ArrayList<>();
508             for (String row : rows) {
509                 CSVParser csvParser = CSVParser.parse(row, CSVFormat.DEFAULT);
510                 List<CSVRecord> csvRecords = csvParser.getRecords();
511                 if (CollectionUtils.isNotEmpty(csvRecords)) {
512                     for (String item : csvRecords.get(0)) {
513                         items.add(item);
514                     }
515                 }
516             }
517             return items.toArray(new String[items.size()]);
518         }
519         return new String[0];
520     }

538     private TaskStatus runTilFinished(Task task) {
539         LOGGER.trace("AnaplanBase: Beginning runTilFinished");
540         try {
541             TaskStatus status = task.getStatus();
542             while (status.getTaskState() != TaskStatus.State.COMPLETE &&
543                     status.getTaskState() != TaskStatus.State.CANCELLED) {
544                 // if busy, nap and check again after
545                 try {
546                     Thread.sleep(SERVER_TASK_POLL_INTERVAL);
547                 } catch (InterruptedException e) {
548                     LOGGER.warn("Interrupted!", e);
549                     // Restore interrupted state...
550                     Thread.currentThread().interrupt();
551                 }
552                 status = task.getStatus();
553             }
554             return status;
