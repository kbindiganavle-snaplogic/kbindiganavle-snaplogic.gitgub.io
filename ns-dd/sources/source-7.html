247     private Connection<Insight> getLastMonthInsights() {
248         Calendar calendar = Calendar.getInstance();

251         if (StringUtils.isNotBlank(startStringDate) && StringUtils.isNotBlank(endStringDate)) {
252             calendar.setTime(startDateTime);
253             lastMonthEndDate = calendar.getTime();
254             calendar.add(Calendar.MONTH, -1);
255             lastMonthStartDate = calendar.getTime();
256         } else {
257             calendar.add(Calendar.MONTH, -1);
258             lastMonthEndDate = calendar.getTime();
259             calendar.add(Calendar.MONTH, -1);
260             lastMonthStartDate = calendar.getTime();
261         }
262         Connection<Insight> lastMonthInsights = facebookClient.fetchConnection(
263                 String.format(ADD, insightId, INSIGHTS_URL), Insight.class,
264                 Parameter.with(SINCE, lastMonthStartDate.toString()),

305      * Method will calculate base growth verses previous month

307     private void calculateBaseGrowth(Connection<Insight> lastMonthInsights) {
308         float previousMonthFans = 0;
309         List<Insight> listInsights = lastMonthInsights.getData();
310         for (int i = 0; i < listInsights.size(); i++) {
311             Insight insight = listInsights.get(i);
312             String insightField = insight.getName();
313             String insightFieldTime = insight.getPeriod();
314             if (fans > 0 && insightField.equals(PAGE_FANS)
315                     && insightFieldTime.equals(OPERATION_LIFETIME)) {
316                 previousMonthFans = insight.getValues().get(0).getInt(VALUE);
317             }
318         }
319         float percentGrowth = 0.00f;
320         if (fans > 0 && previousMonthFans > 0) {
321             percentGrowth = (((fans - previousMonthFans) / previousMonthFans) * 100);
322         }
323         map.put(BASE_GROWTH, percentGrowth);
324     }

341     private void getGenericData(String time, String insightField,
342             String insightFieldTime, JsonObject key) {
343         int talkingAbout = 0;
344         if (insightFieldTime.equals(time)) {
345             switch (insightField) {
346                 case PAGE_IMPRESSIONS_UNIQUE: {
347                     reach = key.getInt(VALUE);
348                     map.put(REACH, reach);
349                     break;
350                 }
351                 case PAGE_IMPRESSIONS: {
352                     facebookUtil.getPageImpressions(map, key);
353                     break;
354                 }
355                 case PAGE_IMPRESSIONS_ORGANIC_UNIQUE: {
356                     map.put(ORGANIC_REACH, key.get(VALUE));
357                     break;
358                 }
359                 case PAGE_POST_IMPRESSIONS: {
360                     map.put(REACH_BY_POST, key.get(VALUE));
361                     break;
362                 }
363                 case PAGE_IMPRESSIONS_VIRAL_UNIQUE: {
364                     map.put(VIRAL_REACH, key.get(VALUE));
365                     break;
366                 }
367                 case PAGE_POSTS_IMPRESSIONS_PAID_UNIQUE: {
368                     map.put(PAID_REACH, key.get(VALUE));
369                     break;
370                 }
371                 case PAGE_NEGATIVE_INDEX: {
372                     int negativeFeed = 0;
373                     negativeFeed = key.getInt(VALUE);
374                     if (negativeFeed > 0 && fans > 0) {
375                         double relevanceIndex = fans / negativeFeed;
376                         map.put(RELEVANCE_INDEX, relevanceIndex);
377                     }
378                     break;
379                 }
380                 case PAGE_STORY_TELLERS: {
381                     talkingAbout = key.getInt(VALUE);
382                     map.put(TALKING_ABOUT, talkingAbout);
383                     break;
384                 }
385                 case PAGE_ENGAGED_USERS: {
386                     int engagedUser = 0;
387                     engagedUser = key.getInt(VALUE);
388                     if (engagedUser > 0 && reach > 0) {
389                         float engIndex = (engagedUser * 100) / reach;
390                         map.put(ENGAGEMENT_INDEX, engIndex);
391                     }
392                     break;
393                 }
394             }
395         }
396         if (insightFieldTime.equals(OPERATION_LIFETIME) && insightField.equals(PAGE_FANS)) {
397             fans = key.getInt(VALUE);
398             map.put(FOLLOWERS_LIFETIME, key.get(VALUE));
399         }
400         if (talkingAbout > 0 && reach > 0) {
401             if (insightFieldTime.equals(time)) {
402                 float result = (talkingAbout * 100) / reach;
403                 map.put(VIRALITY, Float.toString(result));
404             }
405         }
406     }

408     private void getFrequency(String time, String insightField,
409             String insightFieldTime, JsonObject key) {
410         if (insightField.equals(FREQUENCY_DISTRIBUTION)

419                 }
420                 map.put(UNIQUE_USERS_BY_FREQUENCY, frequencyMap);
421             }
422         }
