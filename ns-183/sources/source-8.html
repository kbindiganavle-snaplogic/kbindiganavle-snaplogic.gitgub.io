274     protected void process(Document document, String inputViewName) {
275         writeOutputView = false;
276         String zObject = getString(zObjectExpr.eval(document));
277         try {
278             if (StringUtils.isBlank(zObject)) {
279                 throw new SnapDataException(ERR_EMPTY_ZOBJECT)
280                         .withReason(REASON_ZOBJECT_REQUIRED)
281                         .withResolution(RESOLUTION_ADDRESS_ISSUE);
282             }
283             boolean isActionQuery = ZOBJECT_ACTION_QUERY.contains(zObject.toLowerCase());
284             // initialize OpenAPI related data
285             initOpenAPI(document);
286             // get OpenAPI base address from account
287             initOpenApiBase();
288             // prepare select query
289             String outputFields = getOutputFields(document, isActionQuery, zObject);
290             String query = String.format(SELECT_QUERY, outputFields, zObject);
291             String whereClause = getString(whereClauseExpr.eval(document));
292             if (StringUtils.isNotBlank(whereClause)) {
293                 query = query.concat(String.format(WHERE_CLAUSE, whereClause));
294             }
295             Object limitClause = limitClauseExpr.eval(document);
296             if (limitClause != null && !isActionQuery) {
297                 query = query.concat(String.format(LIMIT_CLAUSE, limitClause));
298             }
299             // if the number of retries is larger than 0, create a temp file
300             if (isStaging) {
301                 File tmpDir = getTmpDir(expressionUtils, propertyValues, errorViews);
302                 tmpFile = File.createTempFile(ZUORA_REST_READ, GZIP, tmpDir);
303             }
304             if (isActionQuery) {
305                 actionQuery(query, document);
306                 return;
307             }
308             // submit a batch job to execute the select query
309             String jobId = submitSelectQueryJob(query, zObject, document);
310             // check the status of the submitted batch job
311             String dataFile = checkJobStatus(jobId, document);
312             // download and parse the batch job result
313             try (GZIPInputStream gzipInputStream = getInputStreamFailSafe(dataFile)) {
314                 // parse JSON and stream the records to output view
315                 Iterator<Object> iterator = new JsonDataIterator(gzipInputStream);
316                 Object outData = null;
317                 boolean isFirst = true;
318                 while (iterator.hasNext()) {
319                     outData = iterator.next();
320                     writeOutputDocument(outData, document);
321                     // Save column names in the cache so that it can be used for the output
322                     // field suggestion. It is necessary because /v1/describe/{object} may have
323                     // missing fields.
324                     if (isFirst && outData instanceof Map &&
325                             StringUtils.startsWithIgnoreCase(query, SELECT_STAR_FROM)) {
326                         Set<String> columns = ((Map<String, Object>) outData).keySet();
327                         COLUMN_NAMES_CACHE.put(zObject,
328                                 columns.stream().sorted().collect(Collectors.toList()));
329                         isFirst = false;
330                     }
331                 }
332             }
333         } catch (SuggestViewAbortException e) {
334             throw e;
335         } catch (SnapDataException e) {
336             errorViews.write(e, document);
337         } catch (Exception e) {
338             errorViews.write((SnapDataException)
339                     new SnapDataException(e, ERR_READ_OBJECT)
340                             .formatWith(zObject)
341                             .withResolution(RESOLUTION_ADDRESS_ISSUE), document);
342         } finally {
343             if (tmpFile != null) {
344                 tmpFile.delete();
345                 tmpFile = null;
346             }
347         }
348     }

377     protected String submitSelectQueryJob(String query, String zObject, Document document)
378             throws Exception {
379         // prepare query job
380         openApiPath = QUERY_JOBS;
381         operation = HttpPost.METHOD_NAME.toLowerCase();
382         Map<String, Object> inputMap = ImmutableMap.of(REQUEST_BODY,
383                 ImmutableMap.of(QUERY, query,
384                         COMPRESSION, GZIP,
385                         OUTPUT_FORMAT, JSON.toUpperCase(),
386                         OUTPUT, ImmutableMap.of(TARGET, S3)));
387         // submit query job
388         outDoc = process(inputMap, document);
389         // get job ID
390         Exception ex = null;
391         try {
392             Object value = compile(ENTITY_DATA_ID).read(outDoc.get(Map.class));
393             if (value instanceof String && StringUtils.isNotBlank((String) value)) {
394                 return  (String) value;
395             }
396         } catch (InvalidPathException e) {
397             ex = e;
398         }
399         throw new SnapDataException(ex, ERR_JOB_ID)

401                 .withReason(REASON_JOB_ID_NOT_FOUND)
402                 .withResolutionAsDefect();

405     private String checkJobStatus(String jobId, Document document) throws Exception {
406         openApiPath = QUERY_JOBS_ID;
407         operation = HttpGet.METHOD_NAME.toLowerCase();
408         Map<String, Object> inputMap = new HashMap<>();
409         compile(PARAM_PATH_JOB_ID)
410                 .write(inputMap, jobId);
411         int interval = 1;
412         String dataFile = null;
413         while (StringUtils.isBlank(dataFile)) {
414             sleepNoInterrupt(interval, TimeUnit.SECONDS, LOG);
415             Document outDoc = process(inputMap, document);
416             Map<String, Object> resultMap = outDoc.get(Map.class);
417             Object value = getValueFrom(resultMap, QUERY_STATUS);
418             String status = value instanceof String ? (String) value : null;
419             switch (status) {
420                 case COMPLETED:
421                     // get the resulting file URL for the batch job
422                     value = getValueFrom(resultMap, ENTITY_DATA_FILE);
423                     dataFile = value instanceof String ? (String) value : null;
424                     break;
425                 case FAILED:
426                     throw new IOException(String.format(ERR_ERROR_CODE,
427                             getValueFrom(resultMap, ENTITY_ERR_MSG),
428                             getValueFrom(resultMap, ENTITY_ERR_CODE)));
429                 case CANCELLED:
430                     throw new SnapDataException(ERR_JOB_CANCELLED)
431                             .formatWith(jobId)
432                             .withReason(REASON_JOB_STOPPED)
433                             .withResolution(RESOLUTION_ADDRESS_ISSUE);
434                 case SUBMITTED:
435                 case ACCEPTED:
436                 case IN_PROGRESS:
437                 default:
438                     // No-Op
439             }
440             interval = interval * 2;
441             if (interval > TIMEOUT_SECONDS) {
442                 throw new SnapDataException(ERR_QUERY_JOB_TIMEOUT)
443                         .formatWith(status)
444                         .withReason(REASON_TAKING_TOO_LONG)
445                         .withResolution(RESOLUTION_ADDRESS_ISSUE);
446             }
447         }
448         return dataFile;
449     }

459     @VisibleForTesting
460     protected GZIPInputStream getGZIPInputStream(String dataFile) throws IOException {
461         InputStream inputStream = null;
462         GZIPInputStream gzipInputStream = null;
463         try {
464             HttpUriRequest httpUriRequest = new HttpGet(dataFile);
465             HttpResponse httpResponse = HTTP_CLIENT.execute(httpUriRequest);
466             int statusCode = httpResponse.getStatusLine().getStatusCode();
467             if (statusCode == SC_OK) {
468                 inputStream = httpResponse.getEntity().getContent();
469                 gzipInputStream = getGZIPInputStream(inputStream);
470             } else {
471                 throw new IOException(ERR_DOWNLOAD_JOB_RESULT);
472             }
473         } catch (IOException e) {
474             IOUtils.closeQuietly(inputStream);
475             IOUtils.closeQuietly(gzipInputStream);
476             throw e;
477         }
478         return gzipInputStream;
479     }

584     private void actionQuery(String query, Document document) throws Exception {
585         openApiPath = ACTION_QUERY;
586         operation = HttpPost.METHOD_NAME.toLowerCase();
587         Map<String, Object> inputMap = new HashMap<>();
588         compile(REQUEST_BODY_QUERY_STRING)
589                 .write(inputMap, query);
590         String queryLocator;
591         boolean isFirst = true;
592         do {
593             // execute the select query and download the records
594             outDoc = process(inputMap, document);
595             Map<String, Object> outData = outDoc.get(Map.class);
596             List<Object> records = compile(ENTITY_RECORDS).read(outData);
597             for (Object record : records) {
598                 writeOutputDocument(record, document);
599             }
600             // prepare to download the next page
601             queryLocator = (String) getValueFrom(outData, ENTITY_QUERY_LOCATOR);
602             if (isFirst) {
603                 openApiPath = ACTION_QUERY_MORE;
604                 isFirst = false;
605             }
606             if (StringUtils.isNotBlank(queryLocator)) {
607                 inputMap.clear();
608                 compile(REQUEST_BODY_QUERY_LOCATOR)
609                         .write(inputMap, queryLocator);
610             }
611         } while (StringUtils.isNotBlank(queryLocator));
612     }

619     private TreeSet<String> getTableSet(PropertyValues propertyValues) {
620         TreeSet<String> tables = null;
621         try {
622             configure(propertyValues);
623             initOpenAPI(null);
624             if (authHeader != null) {
625                 String token = authHeader.getValue();
626                 tables = TABLE_NAMES_CACHE.get(token);
627                 if (CollectionUtils.isEmpty(tables)) {
628                     isStaging = false;
629                     tables = tables == null ? new TreeSet<>() : tables;
630                     initOpenApiBase();
631                     // submit a batch job to execute the 'show tables' query
632                     String jobId = submitSelectQueryJob(SHOW_TABLES, null, null);
633                     // check the status of the submitted batch job
634                     String dataFile = checkJobStatus(jobId, null);
635                     // download and parse the batch job result
636                     try (GZIPInputStream gzipInputStream = getInputStreamFailSafe(dataFile)) {
637                         // parse JSON and stream the records to output view
638                         Iterator<Object> iterator = new JsonDataIterator(gzipInputStream);
639                         while (iterator.hasNext()) {
640                             Object data = iterator.next();
641                             if (data instanceof Map) {
642                                 tables.add(((Map<String, String>) data).get(TABLE));
643                             }
644                         }
645                     }
646                     tables.addAll(ZOBJECT_ACTION_QUERY);
647                     synchronized (TABLE_NAMES_CACHE) {
648                         TABLE_NAMES_CACHE.put(token, tables);
649                     }
650                 }
651             }
652         } catch (Exception e) {
653             throw new ConfigurationException(e, ERR_TABLE_NAMES)
654                     .withReason(e.getMessage())
655                     .withResolution(RESOLUTION_ADDRESS_ISSUE);
656         }
657         if (CollectionUtils.isEmpty(tables)) {
658             throw new ConfigurationException(ERR_TABLE_NAMES)
659                     .withReason(REASON_UNKNOWN)
660                     .withResolution(RESOLUTION_ADDRESS_ISSUE);
661         }
662         return tables;
663     }
