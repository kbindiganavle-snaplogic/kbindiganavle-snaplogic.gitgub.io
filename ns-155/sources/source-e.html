215      * Starts GetQueryIDThread thread, which runs infinitely until snap is stopped.
216      * GetQueryIDThread is responsible for retrieving query IDs from snowflake server.
217      *
218      * @param conn Connection object for Snowflake
219      */
220     protected void startQueryIDThread(Connection conn) {
221         try {
222             queryIDThreadStarted = true;
223             if (Objects.isNull(conn)) {
224                 conn = jdbcOperations.acquireConnection(account);
225             }
226             queryQueue = new LinkedBlockingDeque<>();
227             if (JdbcOperations.SNOWFLAKE.equals(getName())) {
228                 ((SnowflakeOperations) jdbcOperations).setQueryQueue(queryQueue);
229             }
230             queryIDQueue = EvictingQueue.create(1);
231             // Get the session id of current session which can be used in
232             // information_schema.query_history_by_session command for getting information about
233             // the queries running in current session
234             try (Statement statement = conn.createStatement();
235                  ResultSet rs = statement.executeQuery(GET_CURRENT_SESSION)) {
236                 if (rs.next()) {
237                     sessionID = rs.getString(COLUMN_CURRENT_SESSION);
238                 }
239             }
240             createThreadObject();
241             getQueryID.start();
242         } catch (Exception e) {
243             LOGGER.error(ERROR_UNABLE_TO_START_CANCEL_QUERY, e);
244         }
245     }

257      * This method is invoked when snap is stopped/failed/closed
258      * Sends system$cancel_query command to Snowflake.
259      * If the query status is currently running, it gets cancelled.
260      *
261      * @param conn Connection object for Snowflake
262      */
263     protected void stopQuery(Connection conn) {
264         try {
265             LOGGER.debug("Trying to kill the Execute operation");
266             if (!stopedExecuted) {
267                 stopedExecuted = true;
268                 stopQueryIDThread();
269                 SnowflakeQueryData queryInfo = queryIDQueue.poll();
270                 if (Objects.nonNull(queryInfo)) {
271                     String queryIDStop = queryInfo.getQueryID();
272                     if (Objects.isNull(conn)) {
273                         conn = jdbcOperations.acquireConnection(account);
274                     }
275                     try (Statement stmt = conn.createStatement()) {
276                         try (ResultSet resultSet = stmt.executeQuery(String.format
277                                 (FORMAT_CANCEL_QUERY, queryIDStop))) {
278                             //It always retrieves one result
279                             if (resultSet.next()) {
280                                 String status = resultSet.getString(1);
281                                 LOGGER.debug("Message: {} while killing the QueryID: {}", status,
282                                         queryIDStop);
283                             } else {
284                                 LOGGER.debug("No result found while killing the QueryID: {}",
285                                         queryIDStop);
286                             }
287                         }
288                     }
289                 }
290             }

292             LOGGER.error(ERROR_UNABLE_TO_CANCEL_QUERY, e);

487     @Override
488     protected final void process(final Document document, final String inputViewName) {
489         try {
490             processDocument(document, inputViewName);
491         } catch (SuggestViewAbortException | StagingException e) {
492             throw e;
493         } catch (ExecutionException executionException) {
494             if (DIFFERENT_REGION_CONFIGURATION_EXCEPTION
495                     .matcher(executionException.getMessage()).find()) {
496                 throw executionException;
497             }
498             handleSnapException(executionException, document);
499         } catch (SnapException e) {
500             handleSnapException(e, document);
501         } catch (Throwable e) {
502             String reason = e.getCause() != null ? e.getCause().getMessage() : null;
503             reason = reason == null ? String.format(ERR_EXECUTE_SNAP, getName()) : reason;
504             Map<String, Object> wrapper = Maps.newHashMapWithExpectedSize(5);
505             wrapper.put(ERROR, e.getMessage());
506             wrapper.put(STACKTRACE, Throwables.getStackTraceAsString(e.getCause() != null ?
507                     e.getCause() : e));
508             wrapper.put(REASON, reason);
509             wrapper.put(RESOLUTION, PLEASE_ADDRESS_ISSUE);
510             SnapDataException ex = new SnapDataException(documentUtility.newDocument(wrapper), e,
511                     e.getMessage())
512                     .withReason(reason)
513                     .withResolution(PLEASE_ADDRESS_ISSUE);
514             errorViews.write(ex, document);
515         }
516     }

578     @Override
579     public void handle(final LifecycleEvent event) {
580         switch (event) {
581             case FAILURE:
582                 onSnapRunFailure(); // Note: fall through is expected
583             case STOP:
584             case CLOSE:
585             case SUCCESS:
586                 if (jdbcOperations == null) {
587                     LOGGER.debug("(+) jdbcOperations was null");
588                     return;
589                 }
590                 LOGGER.debug("(+) Stopping the QueryIDThread: {}", getQueryID);
591                 stopQueryIDThread();
592                 // get the list of open connections and their corresponding stack traces
593                 Map<ConnectionProxy, String> openConnections = SnapConnection.getOpenConnections();
594                 UUID snapID = jdbcOperations.getSnapID();
595                 if (countOpenSnapsForID(snapID) > 0) {
596                     Set<ConnectionProxy> connectionSet = openConnections.keySet();
597                     Iterator<ConnectionProxy> keyIterator = connectionSet.iterator();
598                     while (keyIterator.hasNext()) {
599                         ConnectionProxy snapConnection = keyIterator.next();
600                         try {
601                             if (snapConnection.hasConnection()
602                                     && snapID == snapConnection.getSnapID()) {
603                                 if (LOGGER.isDebugEnabled()) {
604                                     LOGGER.debug("(+) closing possible leaked connection: @{} {}",
605                                             Integer.toHexString(snapConnection
606                                                     .getConnectionHashCode()),
607                                             openConnections.get(snapConnection));
608                                 }
609                                 keyIterator.remove();
610                                 if (!snapConnection.isClosed()) {
611                                     snapConnection.close();
612                                 } else {
613                                     LOGGER.debug("(+) Connection already closed.");
614                                 }
615                             }
616                         } catch (SQLException e) {
617                             if (snapConnection.hasConnection()) {
618                                 LOGGER.debug("(+) Unable to close connection: @{} {}",
619                                     Integer.toHexString(snapConnection.getConnectionHashCode()),
620                                         openConnections.get(snapConnection));
621                             } else {
622                                 LOGGER.debug("(+) Unable to close connection: @null");
623                             }
624                         }
625                     }
626                 } else {
627                     LOGGER.debug("(+) No leaked connections detected for {}", snapID);
628                 }
629                 break;
630         }
631     }

686      * schema properties. If no schema is specified in the table property, then adds the

693     protected String updateTableName(String schemaName, String tableName) {
694         if (StringUtils.isBlank(tableName)) {
695             return tableName;
696         }
697         String[] tableParts = quotationHandler.splitFullyQualifiedName(tableName, 2);
698         if (StringUtils.isNotBlank(schemaName)) {
699             if (tableParts.length == 1) {
700                 return String.format(TABLE_FORMAT, schemaName, tableName);
701             }
702             if (!schemaName.equals(tableParts[0])) {
703                 // PM has decided to ignore the Schema name property and when the schemaName
704                 // is different from the schema part of the Table name property.
705                 this.schemaName = tableParts[0];
706             }
707         } else {
708             if (tableParts.length > 1) {
709                 schemaNameProvidedInTargetTable = tableParts[0];
710             }
711         }
712         return tableName;
713     }
