317     protected Map<String, Object> processNaturalKeys(
318             Map<String, Object> data, List<String> naturalKeys) {
319         Map<String, Object> naturalKeyMap = new LinkedHashMap<>(naturalKeys.size());
320         List<String> predicates = new ArrayList<>(naturalKeys.size());
321         for (String naturalKey : naturalKeys) {
322             if (!data.containsKey(naturalKey)) {
323                 String error = String.format(ERR_NATURAL_KEY_VALUE_NOT_FOUND, naturalKey);
324                 writeErrorView(data, error, RESOLUTION_NATURAL_KEY_VALUE);
325             }
326             boolean isUpperCaseNaturalKey = isUpperCaseNaturalKey(naturalKey);
327             naturalKeyMap.put(naturalKey, getStringValue(data, naturalKey, true));
328             if (data.get(naturalKey) instanceof Boolean) {
329                 if (!isUpperCaseNaturalKey) {
330                     predicates.add(
331                             String.format(FORMAT_NON_UPPER_CASE_BOOLEAN_PREDICATE,
332                                     naturalKey, data.get(naturalKey)));
333                 } else {
334                     predicates.add(
335                             String.format(FORMAT_BOOLEAN_PREDICATE,
336                                     naturalKey, data.get(naturalKey)));
337                 }
338             } else {
339                 if (!isUpperCaseNaturalKey) {
340                     predicates.add(String.format(FORMAT_NON_UPPER_CASE_PREDICATE, naturalKey));
341                 } else {
342                     predicates.add(String.format(FORMAT_PREDICATE, naturalKey));
343                 }
344             }
345         }
346         Map<String, Object> naturalKeyData = new LinkedHashMap<>();
347         naturalKeyData.put("naturalKeyMap", naturalKeyMap);
348         naturalKeyData.put("predicates", predicates);
349         return naturalKeyData;

353     protected void processDocument(Document document, String inputViewName) {
354         configureRetry(document);
355         data = getMapDataFrom(document);
356         if (data == null) {
357             return;
358         }
359         // initialize current row and value
360         initCurrentRow();
361         // get the natural key map and a list of predicates
362         Map<String,Object> naturalKeyData = processNaturalKeys(data, naturalKeys);
363         Map<String, Object> naturalKeyMap =
364                 (Map<String, Object>) naturalKeyData.get("naturalKeyMap");
365         List<String> predicates = (List<String>) naturalKeyData.get("predicates");
366         // format an and-clause string: key1=? AND key2=? AND ...
367         String andClause = String.format(FORMAT_PARENTHESIS, Joiner.on(AND).join(predicates));
368         // store bind values
369         bindValues.addAll(naturalKeyMap.values());
370         // Converting Date fields to its long time value in Natural Key
371         if (SnowflakeDateFormatOptions.AUTO_FORMAT_DATE_TIMESTAMP.equals(dateFormatOptions)) {
372             for (Map.Entry<String, Object> entry : naturalKeyMap.entrySet()) {
373                 Object input = convertDateAndTimestampToDefault(entry.getValue(),
374                         entry.getKey());
375                 if (dateColumns.containsKey(entry.getKey())) {
376                     entry.setValue(DateUtils.convertDateToTime(input,
377                             dateColumns.get(entry.getKey())));
378                 } else if (dateColumns.containsKey(entry.getKey().toUpperCase())) {
379                     entry.setValue(DateUtils.convertDateToTime(input,
380                             dateColumns.get(entry.getKey().toUpperCase())));
381                 }
382             }
383         }
384         // store input data so that it can be correlated to its lookup result
385         List<InputData> inputDataList = scdInputRecords.get(naturalKeyMap);
386         if (inputDataList == null) {
387             inputDataList = Lists.newArrayList();
388             scdInputRecords.put(naturalKeyMap, inputDataList);
389         }
390         inputDataList.add(new InputData(document));
391         // append an and-clause to the query builder
392         queryBuilder.append(andClause);
393         // if the SQL statement approaches the maximum size, process the lookup operation
394         if (inImmediateMode || (bindValues.size() + naturalKeys.size() > MAX_PARAMETERS)) {
395             processQuery(document);
396             // re-initialize query string builder and input records storage
397             queryBuilder.delete(initialQuerySize, queryBuilder.length());
398             bindValues.clear();
399             scdInputRecords.clear();
400         }
401         // acknowledge document
402         document.acknowledge();
403     }

405     @Override
406     protected void processQuery(final Document document) {
407         Map<String, Object> inputDataKey = new HashMap<>(naturalKeys.size());
408         queryBuilder.delete(queryBuilder.lastIndexOf(OR), queryBuilder.length());
409         queryBuilder.append(String.format(FORMAT_CURRENT_ROW, currentRowField, currentRowValue));
410         currentRowField = QUOTATION_HANDLER.unescape(currentRowField);
411         String sql = queryBuilder.toString();
412         try (DatabaseCursor cursor = jdbcOperations.executeQuery(sql, bindValues, account,
413                 maxRetries, retryInterval)) {
414             while (cursor.hasNext()) {
415                 Map<String, Object> dbData = cursor.fetchOne();
416                 // get the original input data which corresponds to the result
417                 for (String naturalKey : naturalKeys) {
418                     inputDataKey.put(naturalKey, getStringValue(dbData, naturalKey, false));
419                 }
420                 List<InputData> inputDataList = scdInputRecords.get(inputDataKey);
421                 if (inputDataList == null) {
422                     throw new SnapDataException(ERR_INPUT_DATA_NOT_FOUND)
423                             .withReason(REASON_INPUT_DATA_TYPE)
424                             .withResolution(RESOLUTION_INPUT_DATA_TYPE);
425                 }
426                 historizeInputDocuments(inputDataList, dbData, document);
427             }
428         } catch (SnapDataException e) {
429             errorViews.write(e, document);
430             return;
431         } catch (ExecutionException e) {
432             throw e;
433         } catch (SuggestViewAbortException e) {
434             // Max number of output documents produced in preview, should end gracefully
435             return;
436         } catch (Exception e) {
437             throw new ExecutionException(e, CLOSE_CURSOR_FAILED)
438                     .withReason(e.getMessage())
439                     .withResolution(PLEASE_ENSURE_DATABASE_ACCESSIBLE);
440         }
441         // check if there is any input data which did not produce any result
442         for (Map.Entry<Map<String, Object>, List<InputData>> entry : scdInputRecords.entrySet()) {
443             List<InputData> inputDataSet = entry.getValue();
444             // As we are processing all the inputDataSet documents at a time,
445             // Checking first inputData.isResultsProduced() will be enough
446             if (inputDataSet != null && !inputDataSet.isEmpty()
447                     && !inputDataSet.get(0).isResultsProduced()) {
448                 historizeInputDocuments(inputDataSet, null, document);
449             }
450         }
451     }

454      * This is to iterate over all the input documents of Single natural key
455      * and historize them

457     protected void historizeInputDocuments(List<InputData> inputDataList,
458             Map<String, Object> dbData, Document document) {
459         Document inputDoc = null;
460         //This is to evaluate ignoreUnchangedRows properly in case of multiple
461         // same Natural key
462         boolean atleastOneChange = false;
463         for (InputData inputData : inputDataList) {
464             inputData.setResultsProduced(true);
465             // see if any value of Cause-historization fields has changed
466             Map<String, Object> inputMap = inputData.getData();
467             inputDoc = inputData.getDocument();
468             Map<String, String> attributes = Maps.newHashMap();
469             if (attributeExpressions != null) {
470                 attributeExpressions.keySet().forEach(key -> {
471                     String name = QUOTATION_HANDLER.unescape(key.eval(document));
472                     String order = QUOTATION_HANDLER.unescape(attributeExpressions.get(key).eval
473                             (document));
474                     attributes.put(name, order);
475                 });
476             }
477             boolean changed = false;
478             if (dbData != null) {
479                 for (String key : causeFields) {
480                     Object inputValue = inputMap.get(key);
481                     Object dbValue = dbData.get(key);
482                     if (verifyInEquality(dbValue, inputValue, key)) {
483                         changed = true;
484                         break;
485                     }
486                 }
487                 // remove 'identity' columns if any
488                 for (String column : identityColumns) {
489                     dbData.remove(column);
490                 }
491 
492                 if (changed && checkAndHistorizeData(dbData, inputData, inputMap, attributes)) {
493                     continue;
494                 }
495                 // write output view documents
496                 if (changed) {
497                     atleastOneChange = true;
498                     // create a new current row
499                     Map<String, Object> newRow = new LinkedHashMap<>(dbData);
500                     updateMap(newRow, inputMap);
501                     // set the current row to historical
502                     dbData.put(currentRowField, historicalRowValue);
503                     // set the end date on the historical row
504                     setDate(dbData, inputMap, END_DATE_LABEL);
505                     // write the historical row to output view - we should write this historical
506                     // row before new row in the case that more than one change event happens in a
507                     // day for one id value and the start date is part of the natural keys
508                     outputViews.write(documentUtility.newDocumentFor(inputDoc,
509                             dbData));
510                     // set the start date on the new row
511                     setDate(newRow, inputMap, START_DATE_LABEL);
512                     dbData = newRow;
513                 } else if (!ignoreUnchangedRows) {
514                     // update an existing current row
515                     updateMap(dbData, inputMap);
516                 }
517 
518             } else {
519                 Map<String, Object> newRow = new LinkedHashMap<>(inputMap);
520                 atleastOneChange = true;
521                 // set the current row
522                 newRow.put(currentRowField, currentRowValue);
523                 // set the start data
524                 setDate(newRow, inputMap, START_DATE_LABEL);
525                 dbData = newRow;
526             }
527 
528         }
529         if (inputDoc != null && (atleastOneChange || !ignoreUnchangedRows)) {
530             //write the last row to output view
531             outputViews.write(documentUtility.newDocumentFor(inputDoc,
532                     dbData));
533         }
534     }

536     public boolean verifyInEquality(Object dbValue, Object inputValue, String columnName) {
537         boolean changed = false;
538         if (dbValue != null && inputValue != null &&
539                 ObjectUtils.notEqual(dbValue.getClass(), inputValue.getClass())) {
540             try {
541                 switch (dbValue.getClass().getName()) {
542                     case BIGINTEGER:
543                         inputValue = new BigInteger(inputValue.toString());
544                         break;
545                     case DOUBLE:
546                         inputValue = new Double(inputValue.toString());
547                         break;
548                     case BIGDECIMAL:
549                         int dbScale = ((BigDecimal) dbValue).scale();
550                         inputValue = new BigDecimal(inputValue.toString()).setScale(dbScale);
551                         break;
552                     case JODALOCALTIME:
553                         Long dbTimeinMillis = ((LocalTime) dbValue)
554                                 .toDateTime(Instant.EPOCH).getMillis();
555                         inputValue = Time.valueOf(inputValue.toString());
556                         Long inputTimeinMillis = ((Time) inputValue).getTime();
557                         // Add local timezone milliseconds difference to UTC milliseconds
558                         // of the same time.
559                         dbTimeinMillis += ((Time) inputValue).getTimezoneOffset() * 60 * 1000;
560                         return ObjectUtils.notEqual(dbTimeinMillis, inputTimeinMillis);
561                     case TIME:
562                         inputValue = Time.valueOf(inputValue.toString());
563                         break;
564                     case JODALOCALDATE:
565                     case JODADATETIME:
566                     case DATE:
567                     case TIMESTAMP:
568                         Date dateDBValue = null;
569                         if (JODALOCALDATE.equals(dbValue.getClass().getName())) {
570                             dateDBValue = ((LocalDate) dbValue).toDateTimeAtStartOfDay().toDate();
571                         } else if (JODADATETIME.equals(dbValue.getClass().getName())) {
572                             dateDBValue = ((DateTime) dbValue).toDate();
573                         } else {
574                             dateDBValue = (Date) dbValue;
575                         }
576                         // We always get a Joda DateTime Object from a input document,
577                         // But if we get String object, then converting to Joda DataTime
578                         // and proceeding.
579                         inputValue = convertDateAndTimestampToDefault(inputValue
580                                 , columnName);
581                         if (inputValue instanceof String || inputValue instanceof LocalDate) {
582                             inputValue = DateTime.parse(inputValue.toString());
583                         }
584                         Date dateInputValue = null;
585                         if (inputValue instanceof LocalDateTime) {
586                             dateInputValue = ((LocalDateTime) inputValue).toDate();
587                         } else {
588                             dateInputValue = ((DateTime) inputValue).toDate();
589                         }
590                         return (ObjectUtils.notEqual(dateInputValue.getTime(), dateDBValue
591                                 .getTime()));
592                     default:
593                 }
594             } catch (NumberFormatException ex) {
595                 changed = true;
596             } catch (IllegalArgumentException iaex) {
597                 throw new SnapDataException(iaex, FAILED_TO_CONVERT)
598                         .withResolution(PLEASE_ADDRESS_ISSUE);
599             }
600         }
601         if (ObjectUtils.notEqual(inputValue, dbValue)) {
602             changed = true;
603         }
604         // This is to handle case where both input and dbValue are Jodata DateType but different
605         // format
606         if (inputValue instanceof DateTime) {
607             long inputVal = ((DateTime) inputValue).toDate().getTime();
608             long dbVal = ((DateTime) dbValue).toDate().getTime();
609             changed = (inputVal != dbVal);
610 
611         }
612         return changed;
613     }

782             inputData, final Map<String, Object> inputMap, final Map<String, String> attributes) {
783         for (String field : attributes.keySet()) {
784             Object inputValue = inputMap.get(field);
785             Object dbValue = dbData.get(field);
786             // if there are any null values we historize the values as a default behaviour
787             if (inputValue == null || dbValue == null) {
788                 return true;
789             }
790             if (dbValue instanceof BigInteger) {
791                 //This if conditions compares for both ascending and descending values for each
792                 // check.
793                 if ((ASCENDING_ORDER.equalsIgnoreCase(attributes.get(field)) && ((BigInteger)
794                         dbValue).intValue() > ((BigInteger) inputValue).intValue()) ||
795                         (DESCENDING_ORDER.equalsIgnoreCase(attributes.get(field)) && (
796                                 (BigInteger) dbValue).intValue() < ((BigInteger) inputValue)
797                                 .intValue())) {
798                     historizeDocument(inputData, inputMap);
799                     return true;
800                 }
801             } else if (dbValue instanceof BigDecimal) {
802                 if ((ASCENDING_ORDER.equalsIgnoreCase(attributes.get(field)) && ((BigDecimal)
803                         dbValue).doubleValue() > ((BigDecimal) inputValue).doubleValue()) ||
804                         (DESCENDING_ORDER.equalsIgnoreCase(attributes.get(field)) && ((BigDecimal)
805                                 dbValue).doubleValue() < ((BigDecimal) inputValue).doubleValue())) {
806 
807                     historizeDocument(inputData, inputMap);
808                     return true;
809                 }
810             } else if (dbValue instanceof Double) {
811                 if ((ASCENDING_ORDER.equalsIgnoreCase(attributes.get(field)) && (Double) dbValue
812                         > (Double) inputValue) || (DESCENDING_ORDER.equalsIgnoreCase(attributes
813                         .get(field)) && (Double) dbValue < (Double) inputValue)) {
814                     historizeDocument(inputData, inputMap);
815                     return true;
816                 }
817             } else if (dbValue instanceof java.util.Date) {
818                 java.util.Date dateDBValue = (java.util.Date) converterRegistry.from(dbValue, 1);
819                 java.util.Date dateInputValue;
820                 if (inputValue instanceof java.util.Date) {
821                     dateInputValue = (java.util.Date) converterRegistry.from(inputValue, 1);
822                 } else {
823                     //For Joda DateTime Object from a input document
824                     dateInputValue = ((DateTime) inputValue).toDate();
825                 }
826                 if (checkTimeObject(attributes.get(field), dateInputValue.getTime(), dateDBValue
827                         .getTime())) {
828                     historizeDocument(inputData, inputMap);
829                     return true;
830                 }
831             } else if (dbValue instanceof DateTime) {
832                 DateTime dateDBValue = (DateTime) converterRegistry.from(dbValue, 1);
833                 //We always get a Joda DateTime Object from a input document
834                 java.util.Date dateInputValue = ((DateTime) inputValue).toDate();
835                 if (checkTimeObject(attributes.get(field), dateInputValue.getTime(), dateDBValue
836                         .toDate().getTime())) {
837                     historizeDocument(inputData, inputMap);
838                     return true;
839                 }
840             } else if (dbValue instanceof String) {
841                 if ((ASCENDING_ORDER.equalsIgnoreCase(attributes.get(field)) && ((String) dbValue)
842                         .compareTo((String) inputValue) > 0) || (DESCENDING_ORDER.equalsIgnoreCase
843                         (attributes.get(field)) && ((String) dbValue)
844                         .compareTo((String) inputValue) < 0)) {
845                     historizeDocument(inputData, inputMap);
846                     return true;
847                 }
848             } else if (dbValue instanceof LocalDate) {
849                 LocalDate dateDBValue = (LocalDate) converterRegistry.from(dbValue, 1);
850                 //We always get a Joda DateTime Object from a input document
851                 java.util.Date dateInputValue = ((DateTime) inputValue).toDate();
852                 if (checkTimeObject(attributes.get(field), dateInputValue.getTime(), dateDBValue
853                         .toDate().getTime())) {
854                     historizeDocument(inputData, inputMap);
855                     return true;
856                 }
857             } else {
858                 // This will convert all complex types to string a do a comparision
859                 LOGGER.debug("Complex Data Type = {}", dbValue.getClass().getName());
860                 String dbValueStr = dbValue.toString();
861                 String inputValueStr = inputValue.toString();
862                 if ((ASCENDING_ORDER.equalsIgnoreCase(attributes.get(field)) && dbValueStr
863                         .compareTo(inputValueStr) > 0) || (DESCENDING_ORDER.equalsIgnoreCase
864                         (attributes.get(field)) && dbValueStr.compareTo(inputValueStr) < 0)) {
865                     historizeDocument(inputData, inputMap);
866                     return true;
867                 }
868             }
869         }
870         return false;
871     }
