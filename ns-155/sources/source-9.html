235     protected void configureSnap(final PropertyValues propertyValues) {
236         passThrough = Boolean.TRUE.equals(propertyValues.get(PASS_THROUGH_PROP));
237         schemaName = jdbcOperations.normalizeAndEscape((String) propertyValues.get
238                 (SCHEMA_NAME_PROP));
239         tableName = jdbcOperations.normalizeAndEscape((String) propertyValues.get
240                 (TABLE_NAME_PROP));
241         tableName = updateTableName(schemaName, tableName);
242         // get output fields evaluated with pipeline parameters if any
243         List<Map<String, Object>> outputFieldsProp = propertyValues.get(OUTPUT_FIELDS_PATH);
244         List<String> outputFieldsList = Lists.newArrayListWithExpectedSize(outputFieldsProp.size());
245         for (Map<String, Object> outputFieldProp : outputFieldsProp) {
246             ExpressionProperty outputFieldExpression = propertyValues.getExpressionPropertyFor(
247                     outputFieldProp, OUTPUT_FIELD_PROP);
248             outputFieldsList.add((String) outputFieldExpression.eval(null));
249         }
250         // combinedFields are output fields and condition names combined without duplicates, and
251         // will be used to format the output fields in the select statement
252         Set<String> combinedFields = new LinkedHashSet<>(outputFieldsList); // no duplicate
253         // output field
254         // init conditions map of lookup column name and its value expression
255         List<Map<String, Object>> conditionList = propertyValues.get(LOOKUP_CONDITIONS_PATH);
256         conditions = new LinkedHashMap<>(conditionList.size());
257         String databaseDetected = getName();
258         for (Map<String, Object> condition : conditionList) {
259             String name = JsonPathUtil.checkedRead(COLUMN_NAME_PATH, condition);
260             if (StringUtils.isBlank(name)) {
261                 throwExceptionForEmptyProperty(COLUMN_NAME_LABEL);
262             }
263             final String valueStr = JsonPathUtil.checkedRead(COLUMN_VALUE_PATH, condition);
264             if (StringUtils.isBlank(valueStr)) {
265                 throwExceptionForEmptyProperty(COLUMN_VALUE_LABEL);
266             }
267             ExpressionProperty valueExpression = propertyValues.getExpressionPropertyFor(condition,
268                     COLUMN_VALUE_PROP);
269             ExpressionProperty nameExpression = propertyValues.getExpressionPropertyFor(condition,
270                     COLUMN_NAME_PROP);
271             name = nameExpression.eval(null);
272             combinedFields.add(name);
273             // For Oracle Snap, the condition field names need special consideration.
274             if (JdbcOperations.ORACLE.equals(databaseDetected)) {
275                 name = processOracleColumnName(name);
276             }
277             conditions.put(name, valueExpression);
278         }
279         // For Oracle Snap, the output field names need special consideration.
280         if (JdbcOperations.ORACLE.equals(databaseDetected)) {
281             outputFields = new LinkedHashSet<>(outputFieldsList.size());
282             for (String outputField : outputFieldsList) {
283                 outputFields.add(processOracleColumnName(outputField));
284             }
285         } else {
286             outputFields = new LinkedHashSet<>(outputFieldsList);
287         }
288         // For DB Lookup snaps like Oracle, Postgres, Snowflake and Vertica, white space at the end
289         // in value matters so we cannot trim spaces from end in those values
290         if (reservedDBs.contains(databaseDetected)) {
291             isEndSpacesRequired = true;
292         }
293 
294         // init query builder
295         for (String outputField : outputFieldsList) {
296             if (outputField == null) {
297                 throw new ConfigurationException(ERROR_COLUMN_EMPTY)
298                     .withReason(REASON_COLUMN_EMPTY)
299                     .withResolution(RESOLUTION_COLUMN_EMPTY);
300             }
301         }
302 
303         String selectFields = outputFieldsList.isEmpty() ? STAR :
304                 Joiner.on(COMMA).join(combinedFields);
305         queryBuilder.append(String.format(FORMAT_SELECT, selectFields, tableName));
306         initialQuerySize = queryBuilder.length();
307         Connection conn = jdbcOperations.acquireConnection(account);
308         tableMetaData = jdbcOperations.getTableMetaData(conn,
309                 account.getDatabaseConfig().getDatabaseName(), tableName);
310         inImmediateMode = propertyValues.inImmediateMode();
311         if (inImmediateMode && !passThrough) {
312             throw new ConfigurationException(UNSUPPORTED_SNAP_CONFIGURATION)

314                     .withResolution(String.format(SELECT_THE_PASS_THROUGH_OPTION,
315                             PASS_THROUGH_LOOKUP_LABEL));
316         }
317         maxRetryProperty = propertyValues.getAsExpression(MAX_RETRY_PROP);
318         retryIntervalProperty = propertyValues.getAsExpression(RETRY_INTERVAL_PROP);
319     }

343     @SuppressWarnings("unchecked")
344     protected void processDocument(Document document, String inputViewName) {
345         countInputDocs++;
346         configureRetry(document);
347         Map<String, Object> data = getMapDataFrom(document);
348         if (data == null) {
349             document.acknowledge();
350             return;
351         }
352         // get the mapKey to the input data
353         Map<String, Object> mapKey = new LinkedHashMap<>(conditions.size());
354         Map<String, Object> queryKey = new LinkedHashMap<>(conditions.size());
355         for (Map.Entry<String, ExpressionProperty> entry : conditions.entrySet()) {
356             ExpressionProperty valueExpression = entry.getValue();
357             Object value = valueExpression.eval(document);
358             value = getValue(value, entry.getKey(), true);
359             if (!isEndSpacesRequired && (value instanceof String)) {
360                 value = StringUtils.stripEnd((String )value, SPACE);
361             }
362             queryKey.put(entry.getKey(), value);
363             if (SnowflakeDateFormatOptions.AUTO_FORMAT_DATE_TIMESTAMP.equals(dateFormatOptions)) {
364                 String key = quotationHandler.unescape(entry.getKey());
365                 Object input = convertDateAndTimestampToDefault(value, key);
366                 if (dateColumns.containsKey(key)) {
367                     value = DateUtils.convertDateToTime(input, dateColumns.get(key));
368                 } else if (dateColumns.containsKey(key.toUpperCase())) {
369                     value = DateUtils.convertDateToTime(input, dateColumns.get(key.toUpperCase()));
370                 }
371             }
372             mapKey.put(entry.getKey(), value);
373         }
374         List<InputData> inputDataList = inputRecords.get(mapKey);
375         if (inputDataList == null) {
376             // get the list of predicates
377             List<String> predicates = new ArrayList<>(conditions.size());
378             for (Map.Entry<String, Object> entry : queryKey.entrySet()) {
379                 String name = entry.getKey();
380                 // for Oracle Snap, column names in the WHERE clause should be enclosed in double
381                 // quote if they are not in all upper case
382                 if (JdbcOperations.ORACLE.equals(getName()) && !StringUtils.isAllUpperCase(name)) {
383                     name = quotationHandler.escape(name);
384                 }
385                 Object mapVal = entry.getValue();
386                 if (mapVal == null) {
387                     predicates.add(String.format(FORMAT_PREDICATE_NULL, name));
388                 } else {
389                     prepareQuery(queryKey, entry, predicates, name);
390                 }
391 
392                 if (SnowflakeDateFormatOptions.AUTO_FORMAT_DATE_TIMESTAMP
393                         .equals(dateFormatOptions)) {
394                     String unescapeKey = quotationHandler.unescape(name);
395                     if (!dateColumns.containsKey(unescapeKey) &&
396                             !dateColumns.containsKey(unescapeKey.toUpperCase())) {
397                         mapKey.put(entry.getKey(), queryKey.get(entry.getKey()));
398                     }
399                 } else {
400                     mapKey.put(entry.getKey(), queryKey.get(entry.getKey()));
401                 }
402             }
403             //format an and-clause string: key1=? AND key2=? AND ...
404             String andClause = String.format(FORMAT_PARENTHESIS, Joiner.on(AND).join(predicates));
405             // append an and-clause to the query builder
406             queryBuilder.append(andClause);
407             // store bind values
408             for (Object value : queryKey.values()) {
409                 if (value != null) {
410                     bindValues.add(value);
411                 }
412             }
413             // create and store new input data list
414             inputDataList = new ArrayList<>();
415             inputDataList.add(new InputData(document));
416             inputRecords.put(mapKey, inputDataList);
417         } else {
418             // store input data so that it can be merged into output or error view records later
419             inputDataList.add(new InputData(document));
420         }
421         // if the current batch has reached its thresholds, process the batch
422         if (inImmediateMode || (bindValues.size() + mapKey.size() >= MAX_PARAMETERS ||
423                 countInputDocs >= maxInputDocs)) {
424             processQuery(document);
425             // re-initialize query string builder and input records storage
426             queryBuilder.delete(initialQuerySize, queryBuilder.length());
427             bindValues.clear();
428             inputRecords.clear();
429             countInputDocs = 0;
430         }
431     }

456     @Override
457     protected void processQuery(final Document document) {
458         Map<String, Object> inputDataKey = new LinkedHashMap<>(conditions.size());
459         Map<String, Object> inputDataUUIDKey = new LinkedHashMap<>(conditions.size());
460         Map<String, Object> inputDataDateKey = new LinkedHashMap<>(conditions.size());
461         int orIndex = queryBuilder.lastIndexOf(OR);
462         String sql = queryBuilder.substring(0, orIndex);
463         DatabaseCursor cursor = null;
464         try {
465             cursor = executeQuery(sql, bindValues, account, maxRetries, retryInterval);
466             while (cursor.hasNext()) {
467                 Map<String, Object> dbData = cursor.fetchOne();
468                 // pass through the original input data to the output
469                 for (String name : conditions.keySet()) {
470                     Object value = dbData.get(quotationHandler.unescape(name));
471                     value = getValue(value, quotationHandler.unescape(name), false);
472                     if (!isEndSpacesRequired && (value instanceof String)) {
473                         value = StringUtils.stripEnd((String )value, SPACE);
474                     }
475                     inputDataKey.put(name, value);
476                     if (value instanceof LocalDate || value instanceof Date) {
477                         inputDataDateKey.put(name, value.toString());
478                     }
479                     //swat-2004 fix .maintain a seperate map if the uuid is passed as string id
480                     if (value instanceof UUID) {
481                         value = value.toString();
482                         if (convertUniqueIdentifierColumnValueToUpperCase){
483                             value = value.toString().toUpperCase();
484                         }
485                         inputDataUUIDKey.put(name, value);
486                     }
487                 }
488                 List<InputData> inputDataList = inputRecords.get(inputDataKey);
489                 if (CollectionUtils.isEmpty(inputDataList)) {
490                     // convert inputRecords for case-insensitive lookup
491                     convertInputRecordsIgnoreCase();
492                     // try again to find input data which correlates to the lookup result record
493                     inputDataKey = toLowerCase(inputDataKey);
494                     inputDataList = inputRecords.get(inputDataKey);
495                     if (CollectionUtils.isEmpty(inputDataList)) {
496                         // Block to check if there is any date field in string format
497                         Map<String, Object> finalInputDataKey = inputDataKey;
498                         inputDataDateKey.forEach((k, v) -> {
499                             finalInputDataKey.put(k, inputDataDateKey.get(k));
500                         });
501                         inputDataList = inputRecords.get(finalInputDataKey);
502                         // swat-2004 -if it fails again,check if there is any match with uuid as
503                         // string
504                         if (!CollectionUtils.sizeIsEmpty(inputDataUUIDKey)) {
505                             inputDataList = inputRecords.get(inputDataUUIDKey);
506                         }
507                         if (CollectionUtils.isEmpty(inputDataList)) {
508                             // if it fails again, send all input data to the error view
509                             throw new SnapDataException(ERR_INPUT_DATA_NOT_FOUND)
510                                     .withReason(REASON_INPUT_DATA_TYPE)
511                                     .withResolution(RESOLUTION_INPUT_DATA_TYPE);
512                         }
513                     }
514                 }
515                 // write output data with original input data
516                 Map<String, Object> outputData;
517                 if (outputFields.isEmpty()) {
518                     outputData = dbData;
519                 } else {
520                     outputData = new LinkedHashMap<>(outputFields.size());
521                     for (String outputField : outputFields) {
522                         outputField = quotationHandler.unescape(outputField);
523                         outputData.put(outputField, dbData.get(outputField));
524                     }
525                 }
526                 Document outputDoc = documentUtility.newDocument(outputData);
527                 for (InputData inputData : inputDataList) {
528                     inputData.setResultsProduced(true);
529                     outputViews.write(outputDoc, inputData.getDocument());
530                 }
531             }
532         } catch (SnapDataException e) {
533             writeAllInputDataToErrorView(e, document);
534             return;
535         } catch (SuggestViewAbortException e) {
536             // max. number of output documents produced in preview, should end gracefully
537             return;
538         } catch (Exception e) {
539             SnapDataException ex = new SnapDataException(e, ERR_PROCESS_LOOKUP)
540                     .withReason(e.getMessage())
541                     .withResolution(PLEASE_ENSURE_DATABASE_ACCESSIBLE);
542             writeAllInputDataToErrorView(ex, document);
543             return;
544         } finally {
545             IOUtils.closeQuietly(cursor);
546         }
547         // check if there is any input data which did not produce any result
548         for (List<InputData> inputDataList : inputRecords.values()) {
549             for (InputData inputData : inputDataList) {
550                 if (!inputData.isResultsProduced()) {
551                     Document original = inputData.getDocument();
552                     if (passThrough) {
553                         outputViews.write(documentUtility.newDocument(new LinkedHashMap<>()),
554                                 original);
555                     } else {
556                         // write original input data and error message/code to error view
557                         Map<String, Object> errorViewData = new LinkedHashMap<>(ERROR_MAP);
558                         SnapDataException ex = new SnapDataException(
559                                 documentUtility.newDocument(errorViewData), ERR_LOOKUP_RESULT)
560                                 .withReason(REASON_LOOKUP_RESULT)
561                                 .withResolution(RESOLUTION_LOOKUP_RESULT);
562                         errorViews.write(ex, original);
563                     }
564                 }
565             }
566         }
567     }

630     private Object getValue(Object value, String key, boolean forInput) {
631         if (value instanceof String && isIgnoreCase) {
632             return ((String) value).toLowerCase();
633         } else if (value instanceof BigDecimal) {
634             // if stripTrailingZeros() results in engineering notation (e.g. 9E01), convert it to
635             // plain BigDecimal (e.g. 90) by adding zero
636             return ((BigDecimal) value).stripTrailingZeros().add(BigDecimal.ZERO);
637         } else if (value instanceof BigInteger) {
638             return new BigDecimal((BigInteger) value);
639         } else if (value instanceof Number) {
640             return new BigDecimal(value.toString());
641         } else {
642             if (!forInput && SnowflakeDateFormatOptions.AUTO_FORMAT_DATE_TIMESTAMP
643                     .equals(dateFormatOptions)) {
644                 return DateUtils.convertDateToTime(value, dateColumns.get(key));
645             }
646         }
647         if (isPreview) {
648             if (value instanceof LocalDateTime) {
649                 value = new LocalDateTime(value, GJChronology.getInstance(
650                         ((LocalDateTime) value).getChronology().getZone()));
651             }
652         }
653         return value;
654     }
