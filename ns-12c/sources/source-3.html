36     private static final String BEARER_PREFIX2 = "bearer";

38     @Override
39     public Header connect() throws ExecutionException {
40         if (!isHeaderAuthenticated()) {
41             return null;
42         }
43         String token = super.getAccessToken();
44         if (token.startsWith(BEARER_PREFIX1) || token.startsWith(BEARER_PREFIX2)) {
45             return new BasicHeader(AUTH.WWW_AUTH_RESP, token);
46         }
47         return new BasicHeader(AUTH.WWW_AUTH_RESP, String.format(BEARER_TOKEN,
48                 BEARER_PREFIX1, token));
49     }
