367     public void filter(ContainerRequestContext requestContext) {
368         // Process the first redirect property set whose condition expression evaluates to true
369         Optional<RedirectProperties> redirectPropertiesOptional = properties.stream()
370                 .filter(property -> ObjectType.toBoolean(property.condition.eval(null)))
371                 .findFirst();
372 
373         if (redirectPropertiesOptional.isEmpty()) {
374             return;
375         }
376 
377         RedirectProperties redirectProps = redirectPropertiesOptional.get();
378 
379         // Build a response for the HTTP client to handle vs. tasking the GatewayResource
380         Response.ResponseBuilder builder = Response.status(redirectProps.responseStatus.statusType);
381 
382 
383         if (redirectProps.url == null) {
384             // case should be caught in .filter() for missingUrls. Catching here as well.
385             builder.header(HttpHeaders.CONTENT_TYPE, MediaType.APPLICATION_JSON)
386                     .entity(URL_RESP)
387                     .status(Response.Status.BAD_REQUEST);
388             requestContext.abortWith(builder.build());
389             return;
390         }
391 
392         // Response 300 may or may not have a preferred redirect in the Location header per RFC 9110
393         Optional<String> optionalUrl = redirectProps.url.eval(null);
394         if (optionalUrl.isPresent()) {
395             builder.header(HttpHeaders.LOCATION, optionalUrl.get());
396         }
397 
398         if (redirectProps.useOriginalContent) {
399             // Add the headers from the original request
400             MultivaluedMap<String, String> headerMap = requestContext.getHeaders();
401             Set<String> headers = headerMap.keySet();
402 
403             String headerContents;
404             String originalContentType = "";
405             for (String header : headers) {
406                 headerContents = headerMap.getFirst(header);
407                 builder.header(header, headerContents);
408                 if (header.toLowerCase().contains("content-type")) {
409                     originalContentType = headerContents;
410                 }
411             }
412 
413             // Add the query params from the original request in the response header
414             MultivaluedMap<String, String> paramMap = requestContext.getUriInfo()
415                     .getQueryParameters();
416             Set<String> params = paramMap.keySet();
417 
418             String paramValue;
419             for (String param : params) {
420                 paramValue = paramMap.getFirst(param);
421                 builder.header(param, paramValue);
422             }
423 
424             if (requestContext.hasEntity()) {
425                 try (InputStream inputStream = requestContext.getEntityStream()) {
426                     try {
427                         // Attempt to get the media type from the requestContext, may not exist
428                         builder.type(requestContext.getMediaType());
429                     } catch (NotImplementedException e) {
430                         builder.type(originalContentType);
431                     }
432 
433                     builder.entity(inputStream.readAllBytes());
434                 } catch (IOException ignored) {
435                     /*
436                      * Failed to read the original message body. Don't add any content to the
437                      * response. Per RFC 9110, the server is primarily responsible for providing
438                      * Location header and status codes; the user-agent assumes handling the message
439                      * content on the client side. The exception is a Response 300 where the server
440                      * is responsible for providing multiple url addresses in the message body.
441                      * Even so, the server supplies the message body and not the user-agent.
442                      */
443                 }
444             }
445         } else if (redirectProps.containsContentType && redirectProps.containsMessageBody) {
446             // Case for !useOriginalContent
447             builder.type(redirectProps.contentType);
448             builder.entity(redirectProps.messageBody.getBytes(StandardCharsets.UTF_8));
449         }
450 
451         requestContext.abortWith(builder.build());
452     }

512     private void setContentType(RedirectProperties property, Pattern pattern) {
513         if (property.contentTypeExpression == null) {
514             return;
515         }
516 
517         try {
518             // Attempt to evaluate the expression as is
519             Optional<String> res = property.contentTypeExpression.eval(null);
520             if (res.isEmpty()) {
521                 property.contentType = property.contentTypeExpression.getExpression();
522                 return;
523             }
524             property.contentType = res.get();
525         } catch (Exception exception) {
526             // Recovery attempt to acquire the expression in escape() or the string representation
527             String contentTypeExpression = property.contentTypeExpression.getExpression();
528             Matcher matcher = pattern.matcher(contentTypeExpression);
529 
530             if (matcher.find()) {
531                 property.contentType = matcher.group(2); // return the user-defined content-type
532                 return;
533             }
534             property.contentType = contentTypeExpression; // no match, just return the expression
535         }
