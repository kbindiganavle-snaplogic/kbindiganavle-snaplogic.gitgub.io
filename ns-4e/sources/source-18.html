60     /**
61      * Detects the end of a JSON-Path that is embedded in a SQL statement.  The
62      * detector looks for non-alphanumeric characters and symbols that are not
63      * part of JSON-Path syntax (e.g. '=', '+', ...).  This behavior is more
64      * strict than the normal parser, which is pretty lenient, because I think
65      * we want something like '$foo+$bar' to result in a SQL expression where
66      * two values are added together.
67      */
68     protected static final PathParser.TerminatorDetector PATH_TERMINATOR = ch ->
69             !(Character.isAlphabetic(ch) ||
70             Character.isDigit(ch) ||
71             ch == '_' ||
72             ch == '.' ||
73             ch == '(' ||
74             ch == '[');

121     protected int transform(int depth, int start, StringBuilder dest) throws SQLException {
122         ParserState state = ParserState.STATE_SQL;
123         int keywordStart = -1, retval = -1;
124         char last = '\n';
125 
126         for (int lpc = start; lpc < escapedSql.length(); lpc++) {
127             char ch = escapedSql.charAt(lpc);
128 
129             switch (state) {
130                 case STATE_SQL:
131                     if (escapeChar != null && escapeChar.equals(ch)) {
132                         state = ParserState.STATE_ESCAPE_IDENTIFIER;
133                         break;
134                     }
135                     switch (ch) {
136                         case '\'':
137                             state = ParserState.STATE_SINGLE_STRING;
138                             break;
139                         case '"':
140                             state = ParserState.STATE_DOUBLE_STRING;
141                             break;
142                         case '$':
143                             if (lpc > 0) {
144                                 char lastChar = escapedSql.charAt(lpc - 1);
145                                 if (lastChar == '\\') {
146                                     // If it is "\$", remove '\' and break.
147                                     dest.append(escapedSql.substring(start, lpc - 1));
148                                     start = lpc;
149                                     break;
150                                 } else if (Character.isLetterOrDigit(lastChar)) {
151                                     // e.g. metadata$filename in Snowflake
152                                     break;
153                                 }
154                             }
155                             try {
156                                 int varEnd = PathParser.parseUpTo(escapedSql.substring(lpc),
157                                         EMPTY_EXPR_HANDLER, PATH_TERMINATOR);
158                                 dest.append(escapedSql.substring(start, lpc));
159                                 dest.append('?');
160                                 variablePaths.add(escapedSql.substring(lpc, lpc + varEnd));
161                                 lpc = lpc + varEnd;
162                                 start = lpc;
163                                 lpc -= 1; // XXX minus one because the loop adds one.
164                             } catch (InvalidPathException e) {
165                                 throw new SQLSyntaxErrorException(e.getMessage(), e);
166                             }
167                             break;
168                         case '[':
169                             state = ParserState.STATE_BRACKET;
170                             break;
171                         case '-':
172                             if (last == '-') {
173                                 state = ParserState.STATE_SINGLE_LINE_COMMENT;
174                             }
175                             break;
176                         case '*':
177                             if (last == '/') {
178                                 state = ParserState.STATE_MULTI_LINE_COMMENT;
179                             }
180                             break;
181                         default:
182                             break;
183                     }
184                     break;
185                 case STATE_SINGLE_STRING:
186                     if (ch == '\'') {
187                         state = ParserState.STATE_SQL;
188                     }
189                     break;
190                 case STATE_DOUBLE_STRING:
191                     if (ch == '"') {
192                         state = ParserState.STATE_SQL;
193                     }
194                     break;
195                 case STATE_BRACKET:
196                     if (ch == ']') {
197                         state = ParserState.STATE_SQL;
198                     }
199                     break;
200                 case STATE_ESCAPE_IDENTIFIER:
201                     if (escapeChar != null && escapeChar.equals(ch)) {
202                         state = ParserState.STATE_SQL;
203                     }
204                     break;
205                 case STATE_SINGLE_LINE_COMMENT:
206                     switch (ch) {
207                         case '\n':
208                         case '\r':
209                             state = ParserState.STATE_SQL;
210                     }
211                     break;
212                 case STATE_MULTI_LINE_COMMENT:
213                     if (last == '*' && ch == '/') {
214                         state = ParserState.STATE_SQL;
215                     }
216             }
217             last = ch;
218         }
219 
220         /*
221         As shown in the commented-out code segment below, we used to throw SQLSyntaxErrorException
222         if unterminated quote is found. Unterminated quote can occur if quote char is used within
223         comments. We should not attempt to remove all types of comments because the exact
224         definitions of comments are different from one db to another and some database uses
225         a comment syntax to pass the server version info in a comment. If query contains
226         unterminated quote, we should let db server decide if the query is valid or not.
227         if (state != ParserState.STATE_SQL) {
228             throw new SQLSyntaxErrorException(state.getMessage());
229         }
230         */
231 
232         if (start < escapedSql.length()) {
233             dest.append(escapedSql.substring(start, escapedSql.length()));
234         }
235 
236         return retval;
237     }
