37     public static void invalidEndPoint(String endpoint, AmazonClientException e) {
38         String[] error = e.getLocalizedMessage().split(",");
39         boolean endpointError = false;
40         switch (error[0]) {
41             case HTTP_301_Error:
42             case HTTP_304_Error:
43             case HTTP_400_Error:
44             case HTTP_403_Error:
45             case HTTP_404_Error:
46             case HTTP_500_Error:
47             case HTTP_503_Error:
48                 endpointError = true;
49         }
50         if (e.getCause() instanceof UnknownHostException || endpointError) {
51             throw new ExecutionException(e, ERRMSG_INVALID_SETTING)
52                     .withResolution(ERR_RESOLUTION_INVALID_SETTING);
53         }
54     }
