432     private List <SOAPMessageTemplatizer.Input> createTemplatizerInput(final QName operationQName,
433             final boolean isDocStyle, final Input input,
434             List <SOAPMessageTemplatizer.Input> inputs,
435             final InputImpl wsdl11Input, List<String> bodyPartNames) {
436         for (int i = 0; i < wsdl11Input.getParts().size(); i++) {
437             Type type = wsdl11Input.getParts().get(i).getType();
438             if (type != null) {
439                 SOAPMessageTemplatizer.Input templatizerInput = inputFactory
440                         .createTemplatizerInputFrom(operationQName, wsdl11Input.getParts().get(i),
441                                 isDocStyle);
442                 templatizerInput.addLocation(type.getQName());
443                 inputs.add(templatizerInput);
444             } else {
445                 List<SOAPMessageTemplatizer.Input> messageInputs =
446                         inputFactory.createTemplatizerInputListFrom(operationQName,
447                                 input.getParts(), isDocStyle);
448                 if (inputs.size() > 0) {
449                     for (SOAPMessageTemplatizer.Input input1 : messageInputs) {
450                         String localPart = input1.getLocation().getLocalPart();
451                         if (bodyPartNames != null &&
452                                 bodyPartNames.contains(localPart)) {
453                             inputs.add(input1);
454                             bodyPartNames.remove(localPart);
455                         }
456                     }
457                 } else {
458                     inputs = messageInputs;
459                 }
460             }
461         }
462         return inputs;
463     }

492     /**
493      * Ensure we can serialize headers during preview
494      *
495      * @param headers           the headers to be serialized
496      * @param serializedHeaders the serialized headers which can be written as part of the error
497      *                          document
498      */
499     @SuppressWarnings("unchecked")
500     public void serializeHeader(Map<?, Object> headers, Map<String,
501             Object> serializedHeaders) {
502         for (Map.Entry<?, Object> items : headers.entrySet()) {
503             String name = String.valueOf(items.getKey());
504             // scope headers are not good for consumption,
505             // some internal crap being stored for cxf.
506             if (name != null && !name.equalsIgnoreCase(CXF_SCOPES)) {
507                 Object value = items.getValue();
508                 if (value instanceof Map) {
509                     serializeHeader((Map<?, Object>) value, serializedHeaders);
510                 } else if (value instanceof Collection && !((Collection) value).isEmpty()) {
511                     // lets take the first entry to keep it simple,
