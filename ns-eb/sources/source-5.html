225     private org.apache.hadoop.hbase.client.Put constructPutRequest(Document document) {
226         try {
227             Map<String, Object> documentMap = document.get(Map.class);
228             if (!documentMap.containsKey(KEY)) {
229                 throw new SnapDataException(COULD_NOT_CONSTRUCT_REQUEST)
230                         .withReason(KEY_MISSING)
231                         .withResolution(DOCUMENT_VALIDITY);
232             }
233             Object key = documentMap.get(KEY);
234             org.apache.hadoop.hbase.client.Put put = new org.apache.hadoop.hbase.client.Put(Utils
235                     .convertToBytes(key));
236             for (Map.Entry<String, Object> entry : documentMap.entrySet()) {
237                 String entryKey = entry.getKey();
238                 if (KEY.equals(entryKey)) {
239                     continue;
240                 }
241                 // At this point, the entryKey must be a column family name
242                 Object entryValue = entry.getValue();
243                 if (!(entryValue instanceof Map)) {
244                     throw new SnapDataException(COULD_NOT_CONSTRUCT_REQUEST)
245                             .withReason(String.format(COLUMN_FAMILY_VALUE_NOT_A_MAP, entryKey))
246                             .withResolution(DOCUMENT_VALIDITY);
247                 }
248                 Map<String, Object> cqValueMap = (Map<String, Object>) entryValue;
249                 for (Map.Entry<String, Object> cqValueEntry : cqValueMap.entrySet()) {
250                     String cq = cqValueEntry.getKey();
251                     Object value = cqValueEntry.getValue();
252                     put.addColumn(Bytes.toBytes(entryKey), Bytes.toBytes(cq), Utils
253                             .convertToBytes(value));
254                 }
255             }
256             return put;
257         } catch (ClassCastException e) {
258             throw new SnapDataException(e, COULD_NOT_CONSTRUCT_REQUEST)
259                     .withResolution(DOCUMENT_VALIDITY);
260         }
261     }

263     private void executeBatch() {
264         try {
265             table.put(batch);
266         } catch (IOException e) {
267             SnapDataException snapDataException = new SnapDataException(e,
268                     COULD_NOT_PERFORM_OPERATION)
269                     .withResolution(DOCUMENT_VALIDITY);
270             Map<String, Object> operations = new LinkedHashMap<>(1);
271             List<Map<String, Object>> summaryList = new ArrayList<>(batch.size());
272             for (org.apache.hadoop.hbase.client.Put elem : batch) {
273                 Map<String, Object> summary = elem.toMap();
274                 summaryList.add(summary);
275             }
276             operations.put(PUT_OPERATIONS, summaryList);
277             errorViews.write(snapDataException, documentUtility.newDocument(operations));
278         } finally {
279             batch.clear();
280         }
281     }
