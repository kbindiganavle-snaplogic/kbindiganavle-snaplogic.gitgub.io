91     public GraphServiceClient initializeGraphClient(Account<String> account) {
92         String accessToken = account.connect();
93         SimpleAuthProvider authProvider = new SimpleAuthProvider(accessToken);
94         OkHttpClient.Builder httpClientBuilder = HttpClients.createDefault(authProvider)
95                 .newBuilder();
96         final ProxySettings proxySettings = RequestClientUtil.getProxySettings();
97         if (proxySettings.isProxied()) {
98             httpClientBuilder.proxy(new Proxy(Proxy.Type.HTTP,
99                     new InetSocketAddress(proxySettings.getProxyHost(),
100                             proxySettings.getProxyPort())));
101             httpClientBuilder.proxySelector(ProxySelector.getDefault());
102             if (proxySettings.isProxyAuthenticated()) {
103                 Authenticator proxyAuthenticator = new Authenticator() {
104                     @Override
105                     public Request authenticate(@Nullable Route route, @NotNull Response response)
106                             throws IOException {
107                         String credentials = Credentials.basic(proxySettings.getProxyUsername(),
108                                 proxySettings.getProxyPassword());
109                         return response.request().newBuilder()
110                                 .header("Proxy-Authorization", credentials).build();
111                     }
112                 };
113                 httpClientBuilder.proxyAuthenticator(proxyAuthenticator);
114             }
115         }
116         return GraphServiceClient.builder()
117                 .authenticationProvider(authProvider)
118                 .httpClient(httpClientBuilder.build())
119                 .buildClient();
120     }

393      *
394      * @param member DirectoryObject
395      * @return display name of directory object
396      */
397     private String retrieveDisplayName(DirectoryObject member) {
398         JsonElement rawName = member.additionalDataManager().get("displayName");
399         String displayName = Objects.isNull(rawName) ?
400                 StringUtils.EMPTY : rawName.toString();
401         if (displayName.startsWith("\"")) {
402             displayName = displayName.substring(1);
403         }
404         if (displayName.endsWith("\"")) {
405             displayName = displayName.substring(0, displayName.length() - 1);
406         }
407         return displayName;
408     }

430                 teamID = teamID.substring(teamID.indexOf("('") + 2, teamID.indexOf("')"));

434             // consume exception if there is any change in API in future
435             LOG.error("Failed to retrieve Team ID", exception);
