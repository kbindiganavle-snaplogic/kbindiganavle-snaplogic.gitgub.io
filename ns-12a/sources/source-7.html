241     @Override
242     public void execute() throws ExecutionException {
243         /* Ideally there would be utmost one record in DateFilterInputView
244          * while SearchFilterInputview can have multiple records.
245          *
246          * For each record of SearchFilterInputView, the single record of DateFilterInputView
247          * is used.
248          */
249         Map<String, InputView> inputViewMap = inputViews.getAll();
250         InputView dateFilterInputView = inputViewMap
251                 .get(CMP_READ_DATE_FILTER_INPUT_VIEW);
252         InputView searchFilterInputView = inputViewMap.get(CMP_READ_INPUT_VIEW);
253         PeekingDocumentIterator dateFilterInputViewIterator = null;
254         // prepare document from date input view
255         Document dateInputDoc = null;
256         if (dateFilterInputView != null) {
257             dateFilterInputViewIterator = new PeekingDocumentIterator(
258                     inputViews.getDocumentsFrom(dateFilterInputView),
259                     CMP_READ_DATE_FILTER_INPUT_VIEW);
260             if (dateFilterInputViewIterator != null) {
261                 if (dateFilterInputViewIterator.hasNext()) {
262                     dateInputDoc = dateFilterInputViewIterator.next();
263                     // fix for platform 4.5 hasNext() must be called at end of iteration to prevent error
264                     // that not all input documents were consumed, snap currently handles only one date filter document
265                     boolean hasMultipleDateFilterRecords = false;
266                     while (dateFilterInputViewIterator.hasNext()) {
267                         dateFilterInputViewIterator.next();
268                         hasMultipleDateFilterRecords = true;
269                     }
270                     if (hasMultipleDateFilterRecords) {
271                         SnapDataException sde = new SnapDataException(INCORRECT_INPUT_USAGE)
272                                 .withReason(MORE_THAN_ONE_DATE_FILTER_RECORD)
273                                 .withResolution(PLEASE_FIX_INPUT_AND_RETRY);
274                         errorViews.write(sde, dateInputDoc);
275                     }
276                 } else {
277                     // No execution if the user says "do not execute if inputview2 has no records"
278                     if (isNoExecOnInputView2Empty) {
279                         return;
280                     }
281                 }
282             }
283         }
284         PeekingDocumentIterator searchFilterInputViewIterator = null;
285         if (searchFilterInputView != null) {
286             searchFilterInputViewIterator = new PeekingDocumentIterator(
287                     inputViews.getDocumentsFrom(searchFilterInputView),
288                     CMP_READ_INPUT_VIEW);
289         }
290         boolean isFirstExecution = true;
291         do {
292             // prepare document from search input view
293             Document searchInputDoc = null;
294             if (isFirstExecution) {
295                 isFirstExecution = false;
296                 if (searchFilterInputViewIterator != null) {
297                         if (searchFilterInputViewIterator.hasNext()) {
298                             searchInputDoc = searchFilterInputViewIterator.next();
299                         } else {
300                             // No execution if the user says "do not execute if inputview1 has no
301                             //   records"
302                             if (isNoExecOnInputView1Empty) {
303                                 return;
304                             }
305                         }
306                 }
307             } else {
308                 searchInputDoc = searchFilterInputViewIterator.next();
309             }
310             Map<String, String> dates = readFromAndToDateValues(dateInputDoc);
311             String inputFromDateString = dates.get(KEY_DATE_FROM);
312             String inputToDateString = dates.get(KEY_DATE_TO);
313             if (CMP_READ_XML_INCLUDE_DELETED.equalsIgnoreCase(filterName)) {
314                 filterValues.setWriterQueryFilterName(DELETED_FILTER_TAG);
315             } else if (CMP_READ_XML_INCLUDE_ALL_DELETED_AND_NON_DELETED
316                     .equalsIgnoreCase(filterName)) {
317                 filterValues.setWriterQueryFilterName(INCLUDE_NON_DELETED_FILTER_TAG);
318             }
319             if (!openAirReadType.equals(CUSTOM_FIELD_TYPE)) {
320                 updateDateFilterValues(dateInputDoc, inputFromDateString, inputToDateString);
321             }
322             StringBuffer queryBuilderPart = new StringBuffer();
323             queryBuilderPart.append(utilities
324                     .getOpenAirRequestHeader(userCredentials));
325             readOpenAirData(searchInputDoc, false, searchFilterInputViewIterator,
326                     queryBuilderPart.toString());
327             if (openAirReadType.equalsIgnoreCase(USER_ENTITY_TYPE)) {
328                 readOpenAirData(searchInputDoc, true,
329                         searchFilterInputViewIterator, queryBuilderPart.toString());
330             }
331         } while (searchFilterInputViewIterator != null
332                 && searchFilterInputViewIterator.hasNext());
333     }

386     private void updateDateFilterValues(Document dateInputDoc,
387             String inputFromDateString, String inputToDateString) {
388         if (!inputDateFilterType.equals(TYPE_NONE)) {
389             String dateFilterFieldName = getDateFilterFieldName(dateInputDoc);
390             if (StringUtils.isEmpty(dateFilterFieldName)) {
391                 throw new ExecutionException(DATE_FILTER_FIELD_NAME_MISSING_MSG)
392                         .withReason(DATE_FILTER_FIELD_NAME_MISSING_REASON)
393                         .withResolution(DATE_FILTER_FIELD_NAME_MISSING_RESOLUTION);
394             }
395             filterValues.setDateFilterFieldName(utilities
396                     .castAsString(dateFilterFieldName));
397             switch (inputDateFilterType) {
398                 case BEFORE_DATE_VALUE:
399                     if (StringUtils.isEmpty(inputToDateString)) {
400                         throw new ExecutionException(String.format(
401                                 ERR_DATE_VALUE_MISSING_MSG, TO))
402                                 .withReason(String.format(ERR_DATE_VALUE_MISSING_REASON, TO))
403                                 .withResolution(String.format(ERR_DATE_VALUE_MISSING_RESOLUTION,
404                                         TO.toLowerCase()));
405                     }
406                     filterValues.setDateFilterTo(utilities.getDateValue(
407                             inputToDateString,
408                             CMP_READ_DATE_FILTER_TO_DATE_VALUE_LABEL, errorViews,
409                             documentUtility));
410                     filterValues.setDateFilterType(BEFORE_DATE_VALUE);
411                     break;
412                 case BETWEEN_DATE:
413                     if (StringUtils.isEmpty(inputFromDateString) ||
414                             StringUtils.isEmpty(inputToDateString)) {
415                         throw new ExecutionException(
416                                 ERR_BETWEEN_DATE_VALUES_MISSING_MSG)
417                                 .withReason(ERR_BETWEEN_DATE_VALUES_MISSING_REASON)
418                                 .withResolution(ERR_BETWEEN_DATE_VALUES_MISSING_RESOLUTION);
419                     }
420                     filterValues.setDateFilterFrom(utilities.getDateValue(
421                             inputFromDateString,
422                             CMP_READ_DATE_FILTER_FROM_DATE_VALUE_LABEL, errorViews,
423                             documentUtility));
424                     filterValues.setDateFilterTo(utilities.getDateValue(
425                             inputToDateString,
426                             CMP_READ_DATE_FILTER_TO_DATE_VALUE_LABEL, errorViews,
427                             documentUtility));
428                     filterValues.setDateFilterType(BETWEEN_DATE_VALUE);
429                     break;
430                 default:
431                     switch (inputDateFilterType) {
432                         case AFTER_DATE:
433                             filterValues.setDateFilterType(AFTER_DATE_VALUE);
434                             break;
435                         case DATE_EQUAL_TO_DATE:
436                             filterValues.setDateFilterType(DATE_EQUAL_TO_DATE_VALUE);
437                             break;
438                         case DATE_NOT_EQUAL_TO_DATE:
439                             filterValues
440                                     .setDateFilterType(DATE_NOT_EQUAL_TO_DATE_VALUE);
441                             break;
442                     }
443                     if (StringUtils.isEmpty(inputFromDateString)) {
444                         throw new ExecutionException(String.format(
445                                 ERR_DATE_VALUE_MISSING_MSG, FROM))
446                                 .withReason(String.format(ERR_DATE_VALUE_MISSING_REASON, FROM))
447                                 .withResolution(String.format(ERR_DATE_VALUE_MISSING_RESOLUTION,
448                                         FROM.toLowerCase()));
449                     }
450                     filterValues.setDateFilterFrom(utilities.getDateValue(
451                             inputFromDateString,
452                             CMP_READ_DATE_FILTER_FROM_DATE_VALUE_LABEL, errorViews,
453                             documentUtility));
454                     break;
455             }
456             filterValues = utilities.getDateFilterQuery(filterValues);
457         }
