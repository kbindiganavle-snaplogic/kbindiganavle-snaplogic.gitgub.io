149     @Override
150     protected void process(Document document, String inputViewName) {
151         Object data;
152         Map<String, String> inputDoc = null;
153         if (document != null && (data = document.get()) != null) {
154             if (data instanceof Map) {
155                 inputDoc = (Map<String, String>) data;
156             }
157             if(StringUtils.equals(openAirWriteType, OPENAIR_RESOURCE_TYPE_PROJECT)) {
158                 loadProjectStages();
159             }
160             try {
161                 addUpdateRequest.append(String.format(request, openAirWriteType,
162                         utilities.setFieldValues(openAirWriteType, inputDoc, mapProjectstages)));
163             } catch (ExecutionException e) {
164                 writeToErrorView(e, document);
165                 return;
166             }
167             count++;
168             if (count == OPENAIR_READ_LIMIT) {
169                 utilities.executeOpenAirRequest(addUpdateRequest.toString(), userCredentials,
170                         documentUtility, outputViews, errorViews, WRITE_TYPE);
171                 count = 0;
172                 addUpdateRequest = new StringBuffer();
173             }
174         }
175     }

199             for (String inputViewName : provider.getRegisteredViewNames()) {
200                 SchemaBuilder schemaBuilder = provider.getSchemaBuilder(inputViewName);
201                 schemaBuilder.type(SnapType.COMPOSITE);
