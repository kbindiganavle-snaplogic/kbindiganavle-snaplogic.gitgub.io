432     private void consumeMessages(Document document) throws InterruptedException {
433         long maxMsgCountLong = maxMsgCount != null ? maxMsgCount.longValue() : -1;
434         int maxMsgCountInt = maxMsgCount != null ? maxMsgCount.intValue() : -1;
435         long maxMsgPerPollLong = maxMsgPerPoll.longValue();
436         long remMessages;
437         int maxMsgPerPollInt = maxMsgPerPoll.intValue();
438         IterableStream<ServiceBusReceivedMessage> receivedMessages;
439         while(!stopFlag && !Thread.interrupted()) {
440             if (maxMsgPerPollLong > maxMsgCountLong && maxMsgCountLong > 0) {
441                 maxMsgPerPollInt = maxMsgCountInt;
442             }
443             if (maxMsgCountLong > 0) {
444                 remMessages = maxMsgCountLong - messageCountLong;
445                 if (remMessages < maxMsgPerPollInt && remMessages > 0) {
446                     maxMsgPerPollInt = (int) remMessages;
447                 }
448             }
449             isWaitTimeExhausted = true;
450             long numOfMessages = messageCountLong;
451             if (maxMsgPerPollInt > 0) {
452                 receivedMessages = maxWaitTime == null ? receiver.receiveMessages(maxMsgPerPollInt) :
453                         receiver.receiveMessages(maxMsgPerPollInt, Duration.ofSeconds(maxWaitTime.longValue()));
454                 receivedMessages.stream().forEach(m -> {
455                     processMessage(m,document);
456                     isWaitTimeExhausted = false;
457                     messageCountLong += 1;
458                     if (setAcknowledgementMode.equals(AUTO_ACKNOWLEDGE)) {
459                         receiver.complete(m);
460                     } else {
461                         receivedMessagesMap.put(m.getMessageId() + m.getSequenceNumber() + m.getLockToken(), m);
462                         synchronized (receiver) {
463                             try {
464                                 if (ackTimeout != null) {
465                                     receiver.wait(ackTimeout.longValue());
466                                 } else {
467                                     receiver.wait();
468                                 }
469                             } catch (InterruptedException e) {
470                                 LOG.error(ACKNOWLEDGE_LISTENER_THREAD_INTERRUPTED, e);
471                             }
472                         }
473                     }
474                 });
475             }
476             if ((maxMsgCountLong <= messageCountLong && maxMsgCountLong > 0) ||
477                     (isWaitTimeExhausted && maxWaitTime != null)) {
478                 setStopFlag();
479             }
480             if (pollInterval != null) {
481                 Thread.sleep(pollInterval.longValue());
482             }
483             if (messageCountLong - numOfMessages < maxMsgPerPollLong) {
484                 receiver.close();
485                 receiver = null;
486                 if (setDestType.equals(SetDestinationType.TOPIC)) {
487                     createReceiverSubscription(topic,subscriptionValue,prefetchCountInt);
488                 } else {
489                     createReceiverQueue(queue,prefetchCountInt);
490                 }
491             }
492         }
493     }

495     /**
496      * Consumes messages from service bus asynchronously.
497      *
498      * @param document input document
499      *
500      * NOTE 1:
501      *  The message receive loop is processed optimisitically; we process the
502      *  message assuming the ack will succeed (message settled). This will be
503      *  true most of the time.
504      *
505      *  However, if the receive loop encounters a non-retriable exception and
506      *  prefetch is enabled (ie, prefect count > 0), there may be a set of
507      *  unsettled messages. Some of these will be redelivered and assigned
508      *  a new lock token and the delivery count incremented. The rest will be
509      *  dropped. however, this may be less obvious because we optimisitcally
510      *  processed the messages.
511      *
512      *  If the redelivered message is picked up by this consumer, we can detect
513      *  this and not process it again (see LRU cache below). The message value
514      *  (body) is identical, but message delivery metadata (lock token,
515      *  delivery count) is different.
516      *
517      *  If the redelivered message is picked up by a different consumer, it
518      *  won't realize that the message has already been processed by a
519      *  different consumer, so will process it again. This is a dupe in the
520      *  system and not handled at the moment.
521      *
522      *  The hope is to avoid this condition which is why the prefetch count
523      *  is restricted to such a small size. We detected overflow in the service
524      *  bus with higher prefetch sizes which cascades into a variety of service
525      *  bus exceptions and very difficult to reconcile the unsettled messages.
526      *
527      *  See SWAT-3682 for details about research and testing to support
528      *  prefetching.
529      *
530      *  Here is a github issue which acknowledges service bus overflow
531      *  without any resolution (2023-02):
532      *
533      *      Understand the ReceiveLinkHandler deliveries overflow
534      *      https://github.com/Azure/azure-sdk-for-java/issues/33703
535      */
536     private void consumeMessagesAsynchronously(Document document) throws InterruptedException {
537         // LRU cache of processed messages
538         final Set<String> processedMessageIds = Collections.synchronizedSet(
539             Collections.newSetFromMap(new LinkedHashMap<String, Boolean>(){
540                 protected boolean removeEldestEntry(Map.Entry<String, Boolean> eldest) {
541                     return size() > 5_000;
542                 }
543             })
544         );
545 
546         final boolean isTopic = setDestType.equals(SetDestinationType.TOPIC);
547         boolean shouldContinue = true;
548         while (shouldContinue) {
549             try (ServiceBusReceiverAsyncClient asyncReceiver =
550                 isTopic ? createAsyncReceiverSubscription() : createAsyncReceiverQueue()) {
551                 final CountDownLatch errorLatch = new CountDownLatch(1);
552 
553                 final Disposable subscription = asyncReceiver.receiveMessages()
554                     .flatMap(msg -> {
555                         if (!processedMessageIds.contains(msg.getMessageId())) {
556                             // see NOTE #1
557                             processMessage(msg, document);
558                             processedMessageIds.add(msg.getMessageId());
559                         }
560                         Mono<Void> res = null;
561                         if (setAcknowledgeType != null) {
562                             switch (setAcknowledgeType) {
563                                 case DEFER_MESSAGE_PROP:
564                                     res = asyncReceiver.defer(msg);
565                                     break;
566                                 case ABANDON_MESSAGE_PROP:
567                                     res = asyncReceiver.abandon(msg);
568                                     break;
569                                 case ACK_MESSAGE_PROP:
570                                     res = asyncReceiver.complete(msg);
571                                     break;
572                                 case DEADLETTER_MESSAGE_PROP:
573                                     res = asyncReceiver.deadLetter(msg);
574                                     break;
575                                 default:
576                                     res = asyncReceiver.complete(msg);
577                             }
578                         } else {
579                             res = asyncReceiver.abandon(msg);
580                         }
581                         return res;
582                     })
583                     .subscribe(
584                         (ignore) -> LOG.debug("Message processed"),
585                         error -> {
586                             LOG.warn("Retry Exhausted or Non-Retriable error occurred", error);
587                             errorLatch.countDown();
588                         });
589 
590                 if (countDownLatchBigInt == null || countDownLatchBigInt.longValue() < 1) {
591                     // block current thread until there is an error.
592                     errorLatch.await();
593                 } else {
594                     errorLatch.await(countDownLatchBigInt.longValue(), TimeUnit.SECONDS);
595                     shouldContinue = false;
596                 }
597 
598                 // disposing of subscription will cancel receive operation.
599                 if (subscription != null) {
600                     subscription.dispose();
601                 }
602 
603             } // auto-close receiver
604 
605             if (shouldContinue) {
606                 // backoff before consuming again
607                 TimeUnit.SECONDS.sleep(1);
608             }
609         }
610     }

794         @Override
795         public void run() {
796             while (!stopFlag) {
797                 synchronized (ackThreadWaitObj) {
798                     try {
799                         ackThreadWaitObj.wait();
800                     } catch (InterruptedException e) {
801                         LOG.error(ACKNOWLEDGE_LISTENER_THREAD_INTERRUPTED, e);
802                         break;
803                     }
804                 }
805             }
806         }

808         @Override
809         public void update(Observable observable, Object arg) {
810             if (arg != null) {
811                 serviceBusMessage = (ServiceBusMessage) arg;
812                 if (receiver == null) {
813                     throw new ConfigurationException(ERROR_MULTIPLE_CONSUMERS);
814                 }
815                 synchronized (receiver) {
816                     try {
817                         LOG.debug("Received message from acknowledgement snap: {}",serviceBusMessage);
818                         String messageId = serviceBusMessage.getMessageID();
819                         String sequenceNumber = serviceBusMessage.getSequenceNumber();
820                         String lockToken = serviceBusMessage.getLockToken();
821                         String messageKey = messageId+sequenceNumber+lockToken;
822                         if (!StringUtils.isEmpty(messageKey)) {
823                             ServiceBusReceivedMessage serviceBusReceivedMessage = receivedMessagesMap.get(messageKey);
824                             switch (serviceBusMessage.getMessageAction()) {
825                                 case DEFER_MESSAGE_PREFIX:
826                                     receiver.defer(serviceBusReceivedMessage);
827                                     break;
828                                 case ABANDON_MESSAGE_PREFIX:
829                                     receiver.abandon(serviceBusReceivedMessage);
830                                     break;
831                                 case ACKNOWLEDGE_MESSAGE_PREFIX:
832                                     receiver.complete(serviceBusReceivedMessage);
833                                     break;
834                                 case DEADLETTER_MESSAGE_PREFIX:
835                                     receiver.deadLetter(serviceBusReceivedMessage);
836                                     break;
837                             }
838                             receivedMessagesMap.remove(messageKey);
839                             if (((maxMsgCount.longValue() > 0) &&
840                                     (messageCountLong >= maxMsgCount.longValue())) || isWaitTimeExhausted) {
841                                 setStopFlag();
842                             }
843                         }
844                     } catch (Exception e) {
845                         SnapDataException se;
846                         se = new SnapDataException(e.getMessage());
847                         errorViews.write(se);
848                     } finally {
849                         receiver.notifyAll();
850                     }
851                 }
852             }
853         }
