235     private ServiceBusMessage prepareData(Document doc) throws JsonProcessingException {
236         Object messageValue = messageValueExpr.eval(doc);
237         if (messageValue == null) {
238             throw new SnapDataException(ERR_MSG_VAL_NULL_MSG)
239                     .withReason(ERR_MSG_VAL_NULL_REASON)
240                     .withResolution(ERR_MSG_VAL_NULL_RESOLUTION);
241         }
242         String dateTimeString = scheduleMessageExpr == null ? null : scheduleMessageExpr.eval(doc);
243         dateTime = dateTimeString != null ? OffsetDateTime.parse(dateTimeString) : null;
244         ServiceBusMessage message = null;
245         if (msgPropMap.containsKey(CONTENT_TYPE)) {
246             ExpressionProperty contentTypeExpr = msgPropMap.get(CONTENT_TYPE);
247             if (contentTypeExpr != null) {
248                 String contentType = contentTypeExpr.eval(doc);
249                 if (StringUtils.isNotBlank(contentType)) {
250                     if (contentType.contains(CONTENT_TYPE_JSON)) {
251                         message = new ServiceBusMessage(BinaryData.fromObject(messageValue, new DefaultJsonSerializer()));
252                     } else {
253                         message = new ServiceBusMessage(BinaryData.fromString(messageValue.toString()));
254                     }
255                     message = message.setContentType(contentType);
256                 }
257             }
258         } else {
259             if (setContentType.equals(SetContentType.APPLICATION_JSON)) {
260                 message = new ServiceBusMessage(BinaryData.fromObject(messageValue, new DefaultJsonSerializer()));
261                 message.setContentType(CONTENT_TYPE_JSON);
262             }  else {
263                 message = new ServiceBusMessage(BinaryData.fromString(messageValue.toString()));
264                 if (setContentType.equals(SetContentType.APPLICATION_XML)) {
265                     message.setContentType(CONTENT_TYPE_XML);
266                 } else {
267                     message.setContentType(CONTENT_TYPE_PLAIN_TEXT);
268                 }
269             }
270         }
271         if (!msgPropMap.isEmpty()) {
272             message = constructMessage(message, doc);
273         }
274         if (sender == null) {
275             if (setDestType.equals(SetDestinationType.TOPIC)) {
276                 createProducerTopic(topic);
277             } else {
278                 createProducerQueue(queue);
279             }
280         }
281         return message;
282     }

285         Map<String, Object> data = new LinkedHashMap<>();
286         if (dateTime != null) {
287             long seqNum = sender.scheduleMessage(message, dateTime);
288             if (!StringUtils.isEmpty(String.valueOf(seqNum))) {
289                 data.put(SEQUENCE_NUMBER, String.valueOf(seqNum));
290                 data.put(STATUS, "Scheduled");
291                 outputViews.write(documentUtility.newDocument(data), doc);
292                 LOG.debug("Message scheduled: {}", message.getBody());
293             }
294         } else {
295             sender.sendMessage(message);
296             data.put(STATUS, "Sent");
297             outputViews.write(documentUtility.newDocument(data), doc);
298             LOG.debug("Message sent: {}", message);
299         }
