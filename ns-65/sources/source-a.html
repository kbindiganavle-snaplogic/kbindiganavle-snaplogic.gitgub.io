148      * Returns the absolute path of the element being visited
149      * (NOTE: This method doesn't work if a macro is being generated)
150      *
151      * @return an absolute JSON path followed by a '.'
152      */
153     public String getAbsolutePath() {
154         checkIfDisposed();
155         if (StringUtils.isEmpty(currentCollectionPath)) {
156             return peekPath();
157         }
158         // Check if the collection path is the same as the accessor variable
159         if (currentCollectionPath.equals(collectionPathToAccessorMap.get(currentCollectionPath))) {
160             return peekPath();
161         }
162 
163         // If not, then attach the relative path to the collection path
164         String relativePath = peekPath();
165         int firstIndexOfDot = relativePath.indexOf(JsonPathBuilder.CURRENT_ELEMENT);
166         if (firstIndexOfDot != -1) {
167             return currentCollectionPath.concat(relativePath.substring(firstIndexOfDot));
168         }
169         return currentCollectionPath.concat(JsonPathBuilder.CURRENT_ELEMENT);
170     }

271      * @param elementVarName      A name for accessor variable
272      * @param collectionPath      A path for collection
273      * @throws XMLStreamException

275     public void includeForeachDirective(String elementVarName, String collectionPath)
276             throws XMLStreamException {
277         checkIfDisposed();
278         String foreachDirective = String.format(FOREACH_DIRECTIVE_FORMAT, elementVarName,
279                 collectionPath);
280         xmlStreamWriter.writeCharacters(foreachDirective);
281         previousCollectionPathLengthStack.push(currentCollectionPath.length());
282         currentCollectionPath = StringUtils.EMPTY.equals(currentCollectionPath) ? collectionPath :
283                 currentCollectionPath
284                         .concat(collectionPath.substring(collectionPath.indexOf(
285                                 JsonPathBuilder.CURRENT_ELEMENT)));
286         collectionPathToAccessorMap.put(currentCollectionPath, elementVarName);
287     }
