25     /**
26      * Normalizes the localPart : i.e. replaces any '.' with '_', finds the
27      * index of the first alphabet and returns the substring of localPart,
28      * beginning at that index
29      *
30      * @param localPart
31      * @return a normalized localPart
32      */
33     public static String normalizeLocalPart(String localPart) {
34         localPart = localPart.replace(DOT, UNDERSCORE);
35         char[] chars = localPart.toCharArray();
36         int indexOfFirstAlphabet = 0;
37         for (int i = 0; i < chars.length; i++) {
38             if (Character.isUpperCase(chars[i]) || Character.isLowerCase(chars[i])) {
39                 indexOfFirstAlphabet = i;
40                 break;
41             }
42         }
43         return localPart.substring(indexOfFirstAlphabet);
44     }
