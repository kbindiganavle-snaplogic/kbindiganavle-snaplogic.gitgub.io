25 /**
26  * Converts Parquet List Records object into Json Objects
27  */
28 public class ParquetListRecord extends ParquetRecord {
29     @Override
30     protected Object toJsonObject() {
31         List<Object> result = new ArrayList<>(values.size());
32         for (int i = 0; i < values.size(); i++) {
33             result.add(toJsonValue(values.get(i).getValue()));
34             if (result.get(i) instanceof LinkedHashMap) {
35                 LinkedHashMap data = ((LinkedHashMap) result.get(i));
36                 // This is to skip the 3rd element leve of the LIST datatype, this is because we
37                 // use the Spark version of Parquet reader snap's output as the source of truth
38                 if (data.size() == 1) {
39                     result.set(i, data.get(data.keySet().toArray()[0]));
40                 }
41             }
42         }
43         return result;
44     }
