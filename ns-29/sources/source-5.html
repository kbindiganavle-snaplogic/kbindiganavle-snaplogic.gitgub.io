262

264             if (!is.markSupported()) {
265                 LOG.debug("InputStream does not support mark/reset");
266                 return null;

268 
269             // create metadata with hints.
270             final Metadata metadata = new Metadata();
271             if (filename != null) {
272                 metadata.add(Metadata.RESOURCE_NAME_KEY, filename);
273             }
274             if (contentType != null) {
275                 metadata.add(Metadata.CONTENT_TYPE, contentType);
276             }
277 
278             // load buffer, reset stream.
279             final byte[] buffer = new byte[MAX_STREAM_HEADER_SIZE];
280             is.mark(buffer.length);
281             final int n = is.read(buffer);
282             is.reset();
283 
284             if (n == 0) {
285                 return null;
286             }
287 
288             // attempt to identify type of field.
289             try (InputStream bais = new ByteArrayInputStream(buffer, 0, n)) {
290                 final Detector detector = new DefaultDetector();
291                 final MediaType mediaType = detector.detect(is, metadata);
292                 if ((mediaType == null) || DEFAULT_MEDIA_TYPE.equals(mediaType.toString())) {
293                     return null;
294                 }
295                 return mediaType.toString();
296             } catch (IOException e) {
297                 // log and ignore
298                 LOG.debug("error identifying media type", e);
299             }

589          * @param retryInterval - retry interval in seconds
590          * @param maxRetries    - maximum number of retries
591          * @param directory     - directory to be listed
592          * @param isFiles       - is regular file
593          * @param filter        - filter to match file names
594          * @param trueCerts     - true if all certificates are trusted
595          * @param inDoc         - input document
596          *
597          * @return Set of files or directories which matches with the file filter pattern
598          * @throws UnsupportedEncodingException
599          */
600         public List executeListFiles(final int retryInterval, final int maxRetries,
601                                      final String directory, final Boolean isFiles,
602                                      final String filter, final boolean trueCerts,
603                                      final Document inDoc)
604                 throws UnsupportedEncodingException, SnapDataException {
605             setRestParams(directory);
606             this.trueCerts = trueCerts;
607             if (elements[1] != null) {
608                 params.put(DIRECTORY_PARAM, elements[1]);
609             }
610             PathMatcher matcher = getPathMatcher(directory, filter);
611             List<Object> listingFilesDirectories = new ArrayList<>();
612             do {
613                 sendingRequest(APIUtils.Operation.LIST_FILES, retryInterval, maxRetries, inDoc,
614                         null, null, false);
615                 List<Object> list = abfsRestUtility.getListingFilesDirectories();
616                 for (Object obj: list) {
617                     Map<String, Object> document = new LinkedHashMap<>();
618                     Map<String, Object> eachFile = (Map) obj;
619                     String name = (String) eachFile.get(NAME);
620                     name = name.substring(name.lastIndexOf(URL_SEPARATOR) + 1);
621                     String fullPath = StringUtils.join(directory, name);
622                     document.put(FILE_PATH, fullPath);
623                     document.put(FILE_NAME, name);
624                     document.put(OWNER, eachFile.get(OWNER_RES));
625                     document.put(Constants.FILE_PERMISSIONS, eachFile.get(PERMISSIONS));
626                     document.put(LAST_MODIFIED_TIME, eachFile.get(LAST_MODIFIED));
627                     document.put(ETAG_NAME, eachFile.get(ETAG));
628                     if (isFiles == null) {
629                         if (eachFile.get(IS_DIRECTORY) == null && isFileNameMatched(matcher,
630                                 fullPath)) {
631                             document.put(SIZE, eachFile.get(CONTENT_LENGTH));
632                             document.put(TYPE, DEFAULT_VALUE_FILE);
633                             listingFilesDirectories.add(document);
634                         } else if (isFileNameMatched(matcher, fullPath)) {
635                             document.put(SIZE, SIZE_DIRECTORY);
636                             document.put(TYPE, DEFAULT_VALUE_DIR);
637                             listingFilesDirectories.add(document);
638                         }
639                     } else if (isFiles) {
640                         if (eachFile.get(IS_DIRECTORY) == null) {
641                             if (isFileNameMatched(matcher, fullPath)) {
642                                 listingFilesDirectories.add(fullPath);
643                             }
644                         }
645                     } else {
646                         if (eachFile.get(IS_DIRECTORY) != null) {
647                             if (isFileNameMatched(matcher, fullPath)) {
648                                 listingFilesDirectories.add(fullPath);
649                             }
650                         }
651                     }
652                 }
653                 // Check if there is a next page.
654             } while (StringUtils.isNotBlank(abfsRestUtility.getContinuation()));
655             return listingFilesDirectories;
656         }
