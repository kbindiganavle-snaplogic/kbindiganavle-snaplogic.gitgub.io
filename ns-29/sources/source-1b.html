301         //snaps that do not have retry properties will evaluate

305                 MIN_RETRY_INTERVAL);

307         flushInterval = value != null ? value.intValue() * KILO : DEFAULT_FLUSH_INTERVAL_KB;
308         tempFileDir = JfsUtils.getTmpDir(expressionUtils, propertyValues, errorViews);
309         fileProp = propertyValues.getAsExpression(FILE_PROP);
310         dirProp = propertyValues.getAsExpression(DIRECTORY_PROP);
311 
312         if (!isHdfsWrite) {
313             String dir = validateFileProp(null);
314             if (dir == null) {
315                 throw new ConfigurationException(ERR_EMPTY_FILENAME)
316                         .withReason(ERR_BLANK_FILENAME)
317                         .withResolution(RESOLUTION_EMPTY_FILENAME);
318             }
319             try {
320                 initFilename(dir);
321             } catch (SnapDataException ex) {
322                 errorViews.write(ex);

324                 return;

326         }

328                 propertyValues);

348         if (usersAndPermissions != null && !usersAndPermissions.isEmpty()) {

358                             .withResolution(RESOLUTION_INVALID_ENTRY);
359                 }

363                             .withReason(REASON_USER_PERMISSIONS)
364                             .withResolution(RESOLUTION_USER_PERMISSIONS);
365                 }

411     @Override
412     protected void doWork() {
413         if (StringUtils.isBlank(filename) && !isHdfsWrite) {
414             return; // this error has already written to error vew in configure()
415         }
416         for (InputView inputView : inputViews) {
417             for (BinaryInput binaryData : inputViews.binaryInputs(inputView)) {
418                 try {
419                     Document header = binaryData.getHeader();
420                     retries = maxRetries;
421                     if (isHdfsWrite) {
422                         String dir = validateFileProp(header);
423                         if (dir == null) {
424                             errorViews.write((SnapDataException)
425                                     new SnapDataException(ERR_EMPTY_FILENAME)
426                                             .withReason(ERR_BLANK_FILENAME)
427                                             .withResolution(RESOLUTION_EMPTY_FILENAME), header);
428                             continue;
429                         }
430                         // initFilename() internally calls abfsaccount.connect() which will fail If
431                         // there are any connectivity issues, so added ContextualRunnable for retry.
432                         try {
433                             ContextualRunnable runnable = context -> initFilename(dir);
434                             CheckedConsumer<Throwable> failedAttemptListener = e -> LOG.warn(
435                                     "{} - Hdfs Writer failed [file: {}], {}",
436                                     HADOOP_RETRY_SIGNATURE, filename, e.getMessage());
437                             failsafeExecute(runnable, failedAttemptListener, retries,
438                                     retryInterval);
439                         } catch (Exception e) {
440                             Throwable ex = e instanceof FailsafeException ? e.getCause() : e;
441                             if (ex instanceof SnapDataException) {
442                                 errorViews.write((SnapDataException) ex, header);
443                             } else {
444                                 errorViews.write((SnapDataException)
445                                         new SnapDataException(ex, ex.getMessage())
446                                         .withResolution(RESOLUTION_ADDRESS_ISSUE), header);
447                             }
448                             continue;
449                         }
450                     }
451                     if (filename.contains(QUESTION_MARK)) {
452                         errorViews.write((SnapDataException)
453                                 new SnapDataException(ERR_INVALID_FILENAME)
454                                         .formatWith(filenameNoAuth)
455                                         .withReason(REASON_QUESTION_MARK)
456                                         .withResolution(RESOLUTION_ADDRESS_ISSUE));
457                         return;
458                     }
459                     process(header, binaryData.getChannel());
460                 } catch (SuggestViewAbortException e) {
461                     throw e;
462                 } catch (Exception e) {
463                     bytesRead = FAILED_TO_WRITE; // prevents from writing result to output view
464                     SnapDataException ex = new SnapDataException(e, ERR_READ_BINARY_INPUT)
465                             .withReason(e.getMessage())
466                             .withResolution(RESOLUTION_CUSTOMER_SUPPORT);
467                     errorViews.write(ex);
468                 }
469             }
470         }
471     }

473     protected void process(final Document header, final ReadableByteChannel readChannel) {
474         this.header = header;
475         try (InputStream is = Channels.newInputStream(readChannel)) {
476             if (maxRetries > 0) {
477                 InputStream retryInputStream = useTempFile(is, header);
478                 if (retryInputStream != null) {
479                     writeData(this.header, retryInputStream);
480                 } else {
481                     throw new SnapDataException(UNABLE_TO_READ_INPUT_STREAM)
482                             .withResolution(RESOLUTION_CUSTOMER_SUPPORT);
483                 }
484             } else {
485                 writeData(header, is);
486             }
487         } catch (SnapDataException e) {
488             errorViews.write(e, header);
489         } catch (IOException e) {
490             SnapDataException ex = new SnapDataException(e, ERR_READ_BINARY_INPUT)
491                     .withReason(e.getMessage())
492                     .withResolution(RESOLUTION_CUSTOMER_SUPPORT);
493             errorViews.write(ex, header);
494         }
495         // set permissions for the actual file
496         if ((filename.startsWith(UrlUtils.URL_PROTOCOL_ADL) ||
497                 filename.startsWith(UrlUtils.URL_PROTOCOL_HDFS)) && bytesRead != FAILED_TO_WRITE) {
498             setPosixPermissions(getUriFor(filenameNoAuth), filenameNoAuth);
499         } else if (StringUtils.startsWithIgnoreCase(filenameNoAuth, PROTOCOL_ABFS)) {
500             setAbfsPosixPermissions();
501         } else {
502             setPosixPermissions(getUriFor(filename), filenameNoAuth);
503         }
504         if (bytesRead > FAILED_TO_WRITE && isHdfsWrite && isOutputForEachFileWritten) {
505             /*
506                If  the file action is append, and protocol isn't "adl"
507                an error is thrown in the base class configure,  and these
508                errors are written to the error views.
509                The following check prevents, documents being produced for both error
510                and output views.
511               */
512             if (writeResult != WriteResult.APPENDED ||
513                     StringUtils.startsWithIgnoreCase(filename, ADL_SCHEME) ||
514                     StringUtils.startsWithIgnoreCase(filenameNoAuth, PROTOCOL_ABFS)) {
515                 Map<String, String> data = new LinkedHashMap<>(2);
516                 data.put(FILENAME, filenameNoAuth);
517                 data.put(RESULT, writeResult.toString().toLowerCase());
518                 outputViews.write(documentUtility.newDocumentFor(header, data), header);
519             }
520         }
521     }

523     protected void writeData(final Document header, InputStream inputStream) {
524         try {
525             if (isBadConfig) {
526                 return;
527             }
528             // If ABFS protocol
529             if (StringUtils.startsWithIgnoreCase(filenameNoAuth, PROTOCOL_ABFS)) {
530                 writeDataToAbfs(inputStream);
531                 return;
532             }
533 
534             bytesRead = inputStream.read(buffer);
535             if (bytesRead == -1) {
536                 return;
537             }
538 
539             initOutputStream();
540             if (outputStream != null) {
541                 int bytesToFlush = 0;
542                 while (bytesRead != -1) {
543                     // bytesRead could be 0 if there was an interruption,
544                     // so verify if there is data to write
545                     if (bytesRead > 0) {
546                         outputStream.write(buffer, 0, bytesRead);
547                         if (flushInterval >= 0) {
548                             bytesToFlush += bytesRead;
549                             if (bytesToFlush > flushInterval) {
550                                 flushQuietly(outputStream);
551                                 bytesToFlush = 0;
552                             }
553                         }
554                         updateMetrics(bytesRead);
555                     }
556                     bytesRead = inputStream.read(buffer);
557                 }
558                 flushQuietly(outputStream);
559                 LOG.info("Total number of bytes written successfully to the output stream: " +
560                         byteCounter.getStats());
561                 bytesRead = 0;
562                 urlUtils.checkResponse(urlConnection);
563             } else {
564                 // OutputStream is null when the file already exists and users has chosen to
565                 // ignore overwriting it. In this case, inputStream should be closed too soon
566                 // because any CSV/JSON/XML formatter Snap in the upper stream may fail with
567                 // "broken pipe". We should wait until this pipe is cleared.
568                 IOUtils.copyLarge(inputStream, new NullOutputStream());
569             }
570         } catch (SuggestViewAbortException e) {
571             throw e;
572         } catch (SnapException e) {
573             bytesRead = FAILED_TO_WRITE; // prevents from writing result to output view
574             SnapDataException ex = new SnapDataException(e, e.getMessage())
575                     .withReason(e.getReason())
576                     .withResolution(e.getResolution());
577             errorViews.write(ex, header);
578         } catch (Exception e) {
579             if (retries-- > 0) {
580                 retryWriting(inputStream, header, e.getMessage());
581             } else {
582                 throwExceptionCantWrite(e, header);
583             }
584             return;
585         } finally {
586             IOUtils.closeQuietly(inputStream);
587             cleanup(urlConnection);
588             outputStream = null;
589             urlConnection = null;
590         }
591     }

1127             int pos = 0;
1128
