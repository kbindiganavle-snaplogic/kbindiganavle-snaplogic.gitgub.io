138         /*
139         Sample metadata:
140             repeated group map (MAP_KEY_VALUE) {
141               required binary key (UTF8);
142               optional binary value (UTF8);
143             }
144 
145         Sample json string:
146             "map": [
147               {
148                 "key": "pln_reg_number",
149                 "value": "A6E22"
150               },
151               {
152                 "key": "msg_recorded_datetime",
153                 "value": "2017-09-11 22:13:54"
154               }
155             ]
156          */
157         List<Type> kvFields = type.asGroupType().getFields();
158         for (Map.Entry<String, Object> entry : kvMap.entrySet()) {
159             Object value = getValue(kvFields.get(1), entry.getKey(), kvMap, Object.class);
160             if (value == null) {
161                 continue;
162             }
163 
164             recordConsumer.startGroup();
165             recordConsumer.startField(STR_KEY, 0);
166             writePrimitive(kvFields.get(0), entry.getKey());
167             recordConsumer.endField(STR_KEY, 0);
168 
169             recordConsumer.startField(STR_VALUE, 1);
170             writePrimitive(kvFields.get(1), value);
171             recordConsumer.endField(STR_VALUE, 1);
172             recordConsumer.endGroup();
173         }

275                         }

526     private void addPrimitiveToConsumer(Type type, Object value) {
527         PrimitiveType primitiveType = type.asPrimitiveType();
528         switch (primitiveType.getPrimitiveTypeName()) {
529             case INT32:
530                 recordConsumer.addInteger((int) retrieveValue(value, primitiveType));
531                 break;
532             case INT64:
533                 recordConsumer.addLong(retrieveValue(value, primitiveType));
534                 break;
535             case INT96:
536                 if (value instanceof BigInteger) {
537                     BigInteger bigIntegerValue = (BigInteger) value;
538                     byte[] bigIntValueAsBytes = convertBigIntIntoBytes(bigIntegerValue,
539                             TIMESTAMP_BYTE_COUNT);
540                     recordConsumer.addBinary(Binary.fromConstantByteArray(bigIntValueAsBytes));
541                     break;
542                 } else if (value instanceof String) {
543                     recordConsumer.addBinary(Binary.fromString((String) value));
544                     break;
545                 } else {
546                     throw new UnsupportedOperationException(
547                             String.format(ERROR_CONVERSION_TO_BINARY_MSG,
548                                     value.getClass().getSimpleName(),
549                                     primitiveType.getPrimitiveTypeName()));
550                 }
551             case BOOLEAN:
552                 recordConsumer.addBoolean((Boolean) value);
553                 break;
554             case BINARY:
555                 value = parseBinaryBytes(value);
556                 recordConsumer.addBinary(Binary.fromString((String) value));
557                 break;
558             case FIXED_LEN_BYTE_ARRAY:
559                 if (value instanceof BigDecimal) {
560                     byte bigDecimalbytes[] = convertBigDecimalIntoBytes((BigDecimal) value,
561                             primitiveType.getDecimalMetadata());
562                     recordConsumer.addBinary(Binary.fromConstantByteArray(bigDecimalbytes));
563                     break;
564                 } else if (value instanceof String) {
565                     value = parseBinaryBytes(value);
566                     recordConsumer.addBinary(Binary.fromString((String) value));
567                     break;
568                 } else {
569                     throw new UnsupportedOperationException(
570                             String.format(ERROR_CONVERSION_TO_BINARY_MSG,
571                                     value.getClass().getSimpleName(),
572                                     primitiveType.getPrimitiveTypeName()));
573                 }
574             case FLOAT:
575                 recordConsumer.addFloat((getNumber(value)).floatValue());
576                 break;
577             case DOUBLE:
578                 recordConsumer.addDouble((getNumber(value)).doubleValue());
579                 break;
580         }
581     }

583     protected long retrieveValue(Object value, PrimitiveType primitiveType) {
584         OriginalType originalType = primitiveType.getOriginalType();
585         if (OriginalType.DATE.equals(originalType)) {
586             try {
587                 long milliSeconds;
588                 if (value instanceof LocalDate) {
589                     milliSeconds = dateFormatter.parseMillis(
590                             ((LocalDate) value).toString(dateFormatter));
591                 } else {
592                     milliSeconds = dateFormatter.parseMillis((String) value);
593                 }
594                 return  TimeUnit.MILLISECONDS.toDays(milliSeconds);
595             } catch (IllegalArgumentException iex) {
596                 throw new UnsupportedOperationException(
597                         String.format(ERROR_FORMATTING_DATE_FORMAT,
598                                 primitiveType.getName()));
599             }
600         } else if (OriginalType.TIMESTAMP_MILLIS.equals(originalType)){
601             try {
602                 if (value instanceof DateTime) {
603                     return ((DateTime) value).toDate().getTime();
604                 } else {
605                     return getNumber(value).longValue();
606                 }
607             } catch (IllegalArgumentException iex) {
608                 throw new UnsupportedOperationException(
609                         String.format(ERROR_FORMATTING_DATE_FORMAT,
610                                 primitiveType.getName()));
611             }
612         } else if (OriginalType.DECIMAL.equals(originalType)) {
613             DecimalMetadata decimalMetadata = primitiveType.getDecimalMetadata();

649         byte[] targetBytes = new byte[byteCount];

664     protected byte[] convertBigDecimalIntoBytes(BigDecimal value, DecimalMetadata metadata) {
665         int scale = metadata.getScale();
666         if (!TRUNCATE.equals(roundingMode)) {
667             value = value.setScale(scale, getRoundingMode(roundingMode));
668         }
669         BigInteger bigInterValue = value.scaleByPowerOfTen(scale).toBigInteger();
670         byte sourceBytes[] = bigInterValue.toByteArray();
671         int precision = metadata.getPrecision();
672         int expectedBytesCount = getByteLength(precision);
673 
674         if (sourceBytes.length == expectedBytesCount) {
675             return sourceBytes;
676         } else if (sourceBytes.length < expectedBytesCount) {
677             byte[] targetBytes = new byte[expectedBytesCount];
678             fillSourceBytesIntoTarget(bigInterValue, sourceBytes, targetBytes);
679             return targetBytes;
680         } else {
681             throw new SnapDataException(ERROR_DATA_CONVERSION_MSG)
682                     .withReason(ERROR_DATA_CONVERSION_REASON_DECIMAL_TO_BINARY)
683                     .withResolutionAsDefect();
684         }
685     }

688         if (precision == 0) {

692         return (bits + Byte.SIZE - 1) / Byte.SIZE;
