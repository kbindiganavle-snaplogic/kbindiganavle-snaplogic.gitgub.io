392         String successCodes = propertyValues.getAsExpression(SUCCESS_CODES).eval(null);
393         if (!StringUtils.isBlank(successCodes)) {
394             String[] successCodesArray = successCodes.split(SUCCESS_CODE_DELIMITER);
395             successIntegers = new ArrayList<>();
396             for (String aSuccessCodesArray : successCodesArray) {
397                 try {
398                     if (!StringUtils.isBlank(aSuccessCodesArray)) {
399                         successIntegers.add(Integer.valueOf(aSuccessCodesArray));
400                     }
401                 } catch (NumberFormatException nfe) {
402                     throw new ExecutionException(nfe, INVALID_SUCCESS_CODE_MSG)
403                             .withReason(String.format(INVALID_SUCCESS_CODE_RSN, aSuccessCodesArray))
404                             .withResolution(INVALID_SUCCESS_CODE_RES);

406             }
407         }

409         configureRouteErrorProperty(propertyValues);
410         bapiExpr = propertyValues.getAsExpression(I_DOC_READ_BAPI);
411         jCoDestination = getJCoDestination();
412         try {

482     @SuppressWarnings("unchecked")
483     protected void process(Document document, String inputViewName) {
484         if (super.getDocumentContentAsMap(document) == null) {
485             // if not a map: will write to error view if enabled otherwise throw exception
486             return;
487         }
488 
489         docExtension = iDocExtensionTypeProperty.eval(document);
490         if (StringUtils.isNotEmpty(bapiName) && context == null) {
491             return;
492         }
493         if (StringUtils.isEmpty(bapiName)) {
494            context = evalBapiNameAndCreateContext(document, jCoDestination);
495         }
496         if (context == null) {
497             return;
498         }
499         docName = docNameProperty.eval(document);
500         if (docName == null) {
501             SnapDataException snapDataException = new SnapDataException(IDOC_VALUE_IS_NULL)
502                     .withReason(IDOC_VALUE_IS_NULL_REASON)
503                     .withResolution(IDOC_VALUE_IS_NULL_RESOLUTION);
504             errorViews.write(snapDataException, document);
505             return;
506         }
507         pollingIntervalSec = ((BigInteger) pollingIntervalExpr.eval(document))
508                 .intValue();
509         pollingTimeoutSec = ((BigInteger) pollingTimeoutExpr.eval(document))
510                 .intValue();
511         JCoDestination jCoDestination = context.getJCoDestination();
512         IDocDocument iDocDocument = getIDoc(document);
513         if (iDocDocument == null) {
514             return;
515         }
516         SAPDocumentWrapper documentWrapper = new SAPDocumentWrapper(document);
517         Object objectRoot = documentWrapper.get(ROOT);
518         if (!(objectRoot instanceof Map)) {
519             SnapDataException snapDataException = new SnapDataException(NO_ROOT_ELEMENT_FOUND)
520                     .withReason(NO_ROOT_ELEMENT_FOUND)
521                     .withResolution(PLEASE_ENSURE_THE_DOCUMENT_PROVIDES_A_ROOT_ATTRIBUTE);
522             errorViews.write(snapDataException, document);
523             return;
524         }
525         Map<String, Object> rootMap = (Map<String, Object>) objectRoot;
526         if (!isBatch) {
527             processIDocInBatchMode(documentWrapper, rootMap, iDocDocument, document);
528         } else {
529             processIDocListMode(documentWrapper, rootMap, iDocDocument, document);
530         }

533     private RetryPolicy getRetryPolicy(Document document) {
534         return new RetryPolicy()
535                 .retryOn((Throwable throwable) -> {
536                     /**
537                      * If JCoException does not belongs to connection errors then we don't need to
538                      * reprocess them.
539                      * */
540                     if (throwable instanceof JCoException
541                             || throwable instanceof ExecutionException) {
542                         validateJCoErrorAndRetryConnection(throwable, document);
543                     }
544                     return (isConnected && isAConnectionError);
545                 })
546                 .withMaxRetries(maxAttempts)
547                 .withDelay(retryInterval, TimeUnit.SECONDS);
548     }

550     private IDocDocument getIDoc(Document document) {
551         try {
552             return Failsafe.with(getRetryPolicy(document))
553                     .onFailedAttempt(e -> {
554                         LOG.warn("Failed attempt to retry creating IDoc Document: {}", e);
555                     })
556                     .get(context -> {
557                         try {
558                             return sapServices.getIDocFactory()
559                                     .createIDocDocument(
560                                             sapServices.getIDocRepository(jCoDestination),
561                                             docName, docExtension);
562                         } catch (IDocMetaDataUnavailableException e) {
563                             SnapDataException snapDataException =
564                                     new SnapDataException(e, e.getMessage())
565                                     .withReason(SAP_METADATA_UNAVAILABLE)
566                                     .withResolution(INVALID_SNAP_PROPERTIES_RESOLUTION);
567                             errorViews.write(snapDataException, document);
568                             return null;
569                         }
570                     });
571         } catch (Exception e) {
572             Throwable cause = e instanceof FailsafeException ? e.getCause() : e;
573             if (cause instanceof SnapDataException) {
574                 errorViews.write((SnapDataException) cause, document);
575             } else {
576                 SnapDataException sde = new SnapDataException(cause, JCO_EXCEPTION_MSG)
577                         .withResolution(VERIFY_SNAP_PROPERTIES_AND_SAP_RES);
578                 errorViews.write(sde, document);
579             }
580             return null;
581         }
582     }

738     /**
739      * getIDocStatus
740      * <p> get IDoc Status
741      *
742      * @param documentWrapper    as the wrapper that provided the document,
743      *                           the content and the error list if any occur during processing
744      * @param pollingIntervalSec seconds for polling interval
745      * @param pollingTimeoutSec  timeout for polling
746      */
747     private void getIDocStatus(SAPDocumentWrapper documentWrapper, int pollingIntervalSec,
748                                int pollingTimeoutSec) {
749         long startTimeInMs = System.currentTimeMillis();
750         boolean foundStatus = false;
751         while (!foundStatus &&
752                 (System.currentTimeMillis() - startTimeInMs) < pollingTimeoutSec * 1000) {
753             try {
754                 sapExecutor.process(documentWrapper, tidContext, null, false);
755                 LOG.debug(PROCESSED_GETTING_IDOC_NUMBER_FROM_TID_DBG);
756                 Object mapValue = documentWrapper.get(I_DOCNUM);
757                 Object objectDocumentNumber = null;
758                 if (mapValue instanceof List) {
759                     List list = (List) mapValue;
760                     if (!list.isEmpty()) {
761                         if (list.get(list.size() - 1) != null) {
762                             objectDocumentNumber = list.get(list.size() - 1);
763                         }
764                     }
765                 }
766                 if (objectDocumentNumber == null) {
767                     TimeUnit.SECONDS.sleep(pollingIntervalSec);
768                 } else {
769                     LOG.debug(String.format(IDOC_NUMBER_FOR_TID_INF, objectDocumentNumber,
770                             objectDocumentNumber.getClass()));
771                     BigInteger val = new BigDecimal(objectDocumentNumber.toString()).toBigInteger();
772                     documentWrapper.put(DOCUMENT_NUMBER, val);
773                     if (StringUtils.isBlank(bapiName)) {
774                         bapiName = IDOC_READ_COMPLETELY;
775                     }
776 
777                     sapExecutor.process(documentWrapper, context, null, false);
778                     writeSAPTableToDocument(documentWrapper, context.getTableParamList());
779                     Object outputStatus = documentWrapper.get(STATUS_FIELD);
780                     String outputStatusTxt = String.valueOf(documentWrapper.get(STATUS_TEXT_KEY));
781                     if (outputStatus == null) {
782                         throw new ExecutionException(ERROR_GETTING_IDOC_STATUS)
783                                 .withReason(COULD_NOT_GET_STATUS_FOR_IDOC_MSG)
784                                 .withResolution(COULD_NOT_GET_STATUS_FOR_IDOC_RES);
785                     }
786                     Integer integerStatus = Integer.valueOf(outputStatus.toString());
787                     LOG.debug(String.format(IDOC_STATUS_WAS_DBG, integerStatus));
788                     Document documentCreated = documentWrapper.createOutputDocument
789                             (documentUtility);
790                     if (successIntegers != null && successIntegers.contains(integerStatus)) {
791                         outputViews.write(documentCreated);
792                     } else {
793                         if (documentWrapper.hasErrors()) {
794                             SnapDataException snapDataException = documentWrapper.createException(
795                                     documentUtility);
796                             errorViews.write(snapDataException, documentWrapper.getInputDocument());
797                         } else {
798                             SnapDataException snapDataException =
799                                     new SnapDataException(IDOC_WRITE_ERROR)
800                                             .withReason(String.format(
801                                                     STATUS_CODE_NOT_SUCCESSFUL_ONE,
802                                                     integerStatus, outputStatusTxt))
803                                             .withResolution(CHECK_IDOC_SYSTEM_AND_DATA_AND_RETRY);
804                             throw snapDataException;
805                         }
806                     }
807                     foundStatus = true;
808                 }
809             } catch (InterruptedException e) {
810                 throw new ExecutionException(e, ERROR_GETTING_IDOC_STATUS)
811                         .withReason(INTERRUPTED_WHILE_WAITING_FOR_IDOC_NUMBER)
812                         .withResolution(VERIFY_SAP_CONFIGURATION_RES);
813             }
814         }
815         if (!foundStatus) {
816             SnapDataException snapDataException =
817                     new SnapDataException(ERROR_GETTING_IDOC_STATUS)
818                             .withReason(TIMED_OUT_GETTING_IDOC_NUMBER)
819                             .withResolution(VERIFY_SAP_CONFIGURATION_RES);
820             throw snapDataException;
821         }
822     }

871                                 recMetaData.getType(field.getName()),
872                                 field.getValue(), field.getString(),
