91      * Will process the bapi call with the provided context and input.
92      * Will issue a transaction if the executeCommit flag is enabled.
93      *
94      * @param documentWrapper        as the wrapper that provided the document,
95      *                               the content and the error list if any occur during processing
96      * @param context                as the execution context
97      * @param dateTimeFormatter      as the date time formatter
98      * @param isOutputDateTimeAsText as a flag to output date or time value as text instead of
99      *                               date object
100      */
101     @SuppressWarnings("unchecked")
102     public void process(SAPDocumentWrapper documentWrapper, SAPExecutionContext context,
103                         DateTimeFormatter dateTimeFormatter, boolean isOutputDateTimeAsText) {
104         JCoDestination jCoDestination = context.getJCoDestination();
105         JCoRepository jCoRepository = context.getJCoRepository();
106         if (context.isExecuteCommit()) {
107             synchronized (AbstractJCoAccount.localSessionRef.get()) {
108                 beginTransaction(jCoDestination);
109                 LOG.debug("[*] Execute single bapi");
110                 processSingleBapi(documentWrapper, context, dateTimeFormatter,
111                         isOutputDateTimeAsText);
112                 LOG.debug("[*] Running BAPI Commit");
113                 executeCommit(documentWrapper, jCoRepository, jCoDestination);
114                 endTransaction(jCoDestination);
115             }
116         } else {
117             synchronized (AbstractJCoAccount.localSessionRef.get()) {
118                 if (context.startCallSequence()) {
119                     beginTransaction(jCoDestination);
120                 }
121                 processSingleBapi(documentWrapper, context,
122                         dateTimeFormatter, isOutputDateTimeAsText);
123             }
124         }
125     }

127     protected void beginTransaction(JCoDestination jCoDestination) {
128         if (JCoContext.isStateful(jCoDestination)) {
129             LOG.info("A stateful call sequence has been started already in the session. No need " +
130                     "to start again");
131         } else {
132             LOG.debug("[*] Starting a Stateful call sequence");
133             JCoContext.begin(jCoDestination);
134         }
135     }

137     protected void endTransaction(JCoDestination jCoDestination) {
138         try {
139             LOG.debug("[*] Ending a Stateful call sequence");
140             JCoContext.end(jCoDestination);
141         } catch (JCoException e) {
142             throw new ExecutionException(e, Messages.COMMIT_BAPI_CALL_FAILED)
143                     .withResolution(Messages.PLEASE_ADDRESS_MESSAGE_AND_RETRY_AS_NECESSARY);
144         }

148      * Commits a BAPI transaction
149      *
150      * @param documentWrapper as the wrapper that provided the document,
151      *                        the content and the error list if any occur during processing
152      * @param jCoRepository   as the repository
153      * @param jCoDestination  as the destination
154      */
155     private void executeCommit(SAPDocumentWrapper documentWrapper, JCoRepository jCoRepository,
156                                JCoDestination jCoDestination) {
157         try {
158             JCoFunctionTemplate jCoFunctionTemplate =
159                     jCoRepository.getFunctionTemplate(BAPI_TRANSACTION_COMMIT);
160             if (jCoFunctionTemplate == null) {
161                 throw new ExecutionException(Messages.CANNOT_FIND_BAPI_FUNCTION_TEMPLATE)
162                         .formatWith(BAPI_TRANSACTION_COMMIT);
163             }
164             JCoFunction jCoFunction = jCoFunctionTemplate.getFunction();
165             if (jCoFunction == null) {
166                 throw new ExecutionException(Messages.CANNOT_GET_BAPI_FUNCTION)
167                         .formatWith(BAPI_TRANSACTION_COMMIT);
168             }
169             JCoParameterList commitExportParamList = jCoFunction.getExportParameterList();
170             jCoFunction.execute(jCoDestination);
171             if (commitExportParamList != null) {
172                 Map<String, Object> executeResultMap = new HashMap<>();
173                 documentWrapper.put(BAPI_TRANSACTION_COMMIT, executeResultMap);
174                 putSAPResultFieldsToMap(executeResultMap, commitExportParamList,
175                         documentWrapper, false);
176             }
177         } catch (JCoException e) {
178             throw new ExecutionException(e, Messages.COMMIT_TRANSACTION_FAILURE)
179                     .withResolution(Messages.VERIFY_SAP_CONFIGURATION_RES);
180         }
181     }

183     private void processSingleBapi(SAPDocumentWrapper documentWrapper,
184                                    SAPExecutionContext context, DateTimeFormatter dateTimeFormatter,
185                                    boolean isOutputDateTimeAsText) {
186         try {
187             setSAPParameters(context.getImportParamList(), documentWrapper,
188                     dateTimeFormatter);
189             setSAPTableParameters(context.getTableParamList(),
190                     context.getTableParamListMetaData(), documentWrapper, dateTimeFormatter);
191             setSAPParameters(context.getChangingParamList(), documentWrapper,
192                     dateTimeFormatter);
193 
194             context.getJCoFunction().execute(context.getJCoDestination());
195             if (context.getExportParamList() != null) {
196                 putSAPResultFieldsToMap(documentWrapper.getDocumentMap(),
197                         context.getExportParamList(), documentWrapper, isOutputDateTimeAsText);
198             }
199             if (context.getTableParamList() != null) {
200                 putSAPResultFieldsToMap(documentWrapper.getDocumentMap(),
201                         context.getTableParamList(), documentWrapper, isOutputDateTimeAsText);
202             }
203             if (context.getChangingParamList() != null) {
204                 putSAPResultFieldsToMap(documentWrapper.getDocumentMap(),
205                         context.getChangingParamList(), documentWrapper, isOutputDateTimeAsText);
206             }
207         } catch (ConversionException e) {
208             throw new ExecutionException(e, JCO_EXCEPTION_CAUGHT_WHILE_CONVERTING_SAP_RESULTS)
209                     .withReason(JCO_EXCEPTION_CAUGHT_WHILE_CONVERTING_SAP_RESULTS)
210                     .withResolution(Messages.VERIFY_SAP_CONFIGURATION_RES);
211         } catch (JCoException e) {
212             throw new ExecutionException(e, ERROR_EXECUTING_BAPI_FUNCTION)
213                     .withResolution(Messages.VERIFY_SAP_CONFIGURATION_RES);
214         }
215     }

252     private void setTable(SAPDocumentWrapper documentWrapper,
253                           DateTimeFormatter dateTimeFormatter, String strTableName,
254                           Object sapTableObject, JCoTable jCoTable) {
255         if (!(sapTableObject instanceof Collection)) {
256             documentWrapper.addError(String.format(BAD_SAP_TABLE_PARAMETER, strTableName));
257             return;
258         }
259         Collection<Object> sapTableParameterCollection = (Collection<Object>) sapTableObject;
260         int counter = 0;
261         for (Object sapTableRow : sapTableParameterCollection) {
262             if (jCoTable.getNumRows() <= counter) {
263                 jCoTable.appendRow();
264             }
265             if (!(sapTableRow instanceof Map)) {
266                 documentWrapper.addError(String.format(BAD_SAP_TABLE_PARAMETER, strTableName));
267                 return;
268             }
269             Map<String, Object> sapTableRowFieldParameters = (Map<String, Object>) sapTableRow;
270             JCoRecordMetaData jCoRecordMetaData = jCoTable.getRecordMetaData();
271             int structureFieldCount = jCoRecordMetaData.getFieldCount();
272             for (int k = 0; k < structureFieldCount; ++k) {
273                 int fieldType = jCoRecordMetaData.getType(k);
274                 String fieldName = jCoRecordMetaData.getName(k);
275                 Object value = sapTableRowFieldParameters.get(fieldName);
276                 if (value != null) {
277                     setFieldValue(jCoTable, fieldType, fieldName, value, documentWrapper,
278                             dateTimeFormatter);
279                 }
280             }
281             counter++;
282         }

314      *
315      * @param jcoRecord         jcoRecord interface to set field and value
316      * @param type              jco type
317      * @param fieldName         field to set
318      * @param mapValue          value to set, it verified that value is not null

320      *                          the content and the error list if any occur during processing
321      * @param dateTimeFormatter as the date time formatter
322      */
323     private void setFieldValue(JCoRecord jcoRecord, int type, String fieldName,
324                                Object mapValue, SAPDocumentWrapper documentWrapper,
325                                DateTimeFormatter dateTimeFormatter) {
326         if (mapValue == null) {
327             LOG.warn("[*] A NULL value is assigned to field {}. Ignoring", fieldName);
328             return;
329         }
330         try {
331             LOG.debug("[*] Setting field {} of type {} with value {} of type {}", fieldName, type,
332                     mapValue, mapValue.getClass());
333             switch (type) {
334                 case JCoMetaData.TYPE_NUM:
335                     try {
336                         BigInteger bigInteger =
337                                 new BigInteger(String.valueOf(mapValue).replaceFirst(
338                                         REMOVE_LEADING_ZEROS_REGEX, StringUtils.EMPTY));
339                         jcoRecord.setValue(fieldName, bigInteger);
340                     } catch (NumberFormatException numFormatException) {
341                         String message = String.format(INVALID_NUMBER_FOR_FIELD,
342                                 jcoRecord.getMetaData().getName(),
343                                 fieldName, mapValue.getClass().toString());
344                         documentWrapper.addError(message);
345                     }
346                     break;
347                 // https://help.hana.ondemand.com/javadoc/com/sap/conn/jco/JCoRecord.html
348                 case JCoMetaData.TYPE_INT:
349                 case JCoMetaData.TYPE_INT1:
350                 case JCoMetaData.TYPE_INT2:
351                     try {
352                         Integer integer =
353                                 new Integer(String.valueOf(mapValue).replaceFirst(
354                                         REMOVE_LEADING_ZEROS_REGEX, StringUtils.EMPTY));
355                         jcoRecord.setValue(fieldName, integer);
356                     } catch (NumberFormatException numFormatException) {
357                         String message = String.format(INVALID_NUMBER_FOR_FIELD,
358                                 jcoRecord.getMetaData().getName(),
359                                 fieldName, mapValue.getClass().toString());
360                         documentWrapper.addError(message);
361                     }
362                     break;
363                 case JCoMetaData.TYPE_BCD:
364                 case JCoMetaData.TYPE_FLOAT:
365                 case JCoMetaData.TYPE_DECF16:
366                 case JCoMetaData.TYPE_DECF34:
367                     BigDecimal bigDecimal = convertToBigDecimal(mapValue, documentWrapper);
368                     jcoRecord.setValue(fieldName, bigDecimal);
369                     break;
370                 case JCoMetaData.TYPE_DATE:
371                 case JCoMetaData.TYPE_TIME:
372                     Date date = null;
373                     if (mapValue instanceof java.util.Date) {
374                         date = (java.util.Date) mapValue;
375                     } else if (mapValue instanceof DateTime) {
376                         DateTime jodaDateTime = (DateTime) mapValue;
377                         date = jodaDateTime.toDate();
378                     } else if (mapValue instanceof String && dateTimeFormatter != null) {
379                         DateTime jodaDateTime = dateTimeFormatter.parseDateTime((String) mapValue);
380                         date = jodaDateTime.toDate();
381                     }
382                     if (date != null) {
383                         jcoRecord.setValue(fieldName, date);
384                     } else {
385                         if (mapValue instanceof String) {
386                             jcoRecord.setValue(fieldName, mapValue);
387                         }
388                     }
389                     break;
390                 case JCoMetaData.TYPE_XSTRING:
391                 case JCoMetaData.TYPE_BYTE:
392                     if (mapValue instanceof byte[]) {
393                         jcoRecord.setValue(fieldName, (byte[])mapValue);
394                     } else {
395                         String value = String.valueOf(mapValue);
396                         jcoRecord.setValue(fieldName, value.getBytes());
397                     }
398                     break;
399                 case JCoMetaData.TYPE_TABLE:
400                     JCoTable jCoTable = jcoRecord.getTable(fieldName);
401                     if (jCoTable == null) {
402                         LOG.warn("[*] The JCo Record doesn't contain a Table field named {}, " +
403                                 "Ignoring it", fieldName);
404                         break;
405                     }
406                     List<Object> tableValues = (List<Object>) mapValue;
407                     JCoMetaData jCoTableMetaData = jCoTable.getRecordMetaData();
408                     String name = jCoTableMetaData.getName(0);
409                     int fieldType = jCoTableMetaData.getType(0);
410                     int counter = 0;
411                     for (int k = 0; k < tableValues.size(); ++k, jCoTable.nextRow()) {
412                         Object sapTableRow = tableValues.get(k);
413                         if (jCoTable.getNumRows() <= counter) {
414                             jCoTable.appendRow();
415                         }
416                         if (sapTableRow instanceof Map) {
417                             Map<String, Object> sapTableRowFieldParameters = (Map<String,
418                                     Object>) sapTableRow;
419                             JCoRecordMetaData jCoRecordMetaData = jCoTable.getRecordMetaData();
420                             int structureFieldCount = jCoRecordMetaData.getFieldCount();
421                             for (int mapIndex = 0; mapIndex < structureFieldCount; ++mapIndex) {
422                                 int mapFieldType = jCoRecordMetaData.getType(mapIndex);
423                                 String mapFieldName = jCoRecordMetaData.getName(mapIndex);
424                                 Object mapFieldValue = sapTableRowFieldParameters.get(mapFieldName);
425                                 setFieldValue(jCoTable, mapFieldType, mapFieldName,
426                                             mapFieldValue, documentWrapper, dateTimeFormatter);
427 
428                             }
429                         } else {
430                             setFieldValue(jCoTable, fieldType, name, sapTableRow,
431                                     documentWrapper, dateTimeFormatter);
432                         }
433                         counter++;
434                     }
435                     break;
436                 case JCoMetaData.TYPE_STRUCTURE:
437                     if (!(mapValue instanceof Map)) {
438                         documentWrapper.addError(String.format(BAD_SAP_TABLE_PARAMETER,
439                                 fieldName));
440                         return;
441                     }
442                     Map<String, Object> structureValues = (Map<String, Object>) mapValue;
443                     JCoStructure jCoStructure = jcoRecord.getStructure(fieldName);
444                     if (jCoStructure == null) {
445                         LOG.warn("[*] The JCo Record doesn't contain a Structure field named {}." +
446                                 " Ignoring it", fieldName);
447                         break;
448                     }
449                     JCoMetaData jCoMetaData = jCoStructure.getMetaData();
450                     int structureFieldCount = jCoMetaData.getFieldCount();
451                     for (int k = 0; k < structureFieldCount; ++k) {
452                         String structureFieldName = jCoMetaData.getName(k);
453                         int structFieldType = jCoMetaData.getType(k);
454                         Object structFieldValue = structureValues.get(structureFieldName);
455                         setFieldValue(jCoStructure, structFieldType, structureFieldName,
456                                 structFieldValue, documentWrapper, dateTimeFormatter);
457                     }
458                     break;
459                 default:
460                     jcoRecord.setValue(fieldName, String.valueOf(mapValue));
461             }
462         } catch (ExecutionException e) {
463             throw e;
464         } catch (Exception e) {
465             LOG.warn("[*] An error encountered when parsing filed: " + fieldName, e);
466             String message = String.format(Messages.DATA_MAPPING,
467                     jcoRecord.getMetaData().getName(),
468                     fieldName, describeJCoMetaDataType(type), mapValue.toString(), e.getMessage());
469             documentWrapper.addError(message);
470         }
471     }

543 
544         for (int i = 0; i < fieldCount; ++i) {

553                     resultsMap.put(name, structureMap);
554                     LOG.debug("Visiting structured Data {}", name);
555                     for (JCoField field : record) {
556                         putSAPFieldIntoMap(structureMap, field, documentWrapper,
557                                 isOutputDateTimeAsText);
558                     }
559                     break;
560                 case JCoMetaData.TYPE_TABLE:
561                     record = paramList.getTable(name);
562                     LOG.debug("Visiting Table Data {}", name);
563                     JCoTable tableRecord = (JCoTable) record;
564                     tableRecord.firstRow();
565                     int numRows = tableRecord.getNumRows();
566                     List<Object> sapTableRows = Lists.newArrayList();
567                     for (int j = 0; j < numRows; ++j, tableRecord.nextRow()) {
568                         tableRecord.setRow(j);
569                         Map<String, Object> sapRowFieldValues = Maps.newLinkedHashMap();
570                         boolean isScalar = false;
571                         for (JCoField field : tableRecord) {
572                             String fieldName = field.getName();
573                             if (StringUtils.isBlank(fieldName)) {
574                                 isScalar = true;
575                                 Object scalarVal = convertScalarValue(field.getType(), fieldName,
576                                         field.getValue(), field.getString(),
577                                         documentWrapper, isOutputDateTimeAsText);
578                                 sapTableRows.add(scalarVal);
579                             } else {
580                                 putSAPFieldIntoMap(sapRowFieldValues, field, documentWrapper,
581                                         isOutputDateTimeAsText);
582                             }
583                         }
584                         if (!isScalar) {
585                             sapTableRows.add(sapRowFieldValues);
586                         }
587                     }
588                     resultsMap.put(name, sapTableRows);
589                     break;
590                 default:
591                     putSAPFieldValueIntoMap(resultsMap, type,
592                             paramList.getValue(i), paramList.getString(i),
593                             name, documentWrapper,
594                             isOutputDateTimeAsText);
595             }
596         }
597     }

599     /**
600      * Deserialize and store a JCO Field into the given HashMap.
601      *
602      * The field could be a nested structure or table that needs to be deserialized recursively,
603      * otherwise, it would be treated as primitive field
604      *
605      * @param structureMap      The map that is supposed to store the deserialized field data
606      * @param field             The JCO Field
607      * @param documentWrapper   The wrapper that contains the document data
608      * @param isOutputDateTimeAsText   Should we output the date field and time field in raw text
609      */
610     private void putSAPFieldIntoMap(Map<String, Object> structureMap, JCoField field,
611                                         SAPDocumentWrapper documentWrapper,
612                                         boolean isOutputDateTimeAsText) {
613         int type = field.getType();
614         switch(type) {
615             case JCoRecordMetaData.TYPE_TABLE:
616                 LOG.debug("Visiting Nested Table Data {}", field);
617                 JCoTable tableVal = field.getTable();
618                 List<Object> sapTableRows = Lists.newArrayList();
619                 if (!tableVal.isEmpty()) {
620                     tableVal.firstRow();
621                     do {
622                         Map<String, Object> row = Maps.newLinkedHashMap();
623                         for (JCoField tableField :  tableVal) {
624                             putSAPFieldIntoMap(row, tableField, documentWrapper,
625                                     isOutputDateTimeAsText);
626                         }
627                         sapTableRows.add(row);
628                     } while (tableVal.nextRow());
629                 }
630                 structureMap.put(field.getName(), sapTableRows);
631                 break;
632             case JCoRecordMetaData.TYPE_STRUCTURE:
633                 LOG.debug("Visiting Nested Structure Data {}", field);
634                 JCoStructure structVal = field.getStructure();
635                 Map<String, Object> nestedStruct = Maps.newLinkedHashMap();
636                 for (JCoField childField : structVal) {
637                     putSAPFieldIntoMap(nestedStruct,
638                             childField, documentWrapper, isOutputDateTimeAsText);
639                 }
640                 structureMap.put(field.getName(), nestedStruct);
641                 break;
642             default:
643                 // Other field types are handled as primitive data
644                 putSAPFieldValueIntoMap(structureMap, type, field.getValue(), field.getString(),
645                         field.getName(), documentWrapper, isOutputDateTimeAsText);
646         }
647 
648     }

671      * Converts SAP scalar values to SnapLogic values
672      *
673      * @param type                   SAP JCo value type
674      * @param fieldName              SAP JCo field name
675      * @param scalarVal              SAP JCo field value
676      * @param fieldValueString       SAP JCo field value as String
677      * @param documentWrapper        as the document wrapper
678      * @param isOutputDateTimeAsText as a flag to output date or time value as text instead of
679      *                               date object
680      * @return object for SnapLogic document
681      */
682     protected Object convertScalarValue(int type, String fieldName,
683                                         Object scalarVal, String fieldValueString,
684                                         SAPDocumentWrapper documentWrapper,
685                                         boolean isOutputDateTimeAsText) {
686         if (scalarVal == null) {
687             return null;
688         }
689         switch (type) {
690             // going into SAP: TYPE_INT, INT1, and INT2 should be Integer but going out to
691             // SnapLogic can be represented by SnapLogic standard integer type BigInteger
692             case JCoMetaData.TYPE_NUM:
693             case JCoMetaData.TYPE_INT:
694             case JCoMetaData.TYPE_INT1:
695             case JCoMetaData.TYPE_INT2:
696                 try {
697                     scalarVal =
698                             new BigInteger(String.valueOf(scalarVal).replaceFirst(
699                                     REMOVE_LEADING_ZEROS_REGEX, StringUtils.EMPTY));
700                 } catch (NumberFormatException numFormatException) {
701                     String message = String.format("Invalid field value %s for type %s",
702                             fieldName, "Type Number");
703                     documentWrapper.addError(message);
704                 }
705                 break;
706             case JCoMetaData.TYPE_BCD:
707             case JCoMetaData.TYPE_FLOAT:
708             case JCoMetaData.TYPE_DECF16:
709             case JCoMetaData.TYPE_DECF34:
710                 scalarVal = convertToBigDecimal(scalarVal, documentWrapper);
711                 break;
712             case JCoMetaData.TYPE_DATE:
713             case JCoMetaData.TYPE_TIME:
714                 if (isOutputDateTimeAsText) {
715                     scalarVal = fieldValueString;
716                 } else {
717                     DateTime timeStamp = null;
718                     if (scalarVal instanceof java.util.Date) {
719                         timeStamp = new DateTime(scalarVal);
720                     }
721                     if (timeStamp == null) {
722                         documentWrapper.addError(String.format(INVALID_DATE, fieldName));
723                     }
724                     scalarVal = timeStamp;
725                 }
726                 break;
727             case JCoMetaData.TYPE_BYTE:
728             case JCoMetaData.TYPE_XSTRING:
729                 scalarVal = new String((byte[]) scalarVal);
730                 break;
731             default:
732                 scalarVal = String.valueOf(scalarVal);
733         }
734         return scalarVal;
735     }

738     /**
739      * SAP returns string values for numbers, such as 0000 or 0.0000 or 00001.0000.
740      * Here we transform those representations to BigDecimal.ZERO, so that we can compare them
741      * properly.
742      *
743      * @param value           the non null value to be converted
744      * @param documentWrapper as the document wrapper
745      * @return the converted value
746      */
747     private BigDecimal convertToBigDecimal(Object value, SAPDocumentWrapper documentWrapper) {
748         if (value instanceof BigDecimal) {
749             return (BigDecimal) value;
750         }
751         try {
752             String strValue = value.toString();
753             Matcher matcher = NULL_PATTERN.matcher(strValue);
754             if (matcher.find()) {
755                 return BigDecimal.ZERO;
756             }
757             return new BigDecimal(strValue.replaceFirst(REMOVE_LEADING_ZEROS_REGEX,
758                     StringUtils.EMPTY))
759                     .stripTrailingZeros();
760         } catch (NumberFormatException numFormatException) {
761             documentWrapper.addError(numFormatException.getMessage());
762             return null;
763         }
764     }
